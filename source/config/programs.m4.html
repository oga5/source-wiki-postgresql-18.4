4# PGAC_PATH_PROGS
5# ---------------
6# This wrapper for AC_PATH_PROGS behaves like that macro except when
7# VARIABLE is already set; in that case we just accept the value verbatim.
8# (AC_PATH_PROGS would accept it only if it looks like an absolute path.)
9# A desirable future improvement would be to convert a non-absolute-path
10# input into absolute form.
11AC_DEFUN([PGAC_PATH_PROGS],
12[if test -z "$$1"; then
13  AC_PATH_PROGS($@)
14else
15  # Report the value of $1 in configure's output in all cases.
16  AC_MSG_CHECKING([for $1])
17  AC_MSG_RESULT([$$1])
18fi
19])

22# PGAC_PATH_BISON
23# ---------------
24# Look for Bison, set the output variable BISON to its path if found.
25# Reject versions before 2.3 (the earliest version in the buildfarm
26# as of 2022). Note we do not accept other implementations of yacc.
27
28AC_DEFUN([PGAC_PATH_BISON],
29[PGAC_PATH_PROGS(BISON, bison)
30
31if test "$BISON"; then
32  pgac_bison_version=`$BISON --version 2>/dev/null | sed q`
33  AC_MSG_NOTICE([using $pgac_bison_version])
34  if echo "$pgac_bison_version" | $AWK '{ if ([$]4 < 2.3) exit 0; else exit 1;}'
35  then
36    AC_MSG_ERROR([
37*** The installed version of Bison, $BISON, is too old to use with PostgreSQL.
38*** Bison version 2.3 or later is required, but this is $pgac_bison_version.])
39  fi
40  # Bison >=3.0 issues warnings about %name-prefix="base_yy", instead
41  # of the now preferred %name-prefix "base_yy", but the latter
42  # doesn't work with Bison 2.3 or less.  So for now we silence the
43  # deprecation warnings.
44  if echo "$pgac_bison_version" | $AWK '{ if ([$]4 >= 3) exit 0; else exit 1;}'
45  then
46    BISONFLAGS="$BISONFLAGS -Wno-deprecated"
47  fi
48fi
49
50if test -z "$BISON"; then
51  AC_MSG_ERROR([bison not found])
52fi
53dnl We don't need AC_SUBST(BISON) because PGAC_PATH_PROGS did it
54AC_SUBST(BISONFLAGS)
55])# PGAC_PATH_BISON

78# PGAC_LDAP_SAFE
79# --------------
80# PostgreSQL sometimes loads libldap_r and plain libldap into the same
81# process.  Check for OpenLDAP versions known not to tolerate doing so; assume
82# non-OpenLDAP implementations are safe.  The dblink test suite exercises the
83# hazardous interaction directly.
84
85AC_DEFUN([PGAC_LDAP_SAFE],
86[AC_CACHE_CHECK([for compatible LDAP implementation], [pgac_cv_ldap_safe],
87[AC_COMPILE_IFELSE([AC_LANG_PROGRAM(
88[#include <ldap.h>
89#if !defined(LDAP_VENDOR_VERSION) || \
90     (defined(LDAP_API_FEATURE_X_OPENLDAP) && \
91      LDAP_VENDOR_VERSION >= 20424 && LDAP_VENDOR_VERSION <= 20431)
92choke me
93#endif], [])],
94[pgac_cv_ldap_safe=yes],
95[pgac_cv_ldap_safe=no])])
96
97if test "$pgac_cv_ldap_safe" != yes; then
98  AC_MSG_WARN([
99*** With OpenLDAP versions 2.4.24 through 2.4.31, inclusive, each backend
100*** process that loads libpq (via WAL receiver, dblink, or postgres_fdw) and
101*** also uses LDAP will crash on exit.])
102fi])

106# PGAC_CHECK_READLINE
107# -------------------
108# Check for the readline library and dependent libraries, either
109# termcap or curses.  Also try libedit, since NetBSD's is compatible.
110# Add the required flags to LIBS, define HAVE_LIBREADLINE.
111
112AC_DEFUN([PGAC_CHECK_READLINE],
113[AC_REQUIRE([AC_CANONICAL_HOST])
114
115AC_CACHE_CHECK([for library containing readline], [pgac_cv_check_readline],
116[pgac_cv_check_readline=no
117pgac_save_LIBS=$LIBS
118if test x"$with_libedit_preferred" != x"yes"
119then	READLINE_ORDER="-lreadline -ledit"
120else	READLINE_ORDER="-ledit -lreadline"
121fi
122for pgac_rllib in $READLINE_ORDER ; do
123  for pgac_lib in "" " -ltermcap" " -lncurses" " -lcurses" ; do
124    LIBS="${pgac_rllib}${pgac_lib} $pgac_save_LIBS"
125    AC_TRY_LINK_FUNC([readline], [[
126      # Older NetBSD and OpenBSD have a broken linker that does not
127      # recognize dependent libraries; assume curses is needed if we didn't
128      # find any dependency.
129      case $host_os in
130        netbsd* | openbsd*)
131          if test x"$pgac_lib" = x"" ; then
132            pgac_lib=" -lcurses"
133          fi ;;
134      esac
135
136      pgac_cv_check_readline="${pgac_rllib}${pgac_lib}"
137      break
138    ]])
139  done
140  if test "$pgac_cv_check_readline" != no ; then
141    break
142  fi
143done
144LIBS=$pgac_save_LIBS
145])[]dnl AC_CACHE_CHECK
146
147if test "$pgac_cv_check_readline" != no ; then
148  LIBS="$pgac_cv_check_readline $LIBS"
149  AC_DEFINE(HAVE_LIBREADLINE, 1, [Define if you have a function readline library])
150fi
151
152])# PGAC_CHECK_READLINE

156# PGAC_READLINE_VARIABLES
157# -----------------------
158# Some Readline versions lack rl_completion_suppress_quote.
159# Libedit lacks rl_filename_quote_characters and rl_filename_quoting_function
160
161AC_DEFUN([PGAC_READLINE_VARIABLES],
162[AC_CACHE_CHECK([for rl_completion_suppress_quote], pgac_cv_var_rl_completion_suppress_quote,
163[AC_LINK_IFELSE([AC_LANG_PROGRAM([#include <stdio.h>
164#if defined(HAVE_READLINE_READLINE_H)
165#include <readline/readline.h>
166#elif defined(HAVE_EDITLINE_READLINE_H)
167#include <editline/readline.h>
168#elif defined(HAVE_READLINE_H)
169#include <readline.h>
170#endif
171],
172[rl_completion_suppress_quote = 1;])],
173[pgac_cv_var_rl_completion_suppress_quote=yes],
174[pgac_cv_var_rl_completion_suppress_quote=no])])
175if test x"$pgac_cv_var_rl_completion_suppress_quote" = x"yes"; then
176AC_DEFINE(HAVE_RL_COMPLETION_SUPPRESS_QUOTE, 1,
177          [Define to 1 if you have the global variable 'rl_completion_suppress_quote'.])
178fi
179AC_CACHE_CHECK([for rl_filename_quote_characters], pgac_cv_var_rl_filename_quote_characters,
180[AC_LINK_IFELSE([AC_LANG_PROGRAM([#include <stdio.h>
181#if defined(HAVE_READLINE_READLINE_H)
182#include <readline/readline.h>
183#elif defined(HAVE_EDITLINE_READLINE_H)
184#include <editline/readline.h>
185#elif defined(HAVE_READLINE_H)
186#include <readline.h>
187#endif
188],
189[rl_filename_quote_characters = "x";])],
190[pgac_cv_var_rl_filename_quote_characters=yes],
191[pgac_cv_var_rl_filename_quote_characters=no])])
192if test x"$pgac_cv_var_rl_filename_quote_characters" = x"yes"; then
193AC_DEFINE(HAVE_RL_FILENAME_QUOTE_CHARACTERS, 1,
194          [Define to 1 if you have the global variable 'rl_filename_quote_characters'.])
195fi

197[AC_LINK_IFELSE([AC_LANG_PROGRAM([#include <stdio.h>
198#if defined(HAVE_READLINE_READLINE_H)
199#include <readline/readline.h>
200#elif defined(HAVE_EDITLINE_READLINE_H)
201#include <editline/readline.h>
202#elif defined(HAVE_READLINE_H)
203#include <readline.h>
204#endif
205],
206[rl_filename_quoting_function = 0;])],
207[pgac_cv_var_rl_filename_quoting_function=yes],
208[pgac_cv_var_rl_filename_quoting_function=no])])
209if test x"$pgac_cv_var_rl_filename_quoting_function" = x"yes"; then
210AC_DEFINE(HAVE_RL_FILENAME_QUOTING_FUNCTION, 1,
211          [Define to 1 if you have the global variable 'rl_filename_quoting_function'.])
212fi

217# PGAC_CHECK_GETTEXT
218# ------------------
219# We check for bind_textdomain_codeset() not just gettext().  GNU gettext
220# before 0.10.36 does not have that function, and is generally too incomplete
221# to be usable.
222
223AC_DEFUN([PGAC_CHECK_GETTEXT],
224[
225  AC_SEARCH_LIBS(bind_textdomain_codeset, intl, [],
226                 [AC_MSG_ERROR([a gettext implementation is required for NLS])])
227  AC_CHECK_HEADER([libintl.h], [],
228                  [AC_MSG_ERROR([header file <libintl.h> is required for NLS])])
229  PGAC_PATH_PROGS(MSGFMT, msgfmt)
230  AC_ARG_VAR(MSGFMT, [msgfmt program for NLS])dnl
231  if test -z "$MSGFMT"; then
232    AC_MSG_ERROR([msgfmt is required for NLS])
233  fi
234  AC_CACHE_CHECK([for msgfmt flags], pgac_cv_msgfmt_flags,
235[if test x"$MSGFMT" != x"" && "$MSGFMT" --version 2>&1 | grep "GNU" >/dev/null; then
236    pgac_cv_msgfmt_flags=-c
237fi])
238  AC_SUBST(MSGFMT_FLAGS, $pgac_cv_msgfmt_flags)
239  PGAC_PATH_PROGS(MSGMERGE, msgmerge)
240  PGAC_PATH_PROGS(XGETTEXT, xgettext)
241])# PGAC_CHECK_GETTEXT

245# PGAC_CHECK_STRIP
246# ----------------
247# Check for a 'strip' program, and figure out if that program can
248# strip libraries.
249
250AC_DEFUN([PGAC_CHECK_STRIP],
251[
252  AC_CHECK_TOOL(STRIP, strip, :)
253
254  AC_MSG_CHECKING([whether it is possible to strip libraries])
255  if test x"$STRIP" != x"" && "$STRIP" -V 2>&1 | grep "GNU strip" >/dev/null; then
256    STRIP_STATIC_LIB="$STRIP --strip-unneeded"
257    STRIP_SHARED_LIB="$STRIP --strip-unneeded"
258    AC_MSG_RESULT(yes)
259  else
260    case $host_os in
261      darwin*)
262        STRIP="$STRIP -x"
263        STRIP_STATIC_LIB=$STRIP
264        STRIP_SHARED_LIB=$STRIP
265        AC_MSG_RESULT(yes)
266        ;;
267      *)
268        STRIP_STATIC_LIB=:
269        STRIP_SHARED_LIB=:
270        AC_MSG_RESULT(no)
271        ;;
272    esac
273  fi
274  AC_SUBST(STRIP_STATIC_LIB)
275  AC_SUBST(STRIP_SHARED_LIB)
276])# PGAC_CHECK_STRIP

280# PGAC_CHECK_LIBCURL
281# ------------------
282# Check for required libraries and headers, and test to see whether the current
283# installation of libcurl is thread-safe.
284
285AC_DEFUN([PGAC_CHECK_LIBCURL],
286[
287  # libcurl compiler/linker flags are kept separate from the global flags, so
288  # they have to be added back temporarily for the following tests.
289  pgac_save_CPPFLAGS=$CPPFLAGS
290  pgac_save_LDFLAGS=$LDFLAGS
291  pgac_save_LIBS=$LIBS
292
293  CPPFLAGS="$CPPFLAGS $LIBCURL_CPPFLAGS"
294  LDFLAGS="$LDFLAGS $LIBCURL_LDFLAGS"
295
296  AC_CHECK_HEADER(curl/curl.h, [],
297				  [AC_MSG_ERROR([header file <curl/curl.h> is required for --with-libcurl])])
298
299  # LIBCURL_LDLIBS is determined here. Like the compiler flags, it should not
300  # pollute the global LIBS setting.
301  AC_CHECK_LIB(curl, curl_multi_init, [
302				 AC_DEFINE([HAVE_LIBCURL], [1], [Define to 1 if you have the `curl' library (-lcurl).])
303				 AC_SUBST(LIBCURL_LDLIBS, -lcurl)
304			   ],
305			   [AC_MSG_ERROR([library 'curl' does not provide curl_multi_init])])
306
307  LIBS="$LIBCURL_LDLIBS $LIBS"
308
309  # Check to see whether the current platform supports threadsafe Curl
310  # initialization.
311  AC_CACHE_CHECK([for curl_global_init thread safety], [pgac_cv__libcurl_threadsafe_init],
312  [AC_RUN_IFELSE([AC_LANG_PROGRAM([
313#include <curl/curl.h>
314],[
315    curl_version_info_data *info;
316
317    if (curl_global_init(CURL_GLOBAL_ALL))
318        return -1;
319
320    info = curl_version_info(CURLVERSION_NOW);
321#ifdef CURL_VERSION_THREADSAFE
322    if (info->features & CURL_VERSION_THREADSAFE)
323        return 0;
324#endif
325
326    return 1;
327])],
328  [pgac_cv__libcurl_threadsafe_init=yes],
329  [pgac_cv__libcurl_threadsafe_init=no],
330  [pgac_cv__libcurl_threadsafe_init=unknown])])
331  if test x"$pgac_cv__libcurl_threadsafe_init" = xyes ; then
332    AC_DEFINE(HAVE_THREADSAFE_CURL_GLOBAL_INIT, 1,
333              [Define to 1 if curl_global_init() is guaranteed to be thread-safe.])
334  fi
335
336  # Fail if a thread-friendly DNS resolver isn't built.
337  AC_CACHE_CHECK([for curl support for asynchronous DNS], [pgac_cv__libcurl_async_dns],
338  [AC_RUN_IFELSE([AC_LANG_PROGRAM([
339#include <curl/curl.h>
340],[
341    curl_version_info_data *info;
342
343    if (curl_global_init(CURL_GLOBAL_ALL))
344        return -1;
345
346    info = curl_version_info(CURLVERSION_NOW);
347    return (info->features & CURL_VERSION_ASYNCHDNS) ? 0 : 1;
348])],
349  [pgac_cv__libcurl_async_dns=yes],
350  [pgac_cv__libcurl_async_dns=no],
351  [pgac_cv__libcurl_async_dns=unknown])])
352  if test x"$pgac_cv__libcurl_async_dns" = xno ; then
353    AC_MSG_ERROR([
354*** The installed version of libcurl does not support asynchronous DNS
355*** lookups. Rebuild libcurl with the AsynchDNS feature enabled in order
356*** to use it with libpq.])
357  fi
358
359  CPPFLAGS=$pgac_save_CPPFLAGS
360  LDFLAGS=$pgac_save_LDFLAGS
361  LIBS=$pgac_save_LIBS
362])# PGAC_CHECK_LIBCURL
