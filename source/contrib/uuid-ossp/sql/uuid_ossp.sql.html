9-- some quick and dirty field extraction functions
10
11-- this is actually timestamp concatenated with clock sequence, per RFC 4122
12CREATE FUNCTION uuid_timestamp_bits(uuid) RETURNS varbit AS
13$$ SELECT ('x' || substr($1::text, 15, 4) || substr($1::text, 10, 4) ||
14           substr($1::text, 1, 8) || substr($1::text, 20, 4))::bit(80)
15          & x'0FFFFFFFFFFFFFFF3FFF' $$
16LANGUAGE SQL STRICT IMMUTABLE;
17
18CREATE FUNCTION uuid_version_bits(uuid) RETURNS varbit AS
19$$ SELECT ('x' || substr($1::text, 15, 2))::bit(8) & '11110000' $$
20LANGUAGE SQL STRICT IMMUTABLE;
21
22CREATE FUNCTION uuid_reserved_bits(uuid) RETURNS varbit AS
23$$ SELECT ('x' || substr($1::text, 20, 2))::bit(8) & '11000000' $$
24LANGUAGE SQL STRICT IMMUTABLE;
25
26CREATE FUNCTION uuid_multicast_bit(uuid) RETURNS bool AS
27$$ SELECT (('x' || substr($1::text, 25, 2))::bit(8) & '00000001') != '00000000' $$
28LANGUAGE SQL STRICT IMMUTABLE;
29
30CREATE FUNCTION uuid_local_admin_bit(uuid) RETURNS bool AS
31$$ SELECT (('x' || substr($1::text, 25, 2))::bit(8) & '00000010') != '00000000' $$
32LANGUAGE SQL STRICT IMMUTABLE;
33
34CREATE FUNCTION uuid_node(uuid) RETURNS text AS
35$$ SELECT substr($1::text, 25) $$
36LANGUAGE SQL STRICT IMMUTABLE;
37
38-- Ideally, the multicast bit would never be set in V1 output, but the
39-- UUID library may fall back to MC if it can't get the system MAC address.
40-- Also, the local-admin bit might be set (if so, we're probably inside a VM).
41-- So we can't test either bit here.
42SELECT uuid_version_bits(uuid_generate_v1()),
43       uuid_reserved_bits(uuid_generate_v1());
44
45-- Although RFC 4122 only requires the multicast bit to be set in V1MC style
46-- UUIDs, our implementation always sets the local-admin bit as well.
47SELECT uuid_version_bits(uuid_generate_v1mc()),
48       uuid_reserved_bits(uuid_generate_v1mc()),
49       uuid_multicast_bit(uuid_generate_v1mc()),
50       uuid_local_admin_bit(uuid_generate_v1mc());
51
52-- timestamp+clock sequence should be monotonic increasing in v1
53SELECT uuid_timestamp_bits(uuid_generate_v1()) < uuid_timestamp_bits(uuid_generate_v1());
54SELECT uuid_timestamp_bits(uuid_generate_v1mc()) < uuid_timestamp_bits(uuid_generate_v1mc());
55
56-- Ideally, the node value is stable in V1 addresses, but OSSP UUID
57-- falls back to V1MC behavior if it can't get the system MAC address.
58SELECT CASE WHEN uuid_multicast_bit(uuid_generate_v1()) AND
59                 uuid_local_admin_bit(uuid_generate_v1()) THEN
60         true -- punt, no test
61       ELSE
62         uuid_node(uuid_generate_v1()) = uuid_node(uuid_generate_v1())
63       END;
64
65-- In any case, V1MC node addresses should be random.
66SELECT uuid_node(uuid_generate_v1()) <> uuid_node(uuid_generate_v1mc());
67SELECT uuid_node(uuid_generate_v1mc()) <> uuid_node(uuid_generate_v1mc());
68
