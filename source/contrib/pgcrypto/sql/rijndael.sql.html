50-- empty data
51select encrypt('', 'foo', 'aes');
52-- 10 bytes key
53select encrypt('foo', '0123456789', 'aes');

55select encrypt('foo', '0123456789012345678901', 'aes');
56
57-- decrypt
58select encode(decrypt(encrypt('foo', '0123456', 'aes'), '0123456', 'aes'), 'escape');
59-- data not multiple of block size
60select encode(decrypt(encrypt('foo', '0123456', 'aes') || '\x00'::bytea, '0123456', 'aes'), 'escape');
61-- bad padding
62-- (The input value is the result of encrypt_iv('abcdefghijklmnopqrstuvwxyz', '0123456', 'abcd', 'aes')
63-- with the 16th byte changed (s/db/eb/) to corrupt the padding of the last block.)
64select encode(decrypt_iv('\xa21a9c15231465964e3396d32095e67eb52bab05f556a581621dee1b85385789', '0123456', 'abcd', 'aes'), 'escape');
65
66-- iv
67select encrypt_iv('foo', '0123456', 'abcd', 'aes');
68select encode(decrypt_iv('\x2c24cb7da91d6d5699801268b0f5adad', '0123456', 'abcd', 'aes'), 'escape');
69
70-- long message
71select encrypt('Lets try a longer message.', '0123456789', 'aes');
72select encode(decrypt(encrypt('Lets try a longer message.', '0123456789', 'aes'), '0123456789', 'aes'), 'escape');
