9-- We are supposed to pass in a 2-character salt.
10-- error since salt is too short:
11SELECT crypt('password', 'a');
12
13CREATE TABLE ctest (data text, res text, salt text);
14INSERT INTO ctest VALUES ('password', '', '');
15
16UPDATE ctest SET salt = gen_salt('des');
17UPDATE ctest SET res = crypt(data, salt);
18SELECT res = crypt(data, res) AS "worked"
19FROM ctest;
20
21-- check disabling of built in crypto functions
22SET pgcrypto.builtin_crypto_enabled = off;
23UPDATE ctest SET salt = gen_salt('des');
24UPDATE ctest SET res = crypt(data, salt);
25RESET pgcrypto.builtin_crypto_enabled;
26
27DROP TABLE ctest;
