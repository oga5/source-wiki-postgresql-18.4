37#define STEP  (16*1024)

39struct MBuf
40{
41	uint8	   *data;
42	uint8	   *data_end;
43	uint8	   *read_pos;
44	uint8	   *buf_end;
45	bool		no_write;
46	bool		own_data;
47};

49int
50mbuf_avail(MBuf *mbuf)
51{
52	return mbuf->data_end - mbuf->read_pos;
53}
54
55int
56mbuf_size(MBuf *mbuf)
57{
58	return mbuf->data_end - mbuf->data;
59}

61int
62mbuf_free(MBuf *mbuf)
63{
64	if (mbuf->own_data)
65	{
66		px_memset(mbuf->data, 0, mbuf->buf_end - mbuf->data);
67		pfree(mbuf->data);
68	}
69	pfree(mbuf);
70	return 0;
71}

73static void
74prepare_room(MBuf *mbuf, int block_len)
75{
76	uint8	   *newbuf;
77	unsigned	newlen;
78
79	if (mbuf->data_end + block_len <= mbuf->buf_end)
80		return;
81
82	newlen = (mbuf->buf_end - mbuf->data)
83		+ ((block_len + STEP + STEP - 1) & -STEP);
84
85	newbuf = repalloc(mbuf->data, newlen);
86
87	mbuf->buf_end = newbuf + newlen;
88	mbuf->data_end = newbuf + (mbuf->data_end - mbuf->data);
89	mbuf->read_pos = newbuf + (mbuf->read_pos - mbuf->data);
90	mbuf->data = newbuf;
91}

93int
94mbuf_append(MBuf *dst, const uint8 *buf, int len)
95{
96	if (dst->no_write)
97	{
98		px_debug("mbuf_append: no_write");
99		return PXE_BUG;
100	}
101
102	prepare_room(dst, len);
103
104	memcpy(dst->data_end, buf, len);
105	dst->data_end += len;
106
107	return 0;
108}

110MBuf *
111mbuf_create(int len)
112{
113	MBuf	   *mbuf;
114
115	if (!len)
116		len = 8192;
117
118	mbuf = palloc(sizeof *mbuf);
119	mbuf->data = palloc(len);
120	mbuf->buf_end = mbuf->data + len;
121	mbuf->data_end = mbuf->data;
122	mbuf->read_pos = mbuf->data;
123
124	mbuf->no_write = false;
125	mbuf->own_data = true;
126
127	return mbuf;
128}

148int
149mbuf_grab(MBuf *mbuf, int len, uint8 **data_p)
150{
151	if (len > mbuf_avail(mbuf))
152		len = mbuf_avail(mbuf);
153
154	mbuf->no_write = true;
155
156	*data_p = mbuf->read_pos;
157	mbuf->read_pos += len;
158	return len;
159}
160
161int
162mbuf_steal_data(MBuf *mbuf, uint8 **data_p)
163{
164	int			len = mbuf_size(mbuf);
165
166	mbuf->no_write = true;
167	mbuf->own_data = false;
168
169	*data_p = mbuf->data;
170
171	mbuf->data = mbuf->data_end = mbuf->read_pos = mbuf->buf_end = NULL;
172
173	return len;
174}

192{

194	void	   *priv;

196

200		if (res < 0)

213	pf->src = src;
214	if (pf->buflen > 0)
215	{

217		pf->pos = 0;

222		pf->pos = 0;

244/* may return less data than asked, 0 means eof */ view japanese
245int
246pullf_read(PullFilter *pf, int len, uint8 **data_p)
247{
248	int			res;
249
250	if (pf->op->pull)
251	{
252		if (pf->buflen && len > pf->buflen)
253			len = pf->buflen;
254		res = pf->op->pull(pf->priv, pf->src, len, data_p,
255						   pf->buf, pf->buflen);
256	}
257	else
258		res = pullf_read(pf->src, len, data_p);
259	return res;
260}

262int
263pullf_read_max(PullFilter *pf, int len, uint8 **data_p, uint8 *tmpbuf)
264{
265	int			res,
266				total;
267	uint8	   *tmp;
268
269	res = pullf_read(pf, len, data_p);
270	if (res <= 0 || res == len)
271		return res;
272
273	/* read was shorter, use tmpbuf */ view japanese
274	memcpy(tmpbuf, *data_p, res);
275	*data_p = tmpbuf;
276	len -= res;
277	total = res;
278
279	while (len > 0)
280	{
281		res = pullf_read(pf, len, &tmp);
282		if (res < 0)
283		{
284			/* so the caller must clear only on success */ view japanese
285			px_memset(tmpbuf, 0, total);
286			return res;
287		}
288		if (res == 0)
289			break;
290		memcpy(tmpbuf + total, tmp, res);
291		total += res;
292		len -= res;
293	}
294	return total;
295}

297/*
298 * caller wants exactly len bytes and don't bother with references
299 */ view japanese
300int
301pullf_read_fixed(PullFilter *src, int len, uint8 *dst)
302{
303	int			res;
304	uint8	   *p;
305
306	res = pullf_read_max(src, len, &p, dst);
307	if (res < 0)
308		return res;
309	if (res != len)
310	{
311		px_debug("pullf_read_fixed: need=%d got=%d", len, res);
312		return PXE_PGP_CORRUPT_DATA;
313	}
314	if (p != dst)
315		memcpy(dst, p, len);
316	return 0;
317}

346struct PushFilter
347{
348	PushFilter *next;
349	const PushFilterOps *op;
350	int			block_size;
351	uint8	   *buf;
352	int			pos;
353	void	   *priv;

356int
357pushf_create(PushFilter **mp_p, const PushFilterOps *op, void *init_arg, PushFilter *next)
358{
359	PushFilter *mp;
360	void	   *priv;
361	int			res;
362
363	if (op->init != NULL)
364	{
365		res = op->init(next, init_arg, &priv);
366		if (res < 0)
367			return res;
368	}
369	else
370	{
371		priv = init_arg;
372		res = 0;
373	}
374
375	mp = palloc0(sizeof(*mp));
376	mp->block_size = res;
377	mp->op = op;
378	mp->priv = priv;
379	mp->next = next;
380	if (mp->block_size > 0)
381	{
382		mp->buf = palloc(mp->block_size);
383		mp->pos = 0;
384	}
385	else
386	{
387		mp->buf = NULL;
388		mp->pos = 0;
389	}
390	*mp_p = mp;
391	return 0;
392}

423static int
424wrap_process(PushFilter *mp, const uint8 *data, int len)
425{
426	int			res;
427
428	if (mp->op->push != NULL)
429		res = mp->op->push(mp->next, mp->priv, data, len);
430	else
431		res = pushf_write(mp->next, data, len);
432	if (res > 0)
433		return PXE_BUG;
434	return res;
435}

437/* consumes all data, returns len on success */ view japanese
438int
439pushf_write(PushFilter *mp, const uint8 *data, int len)
440{
441	int			need,
442				res;
443
444	/*
445	 * no buffering
446	 */ view japanese
447	if (mp->block_size <= 0)
448		return wrap_process(mp, data, len);
449
450	/*
451	 * try to empty buffer
452	 */ view japanese
453	need = mp->block_size - mp->pos;
454	if (need > 0)
455	{
456		if (len < need)
457		{
458			memcpy(mp->buf + mp->pos, data, len);
459			mp->pos += len;
460			return 0;
461		}
462		memcpy(mp->buf + mp->pos, data, need);
463		len -= need;
464		data += need;
465	}
466
467	/*
468	 * buffer full, process
469	 */ view japanese
470	res = wrap_process(mp, mp->buf, mp->block_size);
471	if (res < 0)
472		return res;
473	mp->pos = 0;
474
475	/*
476	 * now process directly from data
477	 */ view japanese
478	while (len > 0)
479	{
480		if (len > mp->block_size)
481		{
482			res = wrap_process(mp, data, mp->block_size);
483			if (res < 0)
484				return res;
485			data += mp->block_size;
486			len -= mp->block_size;
487		}
488		else
489		{
490			memcpy(mp->buf, data, len);
491			mp->pos += len;
492			break;
493		}
494	}
495	return 0;
496}

498int
499pushf_flush(PushFilter *mp)
500{
501	int			res;
502
503	while (mp)
504	{
505		if (mp->block_size > 0)
506		{
507			res = wrap_process(mp, mp->buf, mp->pos);
508			if (res < 0)
509				return res;
510		}
511
512		if (mp->op->flush)
513		{
514			res = mp->op->flush(mp->next, mp->priv);
515			if (res < 0)
516				return res;
517		}
518
519		mp = mp->next;
520	}
521	return 0;
522}
