32#ifndef _PX_CRYPT_H
33#define _PX_CRYPT_H
34
35/* max room for result */
36#define PX_MAX_CRYPT  128
37
38/* max salt returned by gen_salt() */
39#define PX_MAX_SALT_LEN		128
40
41/* default rounds for xdes salt */
42/* NetBSD bin/passwd/local_passwd.c has (29 * 25)*/
43#define PX_XDES_ROUNDS		(29 * 25)
44
45/* default for blowfish salt */
46#define PX_BF_ROUNDS		6
47
48/* Maximum salt string length of shacrypt.  */
49#define PX_SHACRYPT_SALT_MAX_LEN 16
50
51/* SHA buffer length */
52#define PX_SHACRYPT_DIGEST_MAX_LEN 64
53
54/* calculated buffer size of a buffer to store a shacrypt salt string */
55#define PX_SHACRYPT_SALT_BUF_LEN (3 + 7 + 10 + PX_SHACRYPT_SALT_MAX_LEN + 1)
56
57/*
58 * calculated buffer size of a buffer to store complete result of a shacrypt
59 * digest including salt
60 */
61#define PX_SHACRYPT_BUF_LEN (PX_SHACRYPT_SALT_BUF_LEN + 86 + 1)
62
63/* Default number of rounds of shacrypt if not explicitly specified.  */
64#define PX_SHACRYPT_ROUNDS_DEFAULT 5000
65
66/* Minimum number of rounds of shacrypt.  */
67#define PX_SHACRYPT_ROUNDS_MIN 1000
68
69/* Maximum number of rounds of shacrypt.  */
70#define PX_SHACRYPT_ROUNDS_MAX 999999999
71
72/*
73 * main interface
74 */
75char	   *px_crypt(const char *psw, const char *salt, char *buf, unsigned len);
76int			px_gen_salt(const char *salt_type, char *buf, int rounds);
