3-- test jsonb -> python dict
4CREATE FUNCTION test1(val jsonb) RETURNS int
5LANGUAGE plpython3u
6TRANSFORM FOR TYPE jsonb
7AS $$
8assert isinstance(val, dict)
9assert(val == {'a': 1, 'c': 'NULL'})
10return len(val)
11$$;
12
13SELECT test1('{"a": 1, "c": "NULL"}'::jsonb);
