7-- Check whether any of our opclasses fail amvalidate
8-- ... they will, because of missing cross-type operators
9SELECT amname, opcname
10FROM (SELECT amname, opcname, opc.oid
11      FROM pg_opclass opc LEFT JOIN pg_am am ON am.oid = opcmethod
12      WHERE opc.oid >= 16384
13      ORDER BY 1, 2 OFFSET 0) ss
14WHERE NOT amvalidate(oid);

110-- test non-error-throwing input API
111SELECT str as isn, typ as "type",
112       pg_input_is_valid(str,typ) as ok,
113       errinfo.sql_error_code,
114       errinfo.message,
115       errinfo.detail,
116       errinfo.hint
117FROM (VALUES ('9780123456786', 'UPC'),
118             ('postgresql...','EAN13'),
119             ('9771234567003','ISSN'))
120      AS a(str,typ),
121     LATERAL pg_input_error_info(a.str, a.typ) as errinfo;
