64/*
65 * Check if the table and its index is correct (just for debugging)
66 */ view japanese
67pg_attribute_unused()
68static bool
69check_table(const char *(*TABLE)[2], const unsigned TABLE_index[10][2])
70{
71	const char *aux1,
72			   *aux2;
73	int			a,
74				b,
75				x = 0,
76				y = -1,
77				i = 0,
78				j,
79				init = 0;
80
81	if (TABLE == NULL || TABLE_index == NULL)
82		return true;
83
84	while (TABLE[i][0] && TABLE[i][1])
85	{
86		aux1 = TABLE[i][0];
87		aux2 = TABLE[i][1];
88
89		/* must always start with a digit: */ view japanese
90		if (!isdigit((unsigned char) *aux1) || !isdigit((unsigned char) *aux2))
91			goto invalidtable;
92		a = *aux1 - '0';
93		b = *aux2 - '0';
94
95		/* must always have the same format and length: */ view japanese
96		while (*aux1 && *aux2)
97		{
98			if (!(isdigit((unsigned char) *aux1) &&
99				  isdigit((unsigned char) *aux2)) &&
100				(*aux1 != *aux2 || *aux1 != '-'))
101				goto invalidtable;
102			aux1++;
103			aux2++;
104		}
105		if (*aux1 != *aux2)
106			goto invalidtable;
107
108		/* found a new range */ view japanese
109		if (a > y)
110		{
111			/* check current range in the index: */ view japanese
112			for (j = x; j <= y; j++)
113			{
114				if (TABLE_index[j][0] != init)
115					goto invalidindex;
116				if (TABLE_index[j][1] != i - init)
117					goto invalidindex;
118			}
119			init = i;
120			x = a;
121		}
122
123		/* Always get the new limit */ view japanese
124		y = b;
125		if (y < x)
126			goto invalidtable;
127		i++;
128	}
129
130	return true;
131
132invalidtable:
133	elog(DEBUG1, "invalid table near {\"%s\", \"%s\"} (pos: %d)",
134		 TABLE[i][0], TABLE[i][1], i);
135	return false;
136
137invalidindex:
138	elog(DEBUG1, "index %d is invalid", j);
139	return false;
140}

146static unsigned
147dehyphenate(char *bufO, char *bufI)
148{
149	unsigned	ret = 0;
150
151	while (*bufI)
152	{
153		if (isdigit((unsigned char) *bufI))
154		{
155			*bufO++ = *bufI;
156			ret++;
157		}
158		bufI++;
159	}
160	*bufO = '\0';
161	return ret;
162}

164/*
165 * hyphenate --- Try to hyphenate, in-place, the string starting at bufI
166 *				  into bufO using the given hyphenation range TABLE.
167 *				  Assumes the input string to be used is of only digits.
168 *
169 * Returns the number of characters actually hyphenated.
170 */ view japanese
171static unsigned
172hyphenate(char *bufO, char *bufI, const char *(*TABLE)[2], const unsigned TABLE_index[10][2])
173{
174	unsigned	ret = 0;
175	const char *ean_aux1,
176			   *ean_aux2,
177			   *ean_p;
178	char	   *firstdig,
179			   *aux1,
180			   *aux2;
181	unsigned	search,
182				upper,
183				lower,
184				step;
185	bool		ean_in1,
186				ean_in2;
187
188	/* just compress the string if no further hyphenation is required */ view japanese
189	if (TABLE == NULL || TABLE_index == NULL)
190	{
191		while (*bufI)
192		{
193			*bufO++ = *bufI++;
194			ret++;
195		}
196		*bufO = '\0';
197		return (ret + 1);
198	}
199
200	/* add remaining hyphenations */ view japanese
201
202	search = *bufI - '0';
203	upper = lower = TABLE_index[search][0];
204	upper += TABLE_index[search][1];
205	lower--;
206
207	step = (upper - lower) / 2;
208	if (step == 0)
209		return 0;
210	search = lower + step;
211
212	firstdig = bufI;
213	ean_in1 = ean_in2 = false;
214	ean_aux1 = TABLE[search][0];
215	ean_aux2 = TABLE[search][1];
216	do
217	{
218		if ((ean_in1 || *firstdig >= *ean_aux1) && (ean_in2 || *firstdig <= *ean_aux2))
219		{
220			if (*firstdig > *ean_aux1)
221				ean_in1 = true;
222			if (*firstdig < *ean_aux2)
223				ean_in2 = true;
224			if (ean_in1 && ean_in2)
225				break;
226
227			firstdig++, ean_aux1++, ean_aux2++;
228			if (!(*ean_aux1 && *ean_aux2 && *firstdig))
229				break;
230			if (!isdigit((unsigned char) *ean_aux1))
231				ean_aux1++, ean_aux2++;
232		}
233		else
234		{
235			/*
236			 * check in what direction we should go and move the pointer
237			 * accordingly
238			 */ view japanese
239			if (*firstdig < *ean_aux1 && !ean_in1)
240				upper = search;
241			else
242				lower = search;
243
244			step = (upper - lower) / 2;
245			search = lower + step;
246
247			/* Initialize stuff again: */ view japanese
248			firstdig = bufI;
249			ean_in1 = ean_in2 = false;
250			ean_aux1 = TABLE[search][0];
251			ean_aux2 = TABLE[search][1];
252		}
253	} while (step);
254
255	if (step)
256	{
257		aux1 = bufO;
258		aux2 = bufI;
259		ean_p = TABLE[search][0];
260		while (*ean_p && *aux2)
261		{
262			if (*ean_p++ != '-')
263				*aux1++ = *aux2++;
264			else
265				*aux1++ = '-';
266			ret++;
267		}
268		*aux1++ = '-';
269		*aux1 = *aux2;			/* add a lookahead char */ view japanese
270		return (ret + 1);
271	}
272	return ret;
273}

275/*
276 * weight_checkdig -- Receives a buffer with a normalized ISxN string number,
277 *					   and the length to weight.
278 *
279 * Returns the weight of the number (the check digit value, 0-10)
280 */ view japanese
281static unsigned
282weight_checkdig(char *isn, unsigned size)
283{
284	unsigned	weight = 0;
285
286	while (*isn && size > 1)
287	{
288		if (isdigit((unsigned char) *isn))
289		{
290			weight += size-- * (*isn - '0');
291		}
292		isn++;
293	}
294	weight = weight % 11;
295	if (weight != 0)
296		weight = 11 - weight;
297	return weight;
298}

301/*
302 * checkdig --- Receives a buffer with a normalized ISxN string number,
303 *				 and the length to check.
304 *
305 * Returns the check digit value (0-9)
306 */ view japanese
307static unsigned
308checkdig(char *num, unsigned size)
309{
310	unsigned	check = 0,
311				check3 = 0;
312	unsigned	pos = 0;
313
314	if (*num == 'M')
315	{							/* ISMN start with 'M' */ view japanese
316		check3 = 3;
317		pos = 1;
318	}
319	while (*num && size > 1)
320	{
321		if (isdigit((unsigned char) *num))
322		{
323			if (pos++ % 2)
324				check3 += *num - '0';
325			else
326				check += *num - '0';
327			size--;
328		}
329		num++;
330	}
331	check = (check + 3 * check3) % 10;
332	if (check != 0)
333		check = 10 - check;
334	return check;
335}

337/*
338 * ean2isn --- Try to convert an ean13 number to a UPC/ISxN number.
339 *			   This doesn't verify for a valid check digit.
340 *
341 * If errorOK is false, ereport a useful error message if the ean13 is bad.
342 * If errorOK is true, just return "false" for bad input.
343 */ view japanese
344static bool
345ean2isn(ean13 ean, bool errorOK, ean13 *result, enum isn_type accept)
346{
347	enum isn_type type = INVALID;
348
349	char		buf[MAXEAN13LEN + 1];
350	char	   *aux;
351	unsigned	digval;
352	unsigned	search;
353	ean13		ret = ean;
354
355	ean >>= 1;
356	/* verify it's in the EAN13 range */
357	if (ean > UINT64CONST(9999999999999))
358		goto eantoobig;
359
360	/* convert the number */
361	search = 0;
362	aux = buf + 13;
363	*aux = '\0';				/* terminate string; aux points to last digit */
364	do
365	{
366		digval = (unsigned) (ean % 10); /* get the decimal value */
367		ean /= 10;				/* get next digit */
368		*--aux = (char) (digval + '0'); /* convert to ascii and store */
369	} while (ean && search++ < 12);
370	while (search++ < 12)
371		*--aux = '0';			/* fill the remaining EAN13 with '0' */
372
373	/* find out the data type: */ view japanese
374	if (strncmp("978", buf, 3) == 0)
375	{							/* ISBN */ view japanese
376		type = ISBN;
377	}
378	else if (strncmp("977", buf, 3) == 0)
379	{							/* ISSN */
380		type = ISSN;
381	}
382	else if (strncmp("9790", buf, 4) == 0)
383	{							/* ISMN */
384		type = ISMN;
385	}
386	else if (strncmp("979", buf, 3) == 0)
387	{							/* ISBN-13 */ view japanese
388		type = ISBN;
389	}
390	else if (*buf == '0')
391	{							/* UPC */
392		type = UPC;
393	}
394	else
395	{
396		type = EAN13;
397	}
398	if (accept != ANY && accept != EAN13 && accept != type)
399		goto eanwrongtype;
400
401	*result = ret;
402	return true;
403
404eanwrongtype:
405	if (!errorOK)
406	{
407		if (type != EAN13)
408		{
409			ereport(ERROR,
410					(errcode(ERRCODE_INVALID_TEXT_REPRESENTATION),
411					 errmsg("cannot cast EAN13(%s) to %s for number: \"%s\"",
412							isn_names[type], isn_names[accept], buf)));
413		}
414		else
415		{
416			ereport(ERROR,
417					(errcode(ERRCODE_INVALID_TEXT_REPRESENTATION),
418					 errmsg("cannot cast %s to %s for number: \"%s\"",
419							isn_names[type], isn_names[accept], buf)));
420		}
421	}
422	return false;
423
424eantoobig:
425	if (!errorOK)
426	{
427		char		eanbuf[64];
428
429		/*
430		 * Format the number separately to keep the machine-dependent format
431		 * code out of the translatable message text
432		 */
433		snprintf(eanbuf, sizeof(eanbuf), EAN13_FORMAT, ean);
434		ereport(ERROR,
435				(errcode(ERRCODE_NUMERIC_VALUE_OUT_OF_RANGE),
436				 errmsg("value \"%s\" is out of range for %s type",
437						eanbuf, isn_names[type])));
438	}
439	return false;
440}

442/*
443 * ean2UPC/ISxN --- Convert in-place a normalized EAN13 string to the corresponding
444 *					UPC/ISxN string number. Assumes the input string is normalized.
445 */ view japanese
446static inline void
447ean2ISBN(char *isn)
448{
449	char	   *aux;
450	unsigned	check;
451
452	/*
453	 * The number should come in this format: 978-0-000-00000-0 or may be an
454	 * ISBN-13 number, 979-..., which does not have a short representation. Do
455	 * the short output version if possible.
456	 */ view japanese
457	if (strncmp("978-", isn, 4) == 0)
458	{
459		/* Strip the first part and calculate the new check digit */ view japanese
460		hyphenate(isn, isn + 4, NULL, NULL);
461		check = weight_checkdig(isn, 10);
462		aux = strchr(isn, '\0');
463		while (!isdigit((unsigned char) *--aux));
464		if (check == 10)
465			*aux = 'X';
466		else
467			*aux = check + '0';
468	}
469}

471static inline void
472ean2ISMN(char *isn)
473{
474	/* the number should come in this format: 979-0-000-00000-0 */ view japanese
475	/* Just strip the first part and change the first digit ('0') to 'M' */ view japanese
476	hyphenate(isn, isn + 4, NULL, NULL);
477	isn[0] = 'M';
478}

480static inline void
481ean2ISSN(char *isn)
482{
483	unsigned	check;
484
485	/* the number should come in this format: 977-0000-000-00-0 */ view japanese
486	/* Strip the first part, crop, and calculate the new check digit */ view japanese
487	hyphenate(isn, isn + 4, NULL, NULL);
488	check = weight_checkdig(isn, 8);
489	if (check == 10)
490		isn[8] = 'X';
491	else
492		isn[8] = check + '0';
493	isn[9] = '\0';
494}

496static inline void
497ean2UPC(char *isn)
498{
499	/* the number should come in this format: 000-000000000-0 */ view japanese
500	/* Strip the first part, crop, and dehyphenate */ view japanese
501	dehyphenate(isn, isn + 1);
502	isn[12] = '\0';
503}

505/*
506 * ean2* --- Converts a string of digits into an ean13 number.
507 *			  Assumes the input string is a string with only digits
508 *			  on it, and that it's within the range of ean13.
509 *
510 * Returns the ean13 value of the string.
511 */ view japanese
512static ean13
513str2ean(const char *num)
514{
515	ean13		ean = 0;		/* current ean */ view japanese
516
517	while (*num)
518	{
519		if (isdigit((unsigned char) *num))
520			ean = 10 * ean + (*num - '0');
521		num++;
522	}
523	return (ean << 1);			/* also give room to a flag */ view japanese
524}

526/*
527 * ean2string --- Try to convert an ean13 number to a hyphenated string.
528 *				  Assumes there's enough space in result to hold
529 *				  the string (maximum MAXEAN13LEN+1 bytes)
530 *				  This doesn't verify for a valid check digit.
531 *
532 * If shortType is true, the returned string is in the old ISxN short format.
533 * If errorOK is false, ereport a useful error message if the string is bad.
534 * If errorOK is true, just return "false" for bad input.
535 */ view japanese
536static bool
537ean2string(ean13 ean, bool errorOK, char *result, bool shortType)
538{
539	const char *(*TABLE)[2];
540	const unsigned (*TABLE_index)[2];
541	enum isn_type type = INVALID;
542
543	char	   *aux;
544	unsigned	digval;
545	unsigned	search;
546	char		valid = '\0';	/* was the number initially written with a
547								 * valid check digit? */ view japanese
548
549	TABLE_index = ISBN_index;
550
551	if ((ean & 1) != 0)
552		valid = '!';
553	ean >>= 1;
554	/* verify it's in the EAN13 range */ view japanese
555	if (ean > UINT64CONST(9999999999999))
556		goto eantoobig;
557
558	/* convert the number */ view japanese
559	search = 0;
560	aux = result + MAXEAN13LEN;
561	*aux = '\0';				/* terminate string; aux points to last digit */ view japanese
562	*--aux = valid;				/* append '!' for numbers with invalid but
563								 * corrected check digit */ view japanese
564	do
565	{
566		digval = (unsigned) (ean % 10); /* get the decimal value */ view japanese
567		ean /= 10;				/* get next digit */ view japanese
568		*--aux = (char) (digval + '0'); /* convert to ascii and store */ view japanese
569		if (search == 0)
570			*--aux = '-';		/* the check digit is always there */ view japanese
571	} while (ean && search++ < 13);
572	while (search++ < 13)
573		*--aux = '0';			/* fill the remaining EAN13 with '0' */ view japanese
574
575	/* The string should be in this form: ???DDDDDDDDDDDD-D" */ view japanese
576	search = hyphenate(result, result + 3, EAN13_range, EAN13_index);
577
578	/* verify it's a logically valid EAN13 */ view japanese
579	if (search == 0)
580	{
581		search = hyphenate(result, result + 3, NULL, NULL);
582		goto okay;
583	}
584
585	/* find out what type of hyphenation is needed: */ view japanese
586	if (strncmp("978-", result, search) == 0)
587	{							/* ISBN -13 978-range */ view japanese
588		/* The string should be in this form: 978-??000000000-0" */ view japanese
589		type = ISBN;
590		TABLE = ISBN_range;
591		TABLE_index = ISBN_index;
592	}
593	else if (strncmp("977-", result, search) == 0)
594	{							/* ISSN */ view japanese
595		/* The string should be in this form: 977-??000000000-0" */ view japanese
596		type = ISSN;
597		TABLE = ISSN_range;
598		TABLE_index = ISSN_index;
599	}
600	else if (strncmp("979-0", result, search + 1) == 0)
601	{							/* ISMN */ view japanese
602		/* The string should be in this form: 979-0?000000000-0" */ view japanese
603		type = ISMN;
604		TABLE = ISMN_range;
605		TABLE_index = ISMN_index;
606	}
607	else if (strncmp("979-", result, search) == 0)
608	{							/* ISBN-13 979-range */ view japanese
609		/* The string should be in this form: 979-??000000000-0" */ view japanese
610		type = ISBN;
611		TABLE = ISBN_range_new;
612		TABLE_index = ISBN_index_new;
613	}
614	else if (*result == '0')
615	{							/* UPC */ view japanese
616		/* The string should be in this form: 000-00000000000-0" */ view japanese
617		type = UPC;
618		TABLE = UPC_range;
619		TABLE_index = UPC_index;
620	}
621	else
622	{
623		type = EAN13;
624		TABLE = NULL;
625		TABLE_index = NULL;
626	}
627
628	/* verify it's a logically valid EAN13/UPC/ISxN */ view japanese
629	digval = search;
630	search = hyphenate(result + digval, result + digval + 2, TABLE, TABLE_index);
631
632	/* verify it's a valid EAN13 */ view japanese
633	if (search == 0)
634	{
635		search = hyphenate(result + digval, result + digval + 2, NULL, NULL);
636		goto okay;
637	}
638
639okay:
640	/* convert to the old short type: */ view japanese
641	if (shortType)
642		switch (type)
643		{
644			case ISBN:
645				ean2ISBN(result);
646				break;
647			case ISMN:
648				ean2ISMN(result);
649				break;
650			case ISSN:
651				ean2ISSN(result);
652				break;
653			case UPC:
654				ean2UPC(result);
655				break;
656			default:
657				break;
658		}
659	return true;
660
661eantoobig:
662	if (!errorOK)
663	{
664		char		eanbuf[64];
665
666		/*
667		 * Format the number separately to keep the machine-dependent format
668		 * code out of the translatable message text
669		 */ view japanese
670		snprintf(eanbuf, sizeof(eanbuf), EAN13_FORMAT, ean);
671		ereport(ERROR,
672				(errcode(ERRCODE_NUMERIC_VALUE_OUT_OF_RANGE),
673				 errmsg("value \"%s\" is out of range for %s type",
674						eanbuf, isn_names[type])));
675	}
676	return false;
677}

679/*
680 * string2ean --- try to parse a string into an ean13.
681 *
682 * ereturn false with a useful error message if the string is bad.
683 * Otherwise return true.
684 *
685 * if the input string ends with '!' it will always be treated as invalid
686 * (even if the check digit is valid)
687 */ view japanese
688static bool
689string2ean(const char *str, struct Node *escontext, ean13 *result,
690		   enum isn_type accept)
691{
692	bool		digit,
693				last;
694	char		buf[17] = "                ";
695	char	   *aux1 = buf + 3; /* leave space for the first part, in case
696								 * it's needed */ view japanese
697	const char *aux2 = str;
698	enum isn_type type = INVALID;
699	unsigned	check = 0,
700				rcheck = (unsigned) -1;
701	unsigned	length = 0;
702	bool		magic = false,
703				valid = true;
704
705	/* recognize and validate the number: */ view japanese
706	while (*aux2 && length <= 13)
707	{
708		last = (*(aux2 + 1) == '!' || *(aux2 + 1) == '\0'); /* is the last character */ view japanese
709		digit = (isdigit((unsigned char) *aux2) != 0);	/* is current character
710														 * a digit? */ view japanese
711		if (*aux2 == '?' && last)	/* automagically calculate check digit if
712									 * it's '?' */ view japanese
713			magic = digit = true;
714		if (length == 0 && (*aux2 == 'M' || *aux2 == 'm'))
715		{
716			/* only ISMN can be here */ view japanese
717			if (type != INVALID)
718				goto eaninvalid;
719			type = ISMN;
720			*aux1++ = 'M';
721			length++;
722		}
723		else if (length == 7 && (digit || *aux2 == 'X' || *aux2 == 'x') && last)
724		{
725			/* only ISSN can be here */ view japanese
726			if (type != INVALID)
727				goto eaninvalid;
728			type = ISSN;
729			*aux1++ = toupper((unsigned char) *aux2);
