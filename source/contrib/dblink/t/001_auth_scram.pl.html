3# Test SCRAM authentication when opening a new connection with a foreign
4# server.
5#
6# The test is executed by testing the SCRAM authentifcation on a loopback
7# connection on the same server and with different servers.
8
9use strict;
10use warnings FATAL => 'all';
11use PostgreSQL::Test::Utils;
12use PostgreSQL::Test::Cluster;
13use Test::More;
14
15if (!$use_unix_sockets)
16{
17	plan skip_all => "test requires Unix-domain sockets";
18}
19
20my $user = "user01";
21
22my $db0 = "db0";                               # For node1
23my $db1 = "db1";                               # For node1
24my $db2 = "db2";                               # For node2
25my $fdw_server = "db1_fdw";
26my $fdw_server2 = "db2_fdw";
27my $fdw_invalid_server = "db2_fdw_invalid";    # For invalid fdw options
28my $fdw_invalid_server2 =
29  "db2_fdw_invalid2";    # For invalid scram keys fdw options
30
31my $node1 = PostgreSQL::Test::Cluster->new('node1');
32my $node2 = PostgreSQL::Test::Cluster->new('node2');
33
34$node1->init;
35$node2->init;
36
37$node1->start;
38$node2->start;
39
40# Test setup
41
42$node1->safe_psql('postgres', qq'CREATE USER $user WITH password \'pass\'');
43$node2->safe_psql('postgres', qq'CREATE USER $user WITH password \'pass\'');
44$ENV{PGPASSWORD} = "pass";
45
46$node1->safe_psql('postgres', qq'CREATE DATABASE $db0');
47$node1->safe_psql('postgres', qq'CREATE DATABASE $db1');
48$node2->safe_psql('postgres', qq'CREATE DATABASE $db2');
49
50setup_table($node1, $db1, "t");
51setup_table($node2, $db2, "t2");
52
53$node1->safe_psql($db0, 'CREATE EXTENSION IF NOT EXISTS dblink');
54setup_fdw_server($node1, $db0, $fdw_server, $node1, $db1);
55setup_fdw_server($node1, $db0, $fdw_server2, $node2, $db2);
56setup_invalid_fdw_server($node1, $db0, $fdw_invalid_server, $node2, $db2);
57setup_fdw_server($node1, $db0, $fdw_invalid_server2, $node2, $db2);
58
59setup_user_mapping($node1, $db0, $fdw_server);
60setup_user_mapping($node1, $db0, $fdw_server2);
61setup_user_mapping($node1, $db0, $fdw_invalid_server);
62
63# Make the user have the same SCRAM key on both servers. Forcing to have the
64# same iteration and salt.
65my $rolpassword = $node1->safe_psql('postgres',
66	qq"SELECT rolpassword FROM pg_authid WHERE rolname = '$user';");
67$node2->safe_psql('postgres', qq"ALTER ROLE $user PASSWORD '$rolpassword'");
68
69unlink($node1->data_dir . '/pg_hba.conf');
70unlink($node2->data_dir . '/pg_hba.conf');
71
72$node1->append_conf(
73	'pg_hba.conf', qq{
74local   db0             all                                     scram-sha-256
75local   db1             all                                     scram-sha-256
76}
77);
78$node2->append_conf(
79	'pg_hba.conf', qq{
80local   db2             all                                     scram-sha-256
81}
82);
83
84$node1->restart;
85$node2->restart;
86
87# End of test setup
88
89test_scram_keys_is_not_overwritten($node1, $db0, $fdw_invalid_server2);
90
91test_fdw_auth($node1, $db0, "t", $fdw_server,
92	"SCRAM auth on the same database cluster must succeed");
93
94test_fdw_auth($node1, $db0, "t2", $fdw_server2,
95	"SCRAM auth on a different database cluster must succeed");
96
97test_fdw_auth_with_invalid_overwritten_require_auth($fdw_invalid_server);
98
99# Ensure that trust connections fail without superuser opt-in.
100unlink($node1->data_dir . '/pg_hba.conf');
101unlink($node2->data_dir . '/pg_hba.conf');
102
103$node1->append_conf(
104	'pg_hba.conf', qq{
105local   db0             all                                     scram-sha-256
106local   db1             all                                     trust
107}
108);
109$node2->append_conf(
110	'pg_hba.conf', qq{
111local   all             all                                     password
112}
113);
114
115$node1->restart;
116$node2->restart;
117
118my ($ret, $stdout, $stderr) = $node1->psql(
119	$db0,
120	"SELECT * FROM dblink('$fdw_server', 'SELECT * FROM t') AS t(a int, b int)",
121	connstr => $node1->connstr($db0) . " user=$user");
122
123is($ret, 3, 'loopback trust fails on the same cluster');
124like(
125	$stderr,
126	qr/failed: authentication method requirement "scram-sha-256" failed: server did not complete authentication/,
127	'expected error from loopback trust (same cluster)');
128
129($ret, $stdout, $stderr) = $node1->psql(
130	$db0,
131	"SELECT * FROM dblink('$fdw_server2', 'SELECT * FROM t2') AS t2(a int, b int)",
132	connstr => $node1->connstr($db0) . " user=$user");
133
134is($ret, 3, 'loopback password fails on a different cluster');
135like(
136	$stderr,
137	qr/authentication method requirement "scram-sha-256" failed: server requested a cleartext password/,
138	'expected error from loopback password (different cluster)');

140# Helper functions

142sub test_fdw_auth
143{
144	local $Test::Builder::Level = $Test::Builder::Level + 1;
145
146	my ($node, $db, $tbl, $fdw, $testname) = @_;
147	my $connstr = $node->connstr($db) . qq' user=$user';
148
149	my $ret = $node->safe_psql(
150		$db,
151		qq"SELECT count(1) FROM dblink('$fdw', 'SELECT * FROM $tbl') AS $tbl(a int, b int)",
152		connstr => $connstr);
153
154	is($ret, '10', $testname);
155}
156
157sub test_fdw_auth_with_invalid_overwritten_require_auth
158{
159	local $Test::Builder::Level = $Test::Builder::Level + 1;
160
161	my ($fdw) = @_;
162
163	my ($ret, $stdout, $stderr) = $node1->psql(
164		$db0,
165		"select * from dblink('$fdw', 'select * from t') as t(a int, b int)",
166		connstr => $node1->connstr($db0) . " user=$user");
167
168	is($ret, 3, 'loopback trust fails when overwriting require_auth');
169	like(
170		$stderr,
171		qr/password or GSSAPI delegated credentials required/,
172		'expected error when connecting to a fdw overwriting the require_auth'
173	);
174}
175
176sub test_scram_keys_is_not_overwritten
177{
178	local $Test::Builder::Level = $Test::Builder::Level + 1;
179
180	my ($node, $db, $fdw) = @_;
181
182	my ($ret, $stdout, $stderr) = $node->psql(
183		$db,
184		qq'CREATE USER MAPPING FOR $user SERVER $fdw OPTIONS (user \'$user\', scram_client_key \'key\');',
185		connstr => $node->connstr($db) . " user=$user");
186
187	is($ret, 3, 'user mapping creation fails when using scram_client_key');
188	like(
189		$stderr,
190		qr/ERROR:  invalid option "scram_client_key"/,
191		'user mapping creation fails when using scram_client_key');
192
193	($ret, $stdout, $stderr) = $node->psql(
194		$db,
195		qq'CREATE USER MAPPING FOR $user SERVER $fdw OPTIONS (user \'$user\', scram_server_key \'key\');',
196		connstr => $node->connstr($db) . " user=$user");
197
198	is($ret, 3, 'user mapping creation fails when using scram_server_key');
199	like(
200		$stderr,
201		qr/ERROR:  invalid option "scram_server_key"/,
202		'user mapping creation fails when using scram_server_key');
203}

205sub setup_user_mapping
206{
207	my ($node, $db, $fdw) = @_;
208
209	$node->safe_psql($db,
210		qq'CREATE USER MAPPING FOR $user SERVER $fdw OPTIONS (user \'$user\');'
211	);
212}
213
214sub setup_fdw_server
215{
216	my ($node, $db, $fdw, $fdw_node, $dbname) = @_;
217	my $host = $fdw_node->host;
218	my $port = $fdw_node->port;
219
220	$node->safe_psql(
221		$db, qq'CREATE SERVER $fdw FOREIGN DATA WRAPPER dblink_fdw options (
222		host \'$host\', port \'$port\', dbname \'$dbname\', use_scram_passthrough \'true\') '
223	);
224	$node->safe_psql($db, qq'GRANT USAGE ON FOREIGN SERVER $fdw TO $user;');
225	$node->safe_psql($db, qq'GRANT ALL ON SCHEMA public TO $user');
226}
227
228sub setup_invalid_fdw_server
229{
230	my ($node, $db, $fdw, $fdw_node, $dbname) = @_;
231	my $host = $fdw_node->host;
232	my $port = $fdw_node->port;
233
234	$node->safe_psql(
235		$db, qq'CREATE SERVER $fdw FOREIGN DATA WRAPPER dblink_fdw options (
236		host \'$host\', port \'$port\', dbname \'$dbname\', use_scram_passthrough \'true\', require_auth \'none\') '
237	);
238	$node->safe_psql($db, qq'GRANT USAGE ON FOREIGN SERVER $fdw TO $user;');
239	$node->safe_psql($db, qq'GRANT ALL ON SCHEMA public TO $user');
240}
241
242sub setup_table
243{
244	my ($node, $db, $tbl) = @_;
245
246	$node->safe_psql($db,
247		qq'CREATE TABLE $tbl AS SELECT g as a, g + 1 as b FROM generate_series(1,10) g(g)'
248	);
249	$node->safe_psql($db, qq'GRANT USAGE ON SCHEMA public TO $user');
250	$node->safe_psql($db, qq'GRANT SELECT ON $tbl TO $user');
251}
