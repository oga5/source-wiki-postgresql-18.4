34Datum
35gbt_var_decompress(PG_FUNCTION_ARGS)
36{
37	GISTENTRY  *entry = (GISTENTRY *) PG_GETARG_POINTER(0);
38	GBT_VARKEY *key = (GBT_VARKEY *) PG_DETOAST_DATUM(entry->key);
39
40	if (key != (GBT_VARKEY *) DatumGetPointer(entry->key))
41	{
42		GISTENTRY  *retval = (GISTENTRY *) palloc(sizeof(GISTENTRY));
43
44		gistentryinit(*retval, PointerGetDatum(key),
45					  entry->rel, entry->page,
46					  entry->offset, false);
47
48		PG_RETURN_POINTER(retval);
49	}
50
51	PG_RETURN_POINTER(entry);
52}

54/* Returns a better readable representation of variable key ( sets pointer ) */
55GBT_VARKEY_R
56gbt_var_key_readable(const GBT_VARKEY *k)
57{
58	GBT_VARKEY_R r;
59
60	r.lower = (bytea *) &(((char *) k)[VARHDRSZ]);
61	if (VARSIZE(k) > (VARHDRSZ + (VARSIZE(r.lower))))
62		r.upper = (bytea *) &(((char *) k)[VARHDRSZ + INTALIGN(VARSIZE(r.lower))]);
63	else
64		r.upper = r.lower;
65	return r;
66}

69/*
70 * Create a leaf-entry to store in the index, from a single Datum.
71 */
72static GBT_VARKEY *
73gbt_var_key_from_datum(const struct varlena *u)
74{
75	int32		lowersize = VARSIZE(u);
76	GBT_VARKEY *r;
77
78	r = (GBT_VARKEY *) palloc(lowersize + VARHDRSZ);
79	memcpy(VARDATA(r), u, lowersize);
80	SET_VARSIZE(r, lowersize + VARHDRSZ);
81
82	return r;
83}

85/*
86 * Create an entry to store in the index, from lower and upper bound.
87 */
88GBT_VARKEY *
89gbt_var_key_copy(const GBT_VARKEY_R *u)
90{
91	int32		lowersize = VARSIZE(u->lower);
92	int32		uppersize = VARSIZE(u->upper);
93	GBT_VARKEY *r;
94
95	r = (GBT_VARKEY *) palloc0(INTALIGN(lowersize) + uppersize + VARHDRSZ);
96	memcpy(VARDATA(r), u->lower, lowersize);
97	memcpy(VARDATA(r) + INTALIGN(lowersize), u->upper, uppersize);
98	SET_VARSIZE(r, INTALIGN(lowersize) + uppersize + VARHDRSZ);
99
100	return r;
101}

116/*
117 * returns the common prefix length of a node key
118 *
119 * If the underlying type is character data, the prefix length may point in
120 * the middle of a multibyte character.
121*/
122static int32
123gbt_var_node_cp_len(const GBT_VARKEY *node, const gbtree_vinfo *tinfo)
124{
125	GBT_VARKEY_R r = gbt_var_key_readable(node);
126	int32		i = 0;
127	int32		l_left_to_match = 0;
128	int32		l_total = 0;
129	int32		t1len = VARSIZE(r.lower) - VARHDRSZ;
130	int32		t2len = VARSIZE(r.upper) - VARHDRSZ;
131	int32		ml = Min(t1len, t2len);
132	char	   *p1 = VARDATA(r.lower);
133	char	   *p2 = VARDATA(r.upper);
134	const char *end1 = p1 + t1len;
135	const char *end2 = p2 + t2len;
136
137	if (ml == 0)
138		return 0;
139
140	while (i < ml)
141	{
142		if (tinfo->eml > 1 && l_left_to_match == 0)
143		{
144			l_total = pg_mblen_range(p1, end1);
145			if (l_total != pg_mblen_range(p2, end2))
146			{
147				return i;
148			}
149			l_left_to_match = l_total;
150		}
151		if (*p1 != *p2)
152		{
153			if (tinfo->eml > 1)
154			{
155				int32		l_matched_subset = l_total - l_left_to_match;
156
157				/* end common prefix at final byte of last matching char */
158				return i - l_matched_subset;
159			}
160			else
161			{
162				return i;
163			}
164		}
165
166		p1++;
167		p2++;
168		l_left_to_match--;
169		i++;
170	}
171	return ml;					/* lower == upper */
172}

175/*
176 * returns true, if query matches prefix ( common prefix )
177 */
178static bool
179gbt_bytea_pf_match(const bytea *pf, const bytea *query, const gbtree_vinfo *tinfo)
180{
181	bool		out = false;
182	int32		qlen = VARSIZE(query) - VARHDRSZ;
183	int32		nlen = VARSIZE(pf) - VARHDRSZ;
184
185	if (nlen <= qlen)
186	{
187		char	   *q = VARDATA(query);
188		char	   *n = VARDATA(pf);
189
190		out = (memcmp(q, n, nlen) == 0);
191	}
192
193	return out;
194}

197/*
198 * returns true, if query matches node using common prefix
199 */
200static bool
201gbt_var_node_pf_match(const GBT_VARKEY_R *node, const bytea *query, const gbtree_vinfo *tinfo)
202{
203	return (tinfo->trnc &&
204			(gbt_bytea_pf_match(node->lower, query, tinfo) ||
205			 gbt_bytea_pf_match(node->upper, query, tinfo)));
206}

209/*
210*  truncates / compresses the node key
211*  cpf_length .. common prefix length
212*/
213static GBT_VARKEY *
214gbt_var_node_truncate(const GBT_VARKEY *node, int32 cpf_length, const gbtree_vinfo *tinfo)
215{
216	GBT_VARKEY *out = NULL;
217	GBT_VARKEY_R r = gbt_var_key_readable(node);
218	int32		len1 = VARSIZE(r.lower) - VARHDRSZ;
219	int32		len2 = VARSIZE(r.upper) - VARHDRSZ;
220	int32		si;
221	char	   *out2;
222
223	len1 = Min(len1, (cpf_length + 1));
224	len2 = Min(len2, (cpf_length + 1));
225
226	si = 2 * VARHDRSZ + INTALIGN(len1 + VARHDRSZ) + len2;
227	out = (GBT_VARKEY *) palloc0(si);
228	SET_VARSIZE(out, si);
229
230	memcpy(VARDATA(out), r.lower, len1 + VARHDRSZ);
231	SET_VARSIZE(VARDATA(out), len1 + VARHDRSZ);
232
233	out2 = VARDATA(out) + INTALIGN(len1 + VARHDRSZ);
234	memcpy(out2, r.upper, len2 + VARHDRSZ);
235	SET_VARSIZE(out2, len2 + VARHDRSZ);
236
237	return out;
238}

242void
243gbt_var_bin_union(Datum *u, GBT_VARKEY *e, Oid collation,
244				  const gbtree_vinfo *tinfo, FmgrInfo *flinfo)
245{
246	GBT_VARKEY_R eo = gbt_var_key_readable(e);
247	GBT_VARKEY_R nr;
248
249	if (eo.lower == eo.upper)	/* leaf */
250	{
251		GBT_VARKEY *tmp;
252
253		tmp = gbt_var_leaf2node(e, tinfo, flinfo);
254		if (tmp != e)
255			eo = gbt_var_key_readable(tmp);
256	}
257
258	if (DatumGetPointer(*u))
259	{
260		GBT_VARKEY_R ro = gbt_var_key_readable((GBT_VARKEY *) DatumGetPointer(*u));
261		bool		update = false;
262
263		nr.lower = ro.lower;
264		nr.upper = ro.upper;
265
266		if (tinfo->f_cmp(ro.lower, eo.lower, collation, flinfo) > 0)
267		{
268			nr.lower = eo.lower;
269			update = true;
270		}
271
272		if (tinfo->f_cmp(ro.upper, eo.upper, collation, flinfo) < 0)
273		{
274			nr.upper = eo.upper;
275			update = true;
276		}
277
278		if (update)
279			*u = PointerGetDatum(gbt_var_key_copy(&nr));
280	}
281	else
282	{
283		nr.lower = eo.lower;
284		nr.upper = eo.upper;
285		*u = PointerGetDatum(gbt_var_key_copy(&nr));
286	}
287}

290GISTENTRY *
291gbt_var_compress(GISTENTRY *entry, const gbtree_vinfo *tinfo)
292{
293	GISTENTRY  *retval;
294
295	if (entry->leafkey)
296	{
297		struct varlena *leaf = PG_DETOAST_DATUM(entry->key);
298		GBT_VARKEY *r;
299
300		r = gbt_var_key_from_datum(leaf);
301
302		retval = palloc(sizeof(GISTENTRY));
303		gistentryinit(*retval, PointerGetDatum(r),
304					  entry->rel, entry->page,
305					  entry->offset, true);
306	}
307	else
308		retval = entry;
309
310	return retval;
311}

331GBT_VARKEY *
332gbt_var_union(const GistEntryVector *entryvec, int32 *size, Oid collation,
333			  const gbtree_vinfo *tinfo, FmgrInfo *flinfo)
334{
335	int			i = 0,
336				numranges = entryvec->n;
337	GBT_VARKEY *cur;
338	Datum		out;
339	GBT_VARKEY_R rk;
340
341	*size = sizeof(GBT_VARKEY);
342
343	cur = (GBT_VARKEY *) DatumGetPointer(entryvec->vector[0].key);
344	rk = gbt_var_key_readable(cur);
345	out = PointerGetDatum(gbt_var_key_copy(&rk));
346
347	for (i = 1; i < numranges; i++)
348	{
349		cur = (GBT_VARKEY *) DatumGetPointer(entryvec->vector[i].key);
350		gbt_var_bin_union(&out, cur, collation, tinfo, flinfo);
351	}
352
353
354	/* Truncate (=compress) key */
355	if (tinfo->trnc)
356	{
357		int32		plen;
358		GBT_VARKEY *trc = NULL;
359
360		plen = gbt_var_node_cp_len((GBT_VARKEY *) DatumGetPointer(out), tinfo);
361		trc = gbt_var_node_truncate((GBT_VARKEY *) DatumGetPointer(out), plen + 1, tinfo);
362
363		out = PointerGetDatum(trc);
364	}
365
366	return ((GBT_VARKEY *) DatumGetPointer(out));
367}

370bool
371gbt_var_same(Datum d1, Datum d2, Oid collation,
372			 const gbtree_vinfo *tinfo, FmgrInfo *flinfo)
373{
374	GBT_VARKEY *t1 = (GBT_VARKEY *) DatumGetPointer(d1);
375	GBT_VARKEY *t2 = (GBT_VARKEY *) DatumGetPointer(d2);
376	GBT_VARKEY_R r1,
377				r2;
378
379	r1 = gbt_var_key_readable(t1);
380	r2 = gbt_var_key_readable(t2);
381
382	return (tinfo->f_cmp(r1.lower, r2.lower, collation, flinfo) == 0 &&
383			tinfo->f_cmp(r1.upper, r2.upper, collation, flinfo) == 0);
384}

387float *
388gbt_var_penalty(float *res, const GISTENTRY *o, const GISTENTRY *n,
389				Oid collation, const gbtree_vinfo *tinfo, FmgrInfo *flinfo)
390{
391	GBT_VARKEY *orge = (GBT_VARKEY *) DatumGetPointer(o->key);
392	GBT_VARKEY *newe = (GBT_VARKEY *) DatumGetPointer(n->key);
393	GBT_VARKEY_R ok,
394				nk;
395
396	*res = 0.0;
397
398	nk = gbt_var_key_readable(newe);
399	if (nk.lower == nk.upper)	/* leaf */
400	{
401		GBT_VARKEY *tmp;
402
403		tmp = gbt_var_leaf2node(newe, tinfo, flinfo);
404		if (tmp != newe)
405			nk = gbt_var_key_readable(tmp);
406	}
407	ok = gbt_var_key_readable(orge);
408
409	if ((VARSIZE(ok.lower) - VARHDRSZ) == 0 && (VARSIZE(ok.upper) - VARHDRSZ) == 0)
410		*res = 0.0;
411	else if (!((tinfo->f_cmp(nk.lower, ok.lower, collation, flinfo) >= 0 ||
412				gbt_bytea_pf_match(ok.lower, nk.lower, tinfo)) &&
413			   (tinfo->f_cmp(nk.upper, ok.upper, collation, flinfo) <= 0 ||
414				gbt_bytea_pf_match(ok.upper, nk.upper, tinfo))))
415	{
416		Datum		d = PointerGetDatum(0);
417		double		dres;
418		int32		ol,
419					ul;
420
421		gbt_var_bin_union(&d, orge, collation, tinfo, flinfo);
422		ol = gbt_var_node_cp_len((GBT_VARKEY *) DatumGetPointer(d), tinfo);
423		gbt_var_bin_union(&d, newe, collation, tinfo, flinfo);
424		ul = gbt_var_node_cp_len((GBT_VARKEY *) DatumGetPointer(d), tinfo);
425
426		if (ul < ol)
427		{
428			dres = (ol - ul);	/* reduction of common prefix len */
429		}
430		else
431		{
432			GBT_VARKEY_R uk = gbt_var_key_readable((GBT_VARKEY *) DatumGetPointer(d));
433			unsigned char tmp[4];
434
435			tmp[0] = (unsigned char) (((VARSIZE(ok.lower) - VARHDRSZ) <= ul) ? 0 : (VARDATA(ok.lower)[ul]));
436			tmp[1] = (unsigned char) (((VARSIZE(uk.lower) - VARHDRSZ) <= ul) ? 0 : (VARDATA(uk.lower)[ul]));
437			tmp[2] = (unsigned char) (((VARSIZE(ok.upper) - VARHDRSZ) <= ul) ? 0 : (VARDATA(ok.upper)[ul]));
438			tmp[3] = (unsigned char) (((VARSIZE(uk.upper) - VARHDRSZ) <= ul) ? 0 : (VARDATA(uk.upper)[ul]));
439			dres = abs(tmp[0] - tmp[1]) + abs(tmp[3] - tmp[2]);
440			dres /= 256.0;
441		}
442
443		*res += FLT_MIN;
444		*res += (float) (dres / ((double) (ol + 1)));
445		*res *= (FLT_MAX / (o->rel->rd_att->natts + 1));
446	}
447
448	return res;
449}

452static int
453gbt_vsrt_cmp(const void *a, const void *b, void *arg)
454{
455	GBT_VARKEY_R ar = gbt_var_key_readable(((const Vsrt *) a)->t);
456	GBT_VARKEY_R br = gbt_var_key_readable(((const Vsrt *) b)->t);
457	const gbt_vsrt_arg *varg = (const gbt_vsrt_arg *) arg;
458	int			res;
459
460	res = varg->tinfo->f_cmp(ar.lower, br.lower, varg->collation, varg->flinfo);
461	if (res == 0)
462		return varg->tinfo->f_cmp(ar.upper, br.upper, varg->collation, varg->flinfo);
463
464	return res;
465}

467GIST_SPLITVEC *
468gbt_var_picksplit(const GistEntryVector *entryvec, GIST_SPLITVEC *v,
469				  Oid collation, const gbtree_vinfo *tinfo, FmgrInfo *flinfo)
470{
471	OffsetNumber i,
472				maxoff = entryvec->n - 1;
473	Vsrt	   *arr;
474	int			svcntr = 0,
475				nbytes;
476	char	   *cur;
477	GBT_VARKEY **sv = NULL;
478	gbt_vsrt_arg varg;
479
480	arr = (Vsrt *) palloc((maxoff + 1) * sizeof(Vsrt));
481	nbytes = (maxoff + 2) * sizeof(OffsetNumber);
482	v->spl_left = (OffsetNumber *) palloc(nbytes);
483	v->spl_right = (OffsetNumber *) palloc(nbytes);
484	v->spl_ldatum = PointerGetDatum(0);
485	v->spl_rdatum = PointerGetDatum(0);
486	v->spl_nleft = 0;
487	v->spl_nright = 0;
488
489	sv = palloc(sizeof(bytea *) * (maxoff + 1));
490
491	/* Sort entries */
492
493	for (i = FirstOffsetNumber; i <= maxoff; i = OffsetNumberNext(i))
494	{
495		GBT_VARKEY_R ro;
496
497		cur = (char *) DatumGetPointer(entryvec->vector[i].key);
498		ro = gbt_var_key_readable((GBT_VARKEY *) cur);
499		if (ro.lower == ro.upper)	/* leaf */
500		{
501			sv[svcntr] = gbt_var_leaf2node((GBT_VARKEY *) cur, tinfo, flinfo);
502			arr[i].t = sv[svcntr];
503			if (sv[svcntr] != (GBT_VARKEY *) cur)
504				svcntr++;
505		}
506		else
507			arr[i].t = (GBT_VARKEY *) cur;
508		arr[i].i = i;
509	}
510
511	/* sort */
512	varg.tinfo = tinfo;
513	varg.collation = collation;
514	varg.flinfo = flinfo;
515	qsort_arg(&arr[FirstOffsetNumber],
516			  maxoff - FirstOffsetNumber + 1,
517			  sizeof(Vsrt),
518			  gbt_vsrt_cmp,
519			  &varg);
520
521	/* We do simply create two parts */
522
523	for (i = FirstOffsetNumber; i <= maxoff; i = OffsetNumberNext(i))
524	{
525		if (i <= (maxoff - FirstOffsetNumber + 1) / 2)
526		{
527			gbt_var_bin_union(&v->spl_ldatum, arr[i].t, collation, tinfo, flinfo);
528			v->spl_left[v->spl_nleft] = arr[i].i;
529			v->spl_nleft++;
530		}
531		else
532		{
533			gbt_var_bin_union(&v->spl_rdatum, arr[i].t, collation, tinfo, flinfo);
534			v->spl_right[v->spl_nright] = arr[i].i;
535			v->spl_nright++;
536		}
537	}
538
539	/* Truncate (=compress) key */
540	if (tinfo->trnc)
541	{
542		int32		ll = gbt_var_node_cp_len((GBT_VARKEY *) DatumGetPointer(v->spl_ldatum), tinfo);
543		int32		lr = gbt_var_node_cp_len((GBT_VARKEY *) DatumGetPointer(v->spl_rdatum), tinfo);
544		GBT_VARKEY *dl;
545		GBT_VARKEY *dr;
546
547		ll = Max(ll, lr);
548		ll++;
549
550		dl = gbt_var_node_truncate((GBT_VARKEY *) DatumGetPointer(v->spl_ldatum), ll, tinfo);
551		dr = gbt_var_node_truncate((GBT_VARKEY *) DatumGetPointer(v->spl_rdatum), ll, tinfo);
552		v->spl_ldatum = PointerGetDatum(dl);
553		v->spl_rdatum = PointerGetDatum(dr);
554	}
555
556	return v;
557}

560/*
561 * The GiST consistent method
562 */
563bool
564gbt_var_consistent(GBT_VARKEY_R *key,
565				   const void *query,
566				   StrategyNumber strategy,
567				   Oid collation,
568				   bool is_leaf,
569				   const gbtree_vinfo *tinfo,
570				   FmgrInfo *flinfo)
571{
572	bool		retval = false;
573
574	switch (strategy)
575	{
576		case BTLessEqualStrategyNumber:
577			if (is_leaf)
578				retval = tinfo->f_ge(query, key->lower, collation, flinfo);
579			else
580				retval = tinfo->f_cmp(query, key->lower, collation, flinfo) >= 0
581					|| gbt_var_node_pf_match(key, query, tinfo);
582			break;
583		case BTLessStrategyNumber:
584			if (is_leaf)
585				retval = tinfo->f_gt(query, key->lower, collation, flinfo);
586			else
587				retval = tinfo->f_cmp(query, key->lower, collation, flinfo) >= 0
588					|| gbt_var_node_pf_match(key, query, tinfo);
589			break;
590		case BTEqualStrategyNumber:
591			if (is_leaf)
592				retval = tinfo->f_eq(query, key->lower, collation, flinfo);
593			else
594				retval =
595					(tinfo->f_cmp(key->lower, query, collation, flinfo) <= 0 &&
596					 tinfo->f_cmp(query, key->upper, collation, flinfo) <= 0) ||
597					gbt_var_node_pf_match(key, query, tinfo);
598			break;
599		case BTGreaterStrategyNumber:
600			if (is_leaf)
601				retval = tinfo->f_lt(query, key->upper, collation, flinfo);
602			else
603				retval = tinfo->f_cmp(query, key->upper, collation, flinfo) <= 0
604					|| gbt_var_node_pf_match(key, query, tinfo);
605			break;
606		case BTGreaterEqualStrategyNumber:
607			if (is_leaf)
608				retval = tinfo->f_le(query, key->upper, collation, flinfo);
609			else
610				retval = tinfo->f_cmp(query, key->upper, collation, flinfo) <= 0
611					|| gbt_var_node_pf_match(key, query, tinfo);
612			break;
613		case BtreeGistNotEqualStrategyNumber:
614			retval = !(tinfo->f_eq(query, key->lower, collation, flinfo) &&
615					   tinfo->f_eq(query, key->upper, collation, flinfo));
616			break;
617		default:
618			retval = false;
619	}
620
621	return retval;
622}
