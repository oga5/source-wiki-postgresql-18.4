53static int
54gbt_int4key_cmp(const void *a, const void *b, FmgrInfo *flinfo)
55{
56	int32KEY   *ia = (int32KEY *) (((const Nsrt *) a)->t);
57	int32KEY   *ib = (int32KEY *) (((const Nsrt *) b)->t);
58
59	if (ia->lower == ib->lower)
60	{
61		if (ia->upper == ib->upper)
62			return 0;
63
64		return (ia->upper > ib->upper) ? 1 : -1;
65	}
66
67	return (ia->lower > ib->lower) ? 1 : -1;
68}

95{
96	int32		a = PG_GETARG_INT32(0);
97	int32		b = PG_GETARG_INT32(1);
98	int32		r;
99	int32		ra;
100
101	if (pg_sub_s32_overflow(a, b, &r) ||

103		ereport(ERROR,
104				(errcode(ERRCODE_NUMERIC_VALUE_OUT_OF_RANGE),
105				 errmsg("integer out of range")));

107	ra = abs(r);
