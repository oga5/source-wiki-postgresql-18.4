190Datum
191gbt_text_consistent(PG_FUNCTION_ARGS)
192{
193	GISTENTRY  *entry = (GISTENTRY *) PG_GETARG_POINTER(0);
194	void	   *query = DatumGetTextP(PG_GETARG_DATUM(1));
195	StrategyNumber strategy = (StrategyNumber) PG_GETARG_UINT16(2);
196
197	/* Oid		subtype = PG_GETARG_OID(3); */
198	bool	   *recheck = (bool *) PG_GETARG_POINTER(4);
199	bool		retval;
200	GBT_VARKEY *key = (GBT_VARKEY *) DatumGetPointer(entry->key);
201	GBT_VARKEY_R r = gbt_var_key_readable(key);
202
203	/* All cases served by this function are exact */
204	*recheck = false;
205
206	if (tinfo.eml == 0)
207	{
208		tinfo.eml = pg_database_encoding_max_length();
209	}
210
211	retval = gbt_var_consistent(&r, query, strategy, PG_GET_COLLATION(),
212								GIST_LEAF(entry), &tinfo, fcinfo->flinfo);
213
214	PG_RETURN_BOOL(retval);
215}

286static int
287gbt_text_ssup_cmp(Datum x, Datum y, SortSupport ssup)
288{
289	GBT_VARKEY *key1 = PG_DETOAST_DATUM(x);
290	GBT_VARKEY *key2 = PG_DETOAST_DATUM(y);
291
292	GBT_VARKEY_R arg1 = gbt_var_key_readable(key1);
293	GBT_VARKEY_R arg2 = gbt_var_key_readable(key2);
294	Datum		result;
295
296	/* for leaf items we expect lower == upper, so only compare lower */
297	result = DirectFunctionCall2Coll(bttextcmp,
298									 ssup->ssup_collation,
299									 PointerGetDatum(arg1.lower),
300									 PointerGetDatum(arg2.lower));
301
302	GBT_FREE_IF_COPY(key1, x);
303	GBT_FREE_IF_COPY(key2, y);
304
305	return DatumGetInt32(result);
306}
