59static int
60gbt_macad8key_cmp(const void *a, const void *b, FmgrInfo *flinfo)
61{
62	mac8KEY    *ia = (mac8KEY *) (((const Nsrt *) a)->t);
63	mac8KEY    *ib = (mac8KEY *) (((const Nsrt *) b)->t);
64	int			res;
65
66	res = DatumGetInt32(DirectFunctionCall2(macaddr8_cmp, Macaddr8PGetDatum(&ia->lower), Macaddr8PGetDatum(&ib->lower)));
67	if (res == 0)
68		return DatumGetInt32(DirectFunctionCall2(macaddr8_cmp, Macaddr8PGetDatum(&ia->upper), Macaddr8PGetDatum(&ib->upper)));
69
70	return res;
71}

93static uint64
94mac8_2_uint64(macaddr8 *m)
95{
96	unsigned char *mi = (unsigned char *) m;
97	uint64		res = 0;
98	int			i;
99
100	for (i = 0; i < 8; i++)
101		res += (((uint64) mi[i]) << ((uint64) ((7 - i) * 8)));
102	return res;
103}

154gbt_macad8_penalty(PG_FUNCTION_ARGS)
155{
156	mac8KEY    *origentry = (mac8KEY *) DatumGetPointer(((GISTENTRY *) PG_GETARG_POINTER(0))->key);
157	mac8KEY    *newentry = (mac8KEY *) DatumGetPointer(((GISTENTRY *) PG_GETARG_POINTER(1))->key);
158	float	   *result = (float *) PG_GETARG_POINTER(2);
159	uint64		iorg[2],
160				inew[2];
161
162	iorg[0] = mac8_2_uint64(&origentry->lower);
163	iorg[1] = mac8_2_uint64(&origentry->upper);
164	inew[0] = mac8_2_uint64(&newentry->lower);
165	inew[1] = mac8_2_uint64(&newentry->upper);
166
167	penalty_num(result, iorg[0], iorg[1], inew[0], inew[1]);
168
169	PG_RETURN_POINTER(result);
