64Datum
65gist_translate_cmptype_btree(PG_FUNCTION_ARGS)
66{
67	CompareType cmptype = PG_GETARG_INT32(0);
68
69	switch (cmptype)
70	{
71		case COMPARE_EQ:
72			PG_RETURN_UINT16(BTEqualStrategyNumber);
73		case COMPARE_LT:
74			PG_RETURN_UINT16(BTLessStrategyNumber);
75		case COMPARE_LE:
76			PG_RETURN_UINT16(BTLessEqualStrategyNumber);
77		case COMPARE_GT:
78			PG_RETURN_UINT16(BTGreaterStrategyNumber);
79		case COMPARE_GE:
80			PG_RETURN_UINT16(BTGreaterEqualStrategyNumber);
81		default:
82			PG_RETURN_UINT16(InvalidStrategy);
83	}
