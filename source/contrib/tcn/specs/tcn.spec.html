3# These tests use only self-notifies within a single session,
4# which are convenient because they minimize timing concerns.
5# Whether the NOTIFY mechanism works across sessions is not
6# really tcn's problem.

8setup
9{
10  CREATE TABLE mytable (key int PRIMARY KEY, value text);
11  CREATE TRIGGER tcntrig AFTER INSERT OR UPDATE OR DELETE ON mytable
12    FOR EACH ROW EXECUTE FUNCTION triggered_change_notification(mychannel);
13}
14
15teardown
16{
17  DROP TABLE mytable;
18}
