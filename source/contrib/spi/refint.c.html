23typedef struct
24{
25	char	   *ident;
26	int			nplans;
27	SPIPlanPtr *splan;
28} EPlan;
29
30static EPlan *FPlans = NULL;
31static int	nFPlans = 0;
32static EPlan *PPlans = NULL;
33static int	nPPlans = 0;
34
35static EPlan *find_plan(char *ident, EPlan **eplan, int *nplans);

37/*
38 * check_primary_key () -- check that key in tuple being inserted/updated
39 *			 references existing tuple in "primary" table.
40 * Though it's called without args You have to specify referenced
41 * table/keys while creating trigger:  key field names in triggered table,
42 * referenced table name, referenced key field names:
43 * EXECUTE PROCEDURE
44 * check_primary_key ('Fkey1', 'Fkey2', 'Ptable', 'Pkey1', 'Pkey2').
45 */ view japanese
46
47PG_FUNCTION_INFO_V1(check_primary_key);
48
49Datum
50check_primary_key(PG_FUNCTION_ARGS)
51{
52	TriggerData *trigdata = (TriggerData *) fcinfo->context;
53	Trigger    *trigger;		/* to get trigger name */
54	int			nargs;			/* # of args specified in CREATE TRIGGER */
55	char	  **args;			/* arguments: column names and table name */ view japanese
56	int			nkeys;			/* # of key columns (= nargs / 2) */ view japanese
57	Datum	   *kvals;			/* key values */
58	char	   *relname;		/* referenced relation name */ view japanese
59	Relation	rel;			/* triggered relation */
60	HeapTuple	tuple = NULL;	/* tuple to return */
61	TupleDesc	tupdesc;		/* tuple description */
62	EPlan	   *plan;			/* prepared plan */ view japanese
63	Oid		   *argtypes = NULL;	/* key types to prepare execution plan */
64	bool		isnull;			/* to know is some column NULL or not */
65	char		ident[2 * NAMEDATALEN]; /* to identify myself */
66	int			ret;
67	int			i;
68
69#ifdef	DEBUG_QUERY
70	elog(DEBUG4, "check_primary_key: Enter Function");
71#endif
72
73	/*
74	 * Some checks first...
75	 */
76
77	/* Called by trigger manager ? */
78	if (!CALLED_AS_TRIGGER(fcinfo))
79		/* internal error */
80		elog(ERROR, "check_primary_key: not fired by trigger manager");
81
82	/* Should be called for ROW trigger */
83	if (!TRIGGER_FIRED_FOR_ROW(trigdata->tg_event))
84		/* internal error */
85		elog(ERROR, "check_primary_key: must be fired for row");
86
87	if (!TRIGGER_FIRED_AFTER(trigdata->tg_event))
88		/* internal error */
89		elog(ERROR, "check_primary_key: must be fired by AFTER trigger");
90
91	/* If INSERTion then must check Tuple to being inserted */ view japanese
92	if (TRIGGER_FIRED_BY_INSERT(trigdata->tg_event))
93		tuple = trigdata->tg_trigtuple;
94
95	/* Not should be called for DELETE */ view japanese
96	else if (TRIGGER_FIRED_BY_DELETE(trigdata->tg_event))
97		/* internal error */
98		elog(ERROR, "check_primary_key: cannot process DELETE events");
99
100	/* If UPDATE, then must check new Tuple, not old one */ view japanese
101	else
102		tuple = trigdata->tg_newtuple;
103
104	trigger = trigdata->tg_trigger;
105	nargs = trigger->tgnargs;
106	args = trigger->tgargs;
107
108	if (nargs % 2 != 1)			/* odd number of arguments! */ view japanese
109		/* internal error */
110		elog(ERROR, "check_primary_key: odd number of arguments should be specified");
111
112	nkeys = nargs / 2;
113	relname = args[nkeys];
114	rel = trigdata->tg_relation;
115	tupdesc = rel->rd_att;
116
117	/* Connect to SPI manager */
118	SPI_connect();
119
120	/*
121	 * We use SPI plan preparation feature, so allocate space to place key
122	 * values.
123	 */
124	kvals = (Datum *) palloc(nkeys * sizeof(Datum));
125
126	/*
127	 * Construct ident string as TriggerName $ TriggeredRelationId and try to
128	 * find prepared execution plan.
129	 */ view japanese
130	snprintf(ident, sizeof(ident), "%s$%u", trigger->tgname, rel->rd_id);
131	plan = find_plan(ident, &PPlans, &nPPlans);
132
133	/* if there is no plan then allocate argtypes for preparation */ view japanese
134	if (plan->nplans <= 0)
135		argtypes = (Oid *) palloc(nkeys * sizeof(Oid));
136
137	/* For each column in key ... */
138	for (i = 0; i < nkeys; i++)
139	{
140		/* get index of column in tuple */
141		int			fnumber = SPI_fnumber(tupdesc, args[i]);
142
143		/* Bad guys may give us un-existing column in CREATE TRIGGER */
144		if (fnumber <= 0)
145			ereport(ERROR,
146					(errcode(ERRCODE_UNDEFINED_COLUMN),
147					 errmsg("there is no attribute \"%s\" in relation \"%s\"",
148							args[i], SPI_getrelname(rel))));
149
150		/* Well, get binary (in internal format) value of column */
151		kvals[i] = SPI_getbinval(tuple, tupdesc, fnumber, &isnull);
152
153		/*
154		 * If it's NULL then nothing to do! DON'T FORGET call SPI_finish ()!
155		 * DON'T FORGET return tuple! Executor inserts tuple you're returning!
156		 * If you return NULL then nothing will be inserted!
157		 */
158		if (isnull)
159		{
160			SPI_finish();
161			return PointerGetDatum(tuple);
162		}
163
164		if (plan->nplans <= 0)	/* Get typeId of column */
165			argtypes[i] = SPI_gettypeid(tupdesc, fnumber);
166	}
167
168	/*
169	 * If we have to prepare plan ...
170	 */ view japanese
171	if (plan->nplans <= 0)
172	{
173		SPIPlanPtr	pplan;
174		StringInfoData sql;
175
176		initStringInfo(&sql);
177
178		/*
179		 * Construct query: SELECT 1 FROM _referenced_relation_ WHERE Pkey1 =
180		 * $1 [AND Pkey2 = $2 [...]]
181		 */ view japanese
182		appendStringInfo(&sql, "select 1 from %s where ", relname);
183		for (i = 1; i <= nkeys; i++)
184		{
185			appendStringInfo(&sql, "%s = $%d ", args[i + nkeys], i);
186			if (i < nkeys)
187				appendStringInfoString(&sql, "and ");
188		}
189
190		/* Prepare plan for query */
191		pplan = SPI_prepare(sql.data, nkeys, argtypes);
192		if (pplan == NULL)
193			/* internal error */
194			elog(ERROR, "check_primary_key: SPI_prepare returned %s", SPI_result_code_string(SPI_result));
195
196		/*
197		 * Remember that SPI_prepare places plan in current memory context -
198		 * so, we have to save plan in TopMemoryContext for later use.
199		 */ view japanese
200		if (SPI_keepplan(pplan))
201			/* internal error */
202			elog(ERROR, "check_primary_key: SPI_keepplan failed");
203		plan->splan = (SPIPlanPtr *) MemoryContextAlloc(TopMemoryContext,
204														sizeof(SPIPlanPtr));
205		*(plan->splan) = pplan;
206		plan->nplans = 1;
207
208		pfree(sql.data);
209	}
210
211	/*
212	 * Ok, execute prepared plan.
213	 */ view japanese
214	ret = SPI_execp(*(plan->splan), kvals, NULL, 1);
215	/* we have no NULLs - so we pass   ^^^^   here */ view japanese
216
217	if (ret < 0)
218		/* internal error */
219		elog(ERROR, "check_primary_key: SPI_execp returned %d", ret);
220
221	/*
222	 * If there are no tuples returned by SELECT then ...
223	 */ view japanese
224	if (SPI_processed == 0)
225		ereport(ERROR,
226				(errcode(ERRCODE_TRIGGERED_ACTION_EXCEPTION),
227				 errmsg("tuple references non-existent key"),
228				 errdetail("Trigger \"%s\" found tuple referencing non-existent key in \"%s\".", trigger->tgname, relname)));
229
230	SPI_finish();
231
232	return PointerGetDatum(tuple);
233}

235/*
236 * check_foreign_key () -- check that key in tuple being deleted/updated
237 *			 is not referenced by tuples in "foreign" table(s).
238 * Though it's called without args You have to specify (while creating trigger):
239 * number of references, action to do if key referenced
240 * ('restrict' | 'setnull' | 'cascade'), key field names in triggered
241 * ("primary") table and referencing table(s)/keys:
242 * EXECUTE PROCEDURE
243 * check_foreign_key (2, 'restrict', 'Pkey1', 'Pkey2',
244 * 'Ftable1', 'Fkey11', 'Fkey12', 'Ftable2', 'Fkey21', 'Fkey22').
245 */ view japanese
246
247PG_FUNCTION_INFO_V1(check_foreign_key);
248
249Datum
250check_foreign_key(PG_FUNCTION_ARGS)
251{
252	TriggerData *trigdata = (TriggerData *) fcinfo->context;
253	Trigger    *trigger;		/* to get trigger name */ view japanese
254	int			nargs;			/* # of args specified in CREATE TRIGGER */ view japanese
255	char	  **args;			/* arguments: as described above */ view japanese
256	char	  **args_temp;
257	int			nrefs;			/* number of references (== # of plans) */ view japanese
258	char		action;			/* 'R'estrict | 'S'etnull | 'C'ascade */ view japanese
259	int			nkeys;			/* # of key columns */ view japanese
260	Datum	   *kvals;			/* key values */ view japanese
261	char	   *relname;		/* referencing relation name */ view japanese
262	Relation	rel;			/* triggered relation */ view japanese
263	HeapTuple	trigtuple = NULL;	/* tuple to being changed */ view japanese
264	HeapTuple	newtuple = NULL;	/* tuple to return */ view japanese
265	TupleDesc	tupdesc;		/* tuple description */ view japanese
266	EPlan	   *plan;			/* prepared plan(s) */ view japanese
267	Oid		   *argtypes = NULL;	/* key types to prepare execution plan */ view japanese
268	bool		isnull;			/* to know is some column NULL or not */ view japanese
269	bool		isequal = true; /* are keys in both tuples equal (in UPDATE) */ view japanese
270	char		ident[2 * NAMEDATALEN]; /* to identify myself */ view japanese
271	int			is_update = 0;
272	int			ret;
273	int			i,
274				r;
275
276#ifdef DEBUG_QUERY
277	elog(DEBUG4, "check_foreign_key: Enter Function");
278#endif
279
280	/*
281	 * Some checks first...
282	 */ view japanese
283
284	/* Called by trigger manager ? */ view japanese
285	if (!CALLED_AS_TRIGGER(fcinfo))
286		/* internal error */
287		elog(ERROR, "check_foreign_key: not fired by trigger manager");
288
289	/* Should be called for ROW trigger */ view japanese
290	if (!TRIGGER_FIRED_FOR_ROW(trigdata->tg_event))
291		/* internal error */
292		elog(ERROR, "check_foreign_key: must be fired for row");
293
294	/* Not should be called for INSERT */ view japanese
295	if (TRIGGER_FIRED_BY_INSERT(trigdata->tg_event))
296		/* internal error */
297		elog(ERROR, "check_foreign_key: cannot process INSERT events");
298
299	if (!TRIGGER_FIRED_AFTER(trigdata->tg_event))
300		/* internal error */
301		elog(ERROR, "check_foreign_key: must be fired by AFTER trigger");
302
303	/* Have to check tg_trigtuple - tuple being deleted */ view japanese
304	trigtuple = trigdata->tg_trigtuple;
305
306	/*
307	 * But if this is UPDATE then we have to return tg_newtuple. Also, if key
308	 * in tg_newtuple is the same as in tg_trigtuple then nothing to do.
309	 */ view japanese
310	is_update = 0;
311	if (TRIGGER_FIRED_BY_UPDATE(trigdata->tg_event))
312	{
313		newtuple = trigdata->tg_newtuple;
314		is_update = 1;
315	}
316	trigger = trigdata->tg_trigger;
317	nargs = trigger->tgnargs;
318	args = trigger->tgargs;
319
320	if (nargs < 5)				/* nrefs, action, key, Relation, key - at
321								 * least */ view japanese
322		/* internal error */
323		elog(ERROR, "check_foreign_key: too short %d (< 5) list of arguments", nargs);
324
325	nrefs = pg_strtoint32(args[0]);
326	if (nrefs < 1)
327		/* internal error */
328		elog(ERROR, "check_foreign_key: %d (< 1) number of references specified", nrefs);
329	action = tolower((unsigned char) *(args[1]));
330	if (action != 'r' && action != 'c' && action != 's')
331		/* internal error */
332		elog(ERROR, "check_foreign_key: invalid action %s", args[1]);
333	nargs -= 2;
334	args += 2;
335	nkeys = (nargs - nrefs) / (nrefs + 1);
336	if (nkeys <= 0 || nargs != (nrefs + nkeys * (nrefs + 1)))
337		/* internal error */
338		elog(ERROR, "check_foreign_key: invalid number of arguments %d for %d references",
339			 nargs + 2, nrefs);
340
341	rel = trigdata->tg_relation;
342	tupdesc = rel->rd_att;
343
344	/* Connect to SPI manager */ view japanese
345	SPI_connect();
346
347	/*
348	 * We use SPI plan preparation feature, so allocate space to place key
349	 * values.
350	 */ view japanese
351	kvals = (Datum *) palloc(nkeys * sizeof(Datum));
352
353	/*
354	 * Construct ident string as TriggerName $ TriggeredRelationId $
355	 * OperationType and try to find prepared execution plan(s).
356	 */ view japanese
357	snprintf(ident, sizeof(ident), "%s$%u$%c", trigger->tgname, rel->rd_id, is_update ? 'U' : 'D');
358	plan = find_plan(ident, &FPlans, &nFPlans);
359
360	/* if there is no plan(s) then allocate argtypes for preparation */ view japanese
361	if (plan->nplans <= 0)
362		argtypes = (Oid *) palloc(nkeys * sizeof(Oid));
363
364	/*
365	 * else - check that we have exactly nrefs plan(s) ready
366	 */ view japanese
367	else if (plan->nplans != nrefs)
368		/* internal error */
369		elog(ERROR, "%s: check_foreign_key: # of plans changed in meantime",
370			 trigger->tgname);
371
372	/* For each column in key ... */ view japanese
373	for (i = 0; i < nkeys; i++)
374	{
375		/* get index of column in tuple */ view japanese
376		int			fnumber = SPI_fnumber(tupdesc, args[i]);
377
378		/* Bad guys may give us un-existing column in CREATE TRIGGER */ view japanese
379		if (fnumber <= 0)
380			ereport(ERROR,
381					(errcode(ERRCODE_UNDEFINED_COLUMN),
382					 errmsg("there is no attribute \"%s\" in relation \"%s\"",
383							args[i], SPI_getrelname(rel))));
384
385		/* Well, get binary (in internal format) value of column */ view japanese
386		kvals[i] = SPI_getbinval(trigtuple, tupdesc, fnumber, &isnull);
387
388		/*
389		 * If it's NULL then nothing to do! DON'T FORGET call SPI_finish ()!
390		 * DON'T FORGET return tuple! Executor inserts tuple you're returning!
391		 * If you return NULL then nothing will be inserted!
392		 */ view japanese
393		if (isnull)
394		{
395			SPI_finish();
396			return PointerGetDatum((newtuple == NULL) ? trigtuple : newtuple);
397		}
398
399		/*
400		 * If UPDATE then get column value from new tuple being inserted and
401		 * compare is this the same as old one. For the moment we use string
402		 * presentation of values...
403		 */ view japanese
404		if (newtuple != NULL)
405		{
406			char	   *oldval = SPI_getvalue(trigtuple, tupdesc, fnumber);
407			char	   *newval;
408
409			/* this shouldn't happen! SPI_ERROR_NOOUTFUNC ? */ view japanese
410			if (oldval == NULL)
411				/* internal error */
412				elog(ERROR, "check_foreign_key: SPI_getvalue returned %s", SPI_result_code_string(SPI_result));
413			newval = SPI_getvalue(newtuple, tupdesc, fnumber);
414			if (newval == NULL || strcmp(oldval, newval) != 0)
415				isequal = false;
416		}
417
418		if (plan->nplans <= 0)	/* Get typeId of column */ view japanese
419			argtypes[i] = SPI_gettypeid(tupdesc, fnumber);
420	}
421	args_temp = args;
422	nargs -= nkeys;
423	args += nkeys;
424
425	/*
426	 * If we have to prepare plans ...
427	 */ view japanese
428	if (plan->nplans <= 0)
429	{
430		SPIPlanPtr	pplan;
431		char	  **args2 = args;
432
433		plan->splan = (SPIPlanPtr *) MemoryContextAlloc(TopMemoryContext,
434														nrefs * sizeof(SPIPlanPtr));
435
436		for (r = 0; r < nrefs; r++)
437		{
438			StringInfoData sql;
439
440			initStringInfo(&sql);
441
442			relname = args2[0];
443
444			/*---------
445			 * For 'R'estrict action we construct SELECT query:
446			 *
447			 *	SELECT 1
448			 *	FROM _referencing_relation_
449			 *	WHERE Fkey1 = $1 [AND Fkey2 = $2 [...]]
450			 *
451			 *	to check is tuple referenced or not.
452			 *---------
453			 */ view japanese
454			if (action == 'r')
455				appendStringInfo(&sql, "select 1 from %s where ", relname);
456
457			/*---------
458			 * For 'C'ascade action we construct DELETE query
459			 *
460			 *	DELETE
461			 *	FROM _referencing_relation_
462			 *	WHERE Fkey1 = $1 [AND Fkey2 = $2 [...]]
463			 *
464			 * to delete all referencing tuples.
465			 *---------
466			 */ view japanese
467
468			/*
469			 * Max : Cascade with UPDATE query i create update query that
470			 * updates new key values in referenced tables
471			 */ view japanese
472
473
474			else if (action == 'c')
475			{
476				if (is_update == 1)
477				{
478					int			fn;
479					char	   *nv;
480					int			k;
481
482					appendStringInfo(&sql, "update %s set ", relname);
483					for (k = 1; k <= nkeys; k++)
484					{
485						fn = SPI_fnumber(tupdesc, args_temp[k - 1]);
486						Assert(fn > 0); /* already checked above */ view japanese
487						nv = SPI_getvalue(newtuple, tupdesc, fn);
488
489						appendStringInfo(&sql, " %s = %s ",
490										 args2[k], quote_literal_cstr(nv));
491						if (k < nkeys)
492							appendStringInfoString(&sql, ", ");
493					}
494					appendStringInfoString(&sql, " where ");
495				}
496				else
497					/* DELETE */ view japanese
498					appendStringInfo(&sql, "delete from %s where ", relname);
499			}
500
501			/*
502			 * For 'S'etnull action we construct UPDATE query - UPDATE
503			 * _referencing_relation_ SET Fkey1 null [, Fkey2 null [...]]
504			 * WHERE Fkey1 = $1 [AND Fkey2 = $2 [...]] - to set key columns in
505			 * all referencing tuples to NULL.
506			 */ view japanese
507			else if (action == 's')
508			{
509				appendStringInfo(&sql, "update %s set ", relname);
510				for (i = 1; i <= nkeys; i++)
511				{
512					appendStringInfo(&sql, "%s = null", args2[i]);
513					if (i < nkeys)
514						appendStringInfoString(&sql, ", ");
515				}
516				appendStringInfoString(&sql, " where ");
517			}
518
519			/* Construct WHERE qual */ view japanese
520			for (i = 1; i <= nkeys; i++)
521			{
522				appendStringInfo(&sql, "%s = $%d ", args2[i], i);
523				if (i < nkeys)
524					appendStringInfoString(&sql, "and ");
525			}
526
527			/* Prepare plan for query */ view japanese
528			pplan = SPI_prepare(sql.data, nkeys, argtypes);
529			if (pplan == NULL)
530				/* internal error */
531				elog(ERROR, "check_foreign_key: SPI_prepare returned %s", SPI_result_code_string(SPI_result));
532
533			/*
534			 * Remember that SPI_prepare places plan in current memory context
535			 * - so, we have to save plan in Top memory context for later use.
536			 */ view japanese
537			if (SPI_keepplan(pplan))
538				/* internal error */ view japanese
539				elog(ERROR, "check_foreign_key: SPI_keepplan failed");
540
541			plan->splan[r] = pplan;
542
543			args2 += nkeys + 1; /* to the next relation */
544
545#ifdef DEBUG_QUERY
546			elog(DEBUG4, "check_foreign_key Debug Query is :  %s ", sql.data);
547#endif
548
549			pfree(sql.data);
550		}
551		plan->nplans = nrefs;
552	}
553
554	/*
555	 * If UPDATE and key is not changed ...
556	 */ view japanese
557	if (newtuple != NULL && isequal)
558	{
559		SPI_finish();
560		return PointerGetDatum(newtuple);
561	}
562
563	/*
564	 * Ok, execute prepared plan(s).
565	 */ view japanese
566	for (r = 0; r < nrefs; r++)
567	{
568		/*
569		 * For 'R'estrict we may to execute plan for one tuple only, for other
570		 * actions - for all tuples.
571		 */ view japanese
572		int			tcount = (action == 'r') ? 1 : 0;
573
574		relname = args[0];
575
576		ret = SPI_execp(plan->splan[r], kvals, NULL, tcount);
577		/* we have no NULLs - so we pass   ^^^^  here */ view japanese
578
579		if (ret < 0)
580			ereport(ERROR,
581					(errcode(ERRCODE_TRIGGERED_ACTION_EXCEPTION),
582					 errmsg("SPI_execp returned %d", ret)));
583
584		/* If action is 'R'estrict ... */ view japanese
585		if (action == 'r')
586		{
587			/* If there is tuple returned by SELECT then ... */ view japanese
588			if (SPI_processed > 0)
589				ereport(ERROR,
590						(errcode(ERRCODE_TRIGGERED_ACTION_EXCEPTION),
591						 errmsg("\"%s\": tuple is referenced in \"%s\"",
592								trigger->tgname, relname)));
593		}
594		else
595		{
596#ifdef REFINT_VERBOSE
597			const char *operation;
598
599			if (action == 'c')
600				operation = is_update ? "updated" : "deleted";
601			else
602				operation = "set to null";
603
604			elog(NOTICE, "%s: " UINT64_FORMAT " tuple(s) of %s are %s",
605				 trigger->tgname, SPI_processed, relname, operation);
606#endif
607		}
608		args += nkeys + 1;		/* to the next relation */ view japanese
609	}
610
611	SPI_finish();
612
613	return PointerGetDatum((newtuple == NULL) ? trigtuple : newtuple);
614}

616static EPlan *
617find_plan(char *ident, EPlan **eplan, int *nplans)
618{
619	EPlan	   *newp;
620	int			i;
621	MemoryContext oldcontext;
622
623	/*
624	 * All allocations done for the plans need to happen in a session-safe
625	 * context.
626	 */ view japanese
627	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
628
629	if (*nplans > 0)
630	{
631		for (i = 0; i < *nplans; i++)
632		{
633			if (strcmp((*eplan)[i].ident, ident) == 0)
634				break;
635		}
636		if (i != *nplans)
637		{
638			MemoryContextSwitchTo(oldcontext);
639			return (*eplan + i);
640		}
641		*eplan = (EPlan *) repalloc(*eplan, (i + 1) * sizeof(EPlan));
642		newp = *eplan + i;
643	}
644	else
645	{
