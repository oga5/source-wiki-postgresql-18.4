206-- Repeat the same queries with an extended data set. The data set is the
207-- same that we used before, except that each element in the array is
208-- repeated three times, offset by 1000 and 2000. For example, {1, 5}
209-- becomes {1, 1001, 2001, 5, 1005, 2005}.
210--
211-- That has proven to be unreasonably effective at exercising codepaths in
212-- core GiST code related to splitting parent pages, which is not covered by
213-- other tests. This is a bit out-of-place as the point is to test core GiST
214-- code rather than this extension, but there is no suitable GiST opclass in
215-- core that would reach the same codepaths.
216CREATE TABLE more__int AS SELECT
217   -- Leave alone NULLs, empty arrays and the one row that we use to test
218   -- equality; also skip INT_MAX
219   CASE WHEN a IS NULL OR a = '{}' OR a = '{73,23,20}' THEN a ELSE
220     (select array_agg(u) || array_agg(u + 1000) || array_agg(u + 2000)
221      from unnest(a) u where u < 2000000000)
222   END AS a, a as b
223   FROM test__int;
224CREATE INDEX ON more__int using gist (a gist__int_ops(numranges = 252));
