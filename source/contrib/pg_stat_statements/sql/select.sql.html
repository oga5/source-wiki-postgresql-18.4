177-- GROUP BY [DISTINCT]
178SELECT a, b, c
179FROM (VALUES (1, 2, 3), (4, NULL, 6), (7, 8, 9)) AS t (a, b, c)
180GROUP BY ROLLUP(a, b), rollup(a, c)
181ORDER BY a, b, c;
182SELECT a, b, c
183FROM (VALUES (1, 2, 3), (4, NULL, 6), (7, 8, 9)) AS t (a, b, c)
184GROUP BY DISTINCT ROLLUP(a, b), rollup(a, c)
185ORDER BY a, b, c;

216-- search_path with various schemas and temporary tables
217CREATE SCHEMA pgss_schema_1;
218CREATE SCHEMA pgss_schema_2;
219-- Same attributes.
220CREATE TABLE pgss_schema_1.tab_search_same (a int, b int);
221CREATE TABLE pgss_schema_2.tab_search_same (a int, b int);
222CREATE TEMP TABLE tab_search_same (a int, b int);
223-- Different number of attributes, mapping types
224CREATE TABLE pgss_schema_1.tab_search_diff_1 (a int);
225CREATE TABLE pgss_schema_2.tab_search_diff_1 (a int, b int);
226CREATE TEMP TABLE tab_search_diff_1 (a int, b int, c int);
227-- Same number of attributes, different types
228CREATE TABLE pgss_schema_1.tab_search_diff_2 (a int);
229CREATE TABLE pgss_schema_2.tab_search_diff_2 (a text);
230CREATE TEMP TABLE tab_search_diff_2 (a bigint);
231-- First permanent schema
232SET search_path = 'pgss_schema_1';
233SELECT count(*) FROM tab_search_same;
234SELECT a, b FROM tab_search_same;
235SELECT count(*) FROM tab_search_diff_1;
236SELECT count(*) FROM tab_search_diff_2;
237SELECT a FROM tab_search_diff_2 AS t1;
238SELECT a FROM tab_search_diff_2;
239SELECT a AS a1 FROM tab_search_diff_2;
240-- Second permanent schema
241SET search_path = 'pgss_schema_2';
242SELECT count(*) FROM tab_search_same;
243SELECT a, b FROM tab_search_same;
244SELECT count(*) FROM tab_search_diff_1;
245SELECT count(*) FROM tab_search_diff_2;
246SELECT a FROM tab_search_diff_2 AS t1;
247SELECT a FROM tab_search_diff_2;
248SELECT a AS a1 FROM tab_search_diff_2;
249-- Temporary schema
250SET search_path = 'pg_temp';
251SELECT count(*) FROM tab_search_same;
252SELECT a, b FROM tab_search_same;
253SELECT count(*) FROM tab_search_diff_1;
254SELECT count(*) FROM tab_search_diff_2;
255SELECT a FROM tab_search_diff_2 AS t1;
256SELECT a FROM tab_search_diff_2;
257SELECT a AS a1 FROM tab_search_diff_2;
258RESET search_path;
