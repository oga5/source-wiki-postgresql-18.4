1--
2-- Tests for parallel statistics
3--
4
5SET pg_stat_statements.track_utility = FALSE;
6
7-- encourage use of parallel plans
8SET parallel_setup_cost = 0;
9SET parallel_tuple_cost = 0;
10SET min_parallel_table_scan_size = 0;
11SET max_parallel_workers_per_gather = 2;
12
13CREATE TABLE pgss_parallel_tab (a int);
14
15SELECT pg_stat_statements_reset() IS NOT NULL AS t;
16
17SELECT count(*) FROM pgss_parallel_tab;
18
19SELECT query,
20  parallel_workers_to_launch > 0 AS has_workers_to_launch,
21  parallel_workers_launched > 0 AS has_workers_launched
22  FROM pg_stat_statements
23  WHERE query ~ 'SELECT count'
24  ORDER BY query COLLATE "C";
25
26DROP TABLE pgss_parallel_tab;
