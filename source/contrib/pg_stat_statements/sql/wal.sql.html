4
5SET pg_stat_statements.track_utility = FALSE;
6
7CREATE TABLE pgss_wal_tab (a int, b char(20));
8
9INSERT INTO pgss_wal_tab VALUES(generate_series(1, 10), 'aaa');
10UPDATE pgss_wal_tab SET b = 'bbb' WHERE a > 7;
11DELETE FROM pgss_wal_tab WHERE a > 9;
12DROP TABLE pgss_wal_tab;
13
14-- Check WAL is generated for the above statements
15SELECT query, calls, rows,
16wal_bytes > 0 as wal_bytes_generated,
17wal_records > 0 as wal_records_generated,
18wal_records >= rows as wal_records_ge_rows
19FROM pg_stat_statements ORDER BY query COLLATE "C";
20SELECT pg_stat_statements_reset() IS NOT NULL AS t;
