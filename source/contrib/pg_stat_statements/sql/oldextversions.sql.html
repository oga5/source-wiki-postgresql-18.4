3CREATE EXTENSION pg_stat_statements WITH VERSION '1.4';
4-- Execution of pg_stat_statements_reset() is granted only to
5-- superusers in 1.4, so this fails.
6SET SESSION AUTHORIZATION pg_read_all_stats;
7SELECT pg_stat_statements_reset();
8RESET SESSION AUTHORIZATION;
9
10AlTER EXTENSION pg_stat_statements UPDATE TO '1.5';
11-- Execution of pg_stat_statements_reset() should be granted to
12-- pg_read_all_stats now, so this works.
13SET SESSION AUTHORIZATION pg_read_all_stats;
14SELECT pg_stat_statements_reset();
15RESET SESSION AUTHORIZATION;
16
17-- In 1.6, it got restricted back to superusers.
18AlTER EXTENSION pg_stat_statements UPDATE TO '1.6';
19SET SESSION AUTHORIZATION pg_read_all_stats;
20SELECT pg_stat_statements_reset();
21RESET SESSION AUTHORIZATION;
22SELECT pg_get_functiondef('pg_stat_statements_reset'::regproc);
