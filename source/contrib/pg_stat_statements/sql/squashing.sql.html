43-- prepared statements will also be squashed
44-- the IN and ARRAY forms of this statement will have the same queryId
45SELECT pg_stat_statements_reset() IS NOT NULL AS t;
46PREPARE p1(int, int, int, int, int) AS
47SELECT * FROM test_squash WHERE id IN ($1, $2, $3, $4, $5);
48EXECUTE p1(1, 2, 3, 4, 5);
49DEALLOCATE p1;
50PREPARE p1(int, int, int, int, int) AS
51SELECT * FROM test_squash WHERE id = ANY(ARRAY[$1, $2, $3, $4, $5]);
52EXECUTE p1(1, 2, 3, 4, 5);
53DEALLOCATE p1;
54SELECT query, calls FROM pg_stat_statements ORDER BY query COLLATE "C";

158-- CoerceViaIO

160
161-- Create some dummy type to force CoerceViaIO
162CREATE TYPE casttesttype;
163
164CREATE FUNCTION casttesttype_in(cstring)
165   RETURNS casttesttype
166   AS 'textin'
167   LANGUAGE internal STRICT IMMUTABLE;
168
169CREATE FUNCTION casttesttype_out(casttesttype)
170   RETURNS cstring
171   AS 'textout'
172   LANGUAGE internal STRICT IMMUTABLE;
173
174CREATE TYPE casttesttype (
175   internallength = variable,
176   input = casttesttype_in,
177   output = casttesttype_out,
178   alignment = int4
179);
180
181CREATE CAST (int4 AS casttesttype) WITH INOUT;
182
183CREATE FUNCTION casttesttype_eq(casttesttype, casttesttype)
184returns boolean language sql immutable as $$
185    SELECT true
186$$;
187
188CREATE OPERATOR = (
189    leftarg = casttesttype,
190    rightarg = casttesttype,
191    procedure = casttesttype_eq,
192    commutator = =);
193
194CREATE TABLE test_squash_cast (id int, data casttesttype);
195
196-- Use the introduced type to construct a list of CoerceViaIO around Const
197SELECT pg_stat_statements_reset() IS NOT NULL AS t;
198SELECT * FROM test_squash_cast WHERE data IN

294-- composite function with row expansion
295create table test_composite(x integer);
296CREATE FUNCTION composite_f(a integer[], out x integer, out y integer) returns
297record as $$            begin
298        x = a[1];
299        y = a[2];
300    end;
301$$ language plpgsql;
302SELECT pg_stat_statements_reset() IS NOT NULL AS t;
303SELECT ((composite_f(array[1, 2]))).* FROM test_composite;
304SELECT ((composite_f(array[1, 2, 3]))).* FROM test_composite;
305SELECT ((composite_f(array[1, 2, 3]))).*, 1, 2, 3, ((composite_f(array[1, 2, 3]))).*, 1, 2
306FROM test_composite
307WHERE x IN (1, 2, 3);
308SELECT ((composite_f(array[1, $1, 3]))).*, 1 FROM test_composite \bind 1
309;
310-- ROW() expression with row expansion
311SELECT (ROW(ARRAY[1,2])).*;
312SELECT (ROW(ARRAY[1, 2], ARRAY[1, 2, 3])).*;
313SELECT 1, 2, (ROW(ARRAY[1, 2], ARRAY[1, 2, 3])).*, 3, 4;
314SELECT (ROW(ARRAY[1, 2], ARRAY[1, $1, 3])).*, 1 \bind 1
315;

321;
