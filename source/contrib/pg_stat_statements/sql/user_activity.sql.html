contrib/pg_stat_statements/sql/user_activity.sql
1--
2-- Track user activity and reset them
3--
4
5SET pg_stat_statements.track_utility = TRUE;
6SELECT pg_stat_statements_reset() IS NOT NULL AS t;
7CREATE ROLE regress_stats_user1;
8CREATE ROLE regress_stats_user2;
9
10SET ROLE regress_stats_user1;
11
12SELECT 1 AS "ONE";
13SELECT 1+1 AS "TWO";
14
15RESET ROLE;
16SET ROLE regress_stats_user2;
17
18SELECT 1 AS "ONE";
19SELECT 1+1 AS "TWO";
20
21RESET ROLE;
22SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
23
24--
25-- Don't reset anything if any of the parameter is NULL
26--
27SELECT pg_stat_statements_reset(NULL) IS NOT NULL AS t;
28SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
29
30--
31-- remove query ('SELECT $1+$2 AS "TWO"') executed by regress_stats_user2
32-- in the current_database
33--
34SELECT pg_stat_statements_reset(
35	(SELECT r.oid FROM pg_roles AS r WHERE r.rolname = 'regress_stats_user2'),
36	(SELECT d.oid FROM pg_database As d where datname = current_database()),
37	(SELECT s.queryid FROM pg_stat_statements AS s
38				WHERE s.query = 'SELECT $1+$2 AS "TWO"' LIMIT 1))
39	IS NOT NULL AS t;
40SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
41
42--
43-- remove query ('SELECT $1 AS "ONE"') executed by two users
44--
45SELECT pg_stat_statements_reset(0,0,s.queryid) IS NOT NULL AS t
46	FROM pg_stat_statements AS s WHERE s.query = 'SELECT $1 AS "ONE"';
47SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
48
49--
50-- remove query of a user (regress_stats_user1)
51--
52SELECT pg_stat_statements_reset(r.oid) IS NOT NULL AS t
53		FROM pg_roles AS r WHERE r.rolname = 'regress_stats_user1';
54SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
55
56--
57-- reset all
58--
59SELECT pg_stat_statements_reset(0,0,0) IS NOT NULL AS t;
60SELECT query, calls, rows FROM pg_stat_statements ORDER BY query COLLATE "C";
61
62--
63-- cleanup
64--
65DROP ROLE regress_stats_user1;
66DROP ROLE regress_stats_user2;
67SELECT pg_stat_statements_reset() IS NOT NULL AS t;
