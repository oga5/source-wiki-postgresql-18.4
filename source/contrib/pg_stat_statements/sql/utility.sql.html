155-- CALL
156CREATE OR REPLACE PROCEDURE sum_one(i int) AS $$
157DECLARE
158  r int;
159BEGIN
160  SELECT (i + i)::int INTO r;
161END; $$ LANGUAGE plpgsql;
162CREATE OR REPLACE PROCEDURE sum_two(i int, j int) AS $$
163DECLARE
164  r int;
165BEGIN
166  SELECT (i + j)::int INTO r;
167END; $$ LANGUAGE plpgsql;
168-- Overloaded functions.
169CREATE OR REPLACE PROCEDURE overload(i int) AS $$
170DECLARE
171  r int;
172BEGIN
173  SELECT (i + i)::int INTO r;
174END; $$ LANGUAGE plpgsql;
175CREATE OR REPLACE PROCEDURE overload(i text) AS $$
176DECLARE
177  r text;
178BEGIN
179  SELECT i::text INTO r;
180END; $$ LANGUAGE plpgsql;
181-- Mix of IN/OUT parameters.
182CREATE OR REPLACE PROCEDURE in_out(i int, i2 OUT int, i3 INOUT int) AS $$
183DECLARE
184  r int;
185BEGIN
186  i2 := i;
187  i3 := i3 + i;
188END; $$ LANGUAGE plpgsql;
189SELECT pg_stat_statements_reset() IS NOT NULL AS t;

193CALL sum_two(1,2);
194CALL overload(1);
195CALL overload('A');
