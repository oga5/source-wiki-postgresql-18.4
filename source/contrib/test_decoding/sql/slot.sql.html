96-- Check all copied slots status
97SELECT
98    o.slot_name, o.plugin, o.temporary, c.slot_name, c.plugin, c.temporary
99FROM
100    (SELECT * FROM pg_replication_slots WHERE slot_name LIKE 'orig%') as o
101    LEFT JOIN pg_replication_slots as c ON o.restart_lsn = c.restart_lsn  AND o.confirmed_flush_lsn = c.confirmed_flush_lsn
102WHERE
103    o.slot_name != c.slot_name
104ORDER BY o.slot_name, c.slot_name;
