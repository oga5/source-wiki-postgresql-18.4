1-- Test prepared transactions. When two-phase-commit is enabled, transactions are
2-- decoded at PREPARE time rather than at COMMIT PREPARED time.
3SET synchronous_commit = on;
4SELECT 'init' FROM pg_create_logical_replication_slot('regression_slot', 'test_decoding', false, true);
5
6CREATE TABLE test_prepared1(id integer primary key);
7CREATE TABLE test_prepared2(id integer primary key);
8
9-- Test that decoding happens at PREPARE time when two-phase-commit is enabled.
10-- Decoding after COMMIT PREPARED must have all the commands in the transaction.
11BEGIN;
12INSERT INTO test_prepared1 VALUES (1);
13INSERT INTO test_prepared1 VALUES (2);
14-- should show nothing because the xact has not been prepared yet.
15SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
16PREPARE TRANSACTION 'test_prepared#1';
17-- should show both the above inserts and the PREPARE TRANSACTION.
18SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
19COMMIT PREPARED 'test_prepared#1';
20SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
21
22-- Test that rollback of a prepared xact is decoded.
23BEGIN;
24INSERT INTO test_prepared1 VALUES (3);
25PREPARE TRANSACTION 'test_prepared#2';
26SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
27ROLLBACK PREPARED 'test_prepared#2';
28SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
29
30-- Test prepare of a xact containing ddl. Leaving xact uncommitted for next test.
31BEGIN;
32ALTER TABLE test_prepared1 ADD COLUMN data text;
33INSERT INTO test_prepared1 VALUES (4, 'frakbar');
34PREPARE TRANSACTION 'test_prepared#3';
35-- confirm that exclusive lock from the ALTER command is held on test_prepared1 table
36SELECT 'test_prepared_1' AS relation, locktype, mode
37FROM pg_locks
38WHERE locktype = 'relation'
39  AND relation = 'test_prepared1'::regclass;
40-- The insert should show the newly altered column but not the DDL.
41SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
42
43-- Test that we decode correctly while an uncommitted prepared xact
44-- with ddl exists.
45--
46-- Use a separate table for the concurrent transaction because the lock from
47-- the ALTER will stop us inserting into the other one.
48--
49INSERT INTO test_prepared2 VALUES (5);
50SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
51
52COMMIT PREPARED 'test_prepared#3';
53SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
54-- make sure stuff still works
55INSERT INTO test_prepared1 VALUES (6);
56INSERT INTO test_prepared2 VALUES (7);
57SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
58
59-- Check 'CLUSTER' (as operation that hold exclusive lock) doesn't block
60-- logical decoding.
61BEGIN;
62INSERT INTO test_prepared1 VALUES (8, 'othercol');
63CLUSTER test_prepared1 USING test_prepared1_pkey;
64INSERT INTO test_prepared1 VALUES (9, 'othercol2');
65PREPARE TRANSACTION 'test_prepared_lock';
66
67SELECT 'test_prepared1' AS relation, locktype, mode
68FROM pg_locks
69WHERE locktype = 'relation'
70  AND relation = 'test_prepared1'::regclass;
71-- The above CLUSTER command shouldn't cause a timeout on 2pc decoding.
72\set env_timeout ''
73\getenv env_timeout PG_TEST_TIMEOUT_DEFAULT
74SELECT COALESCE(NULLIF(:'env_timeout', ''), '180') || 's' AS timeout \gset
75SET statement_timeout = :'timeout';
76SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
77RESET statement_timeout;
78COMMIT PREPARED 'test_prepared_lock';
79-- consume the commit
80SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
81
82-- Test savepoints and sub-xacts. Creating savepoints will create
83-- sub-xacts implicitly.
84BEGIN;
85CREATE TABLE test_prepared_savepoint (a int);
86INSERT INTO test_prepared_savepoint VALUES (1);
87SAVEPOINT test_savepoint;
88INSERT INTO test_prepared_savepoint VALUES (2);
89ROLLBACK TO SAVEPOINT test_savepoint;
90PREPARE TRANSACTION 'test_prepared_savepoint';
91-- should show only 1, not 2
92SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');
93COMMIT PREPARED 'test_prepared_savepoint';
94-- consume the commit
95SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');

97-- Test that a GID containing "_nodecode" gets decoded at commit prepared time.

105SELECT data FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1');

107-- Test that accessing a TOAST table is permitted during the decoding of a
108-- prepared transaction.
109
110-- Create a table with a column that uses a TOASTed default value.
111-- (temporarily hide query, to avoid the long CREATE TABLE stmt)
112\set ECHO none
113SELECT 'CREATE TABLE test_tab (a text DEFAULT ''' || string_agg('toast value', '') || ''');' FROM generate_series(1, 4000)
114\gexec
115\set ECHO all
116
117BEGIN;
118INSERT INTO test_tab VALUES('test');
119PREPARE TRANSACTION 'test_toast_table_access';
120
121SELECT count(*) FROM pg_logical_slot_get_changes('regression_slot', NULL, NULL, 'include-xids', '0', 'skip-empty-xacts', '1', 'stream-changes', '1');
122
123COMMIT PREPARED 'test_toast_table_access';
124
125-- consume commit prepared
