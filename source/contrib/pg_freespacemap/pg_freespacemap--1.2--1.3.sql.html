6CREATE OR REPLACE FUNCTION
7  pg_freespace(rel regclass, blkno OUT bigint, avail OUT int2)
8RETURNS SETOF RECORD
9LANGUAGE SQL PARALLEL SAFE
10BEGIN ATOMIC
11  SELECT blkno, pg_freespace($1, blkno) AS avail
12  FROM generate_series('0'::bigint, pg_relation_size($1) / current_setting('block_size'::text)::bigint - '1'::bigint) AS blkno;
13END;
