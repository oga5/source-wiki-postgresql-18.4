6-- Register the new functions.
7CREATE OR REPLACE FUNCTION pg_buffercache_numa_pages()
8RETURNS SETOF RECORD
9AS 'MODULE_PATHNAME', 'pg_buffercache_numa_pages'
10LANGUAGE C PARALLEL SAFE;
11
12-- Create a view for convenient access.
13CREATE VIEW pg_buffercache_numa AS
14	SELECT P.* FROM pg_buffercache_numa_pages() AS P
15	(bufferid integer, os_page_num bigint, numa_node integer);
16
17-- Don't want these to be available to public.
18REVOKE ALL ON FUNCTION pg_buffercache_numa_pages() FROM PUBLIC;
19REVOKE ALL ON pg_buffercache_numa FROM PUBLIC;
20
21GRANT EXECUTE ON FUNCTION pg_buffercache_numa_pages() TO pg_monitor;
22GRANT SELECT ON pg_buffercache_numa TO pg_monitor;
