contrib/pg_walinspect/sql/oldextversions.sql
1-- Test old extension version entry points.
2
3CREATE EXTENSION pg_walinspect WITH VERSION '1.0';
4
5-- Mask DETAIL messages as these could refer to current LSN positions.
6\set VERBOSITY terse
7
8-- List what version 1.0 contains, using a locale-independent sorting.
9SELECT pg_describe_object(classid, objid, 0) AS obj
10  FROM pg_depend
11  WHERE refclassid = 'pg_extension'::regclass AND
12    refobjid = (SELECT oid FROM pg_extension
13                  WHERE extname = 'pg_walinspect') AND deptype = 'e'
14  ORDER BY pg_describe_object(classid, objid, 0) COLLATE "C";
15
16-- Make sure checkpoints don't interfere with the test.
17SELECT 'init' FROM pg_create_physical_replication_slot('regress_pg_walinspect_slot', true, false);
18
19CREATE TABLE sample_tbl(col1 int, col2 int);
20SELECT pg_current_wal_lsn() AS wal_lsn1 \gset
21INSERT INTO sample_tbl SELECT * FROM generate_series(1, 2);
22
23-- Tests for the past functions.
24SELECT COUNT(*) >= 1 AS ok FROM pg_get_wal_records_info_till_end_of_wal(:'wal_lsn1');
25SELECT COUNT(*) >= 1 AS ok FROM pg_get_wal_stats_till_end_of_wal(:'wal_lsn1');
26-- Failures with start LSNs.
27SELECT * FROM pg_get_wal_records_info_till_end_of_wal('FFFFFFFF/FFFFFFFF');
28SELECT * FROM pg_get_wal_stats_till_end_of_wal('FFFFFFFF/FFFFFFFF');
29
30-- Move to new version 1.1.
31ALTER EXTENSION pg_walinspect UPDATE TO '1.1';
32
33-- List what version 1.1 contains.
34\dx+ pg_walinspect
35
36SELECT pg_drop_replication_slot('regress_pg_walinspect_slot');
37
38DROP TABLE sample_tbl;
39DROP EXTENSION pg_walinspect;
