20-- create function to filter unstable results of EXPLAIN
21CREATE FUNCTION explain_filter(text) RETURNS setof text
22LANGUAGE plpgsql AS
23$$
24declare
25    ln text;
26begin
27    for ln in execute $1
28    loop
29        -- Remove the path portion of foreign file names
30        ln := regexp_replace(ln, 'Foreign File: .*/([a-z.]+)$', 'Foreign File: .../\1');
31        return next ln;
32    end loop;
33end;
34$$;

246-- Test DELETE/UPDATE/MERGE on a partitioned table when all partitions
247-- are excluded and only the dummy root result relation remains. The
248-- operation is a no-op but should not fail regardless of whether the
249-- foreign child was processed (pruning off) or not (pruning on).
250DROP TABLE p2;
251SET enable_partition_pruning TO off;
252DELETE FROM pt WHERE false;
253UPDATE pt SET b = 'x' WHERE false;
254MERGE INTO pt t USING (VALUES (1, 'x'::text)) AS s(a, b)
255  ON false WHEN MATCHED THEN UPDATE SET b = s.b;
256
257SET enable_partition_pruning TO on;
258DELETE FROM pt WHERE false;
259UPDATE pt SET b = 'x' WHERE false;
260MERGE INTO pt t USING (VALUES (1, 'x'::text)) AS s(a, b)
261  ON false WHEN MATCHED THEN UPDATE SET b = s.b;
262
263DROP TABLE pt;
