34CREATE FUNCTION bool2perl(bool, bool, bool) RETURNS void
35LANGUAGE plperl
36TRANSFORM FOR TYPE bool, for type boolean  -- duplicate to test ruleutils
37AS $$
38my ($x, $y, $z) = @_;
39
40die("NULL mistransformed") if (defined($z));
41die("TRUE mistransformed to UNDEF") if (!defined($x));
42die("FALSE mistransformed to UNDEF") if (!defined($y));
43die("TRUE mistransformed") if (!$x);
44die("FALSE mistransformed") if ($y);
45$$;
