6/* First we have to remove them from the extension */
7ALTER EXTENSION citext DROP FUNCTION regexp_matches( citext, citext );
8ALTER EXTENSION citext DROP FUNCTION regexp_matches( citext, citext, text );
9
10/* Then we can drop them */
11DROP FUNCTION regexp_matches( citext, citext );
12DROP FUNCTION regexp_matches( citext, citext, text );
13
14/* Now redefine */
15CREATE FUNCTION regexp_matches( citext, citext ) RETURNS SETOF TEXT[] AS $$
16    SELECT pg_catalog.regexp_matches( $1::pg_catalog.text, $2::pg_catalog.text, 'i' );
17$$ LANGUAGE SQL IMMUTABLE STRICT ROWS 1;
18
19CREATE FUNCTION regexp_matches( citext, citext, text ) RETURNS SETOF TEXT[] AS $$
20    SELECT pg_catalog.regexp_matches( $1::pg_catalog.text, $2::pg_catalog.text, CASE WHEN pg_catalog.strpos($3, 'c') = 0 THEN  $3 || 'i' ELSE $3 END );
21$$ LANGUAGE SQL IMMUTABLE STRICT ROWS 10;
