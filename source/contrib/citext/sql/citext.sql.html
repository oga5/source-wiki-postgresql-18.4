7-- Check whether any of our opclasses fail amvalidate
8SELECT amname, opcname
9FROM pg_opclass opc LEFT JOIN pg_am am ON am.oid = opcmethod
10WHERE opc.oid >= 16384 AND NOT amvalidate(opc.oid);

12-- Test the operators and indexing functions
13
14-- Test = and <>.
15SELECT 'a'::citext = 'a'::citext AS t;
16SELECT 'a'::citext = 'A'::citext AS t;
17SELECT 'a'::citext = 'A'::text AS f;        -- text wins the discussion
18SELECT 'a'::citext = 'b'::citext AS f;
19SELECT 'a'::citext = 'ab'::citext AS f;
20SELECT 'a'::citext <> 'ab'::citext AS t;
21
22-- Test > and >=
23SELECT 'B'::citext > 'a'::citext AS t;
24SELECT 'b'::citext >  'A'::citext AS t;
25SELECT 'B'::citext >  'b'::citext AS f;
26SELECT 'B'::citext >= 'b'::citext AS t;
27
28-- Test < and <=
29SELECT 'a'::citext <  'B'::citext AS t;
30SELECT 'a'::citext <= 'B'::citext AS t;
31
32-- Test implicit casting. citext casts to text, but not vice-versa.
33SELECT 'a'::citext = 'a'::text   AS t;
34SELECT 'A'::text  <> 'a'::citext AS t;
35
36SELECT 'B'::citext <  'a'::text AS t;  -- text wins.
37SELECT 'B'::citext <= 'a'::text AS t;  -- text wins.
38
39SELECT 'a'::citext >  'B'::text AS t;  -- text wins.
40SELECT 'a'::citext >= 'B'::text AS t;  -- text wins.
41
42-- Test implicit casting. citext casts to varchar, but not vice-versa.
43SELECT 'a'::citext = 'a'::varchar   AS t;
44SELECT 'A'::varchar  <> 'a'::citext AS t;
45
46SELECT 'B'::citext <  'a'::varchar AS t;  -- varchar wins.
47SELECT 'B'::citext <= 'a'::varchar AS t;  -- varchar wins.
48
49SELECT 'a'::citext >  'B'::varchar AS t;  -- varchar wins.
50SELECT 'a'::citext >= 'B'::varchar AS t;  -- varchar wins.

58-- Check the citext_cmp() function explicitly.
59SELECT citext_cmp('aardvark'::citext, 'aardvark'::citext) AS zero;
60SELECT citext_cmp('aardvark'::citext, 'aardVark'::citext) AS zero;
61SELECT citext_cmp('AARDVARK'::citext, 'AARDVARK'::citext) AS zero;
62SELECT citext_cmp('B'::citext, 'a'::citext) > 0 AS true;

64-- Check the citext_hash() and citext_hash_extended() function explicitly.
65SELECT v as value, citext_hash(v)::bit(32) as standard,
66       citext_hash_extended(v, 0)::bit(32) as extended0,
67       citext_hash_extended(v, 1)::bit(32) as extended1
68FROM   (VALUES (NULL::citext), ('PostgreSQL'), ('eIpUEtqmY89'), ('AXKEJBTK'),
69       ('muop28x03'), ('yi3nm0d73')) x(v)
70WHERE  citext_hash(v)::bit(32) != citext_hash_extended(v, 0)::bit(32)
71       OR citext_hash(v)::bit(32) = citext_hash_extended(v, 1)::bit(32);

142-- LIKE should be case-insensitive
143SELECT name FROM srt WHERE name     LIKE '%a%' ORDER BY name;
144SELECT name FROM srt WHERE name NOT LIKE '%b%' ORDER BY name;
145SELECT name FROM srt WHERE name     LIKE '%A%' ORDER BY name;
146SELECT name FROM srt WHERE name NOT LIKE '%B%' ORDER BY name;

148-- ~~ should be case-insensitive
149SELECT name FROM srt WHERE name ~~  '%a%' ORDER BY name;
150SELECT name FROM srt WHERE name !~~ '%b%' ORDER BY name;
151SELECT name FROM srt WHERE name ~~  '%A%' ORDER BY name;
152SELECT name FROM srt WHERE name !~~ '%B%' ORDER BY name;

154-- ~ should be case-insensitive
155SELECT name FROM srt WHERE name ~  '^a' ORDER BY name;
156SELECT name FROM srt WHERE name !~ 'a$' ORDER BY name;
157SELECT name FROM srt WHERE name ~  '^A' ORDER BY name;
158SELECT name FROM srt WHERE name !~ 'A$' ORDER BY name;
