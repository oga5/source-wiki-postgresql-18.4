8-- It's a bit tricky to test SYSTEM_TIME in a platform-independent way.
9-- We can test the zero-time corner case ...
10SELECT count(*) FROM test_tablesample TABLESAMPLE system_time (0);
11-- ... and we assume that this will finish before running out of time:
12SELECT count(*) FROM test_tablesample TABLESAMPLE system_time (100000);
13
14-- bad parameters should get through planning, but not execution:
15EXPLAIN (COSTS OFF)
16SELECT id FROM test_tablesample TABLESAMPLE system_time (-1);

18SELECT id FROM test_tablesample TABLESAMPLE system_time (-1);
