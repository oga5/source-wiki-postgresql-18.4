1/*-------------------------------------------------------------------------
2 *
3 * trgm_regexp.c
4 *	  Regular expression matching using trigrams.
5 *
6 * The general idea of trigram index support for a regular expression (regex)
7 * search is to transform the regex into a logical expression on trigrams.
8 * For example:
9 *
10 *	 (ab|cd)efg  =>  ((abe & bef) | (cde & def)) & efg
11 *
12 * If a string matches the regex, then it must match the logical expression on
13 * trigrams.  The opposite is not necessarily true, however: a string that
14 * matches the logical expression might not match the original regex.  Such
15 * false positives are removed via recheck, by running the regular regex match
16 * operator on the retrieved heap tuple.
17 *
18 * Since the trigram expression involves both AND and OR operators, we can't
19 * expect the core index machinery to evaluate it completely.  Instead, the
20 * result of regex analysis is a list of trigrams to be sought in the index,
21 * plus a simplified graph that is used by trigramsMatchGraph() to determine
22 * whether a particular indexed value matches the expression.
23 *
24 * Converting a regex to a trigram expression is based on analysis of an
25 * automaton corresponding to the regex.  The algorithm consists of four
26 * stages:
27 *
28 * 1) Compile the regexp to NFA form.  This is handled by the PostgreSQL
29 *	  regexp library, which provides accessors for its opaque regex_t struct
30 *	  to expose the NFA state graph and the "colors" (sets of equivalent
31 *	  characters) used as state transition labels.
32 *
33 * 2) Transform the original NFA into an expanded graph, where arcs
34 *	  are labeled with trigrams that must be present in order to move from
35 *	  one state to another via the arcs.  The trigrams used in this stage
36 *	  consist of colors, not characters, as in the original NFA.
37 *
38 * 3) Expand the color trigrams into regular trigrams consisting of
39 *	  characters.  If too many distinct trigrams are produced, trigrams are
40 *	  eliminated and the graph is simplified until it's simple enough.
41 *
42 * 4) Finally, the resulting graph is packed into a TrgmPackedGraph struct,
43 *	  and returned to the caller.
44 *
45 * 1) Compile the regexp to NFA form
46 * ---------------------------------
47 * The automaton returned by the regexp compiler is a graph where vertices
48 * are "states" and arcs are labeled with colors.  Each color represents
49 * a set of characters, so that all characters assigned to the same color
50 * are interchangeable, so far as matching the regexp is concerned.  There
51 * are two special states: "initial" and "final".  A state can have multiple
52 * outgoing arcs labeled with the same color, which makes the automaton
53 * non-deterministic, because it can be in many states simultaneously.
54 *
55 * Note that this NFA is already lossy compared to the original regexp,
56 * since it ignores some regex features such as lookahead constraints and
57 * backref matching.  This is OK for our purposes since it's still the case
58 * that only strings matching the NFA can possibly satisfy the regexp.
59 *
60 * 2) Transform the original NFA into an expanded graph
61 * ----------------------------------------------------
62 * In the 2nd stage, the automaton is transformed into a graph based on the
63 * original NFA.  Each state in the expanded graph represents a state from
64 * the original NFA, plus a prefix identifying the last two characters
65 * (colors, to be precise) seen before entering the state.  There can be
66 * multiple states in the expanded graph for each state in the original NFA,
67 * depending on what characters can precede it.  A prefix position can be
68 * "unknown" if it's uncertain what the preceding character was, or "blank"
69 * if the character was a non-word character (we don't need to distinguish
70 * which non-word character it was, so just think of all of them as blanks).
71 *
72 * For convenience in description, call an expanded-state identifier
73 * (two prefix colors plus a state number from the original NFA) an
74 * "enter key".
75 *
76 * Each arc of the expanded graph is labeled with a trigram that must be
77 * present in the string to match.  We can construct this from an out-arc of
78 * the underlying NFA state by combining the expanded state's prefix with the
79 * color label of the underlying out-arc, if neither prefix position is
80 * "unknown".  But note that some of the colors in the trigram might be
81 * "blank".  This is OK since we want to generate word-boundary trigrams as
82 * the regular trigram machinery would, if we know that some word characters
83 * must be adjacent to a word boundary in all strings matching the NFA.
84 *
85 * The expanded graph can also have fewer states than the original NFA,
86 * because we don't bother to make a separate state entry unless the state
87 * is reachable by a valid arc.  When an enter key is reachable from a state
88 * of the expanded graph, but we do not know a complete trigram associated
89 * with that transition, we cannot make a valid arc; instead we insert the
90 * enter key into the enterKeys list of the source state.  This effectively
91 * means that the two expanded states are not reliably distinguishable based
92 * on examining trigrams.
93 *
94 * So the expanded graph resembles the original NFA, but the arcs are
95 * labeled with trigrams instead of individual characters, and there may be
96 * more or fewer states.  It is a lossy representation of the original NFA:
97 * any string that matches the original regexp must match the expanded graph,
98 * but the reverse is not true.
99 *
100 * We build the expanded graph through a breadth-first traversal of states
101 * reachable from the initial state.  At each reachable state, we identify the
102 * states reachable from it without traversing a predictable trigram, and add
103 * those states' enter keys to the current state.  Then we generate all
104 * out-arcs leading out of this collection of states that have predictable
105 * trigrams, adding their target states to the queue of states to examine.
106 *
107 * When building the graph, if the number of states or arcs exceed pre-defined
108 * limits, we give up and simply mark any states not yet processed as final
109 * states.  Roughly speaking, that means that we make use of some portion from
110 * the beginning of the regexp.  Also, any colors that have too many member
111 * characters are treated as "unknown", so that we can't derive trigrams
112 * from them.
113 *
114 * 3) Expand the color trigrams into regular trigrams
115 * --------------------------------------------------
116 * The trigrams in the expanded graph are "color trigrams", consisting
117 * of three consecutive colors that must be present in the string. But for
118 * search, we need regular trigrams consisting of characters. In the 3rd
119 * stage, the color trigrams are expanded into regular trigrams. Since each
120 * color can represent many characters, the total number of regular trigrams
121 * after expansion could be very large. Because searching the index for
122 * thousands of trigrams would be slow, and would likely produce so many
123 * false positives that we would have to traverse a large fraction of the
124 * index, the graph is simplified further in a lossy fashion by removing
125 * color trigrams. When a color trigram is removed, the states connected by
126 * any arcs labeled with that trigram are merged.
127 *
128 * Trigrams do not all have equivalent value for searching: some of them are
129 * more frequent and some of them are less frequent. Ideally, we would like
130 * to know the distribution of trigrams, but we don't. But because of padding
131 * we know for sure that the empty character is more frequent than others,
132 * so we can penalize trigrams according to presence of whitespace. The
133 * penalty assigned to each color trigram is the number of simple trigrams
134 * it would produce, times the penalties[] multiplier associated with its
135 * whitespace content. (The penalties[] constants were calculated by analysis
136 * of some real-life text.) We eliminate color trigrams starting with the
137 * highest-penalty one, until we get to a total penalty of no more than
138 * WISH_TRGM_PENALTY. However, we cannot remove a color trigram if that would
139 * lead to merging the initial and final states, so we may not be able to
140 * reach WISH_TRGM_PENALTY. It's still okay so long as we have no more than
141 * MAX_TRGM_COUNT simple trigrams in total, otherwise we fail.
142 *
143 * 4) Pack the graph into a compact representation
144 * -----------------------------------------------
145 * The 2nd and 3rd stages might have eliminated or merged many of the states
146 * and trigrams created earlier, so in this final stage, the graph is
147 * compacted and packed into a simpler struct that contains only the
148 * information needed to evaluate it.
149 *
150 * ALGORITHM EXAMPLE:
151 *
152 * Consider the example regex "ab[cd]".  This regex is transformed into the
153 * following NFA (for simplicity we show colors as their single members):
154 *
155 *					  4#
156 *					c/
157 *		 a	   b	/
158 *	 1* --- 2 ---- 3
159 *					\
160 *					d\
161 *					  5#
162 *
163 * We use * to mark initial state and # to mark final state. It's not depicted,
164 * but states 1, 4, 5 have self-referencing arcs for all possible characters,
165 * because this pattern can match to any part of a string.
166 *
167 * As the result of stage 2 we will have the following graph:
168 *
169 *		  abc	 abd
170 *	 2# <---- 1* ----> 3#
171 *
172 * The process for generating this graph is:
173 * 1) Create state 1 with enter key (UNKNOWN, UNKNOWN, 1).
174 * 2) Add key (UNKNOWN, "a", 2) to state 1.
175 * 3) Add key ("a", "b", 3) to state 1.
176 * 4) Create new state 2 with enter key ("b", "c", 4).  Add an arc
177 *	  from state 1 to state 2 with label trigram "abc".
178 * 5) Mark state 2 final because state 4 of source NFA is marked as final.
179 * 6) Create new state 3 with enter key ("b", "d", 5).  Add an arc
180 *	  from state 1 to state 3 with label trigram "abd".
181 * 7) Mark state 3 final because state 5 of source NFA is marked as final.
182 *

269/*
270 * A "prefix" is information about the colors of the last two characters read
271 * before reaching a specific NFA state.  These colors can have special values
272 * COLOR_UNKNOWN and COLOR_BLANK.  COLOR_UNKNOWN means that we have no
273 * information, for example because we read some character of an unexpandable
274 * color.  COLOR_BLANK means that we read a non-word character.
275 *
276 * We call a prefix ambiguous if at least one of its colors is unknown.  It's
277 * fully ambiguous if both are unknown, partially ambiguous if only the first
278 * is unknown.  (The case of first color known, second unknown is not valid.)
279 *
280 * Wholly- or partly-blank prefixes are mostly handled the same as regular
281 * color prefixes.  This allows us to generate appropriate partly-blank
282 * trigrams when the NFA requires word character(s) to appear adjacent to
283 * non-word character(s).
284 */ view japanese
285typedef int TrgmColor;

514/*
515 * Main entry point to process a regular expression.
516 *
517 * Returns an array of trigrams required by the regular expression, or NULL if
518 * the regular expression was too complex to analyze.  In addition, a packed
519 * graph representation of the regex is returned into *graph.  The results
520 * must be allocated in rcontext (which might or might not be the current
521 * context).
522 */ view japanese
523TRGM *
524createTrgmNFA(text *text_re, Oid collation,
525			  TrgmPackedGraph **graph, MemoryContext rcontext)
526{
527	TRGM	   *trg;
528	regex_t		regex;
529	MemoryContext tmpcontext;
530	MemoryContext oldcontext;
531
532	/*
533	 * This processing generates a great deal of cruft, which we'd like to
534	 * clean up before returning (since this function may be called in a
535	 * query-lifespan memory context).  Make a temp context we can work in so
536	 * that cleanup is easy.
537	 */ view japanese
538	tmpcontext = AllocSetContextCreate(CurrentMemoryContext,
539									   "createTrgmNFA temporary context",
540									   ALLOCSET_DEFAULT_SIZES);
541	oldcontext = MemoryContextSwitchTo(tmpcontext);
542
543	/*
544	 * Stage 1: Compile the regexp into a NFA, using the regexp library.
545	 */ view japanese
546#ifdef IGNORECASE
547	RE_compile(&regex, text_re,
548			   REG_ADVANCED | REG_NOSUB | REG_ICASE, collation);
549#else
550	RE_compile(&regex, text_re,
551			   REG_ADVANCED | REG_NOSUB, collation);
552#endif
553
554	trg = createTrgmNFAInternal(&regex, graph, rcontext);
555
556	/* Clean up all the cruft we created (including regex) */ view japanese
557	MemoryContextSwitchTo(oldcontext);
558	MemoryContextDelete(tmpcontext);
559
560	return trg;
561}

563/*
564 * Body of createTrgmNFA, exclusive of regex compilation/freeing.
565 */ view japanese
566static TRGM *
567createTrgmNFAInternal(regex_t *regex, TrgmPackedGraph **graph,
568					  MemoryContext rcontext)
569{
570	TRGM	   *trg;
571	TrgmNFA		trgmNFA;
572
573	trgmNFA.regex = regex;
574
575	/* Collect color information from the regex */ view japanese
576	getColorInfo(regex, &trgmNFA);
577
578#ifdef TRGM_REGEXP_DEBUG
579	printSourceNFA(regex, trgmNFA.colorInfo, trgmNFA.ncolors);
580#endif
581
582	/*
583	 * Stage 2: Create an expanded graph from the source NFA.
584	 */ view japanese
585	transformGraph(&trgmNFA);
586
587#ifdef TRGM_REGEXP_DEBUG
588	printTrgmNFA(&trgmNFA);
589#endif
590
591	/*
592	 * Fail if we were unable to make a nontrivial graph, ie it is possible to
593	 * get from the initial state to the final state without reading any
594	 * predictable trigram.
595	 */ view japanese
596	if (trgmNFA.initState->flags & TSTATE_FIN)
597		return NULL;
598
599	/*
600	 * Stage 3: Select color trigrams to expand.  Fail if too many trigrams.
601	 */ view japanese
602	if (!selectColorTrigrams(&trgmNFA))
603		return NULL;
604
605	/*
606	 * Stage 4: Expand color trigrams and pack graph into final
607	 * representation.
608	 */ view japanese
609	trg = expandColorTrigrams(&trgmNFA, rcontext);
610
611	*graph = packGraph(&trgmNFA, rcontext);
612
613#ifdef TRGM_REGEXP_DEBUG
614	printTrgmPackedGraph(*graph, trg);
615#endif
616
617	return trg;
618}

620/*
621 * Main entry point for evaluating a graph during index scanning.
622 *
623 * The check[] array is indexed by trigram number (in the array of simple
624 * trigrams returned by createTrgmNFA), and holds true for those trigrams
625 * that are present in the index entry being checked.
626 */ view japanese
627bool
628trigramsMatchGraph(TrgmPackedGraph *graph, bool *check)
629{
630	int			i,
631				j,
632				k,
633				queueIn,
634				queueOut;
635
636	/*
637	 * Reset temporary working areas.
638	 */ view japanese
639	memset(graph->colorTrigramsActive, 0,
640		   sizeof(bool) * graph->colorTrigramsCount);
641	memset(graph->statesActive, 0, sizeof(bool) * graph->statesCount);
642
643	/*
644	 * Check which color trigrams were matched.  A match for any simple
645	 * trigram associated with a color trigram counts as a match of the color
646	 * trigram.
647	 */ view japanese
648	j = 0;
649	for (i = 0; i < graph->colorTrigramsCount; i++)
650	{
651		int			cnt = graph->colorTrigramGroups[i];
652
653		for (k = j; k < j + cnt; k++)
654		{
655			if (check[k])
656			{
657				/*
658				 * Found one matched trigram in the group. Can skip the rest
659				 * of them and go to the next group.
660				 */ view japanese
661				graph->colorTrigramsActive[i] = true;
662				break;
663			}
664		}
665		j = j + cnt;
666	}
667
668	/*
669	 * Initialize the statesQueue to hold just the initial state.  Note:
670	 * statesQueue has room for statesCount entries, which is certainly enough
671	 * since no state will be put in the queue more than once. The
672	 * statesActive array marks which states have been queued.
673	 */ view japanese
674	graph->statesActive[0] = true;
675	graph->statesQueue[0] = 0;
676	queueIn = 0;
677	queueOut = 1;
678
679	/* Process queued states as long as there are any. */ view japanese
680	while (queueIn < queueOut)
681	{
682		int			stateno = graph->statesQueue[queueIn++];
683		TrgmPackedState *state = &graph->states[stateno];
684		int			cnt = state->arcsCount;
685
686		/* Loop over state's out-arcs */ view japanese
687		for (i = 0; i < cnt; i++)
688		{
689			TrgmPackedArc *arc = &state->arcs[i];
690
691			/*
692			 * If corresponding color trigram is present then activate the
693			 * corresponding state.  We're done if that's the final state,
694			 * otherwise queue the state if it's not been queued already.
695			 */ view japanese
696			if (graph->colorTrigramsActive[arc->colorTrgm])
697			{
698				int			nextstate = arc->targetState;
699
700				if (nextstate == 1)
701					return true;	/* success: final state is reachable */ view japanese
702
703				if (!graph->statesActive[nextstate])
704				{
705					graph->statesActive[nextstate] = true;
706					graph->statesQueue[queueOut++] = nextstate;
707				}
708			}
709		}
710	}
711
712	/* Queue is empty, so match fails. */ view japanese
713	return false;
714}

716/*
717 * Compile regex string into struct at *regex.
718 * NB: pg_regfree must be applied to regex if this completes successfully.
719 */ view japanese
720static void
721RE_compile(regex_t *regex, text *text_re, int cflags, Oid collation)
722{
723	int			text_re_len = VARSIZE_ANY_EXHDR(text_re);
724	char	   *text_re_val = VARDATA_ANY(text_re);
725	pg_wchar   *pattern;
726	int			pattern_len;
727	int			regcomp_result;
728	char		errMsg[100];
729
730	/* Convert pattern string to wide characters */ view japanese
731	pattern = (pg_wchar *) palloc((text_re_len + 1) * sizeof(pg_wchar));
732	pattern_len = pg_mb2wchar_with_len(text_re_val,
733									   pattern,
734									   text_re_len);
735
736	/* Compile regex */ view japanese
737	regcomp_result = pg_regcomp(regex,
738								pattern,
739								pattern_len,
740								cflags,
741								collation);
742
743	pfree(pattern);
744
745	if (regcomp_result != REG_OKAY)
746	{
747		/* re didn't compile (no need for pg_regfree, if so) */ view japanese
748		pg_regerror(regcomp_result, regex, errMsg, sizeof(errMsg));
749		ereport(ERROR,
750				(errcode(ERRCODE_INVALID_REGULAR_EXPRESSION),
751				 errmsg("invalid regular expression: %s", errMsg)));
752	}
753}

761/*
762 * Fill TrgmColorInfo structure for each color using regex export functions.
763 */ view japanese
764static void
765getColorInfo(regex_t *regex, TrgmNFA *trgmNFA)
766{
767	int			colorsCount = pg_reg_getnumcolors(regex);
768	int			i;
769
770	trgmNFA->ncolors = colorsCount;
771	trgmNFA->colorInfo = (TrgmColorInfo *)
772		palloc0(colorsCount * sizeof(TrgmColorInfo));
773
774	/*
775	 * Loop over colors, filling TrgmColorInfo about each.  Note we include
776	 * WHITE (0) even though we know it'll be reported as non-expandable.
777	 */ view japanese
778	for (i = 0; i < colorsCount; i++)
779	{
780		TrgmColorInfo *colorInfo = &trgmNFA->colorInfo[i];
781		int			charsCount = pg_reg_getnumcharacters(regex, i);
782		pg_wchar   *chars;
783		int			j;
784
785		if (charsCount < 0 || charsCount > COLOR_COUNT_LIMIT)
786		{
787			/* Non expandable, or too large to work with */ view japanese
788			colorInfo->expandable = false;
789			continue;
790		}
791
792		colorInfo->expandable = true;
793		colorInfo->containsNonWord = false;
794		colorInfo->wordChars = (trgm_mb_char *)
795			palloc(sizeof(trgm_mb_char) * charsCount);
796		colorInfo->wordCharsCount = 0;
797
798		/* Extract all the chars in this color */ view japanese
799		chars = (pg_wchar *) palloc(sizeof(pg_wchar) * charsCount);
800		pg_reg_getcharacters(regex, i, chars, charsCount);
801
802		/*
803		 * Convert characters back to multibyte form, and save only those that
804		 * are word characters.  Set "containsNonWord" if any non-word
805		 * character.  (Note: it'd probably be nicer to keep the chars in
806		 * pg_wchar format for now, but ISWORDCHR wants to see multibyte.)
807		 */ view japanese
808		for (j = 0; j < charsCount; j++)
809		{
810			trgm_mb_char c;
811			int			clen = convertPgWchar(chars[j], &c);
812
813			if (!clen)
814				continue;		/* ok to ignore it altogether */ view japanese
815			if (ISWORDCHR(c.bytes, clen))
816				colorInfo->wordChars[colorInfo->wordCharsCount++] = c;
817			else
818				colorInfo->containsNonWord = true;
819		}
820
821		pfree(chars);
822	}
823}

825/*
826 * Convert pg_wchar to multibyte format.
827 * Returns 0 if the character should be ignored completely, else returns its
828 * byte length.
829 */ view japanese
830static int
831convertPgWchar(pg_wchar c, trgm_mb_char *result)
832{
833	/* "s" has enough space for a multibyte character and a trailing NUL */ view japanese
834	char		s[MAX_MULTIBYTE_CHAR_LEN + 1];
835	int			clen;
836
837	/*
838	 * We can ignore the NUL character, since it can never appear in a PG text
839	 * string.  This avoids the need for various special cases when
840	 * reconstructing trigrams.
841	 */ view japanese
842	if (c == 0)
843		return 0;
844
845	/* Do the conversion, making sure the result is NUL-terminated */ view japanese
846	memset(s, 0, sizeof(s));
847	clen = pg_wchar2mb_with_len(&c, s, 1);
848
849	/*
850	 * In IGNORECASE mode, we can ignore uppercase characters.  We assume that
851	 * the regex engine generated both uppercase and lowercase equivalents
852	 * within each color, since we used the REG_ICASE option; so there's no
853	 * need to process the uppercase version.
854	 *
855	 * XXX this code is dependent on the assumption that str_tolower() works
856	 * the same as the regex engine's internal case folding machinery.  Might
857	 * be wiser to expose pg_wc_tolower and test whether c ==
858	 * pg_wc_tolower(c). On the other hand, the trigrams in the index were
859	 * created using str_tolower(), so we're probably screwed if there's any
860	 * incompatibility anyway.
861	 */ view japanese
862#ifdef IGNORECASE
863	{
864		char	   *lowerCased = str_tolower(s, clen, DEFAULT_COLLATION_OID);
865
866		if (strcmp(lowerCased, s) != 0)
867		{
868			pfree(lowerCased);
869			return 0;
870		}
871		pfree(lowerCased);
872	}
873#endif
874
875	/* Fill result with exactly MAX_MULTIBYTE_CHAR_LEN bytes */ view japanese
876	memcpy(result->bytes, s, MAX_MULTIBYTE_CHAR_LEN);
877	return clen;
878}

882 * Subroutines for expanding original NFA graph into a trigram graph (stage 2).

886/*
887 * Transform the graph, given a regex and extracted color information.
888 *
889 * We create and process a queue of expanded-graph states until all the states
890 * are processed.
891 *
892 * This algorithm may be stopped due to resource limitation. In this case we
893 * force every unprocessed branch to immediately finish with matching (this
894 * can give us false positives but no false negatives) by marking all
895 * unprocessed states as final.
896 */ view japanese
897static void
898transformGraph(TrgmNFA *trgmNFA)
899{
900	HASHCTL		hashCtl;
901	TrgmStateKey initkey;
902	TrgmState  *initstate;
903	ListCell   *lc;
904
905	/* Initialize this stage's workspace in trgmNFA struct */ view japanese
906	trgmNFA->queue = NIL;
907	trgmNFA->keysQueue = NIL;
908	trgmNFA->arcsCount = 0;
909	trgmNFA->overflowed = false;
910
911	/* Create hashtable for states */ view japanese
912	hashCtl.keysize = sizeof(TrgmStateKey);
913	hashCtl.entrysize = sizeof(TrgmState);
914	hashCtl.hcxt = CurrentMemoryContext;
915	trgmNFA->states = hash_create("Trigram NFA",
916								  1024,
917								  &hashCtl,
918								  HASH_ELEM | HASH_BLOBS | HASH_CONTEXT);
919	trgmNFA->nstates = 0;
920
921	/* Create initial state: ambiguous prefix, NFA's initial state */ view japanese
922	MemSet(&initkey, 0, sizeof(initkey));
923	initkey.prefix.colors[0] = COLOR_UNKNOWN;
924	initkey.prefix.colors[1] = COLOR_UNKNOWN;
925	initkey.nstate = pg_reg_getinitialstate(trgmNFA->regex);
926
927	initstate = getState(trgmNFA, &initkey);
928	initstate->flags |= TSTATE_INIT;
929	trgmNFA->initState = initstate;
930
931	/*
932	 * Recursively build the expanded graph by processing queue of states
933	 * (breadth-first search).  getState already put initstate in the queue.
934	 * Note that getState will append new states to the queue within the loop,
935	 * too; this works as long as we don't do repeat fetches using the "lc"
936	 * pointer.
937	 */ view japanese
938	foreach(lc, trgmNFA->queue)
939	{
940		TrgmState  *state = (TrgmState *) lfirst(lc);
941
942		/*
943		 * If we overflowed then just mark state as final.  Otherwise do
944		 * actual processing.
945		 */ view japanese
946		if (trgmNFA->overflowed)
947			state->flags |= TSTATE_FIN;
948		else
949			processState(trgmNFA, state);
950
951		/* Did we overflow? */ view japanese
952		if (trgmNFA->arcsCount > MAX_EXPANDED_ARCS ||
953			hash_get_num_entries(trgmNFA->states) > MAX_EXPANDED_STATES)
954			trgmNFA->overflowed = true;
955	}
956}

958/*
959 * Process one state: add enter keys and then add outgoing arcs.
960 */ view japanese
961static void
962processState(TrgmNFA *trgmNFA, TrgmState *state)
963{
964	ListCell   *lc;
965
966	/* keysQueue should be NIL already, but make sure */ view japanese
967	trgmNFA->keysQueue = NIL;
968
969	/*
970	 * Add state's own key, and then process all keys added to keysQueue until
971	 * queue is finished.  But we can quit if the state gets marked final.
972	 */ view japanese
973	addKey(trgmNFA, state, &state->stateKey);
974	foreach(lc, trgmNFA->keysQueue)
975	{
976		TrgmStateKey *key = (TrgmStateKey *) lfirst(lc);
977
978		if (state->flags & TSTATE_FIN)
979			break;
980		addKey(trgmNFA, state, key);
981	}
982
983	/* Release keysQueue to clean up for next cycle */ view japanese
984	list_free(trgmNFA->keysQueue);
985	trgmNFA->keysQueue = NIL;
986
987	/*
988	 * Add outgoing arcs only if state isn't final (we have no interest in
989	 * outgoing arcs if we already match)
990	 */ view japanese
991	if (!(state->flags & TSTATE_FIN))
992		addArcs(trgmNFA, state);
993}
994
995/*
996 * Add the given enter key into the state's enterKeys list, and determine
997 * whether this should result in any further enter keys being added.
998 * If so, add those keys to keysQueue so that processState will handle them.
999 *
1000 * If the enter key is for the NFA's final state, mark state as TSTATE_FIN.
1001 * This situation means that we can reach the final state from this expanded
1002 * state without reading any predictable trigram, so we must consider this
1003 * state as an accepting one.
1004 *
1005 * The given key could be a duplicate of one already in enterKeys, or be
1006 * redundant with some enterKeys.  So we check that before doing anything.
1007 *
1008 * Note that we don't generate any actual arcs here.  addArcs will do that
1009 * later, after we have identified all the enter keys for this state.
1010 */ view japanese
1011static void
1012addKey(TrgmNFA *trgmNFA, TrgmState *state, TrgmStateKey *key)
1013{
1014	regex_arc_t *arcs;
1015	TrgmStateKey destKey;
1016	ListCell   *cell;
1017	int			i,
1018				arcsCount;
1019
1020	/*
1021	 * Ensure any pad bytes in destKey are zero, since it may get used as a
1022	 * hashtable key by getState.
1023	 */
1024	MemSet(&destKey, 0, sizeof(destKey));
1025
1026	/*
1027	 * Compare key to each existing enter key of the state to check for
1028	 * redundancy.  We can drop either old key(s) or the new key if we find
1029	 * redundancy.
1030	 */ view japanese
1031	foreach(cell, state->enterKeys)
1032	{
1033		TrgmStateKey *existingKey = (TrgmStateKey *) lfirst(cell);
1034
1035		if (existingKey->nstate == key->nstate)
1036		{
1037			if (prefixContains(&existingKey->prefix, &key->prefix))
1038			{
1039				/* This old key already covers the new key. Nothing to do */ view japanese
1040				return;
1041			}
1042			if (prefixContains(&key->prefix, &existingKey->prefix))
1043			{
1044				/*
1045				 * The new key covers this old key. Remove the old key, it's
1046				 * no longer needed once we add this key to the list.
1047				 */ view japanese
1048				state->enterKeys = foreach_delete_current(state->enterKeys,
1049														  cell);
1050			}
1051		}
1052	}
1053
1054	/* No redundancy, so add this key to the state's list */ view japanese
1055	state->enterKeys = lappend(state->enterKeys, key);
1056
1057	/* If state is now known final, mark it and we're done */ view japanese
1058	if (key->nstate == pg_reg_getfinalstate(trgmNFA->regex))
1059	{
1060		state->flags |= TSTATE_FIN;
1061		return;
1062	}
1063
1064	/*
1065	 * Loop through all outgoing arcs of the corresponding state in the
1066	 * original NFA.
1067	 */ view japanese
1068	arcsCount = pg_reg_getnumoutarcs(trgmNFA->regex, key->nstate);
1069	arcs = (regex_arc_t *) palloc(sizeof(regex_arc_t) * arcsCount);
1070	pg_reg_getoutarcs(trgmNFA->regex, key->nstate, arcs, arcsCount);
1071
1072	for (i = 0; i < arcsCount; i++)
1073	{
1074		regex_arc_t *arc = &arcs[i];
1075
1076		if (pg_reg_colorisbegin(trgmNFA->regex, arc->co))
1077		{
1078			/*
1079			 * Start of line/string (^).  Trigram extraction treats start of
1080			 * line same as start of word: double space prefix is added.
1081			 * Hence, make an enter key showing we can reach the arc
1082			 * destination with all-blank prefix.
1083			 */ view japanese
1084			destKey.prefix.colors[0] = COLOR_BLANK;
1085			destKey.prefix.colors[1] = COLOR_BLANK;
1086			destKey.nstate = arc->to;
1087
1088			/* Add enter key to this state */
1089			addKeyToQueue(trgmNFA, &destKey);
1090		}
1091		else if (pg_reg_colorisend(trgmNFA->regex, arc->co))
1092		{
1093			/*
1094			 * End of line/string ($).  We must consider this arc as a
1095			 * transition that doesn't read anything.  The reason for adding
1096			 * this enter key to the state is that if the arc leads to the
1097			 * NFA's final state, we must mark this expanded state as final.
1098			 */ view japanese
1099			destKey.prefix.colors[0] = COLOR_UNKNOWN;
1100			destKey.prefix.colors[1] = COLOR_UNKNOWN;
1101			destKey.nstate = arc->to;
1102
1103			/* Add enter key to this state */ view japanese
1104			addKeyToQueue(trgmNFA, &destKey);
1105		}
1106		else if (arc->co >= 0)
1107		{
1108			/* Regular color (including WHITE) */ view japanese
1109			TrgmColorInfo *colorInfo = &trgmNFA->colorInfo[arc->co];
1110
1111			if (colorInfo->expandable)
1112			{
1113				if (colorInfo->containsNonWord &&
1114					!validArcLabel(key, COLOR_BLANK))
1115				{
1116					/*
1117					 * We can reach the arc destination after reading a
1118					 * non-word character, but the prefix is not something
1119					 * that addArc will accept with COLOR_BLANK, so no trigram
1120					 * arc can get made for this transition.  We must make an
1121					 * enter key to show that the arc destination is
1122					 * reachable.  Set it up with an all-blank prefix, since
1123					 * that corresponds to what the trigram extraction code
1124					 * will do at a word starting boundary.
1125					 */ view japanese
1126					destKey.prefix.colors[0] = COLOR_BLANK;
1127					destKey.prefix.colors[1] = COLOR_BLANK;
1128					destKey.nstate = arc->to;
1129					addKeyToQueue(trgmNFA, &destKey);
1130				}
1131
1132				if (colorInfo->wordCharsCount > 0 &&
1133					!validArcLabel(key, arc->co))
1134				{
1135					/*
1136					 * We can reach the arc destination after reading a word
1137					 * character, but the prefix is not something that addArc
1138					 * will accept, so no trigram arc can get made for this
1139					 * transition.  We must make an enter key to show that the
1140					 * arc destination is reachable.  The prefix for the enter
1141					 * key should reflect the info we have for this arc.
1142					 */ view japanese
1143					destKey.prefix.colors[0] = key->prefix.colors[1];
1144					destKey.prefix.colors[1] = arc->co;
1145					destKey.nstate = arc->to;
1146					addKeyToQueue(trgmNFA, &destKey);
1147				}
1148			}
1149			else
1150			{
1151				/*
1152				 * Unexpandable color.  Add enter key with ambiguous prefix,
1153				 * showing we can reach the destination from this state, but
1154				 * the preceding colors will be uncertain.  (We do not set the
1155				 * first prefix color to key->prefix.colors[1], because a
1156				 * prefix of known followed by unknown is invalid.)
1157				 */ view japanese
1158				destKey.prefix.colors[0] = COLOR_UNKNOWN;
1159				destKey.prefix.colors[1] = COLOR_UNKNOWN;
1160				destKey.nstate = arc->to;
1161				addKeyToQueue(trgmNFA, &destKey);
1162			}
1163		}
1164		else
1165		{
1166			/* RAINBOW: treat as unexpandable color */ view japanese
1167			destKey.prefix.colors[0] = COLOR_UNKNOWN;
1168			destKey.prefix.colors[1] = COLOR_UNKNOWN;
1169			destKey.nstate = arc->to;
1170			addKeyToQueue(trgmNFA, &destKey);
1171		}
1172	}
1173
1174	pfree(arcs);
1175}

1189/*
1190 * Add outgoing arcs from given state, whose enter keys are all now known.
1191 */ view japanese
1192static void
1193addArcs(TrgmNFA *trgmNFA, TrgmState *state)
1194{
1195	TrgmStateKey destKey;
1196	ListCell   *cell;
1197	regex_arc_t *arcs;
1198	int			arcsCount,
1199				i;
1200
1201	/*
1202	 * Ensure any pad bytes in destKey are zero, since it may get used as a
1203	 * hashtable key by getState.
1204	 */ view japanese
1205	MemSet(&destKey, 0, sizeof(destKey));
1206
1207	/*
1208	 * Iterate over enter keys associated with this expanded-graph state. This
1209	 * includes both the state's own stateKey, and any enter keys we added to
1210	 * it during addKey (which represent expanded-graph states that are not
1211	 * distinguishable from this one by means of trigrams).  For each such
1212	 * enter key, examine all the out-arcs of the key's underlying NFA state,
1213	 * and try to make a trigram arc leading to where the out-arc leads.
1214	 * (addArc will deal with whether the arc is valid or not.)
1215	 */ view japanese
1216	foreach(cell, state->enterKeys)
1217	{
1218		TrgmStateKey *key = (TrgmStateKey *) lfirst(cell);
1219
1220		arcsCount = pg_reg_getnumoutarcs(trgmNFA->regex, key->nstate);
1221		arcs = (regex_arc_t *) palloc(sizeof(regex_arc_t) * arcsCount);
1222		pg_reg_getoutarcs(trgmNFA->regex, key->nstate, arcs, arcsCount);
1223
1224		for (i = 0; i < arcsCount; i++)
1225		{
1226			regex_arc_t *arc = &arcs[i];
1227			TrgmColorInfo *colorInfo;
1228
1229			/*
1230			 * Ignore non-expandable colors; addKey already handled the case.
1231			 *
1232			 * We need no special check for WHITE or begin/end pseudocolors
1233			 * here.  We don't need to do any processing for them, and they
1234			 * will be marked non-expandable since the regex engine will have
1235			 * reported them that way.  We do have to watch out for RAINBOW,
1236			 * which has a negative color number.
1237			 */ view japanese
1238			if (arc->co < 0)
1239				continue;
1240			Assert(arc->co < trgmNFA->ncolors);
1241
1242			colorInfo = &trgmNFA->colorInfo[arc->co];
1243			if (!colorInfo->expandable)
1244				continue;
1245
1246			if (colorInfo->containsNonWord)
1247			{
1248				/*
1249				 * Color includes non-word character(s).
1250				 *
1251				 * Generate an arc, treating this transition as occurring on
1252				 * BLANK.  This allows word-ending trigrams to be manufactured
1253				 * if possible.
1254				 */ view japanese
1255				destKey.prefix.colors[0] = key->prefix.colors[1];
1256				destKey.prefix.colors[1] = COLOR_BLANK;
1257				destKey.nstate = arc->to;
1258
1259				addArc(trgmNFA, state, key, COLOR_BLANK, &destKey);
1260			}
1261
1262			if (colorInfo->wordCharsCount > 0)
1263			{
1264				/*
1265				 * Color includes word character(s).
1266				 *
1267				 * Generate an arc.  Color is pushed into prefix of target
1268				 * state.
1269				 */ view japanese
1270				destKey.prefix.colors[0] = key->prefix.colors[1];
1271				destKey.prefix.colors[1] = arc->co;
1272				destKey.nstate = arc->to;
1273
1274				addArc(trgmNFA, state, key, arc->co, &destKey);
1275			}
1276		}
1277
1278		pfree(arcs);
1279	}
1280}

1282/*
1283 * Generate an out-arc of the expanded graph, if it's valid and not redundant.
1284 *
1285 * state: expanded-graph state we want to add an out-arc to
1286 * key: provides prefix colors (key->nstate is not used)
1287 * co: transition color
1288 * destKey: identifier for destination state of expanded graph
1289 */ view japanese
1290static void
1291addArc(TrgmNFA *trgmNFA, TrgmState *state, TrgmStateKey *key,
1292	   TrgmColor co, TrgmStateKey *destKey)
1293{
1294	TrgmArc    *arc;
1295	ListCell   *cell;
1296
1297	/* Do nothing if this wouldn't be a valid arc label trigram */ view japanese
1298	if (!validArcLabel(key, co))
1299		return;
1300
1301	/*
1302	 * Check if we are going to reach key which is covered by a key which is
1303	 * already listed in this state.  If so arc is useless: the NFA can bypass
1304	 * it through a path that doesn't require any predictable trigram, so
1305	 * whether the arc's trigram is present or not doesn't really matter.
1306	 */ view japanese
1307	foreach(cell, state->enterKeys)
1308	{
1309		TrgmStateKey *existingKey = (TrgmStateKey *) lfirst(cell);
1310
1311		if (existingKey->nstate == destKey->nstate &&
1312			prefixContains(&existingKey->prefix, &destKey->prefix))
1313			return;
1314	}
1315
1316	/* Checks were successful, add new arc */ view japanese
1317	arc = (TrgmArc *) palloc(sizeof(TrgmArc));
1318	arc->target = getState(trgmNFA, destKey);
1319	arc->ctrgm.colors[0] = key->prefix.colors[0];
1320	arc->ctrgm.colors[1] = key->prefix.colors[1];
1321	arc->ctrgm.colors[2] = co;
1322
1323	state->arcs = lappend(state->arcs, arc);
1324	trgmNFA->arcsCount++;
1325}

1327/*
1328 * Can we make a valid trigram arc label from the given prefix and arc color?
1329 *
1330 * This is split out so that tests in addKey and addArc will stay in sync.
1331 */ view japanese
1332static bool
1333validArcLabel(TrgmStateKey *key, TrgmColor co)
1334{
1335	/*
1336	 * We have to know full trigram in order to add outgoing arc.  So we can't
1337	 * do it if prefix is ambiguous.
1338	 */ view japanese
1339	if (key->prefix.colors[0] == COLOR_UNKNOWN)
1340		return false;
1341
1342	/* If key->prefix.colors[0] isn't unknown, its second color isn't either */ view japanese
1343	Assert(key->prefix.colors[1] != COLOR_UNKNOWN);
1344	/* And we should not be called with an unknown arc color anytime */ view japanese
1345	Assert(co != COLOR_UNKNOWN);
1346
1347	/*
1348	 * We don't bother with making arcs representing three non-word
1349	 * characters, since that's useless for trigram extraction.
1350	 */ view japanese
1351	if (key->prefix.colors[0] == COLOR_BLANK &&
1352		key->prefix.colors[1] == COLOR_BLANK &&
1353		co == COLOR_BLANK)
1354		return false;
1355
1356	/*
1357	 * We also reject nonblank-blank-anything.  The nonblank-blank-nonblank
1358	 * case doesn't correspond to any trigram the trigram extraction code
1359	 * would make.  The nonblank-blank-blank case is also not possible with
1360	 * RPADDING = 1.  (Note that in many cases we'd fail to generate such a
1361	 * trigram even if it were valid, for example processing "foo bar" will
1362	 * not result in considering the trigram "o  ".  So if you want to support
1363	 * RPADDING = 2, there's more to do than just twiddle this test.)
1364	 */ view japanese
1365	if (key->prefix.colors[0] != COLOR_BLANK &&
1366		key->prefix.colors[1] == COLOR_BLANK)
1367		return false;
1368
1369	/*
1370	 * Other combinations involving blank are valid, in particular we assume
1371	 * blank-blank-nonblank is valid, which presumes that LPADDING is 2.
1372	 *
1373	 * Note: Using again the example "foo bar", we will not consider the
1374	 * trigram "  b", though this trigram would be found by the trigram
1375	 * extraction code.  Since we will find " ba", it doesn't seem worth
1376	 * trying to hack the algorithm to generate the additional trigram.
1377	 */ view japanese
1378
1379	/* arc label is valid */ view japanese
1380	return true;
1381}

1383/*
1384 * Get state of expanded graph for given state key,
1385 * and queue the state for processing if it didn't already exist.
1386 */ view japanese
1387static TrgmState *
1388getState(TrgmNFA *trgmNFA, TrgmStateKey *key)
1389{
1390	TrgmState  *state;
1391	bool		found;
1392
1393	state = (TrgmState *) hash_search(trgmNFA->states, key, HASH_ENTER,
1394									  &found);
1395	if (!found)
1396	{
1397		/* New state: initialize and queue it */ view japanese
1398		state->arcs = NIL;
1399		state->enterKeys = NIL;
1400		state->flags = 0;
1401		/* states are initially given negative numbers */ view japanese
1402		state->snumber = -(++trgmNFA->nstates);
1403		state->parent = NULL;
1404		state->tentFlags = 0;
1405		state->tentParent = NULL;
1406
1407		trgmNFA->queue = lappend(trgmNFA->queue, state);
1408	}
1409	return state;
1410}

1412/*
1413 * Check if prefix1 "contains" prefix2.
1414 *
1415 * "contains" means that any exact prefix (with no ambiguity) that satisfies
1416 * prefix2 also satisfies prefix1.
1417 */ view japanese
1418static bool
1419prefixContains(TrgmPrefix *prefix1, TrgmPrefix *prefix2)
1420{
1421	if (prefix1->colors[1] == COLOR_UNKNOWN)
1422	{
1423		/* Fully ambiguous prefix contains everything */ view japanese
1424		return true;
1425	}
1426	else if (prefix1->colors[0] == COLOR_UNKNOWN)
1427	{
1428		/*
1429		 * Prefix with only first unknown color contains every prefix with
1430		 * same second color.
1431		 */ view japanese
1432		if (prefix1->colors[1] == prefix2->colors[1])
1433			return true;
1434		else
1435			return false;
1436	}
1437	else
1438	{
1439		/* Exact prefix contains only the exact same prefix */ view japanese
1440		if (prefix1->colors[0] == prefix2->colors[0] &&
1441			prefix1->colors[1] == prefix2->colors[1])
1442			return true;
1443		else
1444			return false;
1445	}
1446}

1450 * Subroutines for expanding color trigrams into regular trigrams (stage 3).
1451 *---------------------
1452 */ view japanese

1454/*
1455 * Get vector of all color trigrams in graph and select which of them
1456 * to expand into simple trigrams.
1457 *
1458 * Returns true if OK, false if exhausted resource limits.
1459 */ view japanese
1460static bool
1461selectColorTrigrams(TrgmNFA *trgmNFA)
1462{
1463	HASH_SEQ_STATUS scan_status;
1464	int			arcsCount = trgmNFA->arcsCount,
1465				i;
1466	TrgmState  *state;
1467	ColorTrgmInfo *colorTrgms;
1468	int64		totalTrgmCount;
1469	float4		totalTrgmPenalty;
1470	int			cnumber;
1471
1472	/* Collect color trigrams from all arcs */ view japanese
1473	colorTrgms = (ColorTrgmInfo *) palloc0(sizeof(ColorTrgmInfo) * arcsCount);
1474	trgmNFA->colorTrgms = colorTrgms;
1475
1476	i = 0;
1477	hash_seq_init(&scan_status, trgmNFA->states);
1478	while ((state = (TrgmState *) hash_seq_search(&scan_status)) != NULL)
1479	{
1480		ListCell   *cell;
1481
1482		foreach(cell, state->arcs)
1483		{
1484			TrgmArc    *arc = (TrgmArc *) lfirst(cell);
1485			TrgmArcInfo *arcInfo = (TrgmArcInfo *) palloc(sizeof(TrgmArcInfo));
1486			ColorTrgmInfo *trgmInfo = &colorTrgms[i];
1487
1488			arcInfo->source = state;
1489			arcInfo->target = arc->target;
1490			trgmInfo->ctrgm = arc->ctrgm;
1491			trgmInfo->cnumber = -1;
1492			/* count and penalty will be set below */ view japanese
1493			trgmInfo->expanded = true;
1494			trgmInfo->arcs = list_make1(arcInfo);
1495			i++;
1496		}
1497	}
1498	Assert(i == arcsCount);
1499
1500	/* Remove duplicates, merging their arcs lists */ view japanese
1501	if (arcsCount >= 2)
1502	{
1503		ColorTrgmInfo *p1,
1504				   *p2;
1505
1506		/* Sort trigrams to ease duplicate detection */ view japanese
1507		qsort(colorTrgms, arcsCount, sizeof(ColorTrgmInfo), colorTrgmInfoCmp);
1508
1509		/* p1 is probe point, p2 is last known non-duplicate. */
1510		p2 = colorTrgms;
1511		for (p1 = colorTrgms + 1; p1 < colorTrgms + arcsCount; p1++)
1512		{
1513			if (colorTrgmInfoCmp(p1, p2) > 0)
1514			{
1515				p2++;
1516				*p2 = *p1;
1517			}
1518			else
1519			{
1520				p2->arcs = list_concat(p2->arcs, p1->arcs);
1521			}
1522		}
1523		trgmNFA->colorTrgmsCount = (p2 - colorTrgms) + 1;
1524	}
1525	else
1526	{
1527		trgmNFA->colorTrgmsCount = arcsCount;
1528	}
1529
1530	/*
1531	 * Count number of simple trigrams generated by each color trigram, and
1532	 * also compute a penalty value, which is the number of simple trigrams
1533	 * times a multiplier that depends on its whitespace content.
1534	 *
1535	 * Note: per-color-trigram counts cannot overflow an int so long as
1536	 * COLOR_COUNT_LIMIT is not more than the cube root of INT_MAX, ie about
1537	 * 1290.  However, the grand total totalTrgmCount might conceivably
1538	 * overflow an int, so we use int64 for that within this routine.  Also,
1539	 * penalties are calculated in float4 arithmetic to avoid any overflow
1540	 * worries.
1541	 */ view japanese
1542	totalTrgmCount = 0;
1543	totalTrgmPenalty = 0.0f;
1544	for (i = 0; i < trgmNFA->colorTrgmsCount; i++)
1545	{
1546		ColorTrgmInfo *trgmInfo = &colorTrgms[i];
1547		int			j,
1548					count = 1,
1549					typeIndex = 0;
1550
1551		for (j = 0; j < 3; j++)
1552		{
1553			TrgmColor	c = trgmInfo->ctrgm.colors[j];
1554
1555			typeIndex *= 2;
1556			if (c == COLOR_BLANK)
1557				typeIndex++;
1558			else
1559				count *= trgmNFA->colorInfo[c].wordCharsCount;
1560		}
1561		trgmInfo->count = count;
1562		totalTrgmCount += count;
1563		trgmInfo->penalty = penalties[typeIndex] * (float4) count;
1564		totalTrgmPenalty += trgmInfo->penalty;
1565	}
1566
1567	/* Sort color trigrams in descending order of their penalties */ view japanese
1568	qsort(colorTrgms, trgmNFA->colorTrgmsCount, sizeof(ColorTrgmInfo),
1569		  colorTrgmInfoPenaltyCmp);
1570
1571	/*
1572	 * Remove color trigrams from the graph so long as total penalty of color
1573	 * trigrams exceeds WISH_TRGM_PENALTY.  (If we fail to get down to
1574	 * WISH_TRGM_PENALTY, it's OK so long as total count is no more than
1575	 * MAX_TRGM_COUNT.)  We prefer to remove color trigrams with higher
1576	 * penalty, since those are the most promising for reducing the total
1577	 * penalty.  When removing a color trigram we have to merge states
1578	 * connected by arcs labeled with that trigram.  It's necessary to not
1579	 * merge initial and final states, because our graph becomes useless if
1580	 * that happens; so we cannot always remove the trigram we'd prefer to.
1581	 */ view japanese
1582	for (i = 0; i < trgmNFA->colorTrgmsCount; i++)
1583	{
1584		ColorTrgmInfo *trgmInfo = &colorTrgms[i];
1585		bool		canRemove = true;
1586		ListCell   *cell;
1587
1588		/* Done if we've reached the target */ view japanese
1589		if (totalTrgmPenalty <= WISH_TRGM_PENALTY)
1590			break;
1591
1592#ifdef TRGM_REGEXP_DEBUG
1593		fprintf(stderr, "considering ctrgm %d %d %d, penalty %f, %d arcs\n",
1594				trgmInfo->ctrgm.colors[0],
1595				trgmInfo->ctrgm.colors[1],
1596				trgmInfo->ctrgm.colors[2],
1597				trgmInfo->penalty,
1598				list_length(trgmInfo->arcs));
1599#endif
1600
1601		/*
1602		 * Does any arc of this color trigram connect initial and final
1603		 * states?	If so we can't remove it.
1604		 */ view japanese
1605		foreach(cell, trgmInfo->arcs)
1606		{
1607			TrgmArcInfo *arcInfo = (TrgmArcInfo *) lfirst(cell);
1608			TrgmState  *source = arcInfo->source,
1609					   *target = arcInfo->target;
1610			int			source_flags,
1611						target_flags;
1612
1613#ifdef TRGM_REGEXP_DEBUG
1614			fprintf(stderr, "examining arc to s%d (%x) from s%d (%x)\n",
1615					-target->snumber, target->flags,
1616					-source->snumber, source->flags);
1617#endif
1618
1619			/* examine parent states, if any merging has already happened */ view japanese
1620			while (source->parent)
1621				source = source->parent;
1622			while (target->parent)
1623				target = target->parent;
1624
1625#ifdef TRGM_REGEXP_DEBUG
1626			fprintf(stderr, " ... after completed merges: to s%d (%x) from s%d (%x)\n",
1627					-target->snumber, target->flags,
1628					-source->snumber, source->flags);
1629#endif
1630
1631			/* we must also consider merges we are planning right now */ view japanese
1632			source_flags = source->flags | source->tentFlags;
1633			while (source->tentParent)
1634			{
1635				source = source->tentParent;
1636				source_flags |= source->flags | source->tentFlags;
1637			}
1638			target_flags = target->flags | target->tentFlags;
1639			while (target->tentParent)
1640			{
1641				target = target->tentParent;
1642				target_flags |= target->flags | target->tentFlags;
1643			}
1644
1645#ifdef TRGM_REGEXP_DEBUG
1646			fprintf(stderr, " ... after tentative merges: to s%d (%x) from s%d (%x)\n",
1647					-target->snumber, target_flags,
1648					-source->snumber, source_flags);
1649#endif
1650
1651			/* would fully-merged state have both INIT and FIN set? */ view japanese
1652			if (((source_flags | target_flags) & (TSTATE_INIT | TSTATE_FIN)) ==
1653				(TSTATE_INIT | TSTATE_FIN))
1654			{
1655				canRemove = false;
1656				break;
1657			}
1658
1659			/* ok so far, so remember planned merge */ view japanese
1660			if (source != target)
1661			{
1662#ifdef TRGM_REGEXP_DEBUG
1663				fprintf(stderr, " ... tentatively merging s%d into s%d\n",
1664						-target->snumber, -source->snumber);
1665#endif
1666				target->tentParent = source;
1667				source->tentFlags |= target_flags;
1668			}
1669		}
1670
1671		/*
1672		 * We must reset all the tentFlags/tentParent fields before
1673		 * continuing.  tentFlags could only have become set in states that
1674		 * are the source or parent or tentative parent of one of the current
1675		 * arcs; likewise tentParent could only have become set in states that
1676		 * are the target or parent or tentative parent of one of the current
1677		 * arcs.  There might be some overlap between those sets, but if we
1678		 * clear tentFlags in target states as well as source states, we
1679		 * should be okay even if we visit a state as target before visiting
1680		 * it as a source.
1681		 */ view japanese
1682		foreach(cell, trgmInfo->arcs)
1683		{
1684			TrgmArcInfo *arcInfo = (TrgmArcInfo *) lfirst(cell);
1685			TrgmState  *source = arcInfo->source,
1686					   *target = arcInfo->target;
1687			TrgmState  *ttarget;
1688
1689			/* no need to touch previously-merged states */ view japanese
1690			while (source->parent)
1691				source = source->parent;
1692			while (target->parent)
1693				target = target->parent;
1694
1695			while (source)
1696			{
1697				source->tentFlags = 0;
1698				source = source->tentParent;
1699			}
1700
1701			while ((ttarget = target->tentParent) != NULL)
1702			{
1703				target->tentParent = NULL;
1704				target->tentFlags = 0;	/* in case it was also a source */ view japanese
1705				target = ttarget;
1706			}
1707		}
1708
1709		/* Now, move on if we can't drop this trigram */ view japanese
1710		if (!canRemove)
1711		{
1712#ifdef TRGM_REGEXP_DEBUG
1713			fprintf(stderr, " ... not ok to merge\n");
1714#endif
1715			continue;
1716		}
1717
1718		/* OK, merge states linked by each arc labeled by the trigram */ view japanese
1719		foreach(cell, trgmInfo->arcs)
1720		{
1721			TrgmArcInfo *arcInfo = (TrgmArcInfo *) lfirst(cell);
1722			TrgmState  *source = arcInfo->source,
1723					   *target = arcInfo->target;
1724
1725			while (source->parent)
1726				source = source->parent;
1727			while (target->parent)
1728				target = target->parent;
1729			if (source != target)
1730			{
1731#ifdef TRGM_REGEXP_DEBUG
1732				fprintf(stderr, "merging s%d into s%d\n",
1733						-target->snumber, -source->snumber);
1734#endif
1735				mergeStates(source, target);
1736				/* Assert we didn't merge initial and final states */ view japanese
1737				Assert((source->flags & (TSTATE_INIT | TSTATE_FIN)) !=
1738					   (TSTATE_INIT | TSTATE_FIN));
1739			}
1740		}
1741
1742		/* Mark trigram unexpanded, and update totals */ view japanese
1743		trgmInfo->expanded = false;
1744		totalTrgmCount -= trgmInfo->count;
1745		totalTrgmPenalty -= trgmInfo->penalty;
1746	}
1747
1748	/* Did we succeed in fitting into MAX_TRGM_COUNT? */ view japanese
1749	if (totalTrgmCount > MAX_TRGM_COUNT)
1750		return false;
1751
1752	trgmNFA->totalTrgmCount = (int) totalTrgmCount;
1753
1754	/*
1755	 * Sort color trigrams by colors (will be useful for bsearch in packGraph)
1756	 * and enumerate the color trigrams that are expanded.
1757	 */ view japanese
1758	cnumber = 0;
1759	qsort(colorTrgms, trgmNFA->colorTrgmsCount, sizeof(ColorTrgmInfo),
1760		  colorTrgmInfoCmp);
1761	for (i = 0; i < trgmNFA->colorTrgmsCount; i++)
1762	{
1763		if (colorTrgms[i].expanded)
1764		{
1765			colorTrgms[i].cnumber = cnumber;
1766			cnumber++;
1767		}
1768	}
1769
1770	return true;
1771}

1773/*
1774 * Expand selected color trigrams into regular trigrams.
1775 *
1776 * Returns the TRGM array to be passed to the index machinery.
1777 * The array must be allocated in rcontext.
1778 */ view japanese
1779static TRGM *
1780expandColorTrigrams(TrgmNFA *trgmNFA, MemoryContext rcontext)
1781{
1782	TRGM	   *trg;
1783	trgm	   *p;
1784	int			i;
1785	TrgmColorInfo blankColor;
1786	trgm_mb_char blankChar;
1787
1788	/* Set up "blank" color structure containing a single zero character */ view japanese
1789	memset(blankChar.bytes, 0, sizeof(blankChar.bytes));
1790	blankColor.wordCharsCount = 1;
1791	blankColor.wordChars = &blankChar;
1792
1793	/* Construct the trgm array */ view japanese
1794	trg = (TRGM *)
1795		MemoryContextAllocZero(rcontext,
1796							   TRGMHDRSIZE +
1797							   trgmNFA->totalTrgmCount * sizeof(trgm));
1798	trg->flag = ARRKEY;
1799	SET_VARSIZE(trg, CALCGTSIZE(ARRKEY, trgmNFA->totalTrgmCount));
1800	p = GETARR(trg);
1801	for (i = 0; i < trgmNFA->colorTrgmsCount; i++)
1802	{
1803		ColorTrgmInfo *colorTrgm = &trgmNFA->colorTrgms[i];
1804		TrgmColorInfo *c[3];
1805		trgm_mb_char s[3];
1806		int			j,
1807					i1,
1808					i2,
1809					i3;
1810
1811		/* Ignore any unexpanded trigrams ... */ view japanese
1812		if (!colorTrgm->expanded)
1813			continue;
1814
1815		/* Get colors, substituting the dummy struct for COLOR_BLANK */ view japanese
1816		for (j = 0; j < 3; j++)
1817		{
1818			if (colorTrgm->ctrgm.colors[j] != COLOR_BLANK)
1819				c[j] = &trgmNFA->colorInfo[colorTrgm->ctrgm.colors[j]];
1820			else
1821				c[j] = &blankColor;
1822		}
1823
1824		/* Iterate over all possible combinations of colors' characters */ view japanese
1825		for (i1 = 0; i1 < c[0]->wordCharsCount; i1++)
1826		{
1827			s[0] = c[0]->wordChars[i1];
1828			for (i2 = 0; i2 < c[1]->wordCharsCount; i2++)
1829			{
1830				s[1] = c[1]->wordChars[i2];
1831				for (i3 = 0; i3 < c[2]->wordCharsCount; i3++)
1832				{
1833					s[2] = c[2]->wordChars[i3];
1834					fillTrgm(p, s);
1835					p++;
1836				}
1837			}
1838		}
1839	}
1840
1841	return trg;
1842}
