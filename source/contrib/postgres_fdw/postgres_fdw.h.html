23/*
24 * FDW-specific planner information kept in RelOptInfo.fdw_private for a
25 * postgres_fdw foreign table.  For a baserel, this struct is created by
26 * postgresGetForeignRelSize, although some fields are not filled till later.
27 * postgresGetForeignJoinPaths creates it for a joinrel, and
28 * postgresGetForeignUpperPaths creates it for an upperrel.
29 */ view japanese
30typedef struct PgFdwRelationInfo
31{
32	/*
33	 * True means that the relation can be pushed down. Always true for simple
34	 * foreign scan.
35	 */ view japanese
36	bool		pushdown_safe;
37
38	/*
39	 * Restriction clauses, divided into safe and unsafe to pushdown subsets.
40	 * All entries in these lists should have RestrictInfo wrappers; that
41	 * improves efficiency of selectivity and cost estimation.
42	 */ view japanese
43	List	   *remote_conds;
44	List	   *local_conds;
45
46	/* Actual remote restriction clauses for scan (sans RestrictInfos) */ view japanese
47	List	   *final_remote_exprs;
48
49	/* Bitmap of attr numbers we need to fetch from the remote server. */ view japanese
50	Bitmapset  *attrs_used;
51
52	/* True means that the query_pathkeys is safe to push down */ view japanese
53	bool		qp_is_pushdown_safe;
54
55	/* Cost and selectivity of local_conds. */ view japanese
56	QualCost	local_conds_cost;
57	Selectivity local_conds_sel;
58
59	/* Selectivity of join conditions */ view japanese
60	Selectivity joinclause_sel;
61
62	/* Estimated size and cost for a scan, join, or grouping/aggregation. */ view japanese
63	double		rows;
64	int			width;
65	int			disabled_nodes;
66	Cost		startup_cost;
67	Cost		total_cost;
68
69	/*
70	 * Estimated number of rows fetched from the foreign server, and costs
71	 * excluding costs for transferring those rows from the foreign server.
72	 * These are only used by estimate_path_cost_size().
73	 */ view japanese
74	double		retrieved_rows;
75	Cost		rel_startup_cost;
76	Cost		rel_total_cost;
77
78	/* Options extracted from catalogs. */ view japanese
79	bool		use_remote_estimate;
80	Cost		fdw_startup_cost;
81	Cost		fdw_tuple_cost;
82	List	   *shippable_extensions;	/* OIDs of shippable extensions */ view japanese
83	bool		async_capable;
84
85	/* Cached catalog information. */ view japanese
86	ForeignTable *table;
87	ForeignServer *server;
88	UserMapping *user;			/* only set in use_remote_estimate mode */ view japanese
89
90	int			fetch_size;		/* fetch size for this remote table */ view japanese
91
92	/*
93	 * Name of the relation, for use while EXPLAINing ForeignScan.  It is used
94	 * for join and upper relations but is set for all relations.  For a base
95	 * relation, this is really just the RT index as a string; we convert that
96	 * while producing EXPLAIN output.  For join and upper relations, the name
97	 * indicates which base foreign tables are included and the join type or
98	 * aggregation type used.
99	 */ view japanese
100	char	   *relation_name;
101
102	/* Join information */ view japanese
103	RelOptInfo *outerrel;
104	RelOptInfo *innerrel;
105	JoinType	jointype;
106	/* joinclauses contains only JOIN/ON conditions for an outer join */ view japanese
107	List	   *joinclauses;	/* List of RestrictInfo */ view japanese
108
109	/* Upper relation information */ view japanese
110	UpperRelationKind stage;
111
112	/* Grouping information */ view japanese
113	List	   *grouped_tlist;
114
115	/* Subquery information */ view japanese
116	bool		make_outerrel_subquery; /* do we deparse outerrel as a
117										 * subquery? */ view japanese
118	bool		make_innerrel_subquery; /* do we deparse innerrel as a
119										 * subquery? */ view japanese
120	Relids		lower_subquery_rels;	/* all relids appearing in lower
121										 * subqueries */ view japanese
122	Relids		hidden_subquery_rels;	/* relids, which can't be referred to
123										 * from upper relations, used
124										 * internally for equivalence member
125										 * search */ view japanese
126
127	/*
128	 * Index of the relation.  It is used to create an alias to a subquery
129	 * representing the relation.
130	 */ view japanese
131	int			relation_index;
132} PgFdwRelationInfo;
