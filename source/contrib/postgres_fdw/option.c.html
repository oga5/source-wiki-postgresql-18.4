62/*
63 * Validate the generic options given to a FOREIGN DATA WRAPPER, SERVER,
64 * USER MAPPING or FOREIGN TABLE that uses postgres_fdw.
65 *
66 * Raise an ERROR if the option or its value is considered invalid.
67 */ view japanese
68PG_FUNCTION_INFO_V1(postgres_fdw_validator);
69
70Datum
71postgres_fdw_validator(PG_FUNCTION_ARGS)
72{
73	List	   *options_list = untransformRelOptions(PG_GETARG_DATUM(0));
74	Oid			catalog = PG_GETARG_OID(1);
75	ListCell   *cell;
76
77	/* Build our options lists if we didn't yet. */
78	InitPgFdwOptions();
79
80	/*
81	 * Check that only options supported by postgres_fdw, and allowed for the
82	 * current object type, are given.
83	 */ view japanese
84	foreach(cell, options_list)
85	{
86		DefElem    *def = (DefElem *) lfirst(cell);
87
88		if (!is_valid_option(def->defname, catalog))
89		{
90			/*
91			 * Unknown option specified, complain about it. Provide a hint
92			 * with a valid option that looks similar, if there is one.
93			 */ view japanese
94			PgFdwOption *opt;
95			const char *closest_match;
96			ClosestMatchState match_state;
97			bool		has_valid_options = false;
98
99			initClosestMatch(&match_state, def->defname, 4);
100			for (opt = postgres_fdw_options; opt->keyword; opt++)
101			{
102				if (catalog == opt->optcontext)
103				{
104					has_valid_options = true;
105					updateClosestMatch(&match_state, opt->keyword);
106				}
107			}
108
109			closest_match = getClosestMatch(&match_state);
110			ereport(ERROR,
111					(errcode(ERRCODE_FDW_INVALID_OPTION_NAME),
112					 errmsg("invalid option \"%s\"", def->defname),
113					 has_valid_options ? closest_match ?
114					 errhint("Perhaps you meant the option \"%s\".",
115							 closest_match) : 0 :
116					 errhint("There are no valid options in this context.")));
117		}
118
119		/*
120		 * Validate option value, when we can do so without any context.
121		 */ view japanese
122		if (strcmp(def->defname, "use_remote_estimate") == 0 ||
123			strcmp(def->defname, "updatable") == 0 ||
124			strcmp(def->defname, "truncatable") == 0 ||
125			strcmp(def->defname, "async_capable") == 0 ||
126			strcmp(def->defname, "parallel_commit") == 0 ||
127			strcmp(def->defname, "parallel_abort") == 0 ||
128			strcmp(def->defname, "keep_connections") == 0)
129		{
130			/* these accept only boolean values */ view japanese
131			(void) defGetBoolean(def);
132		}
133		else if (strcmp(def->defname, "fdw_startup_cost") == 0 ||
134				 strcmp(def->defname, "fdw_tuple_cost") == 0)
135		{
136			/*
137			 * These must have a floating point value greater than or equal to
138			 * zero.
139			 */ view japanese
140			char	   *value;
141			double		real_val;
142			bool		is_parsed;
143
144			value = defGetString(def);
145			is_parsed = parse_real(value, &real_val, 0, NULL);
146
147			if (!is_parsed)
148				ereport(ERROR,
149						(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
150						 errmsg("invalid value for floating point option \"%s\": %s",
151								def->defname, value)));
152
153			if (real_val < 0)
154				ereport(ERROR,
155						(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
156						 errmsg("\"%s\" must be a floating point value greater than or equal to zero",
157								def->defname)));
158		}
159		else if (strcmp(def->defname, "extensions") == 0)
160		{
161			/* check list syntax, warn about uninstalled extensions */ view japanese
162			(void) ExtractExtensionList(defGetString(def), true);
163		}
164		else if (strcmp(def->defname, "fetch_size") == 0 ||
165				 strcmp(def->defname, "batch_size") == 0)
166		{
167			char	   *value;
168			int			int_val;
169			bool		is_parsed;
170
171			value = defGetString(def);
172			is_parsed = parse_int(value, &int_val, 0, NULL);
173
174			if (!is_parsed)
175				ereport(ERROR,
176						(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
177						 errmsg("invalid value for integer option \"%s\": %s",
178								def->defname, value)));
179
180			if (int_val <= 0)
181				ereport(ERROR,
182						(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
183						 errmsg("\"%s\" must be an integer value greater than zero",
184								def->defname)));
185		}
186		else if (strcmp(def->defname, "password_required") == 0)
187		{
188			bool		pw_required = defGetBoolean(def);
189
190			/*
191			 * Only the superuser may set this option on a user mapping, or
192			 * alter a user mapping on which this option is set. We allow a
193			 * user to clear this option if it's set - in fact, we don't have
194			 * a choice since we can't see the old mapping when validating an
195			 * alter.
196			 */ view japanese
197			if (!superuser() && !pw_required)
198				ereport(ERROR,
199						(errcode(ERRCODE_INSUFFICIENT_PRIVILEGE),
200						 errmsg("password_required=false is superuser-only"),
201						 errhint("User mappings with the password_required option set to false may only be created or modified by the superuser.")));
202		}
203		else if (strcmp(def->defname, "sslcert") == 0 ||
204				 strcmp(def->defname, "sslkey") == 0)
205		{
206			/* similarly for sslcert / sslkey on user mapping */ view japanese
207			if (catalog == UserMappingRelationId && !superuser())
208				ereport(ERROR,
209						(errcode(ERRCODE_INSUFFICIENT_PRIVILEGE),
210						 errmsg("sslcert and sslkey are superuser-only"),
211						 errhint("User mappings with the sslcert or sslkey options set may only be created or modified by the superuser.")));
212		}
213		else if (strcmp(def->defname, "analyze_sampling") == 0)
214		{
215			char	   *value;
216
217			value = defGetString(def);
218
219			/* we recognize off/auto/random/system/bernoulli */ view japanese
220			if (strcmp(value, "off") != 0 &&
221				strcmp(value, "auto") != 0 &&
222				strcmp(value, "random") != 0 &&
223				strcmp(value, "system") != 0 &&
224				strcmp(value, "bernoulli") != 0)
225				ereport(ERROR,
226						(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
227						 errmsg("invalid value for string option \"%s\": %s",
228								def->defname, value)));
229		}
230	}
231
232	PG_RETURN_VOID();
233}

235/*
236 * Initialize option lists.
237 */ view japanese
238static void
239InitPgFdwOptions(void)
240{
241	int			num_libpq_opts;
242	PQconninfoOption *lopt;
243	PgFdwOption *popt;
244
245	/* non-libpq FDW-specific FDW options */ view japanese
246	static const PgFdwOption non_libpq_options[] = {
247		{"schema_name", ForeignTableRelationId, false},
248		{"table_name", ForeignTableRelationId, false},
249		{"column_name", AttributeRelationId, false},
250		/* use_remote_estimate is available on both server and table */ view japanese
251		{"use_remote_estimate", ForeignServerRelationId, false},
252		{"use_remote_estimate", ForeignTableRelationId, false},
253		/* cost factors */ view japanese
254		{"fdw_startup_cost", ForeignServerRelationId, false},
255		{"fdw_tuple_cost", ForeignServerRelationId, false},
256		/* shippable extensions */ view japanese
257		{"extensions", ForeignServerRelationId, false},
258		/* updatable is available on both server and table */ view japanese
259		{"updatable", ForeignServerRelationId, false},
260		{"updatable", ForeignTableRelationId, false},
261		/* truncatable is available on both server and table */ view japanese
262		{"truncatable", ForeignServerRelationId, false},
263		{"truncatable", ForeignTableRelationId, false},
264		/* fetch_size is available on both server and table */ view japanese
265		{"fetch_size", ForeignServerRelationId, false},
266		{"fetch_size", ForeignTableRelationId, false},
267		/* batch_size is available on both server and table */ view japanese
268		{"batch_size", ForeignServerRelationId, false},
269		{"batch_size", ForeignTableRelationId, false},
270		/* async_capable is available on both server and table */ view japanese
271		{"async_capable", ForeignServerRelationId, false},
272		{"async_capable", ForeignTableRelationId, false},
273		{"parallel_commit", ForeignServerRelationId, false},
274		{"parallel_abort", ForeignServerRelationId, false},
275		{"keep_connections", ForeignServerRelationId, false},
276		{"password_required", UserMappingRelationId, false},
277
278		/* sampling is available on both server and table */ view japanese
279		{"analyze_sampling", ForeignServerRelationId, false},
280		{"analyze_sampling", ForeignTableRelationId, false},
281
282		{"use_scram_passthrough", ForeignServerRelationId, false},
283		{"use_scram_passthrough", UserMappingRelationId, false},
284
285		/*
286		 * sslcert and sslkey are in fact libpq options, but we repeat them
287		 * here to allow them to appear in both foreign server context (when
288		 * we generate libpq options) and user mapping context (from here).
289		 */ view japanese
290		{"sslcert", UserMappingRelationId, true},
291		{"sslkey", UserMappingRelationId, true},
292
293		/*
294		 * gssdelegation is also a libpq option but should be allowed in a
295		 * user mapping context too
296		 */ view japanese
297		{"gssdelegation", UserMappingRelationId, true},
298
299		{NULL, InvalidOid, false}
300	};
301
302	/* Prevent redundant initialization. */ view japanese
303	if (postgres_fdw_options)
304		return;
305
306	/*
307	 * Get list of valid libpq options.
308	 *
309	 * To avoid unnecessary work, we get the list once and use it throughout
310	 * the lifetime of this backend process.  We don't need to care about
311	 * memory context issues, because PQconndefaults allocates with malloc.
312	 */ view japanese
313	libpq_options = PQconndefaults();
314	if (!libpq_options)			/* assume reason for failure is OOM */ view japanese
315		ereport(ERROR,
316				(errcode(ERRCODE_FDW_OUT_OF_MEMORY),
317				 errmsg("out of memory"),
318				 errdetail("Could not get libpq's default connection options.")));
319
320	/* Count how many libpq options are available. */ view japanese
321	num_libpq_opts = 0;
322	for (lopt = libpq_options; lopt->keyword; lopt++)
323		num_libpq_opts++;
324
325	/*
326	 * Construct an array which consists of all valid options for
327	 * postgres_fdw, by appending FDW-specific options to libpq options.
328	 *
329	 * We use plain malloc here to allocate postgres_fdw_options because it
330	 * lives as long as the backend process does.  Besides, keeping
331	 * libpq_options in memory allows us to avoid copying every keyword
332	 * string.
333	 */ view japanese
334	postgres_fdw_options = (PgFdwOption *)
335		malloc(sizeof(PgFdwOption) * num_libpq_opts +
336			   sizeof(non_libpq_options));
337	if (postgres_fdw_options == NULL)
338		ereport(ERROR,
339				(errcode(ERRCODE_FDW_OUT_OF_MEMORY),
340				 errmsg("out of memory")));
341
342	popt = postgres_fdw_options;
343	for (lopt = libpq_options; lopt->keyword; lopt++)
344	{
345		/* Hide debug options, as well as settings we override internally. */ view japanese
346		if (strchr(lopt->dispchar, 'D') ||
347			strcmp(lopt->keyword, "fallback_application_name") == 0 ||
348			strcmp(lopt->keyword, "client_encoding") == 0)
349			continue;
350
351		/*
352		 * Disallow OAuth options for now, since the builtin flow communicates
353		 * on stderr by default and can't cache tokens yet.
354		 */ view japanese
355		if (strncmp(lopt->keyword, "oauth_", strlen("oauth_")) == 0)
356			continue;
357
358		/* We don't have to copy keyword string, as described above. */ view japanese
359		popt->keyword = lopt->keyword;
360
361		/*
362		 * "user" and any secret options are allowed only on user mappings.
363		 * Everything else is a server option.
364		 */ view japanese
365		if (strcmp(lopt->keyword, "user") == 0 || strchr(lopt->dispchar, '*'))
366			popt->optcontext = UserMappingRelationId;
367		else
368			popt->optcontext = ForeignServerRelationId;
369		popt->is_libpq_opt = true;
370
371		popt++;
372	}
373
374	/* Append FDW-specific options and dummy terminator. */ view japanese
375	memcpy(popt, non_libpq_options, sizeof(non_libpq_options));
376}

378/*
379 * Check whether the given option is one of the valid postgres_fdw options.
380 * context is the Oid of the catalog holding the object the option is for.
381 */ view japanese
382static bool
383is_valid_option(const char *keyword, Oid context)
384{
385	PgFdwOption *opt;
386
387	Assert(postgres_fdw_options);	/* must be initialized already */
388
389	for (opt = postgres_fdw_options; opt->keyword; opt++)
390	{
391		if (context == opt->optcontext && strcmp(opt->keyword, keyword) == 0)
392			return true;
393	}
394
395	return false;
396}

398/*
399 * Check whether the given option is one of the valid libpq options.
400 */ view japanese
401static bool
402is_libpq_option(const char *keyword)
403{
404	PgFdwOption *opt;
405
406	Assert(postgres_fdw_options);	/* must be initialized already */ view japanese
407
408	for (opt = postgres_fdw_options; opt->keyword; opt++)
409	{
410		if (opt->is_libpq_opt && strcmp(opt->keyword, keyword) == 0)
411			return true;
412	}
413
414	return false;
415}

417/*
418 * Generate key-value arrays which include only libpq options from the
419 * given list (which can contain any kind of options).  Caller must have
420 * allocated large-enough arrays.  Returns number of options found.
421 */ view japanese
422int
423ExtractConnectionOptions(List *defelems, const char **keywords,
424						 const char **values)
425{
426	ListCell   *lc;
427	int			i;
428
429	/* Build our options lists if we didn't yet. */ view japanese
430	InitPgFdwOptions();
431
432	i = 0;
433	foreach(lc, defelems)
434	{
435		DefElem    *d = (DefElem *) lfirst(lc);
436
437		if (is_libpq_option(d->defname))
438		{
439			keywords[i] = d->defname;
440			values[i] = defGetString(d);
441			i++;
442		}
443	}
444	return i;
445}

447/*
448 * Parse a comma-separated string and return a List of the OIDs of the
449 * extensions named in the string.  If any names in the list cannot be
450 * found, report a warning if warnOnMissing is true, else just silently
451 * ignore them.
452 */ view japanese
453List *
454ExtractExtensionList(const char *extensionsString, bool warnOnMissing)
455{
456	List	   *extensionOids = NIL;
457	List	   *extlist;
458	ListCell   *lc;
459
460	/* SplitIdentifierString scribbles on its input, so pstrdup first */ view japanese
461	if (!SplitIdentifierString(pstrdup(extensionsString), ',', &extlist))
462	{
463		/* syntax error in name list */ view japanese
464		ereport(ERROR,
465				(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
466				 errmsg("parameter \"%s\" must be a list of extension names",
467						"extensions")));
468	}
469
470	foreach(lc, extlist)
471	{
472		const char *extension_name = (const char *) lfirst(lc);
473		Oid			extension_oid = get_extension_oid(extension_name, true);
474
475		if (OidIsValid(extension_oid))
476		{
477			extensionOids = lappend_oid(extensionOids, extension_oid);
478		}
479		else if (warnOnMissing)
480		{
481			ereport(WARNING,
482					(errcode(ERRCODE_UNDEFINED_OBJECT),
483					 errmsg("extension \"%s\" is not installed",
484							extension_name)));
485		}
486	}
487
488	list_free(extlist);
489	return extensionOids;
490}

492/*
493 * Replace escape sequences beginning with % character in the given
494 * application_name with status information, and return it.
495 *
496 * This function always returns a palloc'd string, so the caller is
497 * responsible for pfreeing it.
498 */ view japanese
499char *
500process_pgfdw_appname(const char *appname)
501{
502	const char *p;
503	StringInfoData buf;
504
505	initStringInfo(&buf);
506
507	for (p = appname; *p != '\0'; p++)
508	{
509		if (*p != '%')
510		{
511			/* literal char, just copy */ view japanese
512			appendStringInfoChar(&buf, *p);
513			continue;
514		}
515
516		/* must be a '%', so skip to the next char */ view japanese
517		p++;
518		if (*p == '\0')
519			break;				/* format error - ignore it */
520		else if (*p == '%')
521		{
522			/* string contains %% */ view japanese
523			appendStringInfoChar(&buf, '%');
524			continue;
525		}
526
527		/* process the option */ view japanese
528		switch (*p)
529		{
530			case 'a':
531				appendStringInfoString(&buf, application_name);
532				break;
533			case 'c':
534				appendStringInfo(&buf, "%" PRIx64 ".%x", MyStartTime, MyProcPid);
535				break;
536			case 'C':
537				appendStringInfoString(&buf, cluster_name);
538				break;
539			case 'd':
540				if (MyProcPort)
541				{
542					const char *dbname = MyProcPort->database_name;
543
544					if (dbname)
545						appendStringInfoString(&buf, dbname);
546					else
547						appendStringInfoString(&buf, "[unknown]");
548				}
549				break;
550			case 'p':
551				appendStringInfo(&buf, "%d", MyProcPid);
552				break;
553			case 'u':
554				if (MyProcPort)
555				{
556					const char *username = MyProcPort->user_name;
557
558					if (username)
559						appendStringInfoString(&buf, username);
560					else
561						appendStringInfoString(&buf, "[unknown]");
562				}
563				break;
564			default:
565				/* format error - ignore it */ view japanese
566				break;
567		}
568	}
569
570	return buf.data;
571}
