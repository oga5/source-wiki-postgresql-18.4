36/*
37 * Connection cache hash table entry
38 *
39 * The lookup key in this hash table is the user mapping OID. We use just one
40 * connection per user mapping ID, which ensures that all the scans use the
41 * same snapshot during a query.  Using the user mapping OID rather than
42 * the foreign server OID + user OID avoids creating multiple connections when
43 * the public user mapping applies to all user OIDs.
44 *
45 * The "conn" pointer can be NULL if we don't currently have a live connection.
46 * When we do have a connection, xact_depth tracks the current depth of
47 * transactions and subtransactions open on the remote side.  We need to issue
48 * commands at the same nesting depth on the remote as we're executing at
49 * ourselves, so that rolling back a subtransaction will kill the right
50 * queries and not the wrong ones.
51 */
52typedef Oid ConnCacheKey;
53
54typedef struct ConnCacheEntry
55{
56	ConnCacheKey key;			/* hash key (must be first) */
57	PGconn	   *conn;			/* connection to foreign server, or NULL */
58	/* Remaining fields are invalid when conn is NULL: */
59	int			xact_depth;		/* 0 = no xact open, 1 = main xact open, 2 =
60								 * one level of subxact open, etc */
61	bool		have_prep_stmt; /* have we prepared any stmts in this xact? */
62	bool		have_error;		/* have any subxacts aborted in this xact? */
63	bool		changing_xact_state;	/* xact state change in process */
64	bool		parallel_commit;	/* do we commit (sub)xacts in parallel? */
65	bool		parallel_abort; /* do we abort (sub)xacts in parallel? */
66	bool		invalidated;	/* true if reconnect is pending */
67	bool		keep_connections;	/* setting value of keep_connections
68									 * server option */
69	Oid			serverid;		/* foreign server OID used to get server name */
70	uint32		server_hashvalue;	/* hash value of foreign server OID */
71	uint32		mapping_hashvalue;	/* hash value of user mapping OID */
72	PgFdwConnState state;		/* extra per-connection state */
73} ConnCacheEntry;

189/*
190 * Get a PGconn which can be used to execute queries on the remote PostgreSQL
191 * server with the user's authorization.  A new connection is established
192 * if we don't already have a suitable one, and a transaction is opened at
193 * the right subtransaction nesting depth if we didn't do that already.
194 *
195 * will_prep_stmt must be true if caller intends to create any prepared
196 * statements.  Since those don't go away automatically at transaction end
197 * (not even on error), we need this flag to cue manual cleanup.
198 *
199 * If state is not NULL, *state receives the per-connection state associated
200 * with the PGconn.
201 */
202PGconn *
203GetConnection(UserMapping *user, bool will_prep_stmt, PgFdwConnState **state)
204{
205	bool		found;
206	bool		retry = false;
207	ConnCacheEntry *entry;
208	ConnCacheKey key;
209	MemoryContext ccxt = CurrentMemoryContext;
210
211	/* First time through, initialize connection cache hashtable */
212	if (ConnectionHash == NULL)
213	{
214		HASHCTL		ctl;
215
216		if (pgfdw_we_get_result == 0)
217			pgfdw_we_get_result =
218				WaitEventExtensionNew("PostgresFdwGetResult");
219
220		ctl.keysize = sizeof(ConnCacheKey);
221		ctl.entrysize = sizeof(ConnCacheEntry);
222		ConnectionHash = hash_create("postgres_fdw connections", 8,
223									 &ctl,
224									 HASH_ELEM | HASH_BLOBS);
225
226		/*
227		 * Register some callback functions that manage connection cleanup.
228		 * This should be done just once in each backend.
229		 */
230		RegisterXactCallback(pgfdw_xact_callback, NULL);
231		RegisterSubXactCallback(pgfdw_subxact_callback, NULL);
232		CacheRegisterSyscacheCallback(FOREIGNSERVEROID,
233									  pgfdw_inval_callback, (Datum) 0);
234		CacheRegisterSyscacheCallback(USERMAPPINGOID,
235									  pgfdw_inval_callback, (Datum) 0);
236	}
237
238	/* Set flag that we did GetConnection during the current transaction */
239	xact_got_connection = true;
240
241	/* Create hash key for the entry.  Assume no pad bytes in key struct */
242	key = user->umid;
243
244	/*
245	 * Find or create cached entry for requested connection.
246	 */
247	entry = hash_search(ConnectionHash, &key, HASH_ENTER, &found);
248	if (!found)
249	{
250		/*
251		 * We need only clear "conn" here; remaining fields will be filled
252		 * later when "conn" is set.
253		 */
254		entry->conn = NULL;
255	}
256
257	/* Reject further use of connections which failed abort cleanup. */
258	pgfdw_reject_incomplete_xact_state_change(entry);
259
260	/*
261	 * If the connection needs to be remade due to invalidation, disconnect as
262	 * soon as we're out of all transactions.
263	 */
264	if (entry->conn != NULL && entry->invalidated && entry->xact_depth == 0)
265	{
266		elog(DEBUG3, "closing connection %p for option changes to take effect",
267			 entry->conn);
268		disconnect_pg_server(entry);
269	}
270
271	/*
272	 * If cache entry doesn't have a connection, we have to establish a new
273	 * connection.  (If connect_pg_server throws an error, the cache entry
274	 * will remain in a valid empty state, ie conn == NULL.)
275	 */
276	if (entry->conn == NULL)
277		make_new_connection(entry, user);
278
279	/*
280	 * We check the health of the cached connection here when using it.  In
281	 * cases where we're out of all transactions, if a broken connection is
282	 * detected, we try to reestablish a new connection later.
283	 */
284	PG_TRY();
285	{
286		/* Process a pending asynchronous request if any. */
287		if (entry->state.pendingAreq)
288			process_pending_request(entry->state.pendingAreq);
289		/* Start a new transaction or subtransaction if needed. */
290		begin_remote_xact(entry);
291	}
292	PG_CATCH();
293	{
294		MemoryContext ecxt = MemoryContextSwitchTo(ccxt);
295		ErrorData  *errdata = CopyErrorData();
296
297		/*
298		 * Determine whether to try to reestablish the connection.
299		 *
300		 * After a broken connection is detected in libpq, any error other
301		 * than connection failure (e.g., out-of-memory) can be thrown
302		 * somewhere between return from libpq and the expected ereport() call
303		 * in pgfdw_report_error(). In this case, since PQstatus() indicates
304		 * CONNECTION_BAD, checking only PQstatus() causes the false detection
305		 * of connection failure. To avoid this, we also verify that the
306		 * error's sqlstate is ERRCODE_CONNECTION_FAILURE. Note that also
307		 * checking only the sqlstate can cause another false detection
308		 * because pgfdw_report_error() may report ERRCODE_CONNECTION_FAILURE
309		 * for any libpq-originated error condition.
310		 */
311		if (errdata->sqlerrcode != ERRCODE_CONNECTION_FAILURE ||
312			PQstatus(entry->conn) != CONNECTION_BAD ||
313			entry->xact_depth > 0)
314		{
315			MemoryContextSwitchTo(ecxt);
316			PG_RE_THROW();
317		}
318
319		/* Clean up the error state */
320		FlushErrorState();
321		FreeErrorData(errdata);
322		errdata = NULL;
323
324		retry = true;
325	}
326	PG_END_TRY();
327
328	/*
329	 * If a broken connection is detected, disconnect it, reestablish a new
330	 * connection and retry a new remote transaction. If connection failure is
331	 * reported again, we give up getting a connection.
332	 */
333	if (retry)
334	{
335		Assert(entry->xact_depth == 0);
336
337		ereport(DEBUG3,
338				(errmsg_internal("could not start remote transaction on connection %p",
339								 entry->conn)),
340				errdetail_internal("%s", pchomp(PQerrorMessage(entry->conn))));
341
342		elog(DEBUG3, "closing connection %p to reestablish a new one",
343			 entry->conn);
344		disconnect_pg_server(entry);
345
346		make_new_connection(entry, user);
347
348		begin_remote_xact(entry);
349	}
350
351	/* Remember if caller will prepare statements */
352	entry->have_prep_stmt |= will_prep_stmt;
353
354	/* If caller needs access to the per-connection state, return it. */
355	if (state)
356		*state = &entry->state;
357
358	return entry->conn;
359}

361/*
362 * Reset all transient state fields in the cached connection entry and
363 * establish new connection to the remote server.
364 */
365static void
366make_new_connection(ConnCacheEntry *entry, UserMapping *user)
367{
368	ForeignServer *server = GetForeignServer(user->serverid);
369	ListCell   *lc;
370
371	Assert(entry->conn == NULL);
372
373	/* Reset all transient state fields, to be sure all are clean */
374	entry->xact_depth = 0;
375	entry->have_prep_stmt = false;
376	entry->have_error = false;
377	entry->changing_xact_state = false;
378	entry->invalidated = false;
379	entry->serverid = server->serverid;
380	entry->server_hashvalue =
381		GetSysCacheHashValue1(FOREIGNSERVEROID,
382							  ObjectIdGetDatum(server->serverid));
383	entry->mapping_hashvalue =
384		GetSysCacheHashValue1(USERMAPPINGOID,
385							  ObjectIdGetDatum(user->umid));
386	memset(&entry->state, 0, sizeof(entry->state));
387
388	/*
389	 * Determine whether to keep the connection that we're about to make here
390	 * open even after the transaction using it ends, so that the subsequent
391	 * transactions can re-use it.
392	 *
393	 * By default, all the connections to any foreign servers are kept open.
394	 *
395	 * Also determine whether to commit/abort (sub)transactions opened on the
396	 * remote server in parallel at (sub)transaction end, which is disabled by
397	 * default.
398	 *
399	 * Note: it's enough to determine these only when making a new connection
400	 * because if these settings for it are changed, it will be closed and
401	 * re-made later.
402	 */
403	entry->keep_connections = true;
404	entry->parallel_commit = false;
405	entry->parallel_abort = false;
406	foreach(lc, server->options)
407	{
408		DefElem    *def = (DefElem *) lfirst(lc);
409
410		if (strcmp(def->defname, "keep_connections") == 0)
411			entry->keep_connections = defGetBoolean(def);
412		else if (strcmp(def->defname, "parallel_commit") == 0)
413			entry->parallel_commit = defGetBoolean(def);
414		else if (strcmp(def->defname, "parallel_abort") == 0)
415			entry->parallel_abort = defGetBoolean(def);
416	}
417
418	/* Now try to make the connection */
419	entry->conn = connect_pg_server(server, user);
420
421	elog(DEBUG3, "new postgres_fdw connection %p for server \"%s\" (user mapping oid %u, userid %u)",
422		 entry->conn, server->servername, user->umid, user->userid);
423}

425/*
426 * Check that non-superuser has used password or delegated credentials
427 * to establish connection; otherwise, he's piggybacking on the
428 * postgres server's user identity. See also dblink_security_check()
429 * in contrib/dblink and check_conn_params.
430 */
431static void
432pgfdw_security_check(const char **keywords, const char **values, UserMapping *user, PGconn *conn)
433{
434	/* Superusers bypass the check */
435	if (superuser_arg(user->userid))
436		return;
437
438#ifdef ENABLE_GSS
439	/* Connected via GSSAPI with delegated credentials- all good. */
440	if (PQconnectionUsedGSSAPI(conn) && be_gssapi_get_delegation(MyProcPort))
441		return;
442#endif
443
444	/* Ok if superuser set PW required false. */
445	if (!UserMappingPasswordRequired(user))
446		return;
447
448	/* Connected via PW, with PW required true, and provided non-empty PW. */
449	if (PQconnectionUsedPassword(conn))
450	{
451		/* ok if params contain a non-empty password */
452		for (int i = 0; keywords[i] != NULL; i++)
453		{
454			if (strcmp(keywords[i], "password") == 0 && values[i][0] != '\0')
455				return;
456		}
457	}
458
459	/*
460	 * Ok if SCRAM pass-through is being used and all required SCRAM options
461	 * are set correctly. If pgfdw_has_required_scram_options returns true we
462	 * assume that UseScramPassthrough is also true since SCRAM options are
463	 * only set when UseScramPassthrough is enabled.
464	 */
465	if (MyProcPort != NULL && MyProcPort->has_scram_keys && pgfdw_has_required_scram_options(keywords, values))
466		return;
467
468	ereport(ERROR,
469			(errcode(ERRCODE_S_R_E_PROHIBITED_SQL_STATEMENT_ATTEMPTED),
470			 errmsg("password or GSSAPI delegated credentials required"),
471			 errdetail("Non-superuser cannot connect if the server does not request a password or use GSSAPI with delegated credentials."),
472			 errhint("Target server's authentication method must be changed or password_required=false set in the user mapping attributes.")));
473}

475/*
476 * Connect to remote server using specified server and user mapping properties.
477 */
478static PGconn *
479connect_pg_server(ForeignServer *server, UserMapping *user)
480{
481	PGconn	   *volatile conn = NULL;
482
483	/*
484	 * Use PG_TRY block to ensure closing connection on error.
485	 */
486	PG_TRY();
487	{
488		const char **keywords;
489		const char **values;
490		char	   *appname = NULL;
491		int			n;
492
493		/*
494		 * Construct connection params from generic options of ForeignServer
495		 * and UserMapping.  (Some of them might not be libpq options, in
496		 * which case we'll just waste a few array slots.)  Add 4 extra slots
497		 * for application_name, fallback_application_name, client_encoding,
498		 * end marker, and 3 extra slots for scram keys and required scram
499		 * pass-through options.
500		 */
501		n = list_length(server->options) + list_length(user->options) + 4 + 3;
502		keywords = (const char **) palloc(n * sizeof(char *));
503		values = (const char **) palloc(n * sizeof(char *));
504
505		n = 0;
506		n += ExtractConnectionOptions(server->options,
507									  keywords + n, values + n);
508		n += ExtractConnectionOptions(user->options,
509									  keywords + n, values + n);
510
511		/*
512		 * Use pgfdw_application_name as application_name if set.
513		 *
514		 * PQconnectdbParams() processes the parameter arrays from start to
515		 * end. If any key word is repeated, the last value is used. Therefore
516		 * note that pgfdw_application_name must be added to the arrays after
517		 * options of ForeignServer are, so that it can override
518		 * application_name set in ForeignServer.
519		 */
520		if (pgfdw_application_name && *pgfdw_application_name != '\0')
521		{
522			keywords[n] = "application_name";
523			values[n] = pgfdw_application_name;
524			n++;
525		}
526
527		/*
528		 * Search the parameter arrays to find application_name setting, and
529		 * replace escape sequences in it with status information if found.
530		 * The arrays are searched backwards because the last value is used if
531		 * application_name is repeatedly set.
532		 */
533		for (int i = n - 1; i >= 0; i--)
534		{
535			if (strcmp(keywords[i], "application_name") == 0 &&
536				*(values[i]) != '\0')
537			{
538				/*
539				 * Use this application_name setting if it's not empty string
540				 * even after any escape sequences in it are replaced.
541				 */
542				appname = process_pgfdw_appname(values[i]);
543				if (appname[0] != '\0')
544				{
545					values[i] = appname;
546					break;
547				}
548
549				/*
550				 * This empty application_name is not used, so we set
551				 * values[i] to NULL and keep searching the array to find the
552				 * next one.
553				 */
554				values[i] = NULL;
555				pfree(appname);
556				appname = NULL;
557			}
558		}
559
560		/* Use "postgres_fdw" as fallback_application_name */
561		keywords[n] = "fallback_application_name";
562		values[n] = "postgres_fdw";
563		n++;
564
565		/* Set client_encoding so that libpq can convert encoding properly. */
566		keywords[n] = "client_encoding";
567		values[n] = GetDatabaseEncodingName();
568		n++;
569
570		/* Add required SCRAM pass-through connection options if it's enabled. */
571		if (MyProcPort != NULL && MyProcPort->has_scram_keys && UseScramPassthrough(server, user))
572		{
573			int			len;
574			int			encoded_len;
575
576			keywords[n] = "scram_client_key";
577			len = pg_b64_enc_len(sizeof(MyProcPort->scram_ClientKey));
578			/* don't forget the zero-terminator */
579			values[n] = palloc0(len + 1);
580			encoded_len = pg_b64_encode(MyProcPort->scram_ClientKey,
581										sizeof(MyProcPort->scram_ClientKey),
582										(char *) values[n], len);
583			if (encoded_len < 0)
584				elog(ERROR, "could not encode SCRAM client key");
585			n++;
586
587			keywords[n] = "scram_server_key";
588			len = pg_b64_enc_len(sizeof(MyProcPort->scram_ServerKey));
589			/* don't forget the zero-terminator */
590			values[n] = palloc0(len + 1);
591			encoded_len = pg_b64_encode(MyProcPort->scram_ServerKey,
592										sizeof(MyProcPort->scram_ServerKey),
593										(char *) values[n], len);
594			if (encoded_len < 0)
595				elog(ERROR, "could not encode SCRAM server key");
596			n++;
597
598			/*
599			 * Require scram-sha-256 to ensure that no other auth method is
600			 * used when connecting with foreign server.
601			 */
602			keywords[n] = "require_auth";
603			values[n] = "scram-sha-256";
604			n++;
605		}
606
607		keywords[n] = values[n] = NULL;
608
609		/* Verify the set of connection parameters. */
610		check_conn_params(keywords, values, user);
611
612		/* first time, allocate or get the custom wait event */
613		if (pgfdw_we_connect == 0)
614			pgfdw_we_connect = WaitEventExtensionNew("PostgresFdwConnect");
615
616		/* OK to make connection */
617		conn = libpqsrv_connect_params(keywords, values,
618									   false,	/* expand_dbname */
619									   pgfdw_we_connect);
620
621		if (!conn || PQstatus(conn) != CONNECTION_OK)
622			ereport(ERROR,
623					(errcode(ERRCODE_SQLCLIENT_UNABLE_TO_ESTABLISH_SQLCONNECTION),
624					 errmsg("could not connect to server \"%s\"",
625							server->servername),
626					 errdetail_internal("%s", pchomp(PQerrorMessage(conn)))));
627
628		/* Perform post-connection security checks. */
629		pgfdw_security_check(keywords, values, user, conn);
630
631		/* Prepare new session for use */
632		configure_remote_session(conn);
633
634		if (appname != NULL)
635			pfree(appname);
636		pfree(keywords);
637		pfree(values);
638	}
639	PG_CATCH();
640	{
641		libpqsrv_disconnect(conn);
642		PG_RE_THROW();
643	}
644	PG_END_TRY();
645
646	return conn;
647}

707/*
708 * For non-superusers, insist that the connstr specify a password or that the
709 * user provided their own GSSAPI delegated credentials.  This
710 * prevents a password from being picked up from .pgpass, a service file, the
711 * environment, etc.  We don't want the postgres user's passwords,
712 * certificates, etc to be accessible to non-superusers.  (See also
713 * dblink_connstr_check in contrib/dblink.)
714 */
715static void
716check_conn_params(const char **keywords, const char **values, UserMapping *user)
717{
718	int			i;
719
720	/* no check required if superuser */
721	if (superuser_arg(user->userid))
722		return;
723
724#ifdef ENABLE_GSS
725	/* ok if the user provided their own delegated credentials */
726	if (be_gssapi_get_delegation(MyProcPort))
727		return;
728#endif
729
730	/* ok if params contain a non-empty password */
731	for (i = 0; keywords[i] != NULL; i++)
732	{
733		if (strcmp(keywords[i], "password") == 0 && values[i][0] != '\0')
734			return;
735	}
736
737	/* ok if the superuser explicitly said so at user mapping creation time */
738	if (!UserMappingPasswordRequired(user))
739		return;
740
741	/*
742	 * Ok if SCRAM pass-through is being used and all required scram options
743	 * are set correctly. If pgfdw_has_required_scram_options returns true we
744	 * assume that UseScramPassthrough is also true since SCRAM options are
745	 * only set when UseScramPassthrough is enabled.
746	 */
747	if (MyProcPort != NULL && MyProcPort->has_scram_keys && pgfdw_has_required_scram_options(keywords, values))
748		return;
749
750	ereport(ERROR,
751			(errcode(ERRCODE_S_R_E_PROHIBITED_SQL_STATEMENT_ATTEMPTED),
752			 errmsg("password or GSSAPI delegated credentials required"),
753			 errdetail("Non-superusers must delegate GSSAPI credentials, provide a password, or enable SCRAM pass-through in user mapping.")));
754}

756/*
757 * Issue SET commands to make sure remote session is configured properly.
758 *
759 * We do this just once at connection, assuming nothing will change the
760 * values later.  Since we'll never send volatile function calls to the
761 * remote, there shouldn't be any way to break this assumption from our end.
762 * It's possible to think of ways to break it at the remote end, eg making
763 * a foreign table point to a view that includes a set_config call ---
764 * but once you admit the possibility of a malicious view definition,
765 * there are any number of ways to break things.
766 */
767static void
768configure_remote_session(PGconn *conn)
769{
770	int			remoteversion = PQserverVersion(conn);
771
772	/* Force the search path to contain only pg_catalog (see deparse.c) */
773	do_sql_command(conn, "SET search_path = pg_catalog");
774
775	/*
776	 * Set remote timezone; this is basically just cosmetic, since all
777	 * transmitted and returned timestamptzs should specify a zone explicitly
778	 * anyway.  However it makes the regression test outputs more predictable.
779	 *
780	 * We don't risk setting remote zone equal to ours, since the remote
781	 * server might use a different timezone database.  Instead, use GMT
782	 * (quoted, because very old servers are picky about case).  That's
783	 * guaranteed to work regardless of the remote's timezone database,
784	 * because pg_tzset() hard-wires it (at least in PG 9.2 and later).
785	 */
786	do_sql_command(conn, "SET timezone = 'GMT'");
787
788	/*
789	 * Set values needed to ensure unambiguous data output from remote.  (This
790	 * logic should match what pg_dump does.  See also set_transmission_modes
791	 * in postgres_fdw.c.)
792	 */
793	do_sql_command(conn, "SET datestyle = ISO");
794	if (remoteversion >= 80400)
795		do_sql_command(conn, "SET intervalstyle = postgres");
796	if (remoteversion >= 90000)
797		do_sql_command(conn, "SET extra_float_digits = 3");
798	else
799		do_sql_command(conn, "SET extra_float_digits = 2");
800}

804 */

812static void
813do_sql_command_begin(PGconn *conn, const char *sql)
814{
815	if (!PQsendQuery(conn, sql))
816		pgfdw_report_error(ERROR, NULL, conn, false, sql);

819static void
820do_sql_command_end(PGconn *conn, const char *sql, bool consume_input)
821{
822	PGresult   *res;
823
824	/*
825	 * If requested, consume whatever data is available from the socket. (Note
826	 * that if all data is available, this allows pgfdw_get_result to call
827	 * PQgetResult without forcing the overhead of WaitLatchOrSocket, which
828	 * would be large compared to the overhead of PQconsumeInput.)
829	 */
830	if (consume_input && !PQconsumeInput(conn))
831		pgfdw_report_error(ERROR, NULL, conn, false, sql);
832	res = pgfdw_get_result(conn);
833	if (PQresultStatus(res) != PGRES_COMMAND_OK)
834		pgfdw_report_error(ERROR, res, conn, true, sql);
835	PQclear(res);
836}

838/*
839 * Start remote transaction or subtransaction, if needed.
840 *
841 * Note that we always use at least REPEATABLE READ in the remote session.
842 * This is so that, if a query initiates multiple scans of the same or
843 * different foreign tables, we will get snapshot-consistent results from
844 * those scans.  A disadvantage is that we can't provide sane emulation of
845 * READ COMMITTED behavior --- it would be nice if we had some other way to
846 * control which remote queries share a snapshot.
847 */
848static void
849begin_remote_xact(ConnCacheEntry *entry)
850{
851	int			curlevel = GetCurrentTransactionNestLevel();
852
853	/* Start main transaction if we haven't yet */
854	if (entry->xact_depth <= 0)
855	{
856		const char *sql;
857
858		elog(DEBUG3, "starting remote transaction on connection %p",
859			 entry->conn);
860
861		if (IsolationIsSerializable())
862			sql = "START TRANSACTION ISOLATION LEVEL SERIALIZABLE";
863		else
864			sql = "START TRANSACTION ISOLATION LEVEL REPEATABLE READ";
865		entry->changing_xact_state = true;
866		do_sql_command(entry->conn, sql);
867		entry->xact_depth = 1;
868		entry->changing_xact_state = false;
869	}
870
871	/*
872	 * If we're in a subtransaction, stack up savepoints to match our level.
873	 * This ensures we can rollback just the desired effects when a
874	 * subtransaction aborts.
875	 */
876	while (entry->xact_depth < curlevel)
877	{
878		char		sql[64];
879
880		snprintf(sql, sizeof(sql), "SAVEPOINT s%d", entry->xact_depth + 1);
881		entry->changing_xact_state = true;
882		do_sql_command(entry->conn, sql);
883		entry->xact_depth++;
884		entry->changing_xact_state = false;
885	}
886}

964/*
965 * Report an error we got from the remote server.
966 *
967 * elevel: error level to use (typically ERROR, but might be less)
968 * res: PGresult containing the error
969 * conn: connection we did the query on
970 * clear: if true, PQclear the result (otherwise caller will handle it)
971 * sql: NULL, or text of remote command we tried to execute
972 *
973 * Note: callers that choose not to throw ERROR for a remote error are
974 * responsible for making sure that the associated ConnCacheEntry gets
975 * marked with have_error = true.
976 */
977void
978pgfdw_report_error(int elevel, PGresult *res, PGconn *conn,
979				   bool clear, const char *sql)
980{
981	/* If requested, PGresult must be released before leaving this function. */
982	PG_TRY();
983	{
984		char	   *diag_sqlstate = PQresultErrorField(res, PG_DIAG_SQLSTATE);
985		char	   *message_primary = PQresultErrorField(res, PG_DIAG_MESSAGE_PRIMARY);
986		char	   *message_detail = PQresultErrorField(res, PG_DIAG_MESSAGE_DETAIL);
987		char	   *message_hint = PQresultErrorField(res, PG_DIAG_MESSAGE_HINT);
988		char	   *message_context = PQresultErrorField(res, PG_DIAG_CONTEXT);
989		int			sqlstate;
990
991		if (diag_sqlstate)
992			sqlstate = MAKE_SQLSTATE(diag_sqlstate[0],
993									 diag_sqlstate[1],
994									 diag_sqlstate[2],
995									 diag_sqlstate[3],
996									 diag_sqlstate[4]);
997		else
998			sqlstate = ERRCODE_CONNECTION_FAILURE;
999
1000		/*
1001		 * If we don't get a message from the PGresult, try the PGconn.  This
1002		 * is needed because for connection-level failures, PQgetResult may
1003		 * just return NULL, not a PGresult at all.
1004		 */
1005		if (message_primary == NULL)
1006			message_primary = pchomp(PQerrorMessage(conn));
1007
1008		ereport(elevel,
1009				(errcode(sqlstate),
1010				 (message_primary != NULL && message_primary[0] != '\0') ?
1011				 errmsg_internal("%s", message_primary) :
1012				 errmsg("could not obtain message string for remote error"),
1013				 message_detail ? errdetail_internal("%s", message_detail) : 0,
1014				 message_hint ? errhint("%s", message_hint) : 0,
1015				 message_context ? errcontext("%s", message_context) : 0,
1016				 sql ? errcontext("remote SQL command: %s", sql) : 0));
1017	}
1018	PG_FINALLY();
1019	{
1020		if (clear)
1021			PQclear(res);
1022	}
1023	PG_END_TRY();
1024}

1026/*
1027 * pgfdw_xact_callback --- cleanup at main-transaction end.
1028 *
1029 * This runs just late enough that it must not enter user-defined code
1030 * locally.  (Entering such code on the remote side is fine.  Its remote
1031 * COMMIT TRANSACTION may run deferred triggers.)
1032 */
1033static void
1034pgfdw_xact_callback(XactEvent event, void *arg)
1035{
1036	HASH_SEQ_STATUS scan;
1037	ConnCacheEntry *entry;
1038	List	   *pending_entries = NIL;
1039	List	   *cancel_requested = NIL;
1040
1041	/* Quick exit if no connections were touched in this transaction. */
1042	if (!xact_got_connection)
1043		return;
1044
1045	/*
1046	 * Scan all connection cache entries to find open remote transactions, and
1047	 * close them.
1048	 */
1049	hash_seq_init(&scan, ConnectionHash);
1050	while ((entry = (ConnCacheEntry *) hash_seq_search(&scan)))
1051	{
1052		PGresult   *res;
1053
1054		/* Ignore cache entry if no open connection right now */
1055		if (entry->conn == NULL)
1056			continue;
1057
1058		/* If it has an open remote transaction, try to close it */
1059		if (entry->xact_depth > 0)
1060		{
1061			elog(DEBUG3, "closing remote transaction on connection %p",
1062				 entry->conn);
1063
1064			switch (event)
1065			{
1066				case XACT_EVENT_PARALLEL_PRE_COMMIT:
1067				case XACT_EVENT_PRE_COMMIT:
1068
1069					/*
1070					 * If abort cleanup previously failed for this connection,
1071					 * we can't issue any more commands against it.
1072					 */
1073					pgfdw_reject_incomplete_xact_state_change(entry);
1074
1075					/* Commit all remote transactions during pre-commit */
1076					entry->changing_xact_state = true;
1077					if (entry->parallel_commit)
1078					{
1079						do_sql_command_begin(entry->conn, "COMMIT TRANSACTION");
1080						pending_entries = lappend(pending_entries, entry);
1081						continue;
1082					}
1083					do_sql_command(entry->conn, "COMMIT TRANSACTION");
1084					entry->changing_xact_state = false;
1085
1086					/*
1087					 * If there were any errors in subtransactions, and we
1088					 * made prepared statements, do a DEALLOCATE ALL to make
1089					 * sure we get rid of all prepared statements. This is
1090					 * annoying and not terribly bulletproof, but it's
1091					 * probably not worth trying harder.
1092					 *
1093					 * DEALLOCATE ALL only exists in 8.3 and later, so this
1094					 * constrains how old a server postgres_fdw can
1095					 * communicate with.  We intentionally ignore errors in
1096					 * the DEALLOCATE, so that we can hobble along to some
1097					 * extent with older servers (leaking prepared statements
1098					 * as we go; but we don't really support update operations
1099					 * pre-8.3 anyway).
1100					 */
1101					if (entry->have_prep_stmt && entry->have_error)
1102					{
1103						res = pgfdw_exec_query(entry->conn, "DEALLOCATE ALL",
1104											   NULL);
1105						PQclear(res);
1106					}
1107					entry->have_prep_stmt = false;
1108					entry->have_error = false;
1109					break;
1110				case XACT_EVENT_PRE_PREPARE:
1111
1112					/*
1113					 * We disallow any remote transactions, since it's not
1114					 * very reasonable to hold them open until the prepared
1115					 * transaction is committed.  For the moment, throw error
1116					 * unconditionally; later we might allow read-only cases.
1117					 * Note that the error will cause us to come right back
1118					 * here with event == XACT_EVENT_ABORT, so we'll clean up
1119					 * the connection state at that point.
1120					 */
1121					ereport(ERROR,
1122							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
1123							 errmsg("cannot PREPARE a transaction that has operated on postgres_fdw foreign tables")));
1124					break;
1125				case XACT_EVENT_PARALLEL_COMMIT:
1126				case XACT_EVENT_COMMIT:
1127				case XACT_EVENT_PREPARE:
1128					/* Pre-commit should have closed the open transaction */
1129					elog(ERROR, "missed cleaning up connection during pre-commit");
1130					break;
1131				case XACT_EVENT_PARALLEL_ABORT:
1132				case XACT_EVENT_ABORT:
1133					/* Rollback all remote transactions during abort */
1134					if (entry->parallel_abort)
1135					{
1136						if (pgfdw_abort_cleanup_begin(entry, true,
1137													  &pending_entries,
1138													  &cancel_requested))
1139							continue;
1140					}
1141					else
1142						pgfdw_abort_cleanup(entry, true);
1143					break;
1144			}
1145		}
1146
1147		/* Reset state to show we're out of a transaction */
1148		pgfdw_reset_xact_state(entry, true);
1149	}
1150
1151	/* If there are any pending connections, finish cleaning them up */
1152	if (pending_entries || cancel_requested)
1153	{
1154		if (event == XACT_EVENT_PARALLEL_PRE_COMMIT ||
1155			event == XACT_EVENT_PRE_COMMIT)
1156		{
1157			Assert(cancel_requested == NIL);
1158			pgfdw_finish_pre_commit_cleanup(pending_entries);
1159		}
1160		else
1161		{
1162			Assert(event == XACT_EVENT_PARALLEL_ABORT ||
1163				   event == XACT_EVENT_ABORT);
1164			pgfdw_finish_abort_cleanup(pending_entries, cancel_requested,
1165									   true);
1166		}
1167	}
1168
1169	/*
1170	 * Regardless of the event type, we can now mark ourselves as out of the
1171	 * transaction.  (Note: if we are here during PRE_COMMIT or PRE_PREPARE,
1172	 * this saves a useless scan of the hashtable during COMMIT or PREPARE.)
1173	 */
1174	xact_got_connection = false;
1175
1176	/* Also reset cursor numbering for next transaction */
1177	cursor_number = 0;
1178}

1180/*
1181 * pgfdw_subxact_callback --- cleanup at subtransaction end.
1182 */
1183static void
1184pgfdw_subxact_callback(SubXactEvent event, SubTransactionId mySubid,
1185					   SubTransactionId parentSubid, void *arg)
1186{
1187	HASH_SEQ_STATUS scan;
1188	ConnCacheEntry *entry;
1189	int			curlevel;
1190	List	   *pending_entries = NIL;
1191	List	   *cancel_requested = NIL;
1192
1193	/* Nothing to do at subxact start, nor after commit. */
1194	if (!(event == SUBXACT_EVENT_PRE_COMMIT_SUB ||
1195		  event == SUBXACT_EVENT_ABORT_SUB))
1196		return;
1197
1198	/* Quick exit if no connections were touched in this transaction. */
1199	if (!xact_got_connection)
1200		return;
1201
1202	/*
1203	 * Scan all connection cache entries to find open remote subtransactions
1204	 * of the current level, and close them.
1205	 */
1206	curlevel = GetCurrentTransactionNestLevel();
1207	hash_seq_init(&scan, ConnectionHash);
1208	while ((entry = (ConnCacheEntry *) hash_seq_search(&scan)))
1209	{
1210		char		sql[100];
1211
1212		/*
1213		 * We only care about connections with open remote subtransactions of
1214		 * the current level.
1215		 */
1216		if (entry->conn == NULL || entry->xact_depth < curlevel)
1217			continue;
1218
1219		if (entry->xact_depth > curlevel)
1220			elog(ERROR, "missed cleaning up remote subtransaction at level %d",
1221				 entry->xact_depth);
1222
1223		if (event == SUBXACT_EVENT_PRE_COMMIT_SUB)
1224		{
1225			/*
1226			 * If abort cleanup previously failed for this connection, we
1227			 * can't issue any more commands against it.
1228			 */
1229			pgfdw_reject_incomplete_xact_state_change(entry);
1230
1231			/* Commit all remote subtransactions during pre-commit */
1232			snprintf(sql, sizeof(sql), "RELEASE SAVEPOINT s%d", curlevel);
1233			entry->changing_xact_state = true;
1234			if (entry->parallel_commit)
1235			{
1236				do_sql_command_begin(entry->conn, sql);
1237				pending_entries = lappend(pending_entries, entry);
1238				continue;
1239			}
1240			do_sql_command(entry->conn, sql);
1241			entry->changing_xact_state = false;
1242		}
1243		else
1244		{
1245			/* Rollback all remote subtransactions during abort */
1246			if (entry->parallel_abort)
1247			{
1248				if (pgfdw_abort_cleanup_begin(entry, false,
1249											  &pending_entries,
1250											  &cancel_requested))
1251					continue;
1252			}
1253			else
1254				pgfdw_abort_cleanup(entry, false);
1255		}
1256
1257		/* OK, we're outta that level of subtransaction */
1258		pgfdw_reset_xact_state(entry, false);
1259	}
1260
1261	/* If there are any pending connections, finish cleaning them up */
1262	if (pending_entries || cancel_requested)
1263	{
1264		if (event == SUBXACT_EVENT_PRE_COMMIT_SUB)
1265		{
1266			Assert(cancel_requested == NIL);
1267			pgfdw_finish_pre_subcommit_cleanup(pending_entries, curlevel);
1268		}
1269		else
1270		{
1271			Assert(event == SUBXACT_EVENT_ABORT_SUB);
1272			pgfdw_finish_abort_cleanup(pending_entries, cancel_requested,
1273									   false);
1274		}
1275	}
1276}

1278/*
1279 * Connection invalidation callback function
1280 *
1281 * After a change to a pg_foreign_server or pg_user_mapping catalog entry,
1282 * close connections depending on that entry immediately if current transaction
1283 * has not used those connections yet. Otherwise, mark those connections as
1284 * invalid and then make pgfdw_xact_callback() close them at the end of current
1285 * transaction, since they cannot be closed in the midst of the transaction
1286 * using them. Closed connections will be remade at the next opportunity if
1287 * necessary.
1288 *
1289 * Although most cache invalidation callbacks blow away all the related stuff
1290 * regardless of the given hashvalue, connections are expensive enough that
1291 * it's worth trying to avoid that.
1292 *
1293 * NB: We could avoid unnecessary disconnection more strictly by examining
1294 * individual option values, but it seems too much effort for the gain.
1295 */
1296static void
1297pgfdw_inval_callback(Datum arg, int cacheid, uint32 hashvalue)
1298{
1299	HASH_SEQ_STATUS scan;
1300	ConnCacheEntry *entry;
1301
1302	Assert(cacheid == FOREIGNSERVEROID || cacheid == USERMAPPINGOID);
1303
1304	/* ConnectionHash must exist already, if we're registered */
1305	hash_seq_init(&scan, ConnectionHash);
1306	while ((entry = (ConnCacheEntry *) hash_seq_search(&scan)))
1307	{
1308		/* Ignore invalid entries */
1309		if (entry->conn == NULL)
1310			continue;
1311
1312		/* hashvalue == 0 means a cache reset, must clear all state */
1313		if (hashvalue == 0 ||
1314			(cacheid == FOREIGNSERVEROID &&
1315			 entry->server_hashvalue == hashvalue) ||
1316			(cacheid == USERMAPPINGOID &&
1317			 entry->mapping_hashvalue == hashvalue))
1318		{
1319			/*
1320			 * Close the connection immediately if it's not used yet in this
1321			 * transaction. Otherwise mark it as invalid so that
1322			 * pgfdw_xact_callback() can close it at the end of this
1323			 * transaction.
1324			 */
1325			if (entry->xact_depth == 0)
1326			{
1327				elog(DEBUG3, "discarding connection %p", entry->conn);
1328				disconnect_pg_server(entry);
1329			}
1330			else
1331				entry->invalidated = true;
1332		}
1333	}
1334}

1336/*
1337 * Raise an error if the given connection cache entry is marked as being
1338 * in the middle of an xact state change.  This should be called at which no
1339 * such change is expected to be in progress; if one is found to be in
1340 * progress, it means that we aborted in the middle of a previous state change
1341 * and now don't know what the remote transaction state actually is.
1342 * Such connections can't safely be further used.  Re-establishing the
1343 * connection would change the snapshot and roll back any writes already
1344 * performed, so that's not an option, either. Thus, we must abort.
1345 *
1346 * Note: there might be open cursors that use the connection, so even if the
1347 * connection cache entry is marked as such, we will retain it until abort
1348 * cleanup of the main transaction, to ensure such open cursors can safely
1349 * refer to the PGconn for the connection.
1350 */
1351static void
1352pgfdw_reject_incomplete_xact_state_change(ConnCacheEntry *entry)
1353{
1354	ForeignServer *server;
1355
1356	/* nothing to do for inactive entries and entries of sane state */
1357	if (entry->conn == NULL || !entry->changing_xact_state)
1358		return;
1359
1360	/* find server name to be shown in the message below */
1361	server = GetForeignServer(entry->serverid);
1362
1363	ereport(ERROR,
1364			(errcode(ERRCODE_CONNECTION_EXCEPTION),
1365			 errmsg("connection to server \"%s\" cannot be used due to abort cleanup failure",
1366					server->servername)));
1367}

1369/*
1370 * Reset state to show we're out of a (sub)transaction.
1371 */
1372static void
1373pgfdw_reset_xact_state(ConnCacheEntry *entry, bool toplevel)
1374{
1375	if (toplevel)
1376	{
1377		/* Reset state to show we're out of a transaction */
1378		entry->xact_depth = 0;
1379
1380		/*
1381		 * If the connection isn't in a good idle state, it is marked as
1382		 * invalid or keep_connections option of its server is disabled, then
1383		 * discard it to recover. Next GetConnection will open a new
1384		 * connection.
1385		 */
1386		if (PQstatus(entry->conn) != CONNECTION_OK ||
1387			PQtransactionStatus(entry->conn) != PQTRANS_IDLE ||
1388			entry->changing_xact_state ||
1389			entry->invalidated ||
1390			!entry->keep_connections)
1391		{
1392			elog(DEBUG3, "discarding connection %p", entry->conn);
1393			disconnect_pg_server(entry);
1394		}
1395	}
1396	else
1397	{
1398		/* Reset state to show we're out of a subtransaction */
1399		entry->xact_depth--;
1400	}
1401}

1403/*
1404 * Cancel the currently-in-progress query (whose query text we do not have)
1405 * and ignore the result.  Returns true if we successfully cancel the query
1406 * and discard any pending result, and false if not.
1407 *
1408 * It's not a huge problem if we throw an ERROR here, but if we get into error
1409 * recursion trouble, we'll end up slamming the connection shut, which will
1410 * necessitate failing the entire toplevel transaction even if subtransactions
1411 * were used.  Try to use WARNING where we can.
1412 *
1413 * XXX: if the query was one sent by fetch_more_data_begin(), we could get the
1414 * query text from the pendingAreq saved in the per-connection state, then
1415 * report the query using it.
1416 */
1417static bool
1418pgfdw_cancel_query(PGconn *conn)
1419{
1420	TimestampTz now = GetCurrentTimestamp();
1421	TimestampTz endtime;
1422	TimestampTz retrycanceltime;
1423
1424	/*
1425	 * If it takes too long to cancel the query and discard the result, assume
1426	 * the connection is dead.
1427	 */
1428	endtime = TimestampTzPlusMilliseconds(now, CONNECTION_CLEANUP_TIMEOUT);
1429
1430	/*
1431	 * Also, lose patience and re-issue the cancel request after a little bit.
1432	 * (This serves to close some race conditions.)
1433	 */
1434	retrycanceltime = TimestampTzPlusMilliseconds(now, RETRY_CANCEL_TIMEOUT);
1435
1436	if (!pgfdw_cancel_query_begin(conn, endtime))
1437		return false;
1438	return pgfdw_cancel_query_end(conn, endtime, retrycanceltime, false);
1439}

1441/*
1442 * Submit a cancel request to the given connection, waiting only until
1443 * the given time.
1444 *
1445 * We sleep interruptibly until we receive confirmation that the cancel
1446 * request has been accepted, and if it is, return true; if the timeout
1447 * lapses without that, or the request fails for whatever reason, return
1448 * false.
1449 */
1450static bool
1451pgfdw_cancel_query_begin(PGconn *conn, TimestampTz endtime)
1452{
1453	const char *errormsg = libpqsrv_cancel(conn, endtime);
1454
1455	if (errormsg != NULL)
1456		ereport(WARNING,
1457				errcode(ERRCODE_CONNECTION_FAILURE),
1458				errmsg("could not send cancel request: %s", errormsg));
1459
1460	return errormsg == NULL;
1461}

1463static bool
1464pgfdw_cancel_query_end(PGconn *conn, TimestampTz endtime,
1465					   TimestampTz retrycanceltime, bool consume_input)
1466{
1467	PGresult   *result;
1468	bool		timed_out;
1469
1470	/*
1471	 * If requested, consume whatever data is available from the socket. (Note
1472	 * that if all data is available, this allows pgfdw_get_cleanup_result to
1473	 * call PQgetResult without forcing the overhead of WaitLatchOrSocket,
1474	 * which would be large compared to the overhead of PQconsumeInput.)
1475	 */
1476	if (consume_input && !PQconsumeInput(conn))
1477	{
1478		ereport(WARNING,
1479				(errcode(ERRCODE_CONNECTION_FAILURE),
1480				 errmsg("could not get result of cancel request: %s",
1481						pchomp(PQerrorMessage(conn)))));
1482		return false;
1483	}
1484
1485	/* Get and discard the result of the query. */
1486	if (pgfdw_get_cleanup_result(conn, endtime, retrycanceltime,
1487								 &result, &timed_out))
1488	{
1489		if (timed_out)
1490			ereport(WARNING,
1491					(errmsg("could not get result of cancel request due to timeout")));
1492		else
1493			ereport(WARNING,
1494					(errcode(ERRCODE_CONNECTION_FAILURE),
1495					 errmsg("could not get result of cancel request: %s",
1496							pchomp(PQerrorMessage(conn)))));
1497
1498		return false;
1499	}
1500	PQclear(result);
1501
1502	return true;
1503}

1505/*
1506 * Submit a query during (sub)abort cleanup and wait up to 30 seconds for the
1507 * result.  If the query is executed without error, the return value is true.
1508 * If the query is executed successfully but returns an error, the return
1509 * value is true if and only if ignore_errors is set.  If the query can't be
1510 * sent or times out, the return value is false.
1511 *
1512 * It's not a huge problem if we throw an ERROR here, but if we get into error
1513 * recursion trouble, we'll end up slamming the connection shut, which will
1514 * necessitate failing the entire toplevel transaction even if subtransactions
1515 * were used.  Try to use WARNING where we can.
1516 */
1517static bool
1518pgfdw_exec_cleanup_query(PGconn *conn, const char *query, bool ignore_errors)
1519{
1520	TimestampTz endtime;
1521
1522	/*
1523	 * If it takes too long to execute a cleanup query, assume the connection
1524	 * is dead.  It's fairly likely that this is why we aborted in the first
1525	 * place (e.g. statement timeout, user cancel), so the timeout shouldn't
1526	 * be too long.
1527	 */
1528	endtime = TimestampTzPlusMilliseconds(GetCurrentTimestamp(),
1529										  CONNECTION_CLEANUP_TIMEOUT);
1530
1531	if (!pgfdw_exec_cleanup_query_begin(conn, query))
1532		return false;
1533	return pgfdw_exec_cleanup_query_end(conn, query, endtime,
1534										false, ignore_errors);
1535}
1536
1537static bool
1538pgfdw_exec_cleanup_query_begin(PGconn *conn, const char *query)
1539{
1540	Assert(query != NULL);
1541
1542	/*
1543	 * Submit a query.  Since we don't use non-blocking mode, this also can
1544	 * block.  But its risk is relatively small, so we ignore that for now.
1545	 */
1546	if (!PQsendQuery(conn, query))
1547	{
1548		pgfdw_report_error(WARNING, NULL, conn, false, query);
1549		return false;
1550	}
1551
1552	return true;
1553}

1555static bool
1556pgfdw_exec_cleanup_query_end(PGconn *conn, const char *query,
1557							 TimestampTz endtime, bool consume_input,
1558							 bool ignore_errors)
1559{
1560	PGresult   *result;
1561	bool		timed_out;
1562
1563	Assert(query != NULL);
1564
1565	/*
1566	 * If requested, consume whatever data is available from the socket. (Note
1567	 * that if all data is available, this allows pgfdw_get_cleanup_result to
1568	 * call PQgetResult without forcing the overhead of WaitLatchOrSocket,
1569	 * which would be large compared to the overhead of PQconsumeInput.)
1570	 */
1571	if (consume_input && !PQconsumeInput(conn))
1572	{
1573		pgfdw_report_error(WARNING, NULL, conn, false, query);
1574		return false;
1575	}
1576
1577	/* Get the result of the query. */
1578	if (pgfdw_get_cleanup_result(conn, endtime, endtime, &result, &timed_out))
1579	{
1580		if (timed_out)
1581			ereport(WARNING,
1582					(errmsg("could not get query result due to timeout"),
1583					 errcontext("remote SQL command: %s", query)));
1584		else
1585			pgfdw_report_error(WARNING, NULL, conn, false, query);
1586
1587		return false;
1588	}
1589
1590	/* Issue a warning if not successful. */
1591	if (PQresultStatus(result) != PGRES_COMMAND_OK)
1592	{
1593		pgfdw_report_error(WARNING, result, conn, true, query);
1594		return ignore_errors;
1595	}
1596	PQclear(result);
1597
1598	return true;
1599}

1601/*
1602 * Get, during abort cleanup, the result of a query that is in progress.
1603 * This might be a query that is being interrupted by a cancel request or by
1604 * transaction abort, or it might be a query that was initiated as part of
1605 * transaction abort to get the remote side back to the appropriate state.
1606 *
1607 * endtime is the time at which we should give up and assume the remote side
1608 * is dead.  retrycanceltime is the time at which we should issue a fresh
1609 * cancel request (pass the same value as endtime if this is not wanted).
1610 *
1611 * Returns true if the timeout expired or connection trouble occurred,
1612 * false otherwise.  Sets *result except in case of a true result.
1613 * Sets *timed_out to true only when the timeout expired.
1614 */
1615static bool
1616pgfdw_get_cleanup_result(PGconn *conn, TimestampTz endtime,
1617						 TimestampTz retrycanceltime,
1618						 PGresult **result,
1619						 bool *timed_out)
1620{
1621	volatile bool failed = false;
1622	PGresult   *volatile last_res = NULL;
1623
1624	*result = NULL;
1625	*timed_out = false;
1626
1627	/* In what follows, do not leak any PGresults on an error. */
1628	PG_TRY();
1629	{
1630		int			canceldelta = RETRY_CANCEL_TIMEOUT * 2;
1631
1632		for (;;)
1633		{
1634			PGresult   *res;
1635
1636			while (PQisBusy(conn))
1637			{
1638				int			wc;
1639				TimestampTz now = GetCurrentTimestamp();
1640				long		cur_timeout;
1641
1642				/* If timeout has expired, give up. */
1643				if (now >= endtime)
1644				{
1645					*timed_out = true;
1646					failed = true;
1647					goto exit;
1648				}
1649
1650				/* If we need to re-issue the cancel request, do that. */
1651				if (now >= retrycanceltime)
1652				{
1653					/* We ignore failure to issue the repeated request. */
1654					(void) libpqsrv_cancel(conn, endtime);
1655
1656					/* Recompute "now" in case that took measurable time. */
1657					now = GetCurrentTimestamp();
1658
1659					/* Adjust re-cancel timeout in increasing steps. */
1660					retrycanceltime = TimestampTzPlusMilliseconds(now,
1661																  canceldelta);
1662					canceldelta += canceldelta;
1663				}
1664
1665				/* If timeout has expired, give up, else get sleep time. */
1666				cur_timeout = TimestampDifferenceMilliseconds(now,
1667															  Min(endtime,
1668																  retrycanceltime));
1669				if (cur_timeout <= 0)
1670				{
1671					*timed_out = true;
1672					failed = true;
1673					goto exit;
1674				}
1675
1676				/* first time, allocate or get the custom wait event */
1677				if (pgfdw_we_cleanup_result == 0)
1678					pgfdw_we_cleanup_result = WaitEventExtensionNew("PostgresFdwCleanupResult");
1679
1680				/* Sleep until there's something to do */
1681				wc = WaitLatchOrSocket(MyLatch,
1682									   WL_LATCH_SET | WL_SOCKET_READABLE |
1683									   WL_TIMEOUT | WL_EXIT_ON_PM_DEATH,
1684									   PQsocket(conn),
1685									   cur_timeout, pgfdw_we_cleanup_result);
1686				ResetLatch(MyLatch);
1687
1688				CHECK_FOR_INTERRUPTS();
1689
1690				/* Data available in socket? */
1691				if (wc & WL_SOCKET_READABLE)
1692				{
1693					if (!PQconsumeInput(conn))
1694					{
1695						/* connection trouble */
1696						failed = true;
1697						goto exit;
1698					}
1699				}
1700			}
1701
1702			res = PQgetResult(conn);
1703			if (res == NULL)
1704				break;			/* query is complete */
1705
1706			PQclear(last_res);
1707			last_res = res;
1708		}
1709exit:	;
1710	}
1711	PG_CATCH();
1712	{
1713		PQclear(last_res);
1714		PG_RE_THROW();
1715	}
1716	PG_END_TRY();
1717
1718	if (failed)
1719		PQclear(last_res);
1720	else
1721		*result = last_res;
1722	return failed;
1723}

1725/*
1726 * Abort remote transaction or subtransaction.
1727 *
1728 * "toplevel" should be set to true if toplevel (main) transaction is
1729 * rollbacked, false otherwise.
1730 *
1731 * Set entry->changing_xact_state to false on success, true on failure.
1732 */
1733static void
1734pgfdw_abort_cleanup(ConnCacheEntry *entry, bool toplevel)
1735{
1736	char		sql[100];
1737
1738	/*
1739	 * Don't try to clean up the connection if we're already in error
1740	 * recursion trouble.
1741	 */
1742	if (in_error_recursion_trouble())
1743		entry->changing_xact_state = true;
1744
1745	/*
1746	 * If connection is already unsalvageable, don't touch it further.
1747	 */
1748	if (entry->changing_xact_state)
1749		return;
1750
1751	/*
1752	 * Mark this connection as in the process of changing transaction state.
1753	 */
1754	entry->changing_xact_state = true;
1755
1756	/* Assume we might have lost track of prepared statements */
1757	entry->have_error = true;
1758
1759	/*
1760	 * If a command has been submitted to the remote server by using an
1761	 * asynchronous execution function, the command might not have yet
1762	 * completed.  Check to see if a command is still being processed by the
1763	 * remote server, and if so, request cancellation of the command.
1764	 */
1765	if (PQtransactionStatus(entry->conn) == PQTRANS_ACTIVE &&
1766		!pgfdw_cancel_query(entry->conn))
1767		return;					/* Unable to cancel running query */
1768
1769	CONSTRUCT_ABORT_COMMAND(sql, entry, toplevel);
1770	if (!pgfdw_exec_cleanup_query(entry->conn, sql, false))
1771		return;					/* Unable to abort remote (sub)transaction */
1772
1773	if (toplevel)
1774	{
1775		if (entry->have_prep_stmt && entry->have_error &&
1776			!pgfdw_exec_cleanup_query(entry->conn,
1777									  "DEALLOCATE ALL",
1778									  true))
1779			return;				/* Trouble clearing prepared statements */
1780
1781		entry->have_prep_stmt = false;
1782		entry->have_error = false;
1783	}
1784
1785	/*
1786	 * If pendingAreq of the per-connection state is not NULL, it means that
1787	 * an asynchronous fetch begun by fetch_more_data_begin() was not done
1788	 * successfully and thus the per-connection state was not reset in
1789	 * fetch_more_data(); in that case reset the per-connection state here.
1790	 */
1791	if (entry->state.pendingAreq)
1792		memset(&entry->state, 0, sizeof(entry->state));
1793
1794	/* Disarm changing_xact_state if it all worked */
1795	entry->changing_xact_state = false;
1796}

1798/*
1799 * Like pgfdw_abort_cleanup, submit an abort command or cancel request, but
1800 * don't wait for the result.
1801 *
1802 * Returns true if the abort command or cancel request is successfully issued,
1803 * false otherwise.  If the abort command is successfully issued, the given
1804 * connection cache entry is appended to *pending_entries.  Otherwise, if the
1805 * cancel request is successfully issued, it is appended to *cancel_requested.
1806 */
1807static bool
1808pgfdw_abort_cleanup_begin(ConnCacheEntry *entry, bool toplevel,
1809						  List **pending_entries, List **cancel_requested)
1810{
1811	/*
1812	 * Don't try to clean up the connection if we're already in error
1813	 * recursion trouble.
1814	 */
1815	if (in_error_recursion_trouble())
1816		entry->changing_xact_state = true;
1817
1818	/*
1819	 * If connection is already unsalvageable, don't touch it further.
1820	 */
1821	if (entry->changing_xact_state)
1822		return false;
1823
1824	/*
1825	 * Mark this connection as in the process of changing transaction state.
1826	 */
1827	entry->changing_xact_state = true;
1828
1829	/* Assume we might have lost track of prepared statements */
1830	entry->have_error = true;
1831
1832	/*
1833	 * If a command has been submitted to the remote server by using an
1834	 * asynchronous execution function, the command might not have yet
1835	 * completed.  Check to see if a command is still being processed by the
1836	 * remote server, and if so, request cancellation of the command.
1837	 */
1838	if (PQtransactionStatus(entry->conn) == PQTRANS_ACTIVE)
1839	{
1840		TimestampTz endtime;
1841
1842		endtime = TimestampTzPlusMilliseconds(GetCurrentTimestamp(),
1843											  CONNECTION_CLEANUP_TIMEOUT);
1844		if (!pgfdw_cancel_query_begin(entry->conn, endtime))
1845			return false;		/* Unable to cancel running query */
1846		*cancel_requested = lappend(*cancel_requested, entry);
1847	}
1848	else
1849	{
1850		char		sql[100];
1851
1852		CONSTRUCT_ABORT_COMMAND(sql, entry, toplevel);
1853		if (!pgfdw_exec_cleanup_query_begin(entry->conn, sql))
1854			return false;		/* Unable to abort remote transaction */
1855		*pending_entries = lappend(*pending_entries, entry);
1856	}
1857
1858	return true;
1859}

1861/*
1862 * Finish pre-commit cleanup of connections on each of which we've sent a
1863 * COMMIT command to the remote server.
1864 */
1865static void
1866pgfdw_finish_pre_commit_cleanup(List *pending_entries)
1867{
1868	ConnCacheEntry *entry;
1869	List	   *pending_deallocs = NIL;
1870	ListCell   *lc;
1871
1872	Assert(pending_entries);
1873
1874	/*
1875	 * Get the result of the COMMIT command for each of the pending entries
1876	 */
1877	foreach(lc, pending_entries)
1878	{
1879		entry = (ConnCacheEntry *) lfirst(lc);
1880
1881		Assert(entry->changing_xact_state);
1882
1883		/*
1884		 * We might already have received the result on the socket, so pass
1885		 * consume_input=true to try to consume it first
1886		 */
1887		do_sql_command_end(entry->conn, "COMMIT TRANSACTION", true);
1888		entry->changing_xact_state = false;
1889
1890		/* Do a DEALLOCATE ALL in parallel if needed */
1891		if (entry->have_prep_stmt && entry->have_error)
1892		{
1893			/* Ignore errors (see notes in pgfdw_xact_callback) */
1894			if (PQsendQuery(entry->conn, "DEALLOCATE ALL"))
1895			{
1896				pending_deallocs = lappend(pending_deallocs, entry);
1897				continue;
1898			}
1899		}
1900		entry->have_prep_stmt = false;
1901		entry->have_error = false;
1902
1903		pgfdw_reset_xact_state(entry, true);
1904	}
1905
1906	/* No further work if no pending entries */
1907	if (!pending_deallocs)
1908		return;
1909
1910	/*
1911	 * Get the result of the DEALLOCATE command for each of the pending
1912	 * entries
1913	 */
1914	foreach(lc, pending_deallocs)
1915	{
1916		PGresult   *res;
1917
1918		entry = (ConnCacheEntry *) lfirst(lc);
1919
1920		/* Ignore errors (see notes in pgfdw_xact_callback) */
1921		while ((res = PQgetResult(entry->conn)) != NULL)
1922		{
1923			PQclear(res);
1924			/* Stop if the connection is lost (else we'll loop infinitely) */
1925			if (PQstatus(entry->conn) == CONNECTION_BAD)
1926				break;
1927		}
1928		entry->have_prep_stmt = false;
1929		entry->have_error = false;
1930
1931		pgfdw_reset_xact_state(entry, true);
1932	}
1933}

1936 * Finish pre-subcommit cleanup of connections on each of which we've sent a
1937 * RELEASE command to the remote server.
1938 */
1939static void
1940pgfdw_finish_pre_subcommit_cleanup(List *pending_entries, int curlevel)

1942	ConnCacheEntry *entry;
1943	char		sql[100];
1944	ListCell   *lc;
1945
1946	Assert(pending_entries);
1947
1948	/*
1949	 * Get the result of the RELEASE command for each of the pending entries
1950	 */
1951	snprintf(sql, sizeof(sql), "RELEASE SAVEPOINT s%d", curlevel);
1952	foreach(lc, pending_entries)
1953	{
1954		entry = (ConnCacheEntry *) lfirst(lc);
1955
1956		Assert(entry->changing_xact_state);
1957
1958		/*
1959		 * We might already have received the result on the socket, so pass
1960		 * consume_input=true to try to consume it first
1961		 */
1962		do_sql_command_end(entry->conn, sql, true);
1963		entry->changing_xact_state = false;
1964
1965		pgfdw_reset_xact_state(entry, false);
1966	}
1967}

1969/*
1970 * Finish abort cleanup of connections on each of which we've sent an abort
1971 * command or cancel request to the remote server.
1972 */
1973static void
1974pgfdw_finish_abort_cleanup(List *pending_entries, List *cancel_requested,
1975						   bool toplevel)
1976{
1977	List	   *pending_deallocs = NIL;
1978	ListCell   *lc;
1979
1980	/*
1981	 * For each of the pending cancel requests (if any), get and discard the
1982	 * result of the query, and submit an abort command to the remote server.
1983	 */
1984	if (cancel_requested)
1985	{
1986		foreach(lc, cancel_requested)
1987		{
1988			ConnCacheEntry *entry = (ConnCacheEntry *) lfirst(lc);
1989			TimestampTz now = GetCurrentTimestamp();
1990			TimestampTz endtime;
1991			TimestampTz retrycanceltime;
1992			char		sql[100];
1993
1994			Assert(entry->changing_xact_state);
1995
1996			/*
1997			 * Set end time.  You might think we should do this before issuing
1998			 * cancel request like in normal mode, but that is problematic,
1999			 * because if, for example, it took longer than 30 seconds to
2000			 * process the first few entries in the cancel_requested list, it
2001			 * would cause a timeout error when processing each of the
2002			 * remaining entries in the list, leading to slamming that entry's
2003			 * connection shut.
2004			 */
2005			endtime = TimestampTzPlusMilliseconds(now,
2006												  CONNECTION_CLEANUP_TIMEOUT);
2007			retrycanceltime = TimestampTzPlusMilliseconds(now,
2008														  RETRY_CANCEL_TIMEOUT);
2009
2010			if (!pgfdw_cancel_query_end(entry->conn, endtime,
2011										retrycanceltime, true))
2012			{
2013				/* Unable to cancel running query */
2014				pgfdw_reset_xact_state(entry, toplevel);
2015				continue;
2016			}
2017
2018			/* Send an abort command in parallel if needed */
2019			CONSTRUCT_ABORT_COMMAND(sql, entry, toplevel);
2020			if (!pgfdw_exec_cleanup_query_begin(entry->conn, sql))
2021			{
2022				/* Unable to abort remote (sub)transaction */
2023				pgfdw_reset_xact_state(entry, toplevel);
2024			}
2025			else
2026				pending_entries = lappend(pending_entries, entry);
2027		}
2028	}
2029
2030	/* No further work if no pending entries */
2031	if (!pending_entries)
2032		return;
2033
2034	/*
2035	 * Get the result of the abort command for each of the pending entries
2036	 */
2037	foreach(lc, pending_entries)
2038	{
2039		ConnCacheEntry *entry = (ConnCacheEntry *) lfirst(lc);
2040		TimestampTz endtime;
2041		char		sql[100];
2042
2043		Assert(entry->changing_xact_state);
2044
2045		/*
2046		 * Set end time.  We do this now, not before issuing the command like
2047		 * in normal mode, for the same reason as for the cancel_requested
2048		 * entries.
2049		 */
2050		endtime = TimestampTzPlusMilliseconds(GetCurrentTimestamp(),
2051											  CONNECTION_CLEANUP_TIMEOUT);
2052
2053		CONSTRUCT_ABORT_COMMAND(sql, entry, toplevel);
2054		if (!pgfdw_exec_cleanup_query_end(entry->conn, sql, endtime,
2055										  true, false))
2056		{
2057			/* Unable to abort remote (sub)transaction */
2058			pgfdw_reset_xact_state(entry, toplevel);
2059			continue;
2060		}
2061
2062		if (toplevel)
2063		{
2064			/* Do a DEALLOCATE ALL in parallel if needed */
2065			if (entry->have_prep_stmt && entry->have_error)
2066			{
2067				if (!pgfdw_exec_cleanup_query_begin(entry->conn,
2068													"DEALLOCATE ALL"))
2069				{
2070					/* Trouble clearing prepared statements */
2071					pgfdw_reset_xact_state(entry, toplevel);
2072				}
2073				else
2074					pending_deallocs = lappend(pending_deallocs, entry);
2075				continue;
2076			}
2077			entry->have_prep_stmt = false;
2078			entry->have_error = false;
2079		}
2080
2081		/* Reset the per-connection state if needed */
2082		if (entry->state.pendingAreq)
2083			memset(&entry->state, 0, sizeof(entry->state));
2084
2085		/* We're done with this entry; unset the changing_xact_state flag */
2086		entry->changing_xact_state = false;
2087		pgfdw_reset_xact_state(entry, toplevel);
2088	}
2089
2090	/* No further work if no pending entries */
2091	if (!pending_deallocs)
2092		return;
2093	Assert(toplevel);
2094
2095	/*
2096	 * Get the result of the DEALLOCATE command for each of the pending
2097	 * entries
2098	 */
2099	foreach(lc, pending_deallocs)
2100	{
2101		ConnCacheEntry *entry = (ConnCacheEntry *) lfirst(lc);
2102		TimestampTz endtime;
2103
2104		Assert(entry->changing_xact_state);
2105		Assert(entry->have_prep_stmt);
2106		Assert(entry->have_error);
2107
2108		/*
2109		 * Set end time.  We do this now, not before issuing the command like
2110		 * in normal mode, for the same reason as for the cancel_requested
2111		 * entries.
2112		 */
2113		endtime = TimestampTzPlusMilliseconds(GetCurrentTimestamp(),
2114											  CONNECTION_CLEANUP_TIMEOUT);
2115
2116		if (!pgfdw_exec_cleanup_query_end(entry->conn, "DEALLOCATE ALL",
2117										  endtime, true, true))
2118		{
2119			/* Trouble clearing prepared statements */
2120			pgfdw_reset_xact_state(entry, toplevel);
2121			continue;
2122		}
2123		entry->have_prep_stmt = false;
2124		entry->have_error = false;
2125
2126		/* Reset the per-connection state if needed */
2127		if (entry->state.pendingAreq)
2128			memset(&entry->state, 0, sizeof(entry->state));
2129
2130		/* We're done with this entry; unset the changing_xact_state flag */
2131		entry->changing_xact_state = false;
2132		pgfdw_reset_xact_state(entry, toplevel);
2133	}
2134}

2141/*
2142 * Internal function used by postgres_fdw_get_connections variants.
2143 *
2144 * For API version 1.1, this function takes no input parameter and
2145 * returns a set of records with the following values:
2146 *
2147 * - server_name - server name of active connection. In case the foreign server
2148 *   is dropped but still the connection is active, then the server name will
2149 *   be NULL in output.
2150 * - valid - true/false representing whether the connection is valid or not.
2151 *   Note that connections can become invalid in pgfdw_inval_callback.
2152 *
2153 * For API version 1.2 and later, this function takes an input parameter
2154 * to check a connection status and returns the following
2155 * additional values along with the four values from version 1.1:
2156 *
2157 * - user_name - the local user name of the active connection. In case the
2158 *   user mapping is dropped but the connection is still active, then the
2159 *   user name will be NULL in the output.
2160 * - used_in_xact - true if the connection is used in the current transaction.
2161 * - closed - true if the connection is closed.
2162 * - remote_backend_pid - process ID of the remote backend, on the foreign
2163 *   server, handling the connection.
2164 *
2165 * No records are returned when there are no cached connections at all.
2166 */
2167static void
2168postgres_fdw_get_connections_internal(FunctionCallInfo fcinfo,
2169									  enum pgfdwVersion api_version)
2170{
2171	ReturnSetInfo *rsinfo = (ReturnSetInfo *) fcinfo->resultinfo;
2172	HASH_SEQ_STATUS scan;
2173	ConnCacheEntry *entry;
2174
2175	InitMaterializedSRF(fcinfo, 0);
2176
2177	/* If cache doesn't exist, we return no records */
2178	if (!ConnectionHash)
2179		return;
2180
2181	/* Check we have the expected number of output arguments */
2182	switch (rsinfo->setDesc->natts)
2183	{
2184		case POSTGRES_FDW_GET_CONNECTIONS_COLS_V1_1:
2185			if (api_version != PGFDW_V1_1)
2186				elog(ERROR, "incorrect number of output arguments");
2187			break;
2188		case POSTGRES_FDW_GET_CONNECTIONS_COLS_V1_2:
2189			if (api_version != PGFDW_V1_2)
2190				elog(ERROR, "incorrect number of output arguments");
2191			break;
2192		default:
2193			elog(ERROR, "incorrect number of output arguments");
2194	}
2195
2196	hash_seq_init(&scan, ConnectionHash);
2197	while ((entry = (ConnCacheEntry *) hash_seq_search(&scan)))
2198	{
2199		ForeignServer *server;
2200		Datum		values[POSTGRES_FDW_GET_CONNECTIONS_COLS] = {0};
2201		bool		nulls[POSTGRES_FDW_GET_CONNECTIONS_COLS] = {0};
2202		int			i = 0;
2203
2204		/* We only look for open remote connections */
2205		if (!entry->conn)
2206			continue;
2207
2208		server = GetForeignServerExtended(entry->serverid, FSV_MISSING_OK);
2209
2210		/*
2211		 * The foreign server may have been dropped in current explicit
2212		 * transaction. It is not possible to drop the server from another
2213		 * session when the connection associated with it is in use in the
2214		 * current transaction, if tried so, the drop query in another session
2215		 * blocks until the current transaction finishes.
2216		 *
2217		 * Even though the server is dropped in the current transaction, the
2218		 * cache can still have associated active connection entry, say we
2219		 * call such connections dangling. Since we can not fetch the server
2220		 * name from system catalogs for dangling connections, instead we show
2221		 * NULL value for server name in output.
2222		 *
2223		 * We could have done better by storing the server name in the cache
2224		 * entry instead of server oid so that it could be used in the output.
2225		 * But the server name in each cache entry requires 64 bytes of
2226		 * memory, which is huge, when there are many cached connections and
2227		 * the use case i.e. dropping the foreign server within the explicit
2228		 * current transaction seems rare. So, we chose to show NULL value for
2229		 * server name in output.
2230		 *
2231		 * Such dangling connections get closed either in next use or at the
2232		 * end of current explicit transaction in pgfdw_xact_callback.
2233		 */
2234		if (!server)
2235		{
2236			/*
2237			 * If the server has been dropped in the current explicit
2238			 * transaction, then this entry would have been invalidated in
2239			 * pgfdw_inval_callback at the end of drop server command. Note
2240			 * that this connection would not have been closed in
2241			 * pgfdw_inval_callback because it is still being used in the
2242			 * current explicit transaction. So, assert that here.
2243			 */
2244			Assert(entry->conn && entry->xact_depth > 0 && entry->invalidated);
2245
2246			/* Show null, if no server name was found */
2247			nulls[i++] = true;
2248		}
2249		else
2250			values[i++] = CStringGetTextDatum(server->servername);
2251
2252		if (api_version >= PGFDW_V1_2)
2253		{
2254			HeapTuple	tp;
2255
2256			/* Use the system cache to obtain the user mapping */
2257			tp = SearchSysCache1(USERMAPPINGOID, ObjectIdGetDatum(entry->key));
2258
2259			/*
2260			 * Just like in the foreign server case, user mappings can also be
2261			 * dropped in the current explicit transaction. Therefore, the
2262			 * similar check as in the server case is required.
2263			 */
2264			if (!HeapTupleIsValid(tp))
2265			{
2266				/*
2267				 * If we reach here, this entry must have been invalidated in
2268				 * pgfdw_inval_callback, same as in the server case.
2269				 */
2270				Assert(entry->conn && entry->xact_depth > 0 &&
2271					   entry->invalidated);
2272
2273				nulls[i++] = true;
2274			}
2275			else
2276			{
2277				Oid			userid;
2278
2279				userid = ((Form_pg_user_mapping) GETSTRUCT(tp))->umuser;
2280				values[i++] = CStringGetTextDatum(MappingUserName(userid));
2281				ReleaseSysCache(tp);
2282			}
2283		}
2284
2285		values[i++] = BoolGetDatum(!entry->invalidated);
2286
2287		if (api_version >= PGFDW_V1_2)
2288		{
2289			bool		check_conn = PG_GETARG_BOOL(0);
2290
2291			/* Is this connection used in the current transaction? */
2292			values[i++] = BoolGetDatum(entry->xact_depth > 0);
2293
2294			/*
2295			 * If a connection status check is requested and supported, return
2296			 * whether the connection is closed. Otherwise, return NULL.
2297			 */
2298			if (check_conn && pgfdw_conn_checkable())
2299				values[i++] = BoolGetDatum(pgfdw_conn_check(entry->conn) != 0);
2300			else
2301				nulls[i++] = true;
2302
2303			/* Return process ID of remote backend */
2304			values[i++] = Int32GetDatum(PQbackendPID(entry->conn));
2305		}
2306
2307		tuplestore_putvalues(rsinfo->setResult, rsinfo->setDesc, values, nulls);
2308	}
2309}

2374/*
2375 * Workhorse to disconnect cached connections.
2376 *
2377 * This function scans all the connection cache entries and disconnects
2378 * the open connections whose foreign server OID matches with
2379 * the specified one. If InvalidOid is specified, it disconnects all
2380 * the cached connections.
2381 *
2382 * This function emits a warning for each connection that's used in
2383 * the current transaction and doesn't close it. It returns true if
2384 * it disconnects at least one connection, otherwise false.
2385 *
2386 * Note that this function disconnects even the connections that are
2387 * established by other users in the same local session using different
2388 * user mappings. This leads even non-superuser to be able to close
2389 * the connections established by superusers in the same local session.
2390 *
2391 * XXX As of now we don't see any security risk doing this. But we should
2392 * set some restrictions on that, for example, prevent non-superuser
2393 * from closing the connections established by superusers even
2394 * in the same session?
2395 */
2396static bool
2397disconnect_cached_connections(Oid serverid)
2398{
2399	HASH_SEQ_STATUS scan;
2400	ConnCacheEntry *entry;
2401	bool		all = !OidIsValid(serverid);
2402	bool		result = false;
2403
2404	/*
2405	 * Connection cache hashtable has not been initialized yet in this
2406	 * session, so return false.
2407	 */
2408	if (!ConnectionHash)
2409		return false;
2410
2411	hash_seq_init(&scan, ConnectionHash);
2412	while ((entry = (ConnCacheEntry *) hash_seq_search(&scan)))
2413	{
2414		/* Ignore cache entry if no open connection right now. */
2415		if (!entry->conn)
2416			continue;
2417
2418		if (all || entry->serverid == serverid)
2419		{
2420			/*
2421			 * Emit a warning because the connection to close is used in the
2422			 * current transaction and cannot be disconnected right now.
2423			 */
2424			if (entry->xact_depth > 0)
2425			{
2426				ForeignServer *server;
2427
2428				server = GetForeignServerExtended(entry->serverid,
2429												  FSV_MISSING_OK);
2430
2431				if (!server)
2432				{
2433					/*
2434					 * If the foreign server was dropped while its connection
2435					 * was used in the current transaction, the connection
2436					 * must have been marked as invalid by
2437					 * pgfdw_inval_callback at the end of DROP SERVER command.
2438					 */
2439					Assert(entry->invalidated);
2440
2441					ereport(WARNING,
2442							(errmsg("cannot close dropped server connection because it is still in use")));
2443				}
2444				else
2445					ereport(WARNING,
2446							(errmsg("cannot close connection for server \"%s\" because it is still in use",
2447									server->servername)));
2448			}
2449			else
2450			{
2451				elog(DEBUG3, "discarding connection %p", entry->conn);
2452				disconnect_pg_server(entry);
2453				result = true;
2454			}
2455		}
2456	}
2457
2458	return result;
2459}

2461/*
2462 * Check if the remote server closed the connection.
2463 *
2464 * Returns 1 if the connection is closed, -1 if an error occurred,
2465 * and 0 if it's not closed or if the connection check is unavailable
2466 * on this platform.
2467 */
2468static int
2469pgfdw_conn_check(PGconn *conn)
2470{
2471	int			sock = PQsocket(conn);
2472
2473	if (PQstatus(conn) != CONNECTION_OK || sock == -1)
2474		return -1;
2475
2476#if (defined(HAVE_POLL) && defined(POLLRDHUP))
2477	{
2478		struct pollfd input_fd;
2479		int			result;
2480
2481		input_fd.fd = sock;
2482		input_fd.events = POLLRDHUP;
2483		input_fd.revents = 0;
2484
2485		do
2486			result = poll(&input_fd, 1, 0);
2487		while (result < 0 && errno == EINTR);
2488
2489		if (result < 0)
2490			return -1;
2491
2492		return (input_fd.revents &
2493				(POLLRDHUP | POLLHUP | POLLERR | POLLNVAL)) ? 1 : 0;
2494	}
2495#else
2496	return 0;
2497#endif
2498}

2515/*
2516 * Ensure that require_auth and SCRAM keys are correctly set on values. SCRAM
2517 * keys used to pass-through are coming from the initial connection from the
2518 * client with the server.
2519 *
2520 * All required SCRAM options are set by postgres_fdw, so we just need to
2521 * ensure that these options are not overwritten by the user.
2522 */
2523static bool
2524pgfdw_has_required_scram_options(const char **keywords, const char **values)
2525{
2526	bool		has_scram_server_key = false;
2527	bool		has_scram_client_key = false;
2528	bool		has_require_auth = false;
2529	bool		has_scram_keys = false;
2530
2531	/*
2532	 * Continue iterating even if we found the keys that we need to validate
2533	 * to make sure that there is no other declaration of these keys that can
2534	 * overwrite the first.
2535	 */
2536	for (int i = 0; keywords[i] != NULL; i++)
2537	{
2538		if (strcmp(keywords[i], "scram_client_key") == 0)
2539		{
2540			if (values[i] != NULL && values[i][0] != '\0')
2541				has_scram_client_key = true;
2542			else
2543				has_scram_client_key = false;
2544		}
2545
2546		if (strcmp(keywords[i], "scram_server_key") == 0)
2547		{
2548			if (values[i] != NULL && values[i][0] != '\0')
2549				has_scram_server_key = true;
2550			else
2551				has_scram_server_key = false;
2552		}
2553
2554		if (strcmp(keywords[i], "require_auth") == 0)
2555		{
2556			if (values[i] != NULL && strcmp(values[i], "scram-sha-256") == 0)
2557				has_require_auth = true;
2558			else
2559				has_require_auth = false;
2560		}
2561	}
2562
2563	has_scram_keys = has_scram_client_key && has_scram_server_key && MyProcPort != NULL && MyProcPort->has_scram_keys;
2564
2565	return (has_scram_keys && has_require_auth);
