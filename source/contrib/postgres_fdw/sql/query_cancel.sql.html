6-- Let's test canceling a remote query.  Use a table that does not have
7-- remote_estimate enabled, else there will be multiple queries to the
8-- remote and we might unluckily send the cancel in between two of them.
9-- First let's confirm that the query is actually pushed down.
10EXPLAIN (VERBOSE, COSTS OFF)
11SELECT count(*) FROM ft1 a CROSS JOIN ft1 b CROSS JOIN ft1 c CROSS JOIN ft1 d;
12
13BEGIN;
14-- Make sure that connection is open and set up.
15SELECT count(*) FROM ft1 a;
16-- On most machines, 10ms will be enough to be sure that we've sent the slow
17-- query.  We may sometimes exercise the race condition where we send cancel
18-- before the remote side starts the query, but that's fine too.
19SET LOCAL statement_timeout = '10ms';
20-- This would take very long if not canceled:
21SELECT count(*) FROM ft1 a CROSS JOIN ft1 b CROSS JOIN ft1 c CROSS JOIN ft1 d;
22COMMIT;
