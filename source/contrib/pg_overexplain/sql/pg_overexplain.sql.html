22-- Create a partitioned table.
23CREATE TABLE vegetables (id serial, name text, genus text)
24PARTITION BY LIST (genus);
25CREATE TABLE daucus PARTITION OF vegetables FOR VALUES IN ('daucus');
26CREATE TABLE brassica PARTITION OF vegetables FOR VALUES IN ('brassica');

31		   ('rutabaga', 'brassica'), ('turnip', 'brassica');
32VACUUM ANALYZE vegetables;

34-- We filter relation OIDs out of the test output in order to avoid
35-- test instability. This is currently only needed for EXPLAIN (DEBUG), not
36-- EXPLAIN (RANGE_TABLE). Also suppress actual row counts, which are not
37-- stable (e.g. 1/8 is 0.12 on some buildfarm machines and 0.13 on others).
38CREATE FUNCTION explain_filter(text) RETURNS SETOF text
39LANGUAGE plpgsql AS
40$$
41DECLARE
42    ln text;
43BEGIN
44    FOR ln IN EXECUTE $1
45	LOOP
46		ln := regexp_replace(ln, 'Relation OIDs:( \m\d+\M)+',
47								 'Relation OIDs: NNN...', 'g');
48		ln := regexp_replace(ln, '<Relation-OIDs>( ?\m\d+\M)+</Relation-OIDs>',
49								 '<Relation-OIDs>NNN...</Relation-OIDs>', 'g');
50		ln := regexp_replace(ln, 'actual rows=\d+\.\d+',
51								 'actual rows=N.NN', 'g');
52		RETURN NEXT ln;
53	END LOOP;
54END;
55$$;

94-- Create an index, and then attempt to force a nested loop with inner index
95-- scan so that we can see parameter-related information. Also, let's try
