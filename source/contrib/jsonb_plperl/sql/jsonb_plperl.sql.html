73CREATE FUNCTION roundtrip(val jsonb, ref text = '') RETURNS jsonb
74LANGUAGE plperl
75TRANSFORM FOR TYPE jsonb
76AS $$
77# can't use Data::Dumper, but let's at least check for unexpected ref type
78die 'unexpected '.(ref($_[0]) || 'not a').' reference'
79    if ref($_[0]) ne $_[1];
80return $_[0];
81$$;
82
