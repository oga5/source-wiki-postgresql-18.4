111-- Test that parallel index build produces the same BRIN index as serial build.
112CREATE TABLE brin_parallel_test (a int, b text, c bigint) WITH (fillfactor=40);
113-- Generate a table with a mix of NULLs and non-NULL values (and data suitable
114-- for the different opclasses we build later).
115INSERT INTO brin_parallel_test
116SELECT (CASE WHEN (mod(i,231) = 0) OR (i BETWEEN 3500 AND 4000) THEN NULL ELSE i END),
117       (CASE WHEN (mod(i,233) = 0) OR (i BETWEEN 3750 AND 4250) THEN NULL ELSE encode(sha256(i::text::bytea), 'hex') END),
118       (CASE WHEN (mod(i,233) = 0) OR (i BETWEEN 3850 AND 4500) THEN NULL ELSE (i/100) + mod(i,8) END)
119  FROM generate_series(1,5000) S(i);
120-- Build an index with different opclasses - minmax, bloom and minmax-multi.
121--
122-- For minmax and opclass this is simple, but for minmax-multi we need to be
123-- careful, because the result depends on the order in which values are added
124-- to the summary, which in turn affects how are values merged etc. The order
125-- of merging results from workers has similar effect. All those summaries
126-- should produce correct query results, but it means we can't compare them
127-- using equality (which is what EXCEPT does). To work around this issue, we
128-- generated the data to only have very small number of distinct values per
129-- range, so that no merging is needed. This makes the results deterministic.
130-- build index without parallelism
131SET max_parallel_maintenance_workers = 0;
132CREATE INDEX brin_test_serial_idx ON brin_parallel_test
133 USING brin (a int4_minmax_ops, a int4_bloom_ops, b, c int8_minmax_multi_ops)
134  WITH (pages_per_range=7)
135 WHERE NOT (a BETWEEN 1000 and 1500);
136-- build index using parallelism
137--
138-- Set a couple parameters to force parallel build for small table. There's a
139-- requirement for table size, so disable that. Also, plan_create_index_workers
140-- assumes each worker will use work_mem=32MB for sorting (which works for btree,
141-- but not really for BRIN), so we set maintenance_work_mem for 4 workers.
142SET min_parallel_table_scan_size = 0;
143SET max_parallel_maintenance_workers = 4;
144SET maintenance_work_mem = '128MB';
145CREATE INDEX brin_test_parallel_idx ON brin_parallel_test
146 USING brin (a int4_minmax_ops, a int4_bloom_ops, b, c int8_minmax_multi_ops)
147  WITH (pages_per_range=7)
148 WHERE NOT (a BETWEEN 1000 and 1500);
149SELECT relname, relpages
150  FROM pg_class
151 WHERE relname IN ('brin_test_serial_idx', 'brin_test_parallel_idx')
152 ORDER BY relname;
