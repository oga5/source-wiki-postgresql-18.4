3-- Use a temp table so that effects of VACUUM are predictable
4CREATE TEMP TABLE test1 (a int, b int);
5INSERT INTO test1 VALUES (16777217, 131584);

7VACUUM (DISABLE_PAGE_SKIPPING) test1;  -- set up FSM

45-- tests for decoding of combined flags

52SELECT * FROM heap_tuple_infomask_flags(x'C000'::int, 0);

54-- test all flags of t_infomask and t_infomask2
55SELECT unnest(raw_flags)
56  FROM heap_tuple_infomask_flags(x'FFFF'::int, x'FFFF'::int) ORDER BY 1;
57SELECT unnest(combined_flags)
58  FROM heap_tuple_infomask_flags(x'FFFF'::int, x'FFFF'::int) ORDER BY 1;

67-- check that using any of these functions with a partitioned table or index
68-- would fail
69create table test_partitioned (a int) partition by range (a);
70create index test_partitioned_index on test_partitioned (a);
71select get_raw_page('test_partitioned', 0); -- error about partitioned table
72select get_raw_page('test_partitioned_index', 0); -- error about partitioned index
73
74-- a regular table which is a member of a partition set should work though
75create table test_part1 partition of test_partitioned for values from ( 1 ) to (100);
76select get_raw_page('test_part1', 0); -- get farther and error about empty table
77drop table test_partitioned;
