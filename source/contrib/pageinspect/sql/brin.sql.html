58-- Test that parallel index build produces the same BRIN index as serial build.
59CREATE TABLE brin_parallel_test (a int, b text, c bigint) WITH (fillfactor=40);
60
61-- Generate a table with a mix of NULLs and non-NULL values (and data suitable
62-- for the different opclasses we build later).
63INSERT INTO brin_parallel_test
64SELECT (CASE WHEN (mod(i,231) = 0) OR (i BETWEEN 3500 AND 4000) THEN NULL ELSE i END),
65       (CASE WHEN (mod(i,233) = 0) OR (i BETWEEN 3750 AND 4250) THEN NULL ELSE encode(sha256(i::text::bytea), 'hex') END),
66       (CASE WHEN (mod(i,233) = 0) OR (i BETWEEN 3850 AND 4500) THEN NULL ELSE (i/100) + mod(i,8) END)
67  FROM generate_series(1,5000) S(i);
68
69-- Build an index with different opclasses - minmax, bloom and minmax-multi.
70--
71-- For minmax and opclass this is simple, but for minmax-multi we need to be
72-- careful, because the result depends on the order in which values are added
73-- to the summary, which in turn affects how are values merged etc. The order
74-- of merging results from workers has similar effect. All those summaries
75-- should produce correct query results, but it means we can't compare them
76-- using equality (which is what EXCEPT does). To work around this issue, we
77-- generated the data to only have very small number of distinct values per
78-- range, so that no merging is needed. This makes the results deterministic.
79
80-- build index without parallelism
81SET max_parallel_maintenance_workers = 0;
82CREATE INDEX brin_test_serial_idx ON brin_parallel_test
83 USING brin (a int4_minmax_ops, a int4_bloom_ops, b, c int8_minmax_multi_ops)
84  WITH (pages_per_range=7)
85 WHERE NOT (a BETWEEN 1000 and 1500);
86
87-- build index using parallelism
88--
89-- Set a couple parameters to force parallel build for small table. There's a
90-- requirement for table size, so disable that. Also, plan_create_index_workers
91-- assumes each worker will use work_mem=32MB for sorting (which works for btree,
92-- but not really for BRIN), so we set maintenance_work_mem for 4 workers.
93SET min_parallel_table_scan_size = 0;
94SET max_parallel_maintenance_workers = 4;
95SET maintenance_work_mem = '128MB';
96CREATE INDEX brin_test_parallel_idx ON brin_parallel_test
97 USING brin (a int4_minmax_ops, a int4_bloom_ops, b, c int8_minmax_multi_ops)
98  WITH (pages_per_range=7)
99 WHERE NOT (a BETWEEN 1000 and 1500);
100
101SELECT relname, relpages
102  FROM pg_class
103 WHERE relname IN ('brin_test_serial_idx', 'brin_test_parallel_idx')
104 ORDER BY relname;
105
106-- Check that (A except B) and (B except A) is empty, which means the indexes
107-- are the same.
108
109SELECT * FROM brin_page_items(get_raw_page('brin_test_parallel_idx', 2), 'brin_test_parallel_idx')
110EXCEPT
111SELECT * FROM brin_page_items(get_raw_page('brin_test_serial_idx', 2), 'brin_test_serial_idx');
112
113SELECT * FROM brin_page_items(get_raw_page('brin_test_serial_idx', 2), 'brin_test_serial_idx')
114EXCEPT
115SELECT * FROM brin_page_items(get_raw_page('brin_test_parallel_idx', 2), 'brin_test_parallel_idx');
116
117DROP INDEX brin_test_parallel_idx;
118
119-- force parallel build, but don't allow starting parallel workers to force
120-- fallback to serial build, and repeat the checks
121
122SET max_parallel_workers = 0;
123CREATE INDEX brin_test_parallel_idx ON brin_parallel_test
124 USING brin (a int4_minmax_ops, a int4_bloom_ops, b, c int8_minmax_multi_ops)
125  WITH (pages_per_range=7)
126 WHERE NOT (a BETWEEN 1000 and 1500);

129  FROM pg_class
