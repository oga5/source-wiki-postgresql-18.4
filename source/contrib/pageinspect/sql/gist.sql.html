1-- The gist_page_opaque_info() function prints the page's LSN.
2-- Use an unlogged index, so that the LSN is predictable.
3CREATE UNLOGGED TABLE test_gist AS SELECT point(i,i) p, i::text t FROM
4    generate_series(1,1000) i;
5CREATE INDEX test_gist_idx ON test_gist USING gist (p);
6
7-- Page 0 is the root, the rest are leaf pages
8SELECT * FROM gist_page_opaque_info(get_raw_page('test_gist_idx', 0));
9SELECT * FROM gist_page_opaque_info(get_raw_page('test_gist_idx', 1));
10SELECT * FROM gist_page_opaque_info(get_raw_page('test_gist_idx', 2));
11
12SELECT * FROM gist_page_items(get_raw_page('test_gist_idx', 0), 'test_gist_idx');
13SELECT * FROM gist_page_items(get_raw_page('test_gist_idx', 1), 'test_gist_idx') LIMIT 5;
14
15-- gist_page_items_bytea prints the raw key data as a bytea. The output of that is
16-- platform-dependent (endianness), so omit the actual key data from the output.
17SELECT itemoffset, ctid, itemlen FROM gist_page_items_bytea(get_raw_page('test_gist_idx', 0));
18
19-- Suppress the DETAIL message, to allow the tests to work across various
20-- page sizes and architectures.
21\set VERBOSITY terse
22
23-- Failures with non-GiST index.
24CREATE INDEX test_gist_btree on test_gist(t);
25SELECT gist_page_items(get_raw_page('test_gist_btree', 0), 'test_gist_btree');
26SELECT gist_page_items(get_raw_page('test_gist_btree', 0), 'test_gist_idx');
27
28-- Failure with various modes.
29-- invalid page size
30SELECT gist_page_items_bytea('aaa'::bytea);
31SELECT gist_page_items('aaa'::bytea, 'test_gist_idx'::regclass);
32SELECT gist_page_opaque_info('aaa'::bytea);
33-- invalid special area size
34SELECT * FROM gist_page_opaque_info(get_raw_page('test_gist', 0));
35SELECT gist_page_items_bytea(get_raw_page('test_gist', 0));
36SELECT gist_page_items_bytea(get_raw_page('test_gist_btree', 0));
37\set VERBOSITY default
38
39-- Tests with all-zero pages.
40SHOW block_size \gset
41SELECT gist_page_items_bytea(decode(repeat('00', :block_size), 'hex'));
42SELECT gist_page_items(decode(repeat('00', :block_size), 'hex'), 'test_gist_idx'::regclass);
43SELECT gist_page_opaque_info(decode(repeat('00', :block_size), 'hex'));
44
45-- Test gist_page_items with included columns.
46-- Non-leaf pages contain only the key attributes, and leaf pages contain
47-- the included attributes.
48ALTER TABLE test_gist ADD COLUMN i int DEFAULT NULL;
49CREATE INDEX test_gist_idx_inc ON test_gist
50  USING gist (p) INCLUDE (t, i);
51-- Mask the value of the key attribute to avoid alignment issues.
52SELECT regexp_replace(keys, '\(p\)=\("(.*?)"\)', '(p)=("<val>")') AS keys_nonleaf_1
53  FROM gist_page_items(get_raw_page('test_gist_idx_inc', 0), 'test_gist_idx_inc')
54  WHERE itemoffset = 1;
55SELECT keys AS keys_leaf_1
56  FROM gist_page_items(get_raw_page('test_gist_idx_inc', 1), 'test_gist_idx_inc')
57  WHERE itemoffset = 1;
58
59DROP TABLE test_gist;
