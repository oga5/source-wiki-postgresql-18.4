23-- check that using the module's functions with unsupported relations will fail
24--
25
26-- partitioned tables (the parent ones) don't have visibility maps
27create table test_partitioned (a int) partition by list (a);
28-- these should all fail
29select pg_visibility('test_partitioned', 0);
30select pg_visibility_map('test_partitioned');
31select pg_visibility_map_summary('test_partitioned');
32select pg_check_frozen('test_partitioned');
33select pg_truncate_visibility_map('test_partitioned');
34
35create table test_partition partition of test_partitioned for values in (1);

70-- check some of the allowed relkinds
71create table regular_table (a int, b text);
72alter table regular_table alter column b set storage external;
73insert into regular_table values (1, repeat('one', 1000)), (2, repeat('two', 1000));
74vacuum (disable_page_skipping) regular_table;
75select count(*) > 0 from pg_visibility('regular_table');
76select count(*) > 0 from pg_visibility((select reltoastrelid from pg_class where relname = 'regular_table'));
77truncate regular_table;
