89#define GV_WAITVAL 0
90#define GV_INVAL 1
91#define GV_INESCVAL 2
92#define GV_WAITESCIN 3
93#define GV_WAITESCESCIN 4

95static bool
96get_val(HSParser *state, bool ignoreeq, bool *escaped)
97{
98	int			st = GV_WAITVAL;
99
100	state->wordlen = 32;
101	state->cur = state->word = palloc(state->wordlen);
102	*escaped = false;
103
104	while (1)
105	{
106		if (st == GV_WAITVAL)
107		{
108			if (*(state->ptr) == '"')
109			{
110				*escaped = true;
111				st = GV_INESCVAL;
112			}
113			else if (*(state->ptr) == '\0')
114			{
115				return false;
116			}
117			else if (*(state->ptr) == '=' && !ignoreeq)
118			{
119				PRSSYNTAXERROR;
120			}
121			else if (*(state->ptr) == '\\')
122			{
123				st = GV_WAITESCIN;
124			}
125			else if (!scanner_isspace((unsigned char) *(state->ptr)))
126			{
127				*(state->cur) = *(state->ptr);
128				state->cur++;
129				st = GV_INVAL;
130			}
131		}
132		else if (st == GV_INVAL)
133		{
134			if (*(state->ptr) == '\\')
135			{
136				st = GV_WAITESCIN;
137			}
138			else if (*(state->ptr) == '=' && !ignoreeq)
139			{
140				state->ptr--;
141				return true;
142			}
143			else if (*(state->ptr) == ',' && ignoreeq)
144			{
145				state->ptr--;
146				return true;
147			}
148			else if (scanner_isspace((unsigned char) *(state->ptr)))
149			{
150				return true;
151			}
152			else if (*(state->ptr) == '\0')
153			{
154				state->ptr--;
155				return true;
156			}
157			else
158			{
159				RESIZEPRSBUF;
160				*(state->cur) = *(state->ptr);
161				state->cur++;
162			}
163		}
164		else if (st == GV_INESCVAL)
165		{
166			if (*(state->ptr) == '\\')
167			{
168				st = GV_WAITESCESCIN;
169			}
170			else if (*(state->ptr) == '"')
171			{
172				return true;
173			}
174			else if (*(state->ptr) == '\0')
175			{
176				PRSEOF;
177			}
178			else
179			{
180				RESIZEPRSBUF;
181				*(state->cur) = *(state->ptr);
182				state->cur++;
183			}
184		}
185		else if (st == GV_WAITESCIN)
186		{
187			if (*(state->ptr) == '\0')
188				PRSEOF;
189			RESIZEPRSBUF;
190			*(state->cur) = *(state->ptr);
191			state->cur++;
192			st = GV_INVAL;
193		}
194		else if (st == GV_WAITESCESCIN)
195		{
196			if (*(state->ptr) == '\0')
197				PRSEOF;
198			RESIZEPRSBUF;
199			*(state->cur) = *(state->ptr);
200			state->cur++;
201			st = GV_INESCVAL;
202		}
203		else
204			elog(ERROR, "unrecognized get_val state: %d", st);
205
206		state->ptr++;
207	}
208}

217static bool
218parse_hstore(HSParser *state)
219{
220	int			st = WKEY;
221	bool		escaped = false;
222
223	state->plen = 16;
224	state->pairs = (Pairs *) palloc(sizeof(Pairs) * state->plen);
225	state->pcur = 0;
226	state->ptr = state->begin;
227	state->word = NULL;
228
229	while (1)
230	{
231		if (st == WKEY)
232		{
233			if (!get_val(state, false, &escaped))
234			{
235				if (SOFT_ERROR_OCCURRED(state->escontext))
236					return false;
237				return true;	/* EOF, all okay */ view japanese
238			}
239			if (state->pcur >= state->plen)
240			{
241				state->plen *= 2;
242				state->pairs = (Pairs *) repalloc(state->pairs, sizeof(Pairs) * state->plen);
243			}
244			if (!hstoreCheckKeyLength(state->cur - state->word, state))
245				return false;
246			state->pairs[state->pcur].key = state->word;
247			state->pairs[state->pcur].keylen = state->cur - state->word;
248			state->pairs[state->pcur].val = NULL;
249			state->word = NULL;
250			st = WEQ;
251		}
252		else if (st == WEQ)
253		{
254			if (*(state->ptr) == '=')
255			{
256				st = WGT;
257			}
258			else if (*(state->ptr) == '\0')
259			{
260				PRSEOF;
261			}
262			else if (!scanner_isspace((unsigned char) *(state->ptr)))
263			{
264				PRSSYNTAXERROR;
265			}
266		}
267		else if (st == WGT)
268		{
269			if (*(state->ptr) == '>')
270			{
271				st = WVAL;
272			}
273			else if (*(state->ptr) == '\0')
274			{
275				PRSEOF;
276			}
277			else
278			{
279				PRSSYNTAXERROR;
280			}
281		}
282		else if (st == WVAL)
283		{
284			if (!get_val(state, true, &escaped))
285			{
286				if (SOFT_ERROR_OCCURRED(state->escontext))
287					return false;
288				PRSEOF;
289			}
290			if (!hstoreCheckValLength(state->cur - state->word, state))
291				return false;
292			state->pairs[state->pcur].val = state->word;
293			state->pairs[state->pcur].vallen = state->cur - state->word;
294			state->pairs[state->pcur].isnull = false;
295			state->pairs[state->pcur].needfree = true;
296			if (state->cur - state->word == 4 && !escaped)
297			{
298				state->word[4] = '\0';
299				if (pg_strcasecmp(state->word, "null") == 0)
300					state->pairs[state->pcur].isnull = true;
301			}
302			state->word = NULL;
303			state->pcur++;
304			st = WDEL;
305		}
306		else if (st == WDEL)
307		{
308			if (*(state->ptr) == ',')
309			{
310				st = WKEY;
311			}
312			else if (*(state->ptr) == '\0')
313			{
314				return true;
315			}
316			else if (!scanner_isspace((unsigned char) *(state->ptr)))
317			{
318				PRSSYNTAXERROR;
319			}
320		}
321		else
322			elog(ERROR, "unrecognized parse_hstore state: %d", st);
323
324		state->ptr++;
325	}
326}

328static int
329comparePairs(const void *a, const void *b)
330{
331	const Pairs *pa = a;
332	const Pairs *pb = b;
333
334	if (pa->keylen == pb->keylen)
335	{
336		int			res = memcmp(pa->key, pb->key, pa->keylen);
337
338		if (res)
339			return res;
340
341		/* guarantee that needfree will be later */ view japanese
342		if (pb->needfree == pa->needfree)
343			return 0;
344		else if (pa->needfree)
345			return 1;
346		else
347			return -1;
348	}
349	return (pa->keylen > pb->keylen) ? 1 : -1;
350}
351
352/*
353 * this code still respects pairs.needfree, even though in general
354 * it should never be called in a context where anything needs freeing.
355 * we keep it because (a) those calls are in a rare code path anyway,
356 * and (b) who knows whether they might be needed by some caller.
357 */ view japanese
358int
359hstoreUniquePairs(Pairs *a, int32 l, int32 *buflen)
360{
361	Pairs	   *ptr,
362			   *res;
363
364	*buflen = 0;
365	if (l < 2)
366	{
367		if (l == 1)
368			*buflen = a->keylen + ((a->isnull) ? 0 : a->vallen);
369		return l;
370	}
371
372	qsort(a, l, sizeof(Pairs), comparePairs);
373
374	/*
375	 * We can't use qunique here because we have some clean-up code to run on
376	 * removed elements.
377	 */ view japanese
378	ptr = a + 1;
379	res = a;
380	while (ptr - a < l)
381	{
382		if (ptr->keylen == res->keylen &&
383			memcmp(ptr->key, res->key, res->keylen) == 0)
384		{
385			if (ptr->needfree)
386			{
387				pfree(ptr->key);
388				if (ptr->val != NULL)
389					pfree(ptr->val);
390			}
391		}
392		else
393		{
394			*buflen += res->keylen + ((res->isnull) ? 0 : res->vallen);
395			res++;
396			if (res != ptr)
397				memcpy(res, ptr, sizeof(Pairs));
398		}
399
400		ptr++;
401	}
402
403	*buflen += res->keylen + ((res->isnull) ? 0 : res->vallen);
404	return res + 1 - a;
405}

448HStore *
449hstorePairs(Pairs *pairs, int32 pcount, int32 buflen)
450{
451	HStore	   *out;
452	HEntry	   *entry;
453	char	   *ptr;
454	char	   *buf;
455	int32		len;
456	int32		i;
457
458	len = CALCDATASIZE(pcount, buflen);
459	out = palloc(len);
460	SET_VARSIZE(out, len);
461	HS_SETCOUNT(out, pcount);
462
463	if (pcount == 0)
464		return out;
465
466	entry = ARRPTR(out);
467	buf = ptr = STRPTR(out);
468
469	for (i = 0; i < pcount; i++)
470		HS_ADDITEM(entry, buf, ptr, pairs[i]);
471
472	HS_FINALIZE(out, pcount, buf, ptr);
473
474	return out;
475}

503Datum
504hstore_recv(PG_FUNCTION_ARGS)
505{
506	int32		buflen;
507	HStore	   *out;
508	Pairs	   *pairs;
509	int32		i;
510	int32		pcount;
511	StringInfo	buf = (StringInfo) PG_GETARG_POINTER(0);
512
513	pcount = pq_getmsgint(buf, 4);
514
515	if (pcount == 0)
516	{
517		out = hstorePairs(NULL, 0, 0);
518		PG_RETURN_POINTER(out);
519	}
520
521	if (pcount < 0 || pcount > MaxAllocSize / sizeof(Pairs))
522		ereport(ERROR,
523				(errcode(ERRCODE_PROGRAM_LIMIT_EXCEEDED),
524				 errmsg("number of pairs (%d) exceeds the maximum allowed (%d)",
525						pcount, (int) (MaxAllocSize / sizeof(Pairs)))));
526	pairs = palloc(pcount * sizeof(Pairs));
527
528	for (i = 0; i < pcount; ++i)
529	{
530		int			rawlen = pq_getmsgint(buf, 4);
531		int			len;
532
533		if (rawlen < 0)
534			ereport(ERROR,
535					(errcode(ERRCODE_NULL_VALUE_NOT_ALLOWED),
536					 errmsg("null value not allowed for hstore key")));
537
538		pairs[i].key = pq_getmsgtext(buf, rawlen, &len);
539		pairs[i].keylen = hstoreCheckKeyLen(len);
540		pairs[i].needfree = true;
541
542		rawlen = pq_getmsgint(buf, 4);
543		if (rawlen < 0)
544		{
545			pairs[i].val = NULL;
546			pairs[i].vallen = 0;
547			pairs[i].isnull = true;
548		}
549		else
550		{
551			pairs[i].val = pq_getmsgtext(buf, rawlen, &len);
552			pairs[i].vallen = hstoreCheckValLen(len);
553			pairs[i].isnull = false;
554		}
555	}
556
557	pcount = hstoreUniquePairs(pairs, pcount, &buflen);
558
559	out = hstorePairs(pairs, pcount, buflen);
560
561	PG_RETURN_POINTER(out);
562}

601PG_FUNCTION_INFO_V1(hstore_from_arrays);
602Datum
603hstore_from_arrays(PG_FUNCTION_ARGS)
604{
605	int32		buflen;
606	HStore	   *out;
607	Pairs	   *pairs;
608	Datum	   *key_datums;
609	bool	   *key_nulls;
610	int			key_count;
611	Datum	   *value_datums;
612	bool	   *value_nulls;
613	int			value_count;
614	ArrayType  *key_array;
615	ArrayType  *value_array;
616	int			i;
617
618	if (PG_ARGISNULL(0))
619		PG_RETURN_NULL();
620
621	key_array = PG_GETARG_ARRAYTYPE_P(0);
622
623	Assert(ARR_ELEMTYPE(key_array) == TEXTOID);
624
625	/*
626	 * must check >1 rather than != 1 because empty arrays have 0 dimensions,
627	 * not 1
628	 */ view japanese
629
630	if (ARR_NDIM(key_array) > 1)
631		ereport(ERROR,
632				(errcode(ERRCODE_ARRAY_SUBSCRIPT_ERROR),
633				 errmsg("wrong number of array subscripts")));
634
635	deconstruct_array_builtin(key_array, TEXTOID, &key_datums, &key_nulls, &key_count);
636
637	/* see discussion in hstoreArrayToPairs() */
638	if (key_count > MaxAllocSize / sizeof(Pairs))
639		ereport(ERROR,
640				(errcode(ERRCODE_PROGRAM_LIMIT_EXCEEDED),
641				 errmsg("number of pairs (%d) exceeds the maximum allowed (%d)",
642						key_count, (int) (MaxAllocSize / sizeof(Pairs)))));
643
644	/* value_array might be NULL */ view japanese
645
646	if (PG_ARGISNULL(1))
647	{
648		value_array = NULL;
649		value_count = key_count;
650		value_datums = NULL;
651		value_nulls = NULL;
652	}
653	else
654	{
655		value_array = PG_GETARG_ARRAYTYPE_P(1);
656
657		Assert(ARR_ELEMTYPE(value_array) == TEXTOID);
658
659		if (ARR_NDIM(value_array) > 1)
660			ereport(ERROR,
661					(errcode(ERRCODE_ARRAY_SUBSCRIPT_ERROR),
662					 errmsg("wrong number of array subscripts")));
663
664		if ((ARR_NDIM(key_array) > 0 || ARR_NDIM(value_array) > 0) &&
665			(ARR_NDIM(key_array) != ARR_NDIM(value_array) ||
666			 ARR_DIMS(key_array)[0] != ARR_DIMS(value_array)[0] ||
667			 ARR_LBOUND(key_array)[0] != ARR_LBOUND(value_array)[0]))
668			ereport(ERROR,
669					(errcode(ERRCODE_ARRAY_SUBSCRIPT_ERROR),
670					 errmsg("arrays must have same bounds")));
671
672		deconstruct_array_builtin(value_array, TEXTOID, &value_datums, &value_nulls, &value_count);
673
674		Assert(key_count == value_count);
675	}
676
677	pairs = palloc(key_count * sizeof(Pairs));
678
679	for (i = 0; i < key_count; ++i)
680	{
681		if (key_nulls[i])
682			ereport(ERROR,
683					(errcode(ERRCODE_NULL_VALUE_NOT_ALLOWED),
684					 errmsg("null value not allowed for hstore key")));
685
686		if (!value_nulls || value_nulls[i])
687		{
688			pairs[i].key = VARDATA(key_datums[i]);
689			pairs[i].val = NULL;
690			pairs[i].keylen =
691				hstoreCheckKeyLen(VARSIZE(key_datums[i]) - VARHDRSZ);
692			pairs[i].vallen = 4;
693			pairs[i].isnull = true;
694			pairs[i].needfree = false;
695		}
696		else
697		{
698			pairs[i].key = VARDATA(key_datums[i]);
699			pairs[i].val = VARDATA(value_datums[i]);
700			pairs[i].keylen =
701				hstoreCheckKeyLen(VARSIZE(key_datums[i]) - VARHDRSZ);
702			pairs[i].vallen =
703				hstoreCheckValLen(VARSIZE(value_datums[i]) - VARHDRSZ);
704			pairs[i].isnull = false;
705			pairs[i].needfree = false;
706		}
707	}
708
709	key_count = hstoreUniquePairs(pairs, key_count, &buflen);
710
711	out = hstorePairs(pairs, key_count, buflen);
712
713	PG_RETURN_POINTER(out);
714}

718Datum
719hstore_from_array(PG_FUNCTION_ARGS)
720{
721	ArrayType  *in_array = PG_GETARG_ARRAYTYPE_P(0);
722	int			ndims = ARR_NDIM(in_array);
723	int			count;
724	int32		buflen;
725	HStore	   *out;
726	Pairs	   *pairs;
727	Datum	   *in_datums;
728	bool	   *in_nulls;
729	int			in_count;
730	int			i;
731
732	Assert(ARR_ELEMTYPE(in_array) == TEXTOID);
733
734	switch (ndims)
735	{
736		case 0:
737			out = hstorePairs(NULL, 0, 0);

749				ereport(ERROR,
750						(errcode(ERRCODE_ARRAY_SUBSCRIPT_ERROR),
751						 errmsg("array must have two columns")));
752			break;

754		default:
755			ereport(ERROR,
756					(errcode(ERRCODE_ARRAY_SUBSCRIPT_ERROR),
757					 errmsg("wrong number of array subscripts")));
758	}

760	deconstruct_array_builtin(in_array, TEXTOID, &in_datums, &in_nulls, &in_count);

766		ereport(ERROR,
767				(errcode(ERRCODE_PROGRAM_LIMIT_EXCEEDED),
768				 errmsg("number of pairs (%d) exceeds the maximum allowed (%d)",
769						count, (int) (MaxAllocSize / sizeof(Pairs)))));

771	pairs = palloc(count * sizeof(Pairs));

773	for (i = 0; i < count; ++i)
774	{
775		if (in_nulls[i * 2])
776			ereport(ERROR,
777					(errcode(ERRCODE_NULL_VALUE_NOT_ALLOWED),
778					 errmsg("null value not allowed for hstore key")));
779
780		if (in_nulls[i * 2 + 1])
781		{
782			pairs[i].key = VARDATA(in_datums[i * 2]);
783			pairs[i].val = NULL;
784			pairs[i].keylen =
785				hstoreCheckKeyLen(VARSIZE(in_datums[i * 2]) - VARHDRSZ);
786			pairs[i].vallen = 4;
787			pairs[i].isnull = true;
788			pairs[i].needfree = false;
789		}
790		else
791		{
792			pairs[i].key = VARDATA(in_datums[i * 2]);
793			pairs[i].val = VARDATA(in_datums[i * 2 + 1]);
794			pairs[i].keylen =
795				hstoreCheckKeyLen(VARSIZE(in_datums[i * 2]) - VARHDRSZ);
796			pairs[i].vallen =
797				hstoreCheckValLen(VARSIZE(in_datums[i * 2 + 1]) - VARHDRSZ);
798			pairs[i].isnull = false;
799			pairs[i].needfree = false;
800		}
801	}
802
803	count = hstoreUniquePairs(pairs, count, &buflen);
804
805	out = hstorePairs(pairs, count, buflen);
806
807	PG_RETURN_POINTER(out);
808}
