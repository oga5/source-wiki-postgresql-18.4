222#
223# Rule to make MCS policy work on regression test
224#
225# NOTE: MCS (multi category security) policy was enabled by default, to
226# allow DAC style access control, in the previous selinux policy.
227# However, its definition was changed later, then a limited number of
228# applications are restricted by MCS policy, for container features
229# mainly. The rules below enables MCS policy for domains of regression
230# test also, even if base security policy does not apply. If base policy
231# is old and MCS is enabled in default, rules below does nothing.
232#
233optional_policy(`
234	gen_require(`
235		type sepgsql_trusted_proc_t;
236	')
237	mcs_constrained(sepgsql_regtest_dba_t)
238	mcs_constrained(sepgsql_regtest_superuser_t)
239	mcs_constrained(sepgsql_regtest_user_t)
240	mcs_constrained(sepgsql_regtest_nosuch_t)
241	mcs_constrained(sepgsql_trusted_proc_t)
242
243	mcs_constrained(sepgsql_regtest_pool_t)
244	mcs_constrained(sepgsql_regtest_foo_t)
245	mcs_constrained(sepgsql_regtest_var_t)
246')

248#
249# Rule to execute original trusted procedures
250#
251# These rules intends to allow any valid client types to launch trusted-
252# procedures (including ones causes domain transition to invalid domain)
253# being labeled as sepgsql_regtest_trusted_proc_exec_t and
254# sepgsql_nosuch_trusted_proc_exec_t.
255#
256optional_policy(`
257	gen_require(`
258		attribute sepgsql_client_type;
259	')
260	allow sepgsql_client_type { sepgsql_regtest_trusted_proc_exec_t sepgsql_nosuch_trusted_proc_exec_t }:db_procedure { getattr execute entrypoint };
