3# Common rules for Native Language Support (NLS)
4#
5# If some subdirectory of the source tree wants to provide NLS, it
6# needs to contain a file 'nls.mk' with the following make variable
7# assignments:
8#
9# CATALOG_NAME          -- name of the message catalog (xxx.po); probably
10#                          name of the program
11# GETTEXT_FILES         -- list of source files that contain message strings
12# GETTEXT_TRIGGERS      -- (optional) list of functions that contain
13#                          translatable strings
14# GETTEXT_FLAGS         -- (optional) list of gettext --flag arguments to mark
15#                          function arguments that contain C format strings
16#                          (functions must be listed in TRIGGERS and FLAGS)
17#
18# Also, provide a text file 'po/LINGUAS' with a space-separated list
19# of languages that are provided/supported.
20#
21# That's all, the rest is done here, if --enable-nls was specified.
22#
23# The only user-visible targets here are 'init-po', to make an initial
24# "blank" catalog from program sources, and 'update-po', which is to
25# be called if the messages in the program source have changed, in
26# order to merge the changes into the existing .po files.
27
28
29# existence checked by Makefile.global; otherwise we won't get here
30include $(srcdir)/nls.mk
31
32AVAIL_LANGUAGES := $(shell cat $(srcdir)/po/LINGUAS)
33
34# If user specified the languages he wants in --enable-nls=LANGUAGES,
35# filter out the rest.  Else use all available ones.
36ifdef WANTED_LANGUAGES
37LANGUAGES = $(filter $(WANTED_LANGUAGES), $(AVAIL_LANGUAGES))
38else
39LANGUAGES = $(AVAIL_LANGUAGES)
40endif
41
42PO_FILES = $(addprefix po/, $(addsuffix .po, $(LANGUAGES)))
43ALL_PO_FILES = $(addprefix po/, $(addsuffix .po, $(AVAIL_LANGUAGES)))
44MO_FILES = $(addprefix po/, $(addsuffix .mo, $(LANGUAGES)))
45
46ifdef XGETTEXT
47XGETTEXT += -ctranslator --copyright-holder='PostgreSQL Global Development Group' --msgid-bugs-address=pgsql-bugs@lists.postgresql.org --no-wrap --sort-by-file --package-name='$(CATALOG_NAME) (PostgreSQL)' --package-version='$(MAJORVERSION)'
48endif
49
50ifdef MSGMERGE
51MSGMERGE += --no-wrap --previous --sort-by-file
52endif
53
54# _ is defined in c.h, so it's global
55GETTEXT_TRIGGERS += _
56GETTEXT_FLAGS    += _:1:pass-c-format

73    errhint:1:c-format errhint_plural:1:c-format errhint_plural:2:c-format \

91all-po: $(MO_FILES)
92
93%.mo: %.po
94	$(MSGFMT) $(MSGFMT_FLAGS) -o $@ $<
95
96ifeq ($(word 1,$(GETTEXT_FILES)),+)
97po/$(CATALOG_NAME).pot: $(word 2, $(GETTEXT_FILES)) $(MAKEFILE_LIST)
98ifdef XGETTEXT
99	$(XGETTEXT) -D $(srcdir) -D . -n $(addprefix -k, $(GETTEXT_TRIGGERS)) $(addprefix --flag=, $(GETTEXT_FLAGS)) -f $<
100else
101	@echo "You don't have 'xgettext'."; exit 1
102endif
103else # GETTEXT_FILES
104po/$(CATALOG_NAME).pot: $(GETTEXT_FILES) $(MAKEFILE_LIST)
105# Change to srcdir explicitly, don't rely on $^.  That way we get
106# consistent #: file references in the po files.
107ifdef XGETTEXT
108	$(XGETTEXT) -D $(srcdir) -D . -n $(addprefix -k, $(GETTEXT_TRIGGERS)) $(addprefix --flag=, $(GETTEXT_FLAGS)) $(GETTEXT_FILES)
109else
110	@echo "You don't have 'xgettext'."; exit 1
111endif
112endif # GETTEXT_FILES
113	@$(MKDIR_P) $(dir $@)
114	sed -e '1,18 { s/SOME DESCRIPTIVE TITLE./LANGUAGE message translation file for $(CATALOG_NAME)/;s/PACKAGE/PostgreSQL/g;s/VERSION/$(MAJORVERSION)/g;s/YEAR/'`date +%Y`'/g; }' messages.po >$@
115	rm messages.po

118# catalog name extensions must match behavior of PG_TEXTDOMAIN() in c.h
119install-po: all-po installdirs-po
120ifneq (,$(LANGUAGES))
121	for lang in $(LANGUAGES); do \
122	  $(INSTALL_DATA) po/$$lang.mo '$(DESTDIR)$(localedir)'/$$lang/LC_MESSAGES/$(CATALOG_NAME)$(SO_MAJOR_VERSION)-$(MAJORVERSION).mo || exit 1; \
123	done
124endif
125
126installdirs-po:
127	$(if $(LANGUAGES),$(MKDIR_P) $(foreach lang, $(LANGUAGES), '$(DESTDIR)$(localedir)'/$(lang)/LC_MESSAGES),:)
128
129uninstall-po:
130	$(if $(LANGUAGES),rm -f $(foreach lang, $(LANGUAGES), '$(DESTDIR)$(localedir)'/$(lang)/LC_MESSAGES/$(CATALOG_NAME)$(SO_MAJOR_VERSION)-$(MAJORVERSION).mo),:)

133clean-po:
134	$(if $(MO_FILES),rm -f $(MO_FILES))
135	@$(if $(wildcard po/*.po.new),rm -f po/*.po.new)
136	rm -f po/$(CATALOG_NAME).pot

139init-po: po/$(CATALOG_NAME).pot
140
141
142# For performance reasons, only calculate these when the user actually
143# requested update-po or a specific file.
144ifneq (,$(filter update-po %.po.new,$(MAKECMDGOALS)))
145ifdef PGXS
146ALL_LANGUAGES := $(shell find . -name '*.po' -print | sed 's,^.*/\([^/]*\).po$$,\1,' | LC_ALL=C sort -u)
147all_compendia := $(shell find . -name '*.po' -print | LC_ALL=C sort)
148else
149ALL_LANGUAGES := $(shell find $(top_srcdir) -name '*.po' -print | sed 's,^.*/\([^/]*\).po$$,\1,' | LC_ALL=C sort -u)
150all_compendia := $(shell find $(top_srcdir) -name '*.po' -print | LC_ALL=C sort)
151endif
152else
153ALL_LANGUAGES = $(AVAIL_LANGUAGES)
154all_compendia = FORCE
155FORCE:
156endif
157
158ifdef WANTED_LANGUAGES
159ALL_LANGUAGES := $(filter $(WANTED_LANGUAGES), $(ALL_LANGUAGES))
160endif
161
162update-po: $(ALL_LANGUAGES:%=po/%.po.new)
163
164$(AVAIL_LANGUAGES:%=po/%.po.new): po/%.po.new: po/%.po po/$(CATALOG_NAME).pot $(all_compendia)
165	$(MSGMERGE) --lang=$* $(word 1, $^) $(word 2,$^) -o $@ $(addprefix --compendium=,$(filter %/$*.po,$(wordlist 3,$(words $^),$^)))
166
167# For languages not yet available, merge against oneself, to pick
168# up translations from the compendia.  (Merging against /dev/null
169# doesn't work so well; it inserts the headers from the first-named
170# compendium.)
171po/%.po.new: po/$(CATALOG_NAME).pot $(all_compendia)
172	$(MSGMERGE) --lang=$* $(word 1,$^) $(word 1,$^) -o $@ $(addprefix --compendium=,$(filter %/$*.po,$(wordlist 2,$(words $^),$^)))
