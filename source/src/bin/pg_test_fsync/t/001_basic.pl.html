4use strict;
5use warnings FATAL => 'all';

7use PostgreSQL::Test::Utils;
8use Test::More;

10#########################################
11# Basic checks
12
13program_help_ok('pg_test_fsync');
14program_version_ok('pg_test_fsync');
15program_options_handling_ok('pg_test_fsync');
16
17#########################################
18# Test invalid option combinations
19
20command_fails_like(
21	[ 'pg_test_fsync', '--secs-per-test' => 'a' ],
22	qr/\Qpg_test_fsync: error: invalid argument for option --secs-per-test\E/,
23	'pg_test_fsync: invalid argument for option --secs-per-test');
24command_fails_like(
25	[ 'pg_test_fsync', '--secs-per-test' => '0' ],
26	qr/\Qpg_test_fsync: error: --secs-per-test must be in range 1..4294967295\E/,
27	'pg_test_fsync: --secs-per-test must be in range');
28
29done_testing();
