18# Generate a database with a name made of a range of ASCII characters.
19# Extracted from 002_pg_upgrade.pl.
20sub generate_db
21{
22	my ($node, $prefix, $from_char, $to_char, $suffix) = @_;
23
24	my $dbname = $prefix;
25	for my $i ($from_char .. $to_char)
26	{
27		next if $i == 7 || $i == 10 || $i == 13;    # skip BEL, LF, and CR
28		$dbname = $dbname . sprintf('%c', $i);
29	}
30
31	$dbname .= $suffix;
32
33	# On Windows, older IPC::Run versions can mis-quote command line arguments
34	# containing double quote or backslash
35	$dbname =~ tr/\"\\//d if ($windows_os);
36
37	$node->command_ok(
38		[ 'createdb', $dbname ],
39		"created database with ASCII characters from $from_char to $to_char");
40
41	return $dbname;
42}

121# Set up node P as primary
122my $node_p = PostgreSQL::Test::Cluster->new('node_p');
123my $pconnstr = $node_p->connstr;
124$node_p->init(allows_streaming => 'logical');
125# Disable autovacuum to avoid generating xid during stats update as otherwise
126# the new XID could then be replicated to standby at some random point making
127# slots at primary lag behind standby during slot sync.
128$node_p->append_conf('postgresql.conf', 'autovacuum = off');
129$node_p->start;

131# Set up node F as about-to-fail node
132# Force it to initialize a new cluster instead of copying a
133# previously initdb'd cluster. New cluster has a different system identifier so
134# we can test if the target cluster is a copy of the source cluster.
135my $node_f = PostgreSQL::Test::Cluster->new('node_f');
136$node_f->init(force_initdb => 1, allows_streaming => 'logical');

303# Create failover slot to test its removal
304my $fslotname = 'failover_slot';
305$node_p->safe_psql($db1,
306	"SELECT pg_create_logical_replication_slot('$fslotname', 'pgoutput', false, false, true)"
307);
308$node_s->start;
309# Wait for the standby to catch up so that the standby is not lagging behind
310# the failover slot.
311$node_p->wait_for_replay_catchup($node_s);
312$node_s->safe_psql('postgres', "SELECT pg_sync_replication_slots()");
313my $result = $node_s->safe_psql('postgres',
314	"SELECT slot_name FROM pg_replication_slots WHERE slot_name = '$fslotname' AND synced AND NOT temporary"
315);
316is($result, 'failover_slot', 'failover slot is synced');

318# Insert another row on node P and wait node S to catch up. We
319# intentionally performed this insert after syncing logical slot
320# as otherwise the local slot's (created during synchronization of
321# slot) xmin on standby could be ahead of the remote slot leading
322# to failure in synchronization.
323$node_p->safe_psql($db1, "INSERT INTO tbl1 VALUES('second row')");
324$node_p->wait_for_replay_catchup($node_s);
325
326# Create subscription to test its removal
327my $dummy_sub = 'regress_sub_dummy';
328$node_p->safe_psql($db1,
329	"CREATE SUBSCRIPTION $dummy_sub CONNECTION 'dbname=dummy' PUBLICATION pub_dummy WITH (connect=false)"
330);
331$node_p->wait_for_replay_catchup($node_s);
332
333# Create user-defined publications, wait for streaming replication to sync them
334# to the standby, then verify that '--clean'
335# removes them.
336$node_p->safe_psql(
337	$db1, qq(
338	CREATE PUBLICATION test_pub1 FOR ALL TABLES;
339	CREATE PUBLICATION test_pub2 FOR ALL TABLES;
340));
341
342$node_p->wait_for_replay_catchup($node_s);
343
344is($node_s->safe_psql($db1, "SELECT COUNT(*) FROM pg_publication"),
345	'2', 'two pre-existing publications on subscriber');
