83# Simple test case: type something and see if psql responds as expected
84sub check_completion
85{
86	my ($send, $pattern, $annotation) = @_;
87
88	# report test failures from caller location
89	local $Test::Builder::Level = $Test::Builder::Level + 1;
90
91	# restart per-command timer
92	$h->{timeout}->start($PostgreSQL::Test::Utils::timeout_default);
93
94	# send the data to be sent and wait for its result
95	my $out = $h->query_until($pattern, $send);
96	my $okay = ($out =~ $pattern && !$h->{timeout}->is_expired);
97	ok($okay, $annotation);
98	# for debugging, log actual output if it didn't match
99	local $Data::Dumper::Terse = 1;
100	local $Data::Dumper::Useqq = 1;
101	diag 'Actual output was ' . Dumper($out) . "Did not match \"$pattern\"\n"
102	  if !$okay;
103	return;
104}

106# Clear query buffer to start over
107# (won't work if we are inside a string literal!)
108sub clear_query
109{
110	local $Test::Builder::Level = $Test::Builder::Level + 1;
111
112	check_completion("\\r\n", qr/Query buffer reset.*postgres=# /s,
113		"\\r works");
114	return;
115}
116
117# Clear current line to start over
118# (this will work in an incomplete string literal, but it's less desirable
119# than clear_query because we lose evidence in the history file)
120sub clear_line
121{
122	local $Test::Builder::Level = $Test::Builder::Level + 1;
123
124	check_completion("\025\n", qr/postgres=# /, "control-U works");
125	return;
126}
