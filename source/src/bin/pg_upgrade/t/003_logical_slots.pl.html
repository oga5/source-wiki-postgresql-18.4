3# Tests for upgrading logical replication slots
4
5use strict;
6use warnings FATAL => 'all';
7
8use File::Find qw(find);
9
10use PostgreSQL::Test::Cluster;
11use PostgreSQL::Test::Utils;
12use Test::More;
13
14# Can be changed to test the other modes
15my $mode = $ENV{PG_TEST_PG_UPGRADE_MODE} || '--copy';
16
17# Initialize old cluster
18my $oldpub = PostgreSQL::Test::Cluster->new('oldpub');
19$oldpub->init(allows_streaming => 'logical');
20$oldpub->append_conf('postgresql.conf', 'autovacuum = off');
21
22# Initialize new cluster
23my $newpub = PostgreSQL::Test::Cluster->new('newpub');
24$newpub->init(allows_streaming => 'logical');
25
26# During upgrade, when pg_restore performs CREATE DATABASE, bgwriter or
27# checkpointer may flush buffers and hold a file handle for the system table.
28# So, if later due to some reason we need to re-create the file with the same
29# name like a TRUNCATE command on the same table, then the command will fail
30# if OS (such as older Windows versions) doesn't remove an unlinked file
31# completely till it is open. The probability of seeing this behavior is
32# higher in this test because we use wal_level as logical via
33# allows_streaming => 'logical' which in turn set shared_buffers as 1MB.
34$newpub->append_conf(
35	'postgresql.conf', q{
36bgwriter_lru_maxpages = 0
37checkpoint_timeout = 1h
38});
39
40# Setup a common pg_upgrade command to be used by all the test cases
41my @pg_upgrade_cmd = (
42	'pg_upgrade', '--no-sync',
43	'--old-datadir' => $oldpub->data_dir,
44	'--new-datadir' => $newpub->data_dir,
45	'--old-bindir' => $oldpub->config_data('--bindir'),
46	'--new-bindir' => $newpub->config_data('--bindir'),
47	'--socketdir' => $newpub->host,
48	'--old-port' => $oldpub->port,
49	'--new-port' => $newpub->port,
50	$mode);
51
52# In a VPATH build, we'll be started in the source directory, but we want
53# to run pg_upgrade in the build directory so that any files generated finish
54# in it, like delete_old_cluster.{sh,bat}.
55chdir ${PostgreSQL::Test::Utils::tmp_check};
56
57# ------------------------------
58# TEST: Confirm pg_upgrade fails when the new cluster has wrong GUC values
59
60# Preparations for the subsequent test:
61# 1. Create two slots on the old cluster
62$oldpub->start;
63$oldpub->safe_psql(
64	'postgres', qq[
65	SELECT pg_create_logical_replication_slot('test_slot1', 'test_decoding');
66	SELECT pg_create_logical_replication_slot('test_slot2', 'test_decoding');
67]);
68$oldpub->stop();
69
70# 2. Set 'max_replication_slots' to be less than the number of slots (2)
71#	 present on the old cluster.
72$newpub->append_conf('postgresql.conf', "max_replication_slots = 1");
73
74# pg_upgrade will fail because the new cluster has insufficient
75# max_replication_slots
76command_checks_all(
77	[@pg_upgrade_cmd],
78	1,
79	[
80		qr/"max_replication_slots" \(1\) must be greater than or equal to the number of logical replication slots \(2\) on the old cluster/
81	],
82	[qr//],
83	'run of pg_upgrade where the new cluster has insufficient "max_replication_slots"'
84);
85ok(-d $newpub->data_dir . "/pg_upgrade_output.d",
86	"pg_upgrade_output.d/ not removed after pg_upgrade failure");
87
88# Set 'max_replication_slots' to match the number of slots (2) present on the
89# old cluster. Both slots will be used for subsequent tests.
90$newpub->append_conf('postgresql.conf', "max_replication_slots = 2");
91
92
93# ------------------------------
94# TEST: Confirm pg_upgrade fails when the slot still has unconsumed WAL records
95
96# Preparations for the subsequent test:
97# 1. Generate extra WAL records. At this point neither test_slot1 nor
98#	 test_slot2 has consumed them.
99#
100# 2. Advance the slot test_slot2 up to the current WAL location, but test_slot1
101#	 still has unconsumed WAL records.
102#
103# 3. Emit a non-transactional message. This will cause test_slot2 to detect the
104#	 unconsumed WAL record.
105$oldpub->start;
106$oldpub->safe_psql(
107	'postgres', qq[
108		CREATE TABLE tbl AS SELECT generate_series(1, 10) AS a;
109		SELECT pg_replication_slot_advance('test_slot2', pg_current_wal_lsn());
110		SELECT count(*) FROM pg_logical_emit_message('false', 'prefix', 'This is a non-transactional message');
111]);
112$oldpub->stop;
113
114# pg_upgrade will fail because there are slots still having unconsumed WAL
115# records
116command_checks_all(
117	[@pg_upgrade_cmd],
118	1,
119	[
120		qr/Your installation contains logical replication slots that cannot be upgraded./
121	],
122	[qr//],
123	'run of pg_upgrade of old cluster with slots having unconsumed WAL records'
124);
125
126# Verify the reason why the logical replication slot cannot be upgraded
127my $slots_filename;
128
129# Find a txt file that contains a list of logical replication slots that cannot
130# be upgraded. We cannot predict the file's path because the output directory
131# contains a milliseconds timestamp. File::Find::find must be used.
132find(
133	sub {
134		if ($File::Find::name =~ m/invalid_logical_slots\.txt/)
135		{
136			$slots_filename = $File::Find::name;
137		}
138	},
139	$newpub->data_dir . "/pg_upgrade_output.d");
140
141# Check the file content. Both slots should be reporting that they have
142# unconsumed WAL records.
143like(
144	slurp_file($slots_filename),
145	qr/The slot \"test_slot1\" has not consumed the WAL yet/m,
146	'the previous test failed due to unconsumed WALs');
147like(
148	slurp_file($slots_filename),
149	qr/The slot \"test_slot2\" has not consumed the WAL yet/m,
150	'the previous test failed due to unconsumed WALs');
151
152
153# ------------------------------
154# TEST: Successful upgrade
155
156# Preparations for the subsequent test:
157# 1. Setup logical replication (first, cleanup slots from the previous tests)
158my $old_connstr = $oldpub->connstr . ' dbname=postgres';
159
160$oldpub->start;
161$oldpub->safe_psql(
162	'postgres', qq[
163	SELECT * FROM pg_drop_replication_slot('test_slot1');
164	SELECT * FROM pg_drop_replication_slot('test_slot2');
165	CREATE PUBLICATION regress_pub FOR ALL TABLES;
166]);
167
168# Initialize subscriber cluster
169my $sub = PostgreSQL::Test::Cluster->new('sub');
170$sub->init();
171
172$sub->start;
173$sub->safe_psql(
174	'postgres', qq[
175	CREATE TABLE tbl (a int);
176	CREATE SUBSCRIPTION regress_sub CONNECTION '$old_connstr' PUBLICATION regress_pub WITH (two_phase = 'true', failover = 'true')
177]);
178$sub->wait_for_subscription_sync($oldpub, 'regress_sub');
179
180# Also wait for two-phase to be enabled
181my $twophase_query =
182  "SELECT count(1) = 0 FROM pg_subscription WHERE subtwophasestate NOT IN ('e');";
183$sub->poll_query_until('postgres', $twophase_query)
184  or die "Timed out while waiting for subscriber to enable twophase";
185
186# 2. Temporarily disable the subscription
187$sub->safe_psql('postgres', "ALTER SUBSCRIPTION regress_sub DISABLE");
188$oldpub->stop;
189
190# pg_upgrade should be successful
191command_ok([@pg_upgrade_cmd], 'run of pg_upgrade of old cluster');
192
193# Check that the slot 'regress_sub' has migrated to the new cluster
194$newpub->start;
195my $result = $newpub->safe_psql('postgres',
196	"SELECT slot_name, two_phase, failover FROM pg_replication_slots");
197is($result, qq(regress_sub|t|t), 'check the slot exists on new cluster');
198
199# Update the connection
200my $new_connstr = $newpub->connstr . ' dbname=postgres';
201$sub->safe_psql(
202	'postgres', qq[
203	ALTER SUBSCRIPTION regress_sub CONNECTION '$new_connstr';
204	ALTER SUBSCRIPTION regress_sub ENABLE;
205]);
206
207# Check whether changes on the new publisher get replicated to the subscriber
208$newpub->safe_psql('postgres',
209	"INSERT INTO tbl VALUES (generate_series(11, 20))");
210$newpub->wait_for_catchup('regress_sub');
211$result = $sub->safe_psql('postgres', "SELECT count(*) FROM tbl");
212is($result, qq(20), 'check changes are replicated to the sub');
213
214# Clean up
215$sub->stop();
216$newpub->stop();
217
218done_testing();
