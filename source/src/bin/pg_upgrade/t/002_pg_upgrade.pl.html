23# Generate a database with a name made of a range of ASCII characters.
24sub generate_db
25{
26	my ($node, $prefix, $from_char, $to_char, $suffix) = @_;
27
28	my $dbname = $prefix;
29	for my $i ($from_char .. $to_char)
30	{
31		next if $i == 7 || $i == 10 || $i == 13;    # skip BEL, LF, and CR
32		$dbname = $dbname . sprintf('%c', $i);
33	}
34
35	$dbname .= $suffix;
36	$node->command_ok(
37		[ 'createdb', $dbname ],
38		"created database with ASCII characters from $from_char to $to_char");
39}

41# Filter the contents of a dump before its use in a content comparison for
42# upgrade testing. This returns the path to the filtered dump.
43sub filter_dump
44{
45	my ($is_old, $old_version, $dump_file) = @_;
46	my $dump_contents = slurp_file($dump_file);
47
48	if ($is_old)
49	{
50		$dump_contents = adjust_old_dumpfile($old_version, $dump_contents);
51	}
52	else
53	{
54		$dump_contents = adjust_new_dumpfile($old_version, $dump_contents);
55	}
56
57	my $dump_file_filtered = "${dump_file}_filtered";
58	open(my $dh, '>', $dump_file_filtered)
59	  || die "opening $dump_file_filtered";
60	print $dh $dump_contents;
61	close($dh);
62
63	return $dump_file_filtered;
64}

66# Dump database db from the given node in plain format and adjust it for
67# comparing dumps from the original and the restored database.
68#
69# file_prefix is used to create unique names for all dump files so that they
70# remain available for debugging in case the test fails.
71#
72# adjust_child_columns is passed to adjust_regress_dumpfile() which actually
73# adjusts the dump output.
74#
75# The name of the file containing adjusted dump is returned.
76sub get_dump_for_comparison
77{
78	my ($node, $db, $file_prefix, $adjust_child_columns) = @_;
79
80	my $dumpfile = $tempdir . '/' . $file_prefix . '.sql';
81	my $dump_adjusted = "${dumpfile}_adjusted";
82
83	open(my $dh, '>', $dump_adjusted)
84	  || die "could not open $dump_adjusted for writing $!";
85
86	$node->run_log(
87		[
88			'pg_dump', '--no-sync',
89			'--restrict-key=test',
90			'-d' => $node->connstr($db),
91			'-f' => $dumpfile
92		]);
93
94	print $dh adjust_regress_dumpfile(slurp_file($dumpfile),
95		$adjust_child_columns);
96	close($dh);
97
98	return $dump_adjusted;
99}

101# The test of pg_upgrade requires two clusters, an old one and a new one
102# that gets upgraded.  Before running the upgrade, a logical dump of the
103# old cluster is taken, and a second logical dump of the new one is taken
104# after the upgrade.  The upgrade test passes if there are no differences
105# (after filtering) in these two dumps.
106
107# Testing upgrades with an older version of PostgreSQL requires setting up
108# two environment variables, as of:
109# - "olddump", to point to a dump file that will be used to set up the old
110#   instance to upgrade from.
111# - "oldinstall", to point to the installation path of the old cluster.
112if (   (defined($ENV{olddump}) && !defined($ENV{oldinstall}))
113	|| (!defined($ENV{olddump}) && defined($ENV{oldinstall})))
114{
115	# Not all variables are defined, so leave and die if test is
116	# done with an older installation.
117	die "olddump or oldinstall is undefined";
118}

166# can test that pg_upgrade copies the locale settings of template0

172my $original_datctype = "C";

175if ($oldnode->pg_version >= '17devel')
176{

179	$original_datlocale = "C.UTF-8";
180}

362# Stabilize stats before pg_dump / pg_dumpall. Doing it after initializing
363# the new node gives enough time for autovacuum to update statistics on the
364# old node.

368# Test that dump/restore of the regression database roundtrips cleanly.  This
369# doesn't work well when the nodes are different versions, so skip it in that
370# case.  Note that this isn't a pg_upgrade test, but it's convenient to do it
371# here because we've gone to the trouble of creating the regression database.
372#
373# Do this while the old cluster is running before it is shut down by the
374# upgrade test but after turning its autovacuum off for stable statistics.
375SKIP:
376{
377	my $dstnode = PostgreSQL::Test::Cluster->new('dst_node');
378
379	skip "regress_dump_restore not enabled in PG_TEST_EXTRA"
380	  if (!$ENV{PG_TEST_EXTRA}
381		|| $ENV{PG_TEST_EXTRA} !~ /\bregress_dump_restore\b/);
382	skip "different Postgres versions"
383	  if ($oldnode->pg_version != $dstnode->pg_version);
384	skip "source node not using default install"
385	  if (defined $oldnode->install_path);
386
387	# Setup destination database cluster with the same configuration as the
388	# source cluster to avoid any differences between dumps taken from both the
389	# clusters caused by differences in their configurations.
390	$dstnode->init(%old_node_params);
391	# Avoid unnecessary log noise
392	$dstnode->append_conf('postgresql.conf', 'log_statement = none');
393	# Stabilize stats for comparison.
394	$dstnode->append_conf('postgresql.conf', 'autovacuum = off');
395	$dstnode->start;
396
397	# Use --create in dump and restore commands so that the restored
398	# database has the same configurable variable settings as the original
399	# database so that the dumps taken from both databases taken do not
400	# differ because of locale changes. Additionally this provides test
401	# coverage for --create option.
402	#
403	# Use directory format so that we can use parallel dump/restore.
404	my $dump_file = "$tempdir/regression.dump";
405	$oldnode->command_ok(
406		[
407			'pg_dump', '-Fd', '-j2', '--no-sync',
408			'-d' => $oldnode->connstr('regression'),
409			'--create', '-f' => $dump_file
410		],
411		'pg_dump on source instance');
412
413	$dstnode->command_ok(
414		[ 'pg_restore', '--create', '-j2', '-d' => 'postgres', $dump_file ],
415		'pg_restore to destination instance');
416
417	# Dump original and restored database for comparison.
418	my $src_dump =
419	  get_dump_for_comparison($oldnode, 'regression', 'src_dump', 1);
420	my $dst_dump =
421	  get_dump_for_comparison($dstnode, 'regression', 'dest_dump', 0);
422
423	compare_files($src_dump, $dst_dump,
424		'dump outputs from original and restored regression databases match');
425}

427# Take a dump before performing the upgrade as a base comparison. Note
428# that we need to use pg_dumpall from the new node here.
429my @dump_command = (
430	'pg_dumpall', '--no-sync',
431	'--restrict-key=test',
432	'--dbname' => $oldnode->connstr('postgres'),
433	'--file' => $dump1_file);
434# --extra-float-digits is needed when upgrading from a version older than 11.
435push(@dump_command, '--extra-float-digits', '0')
436  if ($oldnode->pg_version < 12);
437$newnode->command_ok(\@dump_command, 'dump before running pg_upgrade');

439# After dumping, update references to the old source tree's regress.so
440# to point to the new tree.
441if (defined($ENV{oldinstall}))
442{
443	# First, fetch all the references to libraries that are not part
444	# of the default path $libdir.
445	my $output = $oldnode->safe_psql('regression',
446		"SELECT DISTINCT probin::text FROM pg_proc WHERE probin NOT LIKE '\$libdir%';"
447	);
448	chomp($output);
449	my @libpaths = split("\n", $output);
450
451	my $dump_data = slurp_file($dump1_file);
452
453	my $newregresssrc = dirname($ENV{REGRESS_SHLIB});
454	foreach (@libpaths)
455	{
456		my $libpath = $_;
457		$libpath = dirname($libpath);
458		$dump_data =~ s/$libpath/$newregresssrc/g;
459	}
460
461	open my $fh, ">", $dump1_file or die "could not open dump file";
462	print $fh $dump_data;
463	close $fh;
464
465	# This replaces any references to the old tree's regress.so
466	# the new tree's regress.so.  Any references that do *not*
467	# match $libdir are switched so as this request does not
468	# depend on the path of the old source tree.  This is useful
469	# when using an old dump.  Do the operation on all the databases
470	# that allow connections so as this includes the regression
471	# database and anything the user has set up.
472	$output = $oldnode->safe_psql('postgres',
473		"SELECT datname FROM pg_database WHERE datallowconn;");
474	chomp($output);
475	my @datnames = split("\n", $output);
476	foreach (@datnames)
477	{
478		my $datname = $_;
479		$oldnode->safe_psql(
480			$datname, "UPDATE pg_proc SET probin =
481		  regexp_replace(probin, '.*/', '$newregresssrc/')
482		  WHERE probin NOT LIKE '\$libdir/%'");
483	}
484}

486# Create an invalid database, will be deleted below
487$oldnode->safe_psql(
488	'postgres', qq(
489  CREATE DATABASE regression_invalid;
490  UPDATE pg_database SET datconnlimit = -2 WHERE datname = 'regression_invalid';
491));

493# In a VPATH build, we'll be started in the source directory, but we want
494# to run pg_upgrade in the build directory so that any files generated finish
495# in it, like delete_old_cluster.{sh,bat}.
496chdir ${PostgreSQL::Test::Utils::tmp_check};

499$oldnode->stop;

501# Cause a failure at the start of pg_upgrade, this should create the logging

508		'--new-datadir' => $newnode->data_dir,
509		'--old-bindir' => $oldbindir . '/does/not/exist/',
510		'--new-bindir' => $newbindir,
511		'--socketdir' => $newnode->host,
512		'--old-port' => $oldnode->port,
513		'--new-port' => $newnode->port,
514		$mode, '--check',
515	],
516	1,
517	[qr{check for ".*?does/not/exist" failed}],
518	[],
519	'run of pg_upgrade --check for new instance with incorrect binary path');
520ok(-d $newnode->data_dir . "/pg_upgrade_output.d",

522rmtree($newnode->data_dir . "/pg_upgrade_output.d");

527SKIP:
