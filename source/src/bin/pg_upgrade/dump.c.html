12#include "fe_utils/string_utils.h"
13#include "pg_upgrade.h"
14
15void
16generate_old_dump(void)
17{
18	int			dbnum;
19
20	prep_status("Creating dump of global objects");
21
22	/* run new pg_dumpall binary for globals */
23	exec_prog(UTILITY_LOG_FILE, NULL, true, true,
24			  "\"%s/pg_dumpall\" %s%s --globals-only --quote-all-identifiers "
25			  "--binary-upgrade %s --no-sync -f \"%s/%s\"",
26			  new_cluster.bindir, cluster_conn_opts(&old_cluster),
27			  protocol_negotiation_supported(&old_cluster) ? "" : " -d \"max_protocol_version=3.0\"",
28			  log_opts.verbose ? "--verbose" : "",
29			  log_opts.dumpdir,
30			  GLOBALS_DUMP_FILE);
31	check_ok();
32
33	prep_status_progress("Creating dump of database schemas");
34
35	/* create per-db dump files */
36	for (dbnum = 0; dbnum < old_cluster.dbarr.ndbs; dbnum++)
37	{
38		char		sql_file_name[MAXPGPATH],
39					log_file_name[MAXPGPATH];
40		DbInfo	   *old_db = &old_cluster.dbarr.dbs[dbnum];
41		PQExpBufferData connstr,
42					escaped_connstr;
43
44		initPQExpBuffer(&connstr);
45		appendPQExpBufferStr(&connstr, "dbname=");
46		appendConnStrVal(&connstr, old_db->db_name);
47		if (!protocol_negotiation_supported(&old_cluster))
48			appendPQExpBufferStr(&connstr, " max_protocol_version=3.0");
49
50		initPQExpBuffer(&escaped_connstr);
51		appendShellString(&escaped_connstr, connstr.data);
52		termPQExpBuffer(&connstr);
53
54		pg_log(PG_STATUS, "%s", old_db->db_name);
55		snprintf(sql_file_name, sizeof(sql_file_name), DB_DUMP_FILE_MASK, old_db->db_oid);
56		snprintf(log_file_name, sizeof(log_file_name), DB_DUMP_LOG_FILE_MASK, old_db->db_oid);
57
58		parallel_exec_prog(log_file_name, NULL,
59						   "\"%s/pg_dump\" %s --no-data %s %s --quote-all-identifiers "
60						   "--binary-upgrade --format=custom %s --no-sync --file=\"%s/%s\" %s",
61						   new_cluster.bindir, cluster_conn_opts(&old_cluster),
62						   (user_opts.transfer_mode == TRANSFER_MODE_SWAP) ?
63						   "" : "--sequence-data",
64						   log_opts.verbose ? "--verbose" : "",
65						   user_opts.do_statistics ? "--statistics" : "--no-statistics",
66						   log_opts.dumpdir,
67						   sql_file_name, escaped_connstr.data);
68
69		termPQExpBuffer(&escaped_connstr);
70	}
71
72	/* reap all children */
73	while (reap_child(true) == true)
74		;
75
76	end_progress_output();
77	check_ok();
78}
