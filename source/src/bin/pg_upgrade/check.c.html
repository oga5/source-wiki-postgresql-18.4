36/*
37 * DataTypesUsageChecks - definitions of data type checks for the old cluster
38 * in order to determine if an upgrade can be performed.  See the comment on
39 * data_types_usage_checks below for a more detailed description.
40 */
41typedef struct
42{
43	/* Status line to print to the user */
44	const char *status;
45	/* Filename to store report to */
46	const char *report_filename;
47	/* Query to extract the oid of the datatype */
48	const char *base_query;
49	/* Text to store to report in case of error */
50	const char *report_text;
51	/* The latest version where the check applies */
52	int			threshold_version;
53	/* A function pointer for determining if the check applies */
54	DataTypesUsageVersionCheck version_hook;
55} DataTypesUsageChecks;
56
57/*
58 * Special values for threshold_version for indicating that a check applies to
59 * all versions, or that a custom function needs to be invoked to determine
60 * if the check applies.
61 */
62#define MANUAL_CHECK 1
63#define ALL_VERSIONS -1
64
65/*--
66 * Data type usage checks. Each check for problematic data type usage is
67 * defined in this array with metadata, SQL query for finding the data type
68 * and functionality for deciding if the check is applicable to the version
69 * of the old cluster. The struct members are described in detail below:
70 *
71 * status				A oneline string which can be printed to the user to
72 *						inform about progress. Should not end with newline.
73 * report_filename		The filename in which the list of problems detected by
74 *						the check will be printed.
75 * base_query			A query which extracts the Oid of the datatype checked
76 *						for.
77 * report_text			The text which will be printed to the user to explain
78 *						what the check did, and why it failed. The text should
79 *						end with a newline, and does not need to refer to the
80 *						report_filename as that is automatically appended to
81 *						the report with the path to the log folder.
82 * threshold_version	The major version of PostgreSQL for which to run the
83 *						check. Iff the old cluster is less than, or equal to,
84 *						the threshold version then the check will be executed.
85 *						If the old version is greater than the threshold then
86 *						the check is skipped. If the threshold_version is set
87 *						to ALL_VERSIONS then it will be run unconditionally,
88 *						if set to MANUAL_CHECK then the version_hook function
89 *						will be executed in order to determine whether or not
90 *						to run.
91 * version_hook			A function pointer to a version check function of type
92 *						DataTypesUsageVersionCheck which is used to determine
93 *						if the check is applicable to the old cluster. If the
94 *						version_hook returns true then the check will be run,
95 *						else it will be skipped. The function will only be
96 *						executed iff threshold_version is set to MANUAL_CHECK.
97 */

318/*
319 * Private state for check_for_data_types_usage()'s UpgradeTask.
320 */
321struct data_type_check_state
322{
323	DataTypesUsageChecks *check;	/* the check for this step */
324	bool		result;			/* true if check failed for any database */
325	PQExpBuffer *report;		/* buffer for report on failed checks */
326};

328/*
329 * Returns a palloc'd query string for the data type check, for use by
330 * check_for_data_types_usage()'s UpgradeTask.
331 */
332static char *
333data_type_check_query(int checknum)
334{
335	DataTypesUsageChecks *check = &data_types_usage_checks[checknum];
336
337	return psprintf("WITH RECURSIVE oids AS ( "
338	/* start with the type(s) returned by base_query */
339					"	%s "
340					"	UNION ALL "
341					"	SELECT * FROM ( "
342	/* inner WITH because we can only reference the CTE once */
343					"		WITH x AS (SELECT oid FROM oids) "
344	/* domains on any type selected so far */
345					"			SELECT t.oid FROM pg_catalog.pg_type t, x WHERE typbasetype = x.oid AND typtype = 'd' "
346					"			UNION ALL "
347	/* arrays over any type selected so far */
348					"			SELECT t.oid FROM pg_catalog.pg_type t, x WHERE typelem = x.oid AND typtype = 'b' "
349					"			UNION ALL "
350	/* composite types containing any type selected so far */
351					"			SELECT t.oid FROM pg_catalog.pg_type t, pg_catalog.pg_class c, pg_catalog.pg_attribute a, x "
352					"			WHERE t.typtype = 'c' AND "
353					"				  t.oid = c.reltype AND "
354					"				  c.oid = a.attrelid AND "
355					"				  NOT a.attisdropped AND "
356					"				  a.atttypid = x.oid "
357					"			UNION ALL "
358	/* ranges containing any type selected so far */
359					"			SELECT t.oid FROM pg_catalog.pg_type t, pg_catalog.pg_range r, x "
360					"			WHERE t.typtype = 'r' AND r.rngtypid = t.oid AND r.rngsubtype = x.oid"
361					"	) foo "
362					") "
363	/* now look for stored columns of any such type */
364					"SELECT n.nspname, c.relname, a.attname "
365					"FROM	pg_catalog.pg_class c, "
366					"		pg_catalog.pg_namespace n, "
367					"		pg_catalog.pg_attribute a "
368					"WHERE	c.oid = a.attrelid AND "
369					"		NOT a.attisdropped AND "
370					"		a.atttypid IN (SELECT oid FROM oids) AND "
371					"		c.relkind IN ("
372					CppAsString2(RELKIND_RELATION) ", "
373					CppAsString2(RELKIND_MATVIEW) ", "
374					CppAsString2(RELKIND_INDEX) ") AND "
375					"		c.relnamespace = n.oid AND "
376	/* exclude possible orphaned temp tables */
377					"		n.nspname !~ '^pg_temp_' AND "
378					"		n.nspname !~ '^pg_toast_temp_' AND "
379	/* exclude system catalogs, too */
380					"		n.nspname NOT IN ('pg_catalog', 'information_schema')",
381					check->base_query);
382}

384/*
385 * Callback function for processing results of queries for
386 * check_for_data_types_usage()'s UpgradeTask.  If the query returned any rows
387 * (i.e., the check failed), write the details to the report file.
388 */
389static void
390process_data_type_check(DbInfo *dbinfo, PGresult *res, void *arg)
391{
392	struct data_type_check_state *state = (struct data_type_check_state *) arg;
393	int			ntups = PQntuples(res);
394	char		output_path[MAXPGPATH];
395	int			i_nspname = PQfnumber(res, "nspname");
396	int			i_relname = PQfnumber(res, "relname");
397	int			i_attname = PQfnumber(res, "attname");
398	FILE	   *script = NULL;
399
400	AssertVariableIsOfType(&process_data_type_check, UpgradeTaskProcessCB);
401
402	if (ntups == 0)
403		return;
404
405	snprintf(output_path, sizeof(output_path), "%s/%s",
406			 log_opts.basedir,
407			 state->check->report_filename);
408
409	/*
410	 * Make sure we have a buffer to save reports to now that we found a first
411	 * failing check.
412	 */
413	if (*state->report == NULL)
414		*state->report = createPQExpBuffer();
415
416	/*
417	 * If this is the first time we see an error for the check in question
418	 * then print a status message of the failure.
419	 */
420	if (!state->result)
421	{
422		pg_log(PG_REPORT, "failed check: %s", _(state->check->status));
423		appendPQExpBuffer(*state->report, "\n%s\n%s\n    %s\n",
424						  _(state->check->report_text),
425						  _("A list of the problem columns is in the file:"),
426						  output_path);
427	}
428	state->result = true;
429
430	if ((script = fopen_priv(output_path, "a")) == NULL)
431		pg_fatal("could not open file \"%s\": %m", output_path);
432
433	fprintf(script, "In database: %s\n", dbinfo->db_name);
434
435	for (int rowno = 0; rowno < ntups; rowno++)
436		fprintf(script, "  %s.%s.%s\n",
437				PQgetvalue(res, rowno, i_nspname),
438				PQgetvalue(res, rowno, i_relname),
439				PQgetvalue(res, rowno, i_attname));
440
441	fclose(script);
442}

444/*
445 * check_for_data_types_usage()
446 *	Detect whether there are any stored columns depending on given type(s)
447 *
448 * If so, write a report to the given file name and signal a failure to the
449 * user.
450 *
451 * The checks to run are defined in a DataTypesUsageChecks structure where
452 * each check has a metadata for explaining errors to the user, a base_query,
453 * a report filename and a function pointer hook for validating if the check
454 * should be executed given the cluster at hand.
455 *
456 * base_query should be a SELECT yielding a single column named "oid",
457 * containing the pg_type OIDs of one or more types that are known to have
458 * inconsistent on-disk representations across server versions.
459 *
460 * We check for the type(s) in tables, matviews, and indexes, but not views;
461 * there's no storage involved in a view.
462 */
463static void
464check_for_data_types_usage(ClusterInfo *cluster)
465{
466	PQExpBuffer report = NULL;
467	DataTypesUsageChecks *tmp = data_types_usage_checks;
468	int			n_data_types_usage_checks = 0;
469	UpgradeTask *task = upgrade_task_create();
470	char	  **queries = NULL;
471	struct data_type_check_state *states;
472
473	prep_status("Checking data type usage");
474
475	/* Gather number of checks to perform */
476	while (tmp->status != NULL)
477	{
478		n_data_types_usage_checks++;
479		tmp++;
480	}
481
482	/* Allocate memory for queries and for task states */
483	queries = pg_malloc0(sizeof(char *) * n_data_types_usage_checks);
484	states = pg_malloc0(sizeof(struct data_type_check_state) * n_data_types_usage_checks);
485
486	for (int i = 0; i < n_data_types_usage_checks; i++)
487	{
488		DataTypesUsageChecks *check = &data_types_usage_checks[i];
489
490		if (check->threshold_version == MANUAL_CHECK)
491		{
492			Assert(check->version_hook);
493
494			/*
495			 * Make sure that the check applies to the current cluster version
496			 * and skip it if not.
497			 */
498			if (!check->version_hook(cluster))
499				continue;
500		}
501		else if (check->threshold_version != ALL_VERSIONS)
502		{
503			if (GET_MAJOR_VERSION(cluster->major_version) > check->threshold_version)
504				continue;
505		}
506		else
507			Assert(check->threshold_version == ALL_VERSIONS);
508
509		queries[i] = data_type_check_query(i);
510
511		states[i].check = check;
512		states[i].report = &report;
513
514		upgrade_task_add_step(task, queries[i], process_data_type_check,
515							  true, &states[i]);
516	}
517
518	/*
519	 * Connect to each database in the cluster and run all defined checks
520	 * against that database before trying the next one.
521	 */
522	upgrade_task_run(task, cluster);
523	upgrade_task_free(task);
524
525	if (report)
526	{
527		pg_fatal("Data type checks failed: %s", report->data);
528		destroyPQExpBuffer(report);
529	}
530
531	for (int i = 0; i < n_data_types_usage_checks; i++)
532	{
533		if (queries[i])
534			pg_free(queries[i]);
535	}
536	pg_free(queries);
537	pg_free(states);
538
539	check_ok();
540}

542/*
543 * fix_path_separator
544 * For non-Windows, just return the argument.
545 * For Windows convert any forward slash to a backslash
546 * such as is suitable for arguments to builtin commands
547 * like RMDIR and DEL.
548 */
549static char *
550fix_path_separator(char *path)
551{
552#ifdef WIN32
553
554	char	   *result;
555	char	   *c;
556
557	result = pg_strdup(path);
558
559	for (c = result; *c != '\0'; c++)
560		if (*c == '/')
561			*c = '\\';
562
563	return result;
564#else
565
566	return path;
567#endif
568}

588void
589check_and_dump_old_cluster(void)
590{
591	/* -- OLD -- */
592
593	if (!user_opts.live_check)
594		start_postmaster(&old_cluster, true);
595
596	/*
597	 * First check that all databases allow connections since we'll otherwise
598	 * fail in later stages.
599	 */
600	check_for_connection_status(&old_cluster);
601
602	/*
603	 * Extract a list of databases, tables, and logical replication slots from
604	 * the old cluster.
605	 */
606	get_db_rel_and_slot_infos(&old_cluster);
607
608	init_tablespaces();
609
610	get_loadable_libraries();
611
612
613	/*
614	 * Check for various failure cases
615	 */
616	check_is_install_user(&old_cluster);
617	check_for_prepared_transactions(&old_cluster);
618	check_for_isn_and_int8_passing_mismatch(&old_cluster);
619
620	if (GET_MAJOR_VERSION(old_cluster.major_version) >= 1700)
621	{
622		/*
623		 * Logical replication slots can be migrated since PG17. See comments
624		 * atop get_old_cluster_logical_slot_infos().
625		 */
626		check_old_cluster_for_valid_slots();
627
628		/*
629		 * Subscriptions and their dependencies can be migrated since PG17.
630		 * Before that the logical slots are not upgraded, so we will not be
631		 * able to upgrade the logical replication clusters completely.
632		 */
633		get_subscription_count(&old_cluster);
634		check_old_cluster_subscription_state();
635	}
636
637	check_for_data_types_usage(&old_cluster);
638
639	/*
640	 * Unicode updates can affect some objects that use expressions with
641	 * functions dependent on Unicode.
642	 */
643	check_for_unicode_update(&old_cluster);
644
645	/*
646	 * PG 14 changed the function signature of encoding conversion functions.
647	 * Conversions from older versions cannot be upgraded automatically
648	 * because the user-defined functions used by the encoding conversions
649	 * need to be changed to match the new signature.
650	 */
651	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 1300)
652		check_for_user_defined_encoding_conversions(&old_cluster);
653
654	/*
655	 * Pre-PG 14 allowed user defined postfix operators, which are not
656	 * supported anymore.  Verify there are none, iff applicable.
657	 */
658	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 1300)
659		check_for_user_defined_postfix_ops(&old_cluster);
660
661	/*
662	 * PG 14 changed polymorphic functions from anyarray to
663	 * anycompatiblearray.
664	 */
665	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 1300)
666		check_for_incompatible_polymorphics(&old_cluster);
667
668	/*
669	 * Pre-PG 12 allowed tables to be declared WITH OIDS, which is not
670	 * supported anymore. Verify there are none, iff applicable.
671	 */
672	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 1100)
673		check_for_tables_with_oids(&old_cluster);
674
675	/*
676	 * Pre-PG 18 allowed child tables to omit not-null constraints that their
677	 * parents columns have, but schema restore fails for them.  Verify there
678	 * are none, iff applicable.
679	 */
680	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 1800)
681		check_for_not_null_inheritance(&old_cluster);
682
683	/*
684	 * Pre-PG 10 allowed tables with 'unknown' type columns and non WAL logged
685	 * hash indexes
686	 */
687	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 906)
688	{
689		if (user_opts.check)
690			old_9_6_invalidate_hash_indexes(&old_cluster, true);
691	}
692
693	/* 9.5 and below should not have roles starting with pg_ */
694	if (GET_MAJOR_VERSION(old_cluster.major_version) <= 905)
695		check_for_pg_role_prefix(&old_cluster);
696
697	/*
698	 * While not a check option, we do this now because this is the only time
699	 * the old server is running.
700	 */
701	if (!user_opts.check)
702		generate_old_dump();
703
704	if (!user_opts.live_check)
705		stop_postmaster(false);
706}

709void
710check_new_cluster(void)
711{
712	get_db_rel_and_slot_infos(&new_cluster);
713
714	check_new_cluster_is_empty();
715
716	check_loadable_libraries();
717
718	switch (user_opts.transfer_mode)
719	{
720		case TRANSFER_MODE_CLONE:
721			check_file_clone();
722			break;
723		case TRANSFER_MODE_COPY:
724			break;
725		case TRANSFER_MODE_COPY_FILE_RANGE:
726			check_copy_file_range();
727			break;
728		case TRANSFER_MODE_LINK:
729			check_hard_link(TRANSFER_MODE_LINK);
730			break;
731		case TRANSFER_MODE_SWAP:
732
733			/*
734			 * We do the hard link check for --swap, too, since it's an easy
735			 * way to verify the clusters are in the same file system.  This
736			 * allows us to take some shortcuts in the file synchronization
737			 * step.  With some more effort, we could probably support the
738			 * separate-file-system use case, but this mode is unlikely to
739			 * offer much benefit if we have to copy the files across file
740			 * system boundaries.
741			 */
742			check_hard_link(TRANSFER_MODE_SWAP);
743
744			/*
745			 * There are a few known issues with using --swap to upgrade from
746			 * versions older than 10.  For example, the sequence tuple format
747			 * changed in v10, and the visibility map format changed in 9.6.
748			 * While such problems are not insurmountable (and we may have to
749			 * deal with similar problems in the future, anyway), it doesn't
750			 * seem worth the effort to support swap mode for upgrades from
751			 * long-unsupported versions.
752			 */
753			if (GET_MAJOR_VERSION(old_cluster.major_version) < 1000)
754				pg_fatal("Swap mode can only upgrade clusters from PostgreSQL version %s and later.",
755						 "10");
756
757			break;
758	}
759
760	check_is_install_user(&new_cluster);
761
762	check_for_prepared_transactions(&new_cluster);
763
764	check_for_new_tablespace_dir();
765
766	check_new_cluster_logical_replication_slots();
767
768	check_new_cluster_subscription_configuration();
769}

849void
850check_cluster_versions(void)
851{
852	prep_status("Checking cluster versions");
853
854	/* cluster versions should already have been obtained */
855	Assert(old_cluster.major_version != 0);
856	Assert(new_cluster.major_version != 0);
857
858	/*
859	 * We allow upgrades from/to the same major version for alpha/beta
860	 * upgrades
861	 */
862
863	if (GET_MAJOR_VERSION(old_cluster.major_version) < 902)
864		pg_fatal("This utility can only upgrade from PostgreSQL version %s and later.",
865				 "9.2");
866
867	/* Only current PG version is supported as a target */
868	if (GET_MAJOR_VERSION(new_cluster.major_version) != GET_MAJOR_VERSION(PG_VERSION_NUM))
869		pg_fatal("This utility can only upgrade to PostgreSQL version %s.",
870				 PG_MAJORVERSION);
871
872	/*
873	 * We can't allow downgrading because we use the target pg_dump, and
874	 * pg_dump cannot operate on newer database versions, only current and
875	 * older versions.
876	 */
877	if (old_cluster.major_version > new_cluster.major_version)
878		pg_fatal("This utility cannot be used to downgrade to older major PostgreSQL versions.");
879
880	/* Ensure binaries match the designated data directories */
881	if (GET_MAJOR_VERSION(old_cluster.major_version) !=
882		GET_MAJOR_VERSION(old_cluster.bin_version))
883		pg_fatal("Old cluster data and binary directories are from different major versions.");
884	if (GET_MAJOR_VERSION(new_cluster.major_version) !=
885		GET_MAJOR_VERSION(new_cluster.bin_version))
886		pg_fatal("New cluster data and binary directories are from different major versions.");
887
888	/*
889	 * Since from version 18, newly created database clusters always have
890	 * 'signed' default char-signedness, it makes less sense to use
891	 * --set-char-signedness option for upgrading from version 18 or later.
892	 * Users who want to change the default char signedness of the new
893	 * cluster, they can use pg_resetwal manually before the upgrade.
894	 */
895	if (GET_MAJOR_VERSION(old_cluster.major_version) >= 1800 &&
896		user_opts.char_signedness != -1)
897		pg_fatal("The option %s cannot be used for upgrades from PostgreSQL %s and later.",
898				 "--set-char-signedness", "18");
899
900	check_ok();
901}

943 * directory recreated, but they might have forgotten to remove
944 * the new cluster's tablespace directories.  Therefore, check that
945 * new cluster tablespace directories do not already exist.  If
946 * they do, it would cause an error while restoring global objects.

948 * during schema restore.

957

959	{

963				 os_info.old_tablespaces[tblnum],
964				 new_cluster.tablespace_suffix);
965
966		if (stat(new_tablespace_dir, &statbuf) == 0 || errno != ENOENT)
967			pg_fatal("new cluster tablespace directory already exists: \"%s\"",
968					 new_tablespace_dir);
969	}

974/*
975 * create_script_for_old_cluster_deletion()
976 *
977 *	This is particularly useful for tablespace deletion.
978 */
979void
980create_script_for_old_cluster_deletion(char **deletion_script_file_name)
981{
982	FILE	   *script = NULL;
983	int			tblnum;
984	char		old_cluster_pgdata[MAXPGPATH],
985				new_cluster_pgdata[MAXPGPATH];
986	char	   *old_tblspc_suffix;
987
988	*deletion_script_file_name = psprintf("%sdelete_old_cluster.%s",
989										  SCRIPT_PREFIX, SCRIPT_EXT);
990
991	strlcpy(old_cluster_pgdata, old_cluster.pgdata, MAXPGPATH);
992	canonicalize_path(old_cluster_pgdata);
993
994	strlcpy(new_cluster_pgdata, new_cluster.pgdata, MAXPGPATH);
995	canonicalize_path(new_cluster_pgdata);
996
997	/* Some people put the new data directory inside the old one. */
998	if (path_is_prefix_of_path(old_cluster_pgdata, new_cluster_pgdata))
999	{
1000		pg_log(PG_WARNING,
1001			   "\nWARNING:  new data directory should not be inside the old data directory, i.e. %s", old_cluster_pgdata);
1002
1003		/* Unlink file in case it is left over from a previous run. */
1004		unlink(*deletion_script_file_name);
1005		pg_free(*deletion_script_file_name);
1006		*deletion_script_file_name = NULL;
1007		return;
1008	}
1009
1010	/*
1011	 * Some users (oddly) create tablespaces inside the cluster data
1012	 * directory.  We can't create a proper old cluster delete script in that
1013	 * case.
1014	 */
1015	for (tblnum = 0; tblnum < os_info.num_old_tablespaces; tblnum++)
1016	{
1017		char		old_tablespace_dir[MAXPGPATH];
1018
1019		strlcpy(old_tablespace_dir, os_info.old_tablespaces[tblnum], MAXPGPATH);
1020		canonicalize_path(old_tablespace_dir);
1021		if (path_is_prefix_of_path(old_cluster_pgdata, old_tablespace_dir))
1022		{
1023			/* reproduce warning from CREATE TABLESPACE that is in the log */
1024			pg_log(PG_WARNING,
1025				   "\nWARNING:  user-defined tablespace locations should not be inside the data directory, i.e. %s", old_tablespace_dir);
1026
1027			/* Unlink file in case it is left over from a previous run. */
1028			unlink(*deletion_script_file_name);
1029			pg_free(*deletion_script_file_name);
1030			*deletion_script_file_name = NULL;
1031			return;
1032		}
1033	}
1034
1035	prep_status("Creating script to delete old cluster");
1036
1037	if ((script = fopen_priv(*deletion_script_file_name, "w")) == NULL)
1038		pg_fatal("could not open file \"%s\": %m",
1039				 *deletion_script_file_name);
1040
1041#ifndef WIN32
1042	/* add shebang header */
1043	fprintf(script, "#!/bin/sh\n\n");
1044#endif
1045
1046	/* delete old cluster's default tablespace */
1047	fprintf(script, RMDIR_CMD " %c%s%c\n", PATH_QUOTE,
1048			fix_path_separator(old_cluster.pgdata), PATH_QUOTE);
1049
1050	/* delete old cluster's alternate tablespaces */
1051	old_tblspc_suffix = pg_strdup(old_cluster.tablespace_suffix);
1052	fix_path_separator(old_tblspc_suffix);
1053	for (tblnum = 0; tblnum < os_info.num_old_tablespaces; tblnum++)
1054		fprintf(script, RMDIR_CMD " %c%s%s%c\n", PATH_QUOTE,
1055				fix_path_separator(os_info.old_tablespaces[tblnum]),
1056				old_tblspc_suffix, PATH_QUOTE);
1057	pfree(old_tblspc_suffix);
1058
1059	fclose(script);
1060
1061#ifndef WIN32
1062	if (chmod(*deletion_script_file_name, S_IRWXU) != 0)
1063		pg_fatal("could not add execute permission to file \"%s\": %m",
1064				 *deletion_script_file_name);
1065#endif
1066
1067	check_ok();
1068}

1071/*
1072 *	check_is_install_user()
1073 *
1074 *	Check we are the install user, and that the new cluster
1075 *	has no other users.
1076 */
1077static void
1078check_is_install_user(ClusterInfo *cluster)
1079{
1080	PGresult   *res;
1081	PGconn	   *conn = connectToServer(cluster, "template1");
1082
1083	prep_status("Checking database user is the install user");
1084
1085	/* Can't use pg_authid because only superusers can view it. */
1086	res = executeQueryOrDie(conn,
1087							"SELECT rolsuper, oid "
1088							"FROM pg_catalog.pg_roles "
1089							"WHERE rolname = current_user "
1090							"AND rolname !~ '^pg_'");
1091
1092	/*
1093	 * We only allow the install user in the new cluster (see comment below)
1094	 * and we preserve pg_authid.oid, so this must be the install user in the
1095	 * old cluster too.
1096	 */
1097	if (PQntuples(res) != 1 ||
1098		atooid(PQgetvalue(res, 0, 1)) != BOOTSTRAP_SUPERUSERID)
1099		pg_fatal("database user \"%s\" is not the install user",
1100				 os_info.user);
1101
1102	PQclear(res);
1103
1104	res = executeQueryOrDie(conn,
1105							"SELECT COUNT(*) "
1106							"FROM pg_catalog.pg_roles "
1107							"WHERE rolname !~ '^pg_'");
1108
1109	if (PQntuples(res) != 1)
1110		pg_fatal("could not determine the number of users");
1111
1112	/*
1113	 * We only allow the install user in the new cluster because other defined
1114	 * users might match users defined in the old cluster and generate an
1115	 * error during pg_dump restore.
1116	 */
1117	if (cluster == &new_cluster && strcmp(PQgetvalue(res, 0, 0), "1") != 0)
1118		pg_fatal("Only the install user can be defined in the new cluster.");
1119
1120	PQclear(res);
1121
1122	PQfinish(conn);
1123
1124	check_ok();
1125}

1128/*
1129 *	check_for_connection_status
1130 *
1131 *	Ensure that all non-template0 databases allow connections since they
1132 *	otherwise won't be restored; and that template0 explicitly doesn't allow
1133 *	connections since it would make pg_dumpall --globals restore fail.
1134 */
1135static void
1136check_for_connection_status(ClusterInfo *cluster)
1137{
1138	int			dbnum;
1139	PGconn	   *conn_template1;
1140	PGresult   *dbres;
1141	int			ntups;
1142	int			i_datname;
1143	int			i_datallowconn;
1144	int			i_datconnlimit;
1145	FILE	   *script = NULL;
1146	char		output_path[MAXPGPATH];
1147
1148	prep_status("Checking database connection settings");
1149
1150	snprintf(output_path, sizeof(output_path), "%s/%s",
1151			 log_opts.basedir,
1152			 "databases_cannot_connect_to.txt");
1153
1154	conn_template1 = connectToServer(cluster, "template1");
1155
1156	/* get database names */
1157	dbres = executeQueryOrDie(conn_template1,
1158							  "SELECT	datname, datallowconn, datconnlimit "
1159							  "FROM	pg_catalog.pg_database");
1160
1161	i_datname = PQfnumber(dbres, "datname");
1162	i_datallowconn = PQfnumber(dbres, "datallowconn");
1163	i_datconnlimit = PQfnumber(dbres, "datconnlimit");
1164
1165	ntups = PQntuples(dbres);
1166	for (dbnum = 0; dbnum < ntups; dbnum++)
1167	{
1168		char	   *datname = PQgetvalue(dbres, dbnum, i_datname);
1169		char	   *datallowconn = PQgetvalue(dbres, dbnum, i_datallowconn);
1170		char	   *datconnlimit = PQgetvalue(dbres, dbnum, i_datconnlimit);
1171
1172		if (strcmp(datname, "template0") == 0)
1173		{
1174			/* avoid restore failure when pg_dumpall tries to create template0 */
1175			if (strcmp(datallowconn, "t") == 0)
1176				pg_fatal("template0 must not allow connections, "
1177						 "i.e. its pg_database.datallowconn must be false");
1178		}
1179		else
1180		{
1181			/*
1182			 * Avoid datallowconn == false databases from being skipped on
1183			 * restore, and ensure that no databases are marked invalid with
1184			 * datconnlimit == -2.
1185			 */
1186			if ((strcmp(datallowconn, "f") == 0) || strcmp(datconnlimit, "-2") == 0)
1187			{
1188				if (script == NULL && (script = fopen_priv(output_path, "w")) == NULL)
1189					pg_fatal("could not open file \"%s\": %m", output_path);
1190
1191				fprintf(script, "%s\n", datname);
1192			}
1193		}
1194	}
1195
1196	PQclear(dbres);
1197
1198	PQfinish(conn_template1);
1199
1200	if (script)
1201	{
1202		fclose(script);
1203		pg_log(PG_REPORT, "fatal");
1204		pg_fatal("All non-template0 databases must allow connections, i.e. their\n"
1205				 "pg_database.datallowconn must be true and pg_database.datconnlimit\n"
1206				 "must not be -2.  Your installation contains non-template0 databases\n"
1207				 "which cannot be connected to.  Consider allowing connection for all\n"
1208				 "non-template0 databases or drop the databases which do not allow\n"
1209				 "connections.  A list of databases with the problem is in the file:\n"
1210				 "    %s", output_path);
1211	}
1212	else
1213		check_ok();
1214}

1463 *	check_for_incompatible_polymorphics()
1464 *
1465 *	Make sure nothing is using old polymorphic functions with
1466 *	anyarray/anyelement rather than the new anycompatible variants.
1467 */
1468static void
1469check_for_incompatible_polymorphics(ClusterInfo *cluster)
1470{
1471	PQExpBufferData old_polymorphics;
1472	UpgradeTask *task = upgrade_task_create();
1473	UpgradeTaskReport report;
1474	char	   *query;
1475
1476	prep_status("Checking for incompatible polymorphic functions");
1477
1478	report.file = NULL;
1479	snprintf(report.path, sizeof(report.path), "%s/%s",
1480			 log_opts.basedir,
1481			 "incompatible_polymorphics.txt");
1482
1483	/* The set of problematic functions varies a bit in different versions */
1484	initPQExpBuffer(&old_polymorphics);
1485
1486	appendPQExpBufferStr(&old_polymorphics,
1487						 "'array_append(anyarray,anyelement)'"
1488						 ", 'array_cat(anyarray,anyarray)'"
1489						 ", 'array_prepend(anyelement,anyarray)'");
1490
1491	if (GET_MAJOR_VERSION(cluster->major_version) >= 903)
1492		appendPQExpBufferStr(&old_polymorphics,
1493							 ", 'array_remove(anyarray,anyelement)'"
1494							 ", 'array_replace(anyarray,anyelement,anyelement)'");
1495
1496	if (GET_MAJOR_VERSION(cluster->major_version) >= 905)
1497		appendPQExpBufferStr(&old_polymorphics,
1498							 ", 'array_position(anyarray,anyelement)'"
1499							 ", 'array_position(anyarray,anyelement,integer)'"
1500							 ", 'array_positions(anyarray,anyelement)'"

1502
1503	/*
1504	 * The query below hardcodes FirstNormalObjectId as 16384 rather than
1505	 * interpolating that C #define into the query because, if that #define is

1507	 * pre-version 14 servers, not that of some future version.
1508	 */
1509
1510	/* Aggregate transition functions */
1511	query = psprintf("SELECT 'aggregate' AS objkind, p.oid::regprocedure::text AS objname "
1512					 "FROM pg_proc AS p "
1513					 "JOIN pg_aggregate AS a ON a.aggfnoid=p.oid "
1514					 "JOIN pg_proc AS transfn ON transfn.oid=a.aggtransfn "
1515					 "WHERE p.oid >= 16384 "
1516					 "AND a.aggtransfn = ANY(ARRAY[%s]::regprocedure[]) "
1517					 "AND a.aggtranstype = ANY(ARRAY['anyarray', 'anyelement']::regtype[]) "
1518
1519	/* Aggregate final functions */
1520					 "UNION ALL "
1521					 "SELECT 'aggregate' AS objkind, p.oid::regprocedure::text AS objname "
1522					 "FROM pg_proc AS p "
1523					 "JOIN pg_aggregate AS a ON a.aggfnoid=p.oid "
1524					 "JOIN pg_proc AS finalfn ON finalfn.oid=a.aggfinalfn "
1525					 "WHERE p.oid >= 16384 "
1526					 "AND a.aggfinalfn = ANY(ARRAY[%s]::regprocedure[]) "
1527					 "AND a.aggtranstype = ANY(ARRAY['anyarray', 'anyelement']::regtype[]) "
1528
1529	/* Operators */
1530					 "UNION ALL "
1531					 "SELECT 'operator' AS objkind, op.oid::regoperator::text AS objname "
1532					 "FROM pg_operator AS op "
1533					 "WHERE op.oid >= 16384 "
1534					 "AND oprcode = ANY(ARRAY[%s]::regprocedure[]) "
1535					 "AND oprleft = ANY(ARRAY['anyarray', 'anyelement']::regtype[])",
1536					 old_polymorphics.data,
1537					 old_polymorphics.data,
1538					 old_polymorphics.data);
1539
1540	upgrade_task_add_step(task, query, process_incompat_polymorphics,
1541						  true, &report);
1542	upgrade_task_run(task, cluster);
1543	upgrade_task_free(task);
1544
1545	if (report.file)
1546	{
1547		fclose(report.file);
1548		pg_log(PG_REPORT, "fatal");
1549		pg_fatal("Your installation contains user-defined objects that refer to internal\n"
1550				 "polymorphic functions with arguments of type \"anyarray\" or \"anyelement\".\n"
1551				 "These user-defined objects must be dropped before upgrading and restored\n"
1552				 "afterwards, changing them to refer to the new corresponding functions with\n"
1553				 "arguments of type \"anycompatiblearray\" and \"anycompatible\".\n"
1554				 "A list of the problematic objects is in the file:\n"
1555				 "    %s", report.path);
1556	}
1557	else
1558		check_ok();
1559
1560	termPQExpBuffer(&old_polymorphics);
1561	pg_free(query);
1562}

1917/*
1918 * check_for_unicode_update()
1919 *
1920 * Check if the version of Unicode in the old server and the new server
1921 * differ. If so, check for indexes, partitioned tables, or constraints that
1922 * use expressions with functions dependent on Unicode behavior.
1923 */
1924static void
1925check_for_unicode_update(ClusterInfo *cluster)
1926{
1927	UpgradeTaskReport report;
1928	UpgradeTask *task;
1929	const char *query;
1930
1931	/*
1932	 * The builtin provider did not exist prior to version 17. While there are
1933	 * still problems that could potentially be caught from earlier versions,
1934	 * such as an index on NORMALIZE(), we don't check for that here.
1935	 */
1936	if (GET_MAJOR_VERSION(cluster->major_version) < 1700)
1937		return;
1938
1939	prep_status("Checking for objects affected by Unicode update");
1940
1941	if (!unicode_version_changed(cluster))
1942	{
1943		check_ok();
1944		return;
1945	}
1946
1947	report.file = NULL;
1948	snprintf(report.path, sizeof(report.path), "%s/%s",
1949			 log_opts.basedir,
1950			 "unicode_dependent_rels.txt");
1951
1952	query =
1953	/* collations that use built-in Unicode for character semantics */
1954		"WITH collations(collid) AS ( "
1955		"  SELECT oid FROM pg_collation "
1956		"  WHERE collprovider='b' AND colllocale IN ('C.UTF-8','PG_UNICODE_FAST') "
1957	/* include default collation, if appropriate */
1958		"  UNION "
1959		"  SELECT 'pg_catalog.default'::regcollation FROM pg_database "
1960		"  WHERE datname = current_database() AND "
1961		"  datlocprovider='b' AND datlocale IN ('C.UTF-8','PG_UNICODE_FAST') "
1962		"), "
1963	/* functions that use built-in Unicode */
1964		"functions(procid) AS ( "
1965		"  SELECT proc.oid FROM pg_proc proc "
1966		"  WHERE proname IN ('normalize','unicode_assigned','unicode_version','is_normalized') AND "
1967		"        pronamespace='pg_catalog'::regnamespace "
1968		"), "
1969	/* operators that use the input collation for character semantics */
1970		"coll_operators(operid, procid, collid) AS ( "
1971		"  SELECT oper.oid, oper.oprcode, collid FROM pg_operator oper, collations "
1972		"  WHERE oprname IN ('~', '~*', '!~', '!~*', '~~*', '!~~*') AND "
1973		"        oprnamespace='pg_catalog'::regnamespace AND "
1974		"        oprright='pg_catalog.text'::pg_catalog.regtype "
1975		"), "
1976	/* functions that use the input collation for character semantics */
1977		"coll_functions(procid, collid) AS ( "
1978		"  SELECT proc.oid, collid FROM pg_proc proc, collations "
1979		"  WHERE pronamespace='pg_catalog'::regnamespace AND "
1980		"        ((proname IN ('lower','initcap','upper','casefold') AND "
1981		"          pronargs = 1 AND "
1982		"          proargtypes[0] = 'pg_catalog.text'::pg_catalog.regtype) OR "
1983		"         (proname = 'substring' AND pronargs = 2 AND "
1984		"          proargtypes[0] = 'pg_catalog.text'::pg_catalog.regtype AND "
1985		"          proargtypes[1] = 'pg_catalog.text'::pg_catalog.regtype) OR "
1986		"         proname LIKE 'regexp_%') "
1987	/* include functions behind the operators listed above */
1988		"  UNION "
1989		"  SELECT procid, collid FROM coll_operators "
1990		"), "
1991
1992	/*
1993	 * Generate patterns to search a pg_node_tree for the above functions and
1994	 * operators.
1995	 */
1996		"patterns(p) AS ( "
1997		"  SELECT '{FUNCEXPR :funcid ' || procid::text || '[ }]' FROM functions "
1998		"  UNION "
1999		"  SELECT '{OPEXPR :opno ' || operid::text || ' (:\\w+ \\w+ )*' || "
2000		"         ':inputcollid ' || collid::text || '[ }]' FROM coll_operators "
2001		"  UNION "
2002		"  SELECT '{FUNCEXPR :funcid ' || procid::text || ' (:\\w+ \\w+ )*' || "
2003		"         ':inputcollid ' || collid::text || '[ }]' FROM coll_functions "
2004		") "
2005
2006	/*
2007	 * Match the patterns against expressions used for relation contents.
2008	 */
2009		"SELECT reloid, relkind, nspname, relname "
2010		"  FROM ( "
2011		"    SELECT conrelid "
2012		"    FROM pg_constraint, patterns WHERE conbin::text ~ p "
2013		"  UNION "
2014		"    SELECT indexrelid "
2015		"    FROM pg_index, patterns WHERE indexprs::text ~ p OR indpred::text ~ p "
2016		"  UNION "
2017		"    SELECT partrelid "
2018		"    FROM pg_partitioned_table, patterns WHERE partexprs::text ~ p "
2019		"  UNION "
2020		"    SELECT ev_class "
2021		"    FROM pg_rewrite, pg_class, patterns "
2022		"    WHERE ev_class = pg_class.oid AND relkind = 'm' AND ev_action::text ~ p"
2023		"  ) s(reloid), pg_class c, pg_namespace n, pg_database d "
2024		"  WHERE s.reloid = c.oid AND c.relnamespace = n.oid AND "
2025		"        d.datname = current_database() AND "
2026		"        d.encoding = pg_char_to_encoding('UTF8');";
2027
2028	task = upgrade_task_create();
2029	upgrade_task_add_step(task, query,
2030						  process_unicode_update,
2031						  true, &report);
2032	upgrade_task_run(task, cluster);
2033	upgrade_task_free(task);
2034
2035	if (report.file)
2036	{
2037		fclose(report.file);
2038		report_status(PG_WARNING, "warning");
2039		pg_log(PG_WARNING, "Your installation contains relations that might be affected by a new version of Unicode.\n"
2040			   "A list of potentially-affected relations is in the file:\n"
2041			   "    %s", report.path);
2042	}
2043	else
2044		check_ok();
2045}

2270/*
2271 * check_old_cluster_subscription_state()
2272 *
2273 * Verify that the replication origin corresponding to each of the
2274 * subscriptions are present and each of the subscribed tables is in
2275 * 'i' (initialize) or 'r' (ready) state.
2276 */
2277static void
2278check_old_cluster_subscription_state(void)
2279{
2280	UpgradeTask *task = upgrade_task_create();
2281	UpgradeTaskReport report;
2282	const char *query;
2283	PGresult   *res;
2284	PGconn	   *conn;
2285	int			ntup;
2286
2287	prep_status("Checking for subscription state");
2288
2289	report.file = NULL;
2290	snprintf(report.path, sizeof(report.path), "%s/%s",
2291			 log_opts.basedir,
2292			 "subs_invalid.txt");
2293
2294	/*
2295	 * Check that all the subscriptions have their respective replication
2296	 * origin.  This check only needs to run once.
2297	 */
2298	conn = connectToServer(&old_cluster, old_cluster.dbarr.dbs[0].db_name);
2299	res = executeQueryOrDie(conn,
2300							"SELECT d.datname, s.subname "
2301							"FROM pg_catalog.pg_subscription s "
2302							"LEFT OUTER JOIN pg_catalog.pg_replication_origin o "
2303							"	ON o.roname = 'pg_' || s.oid "
2304							"INNER JOIN pg_catalog.pg_database d "
2305							"	ON d.oid = s.subdbid "
2306							"WHERE o.roname IS NULL;");
2307	ntup = PQntuples(res);
2308	for (int i = 0; i < ntup; i++)
2309	{
2310		if (report.file == NULL &&
2311			(report.file = fopen_priv(report.path, "w")) == NULL)
2312			pg_fatal("could not open file \"%s\": %m", report.path);
2313		fprintf(report.file, "The replication origin is missing for database:\"%s\" subscription:\"%s\"\n",
2314				PQgetvalue(res, i, 0),
2315				PQgetvalue(res, i, 1));
2316	}
2317	PQclear(res);
2318	PQfinish(conn);
2319
2320	/*
2321	 * We don't allow upgrade if there is a risk of dangling slot or origin
2322	 * corresponding to initial sync after upgrade.
2323	 *
2324	 * A slot/origin not created yet refers to the 'i' (initialize) state,
2325	 * while 'r' (ready) state refers to a slot/origin created previously but
2326	 * already dropped. These states are supported for pg_upgrade. The other
2327	 * states listed below are not supported:
2328	 *
2329	 * a) SUBREL_STATE_DATASYNC: A relation upgraded while in this state would
2330	 * retain a replication slot and origin. The sync worker spawned after the
2331	 * upgrade cannot drop them because the subscription ID used for the slot
2332	 * and origin name no longer matches.
2333	 *
2334	 * b) SUBREL_STATE_SYNCDONE: A relation upgraded while in this state would
2335	 * retain the replication origin when there is a failure in tablesync
2336	 * worker immediately after dropping the replication slot in the
2337	 * publisher.
2338	 *
2339	 * c) SUBREL_STATE_FINISHEDCOPY: A tablesync worker spawned to work on a
2340	 * relation upgraded while in this state would expect an origin ID with
2341	 * the OID of the subscription used before the upgrade, causing it to
2342	 * fail.
2343	 *
2344	 * d) SUBREL_STATE_SYNCWAIT, SUBREL_STATE_CATCHUP and
2345	 * SUBREL_STATE_UNKNOWN: These states are not stored in the catalog, so we
2346	 * need not allow these states.
2347	 */
2348	query = "SELECT r.srsubstate, s.subname, n.nspname, c.relname "
2349		"FROM pg_catalog.pg_subscription_rel r "
2350		"LEFT JOIN pg_catalog.pg_subscription s"
2351		"   ON r.srsubid = s.oid "
2352		"LEFT JOIN pg_catalog.pg_class c"
2353		"   ON r.srrelid = c.oid "
2354		"LEFT JOIN pg_catalog.pg_namespace n"
2355		"   ON c.relnamespace = n.oid "
2356		"WHERE r.srsubstate NOT IN ('i', 'r') "
2357		"ORDER BY s.subname";
2358
2359	upgrade_task_add_step(task, query, process_old_sub_state_check,
2360						  true, &report);
2361
2362	upgrade_task_run(task, &old_cluster);
2363	upgrade_task_free(task);
2364
2365	if (report.file)
2366	{
2367		fclose(report.file);
2368		pg_log(PG_REPORT, "fatal");
2369		pg_fatal("Your installation contains subscriptions without origin or having relations not in i (initialize) or r (ready) state.\n"
2370				 "You can allow the initial sync to finish for all relations and then restart the upgrade.\n"
2371				 "A list of the problematic subscriptions is in the file:\n"
2372				 "    %s", report.path);
2373	}
2374	else
2375		check_ok();
2376}
