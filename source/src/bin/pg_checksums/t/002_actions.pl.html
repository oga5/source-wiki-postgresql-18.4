16# Utility routine to create and check a table with corrupted checksums
17# on a wanted tablespace.  Note that this stops and starts the node
18# multiple times to perform the checks, leaving the node started
19# at the end.
20sub check_relation_corruption
21{
22	my $node = shift;
23	my $table = shift;
24	my $tablespace = shift;
25	my $pgdata = $node->data_dir;
26
27	# Create table and discover its filesystem location.
28	$node->safe_psql(
29		'postgres',
30		"CREATE TABLE $table AS SELECT a FROM generate_series(1,10000) AS a;
31		ALTER TABLE $table SET (autovacuum_enabled=false);");
32
33	$node->safe_psql('postgres',
34		"ALTER TABLE " . $table . " SET TABLESPACE " . $tablespace . ";");
35
36	my $file_corrupted =
37	  $node->safe_psql('postgres', "SELECT pg_relation_filepath('$table');");
38	my $relfilenode_corrupted = $node->safe_psql('postgres',
39		"SELECT relfilenode FROM pg_class WHERE relname = '$table';");
40
41	$node->stop;
42
43	# Checksums are correct for single relfilenode as the table is not
44	# corrupted yet.
45	command_ok(
46		[
47			'pg_checksums',
48			'--check',
49			'--pgdata' => $pgdata,
50			'--filenode' => $relfilenode_corrupted,
51		],
52		"succeeds for single relfilenode on tablespace $tablespace with offline cluster"
53	);
54
55	# Time to create some corruption
56	$node->corrupt_page_checksum($file_corrupted, 0);
57
58	# Checksum checks on single relfilenode fail
59	$node->command_checks_all(
60		[
61			'pg_checksums',
62			'--check',
63			'--pgdata' => $pgdata,
64			'--filenode' => $relfilenode_corrupted,
65		],
66		1,
67		[qr/Bad checksums:.*1/],
68		[qr/checksum verification failed/],
69		"fails with corrupted data for single relfilenode on tablespace $tablespace"
70	);
71
72	# Global checksum checks fail as well
73	$node->command_checks_all(
74		[ 'pg_checksums', '--check', '--pgdata' => $pgdata ],
75		1,
76		[qr/Bad checksums:.*1/],
77		[qr/checksum verification failed/],
78		"fails with corrupted data on tablespace $tablespace");
79
80	# Drop corrupted table again and make sure there is no more corruption.
81	$node->start;
82	$node->safe_psql('postgres', "DROP TABLE $table;");
83	$node->stop;
84	$node->command_ok(
85		[ 'pg_checksums', '--check', '--pgdata' => $pgdata ],
86		"succeeds again after table drop on tablespace $tablespace");
87
88	$node->start;
89	return;
90}

196# Test postgres -C for an offline cluster.
197# Run-time GUCs are safe to query here.  Note that a lock file is created,
198# then removed, leading to an extra LOG entry showing in stderr.  This uses
199# log_min_messages=fatal to remove any noise.  This test uses a startup
200# wrapped with pg_ctl to allow the case where this runs under a privileged
201# account on Windows.
202command_checks_all(
203	[
204		'pg_ctl', 'start',
205		'--silent',
206		'--pgdata' => $pgdata,
207		'-o' => '-C data_checksums -c log_min_messages=fatal',
208	],
209	1,
210	[qr/^on$/],
211	[qr/could not start server/],
212	'data_checksums=on is reported on an offline cluster');
213
214# Checks cannot happen with an online cluster
215$node->start;
216command_fails([ 'pg_checksums', '--check', '--pgdata' => $pgdata ],
217	"fails with online cluster");

219# Check corruption of table on default tablespace.
220check_relation_corruption($node, 'corrupt1', 'pg_default');

222# Create tablespace to check corruptions in a non-default tablespace.
223my $basedir = $node->basedir;
224my $tablespace_dir = "$basedir/ts_corrupt_dir";
225mkdir($tablespace_dir);
226$node->safe_psql('postgres',
227	"CREATE TABLESPACE ts_corrupt LOCATION '$tablespace_dir';");
228check_relation_corruption($node, 'corrupt2', 'ts_corrupt');

230# Utility routine to check that pg_checksums is able to detect
231# correctly-named relation files filled with some corrupted data.
232sub fail_corrupt
233{
234	my $node = shift;
235	my $file = shift;
236	my $pgdata = $node->data_dir;
237
238	# Create the file with some dummy data in it.
239	my $file_name = "$pgdata/global/$file";
240	append_to_file $file_name, "foo";
241
242	$node->command_checks_all(
243		[ 'pg_checksums', '--check', '--pgdata' => $pgdata ],
244		1,
245		[qr/^$/],
246		[qr/could not read block 0 in file.*$file\":/],
247		"fails for corrupted data in $file");
248
249	# Remove file to prevent future lookup errors on conflicts.
250	unlink $file_name;
251	return;
252}
253
254# Stop instance for the follow-up checks.
255$node->stop;
256
257# Create a fake tablespace location that should not be scanned
258# when verifying checksums.
259mkdir "$tablespace_dir/PG_99_999999991/";
260append_to_file "$tablespace_dir/PG_99_999999991/foo", "123";
261command_ok([ 'pg_checksums', '--check', '--pgdata' => $pgdata ],
262	"succeeds with foreign tablespace");
263
264# Authorized relation files filled with corrupted data cause the
265# checksum checks to fail.  Make sure to use file names different
266# than the previous ones.
267fail_corrupt($node, "99990");
268fail_corrupt($node, "99990.123");
269fail_corrupt($node, "99990_fsm");
270fail_corrupt($node, "99990_init");
271fail_corrupt($node, "99990_vm");
272fail_corrupt($node, "99990_init.123");
273fail_corrupt($node, "99990_fsm.123");
274fail_corrupt($node, "99990_vm.123");
