4package RewindTest;
5
6# Test driver for pg_rewind. Each test consists of a cycle where a new cluster
7# is first created with initdb, and a streaming replication standby is set up
8# to follow the primary. Then the primary is shut down and the standby is
9# promoted, and finally pg_rewind is used to rewind the old primary, using the
10# standby as the source.
11#
12# To run a test, the test script (in t/ subdirectory) calls the functions
13# in this module. These functions should be called in this sequence:
14#
15# 1. setup_cluster - creates a PostgreSQL cluster that runs as the primary
16#
17# 2. start_primary - starts the primary server
18#
19# 3. create_standby - runs pg_basebackup to initialize a standby server, and
20#    sets it up to follow the primary.
21#
22# 4. promote_standby - runs "pg_ctl promote" to promote the standby server.
23# The old primary keeps running.
24#
25# 5. run_pg_rewind - stops the old primary (if it's still running) and runs
26# pg_rewind to synchronize it with the now-promoted standby server.
27#
28# 6. clean_rewind_test - stops both servers used in the test, if they're
29# still running.
30#
31# The test script can use the helper functions primary_psql and standby_psql
32# to run psql against the primary and standby servers, respectively.
33

67sub primary_psql
68{
69	my $cmd = shift;
70	my $dbname = shift || 'postgres';
71
72	system_or_bail 'psql', '--quiet', '--no-psqlrc',
73	  '--dbname' => $node_primary->connstr($dbname),
74	  '--command' => $cmd;
75	return;
76}
77
78sub standby_psql
79{
80	my $cmd = shift;
81	my $dbname = shift || 'postgres';
82
83	system_or_bail 'psql', '--quiet', '--no-psqlrc',
84	  '--dbname' => $node_standby->connstr($dbname),
85	  '--command' => $cmd;
86	return;
87}

89# Run a query against the primary, and check that the output matches what's
90# expected
91sub check_query
92{
93	local $Test::Builder::Level = $Test::Builder::Level + 1;
94
95	my ($query, $expected_stdout, $test_name) = @_;
96	my ($stdout, $stderr);
97
98	# we want just the output, no formatting
99	my $result = run [
100		'psql', '--quiet', '--no-align', '--tuples-only', '--no-psqlrc',
101		'--dbname' => $node_primary->connstr('postgres'),
102		'--command' => $query
103	  ],
104	  '>' => \$stdout,
105	  '2>' => \$stderr;
106
107	is($result, 1, "$test_name: psql exit code");
108	is($stderr, '', "$test_name: psql no stderr");
109	is($stdout, $expected_stdout, "$test_name: query result matches");
110
111	return;
112}

114sub setup_cluster
115{
116	my $extra_name = shift;    # Used to differentiate clusters
117	my $extra = shift;         # Extra params for initdb
118
119	# Initialize primary, data checksums are mandatory
120	$node_primary =
121	  PostgreSQL::Test::Cluster->new(
122		'primary' . ($extra_name ? "_${extra_name}" : ''));
123
124	# Set up pg_hba.conf and pg_ident.conf for the role running
125	# pg_rewind.  This role is used for all the tests, and has
126	# minimal permissions enough to rewind from an online source.
127	$node_primary->init(
128		allows_streaming => 1,
129		extra => $extra,
130		auth_extra => [ '--create-role' => 'rewind_user' ]);
131
132	# Set wal_keep_size to prevent WAL segment recycling after enforced
133	# checkpoints in the tests.
134	$node_primary->append_conf(
135		'postgresql.conf', qq(
136wal_keep_size = 320MB
137allow_in_place_tablespaces = on
138));
139	return;
140}
141
142sub start_primary
143{
144	$node_primary->start;
145
146	# Create custom role which is used to run pg_rewind, and adjust its
147	# permissions to the minimum necessary.
148	$node_primary->safe_psql(
149		'postgres', "
150		CREATE ROLE rewind_user LOGIN;
151		GRANT EXECUTE ON function pg_catalog.pg_ls_dir(text, boolean, boolean)
152		  TO rewind_user;
153		GRANT EXECUTE ON function pg_catalog.pg_stat_file(text, boolean)
154		  TO rewind_user;
155		GRANT EXECUTE ON function pg_catalog.pg_read_binary_file(text)
156		  TO rewind_user;
157		GRANT EXECUTE ON function pg_catalog.pg_read_binary_file(text, bigint, bigint, boolean)
158		  TO rewind_user;");
159
160	#### Now run the test-specific parts to initialize the primary before setting
161	# up standby
162
163	return;
164}

166sub create_standby
167{
168	my $extra_name = shift;
169
170	$node_standby =
171	  PostgreSQL::Test::Cluster->new(
172		'standby' . ($extra_name ? "_${extra_name}" : ''));
173	$node_primary->backup('my_backup');
174	$node_standby->init_from_backup($node_primary, 'my_backup');
175	my $connstr_primary = $node_primary->connstr();
176
177	$node_standby->append_conf(
178		"postgresql.conf", qq(
179primary_conninfo='$connstr_primary'
180));
181
182	$node_standby->set_standby_mode();
183
184	# Start standby
185	$node_standby->start;
186
187	# The standby may have WAL to apply before it matches the primary.  That
188	# is fine, because no test examines the standby before promotion.
189
190	return;
191}
192
193sub promote_standby
194{
195	#### Now run the test-specific parts to run after standby has been started
196	# up standby
197
198	# Wait for the standby to receive and write all WAL.
199	$node_primary->wait_for_catchup($node_standby, 'write');
200
201	# Now promote standby and insert some new data on primary, this will put
202	# the primary out-of-sync with the standby.
203	$node_standby->promote;
204
205	return;
206}

208sub run_pg_rewind
209{
210	my $test_mode = shift;
211	my $primary_pgdata = $node_primary->data_dir;
212	my $standby_pgdata = $node_standby->data_dir;
213	my $standby_connstr = $node_standby->connstr('postgres');
214	my $tmp_folder = PostgreSQL::Test::Utils::tempdir;
215
216	# Append the rewind-specific role to the connection string.
217	$standby_connstr = "$standby_connstr user=rewind_user";
218
219	if ($test_mode eq 'archive')
220	{
221		# pg_rewind is tested with --restore-target-wal by moving all
222		# WAL files to a secondary location.  Note that this leads to
223		# a failure in ensureCleanShutdown(), forcing to the use of
224		# --no-ensure-shutdown in this mode as the initial set of WAL
225		# files needed to ensure a clean restart is gone.  This could
226		# be improved by keeping around only a minimum set of WAL
227		# segments but that would just make the test more costly,
228		# without improving the coverage.  Hence, instead, stop
229		# gracefully the primary here.
230		$node_primary->stop;
231	}
232	else
233	{
234		# Stop the primary and be ready to perform the rewind.  The cluster
235		# needs recovery to finish once, and pg_rewind makes sure that it
236		# happens automatically.
237		$node_primary->stop('immediate');
238	}
239
240	# At this point, the rewind processing is ready to run.
241	# We now have a very simple scenario with a few diverged WAL record.
242	# The real testing begins really now with a bifurcation of the possible
243	# scenarios that pg_rewind supports.
244
245	# Keep a temporary postgresql.conf for primary node or it would be
246	# overwritten during the rewind.
247	copy(
248		"$primary_pgdata/postgresql.conf",
249		"$tmp_folder/primary-postgresql.conf.tmp");
250
251	# Now run pg_rewind
252	if ($test_mode eq "local")
253	{
254
255		# Do rewind using a local pgdata as source
256		# Stop the primary and be ready to perform the rewind
257		$node_standby->stop;
258		command_ok(
259			[
260				'pg_rewind',
261				'--debug',
262				'--source-pgdata' => $standby_pgdata,
263				'--target-pgdata' => $primary_pgdata,
264				'--no-sync',
265				'--config-file' => "$tmp_folder/primary-postgresql.conf.tmp",
266			],
267			'pg_rewind local');
268	}
269	elsif ($test_mode eq "remote")
270	{
271		# Do rewind using a remote connection as source, generating
272		# recovery configuration automatically.
273		command_ok(
274			[
275				'pg_rewind',
276				'--debug',
277				'--source-server' => $standby_connstr,
278				'--target-pgdata' => $primary_pgdata,
279				'--no-sync',
280				'--write-recovery-conf',
281				'--config-file' => "$tmp_folder/primary-postgresql.conf.tmp",
282			],
283			'pg_rewind remote');
284
285		# Check that pg_rewind with dbname and --write-recovery-conf
286		# wrote the dbname in the generated primary_conninfo value.
287		like(slurp_file("$primary_pgdata/postgresql.auto.conf"),
288			qr/dbname=postgres/m, 'recovery conf file sets dbname');
289
290		# Check that standby.signal is here as recovery configuration
291		# was requested.
292		ok( -e "$primary_pgdata/standby.signal",
293			'standby.signal created after pg_rewind');
294
295		# Now, when pg_rewind apparently succeeded with minimal permissions,
296		# add REPLICATION privilege.  So we could test that new standby
297		# is able to connect to the new primary with generated config.
298		$node_standby->safe_psql('postgres',
299			"ALTER ROLE rewind_user WITH REPLICATION;");
300	}
301	elsif ($test_mode eq "archive")
302	{
303
304		# Do rewind using a local pgdata as source and specified
305		# directory with target WAL archive.  The old primary has
306		# to be stopped at this point.
307
308		# Remove the existing archive directory and move all WAL
309		# segments from the old primary to the archives.  These
310		# will be used by pg_rewind.
311		rmtree($node_primary->archive_dir);
312		PostgreSQL::Test::RecursiveCopy::copypath(
313			$node_primary->data_dir . "/pg_wal",
314			$node_primary->archive_dir);
315
316		# Fast way to remove entire directory content
317		rmtree($node_primary->data_dir . "/pg_wal");
318		mkdir($node_primary->data_dir . "/pg_wal");
319
320		# Make sure that directories have the right umask as this is
321		# required by a follow-up check on permissions, and better
322		# safe than sorry.
323		chmod(0700, $node_primary->archive_dir) or die $!;
324		chmod(0700, $node_primary->data_dir . "/pg_wal") or die $!;
325
326		# Add appropriate restore_command to the target cluster
327		$node_primary->enable_restoring($node_primary, 0);
328
329		# Stop the new primary and be ready to perform the rewind.
330		$node_standby->stop;
331
332		# Note the use of --no-ensure-shutdown here.  WAL files are
333		# gone in this mode and the primary has been stopped
334		# gracefully already.  --config-file reuses the original
335		# postgresql.conf as restore_command has been enabled above.
336		command_ok(
337			[
338				'pg_rewind',
339				'--debug',
340				'--source-pgdata' => $standby_pgdata,
341				'--target-pgdata' => $primary_pgdata,
342				'--no-sync',
343				'--no-ensure-shutdown',
344				'--restore-target-wal',
345				'--config-file' => "$primary_pgdata/postgresql.conf",
346			],
347			'pg_rewind archive');
348	}
349	else
350	{
351
352		# Cannot come here normally
353		croak("Incorrect test mode specified");
354	}
355
356	# Now move back postgresql.conf with old settings
357	move(
358		"$tmp_folder/primary-postgresql.conf.tmp",
359		"$primary_pgdata/postgresql.conf");
360
361	chmod(
362		$node_primary->group_access() ? 0640 : 0600,
363		"$primary_pgdata/postgresql.conf")
364	  or BAIL_OUT(
365		"unable to set permissions for $primary_pgdata/postgresql.conf");
366
367	# Plug-in rewound node to the now-promoted standby node
368	if ($test_mode ne "remote")
369	{
370		my $port_standby = $node_standby->port;
371		$node_primary->append_conf(
372			'postgresql.conf', qq(
373primary_conninfo='port=$port_standby'));
374
375		$node_primary->set_standby_mode();
376	}
377
378	# Restart the primary to check that rewind went correctly
379	$node_primary->start;
380
381	#### Now run the test-specific parts to check the result
382
383	return;
384}

386# Clean up after the test. Stop both servers, if they're still running.
387sub clean_rewind_test
388{
389	$node_primary->teardown_node if defined $node_primary;
390	$node_standby->teardown_node if defined $node_standby;
391	return;
