7use strict;
8use warnings FATAL => 'all';
9use PostgreSQL::Test::Utils;
10use Test::More;
11
12program_help_ok('pg_rewind');
13program_version_ok('pg_rewind');
14program_options_handling_ok('pg_rewind');
15
16my $primary_pgdata = PostgreSQL::Test::Utils::tempdir;
17my $standby_pgdata = PostgreSQL::Test::Utils::tempdir;
18command_fails(
19	[
20		'pg_rewind',
21		'--debug',
22		'--target-pgdata' => $primary_pgdata,
23		'--source-pgdata' => $standby_pgdata,
24		'extra_arg1'
25	],
26	'too many arguments');
27command_fails([ 'pg_rewind', '--target-pgdata' => $primary_pgdata ],
28	'no source specified');
29command_fails(
30	[
31		'pg_rewind',
32		'--debug',
33		'--target-pgdata' => $primary_pgdata,
34		'--source-pgdata' => $standby_pgdata,
35		'--source-server' => 'incorrect_source'
36	],
37	'both remote and local sources specified');
38command_fails(
39	[
40		'pg_rewind',
41		'--debug',
42		'--target-pgdata' => $primary_pgdata,
43		'--source-pgdata' => $standby_pgdata,
44		'--write-recovery-conf'
45	],
46	'no local source with --write-recovery-conf');
47
48done_testing();
