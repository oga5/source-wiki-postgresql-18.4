4#
5# Test situation where a target data directory contains
6# WAL records beyond both the last checkpoint and the divergence
7# point:
8#
9# Target WAL (TLI 2):
10#
11# backup ... Checkpoint A ... INSERT 'rewind this'
12#            (TLI 1 -> 2)
13#
14#            ^ last common                        ^ minRecoveryPoint
15#              checkpoint
16#
17# Source WAL (TLI 3):
18#
19# backup ... Checkpoint A ... Checkpoint B ... INSERT 'keep this'
20#            (TLI 1 -> 2)     (TLI 2 -> 3)
21#
22#
23# The last common checkpoint is Checkpoint A. But there is WAL on TLI 2
24# after the last common checkpoint that needs to be rewound. We used to
25# have a bug where minRecoveryPoint was ignored, and pg_rewind concluded
26# that the target doesn't need rewinding in this scenario, because the
27# last checkpoint on the target TLI was an ancestor of the source TLI.
28#
29#
30# This test does not make use of RewindTest as it requires three
31# nodes.
32
33use strict;
34use warnings FATAL => 'all';
35use PostgreSQL::Test::Cluster;
36use PostgreSQL::Test::Utils;
37use Test::More;
38
39use File::Copy;
40
41my $tmp_folder = PostgreSQL::Test::Utils::tempdir;
42
43my $node_1 = PostgreSQL::Test::Cluster->new('node_1');
44$node_1->init(allows_streaming => 1);
45$node_1->append_conf(
46	'postgresql.conf', qq(
47wal_keep_size='100 MB'
48));
49
50$node_1->start;
51
52# Create a couple of test tables
53$node_1->safe_psql('postgres', 'CREATE TABLE public.foo (t TEXT)');
54$node_1->safe_psql('postgres', 'CREATE TABLE public.bar (t TEXT)');
55$node_1->safe_psql('postgres', "INSERT INTO public.bar VALUES ('in both')");
56
57#
58# Create node_2 and node_3 as standbys following node_1
59#
60my $backup_name = 'my_backup';
61$node_1->backup($backup_name);
62
63my $node_2 = PostgreSQL::Test::Cluster->new('node_2');
64$node_2->init_from_backup($node_1, $backup_name, has_streaming => 1);
65$node_2->start;
66
67my $node_3 = PostgreSQL::Test::Cluster->new('node_3');
68$node_3->init_from_backup($node_1, $backup_name, has_streaming => 1);
69$node_3->start;
70
71# Wait until node 3 has connected and caught up
72$node_1->wait_for_catchup('node_3');
73
74#
75# Swap the roles of node_1 and node_3, so that node_1 follows node_3.
76#
77$node_1->stop('fast');
78$node_3->promote;
79
80# reconfigure node_1 as a standby following node_3
81my $node_3_connstr = $node_3->connstr;
82$node_1->append_conf(
83	'postgresql.conf', qq(
84primary_conninfo='$node_3_connstr'
85));
86$node_1->set_standby_mode();
87$node_1->start();
88
89# also reconfigure node_2 to follow node_3
90$node_2->append_conf(
91	'postgresql.conf', qq(
92primary_conninfo='$node_3_connstr'
93));
94$node_2->restart();
95
96#
97# Promote node_1, to create a split-brain scenario.
98#
99
100# make sure node_1 is full caught up with node_3 first
101$node_3->wait_for_catchup('node_1');
102
103$node_1->promote;
104
105#
106# We now have a split-brain with two primaries. Insert a row on both to
107# demonstratively create a split brain. After the rewind, we should only
108# see the insert on 1, as the insert on node 3 is rewound away.
109#
110$node_1->safe_psql('postgres',
111	"INSERT INTO public.foo (t) VALUES ('keep this')");
112# 'bar' is unmodified in node 1, so it won't be overwritten by replaying the
113# WAL from node 1.
114$node_3->safe_psql('postgres',
115	"INSERT INTO public.bar (t) VALUES ('rewind this')");
116
117# Insert more rows in node 1, to bump up the XID counter. Otherwise, if
118# rewind doesn't correctly rewind the changes made on the other node,
119# we might fail to notice if the inserts are invisible because the XIDs
120# are not marked as committed.
121$node_1->safe_psql('postgres',
122	"INSERT INTO public.foo (t) VALUES ('and this')");
123$node_1->safe_psql('postgres',
124	"INSERT INTO public.foo (t) VALUES ('and this too')");
125
126# Wait for node 2 to catch up
127$node_2->poll_query_until('postgres',
128	q|SELECT COUNT(*) > 1 FROM public.bar|, 't');
129
130# At this point node_2 will shut down without a shutdown checkpoint,
131# but with WAL entries beyond the preceding shutdown checkpoint.
132$node_2->stop('fast');
133$node_3->stop('fast');
134
135my $node_2_pgdata = $node_2->data_dir;
136my $node_1_connstr = $node_1->connstr;
137
138# Keep a temporary postgresql.conf or it would be overwritten during the rewind.
139copy(
140	"$node_2_pgdata/postgresql.conf",
141	"$tmp_folder/node_2-postgresql.conf.tmp");
142
143command_ok(
144	[
145		'pg_rewind',
146		'--source-server' => $node_1_connstr,
147		'--target-pgdata' => $node_2_pgdata,
148		'--debug',
149	],
150	'run pg_rewind');
151
152# Now move back postgresql.conf with old settings
153move(
154	"$tmp_folder/node_2-postgresql.conf.tmp",
155	"$node_2_pgdata/postgresql.conf");
156
157$node_2->start;
158
159# Check contents of the test tables after rewind. The rows inserted in node 3
160# before rewind should've been overwritten with the data from node 1.
161my $result;
162$result = $node_2->safe_psql('postgres', 'SELECT * FROM public.foo');
163is( $result, qq(keep this
164and this
165and this too), 'table foo after rewind');
166
167$result = $node_2->safe_psql('postgres', 'SELECT * FROM public.bar');
168is($result, qq(in both), 'table bar after rewind');
169
170done_testing();
