5# Test pg_rewind when the target's pg_wal directory is a symlink.
6#
7use strict;
8use warnings FATAL => 'all';
9use File::Copy;
10use File::Path qw(rmtree);
11use PostgreSQL::Test::Utils;
12use Test::More;
13
14use FindBin;
15use lib $FindBin::RealBin;
16
17use RewindTest;
18
19sub run_test
20{
21	my $test_mode = shift;
22
23	my $primary_xlogdir =
24	  "${PostgreSQL::Test::Utils::tmp_check}/xlog_primary";
25
26	rmtree($primary_xlogdir);
27	RewindTest::setup_cluster($test_mode);
28
29	my $test_primary_datadir = $node_primary->data_dir;
30
31	# turn pg_wal into a symlink
32	print("moving $test_primary_datadir/pg_wal to $primary_xlogdir\n");
33	move("$test_primary_datadir/pg_wal", $primary_xlogdir) or die;
34	dir_symlink($primary_xlogdir, "$test_primary_datadir/pg_wal") or die;
35
36	RewindTest::start_primary();
37
38	# Create a test table and insert a row in primary.
39	primary_psql("CREATE TABLE tbl1 (d text)");
40	primary_psql("INSERT INTO tbl1 VALUES ('in primary')");
41
42	primary_psql("CHECKPOINT");
43
44	RewindTest::create_standby($test_mode);
45
46	# Insert additional data on primary that will be replicated to standby
47	primary_psql("INSERT INTO tbl1 values ('in primary, before promotion')");
48
49	primary_psql('CHECKPOINT');
50
51	RewindTest::promote_standby();
52
53	# Insert a row in the old primary. This causes the primary and standby
54	# to have "diverged", it's no longer possible to just apply the
55	# standby's logs over primary directory - you need to rewind.
56	primary_psql("INSERT INTO tbl1 VALUES ('in primary, after promotion')");
57
58	# Also insert a new row in the standby, which won't be present in the
59	# old primary.
60	standby_psql("INSERT INTO tbl1 VALUES ('in standby, after promotion')");
61
62	RewindTest::run_pg_rewind($test_mode);
63
64	check_query(
65		'SELECT * FROM tbl1',
66		qq(in primary
67in primary, before promotion
68in standby, after promotion
69),
70		'table content');
71
72	RewindTest::clean_rewind_test();
73	return;
74}
75
76# Run the test in both modes
77run_test('local');
78run_test('remote');
79
80done_testing();
