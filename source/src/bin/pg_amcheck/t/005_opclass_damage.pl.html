4# This regression test checks the behavior of the btree validation in the
5# presence of breaking sort order changes.
6#
7use strict;
8use warnings FATAL => 'all';
9use PostgreSQL::Test::Cluster;
10use PostgreSQL::Test::Utils;
11use Test::More;
12
13my $node = PostgreSQL::Test::Cluster->new('test');
14$node->init;
15$node->start;
16
17# Create a custom operator class and an index which uses it.
18$node->safe_psql(
19	'postgres', q(
20	CREATE EXTENSION amcheck;
21
22	CREATE FUNCTION int4_asc_cmp (a int4, b int4) RETURNS int LANGUAGE sql AS $$
23		SELECT CASE WHEN $1 = $2 THEN 0 WHEN $1 > $2 THEN 1 ELSE -1 END; $$;
24
25	CREATE FUNCTION ok_cmp (int4, int4)
26	RETURNS int LANGUAGE sql AS
27	$$
28		SELECT
29			CASE WHEN $1 < $2 THEN -1
30				 WHEN $1 > $2 THEN  1
31				 ELSE 0
32			END;
33	$$;
34
35	CREATE OPERATOR CLASS int4_fickle_ops FOR TYPE int4 USING btree AS
36	    OPERATOR 1 < (int4, int4), OPERATOR 2 <= (int4, int4),
37	    OPERATOR 3 = (int4, int4), OPERATOR 4 >= (int4, int4),
38	    OPERATOR 5 > (int4, int4), FUNCTION 1 int4_asc_cmp(int4, int4);
39
40	CREATE OPERATOR CLASS int4_unique_ops FOR TYPE int4 USING btree AS
41		OPERATOR 1 < (int4, int4), OPERATOR 2 <= (int4, int4),
42		OPERATOR 3 = (int4, int4), OPERATOR 4 >= (int4, int4),
43		OPERATOR 5 > (int4, int4), FUNCTION 1 ok_cmp(int4, int4);
44
45	CREATE TABLE int4tbl (i int4);
46	INSERT INTO int4tbl (SELECT * FROM generate_series(1,1000) gs);
47	CREATE INDEX fickleidx ON int4tbl USING btree (i int4_fickle_ops);
48	CREATE UNIQUE INDEX bttest_unique_idx
49						ON int4tbl
50						USING btree (i int4_unique_ops)
51						WITH (deduplicate_items = off);
52));
53
54# We have not yet broken the index, so we should get no corruption
55$node->command_like([ 'pg_amcheck', '--port' => $node->port, 'postgres' ],
56	qr/^$/,
57	'pg_amcheck all schemas, tables and indexes reports no corruption');
58
59# Change the operator class to use a function which sorts in a different
60# order to corrupt the btree index
61$node->safe_psql(
62	'postgres', q(
63	CREATE FUNCTION int4_desc_cmp (int4, int4) RETURNS int LANGUAGE sql AS $$
64		SELECT CASE WHEN $1 = $2 THEN 0 WHEN $1 > $2 THEN -1 ELSE 1 END; $$;
65	UPDATE pg_catalog.pg_amproc
66		SET amproc = 'int4_desc_cmp'::regproc
67		WHERE amproc = 'int4_asc_cmp'::regproc
68));
69
70# Index corruption should now be reported
71$node->command_checks_all(
72	[ 'pg_amcheck', '--port' => $node->port, 'postgres' ],
73	2,
74	[qr/item order invariant violated for index "fickleidx"/],
75	[],
76	'pg_amcheck all schemas, tables and indexes reports fickleidx corruption'
77);
78
79#
80# Check unique constraints
81#
82
83# Repair broken opclass for check unique tests.
84$node->safe_psql(
85	'postgres', q(
86	UPDATE pg_catalog.pg_amproc
87		SET amproc = 'int4_asc_cmp'::regproc
88		WHERE amproc = 'int4_desc_cmp'::regproc
89));
90
91# We should get no corruptions
92$node->command_like(
93	[ 'pg_amcheck', '--checkunique', '--port' => $node->port, 'postgres' ],
94	qr/^$/,
95	'pg_amcheck all schemas, tables and indexes reports no corruption');
96
97# Break opclass for check unique tests.
98$node->safe_psql(
99	'postgres', q(
100	CREATE FUNCTION bad_cmp (int4, int4)
101	RETURNS int LANGUAGE sql AS
102	$$
103		SELECT
104			CASE WHEN ($1 = 768 AND $2 = 769) OR
105					  ($1 = 769 AND $2 = 768) THEN 0
106				 WHEN $1 < $2 THEN -1
107				 WHEN $1 > $2 THEN  1
108				 ELSE 0
109			END;
110	$$;
111
112	UPDATE pg_catalog.pg_amproc
113		SET amproc = 'bad_cmp'::regproc
114		WHERE amproc = 'ok_cmp'::regproc
115));
116
117# Unique index corruption should now be reported
118$node->command_checks_all(
119	[ 'pg_amcheck', '--checkunique', '--port' => $node->port, 'postgres' ],
120	2,
121	[qr/index uniqueness is violated for index "bttest_unique_idx"/],
122	[],
123	'pg_amcheck all schemas, tables and indexes reports bttest_unique_idx corruption'
124);
125done_testing();
