12# This regression test demonstrates that the pg_amcheck binary correctly
13# identifies specific kinds of corruption within pages.  To test this, we need
14# a mechanism to create corrupt pages with predictable, repeatable corruption.
15# The postgres backend cannot be expected to help us with this, as its design
16# is not consistent with the goal of intentionally corrupting pages.
17#
18# Instead, we create a table to corrupt, and with careful consideration of how
19# postgresql lays out heap pages, we seek to offsets within the page and
20# overwrite deliberately chosen bytes with specific values calculated to
21# corrupt the page in expected ways.  We then verify that pg_amcheck reports
22# the corruption, and that it runs without crashing.  Note that the backend
23# cannot simply be started to run queries against the corrupt table, as the
24# backend will crash, at least for some of the corruption types we generate.
25#
26# Autovacuum potentially touching the table in the background makes the exact
27# behavior of this test harder to reason about.  We turn it off to keep things
28# simpler.  We use a "belt and suspenders" approach, turning it off for the
29# system generally in postgresql.conf, and turning it off specifically for the
30# test table.
31#
32# This test depends on the table being written to the heap file exactly as we
33# expect it to be, so we take care to arrange the columns of the table, and
34# insert rows of the table, that give predictable sizes and locations within
35# the table page.
36#
37# The HeapTupleHeaderData has 23 bytes of fixed size fields before the variable
38# length t_bits[] array.  We have exactly 3 columns in the table, so natts = 3,
39# t_bits is 1 byte long, and t_hoff = MAXALIGN(23 + 1) = 24.
40#
41# We're not too fussy about which datatypes we use for the test, but we do care
42# about some specific properties.  We'd like to test both fixed size and
43# varlena types.  We'd like some varlena data inline and some toasted.  And
44# we'd like the layout of the table such that the datums land at predictable
45# offsets within the tuple.  We choose a structure without padding on all
46# supported architectures:
47#
48#	a BIGINT
49#	b TEXT
50#	c TEXT
51#
52# We always insert a 7-ascii character string into field 'b', which with a
53# 1-byte varlena header gives an 8 byte inline value.  We always insert a long
54# text string in field 'c', long enough to force toast storage.
55#
56# We choose to read and write binary copies of our table's tuples, using perl's
57# pack() and unpack() functions.  Perl uses a packing code system in which:
58#
59#	l = "signed 32-bit Long",
60#	L = "Unsigned 32-bit Long",
61#	S = "Unsigned 16-bit Short",
62#	C = "Unsigned 8-bit Octet",
63#
64# Each tuple in our table has a layout as follows:
65#
66#    xx xx xx xx            t_xmin: xxxx		offset = 0		L
67#    xx xx xx xx            t_xmax: xxxx		offset = 4		L
68#    xx xx xx xx          t_field3: xxxx		offset = 8		L
69#    xx xx                   bi_hi: xx			offset = 12		S
70#    xx xx                   bi_lo: xx			offset = 14		S
71#    xx xx                ip_posid: xx			offset = 16		S
72#    xx xx             t_infomask2: xx			offset = 18		S
73#    xx xx              t_infomask: xx			offset = 20		S
74#    xx                     t_hoff: x			offset = 22		C
75#    xx                     t_bits: x			offset = 23		C
76#    xx xx xx xx xx xx xx xx   'a': xxxxxxxx	offset = 24		LL
77#    xx xx xx xx xx xx xx xx   'b': xxxxxxxx	offset = 32		CCCCCCCC
78#    xx xx xx xx xx xx xx xx   'c': xxxxxxxx	offset = 40		CCllLL
79#    xx xx xx xx xx xx xx xx      : xxxxxxxx	 ...continued
80#    xx xx                        : xx			 ...continued
81#
82# We could choose to read and write columns 'b' and 'c' in other ways, but
83# it is convenient enough to do it this way.  We define packing code
84# constants here, where they can be compared easily against the layout.
85
86use constant HEAPTUPLE_PACK_CODE => 'LLLSSSSSCCLLCCCCCCCCCCllLL';
87use constant HEAPTUPLE_PACK_LENGTH => 58;    # Total size
88
89# Read a tuple of our table from a heap page.
90#
91# Takes an open filehandle to the heap file, and the offset of the tuple.
92#
93# Rather than returning the binary data from the file, unpacks the data into a
94# perl hash with named fields.  These fields exactly match the ones understood
95# by write_tuple(), below.  Returns a reference to this hash.
96#
97sub read_tuple
98{
99	my ($fh, $offset) = @_;
100	my ($buffer, %tup);
101	sysseek($fh, $offset, 0)
102	  or BAIL_OUT("sysseek failed: $!");
103	defined(sysread($fh, $buffer, HEAPTUPLE_PACK_LENGTH))
104	  or BAIL_OUT("sysread failed: $!");
105
106	@_ = unpack(HEAPTUPLE_PACK_CODE, $buffer);
107	%tup = (
108		t_xmin => shift,
109		t_xmax => shift,
110		t_field3 => shift,
111		bi_hi => shift,
112		bi_lo => shift,
113		ip_posid => shift,
114		t_infomask2 => shift,
115		t_infomask => shift,
116		t_hoff => shift,
117		t_bits => shift,
118		a_1 => shift,
119		a_2 => shift,
120		b_header => shift,
121		b_body1 => shift,
122		b_body2 => shift,
123		b_body3 => shift,
124		b_body4 => shift,
125		b_body5 => shift,
126		b_body6 => shift,
127		b_body7 => shift,
128		c_va_header => shift,
129		c_va_vartag => shift,
130		c_va_rawsize => shift,
131		c_va_extinfo => shift,
132		c_va_valueid => shift,
133		c_va_toastrelid => shift);
134	# Stitch together the text for column 'b'
135	$tup{b} = join('', map { chr($tup{"b_body$_"}) } (1 .. 7));
136	return \%tup;
137}

139# Write a tuple of our table to a heap page.
140#
141# Takes an open filehandle to the heap file, the offset of the tuple, and a
142# reference to a hash with the tuple values, as returned by read_tuple().
143# Writes the tuple fields from the hash into the heap file.
144#
145# The purpose of this function is to write a tuple back to disk with some
146# subset of fields modified.  The function does no error checking.  Use
147# cautiously.
148#
149sub write_tuple
150{
151	my ($fh, $offset, $tup) = @_;
152	my $buffer = pack(
153		HEAPTUPLE_PACK_CODE,
154		$tup->{t_xmin}, $tup->{t_xmax},
155		$tup->{t_field3}, $tup->{bi_hi},
156		$tup->{bi_lo}, $tup->{ip_posid},
157		$tup->{t_infomask2}, $tup->{t_infomask},
158		$tup->{t_hoff}, $tup->{t_bits},
159		$tup->{a_1}, $tup->{a_2},
160		$tup->{b_header}, $tup->{b_body1},
161		$tup->{b_body2}, $tup->{b_body3},
162		$tup->{b_body4}, $tup->{b_body5},
163		$tup->{b_body6}, $tup->{b_body7},
164		$tup->{c_va_header}, $tup->{c_va_vartag},
165		$tup->{c_va_rawsize}, $tup->{c_va_extinfo},
166		$tup->{c_va_valueid}, $tup->{c_va_toastrelid});
167	sysseek($fh, $offset, 0)
168	  or BAIL_OUT("sysseek failed: $!");
169	defined(syswrite($fh, $buffer, HEAPTUPLE_PACK_LENGTH))
170	  or BAIL_OUT("syswrite failed: $!");
171	return;
172}

399# Some #define constants from access/htup_details.h for use while corrupting.
400use constant HEAP_HASNULL => 0x0001;
401use constant HEAP_XMAX_LOCK_ONLY => 0x0080;
402use constant HEAP_XMIN_COMMITTED => 0x0100;
403use constant HEAP_XMIN_INVALID => 0x0200;
404use constant HEAP_XMAX_COMMITTED => 0x0400;
405use constant HEAP_XMAX_INVALID => 0x0800;
406use constant HEAP_NATTS_MASK => 0x07FF;
407use constant HEAP_XMAX_IS_MULTI => 0x1000;
408use constant HEAP_KEYS_UPDATED => 0x2000;
409use constant HEAP_HOT_UPDATED => 0x4000;

413# Helper function to generate a regular expression matching the header we
414# expect verify_heapam() to return given which fields we expect to be non-null.
415sub header
416{
417	my ($blkno, $offnum, $attnum) = @_;
418	return
419	  qr/heap table "postgres\.public\.test", block $blkno, offset $offnum, attribute $attnum:\s+/ms
420	  if (defined $attnum);
421	return
422	  qr/heap table "postgres\.public\.test", block $blkno, offset $offnum:\s+/ms
423	  if (defined $offnum);
424	return qr/heap table "postgres\.public\.test", block $blkno:\s+/ms
425	  if (defined $blkno);
426	return qr/heap table "postgres\.public\.test":\s+/ms;
427}

429# Corrupt the tuples, one type of corruption per tuple.  Some types of
430# corruption cause verify_heapam to skip to the next tuple without
431# performing any remaining checks, so we can't exercise the system properly if
432# we focus all our corruption on a single tuple.
433#
434my @expected;
435open($file, '+<', $relpath)
436  or BAIL_OUT("open failed: $!");
437binmode $file;
438
439for (my $tupidx = 0; $tupidx < $ROWCOUNT; $tupidx++)
440{
441	my $offnum = $tupidx + 1;    # offnum is 1-based, not zero-based
442	my $offset = $lp_off[$tupidx];
443	my $header = header(0, $offnum, undef);
444
445	# Read tuple, if there is one.
446	my $tup = $offset == -1 ? undef : read_tuple($file, $offset);
447
448	if ($offnum == 1)
449	{
450		# Corruptly set xmin < relfrozenxid
451		my $xmin = $relfrozenxid - 1;
452		$tup->{t_xmin} = $xmin;
453		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
454		$tup->{t_infomask} &= ~HEAP_XMIN_INVALID;
455
456		# Expected corruption report
457		push @expected,
458		  qr/${header}xmin $xmin precedes relation freeze threshold 0:\d+/;
459	}
460	elsif ($offnum == 2)
461	{
462		# Corruptly set xmin < datfrozenxid
463		my $xmin = 3;
464		$tup->{t_xmin} = $xmin;
465		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
466		$tup->{t_infomask} &= ~HEAP_XMIN_INVALID;
467
468		push @expected,
469		  qr/${$header}xmin $xmin precedes oldest valid transaction ID 0:\d+/;
470	}
471	elsif ($offnum == 3)
472	{
473		# Corruptly set xmin < datfrozenxid, further back, noting circularity
474		# of xid comparison.
475		my $xmin = 4026531839;
476		$tup->{t_xmin} = $xmin;
477		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
478		$tup->{t_infomask} &= ~HEAP_XMIN_INVALID;
479
480		push @expected,
481		  qr/${$header}xmin ${xmin} precedes oldest valid transaction ID 0:\d+/;
482	}
483	elsif ($offnum == 4)
484	{
485		# Corruptly set xmax < relminmxid;
486		my $xmax = 4026531839;
487		$tup->{t_xmax} = $xmax;
488		$tup->{t_infomask} &= ~HEAP_XMAX_INVALID;
489
490		push @expected,
491		  qr/${$header}xmax ${xmax} precedes oldest valid transaction ID 0:\d+/;
492	}
493	elsif ($offnum == 5)
494	{
495		# Corrupt the tuple t_hoff, but keep it aligned properly
496		$tup->{t_hoff} += 128;
497
498		push @expected,
499		  qr/${$header}data begins at offset 152 beyond the tuple length 58/,
500		  qr/${$header}tuple data should begin at byte 24, but actually begins at byte 152 \(3 attributes, no nulls\)/;
501	}
502	elsif ($offnum == 6)
503	{
504		# Corrupt the tuple t_hoff, wrong alignment
505		$tup->{t_hoff} += 3;
506
507		push @expected,
508		  qr/${$header}tuple data should begin at byte 24, but actually begins at byte 27 \(3 attributes, no nulls\)/;
509	}
510	elsif ($offnum == 7)
511	{
512		# Corrupt the tuple t_hoff, underflow but correct alignment
513		$tup->{t_hoff} -= 8;
514
515		push @expected,
516		  qr/${$header}tuple data should begin at byte 24, but actually begins at byte 16 \(3 attributes, no nulls\)/;
517	}
518	elsif ($offnum == 8)
519	{
520		# Corrupt the tuple t_hoff, underflow and wrong alignment
521		$tup->{t_hoff} -= 3;
522
523		push @expected,
524		  qr/${$header}tuple data should begin at byte 24, but actually begins at byte 21 \(3 attributes, no nulls\)/;
525	}
526	elsif ($offnum == 9)
527	{
528		# Corrupt the tuple to look like it has lots of attributes, not just 3
529		$tup->{t_infomask2} |= HEAP_NATTS_MASK;
530
531		push @expected,
532		  qr/${$header}number of attributes 2047 exceeds maximum expected for table 3/;
533	}
534	elsif ($offnum == 10)
535	{
536		# Corrupt the tuple to look like it has lots of attributes, some of
537		# them null.  This falsely creates the impression that the t_bits
538		# array is longer than just one byte, but t_hoff still says otherwise.
539		$tup->{t_infomask} |= HEAP_HASNULL;
540		$tup->{t_infomask2} |= HEAP_NATTS_MASK;
541		$tup->{t_bits} = 0xAA;
542
543		push @expected,
544		  qr/${$header}tuple data should begin at byte 280, but actually begins at byte 24 \(2047 attributes, has nulls\)/;
545	}
546	elsif ($offnum == 11)
547	{
548		# Same as above, but this time t_hoff plays along
549		$tup->{t_infomask} |= HEAP_HASNULL;
550		$tup->{t_infomask2} |= (HEAP_NATTS_MASK & 0x40);
551		$tup->{t_bits} = 0xAA;
552		$tup->{t_hoff} = 32;
553
554		push @expected,
555		  qr/${$header}number of attributes 67 exceeds maximum expected for table 3/;
556	}
557	elsif ($offnum == 12)
558	{
559		# Overwrite column 'b' 1-byte varlena header and initial characters to
560		# look like a long 4-byte varlena
561		#
562		# On little endian machines, bytes ending in two zero bits (xxxxxx00 bytes)
563		# are 4-byte length word, aligned, uncompressed data (up to 1G).  We set the
564		# high six bits to 111111 and the lower two bits to 00, then the next three
565		# bytes with 0xFF using 0xFCFFFFFF.
566		#
567		# On big endian machines, bytes starting in two zero bits (00xxxxxx bytes)
568		# are 4-byte length word, aligned, uncompressed data (up to 1G).  We set the
569		# low six bits to 111111 and the high two bits to 00, then the next three
570		# bytes with 0xFF using 0x3FFFFFFF.
571		#
572		$tup->{b_header} = $ENDIANNESS eq 'little' ? 0xFC : 0x3F;
573		$tup->{b_body1} = 0xFF;
574		$tup->{b_body2} = 0xFF;
575		$tup->{b_body3} = 0xFF;
576
577		$header = header(0, $offnum, 1);
578		push @expected,
579		  qr/${header}attribute with length \d+ ends at offset \d+ beyond total tuple length \d+/;
580	}
581	elsif ($offnum == 13)
582	{
583		# Corrupt the bits in column 'c' toast pointer
584		$tup->{c_va_valueid} = 0xFFFFFFFF;
585
586		$header = header(0, $offnum, 2);
587		push @expected, qr/${header}toast value \d+ not found in toast table/;
588	}
589	elsif ($offnum == 14)
590	{
591		# Set both HEAP_XMAX_COMMITTED and HEAP_XMAX_IS_MULTI
592		$tup->{t_infomask} |= HEAP_XMAX_COMMITTED;
593		$tup->{t_infomask} |= HEAP_XMAX_IS_MULTI;
594		$tup->{t_xmax} = 4;
595
596		push @expected,
597		  qr/${header}multitransaction ID 4 equals or exceeds next valid multitransaction ID 1/;
598	}
599	elsif ($offnum == 15)
600	{
601		# Set both HEAP_XMAX_COMMITTED and HEAP_XMAX_IS_MULTI
602		$tup->{t_infomask} |= HEAP_XMAX_COMMITTED;
603		$tup->{t_infomask} |= HEAP_XMAX_IS_MULTI;
604		$tup->{t_xmax} = 4000000000;
605
606		push @expected,
607		  qr/${header}multitransaction ID 4000000000 precedes relation minimum multitransaction ID threshold 1/;
608	}
609	elsif ($offnum == 16)    # Last offnum must equal ROWCOUNT
610	{
611		# Corruptly set xmin > next_xid to be in the future.
612		my $xmin = 123456;
613		$tup->{t_xmin} = $xmin;
614		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
615		$tup->{t_infomask} &= ~HEAP_XMIN_INVALID;
616
617		push @expected,
618		  qr/${$header}xmin ${xmin} equals or exceeds next valid transaction ID 0:\d+/;
619	}
620	elsif ($offnum == 17)
621	{
622		# at offnum 19 we will unset HEAP_ONLY_TUPLE flag
623		die "offnum $offnum should be a redirect" if defined $tup;
624		push @expected,
625		  qr/${header}redirected line pointer points to a non-heap-only tuple at offset \d+/;
626	}
627	elsif ($offnum == 18)
628	{
629		# rewrite line pointer with lp_off = 17, lp_flags = 2, lp_len = 0.
630		die "offnum $offnum should be a redirect" if defined $tup;
631		sysseek($file, 92, 0) or BAIL_OUT("sysseek failed: $!");
632		syswrite($file,
633			pack("L", $ENDIANNESS eq 'little' ? 0x00010011 : 0x00230000))
634		  or BAIL_OUT("syswrite failed: $!");
635		push @expected,
636		  qr/${header}redirected line pointer points to another redirected line pointer at offset \d+/;
637	}
638	elsif ($offnum == 19)
639	{
640		# unset HEAP_ONLY_TUPLE flag, so that update chain validation will
641		# complain about offset 17
642		$tup->{t_infomask2} &= ~HEAP_ONLY_TUPLE;
643	}
644	elsif ($offnum == 22)
645	{
646		# rewrite line pointer with lp.off = 25, lp_flags = 2, lp_len = 0
647		sysseek($file, 108, 0) or BAIL_OUT("sysseek failed: $!");
648		syswrite($file,
649			pack("L", $ENDIANNESS eq 'little' ? 0x00010019 : 0x00330000))
650		  or BAIL_OUT("syswrite failed: $!");
651		push @expected,
652		  qr/${header}redirect line pointer points to offset \d+, but offset \d+ also points there/;
653	}
654	elsif ($offnum == 28)
655	{
656		$tup->{t_infomask2} &= ~HEAP_HOT_UPDATED;
657		push @expected,
658		  qr/${header}non-heap-only update produced a heap-only tuple at offset \d+/;
659
660		# Save these values so we can insert them into the tuple at offnum 29.
661		$pred_xmax = $tup->{t_xmax};
662		$pred_posid = $tup->{ip_posid};
663	}
664	elsif ($offnum == 29)
665	{
666		# Copy these values from the tuple at offset 28.
667		$tup->{t_xmax} = $pred_xmax;
668		$tup->{ip_posid} = $pred_posid;
669		push @expected,
670		  qr/${header}tuple points to new version at offset \d+, but offset \d+ also points there/;
671	}
672	elsif ($offnum == 30)
673	{
674		# Save xid, so we can insert into into tuple at offset 31.
675		$aborted_xid = $tup->{t_xmax};
676	}
677	elsif ($offnum == 31)
678	{
679		# Set xmin to xmax of tuple at offset 30.
680		$tup->{t_xmin} = $aborted_xid;
681		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
682		push @expected,
683		  qr/${header}tuple with aborted xmin \d+ was updated to produce a tuple at offset \d+ with committed xmin \d+/;
684	}
685	elsif ($offnum == 32)
686	{
687		$tup->{t_infomask2} |= HEAP_ONLY_TUPLE;
688		push @expected,
689		  qr/${header}tuple is root of chain but is marked as heap-only tuple/;
690		push @expected,
691		  qr/${header}tuple is heap only, but not the result of an update/;
692	}
693	elsif ($offnum == 33)
694	{
695		# Tuple at offset 40 is the successor of this one; we'll corrupt it to
696		# be non-heap-only.
697		push @expected,
698		  qr/${header}heap-only update produced a non-heap only tuple at offset \d+/;
699	}
700	elsif ($offnum == 34)
701	{
702		$tup->{t_xmax} = 0;
703		push @expected,
704		  qr/${header}tuple has been HOT updated, but xmax is 0/;
705	}
706	elsif ($offnum == 35)
707	{
708		$tup->{t_xmin} = $in_progress_xid;
709		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
710		push @expected,
711		  qr/${header}tuple with in-progress xmin \d+ was updated to produce a tuple at offset \d+ with committed xmin \d+/;
712	}
713	elsif ($offnum == 36)
714	{
715		# Tuple at offset 43 is the successor of this one; we'll corrupt it to
716		# have xmin = $in_progress_xid. By setting the xmax of this tuple to
717		# the same value, we make it look like an update chain with an
718		# in-progress XID following a committed one.
719		$tup->{t_xmin} = $aborted_xid;
720		$tup->{t_xmax} = $in_progress_xid;
721		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
722		push @expected,
723		  qr/${header}tuple with aborted xmin \d+ was updated to produce a tuple at offset \d+ with in-progress xmin \d+/;
724	}
725	elsif ($offnum == 40)
726	{
727		# Tuple at offset 33 is the predecessor of this one; the error will
728		# be reported there.
729		$tup->{t_infomask2} &= ~HEAP_ONLY_TUPLE;
730	}
731	elsif ($offnum == 43)
732	{
733		# Tuple at offset 36 is the predecessor of this one; the error will
734		# be reported there.
735		$tup->{t_xmin} = $in_progress_xid;
736		$tup->{t_infomask} &= ~HEAP_XMIN_COMMITTED;
737	}
738	else
739	{
740		# The tests for update chain validation end up creating a bunch of
741		# tuples that aren't corrupted in any way e.g. because only one of
742		# the two tuples in the update chain needs to be corrupted for the
743		# test, or because one update chain is being made to erroneously
744		# point into the middle of another that has nothing wrong with it.
745		# In all such cases we need not write the tuple back to the file.
746		next;
747	}
748
749	write_tuple($file, $offset, $tup) if defined $tup;
750}
751close($file)
752  or BAIL_OUT("close failed: $!");
753$node->start;
754
755# Run pg_amcheck against the corrupt table with epoch=0, comparing actual
756# corruption messages against the expected messages
757$node->command_checks_all(
758	[ 'pg_amcheck', '--no-dependent-indexes', '--port' => $port, 'postgres' ],
759	2, [@expected], [], 'Expected corruption message output');
760$node->safe_psql(
761	'postgres', qq(
762                        COMMIT PREPARED 'in_progress_tx';
763        ));
764
765$node->teardown_node;
766$node->clean_node;
767
768done_testing();
