4use strict;
5use warnings FATAL => 'all';
6use PostgreSQL::Test::Utils;
7use Test::More;
8
9my $tempdir = PostgreSQL::Test::Utils::tempdir;
10
11program_help_ok('pg_verifybackup');
12program_version_ok('pg_verifybackup');
13program_options_handling_ok('pg_verifybackup');
14
15command_fails_like(
16	['pg_verifybackup'],
17	qr/no backup directory specified/,
18	'target directory must be specified');
19command_fails_like(
20	[ 'pg_verifybackup', $tempdir ],
21	qr/could not open file.*\/backup_manifest\"/,
22	'pg_verifybackup requires a manifest');
23command_fails_like(
24	[ 'pg_verifybackup', $tempdir, $tempdir ],
25	qr/too many command-line arguments/,
26	'multiple target directories not allowed');
27
28# create fake manifest file
29open(my $fh, '>', "$tempdir/backup_manifest") || die "open: $!";
30close($fh);
31
32# but then try to use an alternate, nonexisting manifest
33command_fails_like(
34	[
35		'pg_verifybackup',
36		'--manifest-path' => "$tempdir/not_the_manifest",
37		$tempdir,
38	],
39	qr/could not open file.*\/not_the_manifest\"/,
40	'pg_verifybackup respects -m flag');
41
42done_testing();
