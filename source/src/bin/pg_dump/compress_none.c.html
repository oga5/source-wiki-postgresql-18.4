26 * Private routines

28
29static void
30ReadDataFromArchiveNone(ArchiveHandle *AH, CompressorState *cs)
31{
32	size_t		cnt;
33	char	   *buf;
34	size_t		buflen;
35
36	buflen = DEFAULT_IO_BUFFER_SIZE;
37	buf = pg_malloc(buflen);
38
39	while ((cnt = cs->readF(AH, &buf, &buflen)))
40	{
41		ahwrite(buf, 1, cnt, AH);
42	}
43
44	free(buf);
45}
46
47
48static void
49WriteDataToArchiveNone(ArchiveHandle *AH, CompressorState *cs,
50					   const void *data, size_t dLen)
51{
52	cs->writeF(AH, data, dLen);
53}
54
55static void
56EndCompressorNone(ArchiveHandle *AH, CompressorState *cs)
57{
58	/* no op */
59}
60
61/*
62 * Public interface
63 */
64
65void
66InitCompressorNone(CompressorState *cs,
67				   const pg_compress_specification compression_spec)
68{
69	cs->readData = ReadDataFromArchiveNone;
70	cs->writeData = WriteDataToArchiveNone;
71	cs->end = EndCompressorNone;
72
73	cs->compression_spec = compression_spec;
74}

86static size_t
87read_none(void *ptr, size_t size, CompressFileHandle *CFH)
88{
89	FILE	   *fp = (FILE *) CFH->private_data;
90	size_t		ret;
91
92	ret = fread(ptr, 1, size, fp);
93	if (ferror(fp))
94		pg_fatal("could not read from input file: %m");
95
96	return ret;
97}
98
99static void
100write_none(const void *ptr, size_t size, CompressFileHandle *CFH)
101{
102	size_t		ret;
103
104	errno = 0;
105	ret = fwrite(ptr, 1, size, (FILE *) CFH->private_data);
106	if (ret != size)
107	{
108		errno = (errno) ? errno : ENOSPC;
109		pg_fatal("could not write to file: %m");
110	}
111}
112
113static const char *
114get_error_none(CompressFileHandle *CFH)
115{
116	return strerror(errno);
117}
118
119static char *
120gets_none(char *ptr, int size, CompressFileHandle *CFH)
121{
122	return fgets(ptr, size, (FILE *) CFH->private_data);
123}
124
125static int
126getc_none(CompressFileHandle *CFH)
127{
128	FILE	   *fp = (FILE *) CFH->private_data;
129	int			ret;
130
131	ret = fgetc(fp);
132	if (ret == EOF)
133	{
134		if (!feof(fp))
135			pg_fatal("could not read from input file: %m");
136		else
137			pg_fatal("could not read from input file: end of file");
138	}
139
140	return ret;
141}
142
143static bool
144close_none(CompressFileHandle *CFH)
145{
146	FILE	   *fp = (FILE *) CFH->private_data;
147	int			ret = 0;
148
149	CFH->private_data = NULL;
150
151	if (fp)
152	{
153		errno = 0;
154		ret = fclose(fp);
155		if (ret != 0)
156			pg_log_error("could not close file: %m");
157	}
158
159	return ret == 0;
160}
161
162static bool
163eof_none(CompressFileHandle *CFH)
164{
165	return feof((FILE *) CFH->private_data) != 0;
166}
167
168static bool
169open_none(const char *path, int fd, const char *mode, CompressFileHandle *CFH)
170{
171	Assert(CFH->private_data == NULL);
172
173	if (fd >= 0)
174		CFH->private_data = fdopen(dup(fd), mode);
175	else
176		CFH->private_data = fopen(path, mode);
177
178	if (CFH->private_data == NULL)
179		return false;
180
181	return true;
182}
183
184static bool
185open_write_none(const char *path, const char *mode, CompressFileHandle *CFH)
186{
187	Assert(CFH->private_data == NULL);
188
189	CFH->private_data = fopen(path, mode);
190	if (CFH->private_data == NULL)
191		return false;
192
193	return true;
194}
195
196/*
197 * Public interface
198 */
199
200void
201InitCompressFileHandleNone(CompressFileHandle *CFH,
202						   const pg_compress_specification compression_spec)
203{
204	CFH->open_func = open_none;
205	CFH->open_write_func = open_write_none;
206	CFH->read_func = read_none;
207	CFH->write_func = write_none;
208	CFH->gets_func = gets_none;
209	CFH->getc_func = getc_none;
210	CFH->close_func = close_none;
211	CFH->eof_func = eof_none;
212	CFH->get_error_func = get_error_none;
213
214	CFH->private_data = NULL;
215}
