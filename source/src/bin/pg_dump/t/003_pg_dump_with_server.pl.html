4use strict;
5use warnings FATAL => 'all';
6
7use PostgreSQL::Test::Cluster;
8use PostgreSQL::Test::Utils;
9use Test::More;
10
11my $tempdir = PostgreSQL::Test::Utils::tempdir;
12
13my $node = PostgreSQL::Test::Cluster->new('main');
14my $port = $node->port;
15
16$node->init;
17$node->start;
18
19#########################################
20# pg_dumpall: newline in database name
21
22$node->safe_psql('postgres', qq{CREATE DATABASE "regress_\nattack"});
23
24my (@cmd, $stdout, $stderr);
25@cmd = ("pg_dumpall", '--port' => $port, '--exclude-database=postgres');
26print("# Running: " . join(" ", @cmd) . "\n");
27my $result = IPC::Run::run \@cmd, '>' => \$stdout, '2>' => \$stderr;
28ok(!$result, "newline in dbname: exit code not 0");
29like(
30	$stderr,
31	qr/shell command argument contains a newline/,
32	"newline in dbname: stderr matches");
33unlike($stdout, qr/^attack/m, "newline in dbname: no comment escape");
34
35#########################################
36# Verify that dumping foreign data includes only foreign tables of
37# matching servers
38
39$node->safe_psql('postgres', "CREATE FOREIGN DATA WRAPPER dummy");
40$node->safe_psql('postgres', "CREATE SERVER s0 FOREIGN DATA WRAPPER dummy");
41$node->safe_psql('postgres', "CREATE SERVER s1 FOREIGN DATA WRAPPER dummy");
42$node->safe_psql('postgres', "CREATE SERVER s2 FOREIGN DATA WRAPPER dummy");
43$node->safe_psql('postgres', "CREATE FOREIGN TABLE t0 (a int) SERVER s0");
44$node->safe_psql('postgres', "CREATE FOREIGN TABLE t1 (a int) SERVER s1");
45
46command_fails_like(
47	[
48		"pg_dump",
49		'--port' => $port,
50		'--include-foreign-data' => 's0',
51		'postgres'
52	],
53	qr/foreign-data wrapper \"dummy\" has no handler\r?\npg_dump: detail: Query was: .*t0/,
54	"correctly fails to dump a foreign table from a dummy FDW");
55
56command_ok(
57	[
58		"pg_dump",
59		'--port' => $port,
60		'--data-only',
61		'--include-foreign-data' => 's2',
62		'postgres'
63	],
64	"dump foreign server with no tables");
65
66done_testing();
