153if (check_pg_config("#define HAVE_LIBZ 1"))
154{
155	command_fails_like(
156		[ 'pg_dump', '-Z', '15' ],
157		qr/\Qpg_dump: error: invalid compression specification: compression algorithm "gzip" expects a compression level between 1 and 9 (default at -1)\E/,
158		'pg_dump: invalid compression specification: must be in range');
159
160	command_fails_like(
161		[ 'pg_dump', '--compress', '1', '--format', 'tar' ],
162		qr/\Qpg_dump: error: compression is not supported by tar archive format\E/,
163		'pg_dump: compression is not supported by tar archive format');
164
165	command_fails_like(
166		[ 'pg_dump', '-Z', 'gzip:nonInt' ],
167		qr/\Qpg_dump: error: invalid compression specification: unrecognized compression option: "nonInt"\E/,
168		'pg_dump: invalid compression specification: must be an integer');
169}
170else
171{
172	# --jobs > 1 forces an error with tar format.
173	command_fails_like(
174		[ 'pg_dump', '--format', 'tar', '-j3' ],
175		qr/\Qpg_dump: error: parallel backup only supported by the directory format\E/,
176		'pg_dump: warning: parallel backup not supported by tar format');
177
178	command_fails_like(
179		[ 'pg_dump', '-Z', 'gzip:nonInt', '--format', 'tar', '-j2' ],
180		qr/\Qpg_dump: error: invalid compression specification: unrecognized compression option\E/,
181		'pg_dump: invalid compression specification: must be an integer');
182}
