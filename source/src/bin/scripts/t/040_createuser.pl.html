4use strict;
5use warnings FATAL => 'all';
6
7use PostgreSQL::Test::Cluster;
8use PostgreSQL::Test::Utils;
9use Test::More;
10
11program_help_ok('createuser');
12program_version_ok('createuser');
13program_options_handling_ok('createuser');
14
15my $node = PostgreSQL::Test::Cluster->new('main');
16$node->init;
17$node->start;
18
19$node->issues_sql_like(
20	[ 'createuser', 'regress_user1' ],
21	qr/statement: CREATE ROLE regress_user1 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS;/,
22	'SQL CREATE USER run');
23$node->issues_sql_like(
24	[ 'createuser', '--no-login', 'regress_role1' ],
25	qr/statement: CREATE ROLE regress_role1 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT NOLOGIN NOREPLICATION NOBYPASSRLS;/,
26	'create a non-login role');
27$node->issues_sql_like(
28	[ 'createuser', '--createrole', 'regress user2' ],
29	qr/statement: CREATE ROLE "regress user2" NOSUPERUSER NOCREATEDB CREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS;/,
30	'create a CREATEROLE user');
31$node->issues_sql_like(
32	[ 'createuser', '--superuser', 'regress_user3' ],
33	qr/statement: CREATE ROLE regress_user3 SUPERUSER CREATEDB CREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS;/,
34	'create a superuser');
35$node->issues_sql_like(
36	[
37		'createuser',
38		'--with-admin' => 'regress_user1',
39		'--with-admin' => 'regress user2',
40		'regress user #4'
41	],
42	qr/statement: CREATE ROLE "regress user #4" NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS ADMIN regress_user1,"regress user2";/,
43	'add a role as a member with admin option of the newly created role');
44$node->issues_sql_like(
45	[
46		'createuser',
47		'REGRESS_USER5',
48		'--with-member' => 'regress_user3',
49		'--with-member' => 'regress user #4'
50	],
51	qr/statement: CREATE ROLE "REGRESS_USER5" NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS ROLE regress_user3,"regress user #4";/,
52	'add a role as a member of the newly created role');
53$node->issues_sql_like(
54	[ 'createuser', '--valid-until' => '2029 12 31', 'regress_user6' ],
55	qr/statement: CREATE ROLE regress_user6 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS VALID UNTIL \'2029 12 31\';/,
56	'create a role with a password expiration date');
57$node->issues_sql_like(
58	[ 'createuser', '--bypassrls', 'regress_user7' ],
59	qr/statement: CREATE ROLE regress_user7 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION BYPASSRLS;/,
60	'create a BYPASSRLS role');
61$node->issues_sql_like(
62	[ 'createuser', '--no-bypassrls', 'regress_user8' ],
63	qr/statement: CREATE ROLE regress_user8 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS;/,
64	'create a role without BYPASSRLS');
65$node->issues_sql_like(
66	[ 'createuser', '--with-admin' => 'regress_user1', 'regress_user9' ],
67	qr/statement: CREATE ROLE regress_user9 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS ADMIN regress_user1;/,
68	'--with-admin');
69$node->issues_sql_like(
70	[ 'createuser', '--with-member' => 'regress_user1', 'regress_user10' ],
71	qr/statement: CREATE ROLE regress_user10 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS ROLE regress_user1;/,
72	'--with-member');
73$node->issues_sql_like(
74	[ 'createuser', '--role' => 'regress_user1', 'regress_user11' ],
75	qr/statement: CREATE ROLE regress_user11 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS IN ROLE regress_user1;/,
76	'--role');
77$node->issues_sql_like(
78	[ 'createuser', 'regress_user12', '--member-of' => 'regress_user1' ],
79	qr/statement: CREATE ROLE regress_user12 NOSUPERUSER NOCREATEDB NOCREATEROLE INHERIT LOGIN NOREPLICATION NOBYPASSRLS IN ROLE regress_user1;/,
80	'--member-of');
81
82$node->command_fails([ 'createuser', 'regress_user1' ],
83	'fails if role already exists');
84$node->command_fails(
85	[
86		'createuser',
87		'regress_user1',
88		'--with-member' => 'regress_user2',
89		'regress_user3'
90	],
91	'fails for too many non-options');
92
93done_testing();
