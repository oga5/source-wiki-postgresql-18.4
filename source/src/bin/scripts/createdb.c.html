20static void help(const char *progname);
21
22
23int
24main(int argc, char *argv[])
25{
26	static struct option long_options[] = {
27		{"host", required_argument, NULL, 'h'},
28		{"port", required_argument, NULL, 'p'},
29		{"username", required_argument, NULL, 'U'},
30		{"no-password", no_argument, NULL, 'w'},
31		{"password", no_argument, NULL, 'W'},
32		{"echo", no_argument, NULL, 'e'},
33		{"owner", required_argument, NULL, 'O'},
34		{"tablespace", required_argument, NULL, 'D'},
35		{"template", required_argument, NULL, 'T'},
36		{"encoding", required_argument, NULL, 'E'},
37		{"strategy", required_argument, NULL, 'S'},
38		{"lc-collate", required_argument, NULL, 1},
39		{"lc-ctype", required_argument, NULL, 2},
40		{"locale", required_argument, NULL, 'l'},
41		{"maintenance-db", required_argument, NULL, 3},
42		{"locale-provider", required_argument, NULL, 4},
43		{"builtin-locale", required_argument, NULL, 5},
44		{"icu-locale", required_argument, NULL, 6},
45		{"icu-rules", required_argument, NULL, 7},
46		{NULL, 0, NULL, 0}
47	};
48
49	const char *progname;
50	int			optindex;
51	int			c;
52
53	const char *dbname = NULL;
54	const char *maintenance_db = NULL;
55	char	   *comment = NULL;
56	char	   *host = NULL;
57	char	   *port = NULL;
58	char	   *username = NULL;
59	enum trivalue prompt_password = TRI_DEFAULT;
60	ConnParams	cparams;
61	bool		echo = false;
62	char	   *owner = NULL;
63	char	   *tablespace = NULL;
64	char	   *template = NULL;
65	char	   *encoding = NULL;
66	char	   *strategy = NULL;
67	char	   *lc_collate = NULL;
68	char	   *lc_ctype = NULL;
69	char	   *locale = NULL;
70	char	   *locale_provider = NULL;
71	char	   *builtin_locale = NULL;
72	char	   *icu_locale = NULL;
73	char	   *icu_rules = NULL;
74
75	PQExpBufferData sql;
76
77	PGconn	   *conn;
78	PGresult   *result;
79
80	pg_logging_init(argv[0]);
81	progname = get_progname(argv[0]);
82	set_pglocale_pgservice(argv[0], PG_TEXTDOMAIN("pgscripts"));
83
84	handle_help_version_opts(argc, argv, "createdb", help);
85
86	while ((c = getopt_long(argc, argv, "D:eE:h:l:O:p:S:T:U:wW", long_options, &optindex)) != -1)
87	{
88		switch (c)
89		{
90			case 'D':
91				tablespace = pg_strdup(optarg);
92				break;
93			case 'e':
94				echo = true;
95				break;
96			case 'E':
97				encoding = pg_strdup(optarg);
98				break;
99			case 'h':
100				host = pg_strdup(optarg);
101				break;
102			case 'l':
103				locale = pg_strdup(optarg);
104				break;
105			case 'O':
106				owner = pg_strdup(optarg);
107				break;
108			case 'p':
109				port = pg_strdup(optarg);
110				break;
111			case 'S':
112				strategy = pg_strdup(optarg);
113				break;
114			case 'T':
115				template = pg_strdup(optarg);
116				break;
117			case 'U':
118				username = pg_strdup(optarg);
119				break;
120			case 'w':
121				prompt_password = TRI_NO;
122				break;
123			case 'W':
124				prompt_password = TRI_YES;
125				break;
126			case 1:
127				lc_collate = pg_strdup(optarg);
128				break;
129			case 2:
130				lc_ctype = pg_strdup(optarg);
131				break;
132			case 3:
133				maintenance_db = pg_strdup(optarg);
134				break;
135			case 4:
136				locale_provider = pg_strdup(optarg);
137				break;
138			case 5:
139				builtin_locale = pg_strdup(optarg);
140				break;
141			case 6:
142				icu_locale = pg_strdup(optarg);
143				break;
144			case 7:
145				icu_rules = pg_strdup(optarg);
146				break;
147			default:
148				/* getopt_long already emitted a complaint */
149				pg_log_error_hint("Try \"%s --help\" for more information.", progname);
150				exit(1);
151		}
152	}
153
154	switch (argc - optind)
155	{
156		case 0:
157			break;
158		case 1:
159			dbname = argv[optind];
160			break;
161		case 2:
162			dbname = argv[optind];
163			comment = argv[optind + 1];
164			break;
165		default:
166			pg_log_error("too many command-line arguments (first is \"%s\")",
167						 argv[optind + 2]);
168			pg_log_error_hint("Try \"%s --help\" for more information.", progname);
169			exit(1);
170	}
171
172	if (encoding)
173	{
174		if (pg_char_to_encoding(encoding) < 0)
175			pg_fatal("\"%s\" is not a valid encoding name", encoding);
176	}
177
178	if (dbname == NULL)
179	{
180		if (getenv("PGDATABASE"))
181			dbname = getenv("PGDATABASE");
182		else if (getenv("PGUSER"))
183			dbname = getenv("PGUSER");
184		else
185			dbname = get_user_name_or_exit(progname);
186	}
187
188	/* No point in trying to use postgres db when creating postgres db. */
189	if (maintenance_db == NULL && strcmp(dbname, "postgres") == 0)
190		maintenance_db = "template1";
191
192	cparams.dbname = maintenance_db;
193	cparams.pghost = host;
194	cparams.pgport = port;
195	cparams.pguser = username;
196	cparams.prompt_password = prompt_password;
197	cparams.override_dbname = NULL;
198
199	conn = connectMaintenanceDatabase(&cparams, progname, echo);
200
201	setFmtEncoding(PQclientEncoding(conn));
202
203	initPQExpBuffer(&sql);
204
205	appendPQExpBuffer(&sql, "CREATE DATABASE %s",
206					  fmtId(dbname));
207
208	if (owner)
209		appendPQExpBuffer(&sql, " OWNER %s", fmtId(owner));
210	if (tablespace)
211		appendPQExpBuffer(&sql, " TABLESPACE %s", fmtId(tablespace));
212	if (encoding)
213	{
214		appendPQExpBufferStr(&sql, " ENCODING ");
215		appendStringLiteralConn(&sql, encoding, conn);
216	}
217	if (strategy)
218		appendPQExpBuffer(&sql, " STRATEGY %s", fmtId(strategy));
219	if (template)
220		appendPQExpBuffer(&sql, " TEMPLATE %s", fmtId(template));
221	if (locale)
222	{
223		appendPQExpBufferStr(&sql, " LOCALE ");
224		appendStringLiteralConn(&sql, locale, conn);
225	}
226	if (builtin_locale)
227	{
228		appendPQExpBufferStr(&sql, " BUILTIN_LOCALE ");
229		appendStringLiteralConn(&sql, builtin_locale, conn);
230	}
231	if (lc_collate)
232	{
233		appendPQExpBufferStr(&sql, " LC_COLLATE ");
234		appendStringLiteralConn(&sql, lc_collate, conn);
235	}
236	if (lc_ctype)
237	{
238		appendPQExpBufferStr(&sql, " LC_CTYPE ");
239		appendStringLiteralConn(&sql, lc_ctype, conn);
240	}
241	if (locale_provider)
242		appendPQExpBuffer(&sql, " LOCALE_PROVIDER %s", fmtId(locale_provider));
243	if (icu_locale)
244	{
245		appendPQExpBufferStr(&sql, " ICU_LOCALE ");
246		appendStringLiteralConn(&sql, icu_locale, conn);
247	}
248	if (icu_rules)
249	{
250		appendPQExpBufferStr(&sql, " ICU_RULES ");
251		appendStringLiteralConn(&sql, icu_rules, conn);
252	}
253
254	appendPQExpBufferChar(&sql, ';');
255
256	if (echo)
257		printf("%s\n", sql.data);
258	result = PQexec(conn, sql.data);
259
260	if (PQresultStatus(result) != PGRES_COMMAND_OK)
261	{
262		pg_log_error("database creation failed: %s", PQerrorMessage(conn));
263		PQfinish(conn);
264		exit(1);
265	}
266
267	PQclear(result);
268
269	if (comment)
270	{
271		printfPQExpBuffer(&sql, "COMMENT ON DATABASE %s IS ", fmtId(dbname));
272		appendStringLiteralConn(&sql, comment, conn);
273		appendPQExpBufferChar(&sql, ';');
274
275		if (echo)
276			printf("%s\n", sql.data);
277		result = PQexec(conn, sql.data);
278
279		if (PQresultStatus(result) != PGRES_COMMAND_OK)
280		{
281			pg_log_error("comment creation failed (database was created): %s",
282						 PQerrorMessage(conn));
283			PQfinish(conn);
284			exit(1);
285		}
286
287		PQclear(result);
288	}
289
290	PQfinish(conn);
291
292	exit(0);
293}

296static void
297help(const char *progname)
298{
299	printf(_("%s creates a PostgreSQL database.\n\n"), progname);
300	printf(_("Usage:\n"));
301	printf(_("  %s [OPTION]... [DBNAME] [DESCRIPTION]\n"), progname);
302	printf(_("\nOptions:\n"));
303	printf(_("  -D, --tablespace=TABLESPACE  default tablespace for the database\n"));
304	printf(_("  -e, --echo                   show the commands being sent to the server\n"));
305	printf(_("  -E, --encoding=ENCODING      encoding for the database\n"));
306	printf(_("  -l, --locale=LOCALE          locale settings for the database\n"));
307	printf(_("      --lc-collate=LOCALE      LC_COLLATE setting for the database\n"));
308	printf(_("      --lc-ctype=LOCALE        LC_CTYPE setting for the database\n"));
309	printf(_("      --builtin-locale=LOCALE  builtin locale setting for the database\n"));
310	printf(_("      --icu-locale=LOCALE      ICU locale setting for the database\n"));
311	printf(_("      --icu-rules=RULES        ICU rules setting for the database\n"));
312	printf(_("      --locale-provider={builtin|libc|icu}\n"
313			 "                               locale provider for the database's default collation\n"));
314	printf(_("  -O, --owner=OWNER            database user to own the new database\n"));
315	printf(_("  -S, --strategy=STRATEGY      database creation strategy wal_log or file_copy\n"));
316	printf(_("  -T, --template=TEMPLATE      template database to copy\n"));
317	printf(_("  -V, --version                output version information, then exit\n"));
318	printf(_("  -?, --help                   show this help, then exit\n"));
319	printf(_("\nConnection options:\n"));
320	printf(_("  -h, --host=HOSTNAME          database server host or socket directory\n"));
321	printf(_("  -p, --port=PORT              database server port\n"));
322	printf(_("  -U, --username=USERNAME      user name to connect as\n"));
323	printf(_("  -w, --no-password            never prompt for password\n"));
324	printf(_("  -W, --password               force password prompt\n"));
325	printf(_("  --maintenance-db=DBNAME      alternate maintenance database\n"));
326	printf(_("\nBy default, a database with the same name as the current user is created.\n"));
327	printf(_("\nReport bugs to <%s>.\n"), PACKAGE_BUGREPORT);
328	printf(_("%s home page: <%s>\n"), PACKAGE_NAME, PACKAGE_URL);
329}
