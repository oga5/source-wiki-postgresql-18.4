38static const RmgrDescData RmgrDescTable[RM_N_BUILTIN_IDS] = {
39#include "access/rmgrlist.h"
40};
41
42#define CUSTOM_NUMERIC_NAME_LEN sizeof("custom###")
43
44static char CustomNumericNames[RM_N_CUSTOM_IDS][CUSTOM_NUMERIC_NAME_LEN] = {0};
45static RmgrDescData CustomRmgrDesc[RM_N_CUSTOM_IDS] = {0};
46static bool CustomRmgrDescInitialized = false;
47
48/*
49 * No information on custom resource managers; just print the ID.
50 */
51static void
52default_desc(StringInfo buf, XLogReaderState *record)
53{
54	appendStringInfo(buf, "rmid: %d", XLogRecGetRmid(record));
55}
56
57/*
58 * No information on custom resource managers; just return NULL and let the
59 * caller handle it.
60 */
61static const char *
62default_identify(uint8 info)
63{
64	return NULL;
65}
66
67/*
68 * We are unable to get the real name of a custom rmgr because the module is
69 * not loaded. Generate a table of rmgrs with numeric names of the form
70 * "custom###", where "###" is the 3-digit resource manager ID.
71 */
72static void
73initialize_custom_rmgrs(void)
74{
75	for (int i = 0; i < RM_N_CUSTOM_IDS; i++)
76	{
77		snprintf(CustomNumericNames[i], CUSTOM_NUMERIC_NAME_LEN,
78				 "custom%03d", i + RM_MIN_CUSTOM_ID);
79		CustomRmgrDesc[i].rm_name = CustomNumericNames[i];
80		CustomRmgrDesc[i].rm_desc = default_desc;
81		CustomRmgrDesc[i].rm_identify = default_identify;
82	}
83	CustomRmgrDescInitialized = true;
84}
85
86const RmgrDescData *
87GetRmgrDesc(RmgrId rmid)
88{
89	Assert(RmgrIdIsValid(rmid));
90
91	if (RmgrIdIsBuiltin(rmid))
92		return &RmgrDescTable[rmid];
93	else
94	{
95		if (!CustomRmgrDescInitialized)
96			initialize_custom_rmgrs();
97		return &CustomRmgrDesc[rmid - RM_MIN_CUSTOM_ID];
98	}
99}
