240# Test for: Display a message that we're skipping data if `from`
241# wasn't a pointer to the start of a record.
242{
243	# Construct a new LSN that is one byte past the original
244	# start_lsn.
245	my ($part1, $part2) = split qr{/}, $start_lsn;
246	my $lsn2 = hex $part2;
247	$lsn2++;
248	my $new_start = sprintf("%s/%X", $part1, $lsn2);
249
250	my ($stdout, $stderr);
251
252	my $result = IPC::Run::run [
253		'pg_waldump',
254		'--start' => $new_start,
255		$node->data_dir . '/pg_wal/' . $start_walfile
256	  ],
257	  '>' => \$stdout,
258	  '2>' => \$stderr;
259	ok($result, "runs with start segment and start LSN specified");
260	like($stderr, qr/first record is after/, 'info message printed');
261}

264# Helper function to test various options.  Pass options as arguments.
265# Output lines are returned as array.
266sub test_pg_waldump
267{
268	local $Test::Builder::Level = $Test::Builder::Level + 1;
269	my @opts = @_;
270
271	my ($stdout, $stderr);
272
273	my $result = IPC::Run::run [
274		'pg_waldump',
275		'--path' => $node->data_dir,
276		'--start' => $start_lsn,
277		'--end' => $end_lsn,
278		@opts
279	  ],
280	  '>' => \$stdout,
281	  '2>' => \$stderr;
282	ok($result, "pg_waldump @opts: runs ok");
283	is($stderr, '', "pg_waldump @opts: no stderr");
284	my @lines = split /\n/, $stdout;
285	ok(@lines > 0, "pg_waldump @opts: some lines are output");
286	return @lines;
287}
