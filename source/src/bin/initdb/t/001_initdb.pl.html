4# To test successful data directory creation with an additional feature, first
5# try to elaborate the "successful creation" test instead of adding a test.
6# Successful initdb consumes much time and I/O.
7
8use strict;
9use warnings FATAL => 'all';
10use Fcntl ':mode';
11use File::stat qw{lstat};
12use PostgreSQL::Test::Cluster;
13use PostgreSQL::Test::Utils;
14use Test::More;
15
16my $tempdir = PostgreSQL::Test::Utils::tempdir;
17my $xlogdir = "$tempdir/pgxlog";
18my $datadir = "$tempdir/data";
19my $supports_syncfs = check_pg_config("#define HAVE_SYNCFS 1");
20
21program_help_ok('initdb');
22program_version_ok('initdb');
23program_options_handling_ok('initdb');
24
25command_fails([ 'initdb', '--sync-only', "$tempdir/nonexistent" ],
26	'sync missing data directory');
27
28mkdir $xlogdir;
29mkdir "$xlogdir/lost+found";
30command_fails([ 'initdb', '--waldir' => $xlogdir, $datadir ],
31	'existing nonempty xlog directory');
32rmdir "$xlogdir/lost+found";
33command_fails(
34	[ 'initdb', '--waldir' => 'pgxlog', $datadir ],
35	'relative xlog directory not allowed');
36
37command_fails([ 'initdb', '--username' => 'pg_test', $datadir ],
38	'role names cannot begin with "pg_"');
39
40mkdir $datadir;
41
42# make sure we run one successful test without a TZ setting so we test
43# initdb's time zone setting code
44{
45
46	# delete local only works from perl 5.12, so use the older way to do this
47	local (%ENV) = %ENV;
48	delete $ENV{TZ};
49
50	# while we are here, also exercise --text-search-config and --set options
51	command_ok(
52		[
53			'initdb', '--no-sync',
54			'--text-search-config' => 'german',
55			'--set' => 'default_text_search_config=german',
56			'--waldir' => $xlogdir,
57			$datadir
58		],
59		'successful creation');
60
61	# Permissions on PGDATA should be default
62  SKIP:
63	{
64		skip "unix-style permissions not supported on Windows", 1
65		  if ($windows_os);
66
67		ok(check_mode_recursive($datadir, 0700, 0600),
68			"check PGDATA permissions");
69	}
70}
71
72# Control file should tell that data checksums are enabled by default.
73command_like(
74	[ 'pg_controldata', $datadir ],
75	qr/Data page checksum version:.*1/,
76	'checksums are enabled in control file');
77
78command_ok([ 'initdb', '--sync-only', $datadir ], 'sync only');
79command_ok([ 'initdb', '--sync-only', '--no-sync-data-files', $datadir ],
80	'--no-sync-data-files');
81command_fails([ 'initdb', $datadir ], 'existing data directory');
82
83if ($supports_syncfs)
84{
85	command_ok(
86		[ 'initdb', '--sync-only', $datadir, '--sync-method' => 'syncfs' ],
87		'sync method syncfs');
88}
89else
90{
91	command_fails(
92		[ 'initdb', '--sync-only', $datadir, '--sync-method' => 'syncfs' ],
93		'sync method syncfs');
94}
95
96# Check group access on PGDATA
97SKIP:
98{
99	skip "unix-style permissions not supported on Windows", 2
100	  if ($windows_os || $Config::Config{osname} eq 'cygwin');
101
102	# Init a new db with group access
103	my $datadir_group = "$tempdir/data_group";
104
105	command_ok([ 'initdb', '--allow-group-access', $datadir_group ],
106		'successful creation with group access');
107
108	ok(check_mode_recursive($datadir_group, 0750, 0640),
109		'check PGDATA permissions');
110}

112# Locale provider tests
113
114if ($ENV{with_icu} eq 'yes')
115{
116	command_fails_like(
117		[
118			'initdb', '--no-sync',
119			'--locale-provider' => 'icu',
120			"$tempdir/data2"
121		],
122		qr/initdb: error: locale must be specified if provider is icu/,
123		'locale provider ICU requires --icu-locale');
124
125	command_ok(
126		[
127			'initdb', '--no-sync',
128			'--locale-provider' => 'icu',
129			'--icu-locale' => 'en',
130			"$tempdir/data3"
131		],
132		'option --icu-locale');
133
134	command_like(
135		[
136			'initdb', '--no-sync',
137			'--auth' => 'trust',
138			'--locale-provider' => 'icu',
139			'--locale' => 'und',
140			'--lc-collate' => 'C',
141			'--lc-ctype' => 'C',
142			'--lc-messages' => 'C',
143			'--lc-numeric' => 'C',
144			'--lc-monetary' => 'C',
145			'--lc-time' => 'C',
146			"$tempdir/data4"
147		],
148		qr/^\s+default collation:\s+und\n/ms,
149		'options --locale-provider=icu --locale=und --lc-*=C');
150
151	command_fails_like(
152		[
153			'initdb', '--no-sync',
154			'--locale-provider' => 'icu',
155			'--icu-locale' => '@colNumeric=lower',
156			"$tempdir/dataX"
157		],
158		qr/could not open collator for locale/,
159		'fails for invalid ICU locale');
160
161	command_fails_like(
162		[
163			'initdb', '--no-sync',
164			'--locale-provider' => 'icu',
165			'--encoding' => 'SQL_ASCII',
166			'--icu-locale' => 'en',
167			"$tempdir/dataX"
168		],
169		qr/error: encoding mismatch/,
170		'fails for encoding not supported by ICU');
171
172	command_fails_like(
173		[
174			'initdb', '--no-sync',
175			'--locale-provider' => 'icu',
176			'--icu-locale' => 'nonsense-nowhere',
177			"$tempdir/dataX"
178		],
179		qr/error: locale "nonsense-nowhere" has unknown language "nonsense"/,
180		'fails for nonsense language');
181
182	command_fails_like(
183		[
184			'initdb', '--no-sync',
185			'--locale-provider' => 'icu',
186			'--icu-locale' => '@colNumeric=lower',
187			"$tempdir/dataX"
188		],
189		qr/could not open collator for locale "und-u-kn-lower": U_ILLEGAL_ARGUMENT_ERROR/,
190		'fails for invalid collation argument');
191}
192else
193{
194	command_fails(
195		[
196			'initdb', '--no-sync',
197			'--locale-provider' => 'icu',
198			"$tempdir/data2"
199		],
200		'locale provider ICU fails since no ICU support');
201}

290command_fails(

294		"$tempdir/dataX"
295	],
296	'fails for invalid --set option');
297
298# Make sure multiple invocations of --set parameters are added case
299# insensitive.
300command_ok(
301	[
302		'initdb', '--no-sync',
303		'--set' => 'work_mem=128',
304		'--set' => 'Work_Mem=256',
305		'--set' => 'WORK_MEM=512',
306		"$tempdir/dataY"
307	],
308	'multiple --set options with different case');
309
310my $conf = slurp_file("$tempdir/dataY/postgresql.conf");
311ok($conf !~ qr/^WORK_MEM = /m, "WORK_MEM should not be configured");
312ok($conf !~ qr/^Work_Mem = /m, "Work_Mem should not be configured");
313ok($conf =~ qr/^work_mem = 512/m, "work_mem should be in config");
314
315# Test the no-data-checksums flag
316my $datadir_nochecksums = "$tempdir/data_no_checksums";
317
318command_ok([ 'initdb', '--no-data-checksums', $datadir_nochecksums ],
319	'successful creation without data checksums');
320
321# Control file should tell that data checksums are disabled.
322command_like(
323	[ 'pg_controldata', $datadir_nochecksums ],
324	qr/Data page checksum version:.*0/,
325	'checksums are disabled in control file');
326
327# pg_checksums fails with checksums disabled. This is
328# not part of the tests included in pg_checksums to save from
329# the creation of an extra instance.
330command_fails(
331	[ 'pg_checksums', '--pgdata' => $datadir_nochecksums ],
332	"pg_checksums fails with data checksum disabled");
333
334done_testing();
