3use strict;
4use warnings FATAL => 'all';
5use File::Compare;
6use PostgreSQL::Test::Cluster;
7use PostgreSQL::Test::Utils;
8use Test::More;
9
10# Set up a new database instance.
11my $node1 = PostgreSQL::Test::Cluster->new('node1');
12$node1->init(has_archiving => 1, allows_streaming => 1);
13$node1->append_conf('postgresql.conf', 'summarize_wal = on');
14$node1->start;
15
16# Create a table and insert a few test rows into it. VACUUM FREEZE it so that
17# autovacuum doesn't induce any future modifications unexpectedly. Then
18# trigger a checkpoint.
19$node1->safe_psql('postgres', <<EOM);
20CREATE TABLE mytable (a int, b text);
21INSERT INTO mytable
22SELECT
23	g, random()::text||random()::text||random()::text||random()::text
24FROM
25	generate_series(1, 400) g;
26VACUUM FREEZE;
27EOM
28
29# Record the current WAL insert LSN.
30my $base_lsn = $node1->safe_psql('postgres', <<EOM);
31SELECT pg_current_wal_insert_lsn()
32EOM
33note("just after insert, WAL insert LSN is $base_lsn");
34
35# Now perform a CHECKPOINT.
36$node1->safe_psql('postgres', <<EOM);
37CHECKPOINT;
38EOM
39
40# Wait for a new summary to show up, one that includes the inserts we just did.
41my $result = $node1->poll_query_until('postgres', <<EOM);
42SELECT EXISTS (
43    SELECT * from pg_available_wal_summaries()
44    WHERE end_lsn >= '$base_lsn'
45)
46EOM
47ok($result, "WAL summarization caught up after insert");
48
49# The WAL summarizer should have generated some IO statistics.
50$node1->poll_query_until(
51	'postgres',
52	q{SELECT sum(reads) > 0 FROM pg_stat_io
53   WHERE backend_type = 'walsummarizer' AND object = 'wal'})
54  or die
55  "Timed out while waiting for WAL summarizer to generate statistics for WAL reads";
56
57# Find the highest LSN that is summarized on disk.
58my $summarized_lsn = $node1->safe_psql('postgres', <<EOM);
59SELECT MAX(end_lsn) AS summarized_lsn FROM pg_available_wal_summaries()
60EOM
61note("after insert, summarized through $summarized_lsn");
62
63# Update a row in the first block of the table and trigger a checkpoint.
64$node1->safe_psql('postgres', <<EOM);
65UPDATE mytable SET b = 'abcdefghijklmnopqrstuvwxyz' || b || '01234567890'
66	WHERE a = 2;
67CHECKPOINT;
68EOM
69
70# Wait for a new summary to show up.
71$result = $node1->poll_query_until('postgres', <<EOM);
72SELECT EXISTS (
73    SELECT * from pg_available_wal_summaries()
74    WHERE end_lsn > '$summarized_lsn'
75)
76EOM
77ok($result, "got new WAL summary after update");
78
79# Figure out the exact details for the new summary file.
80my $details = $node1->safe_psql('postgres', <<EOM);
81SELECT tli, start_lsn, end_lsn from pg_available_wal_summaries()
82	WHERE end_lsn > '$summarized_lsn'
83EOM
84my @lines = split(/\n/, $details);
85is(0 + @lines, 1, "got exactly one new WAL summary");
86my ($tli, $start_lsn, $end_lsn) = split(/\|/, $lines[0]);
87note("examining summary for TLI $tli from $start_lsn to $end_lsn");
88
89# Reconstruct the full pathname for the WAL summary file.
90my $filename = sprintf "%s/pg_wal/summaries/%08s%08s%08s%08s%08s.summary",
91  $node1->data_dir, $tli,
92  split(m@/@, $start_lsn),
93  split(m@/@, $end_lsn);
94ok(-f $filename, "WAL summary file exists");
95
96# Run pg_walsummary on it. We expect exactly two blocks to be modified,
97# block 0 and one other.
98my ($stdout, $stderr) = run_command([ 'pg_walsummary', '-i', $filename ]);
99note($stdout);
100@lines = split(/\n/, $stdout);
101like($stdout, qr/FORK main: block 0$/m, "stdout shows block 0 modified");
102is($stderr, '', 'stderr is empty');
103is(0 + @lines, 2, "UPDATE modified 2 blocks");
104
105done_testing();
