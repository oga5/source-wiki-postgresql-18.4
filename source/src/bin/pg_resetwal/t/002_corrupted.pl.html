4# Tests for handling a corrupted pg_control

6use strict;
7use warnings FATAL => 'all';
8
9use PostgreSQL::Test::Cluster;
10use PostgreSQL::Test::Utils;
11use Test::More;
12
13my $node = PostgreSQL::Test::Cluster->new('main');
14$node->init;
15
16my $pg_control = $node->data_dir . '/global/pg_control';
17my $size = -s $pg_control;
18
19# Read out the head of the file to get PG_CONTROL_VERSION in
20# particular.
21my $data;
22open my $fh, '<', $pg_control or BAIL_OUT($!);
23binmode $fh;
24read $fh, $data, 16 or die $!;
25close $fh;
26
27# Fill pg_control with zeros
28open $fh, '>', $pg_control or BAIL_OUT($!);
29binmode $fh;
30print $fh pack("x[$size]");
31close $fh;
32
33command_checks_all(
34	[ 'pg_resetwal', '--dry-run', $node->data_dir ],
35	0,
36	[qr/pg_control version number/],
37	[
38		qr/pg_resetwal: warning: pg_control exists but is broken or wrong version; ignoring it/
39	],
40	'processes corrupted pg_control all zeroes');
41
42# Put in the previously saved header data.  This uses a different code
43# path internally, allowing us to process a zero WAL segment size.
44open $fh, '>', $pg_control or BAIL_OUT($!);
45binmode $fh;
46print $fh $data, pack("x[" . ($size - 16) . "]");
47close $fh;
48
49command_checks_all(
50	[ 'pg_resetwal', '--dry-run', $node->data_dir ],
51	0,
52	[qr/pg_control version number/],
53	[
54		qr/\Qpg_resetwal: warning: pg_control specifies invalid WAL segment size (0 bytes); proceed with caution\E/
55	],
56	'processes zero WAL segment size');
57
58# now try to run it
59command_fails_like(
60	[ 'pg_resetwal', $node->data_dir ],
61	qr/not proceeding because control file values were guessed/,
62	'does not run when control file values were guessed');
63command_ok(
64	[ 'pg_resetwal', '--force', $node->data_dir ],
65	'runs with force when control file values were guessed');
66
67done_testing();
