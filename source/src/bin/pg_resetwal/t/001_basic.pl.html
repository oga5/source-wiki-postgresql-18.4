184# run with control override options

186my $out = (run_command([ 'pg_resetwal', '--dry-run', $node->data_dir ]))[0];
187$out =~ /^Database block size: *(\d+)$/m or die;
188my $blcksz = $1;
189
190my @cmd = ('pg_resetwal', '--pgdata' => $node->data_dir);
191
192# some not-so-critical hardcoded values
193push @cmd, '--epoch' => 1;
194push @cmd, '--next-wal-file' => '00000001000000320000004B';
195push @cmd, '--next-oid' => 100_000;
196push @cmd, '--wal-segsize' => 1;

198# these use the guidance from the documentation
199
200sub get_slru_files
201{
202	opendir(my $dh, $node->data_dir . '/' . $_[0]) or die $!;
203	my @files = sort grep { /[0-9A-F]+/ } readdir $dh;
204	closedir $dh;
205	return @files;
206}
207
208my (@files, $mult);
209
210@files = get_slru_files('pg_commit_ts');
211# XXX: Should there be a multiplier, similar to the other options?
212# -c argument is "old,new"
213push @cmd,
214  '--commit-timestamp-ids' =>
215  sprintf("%d,%d", hex($files[0]) == 0 ? 3 : hex($files[0]), hex($files[-1]));
216
217@files = get_slru_files('pg_multixact/offsets');
218$mult = 32 * $blcksz / 4;
219# --multixact-ids argument is "new,old"
220push @cmd,
221  '--multixact-ids' => sprintf("%d,%d",
222	(hex($files[-1]) + 1) * $mult,
223	hex($files[0]) == 0 ? 1 : hex($files[0] * $mult));
224
225@files = get_slru_files('pg_multixact/members');
226$mult = 32 * int($blcksz / 20) * 4;
227push @cmd, '--multixact-offset' => (hex($files[-1]) + 1) * $mult;
228
229@files = get_slru_files('pg_xact');
230$mult = 32 * $blcksz * 4;
231push @cmd,
232  '--oldest-transaction-id' =>
233  (hex($files[0]) == 0 ? 3 : hex($files[0]) * $mult),
234  '--next-transaction-id' => ((hex($files[-1]) + 1) * $mult);
235
236command_ok([ @cmd, '--dry-run' ],
237	'runs with control override options, dry run');
238command_ok(\@cmd, 'runs with control override options');
