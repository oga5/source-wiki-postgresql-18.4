4use strict;
5use warnings FATAL => 'all';
6use PostgreSQL::Test::Cluster;
7use PostgreSQL::Test::Utils;
8use Test::More;
9
10program_help_ok('pg_controldata');
11program_version_ok('pg_controldata');
12program_options_handling_ok('pg_controldata');
13command_fails(['pg_controldata'], 'pg_controldata without arguments fails');
14command_fails([ 'pg_controldata', 'nonexistent' ],
15	'pg_controldata with nonexistent directory fails');
16
17my $node = PostgreSQL::Test::Cluster->new('main');
18$node->init;
19
20command_like([ 'pg_controldata', $node->data_dir ],
21	qr/checkpoint/, 'pg_controldata produces output');
22
23
24# check with a corrupted pg_control
25
26my $pg_control = $node->data_dir . '/global/pg_control';
27my $size = -s $pg_control;
28
29open my $fh, '>', $pg_control or BAIL_OUT($!);
30binmode $fh;
31
32# fill file with zeros
33print $fh pack("x[$size]");
34close $fh;
35
36command_checks_all(
37	[ 'pg_controldata', $node->data_dir ],
38	0,
39	[qr/./],
40	[
41		qr/warning: calculated CRC checksum does not match value stored in control file/,
42		qr/warning: invalid WAL segment size/
43	],
44	'pg_controldata with corrupted pg_control');
45
46done_testing();
