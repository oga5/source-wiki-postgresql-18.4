3use strict;
4use warnings FATAL => 'all';
5use File::Compare;
6use PostgreSQL::Test::Cluster;
7use PostgreSQL::Test::Utils;
8use Test::More;
9
10# Can be changed to test the other modes.
11my $mode = $ENV{PG_TEST_PG_COMBINEBACKUP_MODE} || '--copy';
12
13note "testing using mode $mode";
14
15# Set up a new database instance.
16my $primary = PostgreSQL::Test::Cluster->new('primary');
17$primary->init(has_archiving => 1, allows_streaming => 1);
18$primary->append_conf('postgresql.conf', 'summarize_wal = on');
19$primary->start;
20
21# Initial setup.
22$primary->safe_psql('postgres', <<EOM);
23CREATE DATABASE lakh OID = 100000 STRATEGY = FILE_COPY
24EOM
25$primary->safe_psql('lakh', <<EOM);
26CREATE TABLE t1 (a int)
27EOM
28
29# Take a full backup.
30my $backup1path = $primary->backup_dir . '/backup1';
31$primary->command_ok(
32	[
33		'pg_basebackup',
34		'--pgdata' => $backup1path,
35		'--no-sync',
36		'--checkpoint' => 'fast'
37	],
38	"full backup");
39
40# Now make some database changes.
41$primary->safe_psql('postgres', <<EOM);
42DROP DATABASE lakh;
43CREATE DATABASE lakh OID = 100000 STRATEGY = FILE_COPY
44EOM
45
46# Take an incremental backup.
47my $backup2path = $primary->backup_dir . '/backup2';
48$primary->command_ok(
49	[
50		'pg_basebackup',
51		'--pgdata' => $backup2path,
52		'--no-sync',
53		'--checkpoint' => 'fast',
54		'--incremental' => $backup1path . '/backup_manifest'
55	],
56	"incremental backup");
57
58# Recover the incremental backup.
59my $restore = PostgreSQL::Test::Cluster->new('restore');
60$restore->init_from_backup(
61	$primary, 'backup2',
62	combine_with_prior => ['backup1'],
63	combine_mode => $mode);
64$restore->start();
65
66# Query the DB.
67my $stdout;
68my $stderr;
69$restore->psql(
70	'lakh', 'SELECT * FROM t1',
71	stdout => \$stdout,
72	stderr => \$stderr);
73is($stdout, '', 'SELECT * FROM t1: no stdout');
74like(
75	$stderr,
76	qr/relation "t1" does not exist/,
77	'SELECT * FROM t1: stderr missing table');
78
79done_testing();
