14# create a directory for scripts
15my $testname = $0;
16$testname =~ s,.*/,,;

19my $testdir = "$PostgreSQL::Test::Utils::tmp_check/t_${testname}_stuff";
20mkdir $testdir
21  or BAIL_OUT("could not create test directory \"${testdir}\": $!");

23# invoke pgbench
24sub pgbench
25{
26	local $Test::Builder::Level = $Test::Builder::Level + 1;
27
28	my ($opts, $stat, $out, $err, $name) = @_;
29	command_checks_all([ 'pgbench', split(/\s+/, $opts) ],
30		$stat, $out, $err, $name);
31	return;
32}
33
34# invoke pgbench with scripts
35sub pgbench_scripts
36{
37	local $Test::Builder::Level = $Test::Builder::Level + 1;
38
39	my ($opts, $stat, $out, $err, $name, $files) = @_;
40	my @cmd = ('pgbench', split /\s+/, $opts);
41	if (defined $files)
42	{
43		for my $fn (sort keys %$files)
44		{
45			my $filename = $testdir . '/' . $fn;
46
47			# cleanup file weight if any
48			$filename =~ s/\@\d+$//;
49
50			# cleanup from prior runs
51			unlink $filename;
52			append_to_file($filename, $$files{$fn});
53			push @cmd, '--file' => $filename;
54		}
55	}
56	command_checks_all(\@cmd, $stat, $out, $err, $name);
57	return;
58}
