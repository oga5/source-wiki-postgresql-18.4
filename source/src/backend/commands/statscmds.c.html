47/* qsort comparator for the attnums in CreateStatistics */
48static int
49compare_int16(const void *a, const void *b)
50{
51	int			av = *(const int16 *) a;
52	int			bv = *(const int16 *) b;
53
54	/* this can't overflow if int is wider than int16 */
55	return (av - bv);
56}

58/*
59 *		CREATE STATISTICS
60 */
61ObjectAddress
62CreateStatistics(CreateStatsStmt *stmt, bool check_rights)
63{
64	int16		attnums[STATS_MAX_DIMENSIONS];
65	int			nattnums = 0;
66	int			numcols;
67	char	   *namestr;
68	NameData	stxname;
69	Oid			statoid;
70	Oid			namespaceId;
71	Oid			stxowner = GetUserId();
72	HeapTuple	htup;
73	Datum		values[Natts_pg_statistic_ext];
74	bool		nulls[Natts_pg_statistic_ext];
75	int2vector *stxkeys;
76	List	   *stxexprs = NIL;
77	Datum		exprsDatum;
78	Relation	statrel;
79	Relation	rel = NULL;
80	Oid			relid;
81	ObjectAddress parentobject,
82				myself;
83	Datum		types[4];		/* one for each possible type of statistic */
84	int			ntypes;
85	ArrayType  *stxkind;
86	bool		build_ndistinct;
87	bool		build_dependencies;
88	bool		build_mcv;
89	bool		build_expressions;
90	bool		requested_type = false;
91	int			i;
92	ListCell   *cell;
93	ListCell   *cell2;
94
95	Assert(IsA(stmt, CreateStatsStmt));
96
97	/*
98	 * Examine the FROM clause.  Currently, we only allow it to be a single
99	 * simple table, but later we'll probably allow multiple tables and JOIN
100	 * syntax.  The grammar is already prepared for that, so we have to check
101	 * here that what we got is what we can support.
102	 */
103	if (list_length(stmt->relations) != 1)
104		ereport(ERROR,
105				(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
106				 errmsg("only a single relation is allowed in CREATE STATISTICS")));
107
108	foreach(cell, stmt->relations)
109	{
110		Node	   *rln = (Node *) lfirst(cell);
111
112		if (!IsA(rln, RangeVar))
113			ereport(ERROR,
114					(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
115					 errmsg("only a single relation is allowed in CREATE STATISTICS")));
116
117		/*
118		 * CREATE STATISTICS will influence future execution plans but does
119		 * not interfere with currently executing plans.  So it should be
120		 * enough to take only ShareUpdateExclusiveLock on relation,
121		 * conflicting with ANALYZE and other DDL that sets statistical
122		 * information, but not with normal queries.
123		 */
124		rel = relation_openrv((RangeVar *) rln, ShareUpdateExclusiveLock);
125
126		/* Restrict to allowed relation types */
127		if (rel->rd_rel->relkind != RELKIND_RELATION &&
128			rel->rd_rel->relkind != RELKIND_MATVIEW &&
129			rel->rd_rel->relkind != RELKIND_FOREIGN_TABLE &&
130			rel->rd_rel->relkind != RELKIND_PARTITIONED_TABLE)
131			ereport(ERROR,
132					(errcode(ERRCODE_WRONG_OBJECT_TYPE),
133					 errmsg("cannot define statistics for relation \"%s\"",
134							RelationGetRelationName(rel)),
135					 errdetail_relkind_not_supported(rel->rd_rel->relkind)));
136
137		/*
138		 * You must own the relation to create stats on it.
139		 *
140		 * NB: Concurrent changes could cause this function's lookup to find a
141		 * different relation than a previous lookup by the caller, so we must
142		 * perform this check even when check_rights == false.
143		 */
144		if (!object_ownercheck(RelationRelationId, RelationGetRelid(rel), stxowner))
145			aclcheck_error(ACLCHECK_NOT_OWNER, get_relkind_objtype(rel->rd_rel->relkind),
146						   RelationGetRelationName(rel));
147
148		/* Creating statistics on system catalogs is not allowed */
149		if (!allowSystemTableMods && IsSystemRelation(rel))
150			ereport(ERROR,
151					(errcode(ERRCODE_INSUFFICIENT_PRIVILEGE),
152					 errmsg("permission denied: \"%s\" is a system catalog",
153							RelationGetRelationName(rel))));
154	}
155
156	Assert(rel);
157	relid = RelationGetRelid(rel);
158
159	/*
160	 * If the node has a name, split it up and determine creation namespace.
161	 * If not, put the object in the same namespace as the relation, and cons
162	 * up a name for it.  (This can happen either via "CREATE STATISTICS ..."
163	 * or via "CREATE TABLE ... (LIKE)".)
164	 */
165	if (stmt->defnames)
166		namespaceId = QualifiedNameGetCreationNamespace(stmt->defnames,
167														&namestr);
168	else
169	{
170		namespaceId = RelationGetNamespace(rel);
171		namestr = ChooseExtendedStatisticName(RelationGetRelationName(rel),
172											  ChooseExtendedStatisticNameAddition(stmt->exprs),
173											  "stat",
174											  namespaceId);
175	}
176	namestrcpy(&stxname, namestr);
177
178	/*
179	 * Check we have creation rights in target namespace.  Skip check if
180	 * caller doesn't want it.
181	 */
182	if (check_rights)
183	{
184		AclResult	aclresult;
185
186		aclresult = object_aclcheck(NamespaceRelationId, namespaceId,
187									GetUserId(), ACL_CREATE);
188		if (aclresult != ACLCHECK_OK)
189			aclcheck_error(aclresult, OBJECT_SCHEMA,
190						   get_namespace_name(namespaceId));
191	}
192
193	/*
194	 * Deal with the possibility that the statistics object already exists.
195	 */
196	if (SearchSysCacheExists2(STATEXTNAMENSP,
197							  CStringGetDatum(namestr),
198							  ObjectIdGetDatum(namespaceId)))
199	{
200		if (stmt->if_not_exists)
201		{
202			/*
203			 * Since stats objects aren't members of extensions (see comments
204			 * below), no need for checkMembershipInCurrentExtension here.
205			 */
206			ereport(NOTICE,
207					(errcode(ERRCODE_DUPLICATE_OBJECT),
208					 errmsg("statistics object \"%s\" already exists, skipping",
209							namestr)));
210			relation_close(rel, NoLock);
211			return InvalidObjectAddress;
212		}
213
214		ereport(ERROR,
215				(errcode(ERRCODE_DUPLICATE_OBJECT),
216				 errmsg("statistics object \"%s\" already exists", namestr)));
217	}
218
219	/*
220	 * Make sure no more than STATS_MAX_DIMENSIONS columns are used. There
221	 * might be duplicates and so on, but we'll deal with those later.
222	 */
223	numcols = list_length(stmt->exprs);
224	if (numcols > STATS_MAX_DIMENSIONS)
225		ereport(ERROR,
226				(errcode(ERRCODE_TOO_MANY_COLUMNS),
227				 errmsg("cannot have more than %d columns in statistics",
228						STATS_MAX_DIMENSIONS)));
229
230	/*
231	 * Convert the expression list to a simple array of attnums, but also keep
232	 * a list of more complex expressions.  While at it, enforce some
233	 * constraints - we don't allow extended statistics on system attributes,
234	 * and we require the data type to have a less-than operator.
235	 *
236	 * There are many ways to "mask" a simple attribute reference as an
237	 * expression, for example "(a+0)" etc. We can't possibly detect all of
238	 * them, but we handle at least the simple case with the attribute in
239	 * parens. There'll always be a way around this, if the user is determined
240	 * (like the "(a+0)" example), but this makes it somewhat consistent with
241	 * how indexes treat attributes/expressions.
242	 */
243	foreach(cell, stmt->exprs)
244	{
245		StatsElem  *selem = lfirst_node(StatsElem, cell);
246
247		if (selem->name)		/* column reference */
248		{
249			char	   *attname;
250			HeapTuple	atttuple;
251			Form_pg_attribute attForm;
252			TypeCacheEntry *type;
253
254			attname = selem->name;
255
256			atttuple = SearchSysCacheAttName(relid, attname);
257			if (!HeapTupleIsValid(atttuple))
258				ereport(ERROR,
259						(errcode(ERRCODE_UNDEFINED_COLUMN),
260						 errmsg("column \"%s\" does not exist",
261								attname)));
262			attForm = (Form_pg_attribute) GETSTRUCT(atttuple);
263
264			/* Disallow use of system attributes in extended stats */
265			if (attForm->attnum <= 0)
266				ereport(ERROR,
267						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
268						 errmsg("statistics creation on system columns is not supported")));
269
270			/* Disallow use of virtual generated columns in extended stats */
271			if (attForm->attgenerated == ATTRIBUTE_GENERATED_VIRTUAL)
272				ereport(ERROR,
273						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
274						 errmsg("statistics creation on virtual generated columns is not supported")));
275
276			/* Disallow data types without a less-than operator */
277			type = lookup_type_cache(attForm->atttypid, TYPECACHE_LT_OPR);
278			if (type->lt_opr == InvalidOid)
279				ereport(ERROR,
280						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
281						 errmsg("column \"%s\" cannot be used in statistics because its type %s has no default btree operator class",
282								attname, format_type_be(attForm->atttypid))));
283
284			attnums[nattnums] = attForm->attnum;
285			nattnums++;
286			ReleaseSysCache(atttuple);
287		}
288		else if (IsA(selem->expr, Var)) /* column reference in parens */
289		{
290			Var		   *var = (Var *) selem->expr;
291			TypeCacheEntry *type;
292
293			/* Disallow use of system attributes in extended stats */
294			if (var->varattno <= 0)
295				ereport(ERROR,
296						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
297						 errmsg("statistics creation on system columns is not supported")));
298
299			/* Disallow use of virtual generated columns in extended stats */
300			if (get_attgenerated(relid, var->varattno) == ATTRIBUTE_GENERATED_VIRTUAL)
301				ereport(ERROR,
302						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
303						 errmsg("statistics creation on virtual generated columns is not supported")));
304
305			/* Disallow data types without a less-than operator */
306			type = lookup_type_cache(var->vartype, TYPECACHE_LT_OPR);
307			if (type->lt_opr == InvalidOid)
308				ereport(ERROR,
309						(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
310						 errmsg("column \"%s\" cannot be used in statistics because its type %s has no default btree operator class",
311								get_attname(relid, var->varattno, false), format_type_be(var->vartype))));
312
313			attnums[nattnums] = var->varattno;
314			nattnums++;
315		}
316		else					/* expression */
317		{
318			Node	   *expr = selem->expr;
319			Oid			atttype;
320			TypeCacheEntry *type;
321			Bitmapset  *attnums = NULL;
322			int			k;
323
324			Assert(expr != NULL);
325
326			pull_varattnos(expr, 1, &attnums);
327
328			k = -1;
329			while ((k = bms_next_member(attnums, k)) >= 0)
330			{
331				AttrNumber	attnum = k + FirstLowInvalidHeapAttributeNumber;
332
333				/* Disallow expressions referencing system attributes. */
334				if (attnum <= 0)
335					ereport(ERROR,
336							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
337							 errmsg("statistics creation on system columns is not supported")));
338
339				/* Disallow use of virtual generated columns in extended stats */
340				if (get_attgenerated(relid, attnum) == ATTRIBUTE_GENERATED_VIRTUAL)
341					ereport(ERROR,
342							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
343							 errmsg("statistics creation on virtual generated columns is not supported")));
344			}
345
346			/*
347			 * Disallow data types without a less-than operator.
348			 *
349			 * We ignore this for statistics on a single expression, in which
350			 * case we'll build the regular statistics only (and that code can
351			 * deal with such data types).
352			 */
353			if (list_length(stmt->exprs) > 1)
354			{
355				atttype = exprType(expr);
356				type = lookup_type_cache(atttype, TYPECACHE_LT_OPR);
357				if (type->lt_opr == InvalidOid)
358					ereport(ERROR,
359							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
360							 errmsg("expression cannot be used in multivariate statistics because its type %s has no default btree operator class",
361									format_type_be(atttype))));
362			}
363
364			stxexprs = lappend(stxexprs, expr);
365		}
366	}
367
368	/*
369	 * Parse the statistics kinds.
370	 *
371	 * First check that if this is the case with a single expression, there
372	 * are no statistics kinds specified (we don't allow that for the simple
373	 * CREATE STATISTICS form).
374	 */
375	if ((list_length(stmt->exprs) == 1) && (list_length(stxexprs) == 1))
376	{
377		/* statistics kinds not specified */
378		if (stmt->stat_types != NIL)
379			ereport(ERROR,
380					(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
381					 errmsg("when building statistics on a single expression, statistics kinds may not be specified")));
382	}
383
384	/* OK, let's check that we recognize the statistics kinds. */
385	build_ndistinct = false;
386	build_dependencies = false;
387	build_mcv = false;
388	foreach(cell, stmt->stat_types)
389	{
390		char	   *type = strVal(lfirst(cell));
391
392		if (strcmp(type, "ndistinct") == 0)
393		{
394			build_ndistinct = true;
395			requested_type = true;
396		}
397		else if (strcmp(type, "dependencies") == 0)
398		{
399			build_dependencies = true;
400			requested_type = true;
401		}
402		else if (strcmp(type, "mcv") == 0)
403		{
404			build_mcv = true;
405			requested_type = true;
406		}
407		else
408			ereport(ERROR,
409					(errcode(ERRCODE_SYNTAX_ERROR),
410					 errmsg("unrecognized statistics kind \"%s\"",
411							type)));
412	}
413
414	/*
415	 * If no statistic type was specified, build them all (but only when the
416	 * statistics is defined on more than one column/expression).
417	 */
418	if ((!requested_type) && (numcols >= 2))
419	{
420		build_ndistinct = true;
421		build_dependencies = true;
422		build_mcv = true;
423	}
424
425	/*
426	 * When there are non-trivial expressions, build the expression stats
427	 * automatically. This allows calculating good estimates for stats that
428	 * consider per-clause estimates (e.g. functional dependencies).
429	 */
430	build_expressions = (stxexprs != NIL);
431
432	/*
433	 * Check that at least two columns were specified in the statement, or
434	 * that we're building statistics on a single expression.
435	 */
436	if ((numcols < 2) && (list_length(stxexprs) != 1))
437		ereport(ERROR,
438				(errcode(ERRCODE_INVALID_OBJECT_DEFINITION),
439				 errmsg("extended statistics require at least 2 columns")));
440
441	/*
442	 * Sort the attnums, which makes detecting duplicates somewhat easier, and
443	 * it does not hurt (it does not matter for the contents, unlike for
444	 * indexes, for example).
445	 */
446	qsort(attnums, nattnums, sizeof(int16), compare_int16);
447
448	/*
449	 * Check for duplicates in the list of columns. The attnums are sorted so
450	 * just check consecutive elements.
451	 */
452	for (i = 1; i < nattnums; i++)
453	{
454		if (attnums[i] == attnums[i - 1])
455			ereport(ERROR,
456					(errcode(ERRCODE_DUPLICATE_COLUMN),
457					 errmsg("duplicate column name in statistics definition")));
458	}
459
460	/*
461	 * Check for duplicate expressions. We do two loops, counting the
462	 * occurrences of each expression. This is O(N^2) but we only allow small
463	 * number of expressions and it's not executed often.
464	 *
465	 * XXX We don't cross-check attributes and expressions, because it does
466	 * not seem worth it. In principle we could check that expressions don't
467	 * contain trivial attribute references like "(a)", but the reasoning is
468	 * similar to why we don't bother with extracting columns from
469	 * expressions. It's either expensive or very easy to defeat for
470	 * determined user, and there's no risk if we allow such statistics (the
471	 * statistics is useless, but harmless).
472	 */
473	foreach(cell, stxexprs)
474	{
475		Node	   *expr1 = (Node *) lfirst(cell);
476		int			cnt = 0;
477
478		foreach(cell2, stxexprs)
479		{
480			Node	   *expr2 = (Node *) lfirst(cell2);
481
482			if (equal(expr1, expr2))
483				cnt += 1;
484		}
485
486		/* every expression should find at least itself */
487		Assert(cnt >= 1);
488
489		if (cnt > 1)
490			ereport(ERROR,
491					(errcode(ERRCODE_DUPLICATE_COLUMN),
492					 errmsg("duplicate expression in statistics definition")));
493	}
494
495	/* Form an int2vector representation of the sorted column list */
496	stxkeys = buildint2vector(attnums, nattnums);
497
498	/* construct the char array of enabled statistic types */
499	ntypes = 0;
500	if (build_ndistinct)
501		types[ntypes++] = CharGetDatum(STATS_EXT_NDISTINCT);
502	if (build_dependencies)
503		types[ntypes++] = CharGetDatum(STATS_EXT_DEPENDENCIES);
504	if (build_mcv)
505		types[ntypes++] = CharGetDatum(STATS_EXT_MCV);
506	if (build_expressions)
507		types[ntypes++] = CharGetDatum(STATS_EXT_EXPRESSIONS);
508	Assert(ntypes > 0 && ntypes <= lengthof(types));
509	stxkind = construct_array_builtin(types, ntypes, CHAROID);
510
511	/* convert the expressions (if any) to a text datum */
512	if (stxexprs != NIL)
513	{
514		char	   *exprsString;
515
516		exprsString = nodeToString(stxexprs);
517		exprsDatum = CStringGetTextDatum(exprsString);
518		pfree(exprsString);
519	}
520	else
521		exprsDatum = (Datum) 0;
522
523	statrel = table_open(StatisticExtRelationId, RowExclusiveLock);
524
525	/*
526	 * Everything seems fine, so let's build the pg_statistic_ext tuple.
527	 */
528	memset(values, 0, sizeof(values));
529	memset(nulls, false, sizeof(nulls));
530
531	statoid = GetNewOidWithIndex(statrel, StatisticExtOidIndexId,
532								 Anum_pg_statistic_ext_oid);
533	values[Anum_pg_statistic_ext_oid - 1] = ObjectIdGetDatum(statoid);
534	values[Anum_pg_statistic_ext_stxrelid - 1] = ObjectIdGetDatum(relid);
535	values[Anum_pg_statistic_ext_stxname - 1] = NameGetDatum(&stxname);
536	values[Anum_pg_statistic_ext_stxnamespace - 1] = ObjectIdGetDatum(namespaceId);
537	values[Anum_pg_statistic_ext_stxowner - 1] = ObjectIdGetDatum(stxowner);
538	values[Anum_pg_statistic_ext_stxkeys - 1] = PointerGetDatum(stxkeys);
539	nulls[Anum_pg_statistic_ext_stxstattarget - 1] = true;
540	values[Anum_pg_statistic_ext_stxkind - 1] = PointerGetDatum(stxkind);
541
542	values[Anum_pg_statistic_ext_stxexprs - 1] = exprsDatum;
543	if (exprsDatum == (Datum) 0)
544		nulls[Anum_pg_statistic_ext_stxexprs - 1] = true;
545
546	/* insert it into pg_statistic_ext */
547	htup = heap_form_tuple(statrel->rd_att, values, nulls);
548	CatalogTupleInsert(statrel, htup);
549	heap_freetuple(htup);
550
551	relation_close(statrel, RowExclusiveLock);
552
553	/*
554	 * We used to create the pg_statistic_ext_data tuple too, but it's not
555	 * clear what value should the stxdinherit flag have (it depends on
556	 * whether the rel is partitioned, contains data, etc.)
557	 */
558
559	InvokeObjectPostCreateHook(StatisticExtRelationId, statoid, 0);
560
561	/*
562	 * Invalidate relcache so that others see the new statistics object.
563	 */
564	CacheInvalidateRelcache(rel);
565
566	relation_close(rel, NoLock);
567
568	/*
569	 * Add an AUTO dependency on each column used in the stats, so that the
570	 * stats object goes away if any or all of them get dropped.
571	 */
572	ObjectAddressSet(myself, StatisticExtRelationId, statoid);
573
574	/* add dependencies for plain column references */
575	for (i = 0; i < nattnums; i++)
576	{
577		ObjectAddressSubSet(parentobject, RelationRelationId, relid, attnums[i]);
578		recordDependencyOn(&myself, &parentobject, DEPENDENCY_AUTO);
579	}
580
581	/*
582	 * If there are no dependencies on a column, give the statistics object an
583	 * auto dependency on the whole table.  In most cases, this will be
584	 * redundant, but it might not be if the statistics expressions contain no
585	 * Vars (which might seem strange but possible). This is consistent with
586	 * what we do for indexes in index_create.
587	 *
588	 * XXX We intentionally don't consider the expressions before adding this
589	 * dependency, because recordDependencyOnSingleRelExpr may not create any
590	 * dependencies for whole-row Vars.
591	 */
592	if (!nattnums)
593	{
594		ObjectAddressSet(parentobject, RelationRelationId, relid);
595		recordDependencyOn(&myself, &parentobject, DEPENDENCY_AUTO);
596	}
597
598	/*
599	 * Store dependencies on anything mentioned in statistics expressions,
600	 * just like we do for index expressions.
601	 */
602	if (stxexprs)
603		recordDependencyOnSingleRelExpr(&myself,
604										(Node *) stxexprs,
605										relid,
606										DEPENDENCY_NORMAL,
607										DEPENDENCY_AUTO, false);
608
609	/*
610	 * Also add dependencies on namespace and owner.  These are required
611	 * because the stats object might have a different namespace and/or owner
612	 * than the underlying table(s).
613	 */
614	ObjectAddressSet(parentobject, NamespaceRelationId, namespaceId);
615	recordDependencyOn(&myself, &parentobject, DEPENDENCY_NORMAL);
616
617	recordDependencyOnOwner(StatisticExtRelationId, statoid, stxowner);
618
619	/*
620	 * XXX probably there should be a recordDependencyOnCurrentExtension call
621	 * here too, but we'd have to add support for ALTER EXTENSION ADD/DROP
622	 * STATISTICS, which is more work than it seems worth.
623	 */
624
625	/* Add any requested comment */
626	if (stmt->stxcomment != NULL)
627		CreateComments(statoid, StatisticExtRelationId, 0,
628					   stmt->stxcomment);
629
630	/* Return stats object's address */
631	return myself;
632}

634/*
635 *		ALTER STATISTICS
636 */
637ObjectAddress
638AlterStatistics(AlterStatsStmt *stmt)
639{
640	Relation	rel;
641	Oid			stxoid;
642	HeapTuple	oldtup;
643	HeapTuple	newtup;
644	Datum		repl_val[Natts_pg_statistic_ext];
645	bool		repl_null[Natts_pg_statistic_ext];
646	bool		repl_repl[Natts_pg_statistic_ext];
647	ObjectAddress address;
648	int			newtarget = 0;
649	bool		newtarget_default;
650
651	/* -1 was used in previous versions for the default setting */
652	if (stmt->stxstattarget && intVal(stmt->stxstattarget) != -1)
653	{
654		newtarget = intVal(stmt->stxstattarget);
655		newtarget_default = false;
656	}
657	else
658		newtarget_default = true;
659
660	if (!newtarget_default)
661	{
662		/* Limit statistics target to a sane range */
663		if (newtarget < 0)
664		{
665			ereport(ERROR,
666					(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
667					 errmsg("statistics target %d is too low",
668							newtarget)));
669		}
670		else if (newtarget > MAX_STATISTICS_TARGET)
671		{
672			newtarget = MAX_STATISTICS_TARGET;
673			ereport(WARNING,
674					(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
675					 errmsg("lowering statistics target to %d",
676							newtarget)));
677		}
678	}
679
680	/* lookup OID of the statistics object */
681	stxoid = get_statistics_object_oid(stmt->defnames, stmt->missing_ok);
682
683	/*
684	 * If we got here and the OID is not valid, it means the statistics object
685	 * does not exist, but the command specified IF EXISTS. So report this as
686	 * a simple NOTICE and we're done.
687	 */
688	if (!OidIsValid(stxoid))
689	{
690		char	   *schemaname;
691		char	   *statname;
692
693		Assert(stmt->missing_ok);
694
695		DeconstructQualifiedName(stmt->defnames, &schemaname, &statname);
696
697		if (schemaname)
698			ereport(NOTICE,
699					(errmsg("statistics object \"%s.%s\" does not exist, skipping",
700							schemaname, statname)));
701		else
702			ereport(NOTICE,
703					(errmsg("statistics object \"%s\" does not exist, skipping",
704							statname)));
705
706		return InvalidObjectAddress;
707	}
708
709	/* Search pg_statistic_ext */
710	rel = table_open(StatisticExtRelationId, RowExclusiveLock);
711
712	oldtup = SearchSysCache1(STATEXTOID, ObjectIdGetDatum(stxoid));
713	if (!HeapTupleIsValid(oldtup))
714		elog(ERROR, "cache lookup failed for extended statistics object %u", stxoid);
715
716	/* Must be owner of the existing statistics object */
717	if (!object_ownercheck(StatisticExtRelationId, stxoid, GetUserId()))
718		aclcheck_error(ACLCHECK_NOT_OWNER, OBJECT_STATISTIC_EXT,
719					   NameListToString(stmt->defnames));
720
721	/* Build new tuple. */
722	memset(repl_val, 0, sizeof(repl_val));
723	memset(repl_null, false, sizeof(repl_null));
724	memset(repl_repl, false, sizeof(repl_repl));
725
726	/* replace the stxstattarget column */
727	repl_repl[Anum_pg_statistic_ext_stxstattarget - 1] = true;
728	if (!newtarget_default)
729		repl_val[Anum_pg_statistic_ext_stxstattarget - 1] = Int16GetDatum(newtarget);
730	else
731		repl_null[Anum_pg_statistic_ext_stxstattarget - 1] = true;
732
733	newtup = heap_modify_tuple(oldtup, RelationGetDescr(rel),
734							   repl_val, repl_null, repl_repl);
735
736	/* Update system catalog. */
737	CatalogTupleUpdate(rel, &newtup->t_self, newtup);
738
739	InvokeObjectPostAlterHook(StatisticExtRelationId, stxoid, 0);
740
741	ObjectAddressSet(address, StatisticExtRelationId, stxoid);
742
743	/*
744	 * NOTE: because we only support altering the statistics target, not the
745	 * other fields, there is no need to update dependencies.
746	 */
747
748	heap_freetuple(newtup);
749	ReleaseSysCache(oldtup);
750
751	table_close(rel, RowExclusiveLock);
752
753	return address;
754}

756/*
757 * Delete entry in pg_statistic_ext_data catalog. We don't know if the row
758 * exists, so don't error out.
759 */
760void
761RemoveStatisticsDataById(Oid statsOid, bool inh)
762{
763	Relation	relation;
764	HeapTuple	tup;
765
766	relation = table_open(StatisticExtDataRelationId, RowExclusiveLock);
767
768	tup = SearchSysCache2(STATEXTDATASTXOID, ObjectIdGetDatum(statsOid),
769						  BoolGetDatum(inh));
770
771	/* We don't know if the data row for inh value exists. */
772	if (HeapTupleIsValid(tup))
773	{
774		CatalogTupleDelete(relation, &tup->t_self);
775
776		ReleaseSysCache(tup);
777	}
778
779	table_close(relation, RowExclusiveLock);
780}

782/*
783 * Guts of statistics object deletion.
784 */
785void
786RemoveStatisticsById(Oid statsOid)
787{
788	Relation	relation;
789	Relation	rel;
790	HeapTuple	tup;
791	Form_pg_statistic_ext statext;
792	Oid			relid;
793
794	/*
795	 * Delete the pg_statistic_ext tuple.  Also send out a cache inval on the
796	 * associated table, so that dependent plans will be rebuilt.
797	 */
798	relation = table_open(StatisticExtRelationId, RowExclusiveLock);
799
800	tup = SearchSysCache1(STATEXTOID, ObjectIdGetDatum(statsOid));
801
802	if (!HeapTupleIsValid(tup)) /* should not happen */
803		elog(ERROR, "cache lookup failed for statistics object %u", statsOid);
804
805	statext = (Form_pg_statistic_ext) GETSTRUCT(tup);
806	relid = statext->stxrelid;
807
808	/*
809	 * Delete the pg_statistic_ext_data tuples holding the actual statistical
810	 * data. There might be data with/without inheritance, so attempt deleting
811	 * both. We lock the user table first, to prevent other processes (e.g.
812	 * DROP STATISTICS) from removing the row concurrently.
813	 */
814	rel = table_open(relid, ShareUpdateExclusiveLock);
815
816	RemoveStatisticsDataById(statsOid, true);
817	RemoveStatisticsDataById(statsOid, false);
818
819	CacheInvalidateRelcacheByRelid(relid);
820
821	CatalogTupleDelete(relation, &tup->t_self);
822
823	ReleaseSysCache(tup);
824
825	/* Keep lock until the end of the transaction. */
826	table_close(rel, NoLock);
827
828	table_close(relation, RowExclusiveLock);
829}

831/*
832 * Select a nonconflicting name for a new statistics object.
833 *
834 * name1, name2, and label are used the same way as for makeObjectName(),
835 * except that the label can't be NULL; digits will be appended to the label
836 * if needed to create a name that is unique within the specified namespace.
837 *
838 * Returns a palloc'd string.
839 *
840 * Note: it is theoretically possible to get a collision anyway, if someone
841 * else chooses the same name concurrently.  This is fairly unlikely to be
842 * a problem in practice, especially if one is holding a share update
843 * exclusive lock on the relation identified by name1.  However, if choosing
844 * multiple names within a single command, you'd better create the new object
845 * and do CommandCounterIncrement before choosing the next one!
846 */
847static char *
848ChooseExtendedStatisticName(const char *name1, const char *name2,
849							const char *label, Oid namespaceid)
850{
851	int			pass = 0;
852	char	   *stxname = NULL;
853	char		modlabel[NAMEDATALEN];
854
855	/* try the unmodified label first */
856	strlcpy(modlabel, label, sizeof(modlabel));
857
858	for (;;)
859	{
860		Oid			existingstats;
861
862		stxname = makeObjectName(name1, name2, modlabel);
863
864		existingstats = GetSysCacheOid2(STATEXTNAMENSP, Anum_pg_statistic_ext_oid,
865										PointerGetDatum(stxname),
866										ObjectIdGetDatum(namespaceid));
867		if (!OidIsValid(existingstats))
868			break;
869
870		/* found a conflict, so try a new name component */
871		pfree(stxname);
872		snprintf(modlabel, sizeof(modlabel), "%s%d", label, ++pass);
873	}
874
875	return stxname;
876}

878/*
879 * Generate "name2" for a new statistics object given the list of column
880 * names for it.  This will be passed to ChooseExtendedStatisticName along
881 * with the parent table name and a suitable label.
882 *
883 * We know that less than NAMEDATALEN characters will actually be used,
884 * so we can truncate the result once we've generated that many.
885 *
886 * XXX see also ChooseForeignKeyConstraintNameAddition and
887 * ChooseIndexNameAddition.
888 */
889static char *
890ChooseExtendedStatisticNameAddition(List *exprs)
891{
892	char		buf[NAMEDATALEN * 2];
893	int			buflen = 0;
894	ListCell   *lc;
895
896	buf[0] = '\0';
897	foreach(lc, exprs)
898	{
899		StatsElem  *selem = (StatsElem *) lfirst(lc);
900		const char *name;
901
902		/* It should be one of these, but just skip if it happens not to be */
903		if (!IsA(selem, StatsElem))
904			continue;
905
906		name = selem->name;
907
908		if (buflen > 0)
909			buf[buflen++] = '_';	/* insert _ between names */
910
911		/*
912		 * We use fixed 'expr' for expressions, which have empty column names.
913		 * For indexes this is handled in ChooseIndexColumnNames, but we have
914		 * no such function for stats and it does not seem worth adding. If a
915		 * better name is needed, the user can specify it explicitly.
916		 */
917		if (!name)
918			name = "expr";
919
920		/*
921		 * At this point we have buflen <= NAMEDATALEN.  name should be less
922		 * than NAMEDATALEN already, but use strlcpy for paranoia.
923		 */
924		strlcpy(buf + buflen, name, NAMEDATALEN);
925		buflen += strlen(buf + buflen);
926		if (buflen >= NAMEDATALEN)
927			break;
928	}
929	return pstrdup(buf);
930}

932/*
933 * StatisticsGetRelation: given a statistics object's OID, get the OID of
934 * the relation it is defined on.  Uses the system cache.
935 */
936Oid
937StatisticsGetRelation(Oid statId, bool missing_ok)
938{
939	HeapTuple	tuple;
940	Form_pg_statistic_ext stx;
941	Oid			result;
942
943	tuple = SearchSysCache1(STATEXTOID, ObjectIdGetDatum(statId));
944	if (!HeapTupleIsValid(tuple))
945	{
946		if (missing_ok)
947			return InvalidOid;
948		elog(ERROR, "cache lookup failed for statistics object %u", statId);
949	}
950	stx = (Form_pg_statistic_ext) GETSTRUCT(tuple);
951	Assert(stx->oid == statId);
952
953	result = stx->stxrelid;
954	ReleaseSysCache(tuple);
955	return result;
