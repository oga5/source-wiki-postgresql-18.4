35# Test whether first argument is element of the list in the second
36# argument
37sub elem
38{
39	my $x = shift;
40	return grep { $_ eq $x } @_;

99# ARM ABI STABILITY CHECK HERE:
100#
101# In stable branches, set $last_nodetag to the name of the last node type
102# that should receive an auto-generated nodetag number, and $last_nodetag_no
103# to its number.  (Find these values in the last line of the current
104# nodetags.h file.)  The script will then complain if those values don't
105# match reality, providing a cross-check that we haven't broken ABI by
106# adding or removing nodetags.
107# In HEAD, these variables should be left undef, since we don't promise
108# ABI stability during development.
109
110my $last_nodetag = 'WindowObjectData';
111my $last_nodetag_no = 479;

188## check that we have the expected number of files on the command line

192## read input
193
194my $next_input_file = 0;
195foreach my $infile (@ARGV)
196{
197	my $in_struct;
198	my $in_struct_lineno;
199	my $subline;
200	my $is_node_struct;
201	my $supertype;
202	my $supertype_field;
203
204	my $node_attrs = '';
205	my $node_attrs_lineno;
206	my @my_fields;
207	my %my_field_types;
208	my %my_field_attrs;
209
210	# open file with name from command line, which may have a path prefix
211	open my $ifh, '<', $infile or die "could not open \"$infile\": $!";
212
213	# now shorten filename for use below
214	$infile =~ s!.*src/include/!!;
215
216	# check it against next member of @all_input_files
217	die "wrong input file ordering, expected @all_input_files\n"
218	  if ($infile ne $all_input_files[$next_input_file]);
219	$next_input_file++;
220
221	my $raw_file_content = do { local $/; <$ifh> };
222
223	# strip C comments, preserving newlines so we can count lines correctly
224	my $file_content = '';
225	while ($raw_file_content =~ m{^(.*?)(/\*.*?\*/)(.*)$}s)
226	{
227		$file_content .= $1;
228		my $comment = $2;
229		$raw_file_content = $3;
230		$comment =~ tr/\n//cd;
231		$file_content .= $comment;
232	}
233	$file_content .= $raw_file_content;
234
235	my $lineno = 0;
236	my $prevline = '';
237	foreach my $line (split /\n/, $file_content)
238	{
239		# per-physical-line processing
240		$lineno++;
241		chomp $line;
242		$line =~ s/\s*$//;
243		next if $line eq '';
244		next if $line =~ /^#(define|ifdef|endif)/;
245
246		# within a struct, don't process until we have whole logical line
247		if ($in_struct && $subline > 0)
248		{
249			if ($line =~ /;$/)
250			{
251				# found the end, re-attach any previous line(s)
252				$line = $prevline . $line;
253				$prevline = '';
254			}
255			elsif ($prevline eq ''
256				&& $line =~ /^\s*pg_node_attr\(([\w(), ]*)\)$/)
257			{
258				# special case: node-attributes line doesn't end with semi
259			}
260			else
261			{
262				# set it aside for a moment
263				$prevline .= $line . ' ';
264				next;
265			}
266		}
267
268		# we are analyzing a struct definition
269		if ($in_struct)
270		{
271			$subline++;
272
273			# first line should have opening brace
274			if ($subline == 1)
275			{
276				$is_node_struct = 0;
277				$supertype = undef;
278				next if $line eq '{';
279				die "$infile:$lineno: expected opening brace\n";
280			}
281			# second line could be node attributes
282			elsif ($subline == 2
283				&& $line =~ /^\s*pg_node_attr\(([\w(), ]*)\)$/)
284			{
285				$node_attrs = $1;
286				$node_attrs_lineno = $lineno;
287				# hack: don't count the line
288				$subline--;
289				next;
290			}
291			# next line should have node tag or supertype
292			elsif ($subline == 2)
293			{
294				if ($line =~ /^\s*NodeTag\s+type;/)
295				{
296					$is_node_struct = 1;
297					next;
298				}
299				elsif ($line =~ /\s*(\w+)\s+(\w+);/ and elem $1, @node_types)
300				{
301					$is_node_struct = 1;
302					$supertype = $1;
303					$supertype_field = $2;
304					next;
305				}
306			}
307
308			# end of struct
309			if ($line =~ /^\}\s*(?:\Q$in_struct\E\s*)?;$/)
310			{
311				if ($is_node_struct)
312				{
313					# This is the end of a node struct definition.
314					# Save everything we have collected.
315
316					foreach my $attr (split /,\s*/, $node_attrs)
317					{
318						if ($attr eq 'abstract')
319						{
320							push @abstract_types, $in_struct;
321						}
322						elsif ($attr eq 'custom_copy_equal')
323						{
324							push @custom_copy_equal, $in_struct;
325						}
326						elsif ($attr eq 'custom_read_write')
327						{
328							push @custom_read_write, $in_struct;
329						}
330						elsif ($attr eq 'custom_query_jumble')
331						{
332							push @custom_query_jumble, $in_struct;
333						}
334						elsif ($attr eq 'no_copy')
335						{
336							push @no_copy, $in_struct;
337						}
338						elsif ($attr eq 'no_equal')
339						{
340							push @no_equal, $in_struct;
341						}
342						elsif ($attr eq 'no_copy_equal')
343						{
344							push @no_copy, $in_struct;
345							push @no_equal, $in_struct;
346						}
347						elsif ($attr eq 'no_query_jumble')
348						{
349							push @no_query_jumble, $in_struct;
350						}
351						elsif ($attr eq 'no_read')
352						{
353							push @no_read, $in_struct;
354						}
355						elsif ($attr eq 'nodetag_only')
356						{
357							push @nodetag_only, $in_struct;
358						}
359						elsif ($attr eq 'special_read_write')
360						{
361							push @special_read_write, $in_struct;
362						}
363						elsif ($attr =~ /^nodetag_number\((\d+)\)$/)
364						{
365							$manual_nodetag_number{$in_struct} = $1;
366						}
367						else
368						{
369							die
370							  "$infile:$node_attrs_lineno: unrecognized attribute \"$attr\"\n";
371						}
372					}
373
374					# node name
375					push @node_types, $in_struct;
376
377					# field names, types, attributes
378					my @f = @my_fields;
379					my %ft = %my_field_types;
380					my %fa = %my_field_attrs;
381
382					# If there is a supertype, add those fields, too.
383					if ($supertype)
384					{
385						my @superfields;
386						foreach
387						  my $sf (@{ $node_type_info{$supertype}->{fields} })
388						{
389							my $fn = "${supertype_field}.$sf";
390							push @superfields, $fn;
391							$ft{$fn} =
392							  $node_type_info{$supertype}->{field_types}{$sf};
393							if ($node_type_info{$supertype}
394								->{field_attrs}{$sf})
395							{
396								# Copy any attributes, adjusting array_size field references
397								my @newa = @{ $node_type_info{$supertype}
398									  ->{field_attrs}{$sf} };
399								foreach my $a (@newa)
400								{
401									$a =~
402									  s/array_size\((\w+)\)/array_size(${supertype_field}.$1)/;
403								}
404								$fa{$fn} = \@newa;
405							}
406						}
407						unshift @f, @superfields;
408					}
409					# save in global info structure
410					$node_type_info{$in_struct}->{fields} = \@f;
411					$node_type_info{$in_struct}->{field_types} = \%ft;
412					$node_type_info{$in_struct}->{field_attrs} = \%fa;
413
414					# Propagate nodetag_only marking from files to nodes
415					push @nodetag_only, $in_struct
416					  if (elem $infile, @nodetag_only_files);
417
418					# Propagate some node attributes from supertypes
419					if ($supertype)
420					{
421						push @no_copy, $in_struct
422						  if elem $supertype, @no_copy;
423						push @no_equal, $in_struct
424						  if elem $supertype, @no_equal;
425						push @no_read, $in_struct
426						  if elem $supertype, @no_read;
427						push @no_query_jumble, $in_struct
428						  if elem $supertype, @no_query_jumble;
429					}
430				}
431
432				# start new cycle
433				$in_struct = undef;
434				$node_attrs = '';
435				@my_fields = ();
436				%my_field_types = ();
437				%my_field_attrs = ();
438			}
439			# normal struct field
440			elsif ($line =~
441				/^\s*(.+)\s*\b(\w+)(\[[\w\s+]+\])?\s*(?:pg_node_attr\(([\w(), ]*)\))?;/
442			  )
443			{
444				if ($is_node_struct)
445				{
446					my $type = $1;
447					my $name = $2;
448					my $array_size = $3;
449					my $attrs = $4;
450
451					# strip "const"
452					$type =~ s/^const\s*//;
453					# strip trailing space
454					$type =~ s/\s*$//;
455					# strip space between type and "*" (pointer) */
456					$type =~ s/\s+\*$/*/;
457					# strip space between type and "**" (array of pointers) */
458					$type =~ s/\s+\*\*$/**/;
459
460					die
461					  "$infile:$lineno: cannot parse data type in \"$line\"\n"
462					  if $type eq '';
463
464					my @attrs;
465					if ($attrs)
466					{
467						@attrs = split /,\s*/, $attrs;
468						foreach my $attr (@attrs)
469						{
470							if (   $attr !~ /^array_size\(\w+\)$/
471								&& $attr !~ /^copy_as\(\w+\)$/
472								&& $attr !~ /^read_as\(\w+\)$/
473								&& !elem $attr,
474								qw(copy_as_scalar
475								custom_query_jumble
476								equal_as_scalar
477								equal_ignore
478								equal_ignore_if_zero
479								query_jumble_ignore
480								query_jumble_location
481								query_jumble_squash
482								read_write_ignore
483								write_only_relids
484								write_only_nondefault_pathtarget
485								write_only_req_outer))
486							{
487								die
488								  "$infile:$lineno: unrecognized attribute \"$attr\"\n";
489							}
490						}
491					}
492
493					$type = $type . $array_size if $array_size;
494					push @my_fields, $name;
495					$my_field_types{$name} = $type;
496					$my_field_attrs{$name} = \@attrs;
497				}
498			}
499			# function pointer field
500			elsif ($line =~
501				/^\s*([\w\s*]+)\s*\(\*(\w+)\)\s*\((.*)\)\s*(?:pg_node_attr\(([\w(), ]*)\))?;/
502			  )
503			{
504				if ($is_node_struct)
505				{
506					my $type = $1;
507					my $name = $2;
508					my $args = $3;
509					my $attrs = $4;
510
511					my @attrs;
512					if ($attrs)
513					{
514						@attrs = split /,\s*/, $attrs;
515						foreach my $attr (@attrs)
516						{
517							if (   $attr !~ /^copy_as\(\w+\)$/
518								&& $attr !~ /^read_as\(\w+\)$/
519								&& !elem $attr,
520								qw(equal_ignore read_write_ignore))
521							{
522								die
523								  "$infile:$lineno: unrecognized attribute \"$attr\"\n";
524							}
525						}
526					}
527
528					push @my_fields, $name;
529					$my_field_types{$name} = 'function pointer';
530					$my_field_attrs{$name} = \@attrs;
531				}
532			}
533			else
534			{
535				# We're not too picky about what's outside structs,
536				# but we'd better understand everything inside.
537				die "$infile:$lineno: could not parse \"$line\"\n";
538			}
539		}
540		# not in a struct
541		else
542		{
543			# start of a struct?
544			if ($line =~ /^(?:typedef )?struct (\w+)$/ && $1 ne 'Node')
545			{
546				$in_struct = $1;
547				$in_struct_lineno = $lineno;
548				$subline = 0;
549			}
550			# one node type typedef'ed directly from another
551			elsif ($line =~ /^typedef (\w+) (\w+);$/ and elem $1, @node_types)
552			{
553				my $alias_of = $1;
554				my $n = $2;
555
556				# copy everything over
557				push @node_types, $n;
558				my @f = @{ $node_type_info{$alias_of}->{fields} };
559				my %ft = %{ $node_type_info{$alias_of}->{field_types} };
560				my %fa = %{ $node_type_info{$alias_of}->{field_attrs} };
561				$node_type_info{$n}->{fields} = \@f;
562				$node_type_info{$n}->{field_types} = \%ft;
563				$node_type_info{$n}->{field_attrs} = \%fa;
564			}
565			# collect enum names
566			elsif ($line =~ /^typedef enum (\w+)(\s*\/\*.*)?$/)
567			{
568				push @enum_types, $1;
569			}
570		}
571	}
572
573	if ($in_struct)
574	{
575		die
576		  "$infile:$in_struct_lineno: could not find closing brace for struct \"$in_struct\"\n";
577	}
578
579	close $ifh;
580}    # for each file

589  '/*-------------------------------------------------------------------------

591 * %s
592 *    Generated node infrastructure code
593 *
594 * Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
595 * Portions Copyright (c) 1994, Regents of the University of California
596 *
597 * NOTES
598 *  ******************************
599 *  *** DO NOT EDIT THIS FILE! ***
600 *  ******************************
601 *
602 *  It has been GENERATED by src/backend/nodes/gen_node_support.pl
603 *

609# nodetags.h
610
611push @output_files, 'nodetags.h';
612open my $nt, '>', "$output_path/nodetags.h$tmpext"
613  or die "$output_path/nodetags.h$tmpext: $!";
614
615printf $nt $header_comment, 'nodetags.h';
616
617my $tagno = 0;
618my $last_tag = undef;
619foreach my $n (@node_types, @extra_tags)
620{
621	next if elem $n, @abstract_types;
622	if (defined $manual_nodetag_number{$n})
623	{
624		# do not change $tagno or $last_tag
625		print $nt "\tT_${n} = $manual_nodetag_number{$n},\n";
626	}
627	else
628	{
629		$tagno++;
630		$last_tag = $n;
631		print $nt "\tT_${n} = $tagno,\n";
632	}
633}
634
635# verify that last auto-assigned nodetag stays stable
636die "ABI stability break: last nodetag is $last_tag not $last_nodetag\n"
637  if (defined $last_nodetag && $last_nodetag ne $last_tag);
638die
639  "ABI stability break: last nodetag number is $tagno not $last_nodetag_no\n"
640  if (defined $last_nodetag_no && $last_nodetag_no != $tagno);
641
642close $nt;
643
644
645# make #include lines necessary to pull in all the struct definitions
646my $node_includes = '';
647foreach my $infile (sort @ARGV)
648{
649	$infile =~ s!.*src/include/!!;
650	$node_includes .= qq{#include "$infile"\n};
651}
652
653
654# copyfuncs.c, equalfuncs.c
655
656push @output_files, 'copyfuncs.funcs.c';
657open my $cff, '>', "$output_path/copyfuncs.funcs.c$tmpext" or die $!;
658push @output_files, 'equalfuncs.funcs.c';
659open my $eff, '>', "$output_path/equalfuncs.funcs.c$tmpext" or die $!;
660push @output_files, 'copyfuncs.switch.c';
661open my $cfs, '>', "$output_path/copyfuncs.switch.c$tmpext" or die $!;
662push @output_files, 'equalfuncs.switch.c';
663open my $efs, '>', "$output_path/equalfuncs.switch.c$tmpext" or die $!;
664
665printf $cff $header_comment, 'copyfuncs.funcs.c';
666printf $eff $header_comment, 'equalfuncs.funcs.c';
667printf $cfs $header_comment, 'copyfuncs.switch.c';
668printf $efs $header_comment, 'equalfuncs.switch.c';
669
670# add required #include lines to each file set
671print $cff $node_includes;
672print $eff $node_includes;
673
674foreach my $n (@node_types)
675{
676	next if elem $n, @abstract_types;
677	next if elem $n, @nodetag_only;
678	my $struct_no_copy = (elem $n, @no_copy);
679	my $struct_no_equal = (elem $n, @no_equal);
680	next if $struct_no_copy && $struct_no_equal;
681
682	print $cfs "\t\tcase T_${n}:\n"
683	  . "\t\t\tretval = _copy${n}(from);\n"
684	  . "\t\t\tbreak;\n"
685	  unless $struct_no_copy;
686
687	print $efs "\t\tcase T_${n}:\n"

832			}
833		}
834		elsif ($t eq 'function pointer')
835		{

837			print $cff "\tCOPY_SCALAR_FIELD($f);\n" unless $copy_ignore;

839		}

841		elsif (($t =~ /^(\w+)\*$/ or $t =~ /^struct\s+(\w+)\*$/)
842			and elem $1, @node_types)

848			  and !$copy_ignore;

851			  if (elem $1, @no_equal or elem $1, @nodetag_only)
852			  and $1 ne 'List'
853			  and !$equal_ignore;

856			print $eff "\tCOMPARE_NODE_FIELD($f);\n" unless $equal_ignore;
857		}
858		# array (inline)
859		elsif ($t =~ /^\w+\[\w+\]$/)
860		{
861			print $cff "\tCOPY_ARRAY_FIELD($f);\n" unless $copy_ignore;
862			print $eff "\tCOMPARE_ARRAY_FIELD($f);\n" unless $equal_ignore;
863		}
864		elsif ($t eq 'struct CustomPathMethods*'
865			|| $t eq 'struct CustomScanMethods*')
866		{
867			# Fields of these types are required to be a pointer to a
868			# static table of callback functions.  So we don't copy
869			# the table itself, just reference the original one.
870			print $cff "\tCOPY_SCALAR_FIELD($f);\n" unless $copy_ignore;
871			print $eff "\tCOMPARE_SCALAR_FIELD($f);\n" unless $equal_ignore;
872		}
873		else
874		{
875			die
876			  "could not handle type \"$t\" in struct \"$n\" field \"$f\"\n";
877		}
878
879		$previous_fields{$f} = 1;
880	}
881
882	print $cff "
883\treturn newnode;
884}
885" unless $struct_no_copy;
886	print $eff "
887\treturn true;
888}
889" unless $struct_no_equal;
890}

1287" unless $struct_no_query_jumble;

1318		}
