25# Build a small utility static lib for the parser. The generation of the
26# parser is slow, and building this separately avoids other parts of the
27# backend having to wait till gram.h is generated.
28parser_sources = files('parser.c')
29
30backend_scanner = custom_target('scan',
31  input: 'scan.l',
32  output: 'scan.c',
33  command: [flex_cmd, '--no-backup', '--', '-CF', '-p', '-p'],
34)
35generated_sources += backend_scanner
36parser_sources += backend_scanner
37
38backend_parser = custom_target('gram',
39  input: 'gram.y',
40  kwargs: bison_kw,
41)
42generated_sources += backend_parser.to_list()
43parser_sources += backend_parser
44
45parser = static_library('parser',
46  parser_sources,
47  dependencies: [backend_code],
48  include_directories: include_directories('.'),
49  kwargs: internal_lib_args,
50)
51backend_link_with += parser
