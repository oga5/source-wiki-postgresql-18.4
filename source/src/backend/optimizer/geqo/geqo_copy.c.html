39/* geqo_copy
40 *
41 *	 copies one gene to another
42 *
43 */
44void
45geqo_copy(PlannerInfo *root, Chromosome *chromo1, Chromosome *chromo2,
46		  int string_length)
47{
48	int			i;
49
50	for (i = 0; i < string_length; i++)
51		chromo1->string[i] = chromo2->string[i];
52
53	chromo1->worth = chromo2->worth;
54}
