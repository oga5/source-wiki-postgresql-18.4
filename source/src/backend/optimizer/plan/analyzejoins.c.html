37/*
38 * One element of the list passed to query_is_distinct_for_with_collations().
39 * Each entry names a subquery output column that the caller needs to be
40 * distinct over, plus the upper-level equality operator and its input
41 * collation, so that the subquery's own DISTINCT/GROUP BY/set-op clauses can
42 * be compared for compatibility.
43 */
44typedef struct DistinctColInfo
45{
46	int			colno;			/* subquery output column resno */
47	Oid			opid;			/* upper-level equality operator */
48	Oid			collid;			/* input collation of opid */
49} DistinctColInfo;

97/*
98 * remove_useless_joins
99 *		Check for relations that don't actually need to be joined at all,
100 *		and remove them from the query.
101 *
102 * We are passed the current joinlist and return the updated list.  Other
103 * data structures that have to be updated are accessible via "root".
104 */
105List *
106remove_useless_joins(PlannerInfo *root, List *joinlist)
107{
108	ListCell   *lc;
109
110	/*
111	 * We are only interested in relations that are left-joined to, so we can
112	 * scan the join_info_list to find them easily.
113	 */
114restart:
115	foreach(lc, root->join_info_list)
116	{
117		SpecialJoinInfo *sjinfo = (SpecialJoinInfo *) lfirst(lc);
118		int			innerrelid;
119		int			nremoved;
120
121		/* Skip if not removable */
122		if (!join_is_removable(root, sjinfo))
123			continue;
124
125		/*
126		 * Currently, join_is_removable can only succeed when the sjinfo's
127		 * righthand is a single baserel.  Remove that rel from the query and
128		 * joinlist.
129		 */
130		innerrelid = bms_singleton_member(sjinfo->min_righthand);
131
132		remove_leftjoinrel_from_query(root, innerrelid, sjinfo);
133
134		/* We verify that exactly one reference gets removed from joinlist */
135		nremoved = 0;
136		joinlist = remove_rel_from_joinlist(joinlist, innerrelid, &nremoved);
137		if (nremoved != 1)
138			elog(ERROR, "failed to find relation %d in joinlist", innerrelid);
139
140		/*
141		 * We can delete this SpecialJoinInfo from the list too, since it's no
142		 * longer of interest.  (Since we'll restart the foreach loop
143		 * immediately, we don't bother with foreach_delete_current.)
144		 */
145		root->join_info_list = list_delete_cell(root->join_info_list, lc);
146
147		/*
148		 * Restart the scan.  This is necessary to ensure we find all
149		 * removable joins independently of ordering of the join_info_list
150		 * (note that removal of attr_needed bits may make a join appear
151		 * removable that did not before).
152		 */
153		goto restart;
154	}
155
156	return joinlist;
157}

159/*
160 * join_is_removable
161 *	  Check whether we need not perform this special join at all, because
162 *	  it will just duplicate its left input.
163 *
164 * This is true for a left join for which the join condition cannot match
165 * more than one inner-side row.  (There are other possibly interesting
166 * cases, but we don't have the infrastructure to prove them.)  We also
167 * have to check that the inner side doesn't generate any variables needed
168 * above the join.
169 */
170static bool
171join_is_removable(PlannerInfo *root, SpecialJoinInfo *sjinfo)
172{
173	int			innerrelid;
174	RelOptInfo *innerrel;
175	Relids		inputrelids;
176	Relids		joinrelids;
177	List	   *clause_list = NIL;
178	ListCell   *l;
179	int			attroff;
180
181	/*
182	 * Must be a left join to a single baserel, else we aren't going to be
183	 * able to do anything with it.
184	 */
185	if (sjinfo->jointype != JOIN_LEFT)
186		return false;
187
188	if (!bms_get_singleton_member(sjinfo->min_righthand, &innerrelid))
189		return false;
190
191	/*
192	 * Never try to eliminate a left join to the query result rel.  Although
193	 * the case is syntactically impossible in standard SQL, MERGE will build
194	 * a join tree that looks exactly like that.
195	 */
196	if (innerrelid == root->parse->resultRelation)
197		return false;
198
199	innerrel = find_base_rel(root, innerrelid);
200
201	/*
202	 * Before we go to the effort of checking whether any innerrel variables
203	 * are needed above the join, make a quick check to eliminate cases in
204	 * which we will surely be unable to prove uniqueness of the innerrel.
205	 */
206	if (!rel_supports_distinctness(root, innerrel))
207		return false;
208
209	/* Compute the relid set for the join we are considering */
210	inputrelids = bms_union(sjinfo->min_lefthand, sjinfo->min_righthand);
211	Assert(sjinfo->ojrelid != 0);
212	joinrelids = bms_copy(inputrelids);
213	joinrelids = bms_add_member(joinrelids, sjinfo->ojrelid);
214
215	/*
216	 * We can't remove the join if any inner-rel attributes are used above the
217	 * join.  Here, "above" the join includes pushed-down conditions, so we
218	 * should reject if attr_needed includes the OJ's own relid; therefore,
219	 * compare to inputrelids not joinrelids.
220	 *
221	 * As a micro-optimization, it seems better to start with max_attr and
222	 * count down rather than starting with min_attr and counting up, on the
223	 * theory that the system attributes are somewhat less likely to be wanted
224	 * and should be tested last.
225	 */
226	for (attroff = innerrel->max_attr - innerrel->min_attr;
227		 attroff >= 0;
228		 attroff--)
229	{
230		if (!bms_is_subset(innerrel->attr_needed[attroff], inputrelids))
231			return false;
232	}
233
234	/*
235	 * Similarly check that the inner rel isn't needed by any PlaceHolderVars
236	 * that will be used above the join.  The PHV case is a little bit more
237	 * complicated, because PHVs may have been assigned a ph_eval_at location
238	 * that includes the innerrel, yet their contained expression might not
239	 * actually reference the innerrel (it could be just a constant, for
240	 * instance).  If such a PHV is due to be evaluated above the join then it
241	 * needn't prevent join removal.
242	 */
243	foreach(l, root->placeholder_list)
244	{
245		PlaceHolderInfo *phinfo = (PlaceHolderInfo *) lfirst(l);
246
247		if (bms_overlap(phinfo->ph_lateral, innerrel->relids))
248			return false;		/* it references innerrel laterally */
249		if (!bms_overlap(phinfo->ph_eval_at, innerrel->relids))
250			continue;			/* it definitely doesn't reference innerrel */
251		if (bms_is_subset(phinfo->ph_needed, inputrelids))
252			continue;			/* PHV is not used above the join */
253		if (!bms_is_member(sjinfo->ojrelid, phinfo->ph_eval_at))
254			return false;		/* it has to be evaluated below the join */
255
256		/*
257		 * We need to be sure there will still be a place to evaluate the PHV
258		 * if we remove the join, ie that ph_eval_at wouldn't become empty.
259		 */
260		if (!bms_overlap(sjinfo->min_lefthand, phinfo->ph_eval_at))
261			return false;		/* there isn't any other place to eval PHV */
262		/* Check contained expression last, since this is a bit expensive */
263		if (bms_overlap(pull_varnos(root, (Node *) phinfo->ph_var->phexpr),
264						innerrel->relids))
265			return false;		/* contained expression references innerrel */
266	}
267
268	/*
269	 * Search for mergejoinable clauses that constrain the inner rel against
270	 * either the outer rel or a pseudoconstant.  If an operator is
271	 * mergejoinable then it behaves like equality for some btree opclass, so
272	 * it's what we want.  The mergejoinability test also eliminates clauses
273	 * containing volatile functions, which we couldn't depend on.
274	 */
275	foreach(l, innerrel->joininfo)
276	{
277		RestrictInfo *restrictinfo = (RestrictInfo *) lfirst(l);
278
279		/*
280		 * If the current join commutes with some other outer join(s) via
281		 * outer join identity 3, there will be multiple clones of its join
282		 * clauses in the joininfo list.  We want to consider only the
283		 * has_clone form of such clauses.  Processing more than one form
284		 * would be wasteful, and also some of the others would confuse the
285		 * RINFO_IS_PUSHED_DOWN test below.
286		 */
287		if (restrictinfo->is_clone)
288			continue;			/* ignore it */
289
290		/*
291		 * If it's not a join clause for this outer join, we can't use it.
292		 * Note that if the clause is pushed-down, then it is logically from
293		 * above the outer join, even if it references no other rels (it might
294		 * be from WHERE, for example).
295		 */
296		if (RINFO_IS_PUSHED_DOWN(restrictinfo, joinrelids))
297			continue;			/* ignore; not useful here */
298
299		/* Ignore if it's not a mergejoinable clause */
300		if (!restrictinfo->can_join ||
301			restrictinfo->mergeopfamilies == NIL)
302			continue;			/* not mergejoinable */
303
304		/*
305		 * Check if the clause has the form "outer op inner" or "inner op
306		 * outer", and if so mark which side is inner.
307		 */
308		if (!clause_sides_match_join(restrictinfo, sjinfo->min_lefthand,
309									 innerrel->relids))
310			continue;			/* no good for these input relations */
311
312		/* OK, add to list */
313		clause_list = lappend(clause_list, restrictinfo);
314	}
315
316	/*
317	 * Now that we have the relevant equality join clauses, try to prove the
318	 * innerrel distinct.
319	 */
320	if (rel_is_distinct_for(root, innerrel, clause_list, NULL))
321		return true;
322
323	/*
324	 * Some day it would be nice to check for other methods of establishing
325	 * distinctness.
326	 */
327	return false;
328}

331/*
332 * Remove the target rel->relid and references to the target join from the
333 * planner's data structures, having determined that there is no need
334 * to include them in the query. Optionally replace them with subst if subst
335 * is non-negative.
336 *
337 * This function updates only parts needed for both left-join removal and
338 * self-join removal.
339 */
340static void
341remove_rel_from_query(PlannerInfo *root, RelOptInfo *rel,
342					  int subst, SpecialJoinInfo *sjinfo,
343					  Relids joinrelids)
344{
345	int			relid = rel->relid;
346	Index		rti;
347	ListCell   *l;
348
349	/*
350	 * Update all_baserels and related relid sets.
351	 */
352	root->all_baserels = adjust_relid_set(root->all_baserels, relid, subst);
353	root->all_query_rels = adjust_relid_set(root->all_query_rels, relid, subst);
354
355	if (sjinfo != NULL)
356	{
357		root->outer_join_rels = bms_del_member(root->outer_join_rels,
358											   sjinfo->ojrelid);
359		root->all_query_rels = bms_del_member(root->all_query_rels,
360											  sjinfo->ojrelid);
361	}
362
363	/*
364	 * Likewise remove references from SpecialJoinInfo data structures.
365	 *
366	 * This is relevant in case the outer join we're deleting is nested inside
367	 * other outer joins: the upper joins' relid sets have to be adjusted. The
368	 * RHS of the target outer join will be made empty here, but that's OK
369	 * since caller will delete that SpecialJoinInfo entirely.
370	 */
371	foreach(l, root->join_info_list)
372	{
373		SpecialJoinInfo *sjinf = (SpecialJoinInfo *) lfirst(l);
374
375		/*
376		 * initsplan.c is fairly cavalier about allowing SpecialJoinInfos'
377		 * lefthand/righthand relid sets to be shared with other data
378		 * structures.  Ensure that we don't modify the original relid sets.
379		 * (The commute_xxx sets are always per-SpecialJoinInfo though.)
380		 */
381		sjinf->min_lefthand = bms_copy(sjinf->min_lefthand);
382		sjinf->min_righthand = bms_copy(sjinf->min_righthand);
383		sjinf->syn_lefthand = bms_copy(sjinf->syn_lefthand);
384		sjinf->syn_righthand = bms_copy(sjinf->syn_righthand);
385		/* Now remove relid from the sets: */
386		sjinf->min_lefthand = adjust_relid_set(sjinf->min_lefthand, relid, subst);
387		sjinf->min_righthand = adjust_relid_set(sjinf->min_righthand, relid, subst);
388		sjinf->syn_lefthand = adjust_relid_set(sjinf->syn_lefthand, relid, subst);
389		sjinf->syn_righthand = adjust_relid_set(sjinf->syn_righthand, relid, subst);
390
391		if (sjinfo != NULL)
392		{
393			Assert(subst <= 0);
394
395			/* Remove sjinfo->ojrelid bits from the sets: */
396			sjinf->min_lefthand = bms_del_member(sjinf->min_lefthand,
397												 sjinfo->ojrelid);
398			sjinf->min_righthand = bms_del_member(sjinf->min_righthand,
399												  sjinfo->ojrelid);
400			sjinf->syn_lefthand = bms_del_member(sjinf->syn_lefthand,
401												 sjinfo->ojrelid);
402			sjinf->syn_righthand = bms_del_member(sjinf->syn_righthand,
403												  sjinfo->ojrelid);
404			/* relid cannot appear in these fields, but ojrelid can: */
405			sjinf->commute_above_l = bms_del_member(sjinf->commute_above_l,
406													sjinfo->ojrelid);
407			sjinf->commute_above_r = bms_del_member(sjinf->commute_above_r,
408													sjinfo->ojrelid);
409			sjinf->commute_below_l = bms_del_member(sjinf->commute_below_l,
410													sjinfo->ojrelid);
411			sjinf->commute_below_r = bms_del_member(sjinf->commute_below_r,
412													sjinfo->ojrelid);
413		}
414		else
415		{
416			Assert(subst > 0);
417
418			ChangeVarNodesExtended((Node *) sjinf->semi_rhs_exprs, relid, subst,
419								   0, replace_relid_callback);
420		}
421	}
422
423	/*
424	 * Likewise remove references from PlaceHolderVar data structures,
425	 * removing any no-longer-needed placeholders entirely.  We remove PHV
426	 * only for left-join removal.  With self-join elimination, PHVs already
427	 * get moved to the remaining relation, where they might still be needed.
428	 * It might also happen that we skip the removal of some PHVs that could
429	 * be removed.  However, the overhead of extra PHVs is small compared to
430	 * the complexity of analysis needed to remove them.
431	 *
432	 * Removal is a bit trickier than it might seem: we can remove PHVs that
433	 * are used at the target rel and/or in the join qual, but not those that
434	 * are used at join partner rels or above the join.  It's not that easy to
435	 * distinguish PHVs used at partner rels from those used in the join qual,
436	 * since they will both have ph_needed sets that are subsets of
437	 * joinrelids.  However, a PHV used at a partner rel could not have the
438	 * target rel in ph_eval_at, so we check that while deciding whether to
439	 * remove or just update the PHV.  There is no corresponding test in
440	 * join_is_removable because it doesn't need to distinguish those cases.
441	 */
442	foreach(l, root->placeholder_list)
443	{
444		PlaceHolderInfo *phinfo = (PlaceHolderInfo *) lfirst(l);
445
446		Assert(sjinfo == NULL || !bms_is_member(relid, phinfo->ph_lateral));
447		if (sjinfo != NULL &&
448			bms_is_subset(phinfo->ph_needed, joinrelids) &&
449			bms_is_member(relid, phinfo->ph_eval_at) &&
450			!bms_is_member(sjinfo->ojrelid, phinfo->ph_eval_at))
451		{
452			/*
453			 * This code shouldn't be executed if one relation is substituted
454			 * with another: in this case, the placeholder may be employed in
455			 * a filter inside the scan node the SJE removes.
456			 */
457			root->placeholder_list = foreach_delete_current(root->placeholder_list,
458															l);
459			root->placeholder_array[phinfo->phid] = NULL;
460		}
461		else
462		{
463			PlaceHolderVar *phv = phinfo->ph_var;
464
465			phinfo->ph_eval_at = adjust_relid_set(phinfo->ph_eval_at, relid, subst);
466			if (sjinfo != NULL)
467				phinfo->ph_eval_at = adjust_relid_set(phinfo->ph_eval_at,
468													  sjinfo->ojrelid, subst);
469			Assert(!bms_is_empty(phinfo->ph_eval_at));	/* checked previously */
470			/* Reduce ph_needed to contain only "relation 0"; see below */
471			if (bms_is_member(0, phinfo->ph_needed))
472				phinfo->ph_needed = bms_make_singleton(0);
473			else
474				phinfo->ph_needed = NULL;
475
476			phinfo->ph_lateral = adjust_relid_set(phinfo->ph_lateral, relid, subst);
477
478			/*
479			 * ph_lateral might contain rels mentioned in ph_eval_at after the
480			 * replacement, remove them.
481			 */
482			phinfo->ph_lateral = bms_difference(phinfo->ph_lateral, phinfo->ph_eval_at);
483			/* ph_lateral might or might not be empty */
484
485			phv->phrels = adjust_relid_set(phv->phrels, relid, subst);
486			if (sjinfo != NULL)
487				phv->phrels = adjust_relid_set(phv->phrels,
488											   sjinfo->ojrelid, subst);
489			Assert(!bms_is_empty(phv->phrels));
490
491			ChangeVarNodesExtended((Node *) phv->phexpr, relid, subst, 0,
492								   replace_relid_callback);
493
494			Assert(phv->phnullingrels == NULL); /* no need to adjust */
495		}
496	}
497
498	/*
499	 * Likewise remove references from EquivalenceClasses.
500	 */
501	foreach(l, root->eq_classes)
502	{
503		EquivalenceClass *ec = (EquivalenceClass *) lfirst(l);
504
505		if (bms_is_member(relid, ec->ec_relids) ||
506			(sjinfo == NULL || bms_is_member(sjinfo->ojrelid, ec->ec_relids)))
507			remove_rel_from_eclass(ec, sjinfo, relid, subst);
508	}
509
510	/*
511	 * Finally, we must recompute per-Var attr_needed and per-PlaceHolderVar
512	 * ph_needed relid sets.  These have to be known accurately, else we may
513	 * fail to remove other now-removable outer joins.  And our removal of the
514	 * join clause(s) for this outer join may mean that Vars that were
515	 * formerly needed no longer are.  So we have to do this honestly by
516	 * repeating the construction of those relid sets.  We can cheat to one
517	 * small extent: we can avoid re-examining the targetlist and HAVING qual
518	 * by preserving "relation 0" bits from the existing relid sets.  This is
519	 * safe because we'd never remove such references.
520	 *
521	 * So, start by removing all other bits from attr_needed sets and
522	 * lateral_vars lists.  (We already did this above for ph_needed.)
523	 */
524	for (rti = 1; rti < root->simple_rel_array_size; rti++)
525	{
526		RelOptInfo *otherrel = root->simple_rel_array[rti];
527		int			attroff;
528
529		/* there may be empty slots corresponding to non-baserel RTEs */
530		if (otherrel == NULL)
531			continue;
532
533		Assert(otherrel->relid == rti); /* sanity check on array */
534
535		for (attroff = otherrel->max_attr - otherrel->min_attr;
536			 attroff >= 0;
537			 attroff--)
538		{
539			if (bms_is_member(0, otherrel->attr_needed[attroff]))
540				otherrel->attr_needed[attroff] = bms_make_singleton(0);
541			else
542				otherrel->attr_needed[attroff] = NULL;
543		}
544
545		if (subst > 0)
546			ChangeVarNodesExtended((Node *) otherrel->lateral_vars, relid,
547								   subst, 0, replace_relid_callback);
548	}
549}

551/*
552 * Remove the target relid and references to the target join from the
553 * planner's data structures, having determined that there is no need
554 * to include them in the query.
555 *
556 * We are not terribly thorough here.  We only bother to update parts of
557 * the planner's data structures that will actually be consulted later.
558 */
559static void
560remove_leftjoinrel_from_query(PlannerInfo *root, int relid,
561							  SpecialJoinInfo *sjinfo)
562{
563	RelOptInfo *rel = find_base_rel(root, relid);
564	int			ojrelid = sjinfo->ojrelid;
565	Relids		joinrelids;
566	Relids		join_plus_commute;
567	List	   *joininfos;
568	ListCell   *l;
569
570	/* Compute the relid set for the join we are considering */
571	joinrelids = bms_union(sjinfo->min_lefthand, sjinfo->min_righthand);
572	Assert(ojrelid != 0);
573	joinrelids = bms_add_member(joinrelids, ojrelid);
574
575	remove_rel_from_query(root, rel, -1, sjinfo, joinrelids);
576
577	/*
578	 * Remove any joinquals referencing the rel from the joininfo lists.
579	 *
580	 * In some cases, a joinqual has to be put back after deleting its
581	 * reference to the target rel.  This can occur for pseudoconstant and
582	 * outerjoin-delayed quals, which can get marked as requiring the rel in
583	 * order to force them to be evaluated at or above the join.  We can't
584	 * just discard them, though.  Only quals that logically belonged to the
585	 * outer join being discarded should be removed from the query.
586	 *
587	 * We might encounter a qual that is a clone of a deletable qual with some
588	 * outer-join relids added (see deconstruct_distribute_oj_quals).  To
589	 * ensure we get rid of such clones as well, add the relids of all OJs
590	 * commutable with this one to the set we test against for
591	 * pushed-down-ness.
592	 */
593	join_plus_commute = bms_union(joinrelids,
594								  sjinfo->commute_above_r);
595	join_plus_commute = bms_add_members(join_plus_commute,
596										sjinfo->commute_below_l);
597
598	/*
599	 * We must make a copy of the rel's old joininfo list before starting the
600	 * loop, because otherwise remove_join_clause_from_rels would destroy the
601	 * list while we're scanning it.
602	 */
603	joininfos = list_copy(rel->joininfo);
604	foreach(l, joininfos)
605	{
606		RestrictInfo *rinfo = (RestrictInfo *) lfirst(l);
607
608		remove_join_clause_from_rels(root, rinfo, rinfo->required_relids);
609
610		if (RINFO_IS_PUSHED_DOWN(rinfo, join_plus_commute))
611		{
612			/*
613			 * There might be references to relid or ojrelid in the
614			 * RestrictInfo's relid sets, as a consequence of PHVs having had
615			 * ph_eval_at sets that include those.  We already checked above
616			 * that any such PHV is safe (and updated its ph_eval_at), so we
617			 * can just drop those references.
618			 */
619			remove_rel_from_restrictinfo(rinfo, relid, ojrelid);
620
621			/*
622			 * Cross-check that the clause itself does not reference the
623			 * target rel or join.
624			 */
625#ifdef USE_ASSERT_CHECKING
626			{
627				Relids		clause_varnos = pull_varnos(root,
628														(Node *) rinfo->clause);
629
630				Assert(!bms_is_member(relid, clause_varnos));
631				Assert(!bms_is_member(ojrelid, clause_varnos));
632			}
633#endif
634			/* Now throw it back into the joininfo lists */
635			distribute_restrictinfo_to_rels(root, rinfo);
636		}
637	}
638
639	/*
640	 * There may be references to the rel in root->fkey_list, but if so,
641	 * match_foreign_keys_to_quals() will get rid of them.
642	 */
643
644	/*
645	 * Now remove the rel from the baserel array to prevent it from being
646	 * referenced again.  (We can't do this earlier because
647	 * remove_join_clause_from_rels will touch it.)
648	 */
649	root->simple_rel_array[relid] = NULL;
650	root->simple_rte_array[relid] = NULL;
651
652	/* And nuke the RelOptInfo, just in case there's another access path */
653	pfree(rel);
654
655	/*
656	 * Now repeat construction of attr_needed bits coming from all other
657	 * sources.
658	 */
659	rebuild_placeholder_attr_needed(root);
660	rebuild_joinclause_attr_needed(root);
661	rebuild_eclass_attr_needed(root);
662	rebuild_lateral_attr_needed(root);
663}

665/*
666 * Remove any references to relid or ojrelid from the RestrictInfo.
667 *
668 * We only bother to clean out bits in the RestrictInfo's various relid sets,
669 * not nullingrel bits in contained Vars and PHVs.  (This might have to be
670 * improved sometime.)  However, if the RestrictInfo contains an OR clause
671 * we have to also clean up the sub-clauses.
672 */
673static void
674remove_rel_from_restrictinfo(RestrictInfo *rinfo, int relid, int ojrelid)
675{
676	/*
677	 * initsplan.c is fairly cavalier about allowing RestrictInfos to share
678	 * relid sets with other RestrictInfos, and SpecialJoinInfos too.  Make
679	 * sure this RestrictInfo has its own relid sets before we modify them.
680	 * (In present usage, clause_relids is probably not shared, but
681	 * required_relids could be; let's not assume anything.)
682	 */
683	rinfo->clause_relids = bms_copy(rinfo->clause_relids);
684	rinfo->clause_relids = bms_del_member(rinfo->clause_relids, relid);
685	rinfo->clause_relids = bms_del_member(rinfo->clause_relids, ojrelid);
686	/* Likewise for required_relids */
687	rinfo->required_relids = bms_copy(rinfo->required_relids);
688	rinfo->required_relids = bms_del_member(rinfo->required_relids, relid);
689	rinfo->required_relids = bms_del_member(rinfo->required_relids, ojrelid);
690	/* Likewise for incompatible_relids */
691	rinfo->incompatible_relids = bms_copy(rinfo->incompatible_relids);
692	rinfo->incompatible_relids = bms_del_member(rinfo->incompatible_relids, relid);
693	rinfo->incompatible_relids = bms_del_member(rinfo->incompatible_relids, ojrelid);
694	/* Likewise for outer_relids */
695	rinfo->outer_relids = bms_copy(rinfo->outer_relids);
696	rinfo->outer_relids = bms_del_member(rinfo->outer_relids, relid);
697	rinfo->outer_relids = bms_del_member(rinfo->outer_relids, ojrelid);
698	/* Likewise for left_relids */
699	rinfo->left_relids = bms_copy(rinfo->left_relids);
700	rinfo->left_relids = bms_del_member(rinfo->left_relids, relid);
701	rinfo->left_relids = bms_del_member(rinfo->left_relids, ojrelid);
702	/* Likewise for right_relids */
703	rinfo->right_relids = bms_copy(rinfo->right_relids);
704	rinfo->right_relids = bms_del_member(rinfo->right_relids, relid);
705	rinfo->right_relids = bms_del_member(rinfo->right_relids, ojrelid);
706
707	/* If it's an OR, recurse to clean up sub-clauses */
708	if (restriction_is_or_clause(rinfo))
709	{
710		ListCell   *lc;
711
712		Assert(is_orclause(rinfo->orclause));
713		foreach(lc, ((BoolExpr *) rinfo->orclause)->args)
714		{
715			Node	   *orarg = (Node *) lfirst(lc);
716
717			/* OR arguments should be ANDs or sub-RestrictInfos */
718			if (is_andclause(orarg))
719			{
720				List	   *andargs = ((BoolExpr *) orarg)->args;
721				ListCell   *lc2;
722
723				foreach(lc2, andargs)
724				{
725					RestrictInfo *rinfo2 = lfirst_node(RestrictInfo, lc2);
726
727					remove_rel_from_restrictinfo(rinfo2, relid, ojrelid);
728				}
729			}
730			else
731			{
732				RestrictInfo *rinfo2 = castNode(RestrictInfo, orarg);
733
734				remove_rel_from_restrictinfo(rinfo2, relid, ojrelid);
735			}
736		}
737	}
738}

740/*
741 * Remove any references to relid or sjinfo->ojrelid (if sjinfo != NULL)
742 * from the EquivalenceClass.
743 *
744 * Like remove_rel_from_restrictinfo, we don't worry about cleaning out
745 * any nullingrel bits in contained Vars and PHVs.  (This might have to be
746 * improved sometime.)  We do need to fix the EC and EM relid sets to ensure
747 * that implied join equalities will be generated at the appropriate join
748 * level(s).
749 */
750static void
751remove_rel_from_eclass(EquivalenceClass *ec, SpecialJoinInfo *sjinfo,
752					   int relid, int subst)
753{
754	ListCell   *lc;
755
756	/* Fix up the EC's overall relids */
757	ec->ec_relids = adjust_relid_set(ec->ec_relids, relid, subst);
758	if (sjinfo != NULL)
759		ec->ec_relids = adjust_relid_set(ec->ec_relids,
760										 sjinfo->ojrelid, subst);
761
762	/*
763	 * We don't expect any EC child members to exist at this point.  Ensure
764	 * that's the case, otherwise, we might be getting asked to do something
765	 * this function hasn't been coded for.
766	 */
767	Assert(ec->ec_childmembers == NULL);
768
769	/*
770	 * Fix up the member expressions.  Any non-const member that ends with
771	 * empty em_relids must be a Var or PHV of the removed relation.  We don't
772	 * need it anymore, so we can drop it.
773	 */
774	foreach(lc, ec->ec_members)
775	{
776		EquivalenceMember *cur_em = (EquivalenceMember *) lfirst(lc);
777
778		if (bms_is_member(relid, cur_em->em_relids) ||
779			(sjinfo != NULL && bms_is_member(sjinfo->ojrelid,
780											 cur_em->em_relids)))
781		{
782			Assert(!cur_em->em_is_const);
783			cur_em->em_relids = adjust_relid_set(cur_em->em_relids, relid, subst);
784			if (sjinfo != NULL)
785				cur_em->em_relids = adjust_relid_set(cur_em->em_relids,
786													 sjinfo->ojrelid, subst);
787			if (bms_is_empty(cur_em->em_relids))
788				ec->ec_members = foreach_delete_current(ec->ec_members, lc);
789		}
790	}
791
792	/* Fix up the source clauses, in case we can re-use them later */
793	foreach(lc, ec->ec_sources)
794	{
795		RestrictInfo *rinfo = (RestrictInfo *) lfirst(lc);
796
797		if (sjinfo == NULL)
798			ChangeVarNodesExtended((Node *) rinfo, relid, subst, 0,
799								   replace_relid_callback);
800		else
801			remove_rel_from_restrictinfo(rinfo, relid, sjinfo->ojrelid);
802	}
803
804	/*
805	 * Rather than expend code on fixing up any already-derived clauses, just
806	 * drop them.  (At this point, any such clauses would be base restriction
807	 * clauses, which we'd not need anymore anyway.)
808	 */
809	ec_clear_derived_clauses(ec);
810}

812/*
813 * Remove any occurrences of the target relid from a joinlist structure.
814 *
815 * It's easiest to build a whole new list structure, so we handle it that
816 * way.  Efficiency is not a big deal here.
817 *
818 * *nremoved is incremented by the number of occurrences removed (there
819 * should be exactly one, but the caller checks that).
820 */
821static List *
822remove_rel_from_joinlist(List *joinlist, int relid, int *nremoved)
823{
824	List	   *result = NIL;
825	ListCell   *jl;
826
827	foreach(jl, joinlist)
828	{
829		Node	   *jlnode = (Node *) lfirst(jl);
830
831		if (IsA(jlnode, RangeTblRef))
832		{
833			int			varno = ((RangeTblRef *) jlnode)->rtindex;
834
835			if (varno == relid)
836				(*nremoved)++;
837			else
838				result = lappend(result, jlnode);
839		}
840		else if (IsA(jlnode, List))
841		{
842			/* Recurse to handle subproblem */
843			List	   *sublist;
844
845			sublist = remove_rel_from_joinlist((List *) jlnode,
846											   relid, nremoved);
847			/* Avoid including empty sub-lists in the result */
848			if (sublist)
849				result = lappend(result, sublist);
850		}
851		else
852		{
853			elog(ERROR, "unrecognized joinlist node type: %d",
854				 (int) nodeTag(jlnode));
855		}
856	}
857
858	return result;
859}

862/*
863 * reduce_unique_semijoins
864 *		Check for semijoins that can be simplified to plain inner joins
865 *		because the inner relation is provably unique for the join clauses.
866 *
867 * Ideally this would happen during reduce_outer_joins, but we don't have
868 * enough information at that point.
869 *
870 * To perform the strength reduction when applicable, we need only delete
871 * the semijoin's SpecialJoinInfo from root->join_info_list.  (We don't
872 * bother fixing the join type attributed to it in the query jointree,
873 * since that won't be consulted again.)
874 */
875void
876reduce_unique_semijoins(PlannerInfo *root)
877{
878	ListCell   *lc;
879
880	/*
881	 * Scan the join_info_list to find semijoins.
882	 */
883	foreach(lc, root->join_info_list)
884	{
885		SpecialJoinInfo *sjinfo = (SpecialJoinInfo *) lfirst(lc);
886		int			innerrelid;
887		RelOptInfo *innerrel;
888		Relids		joinrelids;
889		List	   *restrictlist;
890
891		/*
892		 * Must be a semijoin to a single baserel, else we aren't going to be
893		 * able to do anything with it.
894		 */
895		if (sjinfo->jointype != JOIN_SEMI)
896			continue;
897
898		if (!bms_get_singleton_member(sjinfo->min_righthand, &innerrelid))
899			continue;
900
901		innerrel = find_base_rel(root, innerrelid);
902
903		/*
904		 * Before we trouble to run generate_join_implied_equalities, make a
905		 * quick check to eliminate cases in which we will surely be unable to
906		 * prove uniqueness of the innerrel.
907		 */
908		if (!rel_supports_distinctness(root, innerrel))
909			continue;
910
911		/* Compute the relid set for the join we are considering */
912		joinrelids = bms_union(sjinfo->min_lefthand, sjinfo->min_righthand);
913		Assert(sjinfo->ojrelid == 0);	/* SEMI joins don't have RT indexes */
914
915		/*
916		 * Since we're only considering a single-rel RHS, any join clauses it
917		 * has must be clauses linking it to the semijoin's min_lefthand.  We
918		 * can also consider EC-derived join clauses.
919		 */
920		restrictlist =
921			list_concat(generate_join_implied_equalities(root,
922														 joinrelids,
923														 sjinfo->min_lefthand,
924														 innerrel,
925														 NULL),
926						innerrel->joininfo);
927
928		/* Test whether the innerrel is unique for those clauses. */
929		if (!innerrel_is_unique(root,
930								joinrelids, sjinfo->min_lefthand, innerrel,
931								JOIN_SEMI, restrictlist, true))
932			continue;
933
934		/* OK, remove the SpecialJoinInfo from the list. */
935		root->join_info_list = foreach_delete_current(root->join_info_list, lc);
936	}
937}

940/*
941 * rel_supports_distinctness
942 *		Could the relation possibly be proven distinct on some set of columns?
943 *
944 * This is effectively a pre-checking function for rel_is_distinct_for().
945 * It must return true if rel_is_distinct_for() could possibly return true
946 * with this rel, but it should not expend a lot of cycles.  The idea is
947 * that callers can avoid doing possibly-expensive processing to compute
948 * rel_is_distinct_for()'s argument lists if the call could not possibly
949 * succeed.
950 */
951static bool
952rel_supports_distinctness(PlannerInfo *root, RelOptInfo *rel)
953{
954	/* We only know about baserels ... */
955	if (rel->reloptkind != RELOPT_BASEREL)
956		return false;
957	if (rel->rtekind == RTE_RELATION)
958	{
959		/*
960		 * For a plain relation, we only know how to prove uniqueness by
961		 * reference to unique indexes.  Make sure there's at least one
962		 * suitable unique index.  It must be immediately enforced, and not a
963		 * partial index. (Keep these conditions in sync with
964		 * relation_has_unique_index_for!)
965		 */
966		ListCell   *lc;
967
968		foreach(lc, rel->indexlist)
969		{
970			IndexOptInfo *ind = (IndexOptInfo *) lfirst(lc);
971
972			if (ind->unique && ind->immediate && ind->indpred == NIL)
973				return true;
974		}
975	}
976	else if (rel->rtekind == RTE_SUBQUERY)
977	{
978		Query	   *subquery = root->simple_rte_array[rel->relid]->subquery;
979
980		/* Check if the subquery has any qualities that support distinctness */
981		if (query_supports_distinctness(subquery))
982			return true;
983	}
984	/* We have no proof rules for any other rtekinds. */
985	return false;
986}

988/*
989 * rel_is_distinct_for
990 *		Does the relation return only distinct rows according to clause_list?
991 *
992 * clause_list is a list of join restriction clauses involving this rel and
993 * some other one.  Return true if no two rows emitted by this rel could
994 * possibly join to the same row of the other rel.
995 *
996 * The caller must have already determined that each condition is a
997 * mergejoinable equality with an expression in this relation on one side, and
998 * an expression not involving this relation on the other.  The transient
999 * outer_is_left flag is used to identify which side references this relation:
1000 * left side if outer_is_left is false, right side if it is true.
1001 *
1002 * Note that the passed-in clause_list may be destructively modified!  This
1003 * is OK for current uses, because the clause_list is built by the caller for
1004 * the sole purpose of passing to this function.
1005 *
1006 * (*extra_clauses) to be set to the right sides of baserestrictinfo clauses,
1007 * looking like "x = const" if distinctness is derived from such clauses, not
1008 * joininfo clauses.  Pass NULL to the extra_clauses if this value is not
1009 * needed.
1010 */
1011static bool
1012rel_is_distinct_for(PlannerInfo *root, RelOptInfo *rel, List *clause_list,
1013					List **extra_clauses)
1014{
1015	/*
1016	 * We could skip a couple of tests here if we assume all callers checked
1017	 * rel_supports_distinctness first, but it doesn't seem worth taking any
1018	 * risk for.
1019	 */
1020	if (rel->reloptkind != RELOPT_BASEREL)
1021		return false;
1022	if (rel->rtekind == RTE_RELATION)
1023	{
1024		/*
1025		 * Examine the indexes to see if we have a matching unique index.
1026		 * relation_has_unique_index_ext automatically adds any usable
1027		 * restriction clauses for the rel, so we needn't do that here.
1028		 */
1029		if (relation_has_unique_index_ext(root, rel, clause_list, NIL, NIL,
1030										  extra_clauses))
1031			return true;
1032	}
1033	else if (rel->rtekind == RTE_SUBQUERY)
1034	{
1035		Index		relid = rel->relid;
1036		Query	   *subquery = root->simple_rte_array[relid]->subquery;
1037		List	   *distinct_cols = NIL;
1038		ListCell   *l;
1039
1040		/*
1041		 * Build the argument list for query_is_distinct_for_with_collations:
1042		 * a list of DistinctColInfo entries, each holding an output column
1043		 * number that the query needs to be distinct over, the equality
1044		 * operator that the column needs to be distinct according to, and
1045		 * that operator's input collation.  The collation matters because the
1046		 * subquery's own DISTINCT / GROUP BY / set-op proves uniqueness under
1047		 * its own collation, which need not agree with the operator's.
1048		 *
1049		 * (XXX we are not considering restriction clauses attached to the
1050		 * subquery; is that worth doing?)
1051		 */
1052		foreach(l, clause_list)
1053		{
1054			RestrictInfo *rinfo = lfirst_node(RestrictInfo, l);
1055			OpExpr	   *opexpr;
1056			Var		   *var;
1057			DistinctColInfo *dcinfo;
1058
1059			/*
1060			 * The caller's mergejoinability test should have selected only
1061			 * OpExprs.  The operator might be a cross-type operator and thus
1062			 * not exactly the same operator the subquery would consider;
1063			 * that's all right since query_is_distinct_for_with_collations
1064			 * can resolve such cases.
1065			 */
1066			opexpr = castNode(OpExpr, rinfo->clause);
1067
1068			/* caller identified the inner side for us */
1069			if (rinfo->outer_is_left)
1070				var = (Var *) get_rightop(rinfo->clause);
1071			else
1072				var = (Var *) get_leftop(rinfo->clause);
1073
1074			/*
1075			 * We may ignore any RelabelType node above the operand.  (There
1076			 * won't be more than one, since eval_const_expressions() has been
1077			 * applied already.)
1078			 */
1079			if (var && IsA(var, RelabelType))
1080				var = (Var *) ((RelabelType *) var)->arg;
1081
1082			/*
1083			 * If inner side isn't a Var referencing a subquery output column,
1084			 * this clause doesn't help us.
1085			 */
1086			if (!var || !IsA(var, Var) ||
1087				var->varno != relid || var->varlevelsup != 0)
1088				continue;
1089
1090			dcinfo = palloc(sizeof(DistinctColInfo));
1091			dcinfo->colno = var->varattno;
1092			dcinfo->opid = opexpr->opno;
1093			dcinfo->collid = opexpr->inputcollid;
1094			distinct_cols = lappend(distinct_cols, dcinfo);
1095		}
1096
1097		if (query_is_distinct_for_with_collations(subquery, distinct_cols))
1098			return true;
1099	}
1100	return false;
1101}

1104/*
1105 * query_supports_distinctness - could the query possibly be proven distinct
1106 *		on some set of output columns?
1107 *
1108 * This is effectively a pre-checking function for query_is_distinct_for().
1109 * It must return true if query_is_distinct_for() could possibly return true
1110 * with this query, but it should not expend a lot of cycles.  The idea is
1111 * that callers can avoid doing possibly-expensive processing to compute
1112 * query_is_distinct_for()'s argument lists if the call could not possibly
1113 * succeed.

1115bool
1116query_supports_distinctness(Query *query)
1117{
1118	/* SRFs break distinctness except with DISTINCT, see below */
1119	if (query->hasTargetSRFs && query->distinctClause == NIL)
1120		return false;
1121
1122	/* check for features we can prove distinctness with */
1123	if (query->distinctClause != NIL ||
1124		query->groupClause != NIL ||
1125		query->groupingSets != NIL ||
1126		query->hasAggs ||
1127		query->havingQual ||
1128		query->setOperations)
1129		return true;
1130
1131	return false;
1132}

1136 *		query_is_distinct_for_with_collations().
1137 *

1166/*
1167 * query_is_distinct_for_with_collations - does query never return duplicates
1168 *		of the specified columns?
1169 *
1170 * query is a not-yet-planned subquery (in current usage, it's always from
1171 * a subquery RTE, which the planner avoids scribbling on).
1172 *
1173 * distinct_cols is a list of DistinctColInfo, one per requested output column.
1174 * Each entry names the subquery output column number we want distinct, the
1175 * upper-level equality operator we'll compare values with, and that operator's
1176 * input collation.  We are interested in whether rows consisting of just these
1177 * columns are certain to be distinct.
1178 *
1179 * "Distinctness" is defined according to whether the corresponding upper-level
1180 * equality operators would think the values are distinct.  (Note: each opid
1181 * could be a cross-type operator, and thus not exactly the equality operator
1182 * that the subquery would use itself.  We use equality_ops_are_compatible() to
1183 * check compatibility.  That looks at opfamily membership for index AMs that
1184 * have declared that they support consistent equality semantics within an
1185 * opfamily, and so should give trustworthy answers for all operators that we
1186 * might need to deal with here.)
1187 *
1188 * The collid must also agree on equality with the collation the subquery's own
1189 * DISTINCT/GROUP BY/set-op uses to deduplicate the column, else the subquery's
1190 * distinctness does not carry over to the caller's equality semantics.  Two
1191 * collations agree on equality if they match or if both are deterministic (in
1192 * which case both reduce equality to byte-equality; see CREATE COLLATION).
1193 */
1194static bool
1195query_is_distinct_for_with_collations(Query *query, List *distinct_cols)
1196{
1197	ListCell   *l;
1198	DistinctColInfo *dcinfo;
1199
1200	/*
1201	 * DISTINCT (including DISTINCT ON) guarantees uniqueness if all the
1202	 * columns in the DISTINCT clause appear in colnos and operator semantics
1203	 * match.  This is true even if there are SRFs in the DISTINCT columns or
1204	 * elsewhere in the tlist.
1205	 */
1206	if (query->distinctClause)
1207	{
1208		foreach(l, query->distinctClause)
1209		{
1210			SortGroupClause *sgc = (SortGroupClause *) lfirst(l);
1211			TargetEntry *tle = get_sortgroupclause_tle(sgc,
1212													   query->targetList);
1213
1214			dcinfo = distinct_col_search(tle->resno, distinct_cols);
1215			if (dcinfo == NULL ||
1216				!equality_ops_are_compatible(dcinfo->opid, sgc->eqop) ||
1217				!collations_agree_on_equality(dcinfo->collid,
1218											  exprCollation((Node *) tle->expr)))
1219				break;			/* exit early if no match */
1220		}
1221		if (l == NULL)			/* had matches for all? */
1222			return true;
1223	}
1224
1225	/*
1226	 * Otherwise, a set-returning function in the query's targetlist can
1227	 * result in returning duplicate rows, despite any grouping that might
1228	 * occur before tlist evaluation.  (If all tlist SRFs are within GROUP BY
1229	 * columns, it would be safe because they'd be expanded before grouping.
1230	 * But it doesn't currently seem worth the effort to check for that.)
1231	 */
1232	if (query->hasTargetSRFs)
1233		return false;
1234
1235	/*
1236	 * Similarly, GROUP BY without GROUPING SETS guarantees uniqueness if all
1237	 * the grouped columns appear in colnos and operator semantics match.
1238	 */
1239	if (query->groupClause && !query->groupingSets)
1240	{
1241		foreach(l, query->groupClause)
1242		{
1243			SortGroupClause *sgc = (SortGroupClause *) lfirst(l);
1244			TargetEntry *tle = get_sortgroupclause_tle(sgc,
1245													   query->targetList);
1246
1247			dcinfo = distinct_col_search(tle->resno, distinct_cols);
1248			if (dcinfo == NULL ||
1249				!equality_ops_are_compatible(dcinfo->opid, sgc->eqop) ||
1250				!collations_agree_on_equality(dcinfo->collid,
1251											  exprCollation((Node *) tle->expr)))
1252				break;			/* exit early if no match */
1253		}
1254		if (l == NULL)			/* had matches for all? */
1255			return true;
1256	}
1257	else if (query->groupingSets)
1258	{
1259		/*
1260		 * If we have grouping sets with expressions, we probably don't have
1261		 * uniqueness and analysis would be hard. Punt.
1262		 */
1263		if (query->groupClause)
1264			return false;
1265
1266		/*
1267		 * If we have no groupClause (therefore no grouping expressions), we
1268		 * might have one or many empty grouping sets. If there's just one,
1269		 * then we're returning only one row and are certainly unique. But
1270		 * otherwise, we know we're certainly not unique.
1271		 */
1272		if (list_length(query->groupingSets) == 1 &&
1273			((GroupingSet *) linitial(query->groupingSets))->kind == GROUPING_SET_EMPTY)
1274			return true;
1275		else
1276			return false;
1277	}
1278	else
1279	{
1280		/*
1281		 * If we have no GROUP BY, but do have aggregates or HAVING, then the
1282		 * result is at most one row so it's surely unique, for any operators.
1283		 */
1284		if (query->hasAggs || query->havingQual)
1285			return true;
1286	}
1287
1288	/*
1289	 * UNION, INTERSECT, EXCEPT guarantee uniqueness of the whole output row,
1290	 * except with ALL.
1291	 */
1292	if (query->setOperations)
1293	{
1294		SetOperationStmt *topop = castNode(SetOperationStmt, query->setOperations);
1295
1296		Assert(topop->op != SETOP_NONE);
1297
1298		if (!topop->all)
1299		{
1300			ListCell   *lg;
1301
1302			/* We're good if all the nonjunk output columns are in colnos */
1303			lg = list_head(topop->groupClauses);
1304			foreach(l, query->targetList)
1305			{
1306				TargetEntry *tle = (TargetEntry *) lfirst(l);
1307				SortGroupClause *sgc;
1308
1309				if (tle->resjunk)
1310					continue;	/* ignore resjunk columns */
1311
1312				/* non-resjunk columns should have grouping clauses */
1313				Assert(lg != NULL);
1314				sgc = (SortGroupClause *) lfirst(lg);
1315				lg = lnext(topop->groupClauses, lg);
1316
1317				dcinfo = distinct_col_search(tle->resno, distinct_cols);
1318				if (dcinfo == NULL ||
1319					!equality_ops_are_compatible(dcinfo->opid, sgc->eqop) ||
1320					!collations_agree_on_equality(dcinfo->collid,
1321												  exprCollation((Node *) tle->expr)))
1322					break;		/* exit early if no match */
1323			}
1324			if (l == NULL)		/* had matches for all? */
1325				return true;
1326		}
1327	}
1328
1329	/*
1330	 * XXX Are there any other cases in which we can easily see the result
1331	 * must be distinct?
1332	 *
1333	 * If you do add more smarts to this function, be sure to update
1334	 * query_supports_distinctness() to match.
1335	 */
1336
1337	return false;
1338}

1341 * distinct_col_search - subroutine for query_is_distinct_for_with_collations
1342 *
1343 * If colno matches the colno field of an entry in distinct_cols, return a
1344 * pointer to that entry; else return NULL.  (Ordinarily distinct_cols would
1345 * not contain duplicate colnos, but if it does, we arbitrarily select the

1361/*
1362 * innerrel_is_unique
1363 *	  Check if the innerrel provably contains at most one tuple matching any
1364 *	  tuple from the outerrel, based on join clauses in the 'restrictlist'.
1365 *
1366 * We need an actual RelOptInfo for the innerrel, but it's sufficient to
1367 * identify the outerrel by its Relids.  This asymmetry supports use of this
1368 * function before joinrels have been built.  (The caller is expected to
1369 * also supply the joinrelids, just to save recalculating that.)
1370 *
1371 * The proof must be made based only on clauses that will be "joinquals"
1372 * rather than "otherquals" at execution.  For an inner join there's no
1373 * difference; but if the join is outer, we must ignore pushed-down quals,
1374 * as those will become "otherquals".  Note that this means the answer might
1375 * vary depending on whether IS_OUTER_JOIN(jointype); since we cache the
1376 * answer without regard to that, callers must take care not to call this
1377 * with jointypes that would be classified differently by IS_OUTER_JOIN().
1378 *
1379 * The actual proof is undertaken by is_innerrel_unique_for(); this function
1380 * is a frontend that is mainly concerned with caching the answers.
1381 * In particular, the force_cache argument allows overriding the internal
1382 * heuristic about whether to cache negative answers; it should be "true"
1383 * if making an inquiry that is not part of the normal bottom-up join search
1384 * sequence.
1385 */
1386bool
1387innerrel_is_unique(PlannerInfo *root,
1388				   Relids joinrelids,
1389				   Relids outerrelids,
1390				   RelOptInfo *innerrel,
1391				   JoinType jointype,
1392				   List *restrictlist,
1393				   bool force_cache)
1394{
1395	return innerrel_is_unique_ext(root, joinrelids, outerrelids, innerrel,
1396								  jointype, restrictlist, force_cache, NULL);
1397}

1399/*
1400 * innerrel_is_unique_ext
1401 *	  Do the same as innerrel_is_unique(), but also set to (*extra_clauses)
1402 *	  additional clauses from a baserestrictinfo list used to prove the
1403 *	  uniqueness.
1404 *
1405 * A non-NULL extra_clauses indicates that we're checking for self-join and
1406 * correspondingly dealing with filtered clauses.
1407 */
1408bool
1409innerrel_is_unique_ext(PlannerInfo *root,
1410					   Relids joinrelids,
1411					   Relids outerrelids,
1412					   RelOptInfo *innerrel,
1413					   JoinType jointype,
1414					   List *restrictlist,
1415					   bool force_cache,
1416					   List **extra_clauses)
1417{
1418	MemoryContext old_context;
1419	ListCell   *lc;
1420	UniqueRelInfo *uniqueRelInfo;
1421	List	   *outer_exprs = NIL;
1422	bool		self_join = (extra_clauses != NULL);
1423
1424	/* Certainly can't prove uniqueness when there are no joinclauses */
1425	if (restrictlist == NIL)
1426		return false;
1427
1428	/*
1429	 * Make a quick check to eliminate cases in which we will surely be unable
1430	 * to prove uniqueness of the innerrel.
1431	 */
1432	if (!rel_supports_distinctness(root, innerrel))
1433		return false;
1434
1435	/*
1436	 * Query the cache to see if we've managed to prove that innerrel is
1437	 * unique for any subset of this outerrel.  For non-self-join search, we
1438	 * don't need an exact match, as extra outerrels can't make the innerrel
1439	 * any less unique (or more formally, the restrictlist for a join to a
1440	 * superset outerrel must be a superset of the conditions we successfully
1441	 * used before). For self-join search, we require an exact match of
1442	 * outerrels because we need extra clauses to be valid for our case. Also,
1443	 * for self-join checking we've filtered the clauses list.  Thus, we can
1444	 * match only the result cached for a self-join search for another
1445	 * self-join check.
1446	 */
1447	foreach(lc, innerrel->unique_for_rels)
1448	{
1449		uniqueRelInfo = (UniqueRelInfo *) lfirst(lc);
1450
1451		if ((!self_join && bms_is_subset(uniqueRelInfo->outerrelids, outerrelids)) ||
1452			(self_join && bms_equal(uniqueRelInfo->outerrelids, outerrelids) &&
1453			 uniqueRelInfo->self_join))
1454		{
1455			if (extra_clauses)
1456				*extra_clauses = uniqueRelInfo->extra_clauses;
1457			return true;		/* Success! */
1458		}
1459	}
1460
1461	/*
1462	 * Conversely, we may have already determined that this outerrel, or some
1463	 * superset thereof, cannot prove this innerrel to be unique.
1464	 */
1465	foreach(lc, innerrel->non_unique_for_rels)
1466	{
1467		Relids		unique_for_rels = (Relids) lfirst(lc);
1468
1469		if (bms_is_subset(outerrelids, unique_for_rels))
1470			return false;
1471	}
1472
1473	/* No cached information, so try to make the proof. */
1474	if (is_innerrel_unique_for(root, joinrelids, outerrelids, innerrel,
1475							   jointype, restrictlist,
1476							   self_join ? &outer_exprs : NULL))
1477	{
1478		/*
1479		 * Cache the positive result for future probes, being sure to keep it
1480		 * in the planner_cxt even if we are working in GEQO.
1481		 *
1482		 * Note: one might consider trying to isolate the minimal subset of
1483		 * the outerrels that proved the innerrel unique.  But it's not worth
1484		 * the trouble, because the planner builds up joinrels incrementally
1485		 * and so we'll see the minimally sufficient outerrels before any
1486		 * supersets of them anyway.
1487		 */
1488		old_context = MemoryContextSwitchTo(root->planner_cxt);
1489		uniqueRelInfo = makeNode(UniqueRelInfo);
1490		uniqueRelInfo->outerrelids = bms_copy(outerrelids);
1491		uniqueRelInfo->self_join = self_join;
1492		uniqueRelInfo->extra_clauses = outer_exprs;
1493		innerrel->unique_for_rels = lappend(innerrel->unique_for_rels,
1494											uniqueRelInfo);
1495		MemoryContextSwitchTo(old_context);
1496
1497		if (extra_clauses)
1498			*extra_clauses = outer_exprs;
1499		return true;			/* Success! */
1500	}
1501	else
1502	{
1503		/*
1504		 * None of the join conditions for outerrel proved innerrel unique, so
1505		 * we can safely reject this outerrel or any subset of it in future
1506		 * checks.
1507		 *
1508		 * However, in normal planning mode, caching this knowledge is totally
1509		 * pointless; it won't be queried again, because we build up joinrels
1510		 * from smaller to larger.  It is useful in GEQO mode, where the
1511		 * knowledge can be carried across successive planning attempts; and
1512		 * it's likely to be useful when using join-search plugins, too. Hence
1513		 * cache when join_search_private is non-NULL.  (Yeah, that's a hack,
1514		 * but it seems reasonable.)
1515		 *
1516		 * Also, allow callers to override that heuristic and force caching;
1517		 * that's useful for reduce_unique_semijoins, which calls here before
1518		 * the normal join search starts.
1519		 */
1520		if (force_cache || root->join_search_private)
1521		{
1522			old_context = MemoryContextSwitchTo(root->planner_cxt);
1523			innerrel->non_unique_for_rels =
1524				lappend(innerrel->non_unique_for_rels,
1525						bms_copy(outerrelids));
1526			MemoryContextSwitchTo(old_context);
1527		}
1528
1529		return false;
1530	}
1531}

1533/*
1534 * is_innerrel_unique_for
1535 *	  Check if the innerrel provably contains at most one tuple matching any
1536 *	  tuple from the outerrel, based on join clauses in the 'restrictlist'.
1537 */
1538static bool
1539is_innerrel_unique_for(PlannerInfo *root,
1540					   Relids joinrelids,
1541					   Relids outerrelids,
1542					   RelOptInfo *innerrel,
1543					   JoinType jointype,
1544					   List *restrictlist,
1545					   List **extra_clauses)
1546{
1547	List	   *clause_list = NIL;
1548	ListCell   *lc;
1549
1550	/*
1551	 * Search for mergejoinable clauses that constrain the inner rel against
1552	 * the outer rel.  If an operator is mergejoinable then it behaves like
1553	 * equality for some btree opclass, so it's what we want.  The
1554	 * mergejoinability test also eliminates clauses containing volatile
1555	 * functions, which we couldn't depend on.
1556	 */
1557	foreach(lc, restrictlist)
1558	{
1559		RestrictInfo *restrictinfo = (RestrictInfo *) lfirst(lc);
1560
1561		/*
1562		 * As noted above, if it's a pushed-down clause and we're at an outer
1563		 * join, we can't use it.
1564		 */
1565		if (IS_OUTER_JOIN(jointype) &&
1566			RINFO_IS_PUSHED_DOWN(restrictinfo, joinrelids))
1567			continue;
1568
1569		/* Ignore if it's not a mergejoinable clause */
1570		if (!restrictinfo->can_join ||
1571			restrictinfo->mergeopfamilies == NIL)
1572			continue;			/* not mergejoinable */
1573
1574		/*
1575		 * Check if the clause has the form "outer op inner" or "inner op
1576		 * outer", and if so mark which side is inner.
1577		 */
1578		if (!clause_sides_match_join(restrictinfo, outerrelids,
1579									 innerrel->relids))
1580			continue;			/* no good for these input relations */
1581
1582		/* OK, add to the list */
1583		clause_list = lappend(clause_list, restrictinfo);
1584	}
1585
1586	/* Let rel_is_distinct_for() do the hard work */
1587	return rel_is_distinct_for(root, innerrel, clause_list, extra_clauses);
1588}

1590/*
1591 * Update EC members to point to the remaining relation instead of the removed
1592 * one, removing duplicates.
1593 *
1594 * Restriction clauses for base relations are already distributed to
1595 * the respective baserestrictinfo lists (see
1596 * generate_implied_equalities_for_column). The above code has already processed
1597 * this list and updated these clauses to reference the remaining
1598 * relation, so that we can skip them here based on their relids.
1599 *
1600 * Likewise, we have already processed the join clauses that join the
1601 * removed relation to the remaining one.
1602 *
1603 * Finally, there might be join clauses tying the removed relation to
1604 * some third relation.  We can't just delete the source clauses and
1605 * regenerate them from the EC because the corresponding equality
1606 * operators might be missing (see the handling of ec_broken).
1607 * Therefore, we will update the references in the source clauses.
1608 *
1609 * Derived clauses can be generated again, so it is simpler just to
1610 * delete them.
1611 */
1612static void
1613update_eclasses(EquivalenceClass *ec, int from, int to)
1614{
1615	List	   *new_members = NIL;
1616	List	   *new_sources = NIL;
1617
1618	/*
1619	 * We don't expect any EC child members to exist at this point.  Ensure
1620	 * that's the case, otherwise, we might be getting asked to do something
1621	 * this function hasn't been coded for.
1622	 */
1623	Assert(ec->ec_childmembers == NULL);
1624
1625	foreach_node(EquivalenceMember, em, ec->ec_members)
1626	{
1627		bool		is_redundant = false;
1628
1629		if (!bms_is_member(from, em->em_relids))
1630		{
1631			new_members = lappend(new_members, em);
1632			continue;
1633		}
1634
1635		em->em_relids = adjust_relid_set(em->em_relids, from, to);
1636		em->em_jdomain->jd_relids = adjust_relid_set(em->em_jdomain->jd_relids, from, to);
1637
1638		/* We only process inner joins */
1639		ChangeVarNodesExtended((Node *) em->em_expr, from, to, 0,
1640							   replace_relid_callback);
1641
1642		foreach_node(EquivalenceMember, other, new_members)
1643		{
1644			if (!equal(em->em_relids, other->em_relids))
1645				continue;
1646
1647			if (equal(em->em_expr, other->em_expr))
1648			{
1649				is_redundant = true;
1650				break;
1651			}
1652		}
1653
1654		if (!is_redundant)
1655			new_members = lappend(new_members, em);
1656	}
1657
1658	list_free(ec->ec_members);
1659	ec->ec_members = new_members;
1660
1661	ec_clear_derived_clauses(ec);
1662
1663	/* Update EC source expressions */
1664	foreach_node(RestrictInfo, rinfo, ec->ec_sources)
1665	{
1666		bool		is_redundant = false;
1667
1668		if (!bms_is_member(from, rinfo->required_relids))
1669		{
1670			new_sources = lappend(new_sources, rinfo);
1671			continue;
1672		}
1673
1674		ChangeVarNodesExtended((Node *) rinfo, from, to, 0,
1675							   replace_relid_callback);
1676
1677		/*
1678		 * After switching the clause to the remaining relation, check it for
1679		 * redundancy with existing ones. We don't have to check for
1680		 * redundancy with derived clauses, because we've just deleted them.
1681		 */
1682		foreach_node(RestrictInfo, other, new_sources)
1683		{
1684			if (!equal(rinfo->clause_relids, other->clause_relids))
1685				continue;
1686
1687			if (equal(rinfo->clause, other->clause))
1688			{
1689				is_redundant = true;
1690				break;
1691			}
1692		}
1693
1694		if (!is_redundant)
1695			new_sources = lappend(new_sources, rinfo);
1696	}
1697
1698	list_free(ec->ec_sources);
1699	ec->ec_sources = new_sources;
1700	ec->ec_relids = adjust_relid_set(ec->ec_relids, from, to);
1701}

1704 * "Logically" compares two RestrictInfo's ignoring the 'rinfo_serial' field,
1705 * which makes almost every RestrictInfo unique.  This type of comparison is
1706 * useful when removing duplicates while moving RestrictInfo's from removed
1707 * relation to remaining relation during self-join elimination.
1708 *
1709 * XXX: In the future, we might remove the 'rinfo_serial' field completely and
1710 * get rid of this function.
1711 */
1712static bool
1713restrict_infos_logically_equal(RestrictInfo *a, RestrictInfo *b)
1714{
1715	int			saved_rinfo_serial = a->rinfo_serial;
1716	bool		result;
1717
1718	a->rinfo_serial = b->rinfo_serial;
1719	result = equal(a, b);
1720	a->rinfo_serial = saved_rinfo_serial;
1721

1725/*
1726 * This function adds all non-redundant clauses to the keeping relation
1727 * during self-join elimination.  That is a contradictory operation. On the
1728 * one hand, we reduce the length of the `restrict` lists, which can
1729 * impact planning or executing time.  Additionally, we improve the
1730 * accuracy of cardinality estimation.  On the other hand, it is one more
1731 * place that can make planning time much longer in specific cases.  It
1732 * would have been better to avoid calling the equal() function here, but
1733 * it's the only way to detect duplicated inequality expressions.
1734 *
1735 * (*keep_rinfo_list) is given by pointer because it might be altered by
1736 * distribute_restrictinfo_to_rels().
1737 */
1738static void
1739add_non_redundant_clauses(PlannerInfo *root,
1740						  List *rinfo_candidates,
1741						  List **keep_rinfo_list,
1742						  Index removed_relid)
1743{
1744	foreach_node(RestrictInfo, rinfo, rinfo_candidates)
1745	{
1746		bool		is_redundant = false;
1747
1748		Assert(!bms_is_member(removed_relid, rinfo->required_relids));
1749
1750		foreach_node(RestrictInfo, src, (*keep_rinfo_list))
1751		{
1752			if (!bms_equal(src->clause_relids, rinfo->clause_relids))
1753				/* Can't compare trivially different clauses */
1754				continue;
1755
1756			if (src == rinfo ||
1757				(rinfo->parent_ec != NULL &&
1758				 src->parent_ec == rinfo->parent_ec) ||
1759				restrict_infos_logically_equal(rinfo, src))
1760			{
1761				is_redundant = true;
1762				break;
1763			}
1764		}
1765		if (!is_redundant)
1766			distribute_restrictinfo_to_rels(root, rinfo);
1767	}
1768}

1770/*
1771 * A custom callback for ChangeVarNodesExtended() providing Self-join
1772 * elimination (SJE) related functionality
1773 *
1774 * SJE needs to skip the RangeTblRef node type.  During SJE's last
1775 * step, remove_rel_from_joinlist() removes remaining RangeTblRefs
1776 * with target relid.  If ChangeVarNodes() replaces the target relid
1777 * before, remove_rel_from_joinlist() would fail to identify the nodes
1778 * to delete.
1779 *
1780 * SJE also needs to change the relids within RestrictInfo's.
1781 */
1782static bool
1783replace_relid_callback(Node *node, ChangeVarNodes_context *context)
1784{
1785	if (IsA(node, RangeTblRef))
1786	{
1787		return true;
1788	}
1789	else if (IsA(node, RestrictInfo))
1790	{
1791		RestrictInfo *rinfo = (RestrictInfo *) node;
1792		int			relid = -1;
1793		bool		is_req_equal =
1794			(rinfo->required_relids == rinfo->clause_relids);
1795		bool		clause_relids_is_multiple =
1796			(bms_membership(rinfo->clause_relids) == BMS_MULTIPLE);
1797
1798		/*
1799		 * Recurse down into clauses if the target relation is present in
1800		 * clause_relids or required_relids.  We must check required_relids
1801		 * because the relation not present in clause_relids might still be
1802		 * present somewhere in orclause.
1803		 */
1804		if (bms_is_member(context->rt_index, rinfo->clause_relids) ||
1805			bms_is_member(context->rt_index, rinfo->required_relids))
1806		{
1807			Relids		new_clause_relids;
1808
1809			ChangeVarNodesWalkExpression((Node *) rinfo->clause, context);
1810			ChangeVarNodesWalkExpression((Node *) rinfo->orclause, context);
1811
1812			new_clause_relids = adjust_relid_set(rinfo->clause_relids,
1813												 context->rt_index,
1814												 context->new_index);
1815
1816			/*
1817			 * Incrementally adjust num_base_rels based on the change of
1818			 * clause_relids, which could contain both base relids and
1819			 * outer-join relids.  This operation is legal until we remove
1820			 * only baserels.
1821			 */
1822			rinfo->num_base_rels -= bms_num_members(rinfo->clause_relids) -
1823				bms_num_members(new_clause_relids);
1824
1825			rinfo->clause_relids = new_clause_relids;
1826			rinfo->left_relids =
1827				adjust_relid_set(rinfo->left_relids, context->rt_index, context->new_index);
1828			rinfo->right_relids =
1829				adjust_relid_set(rinfo->right_relids, context->rt_index, context->new_index);
1830		}
1831
1832		if (is_req_equal)
1833			rinfo->required_relids = rinfo->clause_relids;
1834		else
1835			rinfo->required_relids =
1836				adjust_relid_set(rinfo->required_relids, context->rt_index, context->new_index);
1837
1838		rinfo->outer_relids =
1839			adjust_relid_set(rinfo->outer_relids, context->rt_index, context->new_index);
1840		rinfo->incompatible_relids =
1841			adjust_relid_set(rinfo->incompatible_relids, context->rt_index, context->new_index);
1842
1843		if (rinfo->mergeopfamilies &&
1844			bms_get_singleton_member(rinfo->clause_relids, &relid) &&
1845			clause_relids_is_multiple &&
1846			relid == context->new_index && IsA(rinfo->clause, OpExpr))
1847		{
1848			Expr	   *leftOp;
1849			Expr	   *rightOp;
1850
1851			leftOp = (Expr *) get_leftop(rinfo->clause);
1852			rightOp = (Expr *) get_rightop(rinfo->clause);
1853
1854			/*
1855			 * For self-join elimination, changing varnos could transform
1856			 * "t1.a = t2.a" into "t1.a = t1.a".  That is always true as long
1857			 * as "t1.a" is not null.  We use equal() to check for such a
1858			 * case, and then we replace the qual with a check for not null
1859			 * (NullTest).
1860			 */
1861			if (leftOp != NULL && equal(leftOp, rightOp))
1862			{
1863				NullTest   *ntest = makeNode(NullTest);
1864
1865				ntest->arg = leftOp;
1866				ntest->nulltesttype = IS_NOT_NULL;
1867				ntest->argisrow = false;
1868				ntest->location = -1;
1869				rinfo->clause = (Expr *) ntest;
1870				rinfo->mergeopfamilies = NIL;
1871				rinfo->left_em = NULL;
1872				rinfo->right_em = NULL;
1873			}
1874			Assert(rinfo->orclause == NULL);
1875		}
1876		return true;
1877	}
1878
1879	return false;
1880}

1882/*
1883 * Remove a relation after we have proven that it participates only in an
1884 * unneeded unique self-join.
1885 *
1886 * Replace any links in planner info structures.
1887 *
1888 * Transfer join and restriction clauses from the removed relation to the
1889 * remaining one. We change the Vars of the clause to point to the
1890 * remaining relation instead of the removed one. The clauses that require
1891 * a subset of joinrelids become restriction clauses of the remaining
1892 * relation, and others remain join clauses. We append them to
1893 * baserestrictinfo and joininfo, respectively, trying not to introduce
1894 * duplicates.
1895 *
1896 * We also have to process the 'joinclauses' list here, because it
1897 * contains EC-derived join clauses which must become filter clauses. It
1898 * is not enough to just correct the ECs because the EC-derived
1899 * restrictions are generated before join removal (see
1900 * generate_base_implied_equalities).
1901 *
1902 * NOTE: Remember to keep the code in sync with PlannerInfo to be sure all
1903 * cached relids and relid bitmapsets can be correctly cleaned during the
1904 * self-join elimination procedure.
1905 */
1906static void
1907remove_self_join_rel(PlannerInfo *root, PlanRowMark *kmark, PlanRowMark *rmark,
1908					 RelOptInfo *toKeep, RelOptInfo *toRemove,
1909					 List *restrictlist)
1910{
1911	List	   *joininfos;
1912	ListCell   *lc;
1913	int			i;
1914	List	   *jinfo_candidates = NIL;
1915	List	   *binfo_candidates = NIL;
1916
1917	Assert(toKeep->relid > 0);
1918	Assert(toRemove->relid > 0);
1919
1920	/*
1921	 * Replace the index of the removing table with the keeping one. The
1922	 * technique of removing/distributing restrictinfo is used here to attach
1923	 * just appeared (for keeping relation) join clauses and avoid adding
1924	 * duplicates of those that already exist in the joininfo list.
1925	 */
1926	joininfos = list_copy(toRemove->joininfo);
1927	foreach_node(RestrictInfo, rinfo, joininfos)
1928	{
1929		remove_join_clause_from_rels(root, rinfo, rinfo->required_relids);
1930		ChangeVarNodesExtended((Node *) rinfo, toRemove->relid, toKeep->relid,
1931							   0, replace_relid_callback);
1932
1933		if (bms_membership(rinfo->required_relids) == BMS_MULTIPLE)
1934			jinfo_candidates = lappend(jinfo_candidates, rinfo);
1935		else
1936			binfo_candidates = lappend(binfo_candidates, rinfo);
1937	}
1938
1939	/*
1940	 * Concatenate restrictlist to the list of base restrictions of the
1941	 * removing table just to simplify the replacement procedure: all of them
1942	 * weren't connected to any keeping relations and need to be added to some
1943	 * rels.
1944	 */
1945	toRemove->baserestrictinfo = list_concat(toRemove->baserestrictinfo,
1946											 restrictlist);
1947	foreach_node(RestrictInfo, rinfo, toRemove->baserestrictinfo)
1948	{
1949		ChangeVarNodesExtended((Node *) rinfo, toRemove->relid, toKeep->relid,
1950							   0, replace_relid_callback);
1951
1952		if (bms_membership(rinfo->required_relids) == BMS_MULTIPLE)
1953			jinfo_candidates = lappend(jinfo_candidates, rinfo);
1954		else
1955			binfo_candidates = lappend(binfo_candidates, rinfo);
1956	}
1957
1958	/*
1959	 * Now, add all non-redundant clauses to the keeping relation.
1960	 */
1961	add_non_redundant_clauses(root, binfo_candidates,
1962							  &toKeep->baserestrictinfo, toRemove->relid);
1963	add_non_redundant_clauses(root, jinfo_candidates,
1964							  &toKeep->joininfo, toRemove->relid);
1965
1966	list_free(binfo_candidates);
1967	list_free(jinfo_candidates);
1968
1969	/*
1970	 * Arrange equivalence classes, mentioned removing a table, with the
1971	 * keeping one: varno of removing table should be replaced in members and
1972	 * sources lists. Also, remove duplicated elements if this replacement
1973	 * procedure created them.
1974	 */
1975	i = -1;
1976	while ((i = bms_next_member(toRemove->eclass_indexes, i)) >= 0)
1977	{
1978		EquivalenceClass *ec = (EquivalenceClass *) list_nth(root->eq_classes, i);
1979
1980		update_eclasses(ec, toRemove->relid, toKeep->relid);
1981		toKeep->eclass_indexes = bms_add_member(toKeep->eclass_indexes, i);
1982	}
1983
1984	/*
1985	 * Transfer the targetlist and attr_needed flags.
1986	 */
1987
1988	foreach(lc, toRemove->reltarget->exprs)
1989	{
1990		Node	   *node = lfirst(lc);
1991
1992		ChangeVarNodesExtended(node, toRemove->relid, toKeep->relid, 0,
1993							   replace_relid_callback);
1994		if (!list_member(toKeep->reltarget->exprs, node))
1995			toKeep->reltarget->exprs = lappend(toKeep->reltarget->exprs, node);
1996	}
1997
1998	for (i = toKeep->min_attr; i <= toKeep->max_attr; i++)
1999	{
2000		int			attno = i - toKeep->min_attr;
2001
2002		toRemove->attr_needed[attno] = adjust_relid_set(toRemove->attr_needed[attno],
2003														toRemove->relid, toKeep->relid);
2004		toKeep->attr_needed[attno] = bms_add_members(toKeep->attr_needed[attno],
2005													 toRemove->attr_needed[attno]);
2006	}
2007
2008	/*
2009	 * If the removed relation has a row mark, transfer it to the remaining
2010	 * one.
2011	 *
2012	 * If both rels have row marks, just keep the one corresponding to the
2013	 * remaining relation because we verified earlier that they have the same
2014	 * strength.
2015	 */
2016	if (rmark)
2017	{
2018		if (kmark)
2019		{
2020			Assert(kmark->markType == rmark->markType);
2021
2022			root->rowMarks = list_delete_ptr(root->rowMarks, rmark);
2023		}
2024		else
2025		{
2026			/* Shouldn't have inheritance children here. */
2027			Assert(rmark->rti == rmark->prti);
2028
2029			rmark->rti = rmark->prti = toKeep->relid;
2030		}
2031	}
2032
2033	/*
2034	 * Replace varno in all the query structures, except nodes RangeTblRef
2035	 * otherwise later remove_rel_from_joinlist will yield errors.
2036	 */
2037	ChangeVarNodesExtended((Node *) root->parse, toRemove->relid, toKeep->relid,
2038						   0, replace_relid_callback);
2039
2040	/* Replace links in the planner info */
2041	remove_rel_from_query(root, toRemove, toKeep->relid, NULL, NULL);
2042
2043	/* At last, replace varno in root targetlist and HAVING clause */
2044	ChangeVarNodesExtended((Node *) root->processed_tlist, toRemove->relid,
2045						   toKeep->relid, 0, replace_relid_callback);
2046	ChangeVarNodesExtended((Node *) root->processed_groupClause,
2047						   toRemove->relid, toKeep->relid, 0,
2048						   replace_relid_callback);
2049
2050	adjust_relid_set(root->all_result_relids, toRemove->relid, toKeep->relid);
2051	adjust_relid_set(root->leaf_result_relids, toRemove->relid, toKeep->relid);
2052
2053	/*
2054	 * There may be references to the rel in root->fkey_list, but if so,
2055	 * match_foreign_keys_to_quals() will get rid of them.
2056	 */
2057
2058	/*
2059	 * Finally, remove the rel from the baserel array to prevent it from being
2060	 * referenced again.  (We can't do this earlier because
2061	 * remove_join_clause_from_rels will touch it.)
2062	 */
2063	root->simple_rel_array[toRemove->relid] = NULL;
2064	root->simple_rte_array[toRemove->relid] = NULL;
2065
2066	/* And nuke the RelOptInfo, just in case there's another access path. */
2067	pfree(toRemove);
2068
2069
2070	/*
2071	 * Now repeat construction of attr_needed bits coming from all other
2072	 * sources.
2073	 */
2074	rebuild_placeholder_attr_needed(root);
2075	rebuild_joinclause_attr_needed(root);
2076	rebuild_eclass_attr_needed(root);
2077	rebuild_lateral_attr_needed(root);
2078}

2080/*
2081 * split_selfjoin_quals
2082 *		Processes 'joinquals' by building two lists: one containing the quals
2083 *		where the columns/exprs are on either side of the join match and
2084 *		another one containing the remaining quals.
2085 *
2086 * 'joinquals' must only contain quals for a RTE_RELATION being joined to
2087 * itself.
2088 */
2089static void
2090split_selfjoin_quals(PlannerInfo *root, List *joinquals, List **selfjoinquals,
2091					 List **otherjoinquals, int from, int to)
2092{
2093	List	   *sjoinquals = NIL;
2094	List	   *ojoinquals = NIL;
2095
2096	foreach_node(RestrictInfo, rinfo, joinquals)
2097	{
2098		OpExpr	   *expr;
2099		Node	   *leftexpr;
2100		Node	   *rightexpr;
2101
2102		/* In general, clause looks like F(arg1) = G(arg2) */
2103		if (!rinfo->mergeopfamilies ||
2104			bms_num_members(rinfo->clause_relids) != 2 ||
2105			bms_membership(rinfo->left_relids) != BMS_SINGLETON ||
2106			bms_membership(rinfo->right_relids) != BMS_SINGLETON)
2107		{
2108			ojoinquals = lappend(ojoinquals, rinfo);
2109			continue;
2110		}
2111
2112		expr = (OpExpr *) rinfo->clause;
2113
2114		if (!IsA(expr, OpExpr) || list_length(expr->args) != 2)
2115		{
2116			ojoinquals = lappend(ojoinquals, rinfo);
2117			continue;
2118		}
2119
2120		leftexpr = get_leftop(rinfo->clause);
2121		rightexpr = copyObject(get_rightop(rinfo->clause));
2122
2123		if (leftexpr && IsA(leftexpr, RelabelType))
2124			leftexpr = (Node *) ((RelabelType *) leftexpr)->arg;
2125		if (rightexpr && IsA(rightexpr, RelabelType))
2126			rightexpr = (Node *) ((RelabelType *) rightexpr)->arg;
2127
2128		/*
2129		 * Quite an expensive operation, narrowing the use case. For example,
2130		 * when we have cast of the same var to different (but compatible)
2131		 * types.
2132		 */
2133		ChangeVarNodesExtended(rightexpr,
2134							   bms_singleton_member(rinfo->right_relids),
2135							   bms_singleton_member(rinfo->left_relids), 0,
2136							   replace_relid_callback);
2137
2138		if (equal(leftexpr, rightexpr))
2139			sjoinquals = lappend(sjoinquals, rinfo);
2140		else
2141			ojoinquals = lappend(ojoinquals, rinfo);
2142	}
2143
2144	*selfjoinquals = sjoinquals;
2145	*otherjoinquals = ojoinquals;
2146}

2148/*
2149 * Check for a case when uniqueness is at least partly derived from a
2150 * baserestrictinfo clause. In this case, we have a chance to return only
2151 * one row (if such clauses on both sides of SJ are equal) or nothing (if they
2152 * are different).
2153 */
2154static bool
2155match_unique_clauses(PlannerInfo *root, RelOptInfo *outer, List *uclauses,
2156					 Index relid)
2157{
2158	foreach_node(RestrictInfo, rinfo, uclauses)
2159	{
2160		Expr	   *clause;
2161		Node	   *iclause;
2162		Node	   *c1;
2163		bool		matched = false;
2164
2165		Assert(outer->relid > 0 && relid > 0);
2166
2167		/* Only filters like f(R.x1,...,R.xN) == expr we should consider. */
2168		Assert(bms_is_empty(rinfo->left_relids) ^
2169			   bms_is_empty(rinfo->right_relids));
2170
2171		clause = (Expr *) copyObject(rinfo->clause);
2172		ChangeVarNodesExtended((Node *) clause, relid, outer->relid, 0,
2173							   replace_relid_callback);
2174
2175		iclause = bms_is_empty(rinfo->left_relids) ? get_rightop(clause) :
2176			get_leftop(clause);
2177		c1 = bms_is_empty(rinfo->left_relids) ? get_leftop(clause) :
2178			get_rightop(clause);
2179
2180		/*
2181		 * Compare these left and right sides with the corresponding sides of
2182		 * the outer's filters. If no one is detected - return immediately.
2183		 */
2184		foreach_node(RestrictInfo, orinfo, outer->baserestrictinfo)
2185		{
2186			Node	   *oclause;
2187			Node	   *c2;
2188
2189			if (orinfo->mergeopfamilies == NIL)
2190				/* Don't consider clauses that aren't similar to 'F(X)=G(Y)' */
2191				continue;
2192
2193			Assert(is_opclause(orinfo->clause));
2194
2195			oclause = bms_is_empty(orinfo->left_relids) ?
2196				get_rightop(orinfo->clause) : get_leftop(orinfo->clause);
2197			c2 = (bms_is_empty(orinfo->left_relids) ?
2198				  get_leftop(orinfo->clause) : get_rightop(orinfo->clause));
2199
2200			if (equal(iclause, oclause) && equal(c1, c2))
2201			{
2202				matched = true;
2203				break;
2204			}
2205		}
2206
2207		if (!matched)
2208			return false;
2209	}
2210
2211	return true;
2212}

2214/*
2215 * Find and remove unique self-joins in a group of base relations that have
2216 * the same Oid.
2217 *
2218 * Returns a set of relids that were removed.
2219 */
2220static Relids
2221remove_self_joins_one_group(PlannerInfo *root, Relids relids)
2222{
2223	Relids		result = NULL;
2224	int			k;				/* Index of kept relation */
2225	int			r = -1;			/* Index of removed relation */
2226
2227	while ((r = bms_next_member(relids, r)) > 0)
2228	{
2229		RelOptInfo *rrel = root->simple_rel_array[r];
2230
2231		k = r;
2232
2233		while ((k = bms_next_member(relids, k)) > 0)
2234		{
2235			Relids		joinrelids = NULL;
2236			RelOptInfo *krel = root->simple_rel_array[k];
2237			List	   *restrictlist;
2238			List	   *selfjoinquals;
2239			List	   *otherjoinquals;
2240			ListCell   *lc;
2241			bool		jinfo_check = true;
2242			PlanRowMark *kmark = NULL;
2243			PlanRowMark *rmark = NULL;
2244			List	   *uclauses = NIL;
2245
2246			/* A sanity check: the relations have the same Oid. */
2247			Assert(root->simple_rte_array[k]->relid ==
2248				   root->simple_rte_array[r]->relid);
2249
2250			/*
2251			 * It is impossible to eliminate the join of two relations if they
2252			 * belong to different rules of order. Otherwise, the planner
2253			 * can't find any variants of the correct query plan.
2254			 */
2255			foreach(lc, root->join_info_list)
2256			{
2257				SpecialJoinInfo *info = (SpecialJoinInfo *) lfirst(lc);
2258
2259				if ((bms_is_member(k, info->syn_lefthand) ^
2260					 bms_is_member(r, info->syn_lefthand)) ||
2261					(bms_is_member(k, info->syn_righthand) ^
2262					 bms_is_member(r, info->syn_righthand)))
2263				{
2264					jinfo_check = false;
2265					break;
2266				}
2267			}
2268			if (!jinfo_check)
2269				continue;
2270
2271			/*
2272			 * Check Row Marks equivalence. We can't remove the join if the
2273			 * relations have row marks of different strength (e.g., one is
2274			 * locked FOR UPDATE, and another just has ROW_MARK_REFERENCE for
2275			 * EvalPlanQual rechecking).
2276			 */
2277			foreach(lc, root->rowMarks)
2278			{
2279				PlanRowMark *rowMark = (PlanRowMark *) lfirst(lc);
2280
2281				if (rowMark->rti == r)
2282				{
2283					Assert(rmark == NULL);
2284					rmark = rowMark;
2285				}
2286				else if (rowMark->rti == k)
2287				{
2288					Assert(kmark == NULL);
2289					kmark = rowMark;
2290				}
2291
2292				if (kmark && rmark)
2293					break;
2294			}
2295			if (kmark && rmark && kmark->markType != rmark->markType)
2296				continue;
2297
2298			/*
2299			 * We only deal with base rels here, so their relids bitset
2300			 * contains only one member -- their relid.
2301			 */
2302			joinrelids = bms_add_member(joinrelids, r);
2303			joinrelids = bms_add_member(joinrelids, k);
2304
2305			/*
2306			 * PHVs should not impose any constraints on removing self-joins.
2307			 */
2308
2309			/*
2310			 * At this stage, joininfo lists of inner and outer can contain
2311			 * only clauses required for a superior outer join that can't
2312			 * influence this optimization. So, we can avoid to call the
2313			 * build_joinrel_restrictlist() routine.
2314			 */
2315			restrictlist = generate_join_implied_equalities(root, joinrelids,
2316															rrel->relids,
2317															krel, NULL);

2319				continue;
2320
2321			/*
2322			 * Process restrictlist to separate the self-join quals from the
2323			 * other quals. e.g., "x = x" goes to selfjoinquals and "a = b" to
2324			 * otherjoinquals.
2325			 */

2327								 &otherjoinquals, rrel->relid, krel->relid);
2328
2329			Assert(list_length(restrictlist) ==
2330				   (list_length(selfjoinquals) + list_length(otherjoinquals)));
2331
2332			/*
2333			 * To enable SJE for the only degenerate case without any self
2334			 * join clauses at all, add baserestrictinfo to this list. The
2335			 * degenerate case works only if both sides have the same clause.
2336			 * So doesn't matter which side to add.
2337			 */
2338			selfjoinquals = list_concat(selfjoinquals, krel->baserestrictinfo);
2339

2341			 * Determine if the rrel can duplicate outer rows. We must bypass
2342			 * the unique rel cache here since we're possibly using a subset
2343			 * of join quals. We can use 'force_cache' == true when all join
2344			 * quals are self-join quals.  Otherwise, we could end up putting
2345			 * false negatives in the cache.
2346			 */

2350										&uclauses))

2352
2353			/*
2354			 * 'uclauses' is the copy of outer->baserestrictinfo that are
2355			 * associated with an index.  We proved by matching selfjoinquals
2356			 * to a unique index that the outer relation has at most one
2357			 * matching row for each inner row.  Sometimes that is not enough.
2358			 * e.g. "WHERE s1.b = s2.b AND s1.a = 1 AND s2.a = 2" when the
2359			 * unique index is (a,b).  Having non-empty uclauses, we must
2360			 * validate that the inner baserestrictinfo contains the same
2361			 * expressions, or we won't match the same row on each side of the
2362			 * join.
2363			 */
2364			if (!match_unique_clauses(root, rrel, uclauses, krel->relid))
2365				continue;
2366
2367			/*
2368			 * Remove rrel ReloptInfo from the planner structures and the
2369			 * corresponding row mark.
2370			 */
2371			remove_self_join_rel(root, kmark, rmark, krel, rrel, restrictlist);
2372
2373			result = bms_add_member(result, r);
2374
2375			/* We have removed the outer relation, try the next one. */
2376			break;
2377		}
2378	}
2379
2380	return result;
2381}

2383/*
2384 * Gather indexes of base relations from the joinlist and try to eliminate self
2385 * joins.
2386 */
2387static Relids
2388remove_self_joins_recurse(PlannerInfo *root, List *joinlist, Relids toRemove)
2389{
2390	ListCell   *jl;
2391	Relids		relids = NULL;
2392	SelfJoinCandidate *candidates = NULL;
2393	int			i;
2394	int			j;
2395	int			numRels;
2396
2397	/* Collect indexes of base relations of the join tree */
2398	foreach(jl, joinlist)
2399	{
2400		Node	   *jlnode = (Node *) lfirst(jl);
2401
2402		if (IsA(jlnode, RangeTblRef))
2403		{
2404			int			varno = ((RangeTblRef *) jlnode)->rtindex;
2405			RangeTblEntry *rte = root->simple_rte_array[varno];
2406
2407			/*
2408			 * We only consider ordinary relations as candidates to be
2409			 * removed, and these relations should not have TABLESAMPLE
2410			 * clauses specified.  Removing a relation with TABLESAMPLE clause
2411			 * could potentially change the syntax of the query. Because of
2412			 * UPDATE/DELETE EPQ mechanism, currently Query->resultRelation or
2413			 * Query->mergeTargetRelation associated rel cannot be eliminated.
2414			 */
2415			if (rte->rtekind == RTE_RELATION &&
2416				rte->relkind == RELKIND_RELATION &&
2417				rte->tablesample == NULL &&
2418				varno != root->parse->resultRelation &&
2419				varno != root->parse->mergeTargetRelation)
2420			{
2421				Assert(!bms_is_member(varno, relids));
2422				relids = bms_add_member(relids, varno);
2423			}
2424		}
2425		else if (IsA(jlnode, List))
2426		{
2427			/* Recursively go inside the sub-joinlist */
2428			toRemove = remove_self_joins_recurse(root, (List *) jlnode,
2429												 toRemove);
2430		}
2431		else
2432			elog(ERROR, "unrecognized joinlist node type: %d",
2433				 (int) nodeTag(jlnode));
2434	}
2435
2436	numRels = bms_num_members(relids);
2437
2438	/* Need at least two relations for the join */
2439	if (numRels < 2)
2440		return toRemove;
2441
2442	/*
2443	 * In order to find relations with the same oid we first build an array of
2444	 * candidates and then sort it by oid.
2445	 */
2446	candidates = (SelfJoinCandidate *) palloc(sizeof(SelfJoinCandidate) *
2447											  numRels);
2448	i = -1;
2449	j = 0;
2450	while ((i = bms_next_member(relids, i)) >= 0)
2451	{
2452		candidates[j].relid = i;
2453		candidates[j].reloid = root->simple_rte_array[i]->relid;
2454		j++;
2455	}
2456
2457	qsort(candidates, numRels, sizeof(SelfJoinCandidate),
2458		  self_join_candidates_cmp);
2459
2460	/*
2461	 * Iteratively form a group of relation indexes with the same oid and
2462	 * launch the routine that detects self-joins in this group and removes
2463	 * excessive range table entries.
2464	 *
2465	 * At the end of the iteration, exclude the group from the overall relids
2466	 * list. So each next iteration of the cycle will involve less and less
2467	 * value of relids.
2468	 */
2469	i = 0;
2470	for (j = 1; j < numRels + 1; j++)
2471	{
2472		if (j == numRels || candidates[j].reloid != candidates[i].reloid)
2473		{
2474			if (j - i >= 2)
2475			{
2476				/* Create a group of relation indexes with the same oid */
2477				Relids		group = NULL;
2478				Relids		removed;
2479
2480				while (i < j)
2481				{
2482					group = bms_add_member(group, candidates[i].relid);
2483					i++;
2484				}
2485				relids = bms_del_members(relids, group);
2486
2487				/*
2488				 * Try to remove self-joins from a group of identical entries.
2489				 * Make the next attempt iteratively - if something is deleted
2490				 * from a group, changes in clauses and equivalence classes
2491				 * can give us a chance to find more candidates.
2492				 */
2493				do
2494				{
2495					Assert(!bms_overlap(group, toRemove));
2496					removed = remove_self_joins_one_group(root, group);
2497					toRemove = bms_add_members(toRemove, removed);
2498					group = bms_del_members(group, removed);
2499				} while (!bms_is_empty(removed) &&
2500						 bms_membership(group) == BMS_MULTIPLE);
2501				bms_free(removed);
2502				bms_free(group);
2503			}
2504			else
2505			{
2506				/* Single relation, just remove it from the set */
2507				relids = bms_del_member(relids, candidates[i].relid);
2508				i = j;
2509			}
2510		}
2511	}
2512
2513	Assert(bms_is_empty(relids));
2514
2515	return toRemove;
2516}

2518/*
2519 * Compare self-join candidates by their oids.
2520 */
2521static int
2522self_join_candidates_cmp(const void *a, const void *b)
2523{
2524	const SelfJoinCandidate *ca = (const SelfJoinCandidate *) a;
2525	const SelfJoinCandidate *cb = (const SelfJoinCandidate *) b;
2526
2527	if (ca->reloid != cb->reloid)
2528		return (ca->reloid < cb->reloid ? -1 : 1);
2529	else
2530		return 0;
2531}

2533/*
2534 * Find and remove useless self joins.
2535 *
2536 * Search for joins where a relation is joined to itself. If the join clause
2537 * for each tuple from one side of the join is proven to match the same
2538 * physical row (or nothing) on the other side, that self-join can be
2539 * eliminated from the query.  Suitable join clauses are assumed to be in the
2540 * form of X = X, and can be replaced with NOT NULL clauses.
2541 *
2542 * For the sake of simplicity, we don't apply this optimization to special
2543 * joins. Here is a list of what we could do in some particular cases:
2544 * 'a a1 semi join a a2': is reduced to inner by reduce_unique_semijoins,
2545 * and then removed normally.
2546 * 'a a1 anti join a a2': could simplify to a scan with 'outer quals AND
2547 * (IS NULL on join columns OR NOT inner quals)'.
2548 * 'a a1 left join a a2': could simplify to a scan like inner but without
2549 * NOT NULL conditions on join columns.
2550 * 'a a1 left join (a a2 join b)': can't simplify this, because join to b
2551 * can both remove rows and introduce duplicates.
2552 *
2553 * To search for removable joins, we order all the relations on their Oid,
2554 * go over each set with the same Oid, and consider each pair of relations
2555 * in this set.
2556 *
2557 * To remove the join, we mark one of the participating relations as dead
2558 * and rewrite all references to it to point to the remaining relation.
2559 * This includes modifying RestrictInfos, EquivalenceClasses, and
2560 * EquivalenceMembers. We also have to modify the row marks. The join clauses
2561 * of the removed relation become either restriction or join clauses, based on
2562 * whether they reference any relations not participating in the removed join.
2563 *
2564 * 'joinlist' is the top-level joinlist of the query. If it has any
2565 * references to the removed relations, we update them to point to the
2566 * remaining ones.
2567 */
2568List *
2569remove_useless_self_joins(PlannerInfo *root, List *joinlist)
2570{
2571	Relids		toRemove = NULL;
2572	int			relid = -1;
2573
2574	if (!enable_self_join_elimination || joinlist == NIL ||
2575		(list_length(joinlist) == 1 && !IsA(linitial(joinlist), List)))
2576		return joinlist;
2577
2578	/*
2579	 * Merge pairs of relations participated in self-join. Remove unnecessary
2580	 * range table entries.
2581	 */
2582	toRemove = remove_self_joins_recurse(root, joinlist, toRemove);
2583
2584	if (unlikely(toRemove != NULL))
2585	{
2586		/* At the end, remove orphaned relation links */
2587		while ((relid = bms_next_member(toRemove, relid)) >= 0)
2588		{
2589			int			nremoved = 0;
2590
2591			joinlist = remove_rel_from_joinlist(joinlist, relid, &nremoved);
2592			if (nremoved != 1)
2593				elog(ERROR, "failed to find relation %d in joinlist", relid);
2594		}
2595	}
2596
2597	return joinlist;
