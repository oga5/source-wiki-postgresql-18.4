46# Names of alternative dictionaries for all-ASCII words.  If not
47# listed, the language itself is used.  Note order dependency: Use of
48# some other language as ASCII dictionary must come after creation of

75sub usage

87sub GenerateTsearchFiles
88{
89	my $target = shift;
90	my $outdir_file = "$outdir_path/snowball_create.sql";
91
92	my $F;
93	my $D;
94	my $tmpl = read_file("$input_path/snowball.sql.in");
95
96	if ($depfile)
97	{
98		open($D, '>', "$outdir_path/snowball_create.dep")
99		  || die "Could not write snowball_create.dep";
100	}
101
102	print $D "$outdir_file: $input_path/snowball.sql.in\n" if $depfile;
103	print $D "$outdir_file: $input_path/snowball_func.sql.in\n" if $depfile;
104
105	open($F, '>', $outdir_file)
106	  || die "Could not write snowball_create.sql";
107
108	print $F "-- Language-specific snowball dictionaries\n";
109
110	print $F read_file("$input_path/snowball_func.sql.in");
111
112	foreach my $lang (@languages)
113	{
114		my $asclang = $ascii_languages{$lang} || $lang;
115		my $txt = $tmpl;
116		my $stop = '';
117		my $stopword_path = "$input_path/stopwords/$lang.stop";
118
119		if (-s "$stopword_path")
120		{
121			$stop = ", StopWords=$lang";
122
123			print $D "$outdir_file: $stopword_path\n" if $depfile;
124		}
125
126		$txt =~ s#_LANGNAME_#${lang}#gs;
127		$txt =~ s#_DICTNAME_#${lang}_stem#gs;
128		$txt =~ s#_CFGNAME_#${lang}#gs;
129		$txt =~ s#_ASCDICTNAME_#${asclang}_stem#gs;
130		$txt =~ s#_NONASCDICTNAME_#${lang}_stem#gs;
131		$txt =~ s#_STOPWORDS_#$stop#gs;
132		print $F $txt;
133	}
134	close($F);
135	close($D) if $depfile;
136	return;
137}
138
139
140sub read_file
141{
142	my $filename = shift;
143	my $F;
144	local $/ = undef;
145	open($F, '<', $filename) || die "Could not open file $filename\n";
146	my $txt = <$F>;
147	close($F);
148
149	return $txt;
150}
