43/*
44 * pg_regfree - free an RE (generic function, punts to RE-specific function)
45 *
46 * Ignoring invocation with NULL is a convenience.
47 */
48void
49pg_regfree(regex_t *re)
50{
51	if (re == NULL)
52		return;
53	(*((struct fns *) re->re_fns)->free) (re);
54}
