37/*
38 * The SIGUSR1 signal is multiplexed to support signaling multiple event
39 * types. The specific reason is communicated via flags in shared memory.
40 * We keep a boolean flag for each possible "reason", so that different
41 * reasons can be signaled to a process concurrently.  (However, if the same
42 * reason is signaled more than once nearly simultaneously, the process may
43 * observe it only once.)
44 *
45 * Each process that wants to receive signals registers its process ID
46 * in the ProcSignalSlots array. The array is indexed by ProcNumber to make
47 * slot allocation simple, and to avoid having to search the array when you
48 * know the ProcNumber of the process you're signaling.  (We do support
49 * signaling without ProcNumber, but it's a bit less efficient.)
50 *
51 * The fields in each slot are protected by a spinlock, pss_mutex. pss_pid can
52 * also be read without holding the spinlock, as a quick preliminary check
53 * when searching for a particular PID in the array.
54 *
55 * pss_signalFlags are intended to be set in cases where we don't need to
56 * keep track of whether or not the target process has handled the signal,
57 * but sometimes we need confirmation, as when making a global state change
58 * that cannot be considered complete until all backends have taken notice
59 * of it. For such use cases, we set a bit in pss_barrierCheckMask and then
60 * increment the current "barrier generation"; when the new barrier generation
61 * (or greater) appears in the pss_barrierGeneration flag of every process,
62 * we know that the message has been received everywhere.
63 */
64typedef struct
65{
66	pg_atomic_uint32 pss_pid;
67	int			pss_cancel_key_len; /* 0 means no cancellation is possible */
68	uint8		pss_cancel_key[MAX_CANCEL_KEY_LENGTH];
69	volatile sig_atomic_t pss_signalFlags[NUM_PROCSIGNALS];
70	slock_t		pss_mutex;		/* protects the above fields */
71
72	/* Barrier-related fields (not protected by pss_mutex) */
73	pg_atomic_uint64 pss_barrierGeneration;
74	pg_atomic_uint32 pss_barrierCheckMask;
75	ConditionVariable pss_barrierCV;
76} ProcSignalSlot;
77
78/*
79 * Information that is global to the entire ProcSignal system can be stored
80 * here.
81 *
82 * psh_barrierGeneration is the highest barrier generation in existence.
83 */
84struct ProcSignalHeader
85{
86	pg_atomic_uint64 psh_barrierGeneration;
87	ProcSignalSlot psh_slot[FLEXIBLE_ARRAY_MEMBER];
88};

90/*
91 * We reserve a slot for each possible ProcNumber, plus one for each
92 * possible auxiliary process type.  (This scheme assumes there is not
93 * more than one of any auxiliary process type at a time, except for
94 * IO workers.)
95 */
96#define NumProcSignalSlots	(MaxBackends + NUM_AUXILIARY_PROCS)
97
98/* Check whether the relevant type bit is set in the flags. */
99#define BARRIER_SHOULD_CHECK(flags, type) \
100	(((flags) & (((uint32) 1) << (uint32) (type))) != 0)
101
102/* Clear the relevant type bit from the flags. */
103#define BARRIER_CLEAR_BIT(flags, type) \
104	((flags) &= ~(((uint32) 1) << (uint32) (type)))
105
106NON_EXEC_STATIC ProcSignalHeader *ProcSignal = NULL;
107static ProcSignalSlot *MyProcSignalSlot = NULL;

113/*
114 * ProcSignalShmemSize
115 *		Compute space needed for ProcSignal's shared memory
116 */
117Size
118ProcSignalShmemSize(void)
119{
120	Size		size;
121
122	size = mul_size(NumProcSignalSlots, sizeof(ProcSignalSlot));
123	size = add_size(size, offsetof(ProcSignalHeader, psh_slot));
124	return size;
125}
126
127/*
128 * ProcSignalShmemInit
129 *		Allocate and initialize ProcSignal's shared memory
130 */
131void
132ProcSignalShmemInit(void)
133{
134	Size		size = ProcSignalShmemSize();
135	bool		found;
136
137	ProcSignal = (ProcSignalHeader *)
138		ShmemInitStruct("ProcSignal", size, &found);
139
140	/* If we're first, initialize. */
141	if (!found)
142	{
143		int			i;
144
145		pg_atomic_init_u64(&ProcSignal->psh_barrierGeneration, 0);
146
147		for (i = 0; i < NumProcSignalSlots; ++i)
148		{
149			ProcSignalSlot *slot = &ProcSignal->psh_slot[i];
150
151			SpinLockInit(&slot->pss_mutex);
152			pg_atomic_init_u32(&slot->pss_pid, 0);
153			slot->pss_cancel_key_len = 0;
154			MemSet(slot->pss_signalFlags, 0, sizeof(slot->pss_signalFlags));
155			pg_atomic_init_u64(&slot->pss_barrierGeneration, PG_UINT64_MAX);
156			pg_atomic_init_u32(&slot->pss_barrierCheckMask, 0);
157			ConditionVariableInit(&slot->pss_barrierCV);
158		}
159	}
160}

162/*
163 * ProcSignalInit
164 *		Register the current process in the ProcSignal array
165 */
166void
167ProcSignalInit(const uint8 *cancel_key, int cancel_key_len)
168{
169	ProcSignalSlot *slot;
170	uint64		barrier_generation;
171	uint32		old_pss_pid;
172
173	Assert(cancel_key_len >= 0 && cancel_key_len <= MAX_CANCEL_KEY_LENGTH);
174	if (MyProcNumber < 0)
175		elog(ERROR, "MyProcNumber not set");
176	if (MyProcNumber >= NumProcSignalSlots)
177		elog(ERROR, "unexpected MyProcNumber %d in ProcSignalInit (max %d)", MyProcNumber, NumProcSignalSlots);
178	slot = &ProcSignal->psh_slot[MyProcNumber];
179
180	SpinLockAcquire(&slot->pss_mutex);
181
182	/* Value used for sanity check below */
183	old_pss_pid = pg_atomic_read_u32(&slot->pss_pid);
184
185	/* Clear out any leftover signal reasons */
186	MemSet(slot->pss_signalFlags, 0, NUM_PROCSIGNALS * sizeof(sig_atomic_t));
187
188	/*
189	 * Initialize barrier state. Since we're a brand-new process, there
190	 * shouldn't be any leftover backend-private state that needs to be
191	 * updated. Therefore, we can broadcast the latest barrier generation and
192	 * disregard any previously-set check bits.
193	 *
194	 * NB: This only works if this initialization happens early enough in the
195	 * startup sequence that we haven't yet cached any state that might need
196	 * to be invalidated. That's also why we have a memory barrier here, to be
197	 * sure that any later reads of memory happen strictly after this.
198	 */
199	pg_atomic_write_u32(&slot->pss_barrierCheckMask, 0);
200	barrier_generation =
201		pg_atomic_read_u64(&ProcSignal->psh_barrierGeneration);
202	pg_atomic_write_u64(&slot->pss_barrierGeneration, barrier_generation);
203
204	if (cancel_key_len > 0)
205		memcpy(slot->pss_cancel_key, cancel_key, cancel_key_len);
206	slot->pss_cancel_key_len = cancel_key_len;
207	pg_atomic_write_u32(&slot->pss_pid, MyProcPid);
208
209	SpinLockRelease(&slot->pss_mutex);
210
211	/* Spinlock is released, do the check */
212	if (old_pss_pid != 0)
213		elog(LOG, "process %d taking over ProcSignal slot %d, but it's not empty",
214			 MyProcPid, MyProcNumber);
215
216	/* Remember slot location for CheckProcSignal */
217	MyProcSignalSlot = slot;
218
219	/* Set up to release the slot on process exit */
220	on_shmem_exit(CleanupProcSignalState, (Datum) 0);
221}

223/*
224 * CleanupProcSignalState
225 *		Remove current process from ProcSignal mechanism
226 *
227 * This function is called via on_shmem_exit() during backend shutdown.
228 */
229static void
230CleanupProcSignalState(int status, Datum arg)
231{
232	pid_t		old_pid;
233	ProcSignalSlot *slot = MyProcSignalSlot;
234
235	/*
236	 * Clear MyProcSignalSlot, so that a SIGUSR1 received after this point
237	 * won't try to access it after it's no longer ours (and perhaps even
238	 * after we've unmapped the shared memory segment).
239	 */
240	Assert(MyProcSignalSlot != NULL);
241	MyProcSignalSlot = NULL;
242
243	/* sanity check */
244	SpinLockAcquire(&slot->pss_mutex);
245	old_pid = pg_atomic_read_u32(&slot->pss_pid);
246	if (old_pid != MyProcPid)
247	{
248		/*
249		 * don't ERROR here. We're exiting anyway, and don't want to get into
250		 * infinite loop trying to exit
251		 */
252		SpinLockRelease(&slot->pss_mutex);
253		elog(LOG, "process %d releasing ProcSignal slot %d, but it contains %d",
254			 MyProcPid, (int) (slot - ProcSignal->psh_slot), (int) old_pid);
255		return;					/* XXX better to zero the slot anyway? */
256	}
257
258	/* Mark the slot as unused */
259	pg_atomic_write_u32(&slot->pss_pid, 0);
260	slot->pss_cancel_key_len = 0;
261
262	/*
263	 * Make this slot look like it's absorbed all possible barriers, so that
264	 * no barrier waits block on it.
265	 */
266	pg_atomic_write_u64(&slot->pss_barrierGeneration, PG_UINT64_MAX);
267
268	SpinLockRelease(&slot->pss_mutex);
269
270	ConditionVariableBroadcast(&slot->pss_barrierCV);
271}

273/*
274 * SendProcSignal
275 *		Send a signal to a Postgres process
276 *
277 * Providing procNumber is optional, but it will speed up the operation.
278 *
279 * On success (a signal was sent), zero is returned.
280 * On error, -1 is returned, and errno is set (typically to ESRCH or EPERM).
281 *
282 * Not to be confused with ProcSendSignal
283 */
284int
285SendProcSignal(pid_t pid, ProcSignalReason reason, ProcNumber procNumber)
286{
287	volatile ProcSignalSlot *slot;
288
289	if (procNumber != INVALID_PROC_NUMBER)
290	{
291		Assert(procNumber < NumProcSignalSlots);
292		slot = &ProcSignal->psh_slot[procNumber];
293
294		SpinLockAcquire(&slot->pss_mutex);
295		if (pg_atomic_read_u32(&slot->pss_pid) == pid)
296		{
297			/* Atomically set the proper flag */
298			slot->pss_signalFlags[reason] = true;
299			SpinLockRelease(&slot->pss_mutex);
300			/* Send signal */
301			return kill(pid, SIGUSR1);
302		}
303		SpinLockRelease(&slot->pss_mutex);
304	}
305	else
306	{
307		/*
308		 * procNumber not provided, so search the array using pid.  We search
309		 * the array back to front so as to reduce search overhead.  Passing
310		 * INVALID_PROC_NUMBER means that the target is most likely an
311		 * auxiliary process, which will have a slot near the end of the
312		 * array.
313		 */
314		int			i;
315
316		for (i = NumProcSignalSlots - 1; i >= 0; i--)
317		{
318			slot = &ProcSignal->psh_slot[i];
319
320			if (pg_atomic_read_u32(&slot->pss_pid) == pid)
321			{
322				SpinLockAcquire(&slot->pss_mutex);
323				if (pg_atomic_read_u32(&slot->pss_pid) == pid)
324				{
325					/* Atomically set the proper flag */
326					slot->pss_signalFlags[reason] = true;
327					SpinLockRelease(&slot->pss_mutex);
328					/* Send signal */
329					return kill(pid, SIGUSR1);
330				}
331				SpinLockRelease(&slot->pss_mutex);
332			}
333		}
334	}
335
336	errno = ESRCH;
337	return -1;
338}

340/*
341 * EmitProcSignalBarrier
342 *		Send a signal to every Postgres process
343 *
344 * The return value of this function is the barrier "generation" created
345 * by this operation. This value can be passed to WaitForProcSignalBarrier
346 * to wait until it is known that every participant in the ProcSignal
347 * mechanism has absorbed the signal (or started afterwards).
348 *
349 * Note that it would be a bad idea to use this for anything that happens
350 * frequently, as interrupting every backend could cause a noticeable
351 * performance hit.
352 *
353 * Callers are entitled to assume that this function will not throw ERROR
354 * or FATAL.
355 */
356uint64
357EmitProcSignalBarrier(ProcSignalBarrierType type)
358{
359	uint32		flagbit = 1 << (uint32) type;
360	uint64		generation;
361
362	/*
363	 * Set all the flags.
364	 *
365	 * Note that pg_atomic_fetch_or_u32 has full barrier semantics, so this is
366	 * totally ordered with respect to anything the caller did before, and
367	 * anything that we do afterwards. (This is also true of the later call to
368	 * pg_atomic_add_fetch_u64.)
369	 */
370	for (int i = 0; i < NumProcSignalSlots; i++)
371	{
372		volatile ProcSignalSlot *slot = &ProcSignal->psh_slot[i];
373
374		pg_atomic_fetch_or_u32(&slot->pss_barrierCheckMask, flagbit);
375	}
376
377	/*
378	 * Increment the generation counter.
379	 */
380	generation =
381		pg_atomic_add_fetch_u64(&ProcSignal->psh_barrierGeneration, 1);
382
383	/*
384	 * Signal all the processes, so that they update their advertised barrier
385	 * generation.
386	 *
387	 * Concurrency is not a problem here. Backends that have exited don't
388	 * matter, and new backends that have joined since we entered this
389	 * function must already have current state, since the caller is
390	 * responsible for making sure that the relevant state is entirely visible
391	 * before calling this function in the first place. We still have to wake
392	 * them up - because we can't distinguish between such backends and older
393	 * backends that need to update state - but they won't actually need to
394	 * change any state.
395	 */
396	for (int i = NumProcSignalSlots - 1; i >= 0; i--)
397	{
398		volatile ProcSignalSlot *slot = &ProcSignal->psh_slot[i];
399		pid_t		pid = pg_atomic_read_u32(&slot->pss_pid);
400
401		if (pid != 0)
402		{
403			SpinLockAcquire(&slot->pss_mutex);
404			pid = pg_atomic_read_u32(&slot->pss_pid);
405			if (pid != 0)
406			{
407				/* see SendProcSignal for details */
408				slot->pss_signalFlags[PROCSIG_BARRIER] = true;
409				SpinLockRelease(&slot->pss_mutex);
410				kill(pid, SIGUSR1);
411			}
412			else
413				SpinLockRelease(&slot->pss_mutex);
414		}
415	}
416
417	return generation;
418}

420/*
421 * WaitForProcSignalBarrier - wait until it is guaranteed that all changes
422 * requested by a specific call to EmitProcSignalBarrier() have taken effect.
423 */
424void
425WaitForProcSignalBarrier(uint64 generation)
426{
427	Assert(generation <= pg_atomic_read_u64(&ProcSignal->psh_barrierGeneration));
428
429	elog(DEBUG1,
430		 "waiting for all backends to process ProcSignalBarrier generation "
431		 UINT64_FORMAT,
432		 generation);
433
434	for (int i = NumProcSignalSlots - 1; i >= 0; i--)
435	{
436		ProcSignalSlot *slot = &ProcSignal->psh_slot[i];
437		uint64		oldval;
438
439		/*
440		 * It's important that we check only pss_barrierGeneration here and
441		 * not pss_barrierCheckMask. Bits in pss_barrierCheckMask get cleared
442		 * before the barrier is actually absorbed, but pss_barrierGeneration
443		 * is updated only afterward.
444		 */
445		oldval = pg_atomic_read_u64(&slot->pss_barrierGeneration);
446		while (oldval < generation)
447		{
448			if (ConditionVariableTimedSleep(&slot->pss_barrierCV,
449											5000,
450											WAIT_EVENT_PROC_SIGNAL_BARRIER))
451				ereport(LOG,
452						(errmsg("still waiting for backend with PID %d to accept ProcSignalBarrier",
453								(int) pg_atomic_read_u32(&slot->pss_pid))));
454			oldval = pg_atomic_read_u64(&slot->pss_barrierGeneration);
455		}
456		ConditionVariableCancelSleep();
457	}
458
459	elog(DEBUG1,
460		 "finished waiting for all backends to process ProcSignalBarrier generation "
461		 UINT64_FORMAT,
462		 generation);
463
464	/*
465	 * The caller is probably calling this function because it wants to read
466	 * the shared state or perform further writes to shared state once all
467	 * backends are known to have absorbed the barrier. However, the read of
468	 * pss_barrierGeneration was performed unlocked; insert a memory barrier
469	 * to separate it from whatever follows.
470	 */
471	pg_memory_barrier();
472}

474/*
475 * Handle receipt of an interrupt indicating a global barrier event.
476 *
477 * All the actual work is deferred to ProcessProcSignalBarrier(), because we
478 * cannot safely access the barrier generation inside the signal handler as
479 * 64bit atomics might use spinlock based emulation, even for reads. As this
480 * routine only gets called when PROCSIG_BARRIER is sent that won't cause a
481 * lot of unnecessary work.
482 */
483static void
484HandleProcSignalBarrierInterrupt(void)
485{
486	InterruptPending = true;
487	ProcSignalBarrierPending = true;
488	/* latch will be set by procsignal_sigusr1_handler */
489}
490
491/*
492 * Perform global barrier related interrupt checking.
493 *
494 * Any backend that participates in ProcSignal signaling must arrange to
495 * call this function periodically. It is called from CHECK_FOR_INTERRUPTS(),
496 * which is enough for normal backends, but not necessarily for all types of
497 * background processes.
498 */
499void
500ProcessProcSignalBarrier(void)
501{
502	uint64		local_gen;
503	uint64		shared_gen;
504	volatile uint32 flags;
505
506	Assert(MyProcSignalSlot);
507
508	/* Exit quickly if there's no work to do. */
509	if (!ProcSignalBarrierPending)
510		return;
511	ProcSignalBarrierPending = false;
512
513	/*
514	 * It's not unlikely to process multiple barriers at once, before the
515	 * signals for all the barriers have arrived. To avoid unnecessary work in
516	 * response to subsequent signals, exit early if we already have processed
517	 * all of them.
518	 */
519	local_gen = pg_atomic_read_u64(&MyProcSignalSlot->pss_barrierGeneration);
520	shared_gen = pg_atomic_read_u64(&ProcSignal->psh_barrierGeneration);
521
522	Assert(local_gen <= shared_gen);
523
524	if (local_gen == shared_gen)
525		return;
526
527	/*
528	 * Get and clear the flags that are set for this backend. Note that
529	 * pg_atomic_exchange_u32 is a full barrier, so we're guaranteed that the
530	 * read of the barrier generation above happens before we atomically
531	 * extract the flags, and that any subsequent state changes happen
532	 * afterward.
533	 *
534	 * NB: In order to avoid race conditions, we must zero
535	 * pss_barrierCheckMask first and only afterwards try to do barrier
536	 * processing. If we did it in the other order, someone could send us
537	 * another barrier of some type right after we called the
538	 * barrier-processing function but before we cleared the bit. We would
539	 * have no way of knowing that the bit needs to stay set in that case, so
540	 * the need to call the barrier-processing function again would just get
541	 * forgotten. So instead, we tentatively clear all the bits and then put
542	 * back any for which we don't manage to successfully absorb the barrier.
543	 */
544	flags = pg_atomic_exchange_u32(&MyProcSignalSlot->pss_barrierCheckMask, 0);
545
546	/*
547	 * If there are no flags set, then we can skip doing any real work.
548	 * Otherwise, establish a PG_TRY block, so that we don't lose track of
549	 * which types of barrier processing are needed if an ERROR occurs.
550	 */
551	if (flags != 0)
552	{
553		bool		success = true;
554
555		PG_TRY();
556		{
557			/*
558			 * Process each type of barrier. The barrier-processing functions
559			 * should normally return true, but may return false if the
560			 * barrier can't be absorbed at the current time. This should be
561			 * rare, because it's pretty expensive.  Every single
562			 * CHECK_FOR_INTERRUPTS() will return here until we manage to
563			 * absorb the barrier, and that cost will add up in a hurry.
564			 *
565			 * NB: It ought to be OK to call the barrier-processing functions
566			 * unconditionally, but it's more efficient to call only the ones
567			 * that might need us to do something based on the flags.
568			 */
569			while (flags != 0)
570			{
571				ProcSignalBarrierType type;
572				bool		processed = true;
573
574				type = (ProcSignalBarrierType) pg_rightmost_one_pos32(flags);
575				switch (type)
576				{
577					case PROCSIGNAL_BARRIER_SMGRRELEASE:
578						processed = ProcessBarrierSmgrRelease();
579						break;
580				}
581
582				/*
583				 * To avoid an infinite loop, we must always unset the bit in
584				 * flags.
585				 */
586				BARRIER_CLEAR_BIT(flags, type);
587
588				/*
589				 * If we failed to process the barrier, reset the shared bit
590				 * so we try again later, and set a flag so that we don't bump
591				 * our generation.
592				 */
593				if (!processed)
594				{
595					ResetProcSignalBarrierBits(((uint32) 1) << type);
596					success = false;
597				}
598			}
599		}
600		PG_CATCH();
601		{
602			/*
603			 * If an ERROR occurred, we'll need to try again later to handle
604			 * that barrier type and any others that haven't been handled yet
605			 * or weren't successfully absorbed.
606			 */
607			ResetProcSignalBarrierBits(flags);
608			PG_RE_THROW();
609		}
610		PG_END_TRY();
611
612		/*
613		 * If some barrier types were not successfully absorbed, we will have
614		 * to try again later.
615		 */
616		if (!success)
617			return;
618	}
619
620	/*
621	 * State changes related to all types of barriers that might have been
622	 * emitted have now been handled, so we can update our notion of the
623	 * generation to the one we observed before beginning the updates. If
624	 * things have changed further, it'll get fixed up when this function is
625	 * next called.
626	 */
627	pg_atomic_write_u64(&MyProcSignalSlot->pss_barrierGeneration, shared_gen);
628	ConditionVariableBroadcast(&MyProcSignalSlot->pss_barrierCV);
629}

631/*
632 * If it turns out that we couldn't absorb one or more barrier types, either
633 * because the barrier-processing functions returned false or due to an error,
634 * arrange for processing to be retried later.
635 */
636static void
637ResetProcSignalBarrierBits(uint32 flags)
638{
639	pg_atomic_fetch_or_u32(&MyProcSignalSlot->pss_barrierCheckMask, flags);
640	ProcSignalBarrierPending = true;
641	InterruptPending = true;
642}

644/*
645 * CheckProcSignal - check to see if a particular reason has been
646 * signaled, and clear the signal flag.  Should be called after receiving
647 * SIGUSR1.
648 */
649static bool
650CheckProcSignal(ProcSignalReason reason)
651{
652	volatile ProcSignalSlot *slot = MyProcSignalSlot;
653
654	if (slot != NULL)
655	{
656		/*
657		 * Careful here --- don't clear flag if we haven't seen it set.
658		 * pss_signalFlags is of type "volatile sig_atomic_t" to allow us to
659		 * read it here safely, without holding the spinlock.
660		 */
661		if (slot->pss_signalFlags[reason])
662		{
663			slot->pss_signalFlags[reason] = false;
664			return true;
665		}
666	}
667
668	return false;
669}

671/*
672 * procsignal_sigusr1_handler - handle SIGUSR1 signal.
673 */
674void
675procsignal_sigusr1_handler(SIGNAL_ARGS)
676{
677	if (CheckProcSignal(PROCSIG_CATCHUP_INTERRUPT))
678		HandleCatchupInterrupt();
679
680	if (CheckProcSignal(PROCSIG_NOTIFY_INTERRUPT))
681		HandleNotifyInterrupt();
682
683	if (CheckProcSignal(PROCSIG_PARALLEL_MESSAGE))
684		HandleParallelMessageInterrupt();
685
686	if (CheckProcSignal(PROCSIG_WALSND_INIT_STOPPING))
687		HandleWalSndInitStopping();
688
689	if (CheckProcSignal(PROCSIG_BARRIER))
690		HandleProcSignalBarrierInterrupt();
691
692	if (CheckProcSignal(PROCSIG_LOG_MEMORY_CONTEXT))
693		HandleLogMemoryContextInterrupt();
694
695	if (CheckProcSignal(PROCSIG_PARALLEL_APPLY_MESSAGE))
696		HandleParallelApplyMessageInterrupt();
697
698	if (CheckProcSignal(PROCSIG_SLOTSYNC_MESSAGE))
699		HandleSlotSyncMessageInterrupt();
700
701	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_DATABASE))
702		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_DATABASE);
703
704	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_TABLESPACE))
705		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_TABLESPACE);
706
707	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_LOCK))
708		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_LOCK);
709
710	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_SNAPSHOT))
711		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_SNAPSHOT);
712
713	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_LOGICALSLOT))
714		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_LOGICALSLOT);
715
716	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_STARTUP_DEADLOCK))
717		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_STARTUP_DEADLOCK);
718
719	if (CheckProcSignal(PROCSIG_RECOVERY_CONFLICT_BUFFERPIN))
720		HandleRecoveryConflictInterrupt(PROCSIG_RECOVERY_CONFLICT_BUFFERPIN);
721
722	SetLatch(MyLatch);
723}

725/*
726 * Send a query cancellation signal to backend.
727 *
728 * Note: This is called from a backend process before authentication.  We
729 * cannot take LWLocks yet, but that's OK; we rely on atomic reads of the
730 * fields in the ProcSignal slots.
731 */
732void
733SendCancelRequest(int backendPID, const uint8 *cancel_key, int cancel_key_len)
734{
735	if (backendPID == 0)
736	{
737		ereport(LOG, (errmsg("invalid cancel request with PID 0")));
738		return;
739	}
740
741	/*
742	 * See if we have a matching backend. Reading the pss_pid and
743	 * pss_cancel_key fields is racy, a backend might die and remove itself
744	 * from the array at any time.  The probability of the cancellation key
745	 * matching wrong process is miniscule, however, so we can live with that.
746	 * PIDs are reused too, so sending the signal based on PID is inherently
747	 * racy anyway, although OS's avoid reusing PIDs too soon.
748	 */
749	for (int i = 0; i < NumProcSignalSlots; i++)
750	{
751		ProcSignalSlot *slot = &ProcSignal->psh_slot[i];
752		bool		match;
753
754		if (pg_atomic_read_u32(&slot->pss_pid) != backendPID)
755			continue;
756
757		/* Acquire the spinlock and re-check */
758		SpinLockAcquire(&slot->pss_mutex);
759		if (pg_atomic_read_u32(&slot->pss_pid) != backendPID)
760		{
761			SpinLockRelease(&slot->pss_mutex);
762			continue;
763		}
764		else
765		{
766			match = slot->pss_cancel_key_len == cancel_key_len &&
767				timingsafe_bcmp(slot->pss_cancel_key, cancel_key, cancel_key_len) == 0;
768
769			SpinLockRelease(&slot->pss_mutex);
770
771			if (match)
772			{
773				/* Found a match; signal that backend to cancel current op */
774				ereport(DEBUG2,
775						(errmsg_internal("processing cancel request: sending SIGINT to process %d",
776										 backendPID)));
777
778				/*
779				 * If we have setsid(), signal the backend's whole process
780				 * group
781				 */
782#ifdef HAVE_SETSID
783				kill(-backendPID, SIGINT);
784#else
785				kill(backendPID, SIGINT);
786#endif
787			}
788			else
789			{
790				/* Right PID, wrong key: no way, Jose */
791				ereport(LOG,
792						(errmsg("wrong key in cancel request for process %d",
793								backendPID)));
794			}
795			return;
796		}
797	}
798
799	/* No matching backend */
800	ereport(LOG,
801			(errmsg("PID %d in cancel request did not match any process",
802					backendPID)));
803}
