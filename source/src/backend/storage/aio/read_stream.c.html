1/*-------------------------------------------------------------------------
2 *
3 * read_stream.c
4 *	  Mechanism for accessing buffered relation data with look-ahead
5 *
6 * Code that needs to access relation data typically pins blocks one at a
7 * time, often in a predictable order that might be sequential or data-driven.
8 * Calling the simple ReadBuffer() function for each block is inefficient,
9 * because blocks that are not yet in the buffer pool require I/O operations
10 * that are small and might stall waiting for storage.  This mechanism looks
11 * into the future and calls StartReadBuffers() and WaitReadBuffers() to read
12 * neighboring blocks together and ahead of time, with an adaptive look-ahead
13 * distance.
14 *
15 * A user-provided callback generates a stream of block numbers that is used
16 * to form reads of up to io_combine_limit, by attempting to merge them with a
17 * pending read.  When that isn't possible, the existing pending read is sent
18 * to StartReadBuffers() so that a new one can begin to form.
19 *
20 * The algorithm for controlling the look-ahead distance is based on recent
21 * cache hit and miss history.  When no I/O is necessary, there is no benefit
22 * in looking ahead more than one block.  This is the default initial
23 * assumption, but when blocks needing I/O are streamed, the distance is
24 * increased rapidly to try to benefit from I/O combining and concurrency.  It
25 * is reduced gradually when cached blocks are streamed.
26 *
27 * The main data structure is a circular queue of buffers of size
28 * max_pinned_buffers plus some extra space for technical reasons, ready to be
29 * returned by read_stream_next_buffer().  Each buffer also has an optional
30 * variable sized object that is passed from the callback to the consumer of
31 * buffers.
32 *
33 * Parallel to the queue of buffers, there is a circular queue of in-progress
34 * I/Os that have been started with StartReadBuffers(), and for which
35 * WaitReadBuffers() must be called before returning the buffer.
36 *
37 * For example, if the callback returns block numbers 10, 42, 43, 44, 60 in
38 * successive calls, then these data structures might appear as follows:
39 *
40 *                          buffers buf/data       ios
41 *
42 *                          +----+  +-----+       +--------+
43 *                          |    |  |     |  +----+ 42..44 | <- oldest_io_index
44 *                          +----+  +-----+  |    +--------+
45 *   oldest_buffer_index -> | 10 |  |  ?  |  | +--+ 60..60 |
46 *                          +----+  +-----+  | |  +--------+
47 *                          | 42 |  |  ?  |<-+ |  |        | <- next_io_index
48 *                          +----+  +-----+    |  +--------+
49 *                          | 43 |  |  ?  |    |  |        |
50 *                          +----+  +-----+    |  +--------+
51 *                          | 44 |  |  ?  |    |  |        |
52 *                          +----+  +-----+    |  +--------+
53 *                          | 60 |  |  ?  |<---+
54 *                          +----+  +-----+
55 *     next_buffer_index -> |    |  |     |
56 *                          +----+  +-----+
57 *
58 * In the example, 5 buffers are pinned, and the next buffer to be streamed to
59 * the client is block 10.  Block 10 was a hit and has no associated I/O, but
60 * the range 42..44 requires an I/O wait before its buffers are returned, as
61 * does block 60.
62 *
63 *
64 * Portions Copyright (c) 2024-2025, PostgreSQL Global Development Group
65 * Portions Copyright (c) 1994, Regents of the University of California
66 *
67 * IDENTIFICATION
68 *	  src/backend/storage/aio/read_stream.c
69 *
70 *-------------------------------------------------------------------------
71 */

89/*
90 * State for managing a stream of reads.
91 */
92struct ReadStream
93{
94	int16		max_ios;
95	int16		io_combine_limit;
96	int16		ios_in_progress;
97	int16		queue_size;
98	int16		max_pinned_buffers;
99	int16		forwarded_buffers;
100	int16		pinned_buffers;
101	int16		distance;
102	int16		initialized_buffers;
103	int			read_buffers_flags;
104	bool		sync_mode;		/* using io_method=sync */
105	bool		batch_mode;		/* READ_STREAM_USE_BATCHING */
106	bool		advice_enabled;
107	bool		temporary;
108
109	/*
110	 * One-block buffer to support 'ungetting' a block number, to resolve flow
111	 * control problems when I/Os are split.
112	 */
113	BlockNumber buffered_blocknum;
114
115	/*
116	 * The callback that will tell us which block numbers to read, and an
117	 * opaque pointer that will be pass to it for its own purposes.
118	 */
119	ReadStreamBlockNumberCB callback;
120	void	   *callback_private_data;
121
122	/* Next expected block, for detecting sequential access. */
123	BlockNumber seq_blocknum;
124	BlockNumber seq_until_processed;
125
126	/* The read operation we are currently preparing. */
127	BlockNumber pending_read_blocknum;
128	int16		pending_read_nblocks;
129
130	/* Space for buffers and optional per-buffer private data. */
131	size_t		per_buffer_data_size;
132	void	   *per_buffer_data;
133
134	/* Read operations that have been started but not waited for yet. */
135	InProgressIO *ios;
136	int16		oldest_io_index;
137	int16		next_io_index;
138
139	bool		fast_path;
140
141	/* Circular queue of buffers. */
142	int16		oldest_buffer_index;	/* Next pinned buffer to return */
143	int16		next_buffer_index;	/* Index of next buffer to pin */
144	Buffer		buffers[FLEXIBLE_ARRAY_MEMBER];
145};

157/*
158 * General-use ReadStreamBlockNumberCB for block range scans.  Loops over the
159 * blocks [current_blocknum, last_exclusive).
160 */
161BlockNumber
162block_range_read_stream_cb(ReadStream *stream,
163						   void *callback_private_data,
164						   void *per_buffer_data)
165{
166	BlockRangeReadStreamPrivate *p = callback_private_data;
167
168	if (p->current_blocknum < p->last_exclusive)
169		return p->current_blocknum++;
170
171	return InvalidBlockNumber;
172}

174/*
175 * Ask the callback which block it would like us to read next, with a one block
176 * buffer in front to allow read_stream_unget_block() to work.
177 */
178static inline BlockNumber
179read_stream_get_block(ReadStream *stream, void *per_buffer_data)
180{
181	BlockNumber blocknum;
182
183	blocknum = stream->buffered_blocknum;
184	if (blocknum != InvalidBlockNumber)
185		stream->buffered_blocknum = InvalidBlockNumber;
186	else
187	{
188		/*
189		 * Tell Valgrind that the per-buffer data is undefined.  That replaces
190		 * the "noaccess" state that was set when the consumer moved past this
191		 * entry last time around the queue, and should also catch callbacks
192		 * that fail to initialize data that the buffer consumer later
193		 * accesses.  On the first go around, it is undefined already.
194		 */
195		VALGRIND_MAKE_MEM_UNDEFINED(per_buffer_data,
196									stream->per_buffer_data_size);
197		blocknum = stream->callback(stream,
198									stream->callback_private_data,
199									per_buffer_data);
200	}
201
202	return blocknum;
203}

205/*
206 * In order to deal with buffer shortages and I/O limits after short reads, we
207 * sometimes need to defer handling of a block we've already consumed from the
208 * registered callback until later.
209 */
210static inline void
211read_stream_unget_block(ReadStream *stream, BlockNumber blocknum)
212{
213	/* We shouldn't ever unget more than one block. */
214	Assert(stream->buffered_blocknum == InvalidBlockNumber);
215	Assert(blocknum != InvalidBlockNumber);
216	stream->buffered_blocknum = blocknum;
217}

219/*
220 * Start as much of the current pending read as we can.  If we have to split it
221 * because of the per-backend buffer limit, or the buffer manager decides to
222 * split it, then the pending read is adjusted to hold the remaining portion.
223 *
224 * We can always start a read of at least size one if we have no progress yet.
225 * Otherwise it's possible that we can't start a read at all because of a lack
226 * of buffers, and then false is returned.  Buffer shortages also reduce the
227 * distance to a level that prevents look-ahead until buffers are released.
228 */
229static bool
230read_stream_start_pending_read(ReadStream *stream)
231{
232	bool		need_wait;
233	int			requested_nblocks;
234	int			nblocks;
235	int			flags;
236	int			forwarded;
237	int16		io_index;
238	int16		overflow;
239	int16		buffer_index;
240	int			buffer_limit;
241
242	/* This should only be called with a pending read. */
243	Assert(stream->pending_read_nblocks > 0);
244	Assert(stream->pending_read_nblocks <= stream->io_combine_limit);
245
246	/* We had better not exceed the per-stream buffer limit with this read. */
247	Assert(stream->pinned_buffers + stream->pending_read_nblocks <=
248		   stream->max_pinned_buffers);
249
250#ifdef USE_ASSERT_CHECKING
251	/* We had better not be overwriting an existing pinned buffer. */
252	if (stream->pinned_buffers > 0)
253		Assert(stream->next_buffer_index != stream->oldest_buffer_index);
254	else
255		Assert(stream->next_buffer_index == stream->oldest_buffer_index);
256
257	/*
258	 * Pinned buffers forwarded by a preceding StartReadBuffers() call that
259	 * had to split the operation should match the leading blocks of this
260	 * following StartReadBuffers() call.
261	 */
262	Assert(stream->forwarded_buffers <= stream->pending_read_nblocks);
263	for (int i = 0; i < stream->forwarded_buffers; ++i)
264		Assert(BufferGetBlockNumber(stream->buffers[stream->next_buffer_index + i]) ==
265			   stream->pending_read_blocknum + i);
266
267	/*
268	 * Check that we've cleared the queue/overflow entries corresponding to
269	 * the rest of the blocks covered by this read, unless it's the first go
270	 * around and we haven't even initialized them yet.
271	 */
272	for (int i = stream->forwarded_buffers; i < stream->pending_read_nblocks; ++i)
273		Assert(stream->next_buffer_index + i >= stream->initialized_buffers ||
274			   stream->buffers[stream->next_buffer_index + i] == InvalidBuffer);
275#endif
276
277	/* Do we need to issue read-ahead advice? */
278	flags = stream->read_buffers_flags;
279	if (stream->advice_enabled)
280	{
281		if (stream->pending_read_blocknum == stream->seq_blocknum)
282		{
283			/*
284			 * Sequential:  Issue advice until the preadv() calls have caught
285			 * up with the first advice issued for this sequential region, and
286			 * then stay of the way of the kernel's own read-ahead.
287			 */
288			if (stream->seq_until_processed != InvalidBlockNumber)
289				flags |= READ_BUFFERS_ISSUE_ADVICE;
290		}
291		else
292		{
293			/*
294			 * Random jump:  Note the starting location of a new potential
295			 * sequential region and start issuing advice.  Skip it this time
296			 * if the preadv() follows immediately, eg first block in stream.
297			 */
298			stream->seq_until_processed = stream->pending_read_blocknum;
299			if (stream->pinned_buffers > 0)
300				flags |= READ_BUFFERS_ISSUE_ADVICE;
301		}
302	}
303
304	/*
305	 * How many more buffers is this backend allowed?
306	 *
307	 * Forwarded buffers are already pinned and map to the leading blocks of
308	 * the pending read (the remaining portion of an earlier short read that
309	 * we're about to continue).  They are not counted in pinned_buffers, but
310	 * they are counted as pins already held by this backend according to the
311	 * buffer manager, so they must be added to the limit it grants us.
312	 */
313	if (stream->temporary)
314		buffer_limit = Min(GetAdditionalLocalPinLimit(), PG_INT16_MAX);
315	else
316		buffer_limit = Min(GetAdditionalPinLimit(), PG_INT16_MAX);
317	Assert(stream->forwarded_buffers <= stream->pending_read_nblocks);
318
319	buffer_limit += stream->forwarded_buffers;
320	buffer_limit = Min(buffer_limit, PG_INT16_MAX);
321
322	if (buffer_limit == 0 && stream->pinned_buffers == 0)
323		buffer_limit = 1;		/* guarantee progress */
324
325	/* Does the per-backend limit affect this read? */
326	nblocks = stream->pending_read_nblocks;
327	if (buffer_limit < nblocks)
328	{
329		int16		new_distance;
330
331		/* Shrink distance: no more look-ahead until buffers are released. */
332		new_distance = stream->pinned_buffers + buffer_limit;
333		if (stream->distance > new_distance)
334			stream->distance = new_distance;
335
336		/* Unless we have nothing to give the consumer, stop here. */
337		if (stream->pinned_buffers > 0)
338			return false;
339
340		/* A short read is required to make progress. */
341		nblocks = buffer_limit;
342	}
343
344	/*
345	 * We say how many blocks we want to read, but it may be smaller on return
346	 * if the buffer manager decides to shorten the read.  Initialize buffers
347	 * to InvalidBuffer (= not a forwarded buffer) as input on first use only,
348	 * and keep the original nblocks number so we can check for forwarded
349	 * buffers as output, below.
350	 */
351	buffer_index = stream->next_buffer_index;
352	io_index = stream->next_io_index;
353	while (stream->initialized_buffers < buffer_index + nblocks)
354		stream->buffers[stream->initialized_buffers++] = InvalidBuffer;
355	requested_nblocks = nblocks;
356	need_wait = StartReadBuffers(&stream->ios[io_index].op,
357								 &stream->buffers[buffer_index],
358								 stream->pending_read_blocknum,
359								 &nblocks,
360								 flags);
361	stream->pinned_buffers += nblocks;
362
363	/* Remember whether we need to wait before returning this buffer. */
364	if (!need_wait)
365	{
366		/* Look-ahead distance decays, no I/O necessary. */
367		if (stream->distance > 1)
368			stream->distance--;
369	}
370	else
371	{
372		/*
373		 * Remember to call WaitReadBuffers() before returning head buffer.
374		 * Look-ahead distance will be adjusted after waiting.
375		 */
376		stream->ios[io_index].buffer_index = buffer_index;
377		if (++stream->next_io_index == stream->max_ios)
378			stream->next_io_index = 0;
379		Assert(stream->ios_in_progress < stream->max_ios);
380		stream->ios_in_progress++;
381		stream->seq_blocknum = stream->pending_read_blocknum + nblocks;
382	}
383
384	/*
385	 * How many pins were acquired but forwarded to the next call?  These need
386	 * to be passed to the next StartReadBuffers() call by leaving them
387	 * exactly where they are in the queue, or released if the stream ends
388	 * early.  We need the number for accounting purposes, since they are not
389	 * counted in stream->pinned_buffers but we already hold them.
390	 */
391	forwarded = 0;
392	while (nblocks + forwarded < requested_nblocks &&
393		   stream->buffers[buffer_index + nblocks + forwarded] != InvalidBuffer)
394		forwarded++;
395	stream->forwarded_buffers = forwarded;
396
397	/*
398	 * We gave a contiguous range of buffer space to StartReadBuffers(), but
399	 * we want it to wrap around at queue_size.  Copy overflowing buffers to
400	 * the front of the array where they'll be consumed, but also leave a copy
401	 * in the overflow zone which the I/O operation has a pointer to (it needs
402	 * a contiguous array).  Both copies will be cleared when the buffers are
403	 * handed to the consumer.
404	 */
405	overflow = (buffer_index + nblocks + forwarded) - stream->queue_size;
406	if (overflow > 0)
407	{
408		Assert(overflow < stream->queue_size);	/* can't overlap */
409		memcpy(&stream->buffers[0],
410			   &stream->buffers[stream->queue_size],
411			   sizeof(stream->buffers[0]) * overflow);
412	}
413
414	/* Compute location of start of next read, without using % operator. */
415	buffer_index += nblocks;
416	if (buffer_index >= stream->queue_size)
417		buffer_index -= stream->queue_size;
418	Assert(buffer_index >= 0 && buffer_index < stream->queue_size);
419	stream->next_buffer_index = buffer_index;
420
421	/* Adjust the pending read to cover the remaining portion, if any. */
422	stream->pending_read_blocknum += nblocks;
423	stream->pending_read_nblocks -= nblocks;
424
425	return true;
426}

428static void
429read_stream_look_ahead(ReadStream *stream)
430{
431	/*
432	 * Allow amortizing the cost of submitting IO over multiple IOs. This
433	 * requires that we don't do any operations that could lead to a deadlock
434	 * with staged-but-unsubmitted IO. The callback needs to opt-in to being
435	 * careful.
436	 */
437	if (stream->batch_mode)
438		pgaio_enter_batchmode();
439
440	while (stream->ios_in_progress < stream->max_ios &&
441		   stream->pinned_buffers + stream->pending_read_nblocks < stream->distance)
442	{
443		BlockNumber blocknum;
444		int16		buffer_index;
445		void	   *per_buffer_data;
446
447		if (stream->pending_read_nblocks == stream->io_combine_limit)
448		{
449			read_stream_start_pending_read(stream);
450			continue;
451		}
452
453		/*
454		 * See which block the callback wants next in the stream.  We need to
455		 * compute the index of the Nth block of the pending read including
456		 * wrap-around, but we don't want to use the expensive % operator.
457		 */
458		buffer_index = stream->next_buffer_index + stream->pending_read_nblocks;
459		if (buffer_index >= stream->queue_size)
460			buffer_index -= stream->queue_size;
461		Assert(buffer_index >= 0 && buffer_index < stream->queue_size);
462		per_buffer_data = get_per_buffer_data(stream, buffer_index);
463		blocknum = read_stream_get_block(stream, per_buffer_data);
464		if (blocknum == InvalidBlockNumber)
465		{
466			/* End of stream. */
467			stream->distance = 0;
468			break;
469		}
470
471		/* Can we merge it with the pending read? */
472		if (stream->pending_read_nblocks > 0 &&
473			stream->pending_read_blocknum + stream->pending_read_nblocks == blocknum)
474		{
475			stream->pending_read_nblocks++;
476			continue;
477		}
478
479		/* We have to start the pending read before we can build another. */
480		while (stream->pending_read_nblocks > 0)
481		{
482			if (!read_stream_start_pending_read(stream) ||
483				stream->ios_in_progress == stream->max_ios)
484			{
485				/* We've hit the buffer or I/O limit.  Rewind and stop here. */
486				read_stream_unget_block(stream, blocknum);
487				if (stream->batch_mode)
488					pgaio_exit_batchmode();
489				return;
490			}
491		}
492
493		/* This is the start of a new pending read. */
494		stream->pending_read_blocknum = blocknum;
495		stream->pending_read_nblocks = 1;
496	}
497
498	/*
499	 * We don't start the pending read just because we've hit the distance
500	 * limit, preferring to give it another chance to grow to full
501	 * io_combine_limit size once more buffers have been consumed.  However,
502	 * if we've already reached io_combine_limit, or we've reached the
503	 * distance limit and there isn't anything pinned yet, or the callback has
504	 * signaled end-of-stream, we start the read immediately.  Note that the
505	 * pending read can exceed the distance goal, if the latter was reduced
506	 * after hitting the per-backend buffer limit.
507	 */
508	if (stream->pending_read_nblocks > 0 &&
509		(stream->pending_read_nblocks == stream->io_combine_limit ||
510		 (stream->pending_read_nblocks >= stream->distance &&
511		  stream->pinned_buffers == 0) ||
512		 stream->distance == 0) &&
513		stream->ios_in_progress < stream->max_ios)
514		read_stream_start_pending_read(stream);
515
516	/*
517	 * There should always be something pinned when we leave this function,
518	 * whether started by this call or not, unless we've hit the end of the
519	 * stream.  In the worst case we can always make progress one buffer at a
520	 * time.
521	 */
522	Assert(stream->pinned_buffers > 0 || stream->distance == 0);
523
524	if (stream->batch_mode)
525		pgaio_exit_batchmode();
526}

528/*
529 * Create a new read stream object that can be used to perform the equivalent
530 * of a series of ReadBuffer() calls for one fork of one relation.
531 * Internally, it generates larger vectored reads where possible by looking
532 * ahead.  The callback should return block numbers or InvalidBlockNumber to
533 * signal end-of-stream, and if per_buffer_data_size is non-zero, it may also
534 * write extra data for each block into the space provided to it.  It will
535 * also receive callback_private_data for its own purposes.
536 */
537static ReadStream *
538read_stream_begin_impl(int flags,
539					   BufferAccessStrategy strategy,
540					   Relation rel,
541					   SMgrRelation smgr,
542					   char persistence,
543					   ForkNumber forknum,
544					   ReadStreamBlockNumberCB callback,
545					   void *callback_private_data,
546					   size_t per_buffer_data_size)
547{
548	ReadStream *stream;
549	size_t		size;
550	int16		queue_size;
551	int16		queue_overflow;
552	int			max_ios;
553	int			strategy_pin_limit;
554	uint32		max_pinned_buffers;
555	uint32		max_possible_buffer_limit;
556	Oid			tablespace_id;
557
558	/*
559	 * Decide how many I/Os we will allow to run at the same time.  This
560	 * number also affects how far we look ahead for opportunities to start
561	 * more I/Os.
562	 */
563	tablespace_id = smgr->smgr_rlocator.locator.spcOid;
564	if (!OidIsValid(MyDatabaseId) ||
565		(rel && IsCatalogRelation(rel)) ||
566		IsCatalogRelationOid(smgr->smgr_rlocator.locator.relNumber))
567	{
568		/*
569		 * Avoid circularity while trying to look up tablespace settings or
570		 * before spccache.c is ready.
571		 */
572		max_ios = effective_io_concurrency;
573	}
574	else if (flags & READ_STREAM_MAINTENANCE)
575		max_ios = get_tablespace_maintenance_io_concurrency(tablespace_id);
576	else
577		max_ios = get_tablespace_io_concurrency(tablespace_id);
578
579	/* Cap to INT16_MAX to avoid overflowing below */
580	max_ios = Min(max_ios, PG_INT16_MAX);
581
582	/*
583	 * If starting a multi-block I/O near the end of the queue, we might
584	 * temporarily need extra space for overflowing buffers before they are
585	 * moved to regular circular position.  This is the maximum extra space we
586	 * could need.
587	 */
588	queue_overflow = io_combine_limit - 1;
589
590	/*
591	 * Choose the maximum number of buffers we're prepared to pin.  We try to
592	 * pin fewer if we can, though.  We add one so that we can make progress
593	 * even if max_ios is set to 0 (see also further down).  For max_ios > 0,
594	 * this also allows an extra full I/O's worth of buffers: after an I/O
595	 * finishes we don't want to have to wait for its buffers to be consumed
596	 * before starting a new one.
597	 *
598	 * Be careful not to allow int16 to overflow.  That is possible with the
599	 * current GUC range limits, so this is an artificial limit of ~32k
600	 * buffers and we'd need to adjust the types to exceed that.  We also have
601	 * to allow for the spare entry and the overflow space.
602	 */
603	max_pinned_buffers = (max_ios + 1) * io_combine_limit;
604	max_pinned_buffers = Min(max_pinned_buffers,
605							 PG_INT16_MAX - queue_overflow - 1);
606
607	/* Give the strategy a chance to limit the number of buffers we pin. */
608	strategy_pin_limit = GetAccessStrategyPinLimit(strategy);
609	max_pinned_buffers = Min(strategy_pin_limit, max_pinned_buffers);
610
611	/*
612	 * Also limit our queue to the maximum number of pins we could ever be
613	 * allowed to acquire according to the buffer manager.  We may not really
614	 * be able to use them all due to other pins held by this backend, but
615	 * we'll check that later in read_stream_start_pending_read().
616	 */
617	if (SmgrIsTemp(smgr))
618		max_possible_buffer_limit = GetLocalPinLimit();
619	else
620		max_possible_buffer_limit = GetPinLimit();
621	max_pinned_buffers = Min(max_pinned_buffers, max_possible_buffer_limit);
622
623	/*
624	 * The limit might be zero on a system configured with too few buffers for
625	 * the number of connections.  We need at least one to make progress.
626	 */
627	max_pinned_buffers = Max(1, max_pinned_buffers);
628
629	/*
630	 * We need one extra entry for buffers and per-buffer data, because users
631	 * of per-buffer data have access to the object until the next call to
632	 * read_stream_next_buffer(), so we need a gap between the head and tail
633	 * of the queue so that we don't clobber it.
634	 */
635	queue_size = max_pinned_buffers + 1;
636
637	/*
638	 * Allocate the object, the buffers, the ios and per_buffer_data space in
639	 * one big chunk.  Though we have queue_size buffers, we want to be able
640	 * to assume that all the buffers for a single read are contiguous (i.e.
641	 * don't wrap around halfway through), so we allow temporary overflows of
642	 * up to the maximum possible overflow size.
643	 */
644	size = offsetof(ReadStream, buffers);
645	size += sizeof(Buffer) * (queue_size + queue_overflow);
646	size += sizeof(InProgressIO) * Max(1, max_ios);
647	size += per_buffer_data_size * queue_size;
648	size += MAXIMUM_ALIGNOF * 2;
649	stream = (ReadStream *) palloc(size);
650	memset(stream, 0, offsetof(ReadStream, buffers));
651	stream->ios = (InProgressIO *)
652		MAXALIGN(&stream->buffers[queue_size + queue_overflow]);
653	if (per_buffer_data_size > 0)
654		stream->per_buffer_data = (void *)
655			MAXALIGN(&stream->ios[Max(1, max_ios)]);
656
657	stream->sync_mode = io_method == IOMETHOD_SYNC;
658	stream->batch_mode = flags & READ_STREAM_USE_BATCHING;
659
660#ifdef USE_PREFETCH
661
662	/*
663	 * Read-ahead advice simulating asynchronous I/O with synchronous calls.
664	 * Issue advice only if AIO is not used, direct I/O isn't enabled, the
665	 * caller hasn't promised sequential access (overriding our detection
666	 * heuristics), and max_ios hasn't been set to zero.
667	 */
668	if (stream->sync_mode &&
669		(io_direct_flags & IO_DIRECT_DATA) == 0 &&
670		(flags & READ_STREAM_SEQUENTIAL) == 0 &&
671		max_ios > 0)
672		stream->advice_enabled = true;
673#endif
674
675	/*
676	 * Setting max_ios to zero disables AIO and advice-based pseudo AIO, but
677	 * we still need to allocate space to combine and run one I/O.  Bump it up
678	 * to one, and remember to ask for synchronous I/O only.
679	 */
680	if (max_ios == 0)
681	{
682		max_ios = 1;
683		stream->read_buffers_flags = READ_BUFFERS_SYNCHRONOUSLY;
684	}
685
686	/*
687	 * Capture stable values for these two GUC-derived numbers for the
688	 * lifetime of this stream, so we don't have to worry about the GUCs
689	 * changing underneath us beyond this point.
690	 */
691	stream->max_ios = max_ios;
692	stream->io_combine_limit = io_combine_limit;
693
694	stream->per_buffer_data_size = per_buffer_data_size;
695	stream->max_pinned_buffers = max_pinned_buffers;
696	stream->queue_size = queue_size;
697	stream->callback = callback;
698	stream->callback_private_data = callback_private_data;
699	stream->buffered_blocknum = InvalidBlockNumber;
700	stream->seq_blocknum = InvalidBlockNumber;
701	stream->seq_until_processed = InvalidBlockNumber;
702	stream->temporary = SmgrIsTemp(smgr);
703
704	/*
705	 * Skip the initial ramp-up phase if the caller says we're going to be
706	 * reading the whole relation.  This way we start out assuming we'll be
707	 * doing full io_combine_limit sized reads.
708	 */
709	if (flags & READ_STREAM_FULL)
710		stream->distance = Min(max_pinned_buffers, stream->io_combine_limit);
711	else
712		stream->distance = 1;
713
714	/*
715	 * Since we always access the same relation, we can initialize parts of
716	 * the ReadBuffersOperation objects and leave them that way, to avoid
717	 * wasting CPU cycles writing to them for each read.
718	 */
719	for (int i = 0; i < max_ios; ++i)
720	{
721		stream->ios[i].op.rel = rel;
722		stream->ios[i].op.smgr = smgr;
723		stream->ios[i].op.persistence = persistence;
724		stream->ios[i].op.forknum = forknum;
725		stream->ios[i].op.strategy = strategy;
726	}
727
728	return stream;
729}

780/*
781 * Pull one pinned buffer out of a stream.  Each call returns successive
782 * blocks in the order specified by the callback.  If per_buffer_data_size was
783 * set to a non-zero size, *per_buffer_data receives a pointer to the extra
784 * per-buffer data that the callback had a chance to populate, which remains
785 * valid until the next call to read_stream_next_buffer().  When the stream
786 * runs out of data, InvalidBuffer is returned.  The caller may decide to end
787 * the stream early at any time by calling read_stream_end().
788 */
789Buffer
790read_stream_next_buffer(ReadStream *stream, void **per_buffer_data)
791{
792	Buffer		buffer;
793	int16		oldest_buffer_index;
794
795#ifndef READ_STREAM_DISABLE_FAST_PATH
796
797	/*
798	 * A fast path for all-cached scans.  This is the same as the usual
799	 * algorithm, but it is specialized for no I/O and no per-buffer data, so
800	 * we can skip the queue management code, stay in the same buffer slot and
801	 * use singular StartReadBuffer().
802	 */
803	if (likely(stream->fast_path))
804	{
805		BlockNumber next_blocknum;
806
807		/* Fast path assumptions. */
808		Assert(stream->ios_in_progress == 0);
809		Assert(stream->forwarded_buffers == 0);
810		Assert(stream->pinned_buffers == 1);
811		Assert(stream->distance == 1);
812		Assert(stream->pending_read_nblocks == 0);
813		Assert(stream->per_buffer_data_size == 0);
814		Assert(stream->initialized_buffers > stream->oldest_buffer_index);
815
816		/* We're going to return the buffer we pinned last time. */
817		oldest_buffer_index = stream->oldest_buffer_index;
818		Assert((oldest_buffer_index + 1) % stream->queue_size ==
819			   stream->next_buffer_index);
820		buffer = stream->buffers[oldest_buffer_index];
821		Assert(buffer != InvalidBuffer);
822
823		/* Choose the next block to pin. */
824		next_blocknum = read_stream_get_block(stream, NULL);
825
826		if (likely(next_blocknum != InvalidBlockNumber))
827		{
828			int			flags = stream->read_buffers_flags;
829
830			if (stream->advice_enabled)
831				flags |= READ_BUFFERS_ISSUE_ADVICE;
832
833			/*
834			 * Pin a buffer for the next call.  Same buffer entry, and
835			 * arbitrary I/O entry (they're all free).  We don't have to
836			 * adjust pinned_buffers because we're transferring one to caller
837			 * but pinning one more.
838			 *
839			 * In the fast path we don't need to check the pin limit.  We're
840			 * always allowed at least one pin so that progress can be made,
841			 * and that's all we need here.  Although two pins are momentarily
842			 * held at the same time, the model used here is that the stream
843			 * holds only one, and the other now belongs to the caller.
844			 */
845			if (likely(!StartReadBuffer(&stream->ios[0].op,
846										&stream->buffers[oldest_buffer_index],
847										next_blocknum,
848										flags)))
849			{
850				/* Fast return. */
851				return buffer;
852			}
853
854			/* Next call must wait for I/O for the newly pinned buffer. */
855			stream->oldest_io_index = 0;
856			stream->next_io_index = stream->max_ios > 1 ? 1 : 0;
857			stream->ios_in_progress = 1;
858			stream->ios[0].buffer_index = oldest_buffer_index;
859			stream->seq_blocknum = next_blocknum + 1;
860		}
861		else
862		{
863			/* No more blocks, end of stream. */
864			stream->distance = 0;
865			stream->oldest_buffer_index = stream->next_buffer_index;
866			stream->pinned_buffers = 0;
867			stream->buffers[oldest_buffer_index] = InvalidBuffer;
868		}
869
870		stream->fast_path = false;
871		return buffer;
872	}
873#endif
874
875	if (unlikely(stream->pinned_buffers == 0))
876	{
877		Assert(stream->oldest_buffer_index == stream->next_buffer_index);
878
879		/* End of stream reached?  */
880		if (stream->distance == 0)
881			return InvalidBuffer;
882
883		/*
884		 * The usual order of operations is that we look ahead at the bottom
885		 * of this function after potentially finishing an I/O and making
886		 * space for more, but if we're just starting up we'll need to crank
887		 * the handle to get started.
888		 */
889		read_stream_look_ahead(stream);
890
891		/* End of stream reached? */
892		if (stream->pinned_buffers == 0)
893		{
894			Assert(stream->distance == 0);
895			return InvalidBuffer;
896		}
897	}
898
899	/* Grab the oldest pinned buffer and associated per-buffer data. */
900	Assert(stream->pinned_buffers > 0);
901	oldest_buffer_index = stream->oldest_buffer_index;
902	Assert(oldest_buffer_index >= 0 &&
903		   oldest_buffer_index < stream->queue_size);
904	buffer = stream->buffers[oldest_buffer_index];
905	if (per_buffer_data)
906		*per_buffer_data = get_per_buffer_data(stream, oldest_buffer_index);
907
908	Assert(BufferIsValid(buffer));
909
910	/* Do we have to wait for an associated I/O first? */
911	if (stream->ios_in_progress > 0 &&
912		stream->ios[stream->oldest_io_index].buffer_index == oldest_buffer_index)
913	{
914		int16		io_index = stream->oldest_io_index;
915		int32		distance;	/* wider temporary value, clamped below */
916
917		/* Sanity check that we still agree on the buffers. */
918		Assert(stream->ios[io_index].op.buffers ==
919			   &stream->buffers[oldest_buffer_index]);
920
921		WaitReadBuffers(&stream->ios[io_index].op);
922
923		Assert(stream->ios_in_progress > 0);
924		stream->ios_in_progress--;
925		if (++stream->oldest_io_index == stream->max_ios)
926			stream->oldest_io_index = 0;
927
928		/* Look-ahead distance ramps up rapidly after we do I/O. */
929		distance = stream->distance * 2;
930		distance = Min(distance, stream->max_pinned_buffers);
931		stream->distance = distance;
932
933		/*
934		 * If we've reached the first block of a sequential region we're
935		 * issuing advice for, cancel that until the next jump.  The kernel
936		 * will see the sequential preadv() pattern starting here.
937		 */
938		if (stream->advice_enabled &&
939			stream->ios[io_index].op.blocknum == stream->seq_until_processed)
940			stream->seq_until_processed = InvalidBlockNumber;
941	}
942
943	/*
944	 * We must zap this queue entry, or else it would appear as a forwarded
945	 * buffer.  If it's potentially in the overflow zone (ie from a
946	 * multi-block I/O that wrapped around the queue), also zap the copy.
947	 */
948	stream->buffers[oldest_buffer_index] = InvalidBuffer;
949	if (oldest_buffer_index < stream->io_combine_limit - 1)
950		stream->buffers[stream->queue_size + oldest_buffer_index] =
951			InvalidBuffer;
952
953#if defined(CLOBBER_FREED_MEMORY) || defined(USE_VALGRIND)
954
955	/*
956	 * The caller will get access to the per-buffer data, until the next call.
957	 * We wipe the one before, which is never occupied because queue_size
958	 * allowed one extra element.  This will hopefully trip up client code
959	 * that is holding a dangling pointer to it.
960	 */
961	if (stream->per_buffer_data)
962	{
963		void	   *per_buffer_data;
964
965		per_buffer_data = get_per_buffer_data(stream,
966											  oldest_buffer_index == 0 ?
967											  stream->queue_size - 1 :
968											  oldest_buffer_index - 1);
969
970#if defined(CLOBBER_FREED_MEMORY)
971		/* This also tells Valgrind the memory is "noaccess". */
972		wipe_mem(per_buffer_data, stream->per_buffer_data_size);
973#elif defined(USE_VALGRIND)
974		/* Tell it ourselves. */
975		VALGRIND_MAKE_MEM_NOACCESS(per_buffer_data,
976								   stream->per_buffer_data_size);
977#endif
978	}
979#endif
980
981	/* Pin transferred to caller. */
982	Assert(stream->pinned_buffers > 0);
983	stream->pinned_buffers--;
984
985	/* Advance oldest buffer, with wrap-around. */
986	stream->oldest_buffer_index++;
987	if (stream->oldest_buffer_index == stream->queue_size)
988		stream->oldest_buffer_index = 0;
989
990	/* Prepare for the next call. */
991	read_stream_look_ahead(stream);
992
993#ifndef READ_STREAM_DISABLE_FAST_PATH
994	/* See if we can take the fast path for all-cached scans next time. */
995	if (stream->ios_in_progress == 0 &&
996		stream->forwarded_buffers == 0 &&
997		stream->pinned_buffers == 1 &&
998		stream->distance == 1 &&
999		stream->pending_read_nblocks == 0 &&
1000		stream->per_buffer_data_size == 0)
1001	{
1002		/*
1003		 * The fast path spins on one buffer entry repeatedly instead of
1004		 * rotating through the whole queue and clearing the entries behind
1005		 * it.  If the buffer it starts with happened to be forwarded between
1006		 * StartReadBuffers() calls and also wrapped around the circular queue
1007		 * partway through, then a copy also exists in the overflow zone, and
1008		 * it won't clear it out as the regular path would.  Do that now, so
1009		 * it doesn't need code for that.
1010		 */
1011		if (stream->oldest_buffer_index < stream->io_combine_limit - 1)
1012			stream->buffers[stream->queue_size + stream->oldest_buffer_index] =
1013				InvalidBuffer;
1014
1015		stream->fast_path = true;
1016	}
1017#endif
1018
1019	return buffer;
1020}

1022/*
1023 * Transitional support for code that would like to perform or skip reads
1024 * itself, without using the stream.  Returns, and consumes, the next block
1025 * number that would be read by the stream's look-ahead algorithm, or
1026 * InvalidBlockNumber if the end of the stream is reached.  Also reports the
1027 * strategy that would be used to read it.
1028 */
1029BlockNumber
1030read_stream_next_block(ReadStream *stream, BufferAccessStrategy *strategy)
1031{
1032	*strategy = stream->ios[0].op.strategy;
1033	return read_stream_get_block(stream, NULL);
1034}

1036/*
1037 * Reset a read stream by releasing any queued up buffers, allowing the stream
1038 * to be used again for different blocks.  This can be used to clear an
1039 * end-of-stream condition and start again, or to throw away blocks that were
1040 * speculatively read and read some different blocks instead.
1041 */
1042void
1043read_stream_reset(ReadStream *stream)
1044{
1045	int16		index;
1046	Buffer		buffer;
1047
1048	/* Stop looking ahead. */
1049	stream->distance = 0;
1050
1051	/* Forget buffered block number and fast path state. */
1052	stream->buffered_blocknum = InvalidBlockNumber;
1053	stream->fast_path = false;
1054
1055	/* Unpin anything that wasn't consumed. */
1056	while ((buffer = read_stream_next_buffer(stream, NULL)) != InvalidBuffer)
1057		ReleaseBuffer(buffer);
1058
1059	/* Unpin any unused forwarded buffers. */
1060	index = stream->next_buffer_index;
1061	while (index < stream->initialized_buffers &&
1062		   (buffer = stream->buffers[index]) != InvalidBuffer)
1063	{
1064		Assert(stream->forwarded_buffers > 0);
1065		stream->forwarded_buffers--;
1066		ReleaseBuffer(buffer);
1067
1068		stream->buffers[index] = InvalidBuffer;
1069		if (index < stream->io_combine_limit - 1)
1070			stream->buffers[stream->queue_size + index] = InvalidBuffer;
1071
1072		if (++index == stream->queue_size)
1073			index = 0;
1074	}
1075
1076	Assert(stream->forwarded_buffers == 0);
1077	Assert(stream->pinned_buffers == 0);
1078	Assert(stream->ios_in_progress == 0);
1079
1080	/* Start off assuming data is cached. */
1081	stream->distance = 1;
1082}

1084/*
1085 * Release and free a read stream.
1086 */
1087void
1088read_stream_end(ReadStream *stream)
1089{
1090	read_stream_reset(stream);
1091	pfree(stream);
