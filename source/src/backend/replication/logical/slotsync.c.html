1/*-------------------------------------------------------------------------
2 * slotsync.c
3 *	   Functionality for synchronizing slots to a standby server from the
4 *         primary server.
5 *
6 * Copyright (c) 2024-2025, PostgreSQL Global Development Group
7 *
8 * IDENTIFICATION
9 *	  src/backend/replication/logical/slotsync.c
10 *
11 * This file contains the code for slot synchronization on a physical standby
12 * to fetch logical failover slots information from the primary server, create
13 * the slots on the standby and synchronize them periodically.
14 *
15 * Slot synchronization can be performed either automatically by enabling slot
16 * sync worker or manually by calling SQL function pg_sync_replication_slots().
17 *
18 * If the WAL corresponding to the remote's restart_lsn is not available on the
19 * physical standby or the remote's catalog_xmin precedes the oldest xid for
20 * which it is guaranteed that rows wouldn't have been removed then we cannot
21 * create the local standby slot because that would mean moving the local slot
22 * backward and decoding won't be possible via such a slot. In this case, the
23 * slot will be marked as RS_TEMPORARY. Once the primary server catches up,
24 * the slot will be marked as RS_PERSISTENT (which means sync-ready) after
25 * which slot sync worker can perform the sync periodically or user can call
26 * pg_sync_replication_slots() periodically to perform the syncs.
27 *
28 * If synchronized slots fail to build a consistent snapshot from the
29 * restart_lsn before reaching confirmed_flush_lsn, they would become
30 * unreliable after promotion due to potential data loss from changes
31 * before reaching a consistent point. This can happen because the slots can
32 * be synced at some random time and we may not reach the consistent point
33 * at the same WAL location as the primary. So, we mark such slots as
34 * RS_TEMPORARY. Once the decoding from corresponding LSNs can reach a
35 * consistent point, they will be marked as RS_PERSISTENT.
36 *
37 * The slot sync worker waits for some time before the next synchronization,
38 * with the duration varying based on whether any slots were updated during
39 * the last cycle. Refer to the comments above wait_for_slot_activity() for
40 * more details.
41 *
42 * Any standby synchronized slots will be dropped if they no longer need
43 * to be synchronized. See comment atop drop_local_obsolete_slots() for more
44 * details.
45 *---------------------------------------------------------------------------
46 */

72/*
73 * Struct for sharing information to control slot synchronization.
74 *
75 * The 'pid' is either the slot sync worker's pid or the backend's pid running
76 * the SQL function pg_sync_replication_slots(). On promotion, the startup
77 * process sets 'stopSignaled' and uses this 'pid' to signal the synchronizing
78 * process with PROCSIG_SLOTSYNC_MESSAGE and also to wake it up so that the
79 * process can immediately stop its synchronizing work.
80 * Setting 'stopSignaled' on the other hand is used to handle the race
81 * condition when the postmaster has not noticed the promotion yet and thus may
82 * end up restarting the slot sync worker. If 'stopSignaled' is set, the worker
83 * will exit in such a case. The SQL function pg_sync_replication_slots() will
84 * also error out if this flag is set. Note that we don't need to reset this
85 * variable as after promotion the slot sync worker won't be restarted because
86 * the pmState changes to PM_RUN from PM_HOT_STANDBY and we don't support
87 * demoting primary without restarting the server.
88 * See LaunchMissingBackgroundProcesses.
89 *
90 * The 'syncing' flag is needed to prevent concurrent slot syncs to avoid slot
91 * overwrites.
92 *
93 * The 'last_start_time' is needed by postmaster to start the slot sync worker
94 * once per SLOTSYNC_RESTART_INTERVAL_SEC. In cases where an immediate restart
95 * is expected (e.g., slot sync GUCs change), slot sync worker will reset
96 * last_start_time before exiting, so that postmaster can start the worker
97 * without waiting for SLOTSYNC_RESTART_INTERVAL_SEC.
98 */
99typedef struct SlotSyncCtxStruct
100{
101	pid_t		pid;
102	bool		stopSignaled;
103	bool		syncing;
104	time_t		last_start_time;
105	slock_t		mutex;
106} SlotSyncCtxStruct;

163/*
164 * If necessary, update the local synced slot's metadata based on the data
165 * from the remote slot.
166 *
167 * If no update was needed (the data of the remote slot is the same as the
168 * local slot) return false, otherwise true.
169 *
170 * *found_consistent_snapshot will be true iff the remote slot's LSN or xmin is
171 * modified, and decoding from the corresponding LSN's can reach a
172 * consistent snapshot.
173 *
174 * *remote_slot_precedes will be true if the remote slot's LSN or xmin
175 * precedes locally reserved position.
176 */
177static bool
178update_local_synced_slot(RemoteSlot *remote_slot, Oid remote_dbid,
179						 bool *found_consistent_snapshot,
180						 bool *remote_slot_precedes)
181{
182	ReplicationSlot *slot = MyReplicationSlot;
183	bool		updated_xmin_or_lsn = false;
184	bool		updated_config = false;
185
186	Assert(slot->data.invalidated == RS_INVAL_NONE);
187
188	if (found_consistent_snapshot)
189		*found_consistent_snapshot = false;
190
191	if (remote_slot_precedes)
192		*remote_slot_precedes = false;
193
194	/*
195	 * Don't overwrite if we already have a newer catalog_xmin and
196	 * restart_lsn.
197	 */
198	if (remote_slot->restart_lsn < slot->data.restart_lsn ||
199		TransactionIdPrecedes(remote_slot->catalog_xmin,
200							  slot->data.catalog_xmin))
201	{
202		/*
203		 * This can happen in following situations:
204		 *
205		 * If the slot is temporary, it means either the initial WAL location
206		 * reserved for the local slot is ahead of the remote slot's
207		 * restart_lsn or the initial xmin_horizon computed for the local slot
208		 * is ahead of the remote slot.
209		 *
210		 * If the slot is persistent, both restart_lsn and catalog_xmin of the
211		 * synced slot could still be ahead of the remote slot. Since we use
212		 * slot advance functionality to keep snapbuild/slot updated, it is
213		 * possible that the restart_lsn and catalog_xmin are advanced to a
214		 * later position than it has on the primary. This can happen when
215		 * slot advancing machinery finds running xacts record after reaching
216		 * the consistent state at a later point than the primary where it
217		 * serializes the snapshot and updates the restart_lsn.
218		 *
219		 * We LOG the message if the slot is temporary as it can help the user
220		 * to understand why the slot is not sync-ready. In the case of a
221		 * persistent slot, it would be a more common case and won't directly
222		 * impact the users, so we used DEBUG1 level to log the message.
223		 */
224		ereport(slot->data.persistency == RS_TEMPORARY ? LOG : DEBUG1,
225				errmsg("could not synchronize replication slot \"%s\"",
226					   remote_slot->name),
227				errdetail("Synchronization could lead to data loss, because the remote slot needs WAL at LSN %X/%X and catalog xmin %u, but the standby has LSN %X/%X and catalog xmin %u.",
228						  LSN_FORMAT_ARGS(remote_slot->restart_lsn),
229						  remote_slot->catalog_xmin,
230						  LSN_FORMAT_ARGS(slot->data.restart_lsn),
231						  slot->data.catalog_xmin));
232
233		if (remote_slot_precedes)
234			*remote_slot_precedes = true;
235
236		/*
237		 * Skip updating the configuration. This is required to avoid syncing
238		 * two_phase_at without syncing confirmed_lsn. Otherwise, the prepared
239		 * transaction between old confirmed_lsn and two_phase_at will
240		 * unexpectedly get decoded and sent to the downstream after
241		 * promotion. See comments in ReorderBufferFinishPrepared.
242		 */
243		return false;
244	}
245
246	/*
247	 * Attempt to sync LSNs and xmins only if remote slot is ahead of local
248	 * slot.
249	 */
250	if (remote_slot->confirmed_lsn > slot->data.confirmed_flush ||
251		remote_slot->restart_lsn > slot->data.restart_lsn ||
252		TransactionIdFollows(remote_slot->catalog_xmin,
253							 slot->data.catalog_xmin))
254	{
255		/*
256		 * We can't directly copy the remote slot's LSN or xmin unless there
257		 * exists a consistent snapshot at that point. Otherwise, after
258		 * promotion, the slots may not reach a consistent point before the
259		 * confirmed_flush_lsn which can lead to a data loss. To avoid data
260		 * loss, we let slot machinery advance the slot which ensures that
261		 * snapbuilder/slot statuses are updated properly.
262		 */
263		if (SnapBuildSnapshotExists(remote_slot->restart_lsn))
264		{
265			/*
266			 * Update the slot info directly if there is a serialized snapshot
267			 * at the restart_lsn, as the slot can quickly reach consistency
268			 * at restart_lsn by restoring the snapshot.
269			 */
270			SpinLockAcquire(&slot->mutex);
271			slot->data.restart_lsn = remote_slot->restart_lsn;
272			slot->data.confirmed_flush = remote_slot->confirmed_lsn;
273			slot->data.catalog_xmin = remote_slot->catalog_xmin;
274			SpinLockRelease(&slot->mutex);
275
276			if (found_consistent_snapshot)
277				*found_consistent_snapshot = true;
278
279			updated_xmin_or_lsn = true;
280		}
281		else
282		{
283			XLogRecPtr	old_confirmed_lsn = slot->data.confirmed_flush;
284			XLogRecPtr	old_restart_lsn = slot->data.restart_lsn;
285			XLogRecPtr	old_catalog_xmin = slot->data.catalog_xmin;
286
287			LogicalSlotAdvanceAndCheckSnapState(remote_slot->confirmed_lsn,
288												found_consistent_snapshot);
289
290			/* Sanity check */
291			if (slot->data.confirmed_flush != remote_slot->confirmed_lsn)
292				ereport(ERROR,
293						errmsg_internal("synchronized confirmed_flush for slot \"%s\" differs from remote slot",
294										remote_slot->name),
295						errdetail_internal("Remote slot has LSN %X/%X but local slot has LSN %X/%X.",
296										   LSN_FORMAT_ARGS(remote_slot->confirmed_lsn),
297										   LSN_FORMAT_ARGS(slot->data.confirmed_flush)));
298
299			/*
300			 * It is possible that the slot's xmin or LSNs are not updated,
301			 * when the synced slot has reached consistent snapshot state or
302			 * cannot build one at all.
303			 */
304			updated_xmin_or_lsn = (old_confirmed_lsn != slot->data.confirmed_flush ||
305								   old_restart_lsn != slot->data.restart_lsn ||
306								   old_catalog_xmin != slot->data.catalog_xmin);
307		}
308	}
309
310	if (remote_dbid != slot->data.database ||
311		remote_slot->two_phase != slot->data.two_phase ||
312		remote_slot->failover != slot->data.failover ||
313		strcmp(remote_slot->plugin, NameStr(slot->data.plugin)) != 0 ||
314		remote_slot->two_phase_at != slot->data.two_phase_at)
315	{
316		NameData	plugin_name;
317
318		/* Avoid expensive operations while holding a spinlock. */
319		namestrcpy(&plugin_name, remote_slot->plugin);
320
321		SpinLockAcquire(&slot->mutex);
322		slot->data.plugin = plugin_name;
323		slot->data.database = remote_dbid;
324		slot->data.two_phase = remote_slot->two_phase;
325		slot->data.two_phase_at = remote_slot->two_phase_at;
326		slot->data.failover = remote_slot->failover;
327		SpinLockRelease(&slot->mutex);
328
329		updated_config = true;
330
331		/*
332		 * Ensure that there is no risk of sending prepared transactions
333		 * unexpectedly after the promotion.
334		 */
335		Assert(slot->data.two_phase_at <= slot->data.confirmed_flush);
336	}
337
338	/*
339	 * We have to write the changed xmin to disk *before* we change the
340	 * in-memory value, otherwise after a crash we wouldn't know that some
341	 * catalog tuples might have been removed already.
342	 */
343	if (updated_config || updated_xmin_or_lsn)
344	{
345		ReplicationSlotMarkDirty();
346		ReplicationSlotSave();
347	}
348
349	/*
350	 * Now the new xmin is safely on disk, we can let the global value
351	 * advance. We do not take ProcArrayLock or similar since we only advance
352	 * xmin here and there's not much harm done by a concurrent computation
353	 * missing that.
354	 */
355	if (updated_xmin_or_lsn)
356	{
357		SpinLockAcquire(&slot->mutex);
358		slot->effective_catalog_xmin = remote_slot->catalog_xmin;
359		SpinLockRelease(&slot->mutex);
360
361		ReplicationSlotsComputeRequiredXmin(false);
362		ReplicationSlotsComputeRequiredLSN();
363	}
364
365	return updated_config || updated_xmin_or_lsn;
366}

368/*
369 * Get the list of local logical slots that are synchronized from the
370 * primary server.
371 */
372static List *
373get_local_synced_slots(void)
374{
375	List	   *local_slots = NIL;
376
377	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
378
379	for (int i = 0; i < max_replication_slots; i++)
380	{
381		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
382
383		/* Check if it is a synchronized slot */
384		if (s->in_use && s->data.synced)
385		{
386			Assert(SlotIsLogical(s));
387			local_slots = lappend(local_slots, s);
388		}
389	}
390
391	LWLockRelease(ReplicationSlotControlLock);
392
393	return local_slots;
394}

396/*
397 * Helper function to check if local_slot is required to be retained.
398 *
399 * Return false either if local_slot does not exist in the remote_slots list
400 * or is invalidated while the corresponding remote slot is still valid,
401 * otherwise true.
402 */
403static bool
404local_sync_slot_required(ReplicationSlot *local_slot, List *remote_slots)
405{
406	bool		remote_exists = false;
407	bool		locally_invalidated = false;
408
409	foreach_ptr(RemoteSlot, remote_slot, remote_slots)
410	{
411		if (strcmp(remote_slot->name, NameStr(local_slot->data.name)) == 0)
412		{
413			remote_exists = true;
414
415			/*
416			 * If remote slot is not invalidated but local slot is marked as
417			 * invalidated, then set locally_invalidated flag.
418			 */
419			SpinLockAcquire(&local_slot->mutex);
420			locally_invalidated =
421				(remote_slot->invalidated == RS_INVAL_NONE) &&
422				(local_slot->data.invalidated != RS_INVAL_NONE);
423			SpinLockRelease(&local_slot->mutex);
424
425			break;
426		}
427	}
428
429	return (remote_exists && !locally_invalidated);
430}

432/*
433 * Drop local obsolete slots.
434 *
435 * Drop the local slots that no longer need to be synced i.e. these either do
436 * not exist on the primary or are no longer enabled for failover.
437 *
438 * Additionally, drop any slots that are valid on the primary but got
439 * invalidated on the standby. This situation may occur due to the following
440 * reasons:
441 * - The 'max_slot_wal_keep_size' on the standby is insufficient to retain WAL
442 *   records from the restart_lsn of the slot.
443 * - 'primary_slot_name' is temporarily reset to null and the physical slot is
444 *   removed.
445 * These dropped slots will get recreated in next sync-cycle and it is okay to
446 * drop and recreate such slots as long as these are not consumable on the
447 * standby (which is the case currently).
448 *
449 * Note: Change of 'wal_level' on the primary server to a level lower than
450 * logical may also result in slot invalidation and removal on the standby.
451 * This is because such 'wal_level' change is only possible if the logical
452 * slots are removed on the primary server, so it's expected to see the
453 * slots being invalidated and removed on the standby too (and re-created
454 * if they are re-created on the primary server).
455 */
456static void
457drop_local_obsolete_slots(List *remote_slot_list)
458{
459	List	   *local_slots = get_local_synced_slots();
460
461	foreach_ptr(ReplicationSlot, local_slot, local_slots)
462	{
463		/* Drop the local slot if it is not required to be retained. */
464		if (!local_sync_slot_required(local_slot, remote_slot_list))
465		{
466			bool		synced_slot;
467
468			/*
469			 * Use shared lock to prevent a conflict with
470			 * ReplicationSlotsDropDBSlots(), trying to drop the same slot
471			 * during a drop-database operation.
472			 */
473			LockSharedObject(DatabaseRelationId, local_slot->data.database,
474							 0, AccessShareLock);
475
476			/*
477			 * In the small window between getting the slot to drop and
478			 * locking the database, there is a possibility of a parallel
479			 * database drop by the startup process and the creation of a new
480			 * slot by the user. This new user-created slot may end up using
481			 * the same shared memory as that of 'local_slot'. Thus check if
482			 * local_slot is still the synced one before performing actual
483			 * drop.
484			 */
485			SpinLockAcquire(&local_slot->mutex);
486			synced_slot = local_slot->in_use && local_slot->data.synced;
487			SpinLockRelease(&local_slot->mutex);
488
489			if (synced_slot)
490			{
491				ReplicationSlotAcquire(NameStr(local_slot->data.name), true, false);
492				ReplicationSlotDropAcquired();
493			}
494
495			UnlockSharedObject(DatabaseRelationId, local_slot->data.database,
496							   0, AccessShareLock);
497
498			ereport(LOG,
499					errmsg("dropped replication slot \"%s\" of database with OID %u",
500						   NameStr(local_slot->data.name),
501						   local_slot->data.database));
502		}
503	}
504}

506/*
507 * Reserve WAL for the currently active local slot using the specified WAL
508 * location (restart_lsn).
509 *
510 * If the given WAL location has been removed or is at risk of removal,
511 * reserve WAL using the oldest segment that is non-removable.
512 */
513static void
514reserve_wal_for_local_slot(XLogRecPtr restart_lsn)
515{
516	XLogRecPtr	slot_min_lsn;
517	XLogRecPtr	min_safe_lsn;
518	XLogSegNo	segno;
519	ReplicationSlot *slot = MyReplicationSlot;
520
521	Assert(slot != NULL);
522	Assert(!XLogRecPtrIsValid(slot->data.restart_lsn));
523
524	/*
525	 * Acquire an exclusive lock to prevent the checkpoint process from
526	 * concurrently calculating the minimum slot LSN (see
527	 * CheckPointReplicationSlots), ensuring that if WAL reservation occurs
528	 * first, the checkpoint must wait for the restart_lsn update before
529	 * calculating the minimum LSN.
530	 *
531	 * Note: Unlike ReplicationSlotReserveWal(), this lock does not protect a
532	 * newly synced slot from being invalidated if a concurrent checkpoint has
533	 * invoked CheckPointReplicationSlots() before the WAL reservation here.
534	 * This can happen because the initial restart_lsn received from the
535	 * remote server can precede the redo pointer. Therefore, when selecting
536	 * the initial restart_lsn, we consider using the redo pointer or the
537	 * minimum slot LSN (if those values are greater than the remote
538	 * restart_lsn) instead of relying solely on the remote value.
539	 */
540	LWLockAcquire(ReplicationSlotAllocationLock, LW_EXCLUSIVE);
541
542	/*
543	 * Determine the minimum non-removable LSN by comparing the redo pointer
544	 * with the minimum slot LSN.
545	 *
546	 * The minimum slot LSN is considered because the redo pointer advances at
547	 * every checkpoint, even when replication slots are present on the
548	 * standby. In such scenarios, the redo pointer can exceed the remote
549	 * restart_lsn, while WALs preceding the remote restart_lsn remain
550	 * protected by a local replication slot.
551	 */
552	min_safe_lsn = GetRedoRecPtr();
553	slot_min_lsn = XLogGetReplicationSlotMinimumLSN();
554
555	if (XLogRecPtrIsValid(slot_min_lsn) && min_safe_lsn > slot_min_lsn)
556		min_safe_lsn = slot_min_lsn;
557
558	/*
559	 * If the minimum safe LSN is greater than the given restart_lsn, use it
560	 * as the initial restart_lsn for the newly synced slot. Otherwise, use
561	 * the given remote restart_lsn.
562	 */
563	SpinLockAcquire(&slot->mutex);
564	slot->data.restart_lsn = Max(restart_lsn, min_safe_lsn);
565	SpinLockRelease(&slot->mutex);
566
567	ReplicationSlotsComputeRequiredLSN();
568
569	XLByteToSeg(slot->data.restart_lsn, segno, wal_segment_size);
570	if (XLogGetLastRemovedSegno() >= segno)
571		elog(ERROR, "WAL required by replication slot %s has been removed concurrently",
572			 NameStr(slot->data.name));
573
574	LWLockRelease(ReplicationSlotAllocationLock);
575}

577/*
578 * If the remote restart_lsn and catalog_xmin have caught up with the
579 * local ones, then update the LSNs and persist the local synced slot for
580 * future synchronization; otherwise, do nothing.
581 *
582 * Return true if the slot is marked as RS_PERSISTENT (sync-ready), otherwise
583 * false.
584 */
585static bool
586update_and_persist_local_synced_slot(RemoteSlot *remote_slot, Oid remote_dbid)
587{
588	ReplicationSlot *slot = MyReplicationSlot;
589	bool		found_consistent_snapshot = false;
590	bool		remote_slot_precedes = false;
591
592	(void) update_local_synced_slot(remote_slot, remote_dbid,
593									&found_consistent_snapshot,
594									&remote_slot_precedes);
595
596	/*
597	 * Check if the primary server has caught up. Refer to the comment atop
598	 * the file for details on this check.
599	 */
600	if (remote_slot_precedes)
601	{
602		/*
603		 * The remote slot didn't catch up to locally reserved position.
604		 *
605		 * We do not drop the slot because the restart_lsn can be ahead of the
606		 * current location when recreating the slot in the next cycle. It may
607		 * take more time to create such a slot. Therefore, we keep this slot
608		 * and attempt the synchronization in the next cycle.
609		 */
610		return false;
611	}
612
613	/*
614	 * Don't persist the slot if it cannot reach the consistent point from the
615	 * restart_lsn. See comments atop this file.
616	 */
617	if (!found_consistent_snapshot)
618	{
619		ereport(LOG,
620				errmsg("could not synchronize replication slot \"%s\"", remote_slot->name),
621				errdetail("Synchronization could lead to data loss, because the standby could not build a consistent snapshot to decode WALs at LSN %X/%X.",
622						  LSN_FORMAT_ARGS(slot->data.restart_lsn)));
623
624		return false;
625	}
626
627	ReplicationSlotPersist();
628
629	ereport(LOG,
630			errmsg("newly created replication slot \"%s\" is sync-ready now",
631				   remote_slot->name));
632
633	return true;

636/*
637 * Synchronize a single slot to the given position.
638 *
639 * This creates a new slot if there is no existing one and updates the
640 * metadata of the slot as per the data received from the primary server.
641 *
642 * The slot is created as a temporary slot and stays in the same state until the
643 * remote_slot catches up with locally reserved position and local slot is
644 * updated. The slot is then persisted and is considered as sync-ready for
645 * periodic syncs.
646 *
647 * Returns TRUE if the local slot is updated.
648 */
649static bool
650synchronize_one_slot(RemoteSlot *remote_slot, Oid remote_dbid)
651{
652	ReplicationSlot *slot;
653	XLogRecPtr	latestFlushPtr;
654	bool		slot_updated = false;
655
656	/*
657	 * Make sure that concerned WAL is received and flushed before syncing
658	 * slot to target lsn received from the primary server.
659	 */
660	latestFlushPtr = GetStandbyFlushRecPtr(NULL);
661	if (remote_slot->confirmed_lsn > latestFlushPtr)
662	{
663		/*
664		 * Can get here only if GUC 'synchronized_standby_slots' on the
665		 * primary server was not configured correctly.
666		 */
667		ereport(AmLogicalSlotSyncWorkerProcess() ? LOG : ERROR,
668				errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
669				errmsg("skipping slot synchronization because the received slot sync"
670					   " LSN %X/%X for slot \"%s\" is ahead of the standby position %X/%X",
671					   LSN_FORMAT_ARGS(remote_slot->confirmed_lsn),
672					   remote_slot->name,
673					   LSN_FORMAT_ARGS(latestFlushPtr)));
674
675		return false;
676	}
677
678	/* Search for the named slot */
679	if ((slot = SearchNamedReplicationSlot(remote_slot->name, true)))
680	{
681		bool		synced;
682
683		SpinLockAcquire(&slot->mutex);
684		synced = slot->data.synced;
685		SpinLockRelease(&slot->mutex);
686
687		/* User-created slot with the same name exists, raise ERROR. */
688		if (!synced)
689			ereport(ERROR,
690					errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
691					errmsg("exiting from slot synchronization because same"
692						   " name slot \"%s\" already exists on the standby",
693						   remote_slot->name));
694
695		/*
696		 * The slot has been synchronized before.
697		 *
698		 * It is important to acquire the slot here before checking
699		 * invalidation. If we don't acquire the slot first, there could be a
700		 * race condition that the local slot could be invalidated just after
701		 * checking the 'invalidated' flag here and we could end up
702		 * overwriting 'invalidated' flag to remote_slot's value. See
703		 * InvalidatePossiblyObsoleteSlot() where it invalidates slot directly
704		 * if the slot is not acquired by other processes.
705		 *
706		 * XXX: If it ever turns out that slot acquire/release is costly for
707		 * cases when none of the slot properties is changed then we can do a
708		 * pre-check to ensure that at least one of the slot properties is
709		 * changed before acquiring the slot.
710		 */
711		ReplicationSlotAcquire(remote_slot->name, true, false);
712
713		Assert(slot == MyReplicationSlot);
714
715		/*
716		 * Copy the invalidation cause from remote only if local slot is not
717		 * invalidated locally, we don't want to overwrite existing one.
718		 */
719		if (slot->data.invalidated == RS_INVAL_NONE &&
720			remote_slot->invalidated != RS_INVAL_NONE)
721		{
722			SpinLockAcquire(&slot->mutex);
723			slot->data.invalidated = remote_slot->invalidated;
724			SpinLockRelease(&slot->mutex);
725
726			/* Make sure the invalidated state persists across server restart */
727			ReplicationSlotMarkDirty();
728			ReplicationSlotSave();
729
730			slot_updated = true;
731		}
732
733		/* Skip the sync of an invalidated slot */
734		if (slot->data.invalidated != RS_INVAL_NONE)
735		{
736			ReplicationSlotRelease();
737			return slot_updated;
738		}
739
740		/* Slot not ready yet, let's attempt to make it sync-ready now. */
741		if (slot->data.persistency == RS_TEMPORARY)
742		{
743			slot_updated = update_and_persist_local_synced_slot(remote_slot,
744																remote_dbid);
745		}
746
747		/* Slot ready for sync, so sync it. */
748		else
749		{
750			/*
751			 * Sanity check: As long as the invalidations are handled
752			 * appropriately as above, this should never happen.
753			 *
754			 * We don't need to check restart_lsn here. See the comments in
755			 * update_local_synced_slot() for details.
756			 */
757			if (remote_slot->confirmed_lsn < slot->data.confirmed_flush)
758				ereport(ERROR,
759						errmsg_internal("cannot synchronize local slot \"%s\"",
760										remote_slot->name),
761						errdetail_internal("Local slot's start streaming location LSN(%X/%X) is ahead of remote slot's LSN(%X/%X).",
762										   LSN_FORMAT_ARGS(slot->data.confirmed_flush),
763										   LSN_FORMAT_ARGS(remote_slot->confirmed_lsn)));
764
765			slot_updated = update_local_synced_slot(remote_slot, remote_dbid,
766													NULL, NULL);
767		}
768	}
769	/* Otherwise create the slot first. */
770	else
771	{
772		NameData	plugin_name;
773		TransactionId xmin_horizon = InvalidTransactionId;
774
775		/* Skip creating the local slot if remote_slot is invalidated already */
776		if (remote_slot->invalidated != RS_INVAL_NONE)
777			return false;
778
779		/*
780		 * We create temporary slots instead of ephemeral slots here because
781		 * we want the slots to survive after releasing them. This is done to
782		 * avoid dropping and re-creating the slots in each synchronization
783		 * cycle if the restart_lsn or catalog_xmin of the remote slot has not
784		 * caught up.
785		 */
786		ReplicationSlotCreate(remote_slot->name, true, RS_TEMPORARY,
787							  remote_slot->two_phase,
788							  remote_slot->failover,
789							  true);
790
791		/* For shorter lines. */
792		slot = MyReplicationSlot;
793
794		/* Avoid expensive operations while holding a spinlock. */
795		namestrcpy(&plugin_name, remote_slot->plugin);
796
797		SpinLockAcquire(&slot->mutex);
798		slot->data.database = remote_dbid;
799		slot->data.plugin = plugin_name;
800		SpinLockRelease(&slot->mutex);
801
802		reserve_wal_for_local_slot(remote_slot->restart_lsn);
803
804		LWLockAcquire(ReplicationSlotControlLock, LW_EXCLUSIVE);
805		LWLockAcquire(ProcArrayLock, LW_EXCLUSIVE);
806		xmin_horizon = GetOldestSafeDecodingTransactionId(true);
807		SpinLockAcquire(&slot->mutex);
808		slot->effective_catalog_xmin = xmin_horizon;
809		slot->data.catalog_xmin = xmin_horizon;
810		SpinLockRelease(&slot->mutex);
811		ReplicationSlotsComputeRequiredXmin(true);
812		LWLockRelease(ProcArrayLock);
813		LWLockRelease(ReplicationSlotControlLock);
814
815		update_and_persist_local_synced_slot(remote_slot, remote_dbid);
816
817		slot_updated = true;
818	}
819
820	ReplicationSlotRelease();
821
822	return slot_updated;
823}

825/*
826 * Synchronize slots.
827 *
828 * Gets the failover logical slots info from the primary server and updates
829 * the slots locally. Creates the slots if not present on the standby.
830 *
831 * Returns TRUE if any of the slots gets updated in this sync-cycle.
832 */
833static bool
834synchronize_slots(WalReceiverConn *wrconn)
835{
836#define SLOTSYNC_COLUMN_COUNT 10
837	Oid			slotRow[SLOTSYNC_COLUMN_COUNT] = {TEXTOID, TEXTOID, LSNOID,
838	LSNOID, XIDOID, BOOLOID, LSNOID, BOOLOID, TEXTOID, TEXTOID};
839
840	WalRcvExecResult *res;
841	TupleTableSlot *tupslot;
842	List	   *remote_slot_list = NIL;
843	bool		some_slot_updated = false;
844	bool		started_tx = false;
845	const char *query = "SELECT slot_name, plugin, confirmed_flush_lsn,"
846		" restart_lsn, catalog_xmin, two_phase, two_phase_at, failover,"
847		" database, invalidation_reason"
848		" FROM pg_catalog.pg_replication_slots"
849		" WHERE failover and NOT temporary";
850
851	/* The syscache access in walrcv_exec() needs a transaction env. */
852	if (!IsTransactionState())
853	{
854		StartTransactionCommand();
855		started_tx = true;
856	}
857
858	/* Execute the query */
859	res = walrcv_exec(wrconn, query, SLOTSYNC_COLUMN_COUNT, slotRow);
860	if (res->status != WALRCV_OK_TUPLES)
861		ereport(ERROR,
862				errmsg("could not fetch failover logical slots info from the primary server: %s",
863					   res->err));
864
865	/* Construct the remote_slot tuple and synchronize each slot locally */
866	tupslot = MakeSingleTupleTableSlot(res->tupledesc, &TTSOpsMinimalTuple);
867	while (tuplestore_gettupleslot(res->tuplestore, true, false, tupslot))
868	{
869		bool		isnull;
870		RemoteSlot *remote_slot = palloc0(sizeof(RemoteSlot));
871		Datum		d;
872		int			col = 0;
873
874		remote_slot->name = TextDatumGetCString(slot_getattr(tupslot, ++col,
875															 &isnull));
876		Assert(!isnull);
877
878		remote_slot->plugin = TextDatumGetCString(slot_getattr(tupslot, ++col,
879															   &isnull));
880		Assert(!isnull);
881
882		/*
883		 * It is possible to get null values for LSN and Xmin if slot is
884		 * invalidated on the primary server, so handle accordingly.
885		 */
886		d = slot_getattr(tupslot, ++col, &isnull);
887		remote_slot->confirmed_lsn = isnull ? InvalidXLogRecPtr :
888			DatumGetLSN(d);
889
890		d = slot_getattr(tupslot, ++col, &isnull);
891		remote_slot->restart_lsn = isnull ? InvalidXLogRecPtr : DatumGetLSN(d);
892
893		d = slot_getattr(tupslot, ++col, &isnull);
894		remote_slot->catalog_xmin = isnull ? InvalidTransactionId :
895			DatumGetTransactionId(d);
896
897		remote_slot->two_phase = DatumGetBool(slot_getattr(tupslot, ++col,
898														   &isnull));
899		Assert(!isnull);
900
901		d = slot_getattr(tupslot, ++col, &isnull);
902		remote_slot->two_phase_at = isnull ? InvalidXLogRecPtr : DatumGetLSN(d);
903
904		remote_slot->failover = DatumGetBool(slot_getattr(tupslot, ++col,
905														  &isnull));
906		Assert(!isnull);
907
908		remote_slot->database = TextDatumGetCString(slot_getattr(tupslot,
909																 ++col, &isnull));
910		Assert(!isnull);
911
912		d = slot_getattr(tupslot, ++col, &isnull);
913		remote_slot->invalidated = isnull ? RS_INVAL_NONE :
914			GetSlotInvalidationCause(TextDatumGetCString(d));
915
916		/* Sanity check */
917		Assert(col == SLOTSYNC_COLUMN_COUNT);
918
919		/*
920		 * If restart_lsn, confirmed_lsn or catalog_xmin is invalid but the
921		 * slot is valid, that means we have fetched the remote_slot in its
922		 * RS_EPHEMERAL state. In such a case, don't sync it; we can always
923		 * sync it in the next sync cycle when the remote_slot is persisted
924		 * and has valid lsn(s) and xmin values.
925		 *
926		 * XXX: In future, if we plan to expose 'slot->data.persistency' in
927		 * pg_replication_slots view, then we can avoid fetching RS_EPHEMERAL
928		 * slots in the first place.
929		 */
930		if ((XLogRecPtrIsInvalid(remote_slot->restart_lsn) ||
931			 XLogRecPtrIsInvalid(remote_slot->confirmed_lsn) ||
932			 !TransactionIdIsValid(remote_slot->catalog_xmin)) &&
933			remote_slot->invalidated == RS_INVAL_NONE)
934			pfree(remote_slot);
935		else
936			/* Create list of remote slots */
937			remote_slot_list = lappend(remote_slot_list, remote_slot);
938
939		ExecClearTuple(tupslot);
940	}
941
942	/* Drop local slots that no longer need to be synced. */
943	drop_local_obsolete_slots(remote_slot_list);
944
945	/* Now sync the slots locally */
946	foreach_ptr(RemoteSlot, remote_slot, remote_slot_list)
947	{
948		Oid			remote_dbid = get_database_oid(remote_slot->database, false);
949
950		/*
951		 * Use shared lock to prevent a conflict with
952		 * ReplicationSlotsDropDBSlots(), trying to drop the same slot during
953		 * a drop-database operation.
954		 */
955		LockSharedObject(DatabaseRelationId, remote_dbid, 0, AccessShareLock);
956
957		some_slot_updated |= synchronize_one_slot(remote_slot, remote_dbid);
958
959		UnlockSharedObject(DatabaseRelationId, remote_dbid, 0, AccessShareLock);
960	}
961
962	/* We are done, free remote_slot_list elements */
963	list_free_deep(remote_slot_list);
964
965	walrcv_clear_result(res);
966
967	if (started_tx)
968		CommitTransactionCommand();
969
970	return some_slot_updated;
971}

973/*
974 * Checks the remote server info.
975 *
976 * We ensure that the 'primary_slot_name' exists on the remote server and the
977 * remote server is not a standby node.
978 */
979static void
980validate_remote_info(WalReceiverConn *wrconn)
981{
982#define PRIMARY_INFO_OUTPUT_COL_COUNT 2
983	WalRcvExecResult *res;
984	Oid			slotRow[PRIMARY_INFO_OUTPUT_COL_COUNT] = {BOOLOID, BOOLOID};
985	StringInfoData cmd;
986	bool		isnull;
987	TupleTableSlot *tupslot;
988	bool		remote_in_recovery;
989	bool		primary_slot_valid;
990	bool		started_tx = false;
991
992	initStringInfo(&cmd);
993	appendStringInfo(&cmd,
994					 "SELECT pg_is_in_recovery(), count(*) = 1"
995					 " FROM pg_catalog.pg_replication_slots"
996					 " WHERE slot_type='physical' AND slot_name=%s",
997					 quote_literal_cstr(PrimarySlotName));
998
999	/* The syscache access in walrcv_exec() needs a transaction env. */
1000	if (!IsTransactionState())
1001	{
1002		StartTransactionCommand();
1003		started_tx = true;
1004	}
1005
1006	res = walrcv_exec(wrconn, cmd.data, PRIMARY_INFO_OUTPUT_COL_COUNT, slotRow);
1007	pfree(cmd.data);
1008
1009	if (res->status != WALRCV_OK_TUPLES)
1010		ereport(ERROR,
1011				errmsg("could not fetch primary slot name \"%s\" info from the primary server: %s",
1012					   PrimarySlotName, res->err),
1013				errhint("Check if \"primary_slot_name\" is configured correctly."));
1014
1015	tupslot = MakeSingleTupleTableSlot(res->tupledesc, &TTSOpsMinimalTuple);
1016	if (!tuplestore_gettupleslot(res->tuplestore, true, false, tupslot))
1017		elog(ERROR,
1018			 "failed to fetch tuple for the primary server slot specified by \"primary_slot_name\"");
1019
1020	remote_in_recovery = DatumGetBool(slot_getattr(tupslot, 1, &isnull));
1021	Assert(!isnull);
1022
1023	/*
1024	 * Slot sync is currently not supported on a cascading standby. This is
1025	 * because if we allow it, the primary server needs to wait for all the
1026	 * cascading standbys, otherwise, logical subscribers can still be ahead
1027	 * of one of the cascading standbys which we plan to promote. Thus, to
1028	 * avoid this additional complexity, we restrict it for the time being.
1029	 */
1030	if (remote_in_recovery)
1031		ereport(ERROR,
1032				errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
1033				errmsg("cannot synchronize replication slots from a standby server"));
1034
1035	primary_slot_valid = DatumGetBool(slot_getattr(tupslot, 2, &isnull));
1036	Assert(!isnull);
1037
1038	if (!primary_slot_valid)
1039		ereport(ERROR,
1040				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1041		/* translator: second %s is a GUC variable name */
1042				errmsg("replication slot \"%s\" specified by \"%s\" does not exist on primary server",
1043					   PrimarySlotName, "primary_slot_name"));
1044
1045	ExecClearTuple(tupslot);
1046	walrcv_clear_result(res);
1047
1048	if (started_tx)
1049		CommitTransactionCommand();
1050}

1080/*
1081 * Return true if all necessary GUCs for slot synchronization are set
1082 * appropriately, otherwise, return false.
1083 */
1084bool
1085ValidateSlotSyncParams(int elevel)
1086{
1087	/*
1088	 * Logical slot sync/creation requires wal_level >= logical.
1089	 */
1090	if (wal_level < WAL_LEVEL_LOGICAL)
1091	{
1092		ereport(elevel,
1093				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1094				errmsg("replication slot synchronization requires \"wal_level\" >= \"logical\""));
1095		return false;
1096	}
1097
1098	/*
1099	 * A physical replication slot(primary_slot_name) is required on the
1100	 * primary to ensure that the rows needed by the standby are not removed
1101	 * after restarting, so that the synchronized slot on the standby will not
1102	 * be invalidated.
1103	 */
1104	if (PrimarySlotName == NULL || *PrimarySlotName == '\0')
1105	{
1106		ereport(elevel,
1107				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1108		/* translator: %s is a GUC variable name */
1109				errmsg("replication slot synchronization requires \"%s\" to be set", "primary_slot_name"));
1110		return false;
1111	}
1112
1113	/*
1114	 * hot_standby_feedback must be enabled to cooperate with the physical
1115	 * replication slot, which allows informing the primary about the xmin and
1116	 * catalog_xmin values on the standby.
1117	 */
1118	if (!hot_standby_feedback)
1119	{
1120		ereport(elevel,
1121				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1122		/* translator: %s is a GUC variable name */
1123				errmsg("replication slot synchronization requires \"%s\" to be enabled",
1124					   "hot_standby_feedback"));
1125		return false;
1126	}
1127
1128	/*
1129	 * The primary_conninfo is required to make connection to primary for
1130	 * getting slots information.
1131	 */
1132	if (PrimaryConnInfo == NULL || *PrimaryConnInfo == '\0')
1133	{
1134		ereport(elevel,
1135				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1136		/* translator: %s is a GUC variable name */
1137				errmsg("replication slot synchronization requires \"%s\" to be set",
1138					   "primary_conninfo"));
1139		return false;
1140	}
1141
1142	return true;
1143}

1147 *
1148 * Exit or throw error if relevant GUCs have changed depending on whether
1149 * called from slot sync worker or from the SQL function pg_sync_replication_slots()
1150 */
1151static void
1152slotsync_reread_config(void)
1153{
1154	char	   *old_primary_conninfo = pstrdup(PrimaryConnInfo);
1155	char	   *old_primary_slotname = pstrdup(PrimarySlotName);
1156	bool		old_sync_replication_slots = sync_replication_slots;
1157	bool		old_hot_standby_feedback = hot_standby_feedback;
1158	bool		conninfo_changed;
1159	bool		primary_slotname_changed;
1160	bool		is_slotsync_worker = AmLogicalSlotSyncWorkerProcess();
1161	bool		parameter_changed = false;
1162
1163	if (is_slotsync_worker)
1164		Assert(sync_replication_slots);

1168
1169	conninfo_changed = strcmp(old_primary_conninfo, PrimaryConnInfo) != 0;
1170	primary_slotname_changed = strcmp(old_primary_slotname, PrimarySlotName) != 0;
1171	pfree(old_primary_conninfo);
1172	pfree(old_primary_slotname);
1173
1174	if (old_sync_replication_slots != sync_replication_slots)
1175	{
1176		if (is_slotsync_worker)
1177		{
1178			ereport(LOG,
1179			/* translator: %s is a GUC variable name */
1180					errmsg("replication slot synchronization worker will stop because \"%s\" is disabled",
1181						   "sync_replication_slots"));
1182
1183			proc_exit(0);
1184		}
1185
1186		parameter_changed = true;
1187	}
1188	else
1189	{
1190		if (conninfo_changed ||
1191			primary_slotname_changed ||
1192			(old_hot_standby_feedback != hot_standby_feedback))
1193		{
1194
1195			if (is_slotsync_worker)
1196			{
1197				ereport(LOG,
1198						errmsg("replication slot synchronization worker will restart because of a parameter change"));
1199
1200				/*
1201				 * Reset the last-start time for this worker so that the
1202				 * postmaster can restart it without waiting for
1203				 * SLOTSYNC_RESTART_INTERVAL_SEC.
1204				 */
1205				SlotSyncCtx->last_start_time = 0;
1206
1207				proc_exit(0);
1208			}
1209
1210			parameter_changed = true;
1211		}
1212	}
1213
1214	/*
1215	 * If we have reached here with a parameter change, we must be running in
1216	 * SQL function, emit error in such a case.
1217	 */
1218	if (parameter_changed)
1219	{
1220		Assert(!is_slotsync_worker);
1221		ereport(ERROR,
1222				errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1223				errmsg("replication slot synchronization will stop because of a parameter change"));
1224	}
1225
1226}

1243/*
1244 * Handle a PROCSIG_SLOTSYNC_MESSAGE signal, called from ProcessInterrupts().
1245 *
1246 * If the current process is the slotsync background worker, log a message
1247 * and exit cleanly.  If it is a backend executing pg_sync_replication_slots(),
1248 * raise an error, unless the sync has already finished, in which case there
1249 * is no need to interrupt the caller.
1250 */
1251void
1252ProcessSlotSyncMessage(void)
1253{
1254	SlotSyncShutdownPending = false;
1255
1256	if (AmLogicalSlotSyncWorkerProcess())
1257	{
1258		ereport(LOG,
1259				errmsg("replication slot synchronization worker will stop because promotion is triggered"));
1260		proc_exit(0);
1261	}
1262	else
1263	{
1264		/*
1265		 * If sync has already completed, there is no need to interrupt the
1266		 * caller with an error.
1267		 */
1268		if (!IsSyncingReplicationSlots())
1269			return;

1271		ereport(ERROR,
1272				errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
1273				errmsg("replication slot synchronization will stop because promotion is triggered"));
1274	}
1275}

1290/*
1291 * Cleanup function for slotsync worker.
1292 *
1293 * Called on slotsync worker exit.
1294 */
1295static void
1296slotsync_worker_onexit(int code, Datum arg)
1297{
1298	/*
1299	 * We need to do slots cleanup here just like WalSndErrorCleanup() does.
1300	 *
1301	 * The startup process during promotion invokes ShutDownSlotSync() which
1302	 * waits for slot sync to finish and it does that by checking the
1303	 * 'syncing' flag. Thus the slot sync worker must be done with slots'
1304	 * release and cleanup to avoid any dangling temporary slots or active
1305	 * slots before it marks itself as finished syncing.
1306	 */
1307
1308	/* Make sure active replication slots are released */
1309	if (MyReplicationSlot != NULL)
1310		ReplicationSlotRelease();
1311
1312	/* Also cleanup the temporary slots. */
1313	ReplicationSlotCleanup(false);
1314
1315	SpinLockAcquire(&SlotSyncCtx->mutex);
1316
1317	SlotSyncCtx->pid = InvalidPid;
1318
1319	/*
1320	 * If syncing_slots is true, it indicates that the process errored out
1321	 * without resetting the flag. So, we need to clean up shared memory and
1322	 * reset the flag here.
1323	 */
1324	if (syncing_slots)
1325	{
1326		SlotSyncCtx->syncing = false;
1327		syncing_slots = false;
1328	}
1329
1330	SpinLockRelease(&SlotSyncCtx->mutex);
1331}

1333/*
1334 * Sleep for long enough that we believe it's likely that the slots on primary
1335 * get updated.
1336 *
1337 * If there is no slot activity the wait time between sync-cycles will double
1338 * (to a maximum of 30s). If there is some slot activity the wait time between
1339 * sync-cycles is reset to the minimum (200ms).
1340 */
1341static void
1342wait_for_slot_activity(bool some_slot_updated)
1343{
1344	int			rc;
1345
1346	if (!some_slot_updated)
1347	{
1348		/*
1349		 * No slots were updated, so double the sleep time, but not beyond the
1350		 * maximum allowable value.
1351		 */
1352		sleep_ms = Min(sleep_ms * 2, MAX_SLOTSYNC_WORKER_NAPTIME_MS);
1353	}
1354	else
1355	{
1356		/*
1357		 * Some slots were updated since the last sleep, so reset the sleep
1358		 * time.
1359		 */
1360		sleep_ms = MIN_SLOTSYNC_WORKER_NAPTIME_MS;
1361	}
1362
1363	rc = WaitLatch(MyLatch,
1364				   WL_LATCH_SET | WL_TIMEOUT | WL_EXIT_ON_PM_DEATH,
1365				   sleep_ms,
1366				   WAIT_EVENT_REPLICATION_SLOTSYNC_MAIN);
1367
1368	if (rc & WL_LATCH_SET)
1369		ResetLatch(MyLatch);
1370}

1372/*
1373 * Emit an error if a concurrent sync call is in progress.
1374 * Otherwise, advertise that a sync is in progress.
1375 */
1376static void
1377check_and_set_sync_info(pid_t sync_process_pid)
1378{
1379	SpinLockAcquire(&SlotSyncCtx->mutex);
1380
1381	/*
1382	 * Exit immediately if promotion has been triggered.  This guards against
1383	 * a new worker (or a call to pg_sync_replication_slots()) that starts
1384	 * after the old worker was stopped by ShutDownSlotSync().
1385	 */
1386	if (SlotSyncCtx->stopSignaled)
1387	{
1388		SpinLockRelease(&SlotSyncCtx->mutex);
1389
1390		if (AmLogicalSlotSyncWorkerProcess())
1391		{
1392			ereport(DEBUG1,
1393					errmsg("replication slot synchronization worker will not start because promotion was triggered"));
1394
1395			proc_exit(0);
1396		}
1397		else
1398		{
1399			/*
1400			 * For the backend executing SQL function
1401			 * pg_sync_replication_slots().
1402			 */
1403			ereport(ERROR,
1404					errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
1405					errmsg("replication slot synchronization will not start because promotion was triggered"));
1406		}
1407	}
1408
1409	if (SlotSyncCtx->syncing)
1410	{
1411		SpinLockRelease(&SlotSyncCtx->mutex);
1412		ereport(ERROR,
1413				errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
1414				errmsg("cannot synchronize replication slots concurrently"));
1415	}
1416
1417	/* The pid must not be already assigned in SlotSyncCtx */
1418	Assert(SlotSyncCtx->pid == InvalidPid);
1419
1420	SlotSyncCtx->syncing = true;
1421
1422	/*
1423	 * Advertise the required PID so that the startup process can kill the
1424	 * slot sync process on promotion.
1425	 */
1426	SlotSyncCtx->pid = sync_process_pid;
1427
1428	SpinLockRelease(&SlotSyncCtx->mutex);
1429
1430	syncing_slots = true;
1431}

1447/*
1448 * The main loop of our worker process.
1449 *
1450 * It connects to the primary server, fetches logical failover slots
1451 * information periodically in order to create and sync the slots.
1452 */
1453void
1454ReplSlotSyncWorkerMain(const void *startup_data, size_t startup_data_len)
1455{
1456	WalReceiverConn *wrconn = NULL;
1457	char	   *dbname;
1458	char	   *err;
1459	sigjmp_buf	local_sigjmp_buf;
1460	StringInfoData app_name;
1461
1462	Assert(startup_data_len == 0);
1463
1464	MyBackendType = B_SLOTSYNC_WORKER;
1465
1466	init_ps_display(NULL);
1467
1468	Assert(GetProcessingMode() == InitProcessing);
1469
1470	/*
1471	 * Create a per-backend PGPROC struct in shared memory.  We must do this
1472	 * before we access any shared memory.
1473	 */
1474	InitProcess();
1475
1476	/*
1477	 * Early initialization.
1478	 */
1479	BaseInit();
1480
1481	Assert(SlotSyncCtx != NULL);
1482
1483	/*
1484	 * If an exception is encountered, processing resumes here.
1485	 *
1486	 * We just need to clean up, report the error, and go away.
1487	 *
1488	 * If we do not have this handling here, then since this worker process
1489	 * operates at the bottom of the exception stack, ERRORs turn into FATALs.
1490	 * Therefore, we create our own exception handler to catch ERRORs.
1491	 */
1492	if (sigsetjmp(local_sigjmp_buf, 1) != 0)
1493	{
1494		/* since not using PG_TRY, must reset error stack by hand */
1495		error_context_stack = NULL;
1496
1497		/* Prevents interrupts while cleaning up */
1498		HOLD_INTERRUPTS();
1499
1500		/* Report the error to the server log */
1501		EmitErrorReport();
1502
1503		/*
1504		 * We can now go away.  Note that because we called InitProcess, a
1505		 * callback was registered to do ProcKill, which will clean up
1506		 * necessary state.
1507		 */
1508		proc_exit(0);
1509	}
1510
1511	/* We can now handle ereport(ERROR) */
1512	PG_exception_stack = &local_sigjmp_buf;
1513
1514	/* Setup signal handling */
1515	pqsignal(SIGHUP, SignalHandlerForConfigReload);
1516	pqsignal(SIGINT, StatementCancelHandler);
1517	pqsignal(SIGTERM, die);
1518	pqsignal(SIGFPE, FloatExceptionHandler);
1519	pqsignal(SIGUSR1, procsignal_sigusr1_handler);
1520	pqsignal(SIGUSR2, SIG_IGN);
1521	pqsignal(SIGPIPE, SIG_IGN);
1522	pqsignal(SIGCHLD, SIG_DFL);
1523
1524	check_and_set_sync_info(MyProcPid);
1525
1526	ereport(LOG, errmsg("slot sync worker started"));
1527
1528	/* Register it as soon as SlotSyncCtx->pid is initialized. */
1529	before_shmem_exit(slotsync_worker_onexit, (Datum) 0);
1530
1531	/*
1532	 * Establishes SIGALRM handler and initialize timeout module. It is needed
1533	 * by InitPostgres to register different timeouts.
1534	 */
1535	InitializeTimeouts();
1536
1537	/* Load the libpq-specific functions */
1538	load_file("libpqwalreceiver", false);
1539
1540	/*
1541	 * Unblock signals (they were blocked when the postmaster forked us)
1542	 */
1543	sigprocmask(SIG_SETMASK, &UnBlockSig, NULL);
1544
1545	/*
1546	 * Set always-secure search path, so malicious users can't redirect user
1547	 * code (e.g. operators).
1548	 *
1549	 * It's not strictly necessary since we won't be scanning or writing to
1550	 * any user table locally, but it's good to retain it here for added
1551	 * precaution.
1552	 */
1553	SetConfigOption("search_path", "", PGC_SUSET, PGC_S_OVERRIDE);
1554
1555	dbname = CheckAndGetDbnameFromConninfo();
1556
1557	/*
1558	 * Connect to the database specified by the user in primary_conninfo. We
1559	 * need a database connection for walrcv_exec to work which we use to
1560	 * fetch slot information from the remote node. See comments atop
1561	 * libpqrcv_exec.
1562	 *
1563	 * We do not specify a specific user here since the slot sync worker will
1564	 * operate as a superuser. This is safe because the slot sync worker does
1565	 * not interact with user tables, eliminating the risk of executing
1566	 * arbitrary code within triggers.
1567	 */
1568	InitPostgres(dbname, InvalidOid, NULL, InvalidOid, 0, NULL);
1569
1570	SetProcessingMode(NormalProcessing);
1571
1572	initStringInfo(&app_name);
1573	if (cluster_name[0])
1574		appendStringInfo(&app_name, "%s_%s", cluster_name, "slotsync worker");
1575	else
1576		appendStringInfoString(&app_name, "slotsync worker");
1577
1578	/*
1579	 * Establish the connection to the primary server for slot
1580	 * synchronization.
1581	 */
1582	wrconn = walrcv_connect(PrimaryConnInfo, false, false, false,
1583							app_name.data, &err);
1584
1585	if (!wrconn)
1586		ereport(ERROR,
1587				errcode(ERRCODE_CONNECTION_FAILURE),
1588				errmsg("synchronization worker \"%s\" could not connect to the primary server: %s",
1589					   app_name.data, err));
1590
1591	pfree(app_name.data);
1592
1593	/*
1594	 * Register the disconnection callback.
1595	 *
1596	 * XXX: This can be combined with previous cleanup registration of
1597	 * slotsync_worker_onexit() but that will need the connection to be made
1598	 * global and we want to avoid introducing global for this purpose.
1599	 */
1600	before_shmem_exit(slotsync_worker_disconnect, PointerGetDatum(wrconn));
1601
1602	/*
1603	 * Using the specified primary server connection, check that we are not a
1604	 * cascading standby and slot configured in 'primary_slot_name' exists on
1605	 * the primary server.
1606	 */
1607	validate_remote_info(wrconn);
1608
1609	/* Main loop to synchronize slots */
1610	for (;;)
1611	{
1612		bool		some_slot_updated = false;
1613
1614		CHECK_FOR_INTERRUPTS();
1615
1616		if (ConfigReloadPending)
1617			slotsync_reread_config();
1618
1619		some_slot_updated = synchronize_slots(wrconn);
1620
1621		wait_for_slot_activity(some_slot_updated);
1622	}
1623
1624	/*
1625	 * The slot sync worker can't get here because it will only stop when it
1626	 * receives a stop request from the startup process, or when there is an
1627	 * error.
1628	 */
1629	Assert(false);
1630}

1632/*
1633 * Update the inactive_since property for synced slots.
1634 *
1635 * Note that this function is currently called when we shutdown the slot
1636 * sync machinery.
1637 */
1638static void
1639update_synced_slots_inactive_since(void)
1640{
1641	TimestampTz now = 0;
1642
1643	/*
1644	 * We need to update inactive_since only when we are promoting standby to
1645	 * correctly interpret the inactive_since if the standby gets promoted
1646	 * without a restart. We don't want the slots to appear inactive for a
1647	 * long time after promotion if they haven't been synchronized recently.
1648	 * Whoever acquires the slot, i.e., makes the slot active, will reset it.
1649	 */
1650	if (!StandbyMode)
1651		return;
1652
1653	/* The slot sync worker or the SQL function mustn't be running by now */
1654	Assert((SlotSyncCtx->pid == InvalidPid) && !SlotSyncCtx->syncing);
1655
1656	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1657
1658	for (int i = 0; i < max_replication_slots; i++)
1659	{
1660		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
1661
1662		/* Check if it is a synchronized slot */
1663		if (s->in_use && s->data.synced)
1664		{
1665			Assert(SlotIsLogical(s));
1666
1667			/* The slot must not be acquired by any process */
1668			Assert(s->active_pid == 0);
1669
1670			/* Use the same inactive_since time for all the slots. */
1671			if (now == 0)
1672				now = GetCurrentTimestamp();
1673
1674			ReplicationSlotSetInactiveSince(s, now, true);
1675		}
1676	}
1677
1678	LWLockRelease(ReplicationSlotControlLock);
1679}

1681/*
1682 * Shut down slot synchronization.
1683 *
1684 * This function sets stopSignaled=true and wakes up the slot sync process
1685 * (either worker or backend running the SQL function pg_sync_replication_slots())
1686 * so that worker can exit or the SQL function pg_sync_replication_slots() can
1687 * finish. It also waits till the slot sync worker has exited or
1688 * pg_sync_replication_slots() has finished.
1689 */
1690void
1691ShutDownSlotSync(void)
1692{
1693	pid_t		sync_process_pid;
1694
1695	SpinLockAcquire(&SlotSyncCtx->mutex);
1696
1697	SlotSyncCtx->stopSignaled = true;
1698
1699	/*
1700	 * Return if neither the slot sync worker is running nor the function
1701	 * pg_sync_replication_slots() is executing.
1702	 */
1703	if (!SlotSyncCtx->syncing)
1704	{
1705		SpinLockRelease(&SlotSyncCtx->mutex);
1706		update_synced_slots_inactive_since();
1707		return;
1708	}
1709
1710	sync_process_pid = SlotSyncCtx->pid;
1711
1712	SpinLockRelease(&SlotSyncCtx->mutex);
1713
1714	/*
1715	 * Signal process doing slotsync, if any, asking it to stop.
1716	 */
1717	if (sync_process_pid != InvalidPid)
1718		SendProcSignal(sync_process_pid, PROCSIG_SLOTSYNC_MESSAGE,
1719					   INVALID_PROC_NUMBER);
1720
1721	/* Wait for slot sync to end */
1722	for (;;)
1723	{
1724		int			rc;
1725
1726		/* Wait a bit, we don't expect to have to wait long */
1727		rc = WaitLatch(MyLatch,
1728					   WL_LATCH_SET | WL_TIMEOUT | WL_EXIT_ON_PM_DEATH,
1729					   10L, WAIT_EVENT_REPLICATION_SLOTSYNC_SHUTDOWN);
1730
1731		if (rc & WL_LATCH_SET)
1732		{
1733			ResetLatch(MyLatch);
1734			CHECK_FOR_INTERRUPTS();
1735		}
1736
1737		SpinLockAcquire(&SlotSyncCtx->mutex);
1738
1739		/* Ensure that no process is syncing the slots. */
1740		if (!SlotSyncCtx->syncing)
1741			break;
1742
1743		SpinLockRelease(&SlotSyncCtx->mutex);
1744	}
1745
1746	SpinLockRelease(&SlotSyncCtx->mutex);
1747
1748	update_synced_slots_inactive_since();
1749}

1751/*
1752 * SlotSyncWorkerCanRestart
1753 *
1754 * Returns true if enough time (SLOTSYNC_RESTART_INTERVAL_SEC) has passed
1755 * since it was launched last. Otherwise returns false.
1756 *
1757 * This is a safety valve to protect against continuous respawn attempts if the
1758 * worker is dying immediately at launch. Note that since we will retry to
1759 * launch the worker from the postmaster main loop, we will get another
1760 * chance later.
1761 */
1762bool
1763SlotSyncWorkerCanRestart(void)
1764{
1765	time_t		curtime = time(NULL);
1766
1767	/* Return false if too soon since last start. */
1768	if ((unsigned int) (curtime - SlotSyncCtx->last_start_time) <
1769		(unsigned int) SLOTSYNC_RESTART_INTERVAL_SEC)
1770		return false;
1771
1772	SlotSyncCtx->last_start_time = curtime;
1773
1774	return true;
1775}

1817/*
1818 * Error cleanup callback for slot sync SQL function.
1819 */
1820static void
1821slotsync_failure_callback(int code, Datum arg)
1822{
1823	WalReceiverConn *wrconn = (WalReceiverConn *) DatumGetPointer(arg);
1824
1825	/*
1826	 * We need to do slots cleanup here just like WalSndErrorCleanup() does.
1827	 *
1828	 * The startup process during promotion invokes ShutDownSlotSync() which
1829	 * waits for slot sync to finish and it does that by checking the
1830	 * 'syncing' flag. Thus the SQL function must be done with slots' release
1831	 * and cleanup to avoid any dangling temporary slots or active slots
1832	 * before it marks itself as finished syncing.
1833	 */
1834
1835	/* Make sure active replication slots are released */
1836	if (MyReplicationSlot != NULL)
1837		ReplicationSlotRelease();
1838
1839	/* Also cleanup the synced temporary slots. */
1840	ReplicationSlotCleanup(true);
1841
1842	/*
1843	 * The set syncing_slots indicates that the process errored out without
1844	 * resetting the flag. So, we need to clean up shared memory and reset the
1845	 * flag here.
1846	 */
1847	if (syncing_slots)
1848		reset_syncing_flag();
1849
1850	walrcv_disconnect(wrconn);
1851}

1853/*
1854 * Synchronize the failover enabled replication slots using the specified
1855 * primary server connection.
1856 */
1857void
1858SyncReplicationSlots(WalReceiverConn *wrconn)
1859{
1860	PG_ENSURE_ERROR_CLEANUP(slotsync_failure_callback, PointerGetDatum(wrconn));
1861	{
1862		check_and_set_sync_info(MyProcPid);
1863
1864		validate_remote_info(wrconn);
1865
1866		synchronize_slots(wrconn);
1867
1868		/* Cleanup the synced temporary slots */
1869		ReplicationSlotCleanup(true);
1870
1871		/* We are done with sync, so reset sync flag */
1872		reset_syncing_flag();
1873	}
1874	PG_END_ENSURE_ERROR_CLEANUP(slotsync_failure_callback, PointerGetDatum(wrconn));
