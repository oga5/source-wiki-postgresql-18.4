36%parse-param {char **syncrep_parse_error_msg_p}

44%union
45{
46	char	   *str;
47	List	   *list;
48	SyncRepConfigData *config;
49}

51%token <str> NAME NUM JUNK ANY FIRST
52
53%type <config> result standby_config
54%type <list> standby_list
55%type <str> standby_name

57%start result

59%%
60result:
61		standby_config				{
62										*syncrep_parse_result_p = $1;
63										(void) yynerrs; /* suppress compiler warning */
64									}
65	;
66
67standby_config:
68		standby_list				{ $$ = create_syncrep_config("1", $1, SYNC_REP_PRIORITY); }
69		| NUM '(' standby_list ')'		{ $$ = create_syncrep_config($1, $3, SYNC_REP_PRIORITY); }
70		| ANY NUM '(' standby_list ')'		{ $$ = create_syncrep_config($2, $4, SYNC_REP_QUORUM); }
71		| FIRST NUM '(' standby_list ')'		{ $$ = create_syncrep_config($2, $4, SYNC_REP_PRIORITY); }
72	;
73
74standby_list:
75		standby_name						{ $$ = list_make1($1); }
76		| standby_list ',' standby_name		{ $$ = lappend($1, $3); }
77	;
78
79standby_name:
80		NAME						{ $$ = $1; }
81		| NUM						{ $$ = $1; }
82	;

85static SyncRepConfigData *
86create_syncrep_config(const char *num_sync, List *members, uint8 syncrep_method)
87{
88	SyncRepConfigData *config;
89	int			size;
90	ListCell   *lc;
91	char	   *ptr;
92
93	/* Compute space needed for flat representation */
94	size = offsetof(SyncRepConfigData, member_names);
95	foreach(lc, members)
96	{
97		char	   *standby_name = (char *) lfirst(lc);
98
99		size += strlen(standby_name) + 1;
100	}
101
102	/* And transform the data into flat representation */
103	config = (SyncRepConfigData *) palloc(size);
104
105	config->config_size = size;
106	config->num_sync = atoi(num_sync);
107	config->syncrep_method = syncrep_method;
108	config->nmembers = list_length(members);
109	ptr = config->member_names;
110	foreach(lc, members)
111	{
112		char	   *standby_name = (char *) lfirst(lc);
113
114		strcpy(ptr, standby_name);
115		ptr += strlen(standby_name) + 1;
116	}
117
118	return config;
119}
