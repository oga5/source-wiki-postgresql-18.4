13 * NOTES
14 *
15 * Replication slots are used to keep state about replication streams
16 * originating from this cluster.  Their primary purpose is to prevent the
17 * premature removal of WAL or of old tuple versions in a manner that would
18 * interfere with replication; they are also useful for monitoring purposes.
19 * Slots need to be permanent (to allow restarts), crash-safe, and allocatable
20 * on standbys (to support cascading setups).  The requirement that slots be
21 * usable on standbys precludes storing them in the system catalogs.
22 *
23 * Each replication slot gets its own directory inside the directory
24 * $PGDATA / PG_REPLSLOT_DIR.  Inside that directory the state file will
25 * contain the slot's own data.  Additional data can be stored alongside that
26 * file if required.  While the server is running, the state data is also
27 * cached in memory for efficiency.
28 *
29 * ReplicationSlotAllocationLock must be taken in exclusive mode to allocate
30 * or free a slot. ReplicationSlotControlLock must be taken in shared mode
31 * to iterate over the slots, and in exclusive mode to change the in_use flag
32 * of a slot.  The remaining data in each slot is protected by its mutex.
33 *
34 *-------------------------------------------------------------------------
35 */

182/*
183 * Report shared-memory space needed by ReplicationSlotsShmemInit.
184 */
185Size
186ReplicationSlotsShmemSize(void)
187{
188	Size		size = 0;
189
190	if (max_replication_slots == 0)
191		return size;
192
193	size = offsetof(ReplicationSlotCtlData, replication_slots);
194	size = add_size(size,
195					mul_size(max_replication_slots, sizeof(ReplicationSlot)));
196
197	return size;
198}

200/*
201 * Allocate and initialize shared memory for replication slots.
202 */
203void
204ReplicationSlotsShmemInit(void)
205{
206	bool		found;
207
208	if (max_replication_slots == 0)
209		return;
210
211	ReplicationSlotCtl = (ReplicationSlotCtlData *)
212		ShmemInitStruct("ReplicationSlot Ctl", ReplicationSlotsShmemSize(),
213						&found);
214
215	if (!found)
216	{
217		int			i;
218
219		/* First time through, so initialize */
220		MemSet(ReplicationSlotCtl, 0, ReplicationSlotsShmemSize());
221
222		for (i = 0; i < max_replication_slots; i++)
223		{
224			ReplicationSlot *slot = &ReplicationSlotCtl->replication_slots[i];
225
226			/* everything else is zeroed by the memset above */
227			SpinLockInit(&slot->mutex);
228			LWLockInitialize(&slot->io_in_progress_lock,
229							 LWTRANCHE_REPLICATION_SLOT_IO);
230			ConditionVariableInit(&slot->active_cv);
231		}
232	}
233}

258/*
259 * Check whether the passed slot name is valid and report errors at elevel.
260 *
261 * See comments for ReplicationSlotValidateNameInternal().
262 */
263bool
264ReplicationSlotValidateName(const char *name, int elevel)
265{
266	int			err_code;
267	char	   *err_msg = NULL;
268	char	   *err_hint = NULL;
269
270	if (!ReplicationSlotValidateNameInternal(name, &err_code, &err_msg,
271											 &err_hint))
272	{
273		/*
274		 * Use errmsg_internal() and errhint_internal() instead of errmsg()
275		 * and errhint(), since the messages from
276		 * ReplicationSlotValidateNameInternal() are already translated. This
277		 * avoids double translation.
278		 */
279		ereport(elevel,
280				errcode(err_code),
281				errmsg_internal("%s", err_msg),
282				(err_hint != NULL) ? errhint_internal("%s", err_hint) : 0);
283
284		pfree(err_msg);
285		if (err_hint != NULL)
286			pfree(err_hint);
287		return false;
288	}
289
290	return true;
291}

293/*
294 * Check whether the passed slot name is valid.
295 *
296 * Slot names may consist out of [a-z0-9_]{1,NAMEDATALEN-1} which should allow
297 * the name to be used as a directory name on every supported OS.
298 *
299 * Returns true if the slot name is valid. Otherwise, returns false and stores
300 * the error code, error message, and optional hint in err_code, err_msg, and
301 * err_hint, respectively. The caller is responsible for freeing err_msg and
302 * err_hint, which are palloc'd.
303 */
304bool
305ReplicationSlotValidateNameInternal(const char *name, int *err_code,
306									char **err_msg, char **err_hint)
307{
308	const char *cp;
309
310	if (strlen(name) == 0)
311	{
312		*err_code = ERRCODE_INVALID_NAME;
313		*err_msg = psprintf(_("replication slot name \"%s\" is too short"), name);
314		*err_hint = NULL;
315		return false;
316	}
317
318	if (strlen(name) >= NAMEDATALEN)
319	{
320		*err_code = ERRCODE_NAME_TOO_LONG;
321		*err_msg = psprintf(_("replication slot name \"%s\" is too long"), name);
322		*err_hint = NULL;
323		return false;
324	}
325
326	for (cp = name; *cp; cp++)
327	{
328		if (!((*cp >= 'a' && *cp <= 'z')
329			  || (*cp >= '0' && *cp <= '9')
330			  || (*cp == '_')))
331		{
332			*err_code = ERRCODE_INVALID_NAME;
333			*err_msg = psprintf(_("replication slot name \"%s\" contains invalid character"), name);
334			*err_hint = psprintf(_("Replication slot names may only contain lower case letters, numbers, and the underscore character."));
335			return false;
336		}
337	}
338	return true;
339}

341/*
342 * Create a new replication slot and mark it as used by this backend.
343 *
344 * name: Name of the slot
345 * db_specific: logical decoding is db specific; if the slot is going to
346 *	   be used for that pass true, otherwise false.
347 * two_phase: If enabled, allows decoding of prepared transactions.
348 * failover: If enabled, allows the slot to be synced to standbys so
349 *     that logical replication can be resumed after failover.
350 * synced: True if the slot is synchronized from the primary server.
351 */
352void
353ReplicationSlotCreate(const char *name, bool db_specific,
354					  ReplicationSlotPersistency persistency,
355					  bool two_phase, bool failover, bool synced)
356{
357	ReplicationSlot *slot = NULL;
358	int			i;
359
360	Assert(MyReplicationSlot == NULL);
361
362	ReplicationSlotValidateName(name, ERROR);
363
364	if (failover)
365	{
366		/*
367		 * Do not allow users to create the failover enabled slots on the
368		 * standby as we do not support sync to the cascading standby.
369		 *
370		 * However, failover enabled slots can be created during slot
371		 * synchronization because we need to retain the same values as the
372		 * remote slot.
373		 */
374		if (RecoveryInProgress() && !IsSyncingReplicationSlots())
375			ereport(ERROR,
376					errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
377					errmsg("cannot enable failover for a replication slot created on the standby"));
378
379		/*
380		 * Do not allow users to create failover enabled temporary slots,
381		 * because temporary slots will not be synced to the standby.
382		 *
383		 * However, failover enabled temporary slots can be created during
384		 * slot synchronization. See the comments atop slotsync.c for details.
385		 */
386		if (persistency == RS_TEMPORARY && !IsSyncingReplicationSlots())
387			ereport(ERROR,
388					errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
389					errmsg("cannot enable failover for a temporary replication slot"));
390	}
391
392	/*
393	 * If some other backend ran this code concurrently with us, we'd likely
394	 * both allocate the same slot, and that would be bad.  We'd also be at
395	 * risk of missing a name collision.  Also, we don't want to try to create
396	 * a new slot while somebody's busy cleaning up an old one, because we
397	 * might both be monkeying with the same directory.
398	 */
399	LWLockAcquire(ReplicationSlotAllocationLock, LW_EXCLUSIVE);
400
401	/*
402	 * Check for name collision, and identify an allocatable slot.  We need to
403	 * hold ReplicationSlotControlLock in shared mode for this, so that nobody
404	 * else can change the in_use flags while we're looking at them.
405	 */
406	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
407	for (i = 0; i < max_replication_slots; i++)
408	{
409		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
410
411		if (s->in_use && strcmp(name, NameStr(s->data.name)) == 0)
412			ereport(ERROR,
413					(errcode(ERRCODE_DUPLICATE_OBJECT),
414					 errmsg("replication slot \"%s\" already exists", name)));
415		if (!s->in_use && slot == NULL)
416			slot = s;
417	}
418	LWLockRelease(ReplicationSlotControlLock);
419
420	/* If all slots are in use, we're out of luck. */
421	if (slot == NULL)
422		ereport(ERROR,
423				(errcode(ERRCODE_CONFIGURATION_LIMIT_EXCEEDED),
424				 errmsg("all replication slots are in use"),
425				 errhint("Free one or increase \"max_replication_slots\".")));
426
427	/*
428	 * Since this slot is not in use, nobody should be looking at any part of
429	 * it other than the in_use field unless they're trying to allocate it.
430	 * And since we hold ReplicationSlotAllocationLock, nobody except us can
431	 * be doing that.  So it's safe to initialize the slot.
432	 */
433	Assert(!slot->in_use);
434	Assert(slot->active_pid == 0);
435
436	/* first initialize persistent data */
437	memset(&slot->data, 0, sizeof(ReplicationSlotPersistentData));
438	namestrcpy(&slot->data.name, name);
439	slot->data.database = db_specific ? MyDatabaseId : InvalidOid;
440	slot->data.persistency = persistency;
441	slot->data.two_phase = two_phase;
442	slot->data.two_phase_at = InvalidXLogRecPtr;
443	slot->data.failover = failover;
444	slot->data.synced = synced;
445
446	/* and then data only present in shared memory */
447	slot->just_dirtied = false;
448	slot->dirty = false;
449	slot->effective_xmin = InvalidTransactionId;
450	slot->effective_catalog_xmin = InvalidTransactionId;
451	slot->candidate_catalog_xmin = InvalidTransactionId;
452	slot->candidate_xmin_lsn = InvalidXLogRecPtr;
453	slot->candidate_restart_valid = InvalidXLogRecPtr;
454	slot->candidate_restart_lsn = InvalidXLogRecPtr;
455	slot->last_saved_confirmed_flush = InvalidXLogRecPtr;
456	slot->last_saved_restart_lsn = InvalidXLogRecPtr;
457	slot->inactive_since = 0;
458
459	/*
460	 * Create the slot on disk.  We haven't actually marked the slot allocated
461	 * yet, so no special cleanup is required if this errors out.
462	 */
463	CreateSlotOnDisk(slot);
464
465	/*
466	 * We need to briefly prevent any other backend from iterating over the
467	 * slots while we flip the in_use flag. We also need to set the active
468	 * flag while holding the ControlLock as otherwise a concurrent
469	 * ReplicationSlotAcquire() could acquire the slot as well.
470	 */
471	LWLockAcquire(ReplicationSlotControlLock, LW_EXCLUSIVE);
472
473	slot->in_use = true;
474
475	/* We can now mark the slot active, and that makes it our slot. */
476	SpinLockAcquire(&slot->mutex);
477	Assert(slot->active_pid == 0);
478	slot->active_pid = MyProcPid;
479	SpinLockRelease(&slot->mutex);
480	MyReplicationSlot = slot;
481
482	LWLockRelease(ReplicationSlotControlLock);
483
484	/*
485	 * Create statistics entry for the new logical slot. We don't collect any
486	 * stats for physical slots, so no need to create an entry for the same.
487	 * See ReplicationSlotDropPtr for why we need to do this before releasing
488	 * ReplicationSlotAllocationLock.
489	 */
490	if (SlotIsLogical(slot))
491		pgstat_create_replslot(slot);
492
493	/*
494	 * Now that the slot has been marked as in_use and active, it's safe to
495	 * let somebody else try to allocate a slot.
496	 */
497	LWLockRelease(ReplicationSlotAllocationLock);
498
499	/* Let everybody know we've modified this slot */
500	ConditionVariableBroadcast(&slot->active_cv);
501}

503/*
504 * Search for the named replication slot.
505 *
506 * Return the replication slot if found, otherwise NULL.
507 */
508ReplicationSlot *
509SearchNamedReplicationSlot(const char *name, bool need_lock)
510{
511	int			i;
512	ReplicationSlot *slot = NULL;
513
514	if (need_lock)
515		LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
516
517	for (i = 0; i < max_replication_slots; i++)
518	{
519		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
520
521		if (s->in_use && strcmp(name, NameStr(s->data.name)) == 0)
522		{
523			slot = s;
524			break;
525		}
526	}
527
528	if (need_lock)
529		LWLockRelease(ReplicationSlotControlLock);
530
531	return slot;
532}

536 * ReplicationSlotCtl->replication_slots.
537 *
538 * This is mainly useful to have an efficient key for storing replication slot

541int

544	Assert(slot >= ReplicationSlotCtl->replication_slots &&

550/*
551 * If the slot at 'index' is unused, return false. Otherwise 'name' is set to
552 * the slot's name and true is returned.
553 *
554 * This likely is only useful for pgstat_replslot.c during shutdown, in other
555 * cases there are obvious TOCTOU issues.
556 */
557bool
558ReplicationSlotName(int index, Name name)
559{
560	ReplicationSlot *slot;
561	bool		found;
562
563	slot = &ReplicationSlotCtl->replication_slots[index];
564
565	/*
566	 * Ensure that the slot cannot be dropped while we copy the name. Don't
567	 * need the spinlock as the name of an existing slot cannot change.
568	 */
569	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
570	found = slot->in_use;
571	if (slot->in_use)
572		namestrcpy(name, NameStr(slot->data.name));
573	LWLockRelease(ReplicationSlotControlLock);
574
575	return found;
576}

578/*
579 * Find a previously created slot and mark it as used by this process.
580 *
581 * An error is raised if nowait is true and the slot is currently in use. If
582 * nowait is false, we sleep until the slot is released by the owning process.
583 *
584 * An error is raised if error_if_invalid is true and the slot is found to
585 * be invalid. It should always be set to true, except when we are temporarily
586 * acquiring the slot and don't intend to change it.
587 */
588void
589ReplicationSlotAcquire(const char *name, bool nowait, bool error_if_invalid)
590{
591	ReplicationSlot *s;
592	int			active_pid;
593
594	Assert(name != NULL);
595
596retry:
597	Assert(MyReplicationSlot == NULL);
598
599	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
600
601	/* Check if the slot exits with the given name. */
602	s = SearchNamedReplicationSlot(name, false);
603	if (s == NULL || !s->in_use)
604	{
605		LWLockRelease(ReplicationSlotControlLock);
606
607		ereport(ERROR,
608				(errcode(ERRCODE_UNDEFINED_OBJECT),
609				 errmsg("replication slot \"%s\" does not exist",
610						name)));
611	}
612
613	/*
614	 * This is the slot we want; check if it's active under some other
615	 * process.  In single user mode, we don't need this check.
616	 */
617	if (IsUnderPostmaster)
618	{
619		/*
620		 * Get ready to sleep on the slot in case it is active.  (We may end
621		 * up not sleeping, but we don't want to do this while holding the
622		 * spinlock.)
623		 */
624		if (!nowait)
625			ConditionVariablePrepareToSleep(&s->active_cv);
626
627		/*
628		 * It is important to reset the inactive_since under spinlock here to
629		 * avoid race conditions with slot invalidation. See comments related
630		 * to inactive_since in InvalidatePossiblyObsoleteSlot.
631		 */
632		SpinLockAcquire(&s->mutex);
633		if (s->active_pid == 0)
634			s->active_pid = MyProcPid;
635		active_pid = s->active_pid;
636		ReplicationSlotSetInactiveSince(s, 0, false);
637		SpinLockRelease(&s->mutex);
638	}
639	else
640	{
641		s->active_pid = active_pid = MyProcPid;
642		ReplicationSlotSetInactiveSince(s, 0, true);
643	}
644	LWLockRelease(ReplicationSlotControlLock);
645
646	/*
647	 * If we found the slot but it's already active in another process, we
648	 * wait until the owning process signals us that it's been released, or
649	 * error out.
650	 */
651	if (active_pid != MyProcPid)
652	{
653		if (!nowait)
654		{
655			/* Wait here until we get signaled, and then restart */
656			ConditionVariableSleep(&s->active_cv,
657								   WAIT_EVENT_REPLICATION_SLOT_DROP);
658			ConditionVariableCancelSleep();
659			goto retry;
660		}
661
662		ereport(ERROR,
663				(errcode(ERRCODE_OBJECT_IN_USE),
664				 errmsg("replication slot \"%s\" is active for PID %d",
665						NameStr(s->data.name), active_pid)));
666	}
667	else if (!nowait)
668		ConditionVariableCancelSleep(); /* no sleep needed after all */
669
670	/* We made this slot active, so it's ours now. */
671	MyReplicationSlot = s;
672
673	/*
674	 * We need to check for invalidation after making the slot ours to avoid
675	 * the possible race condition with the checkpointer that can otherwise
676	 * invalidate the slot immediately after the check.
677	 */
678	if (error_if_invalid && s->data.invalidated != RS_INVAL_NONE)
679		ereport(ERROR,
680				errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
681				errmsg("can no longer access replication slot \"%s\"",
682					   NameStr(s->data.name)),
683				errdetail("This replication slot has been invalidated due to \"%s\".",
684						  GetSlotInvalidationCauseName(s->data.invalidated)));
685
686	/* Let everybody know we've modified this slot */
687	ConditionVariableBroadcast(&s->active_cv);
688
689	/*
690	 * The call to pgstat_acquire_replslot() protects against stats for a
691	 * different slot, from before a restart or such, being present during
692	 * pgstat_report_replslot().
693	 */
694	if (SlotIsLogical(s))
695		pgstat_acquire_replslot(s);
696
697
698	if (am_walsender)
699	{
700		ereport(log_replication_commands ? LOG : DEBUG1,
701				SlotIsLogical(s)
702				? errmsg("acquired logical replication slot \"%s\"",
703						 NameStr(s->data.name))
704				: errmsg("acquired physical replication slot \"%s\"",
705						 NameStr(s->data.name)));
706	}
707}

709/*
710 * Release the replication slot that this backend considers to own.
711 *
712 * This or another backend can re-acquire the slot later.
713 * Resources this slot requires will be preserved.
714 */
715void
716ReplicationSlotRelease(void)
717{
718	ReplicationSlot *slot = MyReplicationSlot;
719	char	   *slotname = NULL;	/* keep compiler quiet */
720	bool		is_logical = false; /* keep compiler quiet */
721	TimestampTz now = 0;
722
723	Assert(slot != NULL && slot->active_pid != 0);
724
725	if (am_walsender)
726	{
727		slotname = pstrdup(NameStr(slot->data.name));
728		is_logical = SlotIsLogical(slot);
729	}
730
731	if (slot->data.persistency == RS_EPHEMERAL)
732	{
733		/*
734		 * Delete the slot. There is no !PANIC case where this is allowed to
735		 * fail, all that may happen is an incomplete cleanup of the on-disk
736		 * data.
737		 */
738		ReplicationSlotDropAcquired();
739	}
740
741	/*
742	 * If slot needed to temporarily restrain both data and catalog xmin to
743	 * create the catalog snapshot, remove that temporary constraint.
744	 * Snapshots can only be exported while the initial snapshot is still
745	 * acquired.
746	 */
747	if (!TransactionIdIsValid(slot->data.xmin) &&
748		TransactionIdIsValid(slot->effective_xmin))
749	{
750		SpinLockAcquire(&slot->mutex);
751		slot->effective_xmin = InvalidTransactionId;
752		SpinLockRelease(&slot->mutex);
753		ReplicationSlotsComputeRequiredXmin(false);
754	}
755
756	/*
757	 * Set the time since the slot has become inactive. We get the current
758	 * time beforehand to avoid system call while holding the spinlock.
759	 */
760	now = GetCurrentTimestamp();
761
762	if (slot->data.persistency == RS_PERSISTENT)
763	{
764		/*
765		 * Mark persistent slot inactive.  We're not freeing it, just
766		 * disconnecting, but wake up others that may be waiting for it.
767		 */
768		SpinLockAcquire(&slot->mutex);
769		slot->active_pid = 0;
770		ReplicationSlotSetInactiveSince(slot, now, false);
771		SpinLockRelease(&slot->mutex);
772		ConditionVariableBroadcast(&slot->active_cv);
773	}
774	else
775		ReplicationSlotSetInactiveSince(slot, now, true);
776
777	MyReplicationSlot = NULL;
778
779	/* might not have been set when we've been a plain slot */
780	LWLockAcquire(ProcArrayLock, LW_EXCLUSIVE);
781	MyProc->statusFlags &= ~PROC_IN_LOGICAL_DECODING;
782	ProcGlobal->statusFlags[MyProc->pgxactoff] = MyProc->statusFlags;
783	LWLockRelease(ProcArrayLock);
784
785	if (am_walsender)
786	{
787		ereport(log_replication_commands ? LOG : DEBUG1,
788				is_logical
789				? errmsg("released logical replication slot \"%s\"",
790						 slotname)
791				: errmsg("released physical replication slot \"%s\"",
792						 slotname));
793
794		pfree(slotname);
795	}
796}

798/*
799 * Cleanup temporary slots created in current session.
800 *
801 * Cleanup only synced temporary slots if 'synced_only' is true, else
802 * cleanup all temporary slots.
803 */
804void
805ReplicationSlotCleanup(bool synced_only)
806{
807	int			i;
808
809	Assert(MyReplicationSlot == NULL);
810
811restart:
812	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
813	for (i = 0; i < max_replication_slots; i++)
814	{
815		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
816
817		if (!s->in_use)
818			continue;
819
820		SpinLockAcquire(&s->mutex);
821		if ((s->active_pid == MyProcPid &&
822			 (!synced_only || s->data.synced)))
823		{
824			Assert(s->data.persistency == RS_TEMPORARY);
825			SpinLockRelease(&s->mutex);
826			LWLockRelease(ReplicationSlotControlLock);	/* avoid deadlock */
827
828			ReplicationSlotDropPtr(s);
829
830			ConditionVariableBroadcast(&s->active_cv);
831			goto restart;
832		}
833		else
834			SpinLockRelease(&s->mutex);
835	}
836
837	LWLockRelease(ReplicationSlotControlLock);
838}

841 * Permanently drop replication slot identified by the passed in name.

856				errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),

863/*
864 * Change the definition of the slot identified by the specified name.
865 *
866 * Altering the two_phase property of a slot requires caution on the
867 * client-side. Enabling it at any random point during decoding has the
868 * risk that transactions prepared before this change may be skipped by
869 * the decoder, leading to missing prepare records on the client. So, we
870 * enable it for subscription related slots only once the initial tablesync
871 * is finished. See comments atop worker.c. Disabling it is safe only when
872 * there are no pending prepared transaction, otherwise, the changes of
873 * already prepared transactions can be replicated again along with their
874 * corresponding commit leading to duplicate data or errors.
875 */
876void
877ReplicationSlotAlter(const char *name, const bool *failover,
878					 const bool *two_phase)
879{
880	bool		update_slot = false;
881
882	Assert(MyReplicationSlot == NULL);
883	Assert(failover || two_phase);
884
885	ReplicationSlotAcquire(name, false, true);
886
887	if (SlotIsPhysical(MyReplicationSlot))
888		ereport(ERROR,
889				errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
890				errmsg("cannot use %s with a physical replication slot",
891					   "ALTER_REPLICATION_SLOT"));
892
893	if (RecoveryInProgress())
894	{
895		/*
896		 * Do not allow users to alter the slots which are currently being
897		 * synced from the primary to the standby.
898		 */
899		if (MyReplicationSlot->data.synced)
900			ereport(ERROR,
901					errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
902					errmsg("cannot alter replication slot \"%s\"", name),
903					errdetail("This replication slot is being synchronized from the primary server."));
904
905		/*
906		 * Do not allow users to enable failover on the standby as we do not
907		 * support sync to the cascading standby.
908		 */
909		if (failover && *failover)
910			ereport(ERROR,
911					errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
912					errmsg("cannot enable failover for a replication slot"
913						   " on the standby"));
914	}
915
916	if (failover)
917	{
918		/*
919		 * Do not allow users to enable failover for temporary slots as we do
920		 * not support syncing temporary slots to the standby.
921		 */
922		if (*failover && MyReplicationSlot->data.persistency == RS_TEMPORARY)
923			ereport(ERROR,
924					errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
925					errmsg("cannot enable failover for a temporary replication slot"));
926
927		if (MyReplicationSlot->data.failover != *failover)
928		{
929			SpinLockAcquire(&MyReplicationSlot->mutex);
930			MyReplicationSlot->data.failover = *failover;
931			SpinLockRelease(&MyReplicationSlot->mutex);
932
933			update_slot = true;
934		}
935	}
936
937	if (two_phase && MyReplicationSlot->data.two_phase != *two_phase)
938	{
939		SpinLockAcquire(&MyReplicationSlot->mutex);
940		MyReplicationSlot->data.two_phase = *two_phase;
941		SpinLockRelease(&MyReplicationSlot->mutex);
942
943		update_slot = true;
944	}
945
946	if (update_slot)
947	{
948		ReplicationSlotMarkDirty();
949		ReplicationSlotSave();
950	}
951
952	ReplicationSlotRelease();
953}

971/*
972 * Permanently drop the replication slot which will be released by the point
973 * this function returns.
974 */
975static void
976ReplicationSlotDropPtr(ReplicationSlot *slot)
977{
978	char		path[MAXPGPATH];
979	char		tmppath[MAXPGPATH];
980
981	/*
982	 * If some other backend ran this code concurrently with us, we might try
983	 * to delete a slot with a certain name while someone else was trying to
984	 * create a slot with the same name.
985	 */
986	LWLockAcquire(ReplicationSlotAllocationLock, LW_EXCLUSIVE);
987
988	/* Generate pathnames. */
989	sprintf(path, "%s/%s", PG_REPLSLOT_DIR, NameStr(slot->data.name));
990	sprintf(tmppath, "%s/%s.tmp", PG_REPLSLOT_DIR, NameStr(slot->data.name));
991
992	/*
993	 * Rename the slot directory on disk, so that we'll no longer recognize
994	 * this as a valid slot.  Note that if this fails, we've got to mark the
995	 * slot inactive before bailing out.  If we're dropping an ephemeral or a
996	 * temporary slot, we better never fail hard as the caller won't expect
997	 * the slot to survive and this might get called during error handling.
998	 */
999	if (rename(path, tmppath) == 0)
1000	{
1001		/*
1002		 * We need to fsync() the directory we just renamed and its parent to
1003		 * make sure that our changes are on disk in a crash-safe fashion.  If
1004		 * fsync() fails, we can't be sure whether the changes are on disk or
1005		 * not.  For now, we handle that by panicking;
1006		 * StartupReplicationSlots() will try to straighten it out after
1007		 * restart.
1008		 */
1009		START_CRIT_SECTION();
1010		fsync_fname(tmppath, true);
1011		fsync_fname(PG_REPLSLOT_DIR, true);
1012		END_CRIT_SECTION();
1013	}
1014	else
1015	{
1016		bool		fail_softly = slot->data.persistency != RS_PERSISTENT;
1017
1018		SpinLockAcquire(&slot->mutex);
1019		slot->active_pid = 0;
1020		SpinLockRelease(&slot->mutex);
1021
1022		/* wake up anyone waiting on this slot */
1023		ConditionVariableBroadcast(&slot->active_cv);
1024
1025		ereport(fail_softly ? WARNING : ERROR,
1026				(errcode_for_file_access(),
1027				 errmsg("could not rename file \"%s\" to \"%s\": %m",
1028						path, tmppath)));
1029	}
1030
1031	/*
1032	 * The slot is definitely gone.  Lock out concurrent scans of the array
1033	 * long enough to kill it.  It's OK to clear the active PID here without
1034	 * grabbing the mutex because nobody else can be scanning the array here,
1035	 * and nobody can be attached to this slot and thus access it without
1036	 * scanning the array.
1037	 *
1038	 * Also wake up processes waiting for it.
1039	 */
1040	LWLockAcquire(ReplicationSlotControlLock, LW_EXCLUSIVE);
1041	slot->active_pid = 0;
1042	slot->in_use = false;
1043	LWLockRelease(ReplicationSlotControlLock);
1044	ConditionVariableBroadcast(&slot->active_cv);
1045
1046	/*
1047	 * Slot is dead and doesn't prevent resource removal anymore, recompute
1048	 * limits.
1049	 */
1050	ReplicationSlotsComputeRequiredXmin(false);
1051	ReplicationSlotsComputeRequiredLSN();
1052
1053	/*
1054	 * If removing the directory fails, the worst thing that will happen is
1055	 * that the user won't be able to create a new slot with the same name
1056	 * until the next server restart.  We warn about it, but that's all.
1057	 */
1058	if (!rmtree(tmppath, true))
1059		ereport(WARNING,
1060				(errmsg("could not remove directory \"%s\"", tmppath)));
1061
1062	/*
1063	 * Drop the statistics entry for the replication slot.  Do this while
1064	 * holding ReplicationSlotAllocationLock so that we don't drop a
1065	 * statistics entry for another slot with the same name just created in
1066	 * another session.
1067	 */
1068	if (SlotIsLogical(slot))
1069		pgstat_drop_replslot(slot);
1070
1071	/*
1072	 * We release this at the very end, so that nobody starts trying to create
1073	 * a slot while we're still cleaning up the detritus of the old one.
1074	 */
1075	LWLockRelease(ReplicationSlotAllocationLock);
1076}

1133/*
1134 * Compute the oldest xmin across all slots and store it in the ProcArray.
1135 *
1136 * If already_locked is true, both the ReplicationSlotControlLock and the
1137 * ProcArrayLock have already been acquired exclusively. It is crucial that the
1138 * caller first acquires the ReplicationSlotControlLock, followed by the
1139 * ProcArrayLock, to prevent any undetectable deadlocks since this function
1140 * acquires them in that order.
1141 */
1142void
1143ReplicationSlotsComputeRequiredXmin(bool already_locked)
1144{
1145	int			i;
1146	TransactionId agg_xmin = InvalidTransactionId;
1147	TransactionId agg_catalog_xmin = InvalidTransactionId;
1148
1149	Assert(ReplicationSlotCtl != NULL);
1150	Assert(!already_locked ||
1151		   (LWLockHeldByMeInMode(ReplicationSlotControlLock, LW_EXCLUSIVE) &&
1152			LWLockHeldByMeInMode(ProcArrayLock, LW_EXCLUSIVE)));
1153
1154	/*
1155	 * Hold the ReplicationSlotControlLock until after updating the slot xmin
1156	 * values, so no backend updates the initial xmin for newly created slot
1157	 * concurrently. A shared lock is used here to minimize lock contention,
1158	 * especially when many slots exist and advancements occur frequently.
1159	 * This is safe since an exclusive lock is taken during initial slot xmin
1160	 * update in slot creation.
1161	 *
1162	 * One might think that we can hold the ProcArrayLock exclusively and
1163	 * update the slot xmin values, but it could increase lock contention on
1164	 * the ProcArrayLock, which is not great since this function can be called
1165	 * at non-negligible frequency.
1166	 *
1167	 * Concurrent invocation of this function may cause the computed slot xmin
1168	 * to regress. However, this is harmless because tuples prior to the most
1169	 * recent xmin are no longer useful once advancement occurs (see
1170	 * LogicalConfirmReceivedLocation where the slot's xmin value is flushed
1171	 * before updating the effective_xmin). Thus, such regression merely
1172	 * prevents VACUUM from prematurely removing tuples without causing the
1173	 * early deletion of required data.
1174	 */
1175	if (!already_locked)
1176		LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1177
1178	for (i = 0; i < max_replication_slots; i++)
1179	{
1180		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
1181		TransactionId effective_xmin;
1182		TransactionId effective_catalog_xmin;
1183		bool		invalidated;
1184
1185		if (!s->in_use)
1186			continue;
1187
1188		SpinLockAcquire(&s->mutex);
1189		effective_xmin = s->effective_xmin;
1190		effective_catalog_xmin = s->effective_catalog_xmin;
1191		invalidated = s->data.invalidated != RS_INVAL_NONE;
1192		SpinLockRelease(&s->mutex);
1193
1194		/* invalidated slots need not apply */
1195		if (invalidated)
1196			continue;
1197
1198		/* check the data xmin */
1199		if (TransactionIdIsValid(effective_xmin) &&
1200			(!TransactionIdIsValid(agg_xmin) ||
1201			 TransactionIdPrecedes(effective_xmin, agg_xmin)))
1202			agg_xmin = effective_xmin;
1203
1204		/* check the catalog xmin */
1205		if (TransactionIdIsValid(effective_catalog_xmin) &&
1206			(!TransactionIdIsValid(agg_catalog_xmin) ||
1207			 TransactionIdPrecedes(effective_catalog_xmin, agg_catalog_xmin)))
1208			agg_catalog_xmin = effective_catalog_xmin;
1209	}
1210
1211	ProcArraySetReplicationSlotXmin(agg_xmin, agg_catalog_xmin, already_locked);
1212
1213	if (!already_locked)
1214		LWLockRelease(ReplicationSlotControlLock);
1215}

1217/*
1218 * Compute the oldest restart LSN across all slots and inform xlog module.
1219 *
1220 * Note: while max_slot_wal_keep_size is theoretically relevant for this
1221 * purpose, we don't try to account for that, because this module doesn't
1222 * know what to compare against.
1223 */
1224void
1225ReplicationSlotsComputeRequiredLSN(void)
1226{
1227	int			i;
1228	XLogRecPtr	min_required = InvalidXLogRecPtr;
1229
1230	Assert(ReplicationSlotCtl != NULL);
1231
1232	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1233	for (i = 0; i < max_replication_slots; i++)
1234	{
1235		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
1236		XLogRecPtr	restart_lsn;
1237		XLogRecPtr	last_saved_restart_lsn;
1238		bool		invalidated;
1239		ReplicationSlotPersistency persistency;
1240
1241		if (!s->in_use)
1242			continue;
1243
1244		SpinLockAcquire(&s->mutex);
1245		persistency = s->data.persistency;
1246		restart_lsn = s->data.restart_lsn;
1247		invalidated = s->data.invalidated != RS_INVAL_NONE;
1248		last_saved_restart_lsn = s->last_saved_restart_lsn;
1249		SpinLockRelease(&s->mutex);
1250
1251		/* invalidated slots need not apply */
1252		if (invalidated)
1253			continue;
1254
1255		/*
1256		 * For persistent slot use last_saved_restart_lsn to compute the
1257		 * oldest LSN for removal of WAL segments.  The segments between
1258		 * last_saved_restart_lsn and restart_lsn might be needed by a
1259		 * persistent slot in the case of database crash.  Non-persistent
1260		 * slots can't survive the database crash, so we don't care about
1261		 * last_saved_restart_lsn for them.
1262		 */
1263		if (persistency == RS_PERSISTENT)
1264		{
1265			if (last_saved_restart_lsn != InvalidXLogRecPtr &&
1266				restart_lsn > last_saved_restart_lsn)
1267			{
1268				restart_lsn = last_saved_restart_lsn;
1269			}
1270		}
1271
1272		if (restart_lsn != InvalidXLogRecPtr &&
1273			(min_required == InvalidXLogRecPtr ||
1274			 restart_lsn < min_required))
1275			min_required = restart_lsn;
1276	}
1277	LWLockRelease(ReplicationSlotControlLock);
1278
1279	XLogSetReplicationSlotMinimumLSN(min_required);
1280}

1282/*
1283 * Compute the oldest WAL LSN required by *logical* decoding slots..
1284 *
1285 * Returns InvalidXLogRecPtr if logical decoding is disabled or no logical
1286 * slots exist.
1287 *
1288 * NB: this returns a value >= ReplicationSlotsComputeRequiredLSN(), since it
1289 * ignores physical replication slots.
1290 *
1291 * The results aren't required frequently, so we don't maintain a precomputed
1292 * value like we do for ComputeRequiredLSN() and ComputeRequiredXmin().
1293 */
1294XLogRecPtr
1295ReplicationSlotsComputeLogicalRestartLSN(void)
1296{
1297	XLogRecPtr	result = InvalidXLogRecPtr;
1298	int			i;
1299
1300	if (max_replication_slots <= 0)
1301		return InvalidXLogRecPtr;
1302
1303	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1304
1305	for (i = 0; i < max_replication_slots; i++)
1306	{
1307		ReplicationSlot *s;
1308		XLogRecPtr	restart_lsn;
1309		XLogRecPtr	last_saved_restart_lsn;
1310		bool		invalidated;
1311		ReplicationSlotPersistency persistency;
1312
1313		s = &ReplicationSlotCtl->replication_slots[i];
1314
1315		/* cannot change while ReplicationSlotCtlLock is held */
1316		if (!s->in_use)
1317			continue;
1318
1319		/* we're only interested in logical slots */
1320		if (!SlotIsLogical(s))
1321			continue;
1322
1323		/* read once, it's ok if it increases while we're checking */
1324		SpinLockAcquire(&s->mutex);
1325		persistency = s->data.persistency;
1326		restart_lsn = s->data.restart_lsn;
1327		invalidated = s->data.invalidated != RS_INVAL_NONE;
1328		last_saved_restart_lsn = s->last_saved_restart_lsn;
1329		SpinLockRelease(&s->mutex);
1330
1331		/* invalidated slots need not apply */
1332		if (invalidated)
1333			continue;
1334
1335		/*
1336		 * For persistent slot use last_saved_restart_lsn to compute the
1337		 * oldest LSN for removal of WAL segments.  The segments between
1338		 * last_saved_restart_lsn and restart_lsn might be needed by a
1339		 * persistent slot in the case of database crash.  Non-persistent
1340		 * slots can't survive the database crash, so we don't care about
1341		 * last_saved_restart_lsn for them.
1342		 */
1343		if (persistency == RS_PERSISTENT)
1344		{
1345			if (last_saved_restart_lsn != InvalidXLogRecPtr &&
1346				restart_lsn > last_saved_restart_lsn)
1347			{
1348				restart_lsn = last_saved_restart_lsn;
1349			}
1350		}
1351
1352		if (restart_lsn == InvalidXLogRecPtr)
1353			continue;
1354
1355		if (result == InvalidXLogRecPtr ||
1356			restart_lsn < result)
1357			result = restart_lsn;
1358	}
1359
1360	LWLockRelease(ReplicationSlotControlLock);
1361
1362	return result;
1363}

1365/*
1366 * ReplicationSlotsCountDBSlots -- count the number of slots that refer to the
1367 * passed database oid.
1368 *
1369 * Returns true if there are any slots referencing the database. *nslots will

1371 * currently active.

1375{
1376	int			i;
1377
1378	*nslots = *nactive = 0;
1379
1380	if (max_replication_slots <= 0)
1381		return false;
1382
1383	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1384	for (i = 0; i < max_replication_slots; i++)
1385	{
1386		ReplicationSlot *s;
1387
1388		s = &ReplicationSlotCtl->replication_slots[i];
1389
1390		/* cannot change while ReplicationSlotCtlLock is held */
1391		if (!s->in_use)
1392			continue;
1393
1394		/* only logical slots are database specific, skip */
1395		if (!SlotIsLogical(s))
1396			continue;
1397
1398		/* not our database, skip */
1399		if (s->data.database != dboid)
1400			continue;
1401
1402		/* NB: intentionally counting invalidated slots */
1403
1404		/* count slots with spinlock held */
1405		SpinLockAcquire(&s->mutex);
1406		(*nslots)++;
1407		if (s->active_pid != 0)
1408			(*nactive)++;
1409		SpinLockRelease(&s->mutex);
1410	}
1411	LWLockRelease(ReplicationSlotControlLock);
1412
1413	if (*nslots > 0)
1414		return true;
1415	return false;
1416}

1418/*
1419 * ReplicationSlotsDropDBSlots -- Drop all db-specific slots relating to the
1420 * passed database oid. The caller should hold an exclusive lock on the
1421 * pg_database oid for the database to prevent creation of new slots on the db
1422 * or replay from existing slots.
1423 *
1424 * Another session that concurrently acquires an existing slot on the target DB
1425 * (most likely to drop it) may cause this function to ERROR. If that happens
1426 * it may have dropped some but not all slots.
1427 *
1428 * This routine isn't as efficient as it could be - but we don't drop
1429 * databases often, especially databases with lots of slots.
1430 */
1431void
1432ReplicationSlotsDropDBSlots(Oid dboid)
1433{
1434	int			i;
1435
1436	if (max_replication_slots <= 0)
1437		return;
1438
1439restart:
1440	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
1441	for (i = 0; i < max_replication_slots; i++)
1442	{
1443		ReplicationSlot *s;
1444		char	   *slotname;
1445		int			active_pid;
1446
1447		s = &ReplicationSlotCtl->replication_slots[i];
1448
1449		/* cannot change while ReplicationSlotCtlLock is held */
1450		if (!s->in_use)
1451			continue;
1452
1453		/* only logical slots are database specific, skip */
1454		if (!SlotIsLogical(s))
1455			continue;
1456
1457		/* not our database, skip */
1458		if (s->data.database != dboid)
1459			continue;
1460
1461		/* NB: intentionally including invalidated slots */
1462
1463		/* acquire slot, so ReplicationSlotDropAcquired can be reused  */
1464		SpinLockAcquire(&s->mutex);
1465		/* can't change while ReplicationSlotControlLock is held */
1466		slotname = NameStr(s->data.name);
1467		active_pid = s->active_pid;
1468		if (active_pid == 0)
1469		{
1470			MyReplicationSlot = s;
1471			s->active_pid = MyProcPid;
1472		}
1473		SpinLockRelease(&s->mutex);
1474
1475		/*
1476		 * Even though we hold an exclusive lock on the database object a
1477		 * logical slot for that DB can still be active, e.g. if it's
1478		 * concurrently being dropped by a backend connected to another DB.
1479		 *
1480		 * That's fairly unlikely in practice, so we'll just bail out.
1481		 *
1482		 * The slot sync worker holds a shared lock on the database before
1483		 * operating on synced logical slots to avoid conflict with the drop
1484		 * happening here. The persistent synced slots are thus safe but there
1485		 * is a possibility that the slot sync worker has created a temporary
1486		 * slot (which stays active even on release) and we are trying to drop
1487		 * that here. In practice, the chances of hitting this scenario are
1488		 * less as during slot synchronization, the temporary slot is
1489		 * immediately converted to persistent and thus is safe due to the
1490		 * shared lock taken on the database. So, we'll just bail out in such
1491		 * a case.
1492		 *
1493		 * XXX: We can consider shutting down the slot sync worker before
1494		 * trying to drop synced temporary slots here.
1495		 */
1496		if (active_pid)
1497			ereport(ERROR,
1498					(errcode(ERRCODE_OBJECT_IN_USE),
1499					 errmsg("replication slot \"%s\" is active for PID %d",
1500							slotname, active_pid)));
1501
1502		/*
1503		 * To avoid duplicating ReplicationSlotDropAcquired() and to avoid
1504		 * holding ReplicationSlotControlLock over filesystem operations,
1505		 * release ReplicationSlotControlLock and use
1506		 * ReplicationSlotDropAcquired.
1507		 *
1508		 * As that means the set of slots could change, restart scan from the
1509		 * beginning each time we release the lock.
1510		 */
1511		LWLockRelease(ReplicationSlotControlLock);
1512		ReplicationSlotDropAcquired();
1513		goto restart;
1514	}
1515	LWLockRelease(ReplicationSlotControlLock);
1516}

1556/*
1557 * Reserve WAL for the currently active slot.
1558 *
1559 * Compute and set restart_lsn in a manner that's appropriate for the type of
1560 * the slot and concurrency safe.
1561 */
1562void
1563ReplicationSlotReserveWal(void)
1564{
1565	ReplicationSlot *slot = MyReplicationSlot;
1566	XLogSegNo	segno;
1567	XLogRecPtr	restart_lsn;
1568
1569	Assert(slot != NULL);
1570	Assert(slot->data.restart_lsn == InvalidXLogRecPtr);
1571	Assert(slot->last_saved_restart_lsn == InvalidXLogRecPtr);
1572
1573	/*
1574	 * The replication slot mechanism is used to prevent the removal of
1575	 * required WAL.
1576	 *
1577	 * Acquire an exclusive lock to prevent the checkpoint process from
1578	 * concurrently computing the minimum slot LSN (see
1579	 * CheckPointReplicationSlots). This ensures that the WAL reserved for
1580	 * replication cannot be removed during a checkpoint.
1581	 *
1582	 * The mechanism is reliable because if WAL reservation occurs first, the
1583	 * checkpoint must wait for the restart_lsn update before determining the
1584	 * minimum non-removable LSN. On the other hand, if the checkpoint happens
1585	 * first, subsequent WAL reservations will select positions at or beyond
1586	 * the redo pointer of that checkpoint.
1587	 */
1588	LWLockAcquire(ReplicationSlotAllocationLock, LW_EXCLUSIVE);
1589
1590	/*
1591	 * For logical slots log a standby snapshot and start logical decoding at
1592	 * exactly that position. That allows the slot to start up more quickly.
1593	 * But on a standby we cannot do WAL writes, so just use the replay
1594	 * pointer; effectively, an attempt to create a logical slot on standby
1595	 * will cause it to wait for an xl_running_xact record to be logged
1596	 * independently on the primary, so that a snapshot can be built using the
1597	 * record.
1598	 *
1599	 * None of this is needed (or indeed helpful) for physical slots as
1600	 * they'll start replay at the last logged checkpoint anyway. Instead,
1601	 * return the location of the last redo LSN, where a base backup has to
1602	 * start replay at.
1603	 */
1604	if (SlotIsPhysical(slot))
1605		restart_lsn = GetRedoRecPtr();
1606	else if (RecoveryInProgress())
1607		restart_lsn = GetXLogReplayRecPtr(NULL);
1608	else
1609		restart_lsn = GetXLogInsertRecPtr();
1610
1611	SpinLockAcquire(&slot->mutex);
1612	slot->data.restart_lsn = restart_lsn;
1613	SpinLockRelease(&slot->mutex);
1614
1615	/* prevent WAL removal as fast as possible */
1616	ReplicationSlotsComputeRequiredLSN();
1617
1618	/* Checkpoint shouldn't remove the required WAL. */
1619	XLByteToSeg(slot->data.restart_lsn, segno, wal_segment_size);
1620	if (XLogGetLastRemovedSegno() >= segno)
1621		elog(ERROR, "WAL required by replication slot %s has been removed concurrently",
1622			 NameStr(slot->data.name));
1623
1624	LWLockRelease(ReplicationSlotAllocationLock);
1625
1626	if (!RecoveryInProgress() && SlotIsLogical(slot))
1627	{
1628		XLogRecPtr	flushptr;
1629
1630		/* make sure we have enough information to start */
1631		flushptr = LogStandbySnapshot();
1632
1633		/* and make sure it's fsynced to disk */
1634		XLogFlush(flushptr);
1635	}
1636}

1638/*
1639 * Report that replication slot needs to be invalidated
1640 */
1641static void
1642ReportSlotInvalidation(ReplicationSlotInvalidationCause cause,
1643					   bool terminating,
1644					   int pid,
1645					   NameData slotname,
1646					   XLogRecPtr restart_lsn,
1647					   XLogRecPtr oldestLSN,
1648					   TransactionId snapshotConflictHorizon,
1649					   long slot_idle_seconds)
1650{
1651	StringInfoData err_detail;
1652	StringInfoData err_hint;
1653
1654	initStringInfo(&err_detail);
1655	initStringInfo(&err_hint);
1656
1657	switch (cause)
1658	{
1659		case RS_INVAL_WAL_REMOVED:
1660			{
1661				uint64		ex = oldestLSN - restart_lsn;
1662
1663				appendStringInfo(&err_detail,
1664								 ngettext("The slot's restart_lsn %X/%X exceeds the limit by %" PRIu64 " byte.",
1665										  "The slot's restart_lsn %X/%X exceeds the limit by %" PRIu64 " bytes.",
1666										  ex),
1667								 LSN_FORMAT_ARGS(restart_lsn),
1668								 ex);
1669				/* translator: %s is a GUC variable name */
1670				appendStringInfo(&err_hint, _("You might need to increase \"%s\"."),
1671								 "max_slot_wal_keep_size");
1672				break;
1673			}
1674		case RS_INVAL_HORIZON:
1675			appendStringInfo(&err_detail, _("The slot conflicted with xid horizon %u."),
1676							 snapshotConflictHorizon);
1677			break;
1678
1679		case RS_INVAL_WAL_LEVEL:
1680			appendStringInfoString(&err_detail, _("Logical decoding on standby requires \"wal_level\" >= \"logical\" on the primary server."));
1681			break;
1682
1683		case RS_INVAL_IDLE_TIMEOUT:
1684			{
1685				/* translator: %s is a GUC variable name */
1686				appendStringInfo(&err_detail, _("The slot's idle time of %lds exceeds the configured \"%s\" duration of %ds."),
1687								 slot_idle_seconds, "idle_replication_slot_timeout",
1688								 idle_replication_slot_timeout_secs);
1689				/* translator: %s is a GUC variable name */
1690				appendStringInfo(&err_hint, _("You might need to increase \"%s\"."),
1691								 "idle_replication_slot_timeout");
1692				break;
1693			}
1694		case RS_INVAL_NONE:
1695			pg_unreachable();
1696	}
1697
1698	ereport(LOG,
1699			terminating ?
1700			errmsg("terminating process %d to release replication slot \"%s\"",
1701				   pid, NameStr(slotname)) :
1702			errmsg("invalidating obsolete replication slot \"%s\"",
1703				   NameStr(slotname)),
1704			errdetail_internal("%s", err_detail.data),
1705			err_hint.len ? errhint("%s", err_hint.data) : 0);
1706
1707	pfree(err_detail.data);
1708	pfree(err_hint.data);
1709}

1711/*
1712 * Can we invalidate an idle replication slot?
1713 *
1714 * Idle timeout invalidation is allowed only when:
1715 *
1716 * 1. Idle timeout is set
1717 * 2. Slot has reserved WAL
1718 * 3. Slot is inactive
1719 * 4. The slot is not being synced from the primary while the server is in
1720 *	  recovery. This is because synced slots are always considered to be
1721 *	  inactive because they don't perform logical decoding to produce changes.
1722 */
1723static inline bool
1724CanInvalidateIdleSlot(ReplicationSlot *s)
1725{
1726	return (idle_replication_slot_timeout_secs != 0 &&
1727			!XLogRecPtrIsInvalid(s->data.restart_lsn) &&
1728			s->inactive_since > 0 &&
1729			!(RecoveryInProgress() && s->data.synced));
1730}
1731
1732/*
1733 * DetermineSlotInvalidationCause - Determine the cause for which a slot
1734 * becomes invalid among the given possible causes.
1735 *
1736 * This function sequentially checks all possible invalidation causes and
1737 * returns the first one for which the slot is eligible for invalidation.
1738 */
1739static ReplicationSlotInvalidationCause
1740DetermineSlotInvalidationCause(uint32 possible_causes, ReplicationSlot *s,
1741							   XLogRecPtr oldestLSN, Oid dboid,
1742							   TransactionId snapshotConflictHorizon,
1743							   TimestampTz *inactive_since, TimestampTz now)
1744{
1745	Assert(possible_causes != RS_INVAL_NONE);
1746
1747	if (possible_causes & RS_INVAL_WAL_REMOVED)
1748	{
1749		XLogRecPtr	restart_lsn = s->data.restart_lsn;
1750
1751		if (restart_lsn != InvalidXLogRecPtr &&
1752			restart_lsn < oldestLSN)
1753			return RS_INVAL_WAL_REMOVED;
1754	}
1755
1756	if (possible_causes & RS_INVAL_HORIZON)
1757	{
1758		/* invalid DB oid signals a shared relation */
1759		if (SlotIsLogical(s) &&
1760			(dboid == InvalidOid || dboid == s->data.database))
1761		{
1762			TransactionId effective_xmin = s->effective_xmin;
1763			TransactionId catalog_effective_xmin = s->effective_catalog_xmin;
1764
1765			if (TransactionIdIsValid(effective_xmin) &&
1766				TransactionIdPrecedesOrEquals(effective_xmin,
1767											  snapshotConflictHorizon))
1768				return RS_INVAL_HORIZON;
1769			else if (TransactionIdIsValid(catalog_effective_xmin) &&
1770					 TransactionIdPrecedesOrEquals(catalog_effective_xmin,
1771												   snapshotConflictHorizon))
1772				return RS_INVAL_HORIZON;
1773		}
1774	}
1775
1776	if (possible_causes & RS_INVAL_WAL_LEVEL)
1777	{
1778		if (SlotIsLogical(s))
1779			return RS_INVAL_WAL_LEVEL;
1780	}
1781
1782	if (possible_causes & RS_INVAL_IDLE_TIMEOUT)
1783	{
1784		Assert(now > 0);
1785
1786		if (CanInvalidateIdleSlot(s))
1787		{
1788			/*
1789			 * Simulate the invalidation due to idle_timeout to test the
1790			 * timeout behavior promptly, without waiting for it to trigger
1791			 * naturally.
1792			 */
1793#ifdef USE_INJECTION_POINTS
1794			if (IS_INJECTION_POINT_ATTACHED("slot-timeout-inval"))
1795			{
1796				*inactive_since = 0;	/* since the beginning of time */
1797				return RS_INVAL_IDLE_TIMEOUT;
1798			}
1799#endif
1800
1801			/*
1802			 * Check if the slot needs to be invalidated due to
1803			 * idle_replication_slot_timeout GUC.
1804			 */
1805			if (TimestampDifferenceExceedsSeconds(s->inactive_since, now,
1806												  idle_replication_slot_timeout_secs))
1807			{
1808				*inactive_since = s->inactive_since;
1809				return RS_INVAL_IDLE_TIMEOUT;
1810			}
1811		}
1812	}
1813
1814	return RS_INVAL_NONE;
1815}

1817/*
1818 * Helper for InvalidateObsoleteReplicationSlots
1819 *
1820 * Acquires the given slot and mark it invalid, if necessary and possible.
1821 *
1822 * Returns whether ReplicationSlotControlLock was released in the interim (and
1823 * in that case we're not holding the lock at return, otherwise we are).
1824 *
1825 * Sets *invalidated true if the slot was invalidated. (Untouched otherwise.)
1826 *
1827 * This is inherently racy, because we release the LWLock
1828 * for syscalls, so caller must restart if we return true.
1829 */
1830static bool
1831InvalidatePossiblyObsoleteSlot(uint32 possible_causes,
1832							   ReplicationSlot *s,
1833							   XLogRecPtr oldestLSN,
1834							   Oid dboid, TransactionId snapshotConflictHorizon,
1835							   bool *invalidated)
1836{
1837	int			last_signaled_pid = 0;
1838	bool		released_lock = false;
1839	TimestampTz inactive_since = 0;
1840
1841	for (;;)
1842	{
1843		XLogRecPtr	restart_lsn;
1844		NameData	slotname;
1845		int			active_pid = 0;
1846		ReplicationSlotInvalidationCause invalidation_cause = RS_INVAL_NONE;
1847		TimestampTz now = 0;
1848		long		slot_idle_secs = 0;
1849
1850		Assert(LWLockHeldByMeInMode(ReplicationSlotControlLock, LW_SHARED));
1851
1852		if (!s->in_use)
1853		{
1854			if (released_lock)
1855				LWLockRelease(ReplicationSlotControlLock);
1856			break;
1857		}
1858
1859		if (possible_causes & RS_INVAL_IDLE_TIMEOUT)
1860		{
1861			/*
1862			 * Assign the current time here to avoid system call overhead
1863			 * while holding the spinlock in subsequent code.
1864			 */
1865			now = GetCurrentTimestamp();
1866		}
1867
1868		/*
1869		 * Check if the slot needs to be invalidated. If it needs to be
1870		 * invalidated, and is not currently acquired, acquire it and mark it
1871		 * as having been invalidated.  We do this with the spinlock held to
1872		 * avoid race conditions -- for example the restart_lsn could move
1873		 * forward, or the slot could be dropped.
1874		 */
1875		SpinLockAcquire(&s->mutex);
1876
1877		restart_lsn = s->data.restart_lsn;
1878
1879		/* we do nothing if the slot is already invalid */
1880		if (s->data.invalidated == RS_INVAL_NONE)
1881			invalidation_cause = DetermineSlotInvalidationCause(possible_causes,
1882																s, oldestLSN,
1883																dboid,
1884																snapshotConflictHorizon,
1885																&inactive_since,
1886																now);
1887
1888		/* if there's no invalidation, we're done */
1889		if (invalidation_cause == RS_INVAL_NONE)
1890		{
1891			SpinLockRelease(&s->mutex);
1892			if (released_lock)
1893				LWLockRelease(ReplicationSlotControlLock);
1894			break;
1895		}
1896
1897		slotname = s->data.name;
1898		active_pid = s->active_pid;
1899
1900		/*
1901		 * If the slot can be acquired, do so and mark it invalidated
1902		 * immediately.  Otherwise we'll signal the owning process, below, and
1903		 * retry.
1904		 *
1905		 * Note: Unlike other slot attributes, slot's inactive_since can't be
1906		 * changed until the acquired slot is released or the owning process
1907		 * is terminated. So, the inactive slot can only be invalidated
1908		 * immediately without being terminated.
1909		 */
1910		if (active_pid == 0)
1911		{
1912			MyReplicationSlot = s;
1913			s->active_pid = MyProcPid;
1914			s->data.invalidated = invalidation_cause;
1915
1916			/*
1917			 * XXX: We should consider not overwriting restart_lsn and instead
1918			 * just rely on .invalidated.
1919			 */
1920			if (invalidation_cause == RS_INVAL_WAL_REMOVED)
1921			{
1922				s->data.restart_lsn = InvalidXLogRecPtr;
1923				s->last_saved_restart_lsn = InvalidXLogRecPtr;
1924			}
1925
1926			/* Let caller know */
1927			*invalidated = true;
1928		}
1929
1930		SpinLockRelease(&s->mutex);
1931
1932		/*
1933		 * Calculate the idle time duration of the slot if slot is marked
1934		 * invalidated with RS_INVAL_IDLE_TIMEOUT.
1935		 */
1936		if (invalidation_cause == RS_INVAL_IDLE_TIMEOUT)
1937		{
1938			int			slot_idle_usecs;
1939
1940			TimestampDifference(inactive_since, now, &slot_idle_secs,
1941								&slot_idle_usecs);
1942		}
1943
1944		if (active_pid != 0)
1945		{
1946			/*
1947			 * Prepare the sleep on the slot's condition variable before
1948			 * releasing the lock, to close a possible race condition if the
1949			 * slot is released before the sleep below.
1950			 */
1951			ConditionVariablePrepareToSleep(&s->active_cv);
1952
1953			LWLockRelease(ReplicationSlotControlLock);
1954			released_lock = true;
1955
1956			/*
1957			 * Signal to terminate the process that owns the slot, if we
1958			 * haven't already signalled it.  (Avoidance of repeated
1959			 * signalling is the only reason for there to be a loop in this
1960			 * routine; otherwise we could rely on caller's restart loop.)
1961			 *
1962			 * There is the race condition that other process may own the slot
1963			 * after its current owner process is terminated and before this
1964			 * process owns it. To handle that, we signal only if the PID of
1965			 * the owning process has changed from the previous time. (This
1966			 * logic assumes that the same PID is not reused very quickly.)
1967			 */
1968			if (last_signaled_pid != active_pid)
1969			{
1970				ReportSlotInvalidation(invalidation_cause, true, active_pid,
1971									   slotname, restart_lsn,
1972									   oldestLSN, snapshotConflictHorizon,
1973									   slot_idle_secs);
1974
1975				if (MyBackendType == B_STARTUP)
1976					(void) SendProcSignal(active_pid,
1977										  PROCSIG_RECOVERY_CONFLICT_LOGICALSLOT,
1978										  INVALID_PROC_NUMBER);
1979				else
1980					(void) kill(active_pid, SIGTERM);
1981
1982				last_signaled_pid = active_pid;
1983			}
1984
1985			/* Wait until the slot is released. */
1986			ConditionVariableSleep(&s->active_cv,
1987								   WAIT_EVENT_REPLICATION_SLOT_DROP);
1988
1989			/*
1990			 * Re-acquire lock and start over; we expect to invalidate the
1991			 * slot next time (unless another process acquires the slot in the
1992			 * meantime).
1993			 *
1994			 * Note: It is possible for a slot to advance its restart_lsn or
1995			 * xmin values sufficiently between when we release the mutex and
1996			 * when we recheck, moving from a conflicting state to a non
1997			 * conflicting state.  This is intentional and safe: if the slot
1998			 * has caught up while we're busy here, the resources we were
1999			 * concerned about (WAL segments or tuples) have not yet been
2000			 * removed, and there's no reason to invalidate the slot.
2001			 */
2002			LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
2003			continue;
2004		}
2005		else
2006		{
2007			/*
2008			 * We hold the slot now and have already invalidated it; flush it
2009			 * to ensure that state persists.
2010			 *
2011			 * Don't want to hold ReplicationSlotControlLock across file
2012			 * system operations, so release it now but be sure to tell caller
2013			 * to restart from scratch.
2014			 */
2015			LWLockRelease(ReplicationSlotControlLock);
2016			released_lock = true;
2017
2018			/* Make sure the invalidated state persists across server restart */
2019			ReplicationSlotMarkDirty();
2020			ReplicationSlotSave();
2021			ReplicationSlotRelease();
2022

2038/*
2039 * Invalidate slots that require resources about to be removed.
2040 *
2041 * Returns true when any slot have got invalidated.
2042 *
2043 * Whether a slot needs to be invalidated depends on the invalidation cause.
2044 * A slot is invalidated if it:
2045 * - RS_INVAL_WAL_REMOVED: requires a LSN older than the given segment
2046 * - RS_INVAL_HORIZON: requires a snapshot <= the given horizon in the given
2047 *   db; dboid may be InvalidOid for shared relations
2048 * - RS_INVAL_WAL_LEVEL: is logical and wal_level is insufficient
2049 * - RS_INVAL_IDLE_TIMEOUT: has been idle longer than the configured
2050 *   "idle_replication_slot_timeout" duration.
2051 *
2052 * Note: This function attempts to invalidate the slot for multiple possible
2053 * causes in a single pass, minimizing redundant iterations. The "cause"
2054 * parameter can be a MASK representing one or more of the defined causes.
2055 *
2056 * NB - this runs as part of checkpoint, so avoid raising errors if possible.
2057 */
2058bool
2059InvalidateObsoleteReplicationSlots(uint32 possible_causes,
2060								   XLogSegNo oldestSegno, Oid dboid,
2061								   TransactionId snapshotConflictHorizon)
2062{
2063	XLogRecPtr	oldestLSN;
2064	bool		invalidated = false;
2065
2066	Assert(!(possible_causes & RS_INVAL_HORIZON) || TransactionIdIsValid(snapshotConflictHorizon));
2067	Assert(!(possible_causes & RS_INVAL_WAL_REMOVED) || oldestSegno > 0);
2068	Assert(possible_causes != RS_INVAL_NONE);
2069
2070	if (max_replication_slots == 0)
2071		return invalidated;
2072
2073	XLogSegNoOffsetToRecPtr(oldestSegno, 0, wal_segment_size, oldestLSN);
2074
2075restart:
2076	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
2077	for (int i = 0; i < max_replication_slots; i++)
2078	{
2079		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
2080
2081		if (!s->in_use)
2082			continue;
2083
2084		/* Prevent invalidation of logical slots during binary upgrade */
2085		if (SlotIsLogical(s) && IsBinaryUpgrade)
2086			continue;
2087
2088		if (InvalidatePossiblyObsoleteSlot(possible_causes, s, oldestLSN, dboid,
2089										   snapshotConflictHorizon,
2090										   &invalidated))
2091		{
2092			/* if the lock was released, start from scratch */
2093			goto restart;
2094		}
2095	}
2096	LWLockRelease(ReplicationSlotControlLock);
2097
2098	/*
2099	 * If any slots have been invalidated, recalculate the resource limits.
2100	 */
2101	if (invalidated)
2102	{
2103		ReplicationSlotsComputeRequiredXmin(false);
2104		ReplicationSlotsComputeRequiredLSN();
2105	}
2106
2107	return invalidated;
2108}

2110/*
2111 * Flush all replication slots to disk.
2112 *
2113 * It is convenient to flush dirty replication slots at the time of checkpoint.
2114 * Additionally, in case of a shutdown checkpoint, we also identify the slots
2115 * for which the confirmed_flush LSN has been updated since the last time it
2116 * was saved and flush them.
2117 */
2118void
2119CheckPointReplicationSlots(bool is_shutdown)
2120{
2121	int			i;
2122	bool		last_saved_restart_lsn_updated = false;
2123
2124	elog(DEBUG1, "performing replication slot checkpoint");
2125
2126	/*
2127	 * Prevent any slot from being created/dropped while we're active. As we
2128	 * explicitly do *not* want to block iterating over replication_slots or
2129	 * acquiring a slot we cannot take the control lock - but that's OK,
2130	 * because holding ReplicationSlotAllocationLock is strictly stronger, and
2131	 * enough to guarantee that nobody can change the in_use bits on us.
2132	 *
2133	 * Additionally, acquiring the Allocation lock is necessary to serialize
2134	 * the slot flush process with concurrent slot WAL reservation. This
2135	 * ensures that the WAL position being reserved is either flushed to disk
2136	 * or is beyond or equal to the redo pointer of the current checkpoint
2137	 * (See ReplicationSlotReserveWal for details).
2138	 */
2139	LWLockAcquire(ReplicationSlotAllocationLock, LW_SHARED);
2140
2141	for (i = 0; i < max_replication_slots; i++)
2142	{
2143		ReplicationSlot *s = &ReplicationSlotCtl->replication_slots[i];
2144		char		path[MAXPGPATH];
2145
2146		if (!s->in_use)
2147			continue;
2148
2149		/* save the slot to disk, locking is handled in SaveSlotToPath() */
2150		sprintf(path, "%s/%s", PG_REPLSLOT_DIR, NameStr(s->data.name));
2151
2152		/*
2153		 * Slot's data is not flushed each time the confirmed_flush LSN is
2154		 * updated as that could lead to frequent writes.  However, we decide
2155		 * to force a flush of all logical slot's data at the time of shutdown
2156		 * if the confirmed_flush LSN is changed since we last flushed it to
2157		 * disk.  This helps in avoiding an unnecessary retreat of the
2158		 * confirmed_flush LSN after restart.
2159		 */
2160		if (is_shutdown && SlotIsLogical(s))
2161		{
2162			SpinLockAcquire(&s->mutex);
2163
2164			if (s->data.invalidated == RS_INVAL_NONE &&
2165				s->data.confirmed_flush > s->last_saved_confirmed_flush)
2166			{
2167				s->just_dirtied = true;
2168				s->dirty = true;
2169			}
2170			SpinLockRelease(&s->mutex);
2171		}
2172
2173		/*
2174		 * Track if we're going to update slot's last_saved_restart_lsn. We
2175		 * need this to know if we need to recompute the required LSN.
2176		 */
2177		if (s->last_saved_restart_lsn != s->data.restart_lsn)
2178			last_saved_restart_lsn_updated = true;
2179
2180		SaveSlotToPath(s, path, LOG);
2181	}
2182	LWLockRelease(ReplicationSlotAllocationLock);
2183
2184	/*
2185	 * Recompute the required LSN if SaveSlotToPath() updated
2186	 * last_saved_restart_lsn for any slot.
2187	 */
2188	if (last_saved_restart_lsn_updated)
2189		ReplicationSlotsComputeRequiredLSN();
2190}

2250/* ----
2251 * Manipulation of on-disk state of replication slots
2252 *
2253 * NB: none of the routines below should take any notice whether a slot is the
2254 * current one or not, that's all handled a layer above.

2259{

2264	/*
2265	 * No need to take out the io_in_progress_lock, nobody else can see this
2266	 * slot yet, so nobody else will write. We're reusing SaveSlotToPath which
2267	 * takes out the lock, if we'd take the lock here, we'd deadlock.

2297		ereport(ERROR,

2300						tmppath, path)));
2301

2307	START_CRIT_SECTION();

2312	END_CRIT_SECTION();
2313}

2638	 * CheckSlotRequirements() and CheckLogicalDecodingRequirements().

2640	if (cp.slotdata.database != InvalidOid)
2641	{
2642		if (wal_level < WAL_LEVEL_LOGICAL)
2643			ereport(FATAL,
2644					(errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
2645					 errmsg("logical replication slot \"%s\" exists, but \"wal_level\" < \"logical\"",
2646							NameStr(cp.slotdata.name)),
2647					 errhint("Change \"wal_level\" to be \"logical\" or higher.")));
2648
2649		/*
2650		 * In standby mode, the hot standby must be enabled. This check is
2651		 * necessary to ensure logical slots are invalidated when they become
2652		 * incompatible due to insufficient wal_level. Otherwise, if the
2653		 * primary reduces wal_level < logical while hot standby is disabled,
2654		 * logical slots would remain valid even after promotion.
2655		 */
2656		if (StandbyMode && !EnableHotStandby)
2657			ereport(FATAL,
2658					(errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
2659					 errmsg("logical replication slot \"%s\" exists on the standby, but \"hot_standby\" = \"off\"",
2660							NameStr(cp.slotdata.name)),
2661					 errhint("Change \"hot_standby\" to be \"on\".")));
2662	}
2663	else if (wal_level < WAL_LEVEL_REPLICA)
2664		ereport(FATAL,
2665				(errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
2666				 errmsg("physical replication slot \"%s\" exists, but \"wal_level\" < \"replica\"",
2667						NameStr(cp.slotdata.name)),
2668				 errhint("Change \"wal_level\" to be \"replica\" or higher.")));
2669
2670	/* nothing can be active yet, don't lock anything */
2671	for (i = 0; i < max_replication_slots; i++)
2672	{
2673		ReplicationSlot *slot;
2674
2675		slot = &ReplicationSlotCtl->replication_slots[i];
2676
2677		if (slot->in_use)
2678			continue;
2679
2680		/* restore the entire set of persistent data */
2681		memcpy(&slot->data, &cp.slotdata,
2682			   sizeof(ReplicationSlotPersistentData));
2683
2684		/* initialize in memory state */
2685		slot->effective_xmin = cp.slotdata.xmin;
2686		slot->effective_catalog_xmin = cp.slotdata.catalog_xmin;
2687		slot->last_saved_confirmed_flush = cp.slotdata.confirmed_flush;
2688		slot->last_saved_restart_lsn = cp.slotdata.restart_lsn;
2689
2690		slot->candidate_catalog_xmin = InvalidTransactionId;
2691		slot->candidate_xmin_lsn = InvalidXLogRecPtr;
2692		slot->candidate_restart_lsn = InvalidXLogRecPtr;
2693		slot->candidate_restart_valid = InvalidXLogRecPtr;
2694
2695		slot->in_use = true;
2696		slot->active_pid = 0;
2697
2698		/*
2699		 * Set the time since the slot has become inactive after loading the
2700		 * slot from the disk into memory. Whoever acquires the slot i.e.
2701		 * makes the slot active will reset it. Use the same inactive_since
2702		 * time for all the slots.
2703		 */
2704		if (now == 0)
2705			now = GetCurrentTimestamp();
2706
2707		ReplicationSlotSetInactiveSince(slot, now, false);
2708
2709		restored = true;
2710		break;
2711	}
2712
2713	if (!restored)
2714		ereport(FATAL,
2715				(errmsg("too many replication slots active before shutdown"),
2716				 errhint("Increase \"max_replication_slots\" and try again.")));
2717}

2893/*
2894 * Return true if the slots specified in synchronized_standby_slots have caught up to
2895 * the given WAL location, false otherwise.
2896 *
2897 * The elevel parameter specifies the error level used for logging messages
2898 * related to slots that do not exist, are invalidated, or are inactive.
2899 */
2900bool
2901StandbySlotsHaveCaughtup(XLogRecPtr wait_for_lsn, int elevel)
2902{
2903	const char *name;
2904	int			caught_up_slot_num = 0;
2905	XLogRecPtr	min_restart_lsn = InvalidXLogRecPtr;
2906
2907	/*
2908	 * Don't need to wait for the standbys to catch up if there is no value in
2909	 * synchronized_standby_slots.
2910	 */
2911	if (synchronized_standby_slots_config == NULL)
2912		return true;
2913
2914	/*
2915	 * Don't need to wait for the standbys to catch up if we are on a standby
2916	 * server, since we do not support syncing slots to cascading standbys.
2917	 */
2918	if (RecoveryInProgress())
2919		return true;
2920
2921	/*
2922	 * Don't need to wait for the standbys to catch up if they are already
2923	 * beyond the specified WAL location.
2924	 */
2925	if (!XLogRecPtrIsInvalid(ss_oldest_flush_lsn) &&
2926		ss_oldest_flush_lsn >= wait_for_lsn)
2927		return true;
2928
2929	/*
2930	 * To prevent concurrent slot dropping and creation while filtering the
2931	 * slots, take the ReplicationSlotControlLock outside of the loop.
2932	 */
2933	LWLockAcquire(ReplicationSlotControlLock, LW_SHARED);
2934
2935	name = synchronized_standby_slots_config->slot_names;
2936	for (int i = 0; i < synchronized_standby_slots_config->nslotnames; i++)
2937	{
2938		XLogRecPtr	restart_lsn;
2939		bool		invalidated;
2940		bool		inactive;
2941		ReplicationSlot *slot;
2942
2943		slot = SearchNamedReplicationSlot(name, false);
2944
2945		/*
2946		 * If a slot name provided in synchronized_standby_slots does not
2947		 * exist, report a message and exit the loop.
2948		 */
2949		if (!slot)
2950		{
2951			ereport(elevel,
2952					errcode(ERRCODE_INVALID_PARAMETER_VALUE),
2953					errmsg("replication slot \"%s\" specified in parameter \"%s\" does not exist",
2954						   name, "synchronized_standby_slots"),
2955					errdetail("Logical replication is waiting on the standby associated with replication slot \"%s\".",
2956							  name),
2957					errhint("Create the replication slot \"%s\" or amend parameter \"%s\".",
2958							name, "synchronized_standby_slots"));
2959			break;
2960		}
2961
2962		/* Same as above: if a slot is not physical, exit the loop. */
2963		if (SlotIsLogical(slot))
2964		{
2965			ereport(elevel,
2966					errcode(ERRCODE_INVALID_PARAMETER_VALUE),
2967					errmsg("cannot specify logical replication slot \"%s\" in parameter \"%s\"",
2968						   name, "synchronized_standby_slots"),
2969					errdetail("Logical replication is waiting for correction on replication slot \"%s\".",
2970							  name),
2971					errhint("Remove the logical replication slot \"%s\" from parameter \"%s\".",
2972							name, "synchronized_standby_slots"));
2973			break;
2974		}
2975
2976		SpinLockAcquire(&slot->mutex);
2977		restart_lsn = slot->data.restart_lsn;
2978		invalidated = slot->data.invalidated != RS_INVAL_NONE;
2979		inactive = slot->active_pid == 0;
2980		SpinLockRelease(&slot->mutex);
2981
2982		if (invalidated)
2983		{
2984			/* Specified physical slot has been invalidated */
2985			ereport(elevel,
2986					errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
2987					errmsg("physical replication slot \"%s\" specified in parameter \"%s\" has been invalidated",
2988						   name, "synchronized_standby_slots"),
2989					errdetail("Logical replication is waiting on the standby associated with replication slot \"%s\".",
2990							  name),
2991					errhint("Drop and recreate the replication slot \"%s\", or amend parameter \"%s\".",
2992							name, "synchronized_standby_slots"));
2993			break;
2994		}
2995
2996		if (XLogRecPtrIsInvalid(restart_lsn) || restart_lsn < wait_for_lsn)
2997		{
2998			/* Log a message if no active_pid for this physical slot */
2999			if (inactive)
3000				ereport(elevel,
3001						errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
3002						errmsg("replication slot \"%s\" specified in parameter \"%s\" does not have active_pid",
3003							   name, "synchronized_standby_slots"),
3004						errdetail("Logical replication is waiting on the standby associated with replication slot \"%s\".",
3005								  name),
3006						errhint("Start the standby associated with the replication slot \"%s\", or amend parameter \"%s\".",
3007								name, "synchronized_standby_slots"));
3008
3009			/* Continue if the current slot hasn't caught up. */
3010			break;
3011		}
3012
3013		Assert(restart_lsn >= wait_for_lsn);
3014
3015		if (XLogRecPtrIsInvalid(min_restart_lsn) ||
3016			min_restart_lsn > restart_lsn)
3017			min_restart_lsn = restart_lsn;
3018
3019		caught_up_slot_num++;
3020
3021		name += strlen(name) + 1;
3022	}
3023
3024	LWLockRelease(ReplicationSlotControlLock);
3025
3026	/*
3027	 * Return false if not all the standbys have caught up to the specified
3028	 * WAL location.
3029	 */
3030	if (caught_up_slot_num != synchronized_standby_slots_config->nslotnames)
3031		return false;
3032
3033	/* The ss_oldest_flush_lsn must not retreat. */
3034	Assert(XLogRecPtrIsInvalid(ss_oldest_flush_lsn) ||
3035		   min_restart_lsn >= ss_oldest_flush_lsn);
3036
3037	ss_oldest_flush_lsn = min_restart_lsn;
3038
3039	return true;
3040}

3042/*
3043 * Wait for physical standbys to confirm receiving the given lsn.
3044 *
3045 * Used by logical decoding SQL functions. It waits for physical standbys
3046 * corresponding to the physical slots specified in the synchronized_standby_slots GUC.
3047 */
3048void
3049WaitForStandbyConfirmation(XLogRecPtr wait_for_lsn)
3050{
3051	/*
3052	 * Don't need to wait for the standby to catch up if the current acquired
3053	 * slot is not a logical failover slot, or there is no value in
3054	 * synchronized_standby_slots.
3055	 */
3056	if (!MyReplicationSlot->data.failover || !synchronized_standby_slots_config)
3057		return;
3058
3059	ConditionVariablePrepareToSleep(&WalSndCtl->wal_confirm_rcv_cv);
3060
3061	for (;;)
3062	{
3063		CHECK_FOR_INTERRUPTS();
3064
3065		if (ConfigReloadPending)
3066		{
3067			ConfigReloadPending = false;
3068			ProcessConfigFile(PGC_SIGHUP);
3069		}
3070
3071		/* Exit if done waiting for every slot. */
3072		if (StandbySlotsHaveCaughtup(wait_for_lsn, WARNING))
3073			break;
3074
3075		/*
3076		 * Wait for the slots in the synchronized_standby_slots to catch up,
3077		 * but use a timeout (1s) so we can also check if the
3078		 * synchronized_standby_slots has been changed.
3079		 */
3080		ConditionVariableTimedSleep(&WalSndCtl->wal_confirm_rcv_cv, 1000,
3081									WAIT_EVENT_WAIT_FOR_STANDBY_CONFIRMATION);
3082	}
3083
3084	ConditionVariableCancelSleep();
