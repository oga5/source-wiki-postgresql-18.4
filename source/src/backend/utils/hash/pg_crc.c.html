102/*
103 * SQL-callable functions
104 */
105
106Datum
107crc32_bytea(PG_FUNCTION_ARGS)
108{
109	bytea	   *in = PG_GETARG_BYTEA_PP(0);
110	pg_crc32	crc;
111
112	INIT_TRADITIONAL_CRC32(crc);
113	COMP_TRADITIONAL_CRC32(crc, VARDATA_ANY(in), VARSIZE_ANY_EXHDR(in));
114	FIN_TRADITIONAL_CRC32(crc);
115
116	PG_RETURN_INT64(crc);
117}
118
119Datum
120crc32c_bytea(PG_FUNCTION_ARGS)
121{
122	bytea	   *in = PG_GETARG_BYTEA_PP(0);
123	pg_crc32c	crc;
124
125	INIT_CRC32C(crc);
126	COMP_CRC32C(crc, VARDATA_ANY(in), VARSIZE_ANY_EXHDR(in));
127	FIN_CRC32C(crc);
128
129	PG_RETURN_INT64(crc);
130}
