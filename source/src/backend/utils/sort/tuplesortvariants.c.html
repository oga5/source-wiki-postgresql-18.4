179Tuplesortstate *
180tuplesort_begin_heap(TupleDesc tupDesc,
181					 int nkeys, AttrNumber *attNums,
182					 Oid *sortOperators, Oid *sortCollations,
183					 bool *nullsFirstFlags,
184					 int workMem, SortCoordinate coordinate, int sortopt)
185{
186	Tuplesortstate *state = tuplesort_begin_common(workMem, coordinate,
187												   sortopt);
188	TuplesortPublic *base = TuplesortstateGetPublic(state);
189	MemoryContext oldcontext;
190	int			i;
191
192	oldcontext = MemoryContextSwitchTo(base->maincontext);
193
194	Assert(nkeys > 0);
195
196	if (trace_sort)
197		elog(LOG,
198			 "begin tuple sort: nkeys = %d, workMem = %d, randomAccess = %c",
199			 nkeys, workMem, sortopt & TUPLESORT_RANDOMACCESS ? 't' : 'f');
200
201	base->nKeys = nkeys;
202
203	TRACE_POSTGRESQL_SORT_START(HEAP_SORT,
204								false,	/* no unique check */
205								nkeys,
206								workMem,
207								sortopt & TUPLESORT_RANDOMACCESS,
208								PARALLEL_SORT(coordinate));
209
210	base->removeabbrev = removeabbrev_heap;
211	base->comparetup = comparetup_heap;
212	base->comparetup_tiebreak = comparetup_heap_tiebreak;
213	base->writetup = writetup_heap;
214	base->readtup = readtup_heap;
215	base->haveDatum1 = true;
216	base->arg = tupDesc;		/* assume we need not copy tupDesc */
217
218	/* Prepare SortSupport data for each column */
219	base->sortKeys = (SortSupport) palloc0(nkeys * sizeof(SortSupportData));
220
221	for (i = 0; i < nkeys; i++)
222	{
223		SortSupport sortKey = base->sortKeys + i;
224
225		Assert(attNums[i] != 0);
226		Assert(sortOperators[i] != 0);
227
228		sortKey->ssup_cxt = CurrentMemoryContext;
229		sortKey->ssup_collation = sortCollations[i];
230		sortKey->ssup_nulls_first = nullsFirstFlags[i];
231		sortKey->ssup_attno = attNums[i];
232		/* Convey if abbreviation optimization is applicable in principle */
233		sortKey->abbreviate = (i == 0 && base->haveDatum1);
234
235		PrepareSortSupportFromOrderingOp(sortOperators[i], sortKey);
236	}
237
238	/*
239	 * The "onlyKey" optimization cannot be used with abbreviated keys, since
240	 * tie-breaker comparisons may be required.  Typically, the optimization
241	 * is only of value to pass-by-value types anyway, whereas abbreviated
242	 * keys are typically only of value to pass-by-reference types.
243	 */
244	if (nkeys == 1 && !base->sortKeys->abbrev_converter)
245		base->onlyKey = base->sortKeys;
246
247	MemoryContextSwitchTo(oldcontext);
248
249	return state;
250}

252Tuplesortstate *
253tuplesort_begin_cluster(TupleDesc tupDesc,
254						Relation indexRel,
255						int workMem,
256						SortCoordinate coordinate, int sortopt)
257{
258	Tuplesortstate *state = tuplesort_begin_common(workMem, coordinate,
259												   sortopt);
260	TuplesortPublic *base = TuplesortstateGetPublic(state);
261	BTScanInsert indexScanKey;
262	MemoryContext oldcontext;
263	TuplesortClusterArg *arg;
264	int			i;
265
266	Assert(indexRel->rd_rel->relam == BTREE_AM_OID);
267
268	oldcontext = MemoryContextSwitchTo(base->maincontext);
269	arg = (TuplesortClusterArg *) palloc0(sizeof(TuplesortClusterArg));
270
271	if (trace_sort)
272		elog(LOG,
273			 "begin tuple sort: nkeys = %d, workMem = %d, randomAccess = %c",
274			 RelationGetNumberOfAttributes(indexRel),
275			 workMem, sortopt & TUPLESORT_RANDOMACCESS ? 't' : 'f');
276
277	base->nKeys = IndexRelationGetNumberOfKeyAttributes(indexRel);
278
279	TRACE_POSTGRESQL_SORT_START(CLUSTER_SORT,
280								false,	/* no unique check */
281								base->nKeys,
282								workMem,
283								sortopt & TUPLESORT_RANDOMACCESS,
284								PARALLEL_SORT(coordinate));
285
286	base->removeabbrev = removeabbrev_cluster;
287	base->comparetup = comparetup_cluster;
288	base->comparetup_tiebreak = comparetup_cluster_tiebreak;
289	base->writetup = writetup_cluster;
290	base->readtup = readtup_cluster;
291	base->freestate = freestate_cluster;
292	base->arg = arg;
293
294	arg->indexInfo = BuildIndexInfo(indexRel);
295
296	/*
297	 * If we don't have a simple leading attribute, we don't currently
298	 * initialize datum1, so disable optimizations that require it.
299	 */
300	if (arg->indexInfo->ii_IndexAttrNumbers[0] == 0)
301		base->haveDatum1 = false;
302	else
303		base->haveDatum1 = true;
304
305	arg->tupDesc = tupDesc;		/* assume we need not copy tupDesc */
306
307	indexScanKey = _bt_mkscankey(indexRel, NULL);
308
309	if (arg->indexInfo->ii_Expressions != NULL)
310	{
311		TupleTableSlot *slot;
312		ExprContext *econtext;
313
314		/*
315		 * We will need to use FormIndexDatum to evaluate the index
316		 * expressions.  To do that, we need an EState, as well as a
317		 * TupleTableSlot to put the table tuples into.  The econtext's
318		 * scantuple has to point to that slot, too.
319		 */
320		arg->estate = CreateExecutorState();
321		slot = MakeSingleTupleTableSlot(tupDesc, &TTSOpsHeapTuple);
322		econtext = GetPerTupleExprContext(arg->estate);
323		econtext->ecxt_scantuple = slot;
324	}
325
326	/* Prepare SortSupport data for each column */
327	base->sortKeys = (SortSupport) palloc0(base->nKeys *
328										   sizeof(SortSupportData));
329
330	for (i = 0; i < base->nKeys; i++)
331	{
332		SortSupport sortKey = base->sortKeys + i;
333		ScanKey		scanKey = indexScanKey->scankeys + i;
334		bool		reverse;
335
336		sortKey->ssup_cxt = CurrentMemoryContext;
337		sortKey->ssup_collation = scanKey->sk_collation;
338		sortKey->ssup_nulls_first =
339			(scanKey->sk_flags & SK_BT_NULLS_FIRST) != 0;
340		sortKey->ssup_attno = scanKey->sk_attno;
341		/* Convey if abbreviation optimization is applicable in principle */
342		sortKey->abbreviate = (i == 0 && base->haveDatum1);
343
344		Assert(sortKey->ssup_attno != 0);
345
346		reverse = (scanKey->sk_flags & SK_BT_DESC) != 0;
347
348		PrepareSortSupportFromIndexRel(indexRel, reverse, sortKey);
349	}
350
351	pfree(indexScanKey);
352
353	MemoryContextSwitchTo(oldcontext);
354
355	return state;
356}

581Tuplesortstate *
582tuplesort_begin_index_gin(Relation heapRel,
583						  Relation indexRel,
584						  int workMem, SortCoordinate coordinate,
585						  int sortopt)
586{
587	Tuplesortstate *state = tuplesort_begin_common(workMem, coordinate,
588												   sortopt);
589	TuplesortPublic *base = TuplesortstateGetPublic(state);
590	MemoryContext oldcontext;
591	int			i;
592	TupleDesc	desc = RelationGetDescr(indexRel);
593
594	oldcontext = MemoryContextSwitchTo(base->maincontext);
595
596#ifdef TRACE_SORT
597	if (trace_sort)
598		elog(LOG,
599			 "begin index sort: workMem = %d, randomAccess = %c",
600			 workMem,
601			 sortopt & TUPLESORT_RANDOMACCESS ? 't' : 'f');
602#endif
603
604	/*
605	 * Multi-column GIN indexes expand the row into a separate index entry for
606	 * attribute, and that's what we write into the tuplesort. But we still
607	 * need to initialize sortsupport for all the attributes.
608	 */
609	base->nKeys = IndexRelationGetNumberOfKeyAttributes(indexRel);
610
611	/* Prepare SortSupport data for each column */
612	base->sortKeys = (SortSupport) palloc0(base->nKeys *
613										   sizeof(SortSupportData));
614
615	for (i = 0; i < base->nKeys; i++)
616	{
617		SortSupport sortKey = base->sortKeys + i;
618		Form_pg_attribute att = TupleDescAttr(desc, i);
619		Oid			cmpFunc;
620
621		sortKey->ssup_cxt = CurrentMemoryContext;
622		sortKey->ssup_collation = indexRel->rd_indcollation[i];
623		sortKey->ssup_nulls_first = false;
624		sortKey->ssup_attno = i + 1;
625		sortKey->abbreviate = false;
626
627		Assert(sortKey->ssup_attno != 0);
628
629		if (!OidIsValid(sortKey->ssup_collation))
630			sortKey->ssup_collation = DEFAULT_COLLATION_OID;
631
632		/*
633		 * If the compare proc isn't specified in the opclass definition, look
634		 * up the index key type's default btree comparator.
635		 */
636		cmpFunc = index_getprocid(indexRel, i + 1, GIN_COMPARE_PROC);
637		if (cmpFunc == InvalidOid)
638		{
639			TypeCacheEntry *typentry;
640
641			typentry = lookup_type_cache(att->atttypid,
642										 TYPECACHE_CMP_PROC_FINFO);
643			if (!OidIsValid(typentry->cmp_proc_finfo.fn_oid))
644				ereport(ERROR,
645						(errcode(ERRCODE_UNDEFINED_FUNCTION),
646						 errmsg("could not identify a comparison function for type %s",
647								format_type_be(att->atttypid))));
648
649			cmpFunc = typentry->cmp_proc_finfo.fn_oid;
650		}
651
652		PrepareSortSupportComparisonShim(cmpFunc, sortKey);
653	}
654
655	base->removeabbrev = removeabbrev_index_gin;
656	base->comparetup = comparetup_index_gin;
657	base->writetup = writetup_index_gin;
658	base->readtup = readtup_index_gin;
659	base->haveDatum1 = false;

661
662	MemoryContextSwitchTo(oldcontext);
663

665}

667Tuplesortstate *
668tuplesort_begin_datum(Oid datumType, Oid sortOperator, Oid sortCollation,
669					  bool nullsFirstFlag, int workMem,
670					  SortCoordinate coordinate, int sortopt)
671{
672	Tuplesortstate *state = tuplesort_begin_common(workMem, coordinate,
673												   sortopt);
674	TuplesortPublic *base = TuplesortstateGetPublic(state);
675	TuplesortDatumArg *arg;
676	MemoryContext oldcontext;
677	int16		typlen;
678	bool		typbyval;
679
680	oldcontext = MemoryContextSwitchTo(base->maincontext);
681	arg = (TuplesortDatumArg *) palloc(sizeof(TuplesortDatumArg));
682
683	if (trace_sort)
684		elog(LOG,
685			 "begin datum sort: workMem = %d, randomAccess = %c",
686			 workMem, sortopt & TUPLESORT_RANDOMACCESS ? 't' : 'f');
687
688	base->nKeys = 1;			/* always a one-column sort */
689
690	TRACE_POSTGRESQL_SORT_START(DATUM_SORT,
691								false,	/* no unique check */
692								1,
693								workMem,
694								sortopt & TUPLESORT_RANDOMACCESS,
695								PARALLEL_SORT(coordinate));
696
697	base->removeabbrev = removeabbrev_datum;
698	base->comparetup = comparetup_datum;
699	base->comparetup_tiebreak = comparetup_datum_tiebreak;
700	base->writetup = writetup_datum;
701	base->readtup = readtup_datum;
702	base->haveDatum1 = true;
703	base->arg = arg;
704
705	arg->datumType = datumType;
706
707	/* lookup necessary attributes of the datum type */
708	get_typlenbyval(datumType, &typlen, &typbyval);
709	arg->datumTypeLen = typlen;
710	base->tuples = !typbyval;
711
712	/* Prepare SortSupport data */
713	base->sortKeys = (SortSupport) palloc0(sizeof(SortSupportData));
714
715	base->sortKeys->ssup_cxt = CurrentMemoryContext;
716	base->sortKeys->ssup_collation = sortCollation;
717	base->sortKeys->ssup_nulls_first = nullsFirstFlag;
718
719	/*
720	 * Abbreviation is possible here only for by-reference types.  In theory,
721	 * a pass-by-value datatype could have an abbreviated form that is cheaper
722	 * to compare.  In a tuple sort, we could support that, because we can
723	 * always extract the original datum from the tuple as needed.  Here, we
724	 * can't, because a datum sort only stores a single copy of the datum; the
725	 * "tuple" field of each SortTuple is NULL.
726	 */
727	base->sortKeys->abbreviate = !typbyval;
728
729	PrepareSortSupportFromOrderingOp(sortOperator, base->sortKeys);
730
731	/*
732	 * The "onlyKey" optimization cannot be used with abbreviated keys, since
733	 * tie-breaker comparisons may be required.  Typically, the optimization
734	 * is only of value to pass-by-value types anyway, whereas abbreviated
735	 * keys are typically only of value to pass-by-reference types.
736	 */
737	if (!base->sortKeys->abbrev_converter)
738		base->onlyKey = base->sortKeys;
739
740	MemoryContextSwitchTo(oldcontext);
741
742	return state;
743}

935 *
936 * If the Datum is pass-by-ref type, the value will be copied.
937 */
938void
939tuplesort_putdatum(Tuplesortstate *state, Datum val, bool isNull)
940{
941	TuplesortPublic *base = TuplesortstateGetPublic(state);
942	MemoryContext oldcontext = MemoryContextSwitchTo(base->tuplecontext);
943	TuplesortDatumArg *arg = (TuplesortDatumArg *) base->arg;
944	SortTuple	stup;
945
946	/*
947	 * Pass-by-value types or null values are just stored directly in
948	 * stup.datum1 (and stup.tuple is not used and set to NULL).
949	 *
950	 * Non-null pass-by-reference values need to be copied into memory we
951	 * control, and possibly abbreviated. The copied value is pointed to by
952	 * stup.tuple and is treated as the canonical copy (e.g. to return via
953	 * tuplesort_getdatum or when writing to tape); stup.datum1 gets the

962		 * and to support cheap inequality tests for NULL abbreviated keys).
963		 */

965		stup.isnull1 = isNull;
966		stup.tuple = NULL;		/* no separate storage */
967	}
968	else
969	{
970		stup.isnull1 = false;
971		stup.datum1 = datumCopy(val, false, arg->datumTypeLen);
972		stup.tuple = DatumGetPointer(stup.datum1);
973	}
974
975	tuplesort_puttuple_common(state, &stup,
976							  base->tuples &&
977							  base->sortKeys->abbrev_converter && !isNull, 0);
978

980}

982/*
983 * Fetch the next tuple in either forward or back direction.
984 * If successful, put tuple in slot and return true; else, clear the slot
985 * and return false.
986 *
987 * Caller may optionally be passed back abbreviated value (on true return
988 * value) when abbreviation was used, which can be used to cheaply avoid
989 * equality checks that might otherwise be required.  Caller can safely make a
990 * determination of "non-equal tuple" based on simple binary inequality.  A
991 * NULL value in leading attribute will set abbreviated value to zeroed
992 * representation, which caller may rely on in abbreviated inequality check.
993 *
994 * If copy is true, the slot receives a tuple that's been copied into the
995 * caller's memory context, so that it will stay valid regardless of future
996 * manipulations of the tuplesort's state (up to and including deleting the
997 * tuplesort).  If copy is false, the slot will just receive a pointer to a
998 * tuple held within the tuplesort, which is more efficient, but only safe for
999 * callers that are prepared to have any subsequent manipulation of the
1000 * tuplesort's state invalidate slot contents.
1001 */
1002bool
1003tuplesort_gettupleslot(Tuplesortstate *state, bool forward, bool copy,
1004					   TupleTableSlot *slot, Datum *abbrev)
1005{
1006	TuplesortPublic *base = TuplesortstateGetPublic(state);
1007	MemoryContext oldcontext = MemoryContextSwitchTo(base->sortcontext);
1008	SortTuple	stup;
1009
1010	if (!tuplesort_gettuple_common(state, forward, &stup))
1011		stup.tuple = NULL;
1012
1013	MemoryContextSwitchTo(oldcontext);
1014
1015	if (stup.tuple)
1016	{
1017		/* Record abbreviated key for caller */
1018		if (base->sortKeys->abbrev_converter && abbrev)
1019			*abbrev = stup.datum1;
1020
1021		if (copy)
1022			stup.tuple = heap_copy_minimal_tuple((MinimalTuple) stup.tuple, 0);
1023
1024		ExecStoreMinimalTuple((MinimalTuple) stup.tuple, slot, copy);
1025		return true;
1026	}
1027	else
1028	{
1029		ExecClearTuple(slot);
1030		return false;
1031	}
1032}

1239static int
1240comparetup_heap_tiebreak(const SortTuple *a, const SortTuple *b, Tuplesortstate *state)
1241{
1242	TuplesortPublic *base = TuplesortstateGetPublic(state);
1243	SortSupport sortKey = base->sortKeys;
1244	HeapTupleData ltup;
1245	HeapTupleData rtup;
1246	TupleDesc	tupDesc;
1247	int			nkey;
1248	int32		compare;
1249	AttrNumber	attno;
1250	Datum		datum1,
1251				datum2;
1252	bool		isnull1,
1253				isnull2;
1254
1255	ltup.t_len = ((MinimalTuple) a->tuple)->t_len + MINIMAL_TUPLE_OFFSET;
1256	ltup.t_data = (HeapTupleHeader) ((char *) a->tuple - MINIMAL_TUPLE_OFFSET);
1257	rtup.t_len = ((MinimalTuple) b->tuple)->t_len + MINIMAL_TUPLE_OFFSET;
1258	rtup.t_data = (HeapTupleHeader) ((char *) b->tuple - MINIMAL_TUPLE_OFFSET);
1259	tupDesc = (TupleDesc) base->arg;
1260
1261	if (sortKey->abbrev_converter)
1262	{
1263		attno = sortKey->ssup_attno;
1264
1265		datum1 = heap_getattr(&ltup, attno, tupDesc, &isnull1);
1266		datum2 = heap_getattr(&rtup, attno, tupDesc, &isnull2);
1267

1273	}

1280		datum1 = heap_getattr(&ltup, attno, tupDesc, &isnull1);
1281		datum2 = heap_getattr(&rtup, attno, tupDesc, &isnull2);
1282
1283		compare = ApplySortComparator(datum1, isnull1,
1284									  datum2, isnull2,
1285									  sortKey);
1286		if (compare != 0)
1287			return compare;
1288	}

1383static int
1384comparetup_cluster_tiebreak(const SortTuple *a, const SortTuple *b,
1385							Tuplesortstate *state)
1386{
1387	TuplesortPublic *base = TuplesortstateGetPublic(state);
1388	TuplesortClusterArg *arg = (TuplesortClusterArg *) base->arg;
1389	SortSupport sortKey = base->sortKeys;
1390	HeapTuple	ltup;
1391	HeapTuple	rtup;
1392	TupleDesc	tupDesc;
1393	int			nkey;
1394	int32		compare = 0;
1395	Datum		datum1,
1396				datum2;
1397	bool		isnull1,
1398				isnull2;
1399
1400	ltup = (HeapTuple) a->tuple;
1401	rtup = (HeapTuple) b->tuple;
1402	tupDesc = arg->tupDesc;
1403
1404	/* Compare the leading sort key, if it's simple */
1405	if (base->haveDatum1)
1406	{
1407		if (sortKey->abbrev_converter)
1408		{
1409			AttrNumber	leading = arg->indexInfo->ii_IndexAttrNumbers[0];
1410
1411			datum1 = heap_getattr(ltup, leading, tupDesc, &isnull1);
1412			datum2 = heap_getattr(rtup, leading, tupDesc, &isnull2);
1413
1414			compare = ApplySortAbbrevFullComparator(datum1, isnull1,
1415													datum2, isnull2,
1416													sortKey);
1417		}
1418		if (compare != 0 || base->nKeys == 1)
1419			return compare;
1420		/* Compare additional columns the hard way */
1421		sortKey++;
1422		nkey = 1;
1423	}
1424	else
1425	{
1426		/* Must compare all keys the hard way */
1427		nkey = 0;
1428	}
1429
1430	if (arg->indexInfo->ii_Expressions == NULL)
1431	{
1432		/* If not expression index, just compare the proper heap attrs */
1433
1434		for (; nkey < base->nKeys; nkey++, sortKey++)
1435		{
1436			AttrNumber	attno = arg->indexInfo->ii_IndexAttrNumbers[nkey];
1437
1438			datum1 = heap_getattr(ltup, attno, tupDesc, &isnull1);
1439			datum2 = heap_getattr(rtup, attno, tupDesc, &isnull2);
1440
1441			compare = ApplySortComparator(datum1, isnull1,
1442										  datum2, isnull2,
1443										  sortKey);
1444			if (compare != 0)
1445				return compare;
1446		}
1447	}
1448	else
1449	{
1450		/*
1451		 * In the expression index case, compute the whole index tuple and
1452		 * then compare values.  It would perhaps be faster to compute only as
1453		 * many columns as we need to compare, but that would require
1454		 * duplicating all the logic in FormIndexDatum.
1455		 */
1456		Datum		l_index_values[INDEX_MAX_KEYS];
1457		bool		l_index_isnull[INDEX_MAX_KEYS];
1458		Datum		r_index_values[INDEX_MAX_KEYS];
1459		bool		r_index_isnull[INDEX_MAX_KEYS];
1460		TupleTableSlot *ecxt_scantuple;
1461
1462		/* Reset context each time to prevent memory leakage */
1463		ResetPerTupleExprContext(arg->estate);
1464
1465		ecxt_scantuple = GetPerTupleExprContext(arg->estate)->ecxt_scantuple;
1466
1467		ExecStoreHeapTuple(ltup, ecxt_scantuple, false);
1468		FormIndexDatum(arg->indexInfo, ecxt_scantuple, arg->estate,
1469					   l_index_values, l_index_isnull);
1470
1471		ExecStoreHeapTuple(rtup, ecxt_scantuple, false);
1472		FormIndexDatum(arg->indexInfo, ecxt_scantuple, arg->estate,
1473					   r_index_values, r_index_isnull);
1474
1475		for (; nkey < base->nKeys; nkey++, sortKey++)
1476		{
1477			compare = ApplySortComparator(l_index_values[nkey],
1478										  l_index_isnull[nkey],
1479										  r_index_values[nkey],
1480										  r_index_isnull[nkey],
1481										  sortKey);
1482			if (compare != 0)
1483				return compare;
1484		}
1485	}
1486
1487	return 0;
1488}

1601static int
1602comparetup_index_btree_tiebreak(const SortTuple *a, const SortTuple *b,
1603								Tuplesortstate *state)
1604{
1605	TuplesortPublic *base = TuplesortstateGetPublic(state);
1606	TuplesortIndexBTreeArg *arg = (TuplesortIndexBTreeArg *) base->arg;
1607	SortSupport sortKey = base->sortKeys;
1608	IndexTuple	tuple1;
1609	IndexTuple	tuple2;
1610	int			keysz;
1611	TupleDesc	tupDes;
1612	bool		equal_hasnull = false;
1613	int			nkey;
1614	int32		compare;
1615	Datum		datum1,
1616				datum2;
1617	bool		isnull1,
1618				isnull2;
1619
1620	tuple1 = (IndexTuple) a->tuple;
1621	tuple2 = (IndexTuple) b->tuple;
1622	keysz = base->nKeys;
1623	tupDes = RelationGetDescr(arg->index.indexRel);
1624
1625	if (sortKey->abbrev_converter)
1626	{
1627		datum1 = index_getattr(tuple1, 1, tupDes, &isnull1);
1628		datum2 = index_getattr(tuple2, 1, tupDes, &isnull2);
1629
1630		compare = ApplySortAbbrevFullComparator(datum1, isnull1,
1631												datum2, isnull2,
1632												sortKey);
1633		if (compare != 0)
1634			return compare;
1635	}
1636
1637	/* they are equal, so we only need to examine one null flag */
1638	if (a->isnull1)
1639		equal_hasnull = true;
1640
1641	sortKey++;
1642	for (nkey = 2; nkey <= keysz; nkey++, sortKey++)
1643	{
1644		datum1 = index_getattr(tuple1, nkey, tupDes, &isnull1);
1645		datum2 = index_getattr(tuple2, nkey, tupDes, &isnull2);
1646
1647		compare = ApplySortComparator(datum1, isnull1,
1648									  datum2, isnull2,
1649									  sortKey);
1650		if (compare != 0)
1651			return compare;		/* done when we find unequal attributes */
1652
1653		/* they are equal, so we only need to examine one null flag */
1654		if (isnull1)
1655			equal_hasnull = true;
1656	}
1657
1658	/*
1659	 * If btree has asked us to enforce uniqueness, complain if two equal
1660	 * tuples are detected (unless there was at least one NULL field and NULLS
1661	 * NOT DISTINCT was not set).
1662	 *
1663	 * It is sufficient to make the test here, because if two tuples are equal
1664	 * they *must* get compared at some stage of the sort --- otherwise the
1665	 * sort algorithm wouldn't have checked whether one must appear before the
1666	 * other.
1667	 */
1668	if (arg->enforceUnique && !(!arg->uniqueNullsNotDistinct && equal_hasnull))
1669	{
1670		Datum		values[INDEX_MAX_KEYS];
1671		bool		isnull[INDEX_MAX_KEYS];
1672		char	   *key_desc;
1673
1674		/*
1675		 * Some rather brain-dead implementations of qsort (such as the one in
1676		 * QNX 4) will sometimes call the comparison routine to compare a
1677		 * value to itself, but we always use our own implementation, which
1678		 * does not.
1679		 */
1680		Assert(tuple1 != tuple2);
1681
1682		index_deform_tuple(tuple1, tupDes, values, isnull);
1683
1684		key_desc = BuildIndexValueDescription(arg->index.indexRel, values, isnull);
1685
1686		ereport(ERROR,
1687				(errcode(ERRCODE_UNIQUE_VIOLATION),
1688				 errmsg("could not create unique index \"%s\"",
1689						RelationGetRelationName(arg->index.indexRel)),
1690				 key_desc ? errdetail("Key %s is duplicated.", key_desc) :
1691				 errdetail("Duplicate keys exist."),
1692				 errtableconstraint(arg->index.heapRel,
1693									RelationGetRelationName(arg->index.indexRel))));
1694	}
1695
1696	/*
1697	 * If key values are equal, we sort on ItemPointer.  This is required for
1698	 * btree indexes, since heap TID is treated as an implicit last key
1699	 * attribute in order to ensure that all keys in the index are physically
1700	 * unique.
1701	 */
1702	{
1703		BlockNumber blk1 = ItemPointerGetBlockNumber(&tuple1->t_tid);
1704		BlockNumber blk2 = ItemPointerGetBlockNumber(&tuple2->t_tid);
1705
1706		if (blk1 != blk2)
1707			return (blk1 < blk2) ? -1 : 1;
1708	}
1709	{
1710		OffsetNumber pos1 = ItemPointerGetOffsetNumber(&tuple1->t_tid);
1711		OffsetNumber pos2 = ItemPointerGetOffsetNumber(&tuple2->t_tid);
1712
1713		if (pos1 != pos2)
1714			return (pos1 < pos2) ? -1 : 1;
1715	}
1716
1717	/* ItemPointer values should never be equal */
1718	Assert(false);
1719
1720	return 0;
1721}

1723static int
1724comparetup_index_hash(const SortTuple *a, const SortTuple *b,
1725					  Tuplesortstate *state)
1726{
1727	Bucket		bucket1;
1728	Bucket		bucket2;
1729	uint32		hash1;
1730	uint32		hash2;
1731	IndexTuple	tuple1;
1732	IndexTuple	tuple2;
1733	TuplesortPublic *base = TuplesortstateGetPublic(state);
1734	TuplesortIndexHashArg *arg = (TuplesortIndexHashArg *) base->arg;
1735
1736	/*
1737	 * Fetch hash keys and mask off bits we don't want to sort by, so that the
1738	 * initial sort is just on the bucket number.  We know that the first
1739	 * column of the index tuple is the hash key.
1740	 */
1741	Assert(!a->isnull1);
1742	bucket1 = _hash_hashkey2bucket(DatumGetUInt32(a->datum1),
1743								   arg->max_buckets, arg->high_mask,
1744								   arg->low_mask);
1745	Assert(!b->isnull1);
1746	bucket2 = _hash_hashkey2bucket(DatumGetUInt32(b->datum1),
1747								   arg->max_buckets, arg->high_mask,
1748								   arg->low_mask);
1749	if (bucket1 > bucket2)
1750		return 1;
1751	else if (bucket1 < bucket2)
1752		return -1;
1753
1754	/*
1755	 * If bucket values are equal, sort by hash values.  This allows us to
1756	 * insert directly onto bucket/overflow pages, where the index tuples are
1757	 * stored in hash order to allow fast binary search within each page.
1758	 */
1759	hash1 = DatumGetUInt32(a->datum1);
1760	hash2 = DatumGetUInt32(b->datum1);
1761	if (hash1 > hash2)
1762		return 1;
1763	else if (hash1 < hash2)
1764		return -1;
1765
1766	/*
1767	 * If hash values are equal, we sort on ItemPointer.  This does not affect
1768	 * validity of the finished index, but it may be useful to have index
1769	 * scans in physical order.
1770	 */
1771	tuple1 = (IndexTuple) a->tuple;
1772	tuple2 = (IndexTuple) b->tuple;
1773
1774	{
1775		BlockNumber blk1 = ItemPointerGetBlockNumber(&tuple1->t_tid);
1776		BlockNumber blk2 = ItemPointerGetBlockNumber(&tuple2->t_tid);
1777
1778		if (blk1 != blk2)
1779			return (blk1 < blk2) ? -1 : 1;
1780	}
1781	{
1782		OffsetNumber pos1 = ItemPointerGetOffsetNumber(&tuple1->t_tid);
1783		OffsetNumber pos2 = ItemPointerGetOffsetNumber(&tuple2->t_tid);
1784
1785		if (pos1 != pos2)
1786			return (pos1 < pos2) ? -1 : 1;
1787	}
1788
1789	/* ItemPointer values should never be equal */
1790	Assert(false);
1791
1792	return 0;
1793}
