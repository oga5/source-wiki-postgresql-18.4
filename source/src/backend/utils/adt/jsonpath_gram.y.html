158	| GREATEREQUAL_P				{ $$ = jpiGreaterOrEqual; }

163	| EXISTS_P '(' expr ')'			{ $$ = makeItemUnary(jpiExists, $3); }
164	;

168	| expr comp_op expr				{ $$ = makeItemBinary($2, $1, $3); }
169	| predicate AND_P predicate		{ $$ = makeItemBinary(jpiAnd, $1, $3); }
170	| predicate OR_P predicate		{ $$ = makeItemBinary(jpiOr, $1, $3); }
171	| NOT_P delimited_predicate		{ $$ = makeItemUnary(jpiNot, $2); }
172	| '(' predicate ')' IS_P UNKNOWN_P
173									{ $$ = makeItemUnary(jpiIsUnknown, $2); }
174	| expr STARTS_P WITH_P starts_with_initial
175									{ $$ = makeItemBinary(jpiStartsWith, $1, $4); }
176	| expr LIKE_REGEX_P STRING_P

189	}

384static JsonPathParseItem *
385makeItemType(JsonPathItemType type)
386{
387	JsonPathParseItem *v = palloc(sizeof(*v));
388
389	CHECK_FOR_INTERRUPTS();
390
391	v->type = type;
392	v->next = NULL;
393
394	return v;
395}
396
397static JsonPathParseItem *
398makeItemString(JsonPathString *s)
399{
400	JsonPathParseItem *v;
401
402	if (s == NULL)
403	{
404		v = makeItemType(jpiNull);
405	}
406	else
407	{
408		v = makeItemType(jpiString);
409		v->value.string.val = s->val;
410		v->value.string.len = s->len;
411	}
412
413	return v;
414}

416static JsonPathParseItem *
417makeItemVariable(JsonPathString *s)
418{
419	JsonPathParseItem *v;
420
421	v = makeItemType(jpiVariable);
422	v->value.string.val = s->val;
423	v->value.string.len = s->len;
424
425	return v;
426}

464static JsonPathParseItem *
465makeItemBinary(JsonPathItemType type, JsonPathParseItem *la, JsonPathParseItem *ra)
466{
467	JsonPathParseItem *v = makeItemType(type);
468
469	v->value.args.left = la;
470	v->value.args.right = ra;

472	return v;

474
475static JsonPathParseItem *
476makeItemUnary(JsonPathItemType type, JsonPathParseItem *a)
477{
478	JsonPathParseItem *v;
479
480	if (type == jpiPlus && a->type == jpiNumeric && !a->next)
481		return a;
482
483	if (type == jpiMinus && a->type == jpiNumeric && !a->next)
484	{
485		v = makeItemType(jpiNumeric);
486		v->value.numeric =
487			DatumGetNumeric(DirectFunctionCall1(numeric_uminus,
488												NumericGetDatum(a->value.numeric)));
489		return v;
490	}
491
492	v = makeItemType(type);
493
494	v->value.arg = a;
495
496	return v;
497}

499static JsonPathParseItem *
500makeItemList(List *list)
501{
502	JsonPathParseItem *head,
503			   *end;
504	ListCell   *cell;
505
506	head = end = (JsonPathParseItem *) linitial(list);
507
508	if (list_length(list) == 1)
509		return head;
510
511	/* append items to the end of already existing list */
512	while (end->next)
513		end = end->next;
514
515	for_each_from(cell, list, 1)
516	{
517		JsonPathParseItem *c = (JsonPathParseItem *) lfirst(cell);
518
519		end->next = c;
520		end = c;
521	}
522
523	return head;
524}
525
526static JsonPathParseItem *
527makeIndexArray(List *list)
528{
529	JsonPathParseItem *v = makeItemType(jpiIndexArray);
530	ListCell   *cell;
531	int			i = 0;
532
533	Assert(list != NIL);
534	v->value.array.nelems = list_length(list);
535
536	v->value.array.elems = palloc(sizeof(v->value.array.elems[0]) *
537								  v->value.array.nelems);
538
539	foreach(cell, list)
540	{
541		JsonPathParseItem *jpi = lfirst(cell);
542
543		Assert(jpi->type == jpiSubscript);
544
545		v->value.array.elems[i].from = jpi->value.args.left;
546		v->value.array.elems[i++].to = jpi->value.args.right;
547	}
548
549	return v;
550}

552static JsonPathParseItem *
553makeAny(int first, int last)
554{
555	JsonPathParseItem *v = makeItemType(jpiAny);
556
557	v->value.anybounds.first = (first >= 0) ? first : PG_UINT32_MAX;
558	v->value.anybounds.last = (last >= 0) ? last : PG_UINT32_MAX;
559
560	return v;
561}

563static bool
564makeItemLikeRegex(JsonPathParseItem *expr, JsonPathString *pattern,
565				  JsonPathString *flags, JsonPathParseItem **result,
566				  struct Node *escontext)
567{
568	JsonPathParseItem *v = makeItemType(jpiLikeRegex);
569	int			i;
570	int			cflags;
571
572	v->value.like_regex.expr = expr;
573	v->value.like_regex.pattern = pattern->val;
574	v->value.like_regex.patternlen = pattern->len;
575
576	/* Parse the flags string, convert to bitmask.  Duplicate flags are OK. */
577	v->value.like_regex.flags = 0;
578	for (i = 0; flags && i < flags->len; i++)
579	{
580		switch (flags->val[i])
581		{
582			case 'i':
583				v->value.like_regex.flags |= JSP_REGEX_ICASE;
584				break;
585			case 's':
586				v->value.like_regex.flags |= JSP_REGEX_DOTALL;
587				break;
588			case 'm':
589				v->value.like_regex.flags |= JSP_REGEX_MLINE;
590				break;
591			case 'x':
592				v->value.like_regex.flags |= JSP_REGEX_WSPACE;
593				break;
594			case 'q':
595				v->value.like_regex.flags |= JSP_REGEX_QUOTE;
596				break;
597			default:
598				ereturn(escontext, false,
599						(errcode(ERRCODE_SYNTAX_ERROR),
600						 errmsg("invalid input syntax for type %s", "jsonpath"),
601						 errdetail("Unrecognized flag character \"%.*s\" in LIKE_REGEX predicate.",
602								   pg_mblen_range(flags->val + i, flags->val + flags->len),
603								   flags->val + i)));
604				break;
605		}
606	}
607
608	/* Convert flags to what pg_regcomp needs */
609	if (!jspConvertRegexFlags(v->value.like_regex.flags, &cflags, escontext))
610		return false;
611
612	/* check regex validity */
613	{
614		regex_t		re_tmp;
615		pg_wchar   *wpattern;
616		int			wpattern_len;
617		int			re_result;
618
619		wpattern = (pg_wchar *) palloc((pattern->len + 1) * sizeof(pg_wchar));
620		wpattern_len = pg_mb2wchar_with_len(pattern->val,
621											wpattern,
622											pattern->len);
623
624		if ((re_result = pg_regcomp(&re_tmp, wpattern, wpattern_len, cflags,
625									DEFAULT_COLLATION_OID)) != REG_OKAY)
626		{
627			char		errMsg[100];
628
629			pg_regerror(re_result, &re_tmp, errMsg, sizeof(errMsg));
630			ereturn(escontext, false,
631					(errcode(ERRCODE_INVALID_REGULAR_EXPRESSION),
632					 errmsg("invalid regular expression: %s", errMsg)));
633		}
634
635		pg_regfree(&re_tmp);
636	}
637
638	*result = v;
639
640	return true;
641}

643/*
644 * Convert from XQuery regex flags to those recognized by our regex library.
645 */
646bool
647jspConvertRegexFlags(uint32 xflags, int *result, struct Node *escontext)
648{
649	/* By default, XQuery is very nearly the same as Spencer's AREs */
650	int			cflags = REG_ADVANCED;
651
652	/* Ignore-case means the same thing, too, modulo locale issues */
653	if (xflags & JSP_REGEX_ICASE)
654		cflags |= REG_ICASE;
655
656	/* Per XQuery spec, if 'q' is specified then 'm', 's', 'x' are ignored */
657	if (xflags & JSP_REGEX_QUOTE)
658	{
659		cflags &= ~REG_ADVANCED;
660		cflags |= REG_QUOTE;
661	}
662	else
663	{
664		/* Note that dotall mode is the default in POSIX */
665		if (!(xflags & JSP_REGEX_DOTALL))
666			cflags |= REG_NLSTOP;
667		if (xflags & JSP_REGEX_MLINE)
668			cflags |= REG_NLANCH;
669
670		/*
671		 * XQuery's 'x' mode is related to Spencer's expanded mode, but it's
672		 * not really enough alike to justify treating JSP_REGEX_WSPACE as
673		 * REG_EXPANDED.  For now we treat 'x' as unimplemented; perhaps in
674		 * future we'll modify the regex library to have an option for
675		 * XQuery-style ignore-whitespace mode.
676		 */
677		if (xflags & JSP_REGEX_WSPACE)
678			ereturn(escontext, false,
679					(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
680					 errmsg("XQuery \"x\" flag (expanded regular expressions) is not implemented")));
681	}
682
683	*result = cflags;
684
685	return true;
