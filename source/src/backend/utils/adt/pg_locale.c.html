12/*----------
13 * Here is how the locale stuff is handled: LC_COLLATE and LC_CTYPE
14 * are fixed at CREATE DATABASE time, stored in pg_database, and cannot
15 * be changed. Thus, the effects of strcoll(), strxfrm(), isupper(),
16 * toupper(), etc. are always in the same fixed locale.
17 *
18 * LC_MESSAGES is settable at run time and will take effect
19 * immediately.
20 *
21 * The other categories, LC_MONETARY, LC_NUMERIC, and LC_TIME are
22 * permanently set to "C", and then we use temporary locale_t
23 * objects when we need to look up locale data based on the GUCs
24 * of the same name.  Information is cached when the GUCs change.
25 * The cached information is only used by the formatting functions
26 * (to_char, etc.) and the money type.  For the user, this should all be
27 * transparent.
28 *----------
29 */

183/*
184 * pg_perm_setlocale
185 *
186 * This wraps the libc function setlocale(), with two additions.  First, when
187 * changing LC_CTYPE, update gettext's encoding for the current message
188 * domain.  GNU gettext automatically tracks LC_CTYPE on most platforms, but
189 * not on Windows.  Second, if the operation is successful, the corresponding
190 * LC_XXX environment variable is set to match.  By setting the environment
191 * variable, we ensure that any subsequent use of setlocale(..., "") will
192 * preserve the settings made through this routine.  Of course, LC_ALL must
193 * also be unset to fully ensure that, but that has to be done elsewhere after
194 * all the individual LC_XXX variables have been set correctly.  (Thank you
195 * Perl for making this kluge necessary.)
196 */
197char *
198pg_perm_setlocale(int category, const char *locale)
199{
200	char	   *result;
201	const char *envvar;
202
203#ifndef WIN32
204	result = setlocale(category, locale);
205#else
206
207	/*
208	 * On Windows, setlocale(LC_MESSAGES) does not work, so just assume that
209	 * the given value is good and set it in the environment variables. We
210	 * must ignore attempts to set to "", which means "keep using the old
211	 * environment value".
212	 */
213#ifdef LC_MESSAGES
214	if (category == LC_MESSAGES)
215	{
216		result = (char *) locale;
217		if (locale == NULL || locale[0] == '\0')
218			return result;
219	}
220	else
221#endif
222		result = setlocale(category, locale);
223#endif							/* WIN32 */
224
225	if (result == NULL)
226		return result;			/* fall out immediately on failure */
227
228	/*
229	 * Use the right encoding in translated messages.  Under ENABLE_NLS, let
230	 * pg_bind_textdomain_codeset() figure it out.  Under !ENABLE_NLS, message
231	 * format strings are ASCII, but database-encoding strings may enter the
232	 * message via %s.  This makes the overall message encoding equal to the
233	 * database encoding.
234	 */
235	if (category == LC_CTYPE)
236	{
237		static char save_lc_ctype[LOCALE_NAME_BUFLEN];
238
239		/* copy setlocale() return value before callee invokes it again */
240		strlcpy(save_lc_ctype, result, sizeof(save_lc_ctype));
241		result = save_lc_ctype;
242
243#ifdef ENABLE_NLS
244		SetMessageEncoding(pg_bind_textdomain_codeset(textdomain(NULL)));
245#else
246		SetMessageEncoding(GetDatabaseEncoding());
247#endif
248	}
249
250	switch (category)
251	{
252		case LC_COLLATE:
253			envvar = "LC_COLLATE";
254			break;
255		case LC_CTYPE:
256			envvar = "LC_CTYPE";
257			break;
258#ifdef LC_MESSAGES
259		case LC_MESSAGES:
260			envvar = "LC_MESSAGES";
261#ifdef WIN32
262			result = IsoLocaleName(locale);
263			if (result == NULL)
264				result = (char *) locale;
265			elog(DEBUG3, "IsoLocaleName() executed; locale: \"%s\"", result);
266#endif							/* WIN32 */
267			break;
268#endif							/* LC_MESSAGES */
269		case LC_MONETARY:
270			envvar = "LC_MONETARY";
271			break;
272		case LC_NUMERIC:
273			envvar = "LC_NUMERIC";
274			break;
275		case LC_TIME:
276			envvar = "LC_TIME";
277			break;
278		default:
279			elog(FATAL, "unrecognized LC category: %d", category);
280			return NULL;		/* keep compiler quiet */
281	}
282
283	if (setenv(envvar, result, 1) != 0)
284		return NULL;
285
286	return result;
287}

290/*
291 * Is the locale name valid for the locale category?
292 *
293 * If successful, and canonname isn't NULL, a palloc'd copy of the locale's
294 * canonical name is stored there.  This is especially useful for figuring out
295 * what locale name "" means (ie, the server environment value).  (Actually,
296 * it seems that on most implementations that's the only thing it's good for;
297 * we could wish that setlocale gave back a canonically spelled version of
298 * the locale name, but typically it doesn't.)
299 */
300bool
301check_locale(int category, const char *locale, char **canonname)
302{
303	char	   *save;
304	char	   *res;
305
306	/* Don't let Windows' non-ASCII locale names in. */
307	if (!pg_is_ascii(locale))
308	{
309		ereport(WARNING,
310				(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
311				 errmsg("locale name \"%s\" contains non-ASCII characters",
312						locale)));
313		return false;
314	}
315
316	if (canonname)
317		*canonname = NULL;		/* in case of failure */
318
319	save = setlocale(category, NULL);
320	if (!save)
321		return false;			/* won't happen, we hope */
322
323	/* save may be pointing at a modifiable scratch variable, see above. */
324	save = pstrdup(save);
325
326	/* set the locale with setlocale, to see if it accepts it. */
327	res = setlocale(category, locale);
328
329	/* save canonical name if requested. */
330	if (res && canonname)
331		*canonname = pstrdup(res);
332
333	/* restore old value. */
334	if (!setlocale(category, save))
335		elog(WARNING, "failed to restore old locale \"%s\"", save);
336	pfree(save);
337
338	/* Don't let Windows' non-ASCII locale names out. */
339	if (canonname && *canonname && !pg_is_ascii(*canonname))
340	{
341		ereport(WARNING,
342				(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
343				 errmsg("locale name \"%s\" contains non-ASCII characters",
344						*canonname)));
345		pfree(*canonname);
346		*canonname = NULL;
347		return false;
348	}
349
350	return (res != NULL);
351}

401/*
402 * We allow LC_MESSAGES to actually be set globally.
403 *
404 * Note: we normally disallow value = "" because it wouldn't have consistent
405 * semantics (it'd effectively just use the previous value).  However, this
406 * is the value passed for PGC_S_DEFAULT, so don't complain in that case,
407 * not even if the attempted setting fails due to invalid environment value.
408 * The idea there is just to accept the environment setting *if possible*
409 * during startup, until we can read the proper value from postgresql.conf.
410 */
411bool
412check_locale_messages(char **newval, void **extra, GucSource source)
413{
414	if (**newval == '\0')
415	{
416		if (source == PGC_S_DEFAULT)
417			return true;
418		else
419			return false;
420	}
421
422	/*
423	 * LC_MESSAGES category does not exist everywhere, but accept it anyway
424	 *
425	 * On Windows, we can't even check the value, so accept blindly
426	 */
427#if defined(LC_MESSAGES) && !defined(WIN32)
428	return check_locale(LC_MESSAGES, *newval, NULL);
429#else
430	return true;
431#endif
432}

434void
435assign_locale_messages(const char *newval, void *extra)
436{
437	/*
438	 * LC_MESSAGES category does not exist everywhere, but accept it anyway.
439	 * We ignore failure, as per comment above.
440	 */
441#ifdef LC_MESSAGES
442	(void) pg_perm_setlocale(LC_MESSAGES, newval);
443#endif
444}

448 * Frees the malloced content of a struct lconv.  (But not the struct
449 * itself.)  It's important that this not throw elog(ERROR).
450 */
451static void
452free_struct_lconv(struct lconv *s)
453{
454	free(s->decimal_point);
455	free(s->thousands_sep);
456	free(s->grouping);
457	free(s->int_curr_symbol);
458	free(s->currency_symbol);
459	free(s->mon_decimal_point);

461	free(s->mon_grouping);
462	free(s->positive_sign);
463	free(s->negative_sign);

466/*
467 * Check that all fields of a struct lconv (or at least, the ones we care
468 * about) are non-NULL.  The field list must match free_struct_lconv().
469 */
470static bool
471struct_lconv_is_valid(struct lconv *s)
472{
473	if (s->decimal_point == NULL)
474		return false;
475	if (s->thousands_sep == NULL)
476		return false;
477	if (s->grouping == NULL)
478		return false;
479	if (s->int_curr_symbol == NULL)
480		return false;
481	if (s->currency_symbol == NULL)
482		return false;
483	if (s->mon_decimal_point == NULL)
484		return false;
485	if (s->mon_thousands_sep == NULL)
486		return false;
487	if (s->mon_grouping == NULL)
488		return false;
489	if (s->positive_sign == NULL)
490		return false;
491	if (s->negative_sign == NULL)
492		return false;
493	return true;
494}

497/*
498 * Convert the strdup'd string at *str from the specified encoding to the
499 * database encoding.
500 */
501static void
502db_encoding_convert(int encoding, char **str)
503{
504	char	   *pstr;
505	char	   *mstr;
506
507	/* convert the string to the database encoding */
508	pstr = pg_any_to_server(*str, strlen(*str), encoding);
509	if (pstr == *str)
510		return;					/* no conversion happened */
511
512	/* need it malloc'd not palloc'd */
513	mstr = strdup(pstr);
514	if (mstr == NULL)
515		ereport(ERROR,
516				(errcode(ERRCODE_OUT_OF_MEMORY),
517				 errmsg("out of memory")));
518
519	/* replace old string */
520	free(*str);
521	*str = mstr;
522
523	pfree(pstr);
524}

527/*
528 * Return the POSIX lconv struct (contains number/money formatting
529 * information) with locale information for all categories.
530 */
531struct lconv *
532PGLC_localeconv(void)
533{
534	static struct lconv CurrentLocaleConv;
535	static bool CurrentLocaleConvAllocated = false;
536	struct lconv *extlconv;
537	struct lconv tmp;
538	struct lconv worklconv = {0};
539
540	/* Did we do it already? */
541	if (CurrentLocaleConvValid)
542		return &CurrentLocaleConv;
543
544	/* Free any already-allocated storage */
545	if (CurrentLocaleConvAllocated)
546	{
547		free_struct_lconv(&CurrentLocaleConv);
548		CurrentLocaleConvAllocated = false;
549	}
550
551	/*
552	 * Use thread-safe method of obtaining a copy of lconv from the operating
553	 * system.
554	 */
555	if (pg_localeconv_r(locale_monetary,
556						locale_numeric,
557						&tmp) != 0)
558		elog(ERROR,
559			 "could not get lconv for LC_MONETARY = \"%s\", LC_NUMERIC = \"%s\": %m",
560			 locale_monetary, locale_numeric);
561
562	/* Must copy data now so we can re-encode it. */
563	extlconv = &tmp;
564	worklconv.decimal_point = strdup(extlconv->decimal_point);
565	worklconv.thousands_sep = strdup(extlconv->thousands_sep);
566	worklconv.grouping = strdup(extlconv->grouping);
567	worklconv.int_curr_symbol = strdup(extlconv->int_curr_symbol);
568	worklconv.currency_symbol = strdup(extlconv->currency_symbol);
569	worklconv.mon_decimal_point = strdup(extlconv->mon_decimal_point);
570	worklconv.mon_thousands_sep = strdup(extlconv->mon_thousands_sep);
571	worklconv.mon_grouping = strdup(extlconv->mon_grouping);
572	worklconv.positive_sign = strdup(extlconv->positive_sign);
573	worklconv.negative_sign = strdup(extlconv->negative_sign);
574	/* Copy scalar fields as well */
575	worklconv.int_frac_digits = extlconv->int_frac_digits;
576	worklconv.frac_digits = extlconv->frac_digits;
577	worklconv.p_cs_precedes = extlconv->p_cs_precedes;
578	worklconv.p_sep_by_space = extlconv->p_sep_by_space;
579	worklconv.n_cs_precedes = extlconv->n_cs_precedes;
580	worklconv.n_sep_by_space = extlconv->n_sep_by_space;
581	worklconv.p_sign_posn = extlconv->p_sign_posn;
582	worklconv.n_sign_posn = extlconv->n_sign_posn;
583
584	/* Free the contents of the object populated by pg_localeconv_r(). */
585	pg_localeconv_free(&tmp);
586
587	/* If any of the preceding strdup calls failed, complain now. */
588	if (!struct_lconv_is_valid(&worklconv))
589		ereport(ERROR,
590				(errcode(ERRCODE_OUT_OF_MEMORY),
591				 errmsg("out of memory")));
592
593	PG_TRY();
594	{
595		int			encoding;
596
597		/*
598		 * Now we must perform encoding conversion from whatever's associated
599		 * with the locales into the database encoding.  If we can't identify
600		 * the encoding implied by LC_NUMERIC or LC_MONETARY (ie we get -1),
601		 * use PG_SQL_ASCII, which will result in just validating that the
602		 * strings are OK in the database encoding.
603		 */
604		encoding = pg_get_encoding_from_locale(locale_numeric, true);
605		if (encoding < 0)
606			encoding = PG_SQL_ASCII;
607
608		db_encoding_convert(encoding, &worklconv.decimal_point);
609		db_encoding_convert(encoding, &worklconv.thousands_sep);
610		/* grouping is not text and does not require conversion */
611
612		encoding = pg_get_encoding_from_locale(locale_monetary, true);
613		if (encoding < 0)
614			encoding = PG_SQL_ASCII;
615
616		db_encoding_convert(encoding, &worklconv.int_curr_symbol);
617		db_encoding_convert(encoding, &worklconv.currency_symbol);
618		db_encoding_convert(encoding, &worklconv.mon_decimal_point);
619		db_encoding_convert(encoding, &worklconv.mon_thousands_sep);
620		/* mon_grouping is not text and does not require conversion */
621		db_encoding_convert(encoding, &worklconv.positive_sign);
622		db_encoding_convert(encoding, &worklconv.negative_sign);
623	}
624	PG_CATCH();
625	{
626		free_struct_lconv(&worklconv);
627		PG_RE_THROW();
628	}
629	PG_END_TRY();
630
631	/*
632	 * Everything is good, so save the results.
633	 */
634	CurrentLocaleConv = worklconv;
635	CurrentLocaleConvAllocated = true;
636	CurrentLocaleConvValid = true;
637	return &CurrentLocaleConv;
638}

640#ifdef WIN32
641/*
642 * On Windows, strftime() returns its output in encoding CP_ACP (the default
643 * operating system codepage for the computer), which is likely different
644 * from SERVER_ENCODING.  This is especially important in Japanese versions
645 * of Windows which will use SJIS encoding, which we don't support as a
646 * server encoding.
647 *
648 * So, instead of using strftime(), use wcsftime() to return the value in
649 * wide characters (internally UTF16) and then convert to UTF8, which we
650 * know how to handle directly.
651 *
652 * Note that this only affects the calls to strftime() in this file, which are
653 * used to get the locale-aware strings. Other parts of the backend use
654 * pg_strftime(), which isn't locale-aware and does not need to be replaced.
655 */
656static size_t
657strftime_l_win32(char *dst, size_t dstlen,
658				 const char *format, const struct tm *tm, locale_t locale)
659{
660	size_t		len;
661	wchar_t		wformat[8];		/* formats used below need 3 chars */
662	wchar_t		wbuf[MAX_L10N_DATA];
663
664	/*
665	 * Get a wchar_t version of the format string.  We only actually use
666	 * plain-ASCII formats in this file, so we can say that they're UTF8.
667	 */
668	len = MultiByteToWideChar(CP_UTF8, 0, format, -1,
669							  wformat, lengthof(wformat));
670	if (len == 0)
671		elog(ERROR, "could not convert format string from UTF-8: error code %lu",
672			 GetLastError());
673
674	len = _wcsftime_l(wbuf, MAX_L10N_DATA, wformat, tm, locale);
675	if (len == 0)
676	{
677		/*
678		 * wcsftime failed, possibly because the result would not fit in
679		 * MAX_L10N_DATA.  Return 0 with the contents of dst unspecified.
680		 */
681		return 0;
682	}
683
684	len = WideCharToMultiByte(CP_UTF8, 0, wbuf, len, dst, dstlen - 1,
685							  NULL, NULL);
686	if (len == 0)
687		elog(ERROR, "could not convert string to UTF-8: error code %lu",
688			 GetLastError());
689
690	dst[len] = '\0';
691
692	return len;
693}

699/*
700 * Subroutine for cache_locale_time().
701 * Convert the given string from encoding "encoding" to the database
702 * encoding, and store the result at *dst, replacing any previous value.
703 */
704static void
705cache_single_string(char **dst, const char *src, int encoding)
706{
707	char	   *ptr;
708	char	   *olddst;
709
710	/* Convert the string to the database encoding, or validate it's OK */
711	ptr = pg_any_to_server(src, strlen(src), encoding);
712
713	/* Store the string in long-lived storage, replacing any previous value */
714	olddst = *dst;
715	*dst = MemoryContextStrdup(TopMemoryContext, ptr);
716	if (olddst)
717		pfree(olddst);
718
719	/* Might as well clean up any palloc'd conversion result, too */
720	if (ptr != src)
721		pfree(ptr);
722}

724/*
725 * Update the lc_time localization cache variables if needed.
726 */
727void
728cache_locale_time(void)
729{
730	char		buf[(2 * 7 + 2 * 12) * MAX_L10N_DATA];
731	char	   *bufptr;
732	time_t		timenow;
733	struct tm  *timeinfo;
734	struct tm	timeinfobuf;
735	bool		strftimefail = false;
736	int			encoding;
737	int			i;
738	locale_t	locale;
739
740	/* did we do this already? */
741	if (CurrentLCTimeValid)
742		return;
743
744	elog(DEBUG3, "cache_locale_time() executed; locale: \"%s\"", locale_time);
745
746	errno = ENOENT;
747#ifdef WIN32
748	locale = _create_locale(LC_ALL, locale_time);
749	if (locale == (locale_t) 0)
750		_dosmaperr(GetLastError());
751#else
752	locale = newlocale(LC_ALL_MASK, locale_time, (locale_t) 0);
753#endif
754	if (!locale)
755		report_newlocale_failure(locale_time);
756
757	/* We use times close to current time as data for strftime(). */
758	timenow = time(NULL);
759	timeinfo = gmtime_r(&timenow, &timeinfobuf);
760
761	/* Store the strftime results in MAX_L10N_DATA-sized portions of buf[] */
762	bufptr = buf;
763
764	/*
765	 * MAX_L10N_DATA is sufficient buffer space for every known locale, and
766	 * POSIX defines no strftime() errors.  (Buffer space exhaustion is not an
767	 * error.)  An implementation might report errors (e.g. ENOMEM) by
768	 * returning 0 (or, less plausibly, a negative value) and setting errno.
769	 * Report errno just in case the implementation did that, but clear it in
770	 * advance of the calls so we don't emit a stale, unrelated errno.
771	 */
772	errno = 0;
773
774	/* localized days */
775	for (i = 0; i < 7; i++)
776	{
777		timeinfo->tm_wday = i;
778		if (strftime_l(bufptr, MAX_L10N_DATA, "%a", timeinfo, locale) <= 0)
779			strftimefail = true;
780		bufptr += MAX_L10N_DATA;
781		if (strftime_l(bufptr, MAX_L10N_DATA, "%A", timeinfo, locale) <= 0)
782			strftimefail = true;
783		bufptr += MAX_L10N_DATA;
784	}
785
786	/* localized months */
787	for (i = 0; i < 12; i++)
788	{
789		timeinfo->tm_mon = i;
790		timeinfo->tm_mday = 1;	/* make sure we don't have invalid date */
791		if (strftime_l(bufptr, MAX_L10N_DATA, "%b", timeinfo, locale) <= 0)
792			strftimefail = true;
793		bufptr += MAX_L10N_DATA;
794		if (strftime_l(bufptr, MAX_L10N_DATA, "%B", timeinfo, locale) <= 0)
795			strftimefail = true;
796		bufptr += MAX_L10N_DATA;
797	}
798
799#ifdef WIN32
800	_free_locale(locale);
801#else
802	freelocale(locale);
803#endif
804
805	/*
806	 * At this point we've done our best to clean up, and can throw errors, or
807	 * call functions that might throw errors, with a clean conscience.
808	 */
809	if (strftimefail)
810		elog(ERROR, "strftime_l() failed");
811
812#ifndef WIN32
813
814	/*
815	 * As in PGLC_localeconv(), we must convert strftime()'s output from the
816	 * encoding implied by LC_TIME to the database encoding.  If we can't
817	 * identify the LC_TIME encoding, just perform encoding validation.
818	 */
819	encoding = pg_get_encoding_from_locale(locale_time, true);
820	if (encoding < 0)
821		encoding = PG_SQL_ASCII;
822
823#else
824
825	/*
826	 * On Windows, strftime_win32() always returns UTF8 data, so convert from
827	 * that if necessary.
828	 */
829	encoding = PG_UTF8;
830
831#endif							/* WIN32 */
832
833	bufptr = buf;
834
835	/* localized days */
836	for (i = 0; i < 7; i++)
837	{
838		cache_single_string(&localized_abbrev_days[i], bufptr, encoding);
839		bufptr += MAX_L10N_DATA;
840		cache_single_string(&localized_full_days[i], bufptr, encoding);
841		bufptr += MAX_L10N_DATA;
842	}
843	localized_abbrev_days[7] = NULL;
844	localized_full_days[7] = NULL;
845
846	/* localized months */
847	for (i = 0; i < 12; i++)
848	{
849		cache_single_string(&localized_abbrev_months[i], bufptr, encoding);
850		bufptr += MAX_L10N_DATA;
851		cache_single_string(&localized_full_months[i], bufptr, encoding);
852		bufptr += MAX_L10N_DATA;
853	}
854	localized_abbrev_months[12] = NULL;
855	localized_full_months[12] = NULL;
856
857	CurrentLCTimeValid = true;
858}

862/*
863 * Convert a Windows setlocale() argument to a Unix-style one.
864 *
865 * Regardless of platform, we install message catalogs under a Unix-style
866 * LL[_CC][.ENCODING][@VARIANT] naming convention.  Only LC_MESSAGES settings
867 * following that style will elicit localized interface strings.
868 *
869 * Before Visual Studio 2012 (msvcr110.dll), Windows setlocale() accepted "C"
870 * (but not "c") and strings of the form <Language>[_<Country>][.<CodePage>],
871 * case-insensitive.  setlocale() returns the fully-qualified form; for
872 * example, setlocale("thaI") returns "Thai_Thailand.874".  Internally,
873 * setlocale() and _create_locale() select a "locale identifier"[1] and store
874 * it in an undocumented _locale_t field.  From that LCID, we can retrieve the
875 * ISO 639 language and the ISO 3166 country.  Character encoding does not
876 * matter, because the server and client encodings govern that.
877 *
878 * Windows Vista introduced the "locale name" concept[2], closely following
879 * RFC 4646.  Locale identifiers are now deprecated.  Starting with Visual
880 * Studio 2012, setlocale() accepts locale names in addition to the strings it
881 * accepted historically.  It does not standardize them; setlocale("Th-tH")
882 * returns "Th-tH".  setlocale(category, "") still returns a traditional
883 * string.  Furthermore, msvcr110.dll changed the undocumented _locale_t
884 * content to carry locale names instead of locale identifiers.
885 *
886 * Visual Studio 2015 should still be able to do the same as Visual Studio
887 * 2012, but the declaration of locale_name is missing in _locale_t, causing
888 * this code compilation to fail, hence this falls back instead on to
889 * enumerating all system locales by using EnumSystemLocalesEx to find the
890 * required locale name.  If the input argument is in Unix-style then we can
891 * get ISO Locale name directly by using GetLocaleInfoEx() with LCType as
892 * LOCALE_SNAME.
893 *
894 * This function returns a pointer to a static buffer bearing the converted
895 * name or NULL if conversion fails.
896 *
897 * [1] https://docs.microsoft.com/en-us/windows/win32/intl/locale-identifiers
898 * [2] https://docs.microsoft.com/en-us/windows/win32/intl/locale-names
899 */
900
901/*
902 * Callback function for EnumSystemLocalesEx() in get_iso_localename().
903 *
904 * This function enumerates all system locales, searching for one that matches
905 * an input with the format: <Language>[_<Country>], e.g.
906 * English[_United States]
907 *
908 * The input is a three wchar_t array as an LPARAM. The first element is the
909 * locale_name we want to match, the second element is an allocated buffer
910 * where the Unix-style locale is copied if a match is found, and the third
911 * element is the search status, 1 if a match was found, 0 otherwise.
912 */
913static BOOL CALLBACK
914search_locale_enum(LPWSTR pStr, DWORD dwFlags, LPARAM lparam)
915{
916	wchar_t		test_locale[LOCALE_NAME_MAX_LENGTH];
917	wchar_t   **argv;
918
919	(void) (dwFlags);
920
921	argv = (wchar_t **) lparam;
922	*argv[2] = (wchar_t) 0;
923
924	memset(test_locale, 0, sizeof(test_locale));
925
926	/* Get the name of the <Language> in English */
927	if (GetLocaleInfoEx(pStr, LOCALE_SENGLISHLANGUAGENAME,
928						test_locale, LOCALE_NAME_MAX_LENGTH))
929	{
930		/*
931		 * If the enumerated locale does not have a hyphen ("en") OR the
932		 * locale_name input does not have an underscore ("English"), we only
933		 * need to compare the <Language> tags.
934		 */
935		if (wcsrchr(pStr, '-') == NULL || wcsrchr(argv[0], '_') == NULL)
936		{
937			if (_wcsicmp(argv[0], test_locale) == 0)
938			{
939				wcscpy(argv[1], pStr);
940				*argv[2] = (wchar_t) 1;
941				return FALSE;
942			}
943		}
944
945		/*
946		 * We have to compare a full <Language>_<Country> tag, so we append
947		 * the underscore and name of the country/region in English, e.g.
948		 * "English_United States".
949		 */
950		else
951		{
952			size_t		len;
953
954			wcscat(test_locale, L"_");
955			len = wcslen(test_locale);
956			if (GetLocaleInfoEx(pStr, LOCALE_SENGLISHCOUNTRYNAME,
957								test_locale + len,
958								LOCALE_NAME_MAX_LENGTH - len))
959			{
960				if (_wcsicmp(argv[0], test_locale) == 0)
961				{
962					wcscpy(argv[1], pStr);
963					*argv[2] = (wchar_t) 1;
964					return FALSE;
965				}
966			}
967		}
968	}
969
970	return TRUE;
971}

973/*
974 * This function converts a Windows locale name to an ISO formatted version
975 * for Visual Studio 2015 or greater.
976 *
977 * Returns NULL, if no valid conversion was found.
978 */
979static char *
980get_iso_localename(const char *winlocname)
981{
982	wchar_t		wc_locale_name[LOCALE_NAME_MAX_LENGTH];
983	wchar_t		buffer[LOCALE_NAME_MAX_LENGTH];
984	static char iso_lc_messages[LOCALE_NAME_MAX_LENGTH];
985	const char *period;
986	int			len;
987	int			ret_val;
988
989	/*
990	 * Valid locales have the following syntax:
991	 * <Language>[_<Country>[.<CodePage>]]
992	 *
993	 * GetLocaleInfoEx can only take locale name without code-page and for the
994	 * purpose of this API the code-page doesn't matter.
995	 */
996	period = strchr(winlocname, '.');
997	if (period != NULL)
998		len = period - winlocname;
999	else
1000		len = pg_mbstrlen(winlocname);
1001
1002	memset(wc_locale_name, 0, sizeof(wc_locale_name));
1003	memset(buffer, 0, sizeof(buffer));
1004	MultiByteToWideChar(CP_ACP, 0, winlocname, len, wc_locale_name,
1005						LOCALE_NAME_MAX_LENGTH);
1006
1007	/*
1008	 * If the lc_messages is already a Unix-style string, we have a direct
1009	 * match with LOCALE_SNAME, e.g. en-US, en_US.
1010	 */
1011	ret_val = GetLocaleInfoEx(wc_locale_name, LOCALE_SNAME, (LPWSTR) &buffer,
1012							  LOCALE_NAME_MAX_LENGTH);
1013	if (!ret_val)
1014	{
1015		/*
1016		 * Search for a locale in the system that matches language and country
1017		 * name.
1018		 */
1019		wchar_t    *argv[3];
1020
1021		argv[0] = wc_locale_name;
1022		argv[1] = buffer;
1023		argv[2] = (wchar_t *) &ret_val;
1024		EnumSystemLocalesEx(search_locale_enum, LOCALE_WINDOWS, (LPARAM) argv,
1025							NULL);
1026	}
1027
1028	if (ret_val)
1029	{
1030		size_t		rc;
1031		char	   *hyphen;
1032
1033		/* Locale names use only ASCII, any conversion locale suffices. */
1034		rc = wchar2char(iso_lc_messages, buffer, sizeof(iso_lc_messages), NULL);
1035		if (rc == -1 || rc == sizeof(iso_lc_messages))
1036			return NULL;
1037
1038		/*
1039		 * Since the message catalogs sit on a case-insensitive filesystem, we
1040		 * need not standardize letter case here.  So long as we do not ship
1041		 * message catalogs for which it would matter, we also need not
1042		 * translate the script/variant portion, e.g.  uz-Cyrl-UZ to
1043		 * uz_UZ@cyrillic.  Simply replace the hyphen with an underscore.
1044		 */
1045		hyphen = strchr(iso_lc_messages, '-');
1046		if (hyphen)
1047			*hyphen = '_';
1048		return iso_lc_messages;
1049	}
1050
1051	return NULL;
1052}

1057	static char iso_lc_messages[LOCALE_NAME_MAX_LENGTH];

1079	pg_locale_t result;
1080	Datum		datum;

1082

1098	result->is_default = false;
1099
1100	Assert((result->collate_is_c && result->collate == NULL) ||
1101		   (!result->collate_is_c && result->collate != NULL));
1102
1103	datum = SysCacheGetAttr(COLLOID, tp, Anum_pg_collation_collversion,
1104							&isnull);

1106	{
1107		char	   *actual_versionstr;
1108		char	   *collversionstr;

1110		collversionstr = TextDatumGetCString(datum);
1111

1113			datum = SysCacheGetAttrNotNull(COLLOID, tp, Anum_pg_collation_collcollate);
1114		else
1115			datum = SysCacheGetAttrNotNull(COLLOID, tp, Anum_pg_collation_colllocale);

1118														 TextDatumGetCString(datum));
1119		if (!actual_versionstr)
1120		{
1121			/*
1122			 * This could happen when specifying a version in CREATE COLLATION
1123			 * but the provider does not support versioning, or manually
1124			 * creating a mess in the catalogs.
1125			 */
1126			ereport(ERROR,
1127					(errmsg("collation \"%s\" has no actual version, but a version was recorded",

1129		}
1130
1131		if (strcmp(actual_versionstr, collversionstr) != 0)
1132			ereport(WARNING,
1133					(errmsg("collation \"%s\" has version mismatch",

1137							   collversionstr, actual_versionstr),
1138					 errhint("Rebuild all objects affected by this collation and run "
1139							 "ALTER COLLATION %s REFRESH VERSION, "
1140							 "or build PostgreSQL with the right library version.",
1141							 quote_qualified_identifier(get_namespace_name(collform->collnamespace),
1142														NameStr(collform->collname)))));
1143	}

1145	ReleaseSysCache(tp);
1146

1148}

1187/*
1188 * Create a pg_locale_t from a collation OID.  Results are cached for the
1189 * lifetime of the backend.  Thus, do not free the result with freelocale().
1190 *
1191 * For simplicity, we always generate COLLATE + CTYPE even though we
1192 * might only need one of them.  Since this is called only once per session,
1193 * it shouldn't cost much.
1194 */
1195pg_locale_t
1196pg_newlocale_from_collation(Oid collid)
1197{
1198	collation_cache_entry *cache_entry;
1199	bool		found;
1200
1201	if (collid == DEFAULT_COLLATION_OID)
1202		return default_locale;
1203
1204	/*
1205	 * Some callers expect C_COLLATION_OID to succeed even without catalog
1206	 * access.
1207	 */
1208	if (collid == C_COLLATION_OID)
1209		return &c_locale;
1210
1211	if (!OidIsValid(collid))
1212		elog(ERROR, "cache lookup failed for collation %u", collid);
1213
1214	AssertCouldGetRelation();
1215
1216	if (last_collation_cache_oid == collid)
1217		return last_collation_cache_locale;
1218
1219	if (CollationCache == NULL)
1220	{
1221		CollationCacheContext = AllocSetContextCreate(TopMemoryContext,
1222													  "collation cache",
1223													  ALLOCSET_DEFAULT_SIZES);
1224		CollationCache = collation_cache_create(CollationCacheContext,
1225												16, NULL);
1226	}
1227
1228	cache_entry = collation_cache_insert(CollationCache, collid, &found);
1229	if (!found)
1230	{
1231		/*
1232		 * Make sure cache entry is marked invalid, in case we fail before
1233		 * setting things.
1234		 */
1235		cache_entry->locale = 0;
1236	}
1237
1238	if (cache_entry->locale == 0)
1239	{
1240		cache_entry->locale = create_pg_locale(collid, CollationCacheContext);
1241	}
1242
1243	last_collation_cache_oid = collid;
1244	last_collation_cache_locale = cache_entry->locale;
1245
1246	return cache_entry->locale;
1247}

1481/*
1482 * Return required encoding ID for the given locale, or -1 if any encoding is
1483 * valid for the locale.
1484 */
1485int
1486builtin_locale_encoding(const char *locale)
1487{
1488	if (strcmp(locale, "C") == 0)
1489		return -1;
1490	else if (strcmp(locale, "C.UTF-8") == 0)
1491		return PG_UTF8;
1492	else if (strcmp(locale, "PG_UNICODE_FAST") == 0)
1493		return PG_UTF8;
1494
1495
1496	ereport(ERROR,
1497			(errcode(ERRCODE_WRONG_OBJECT_TYPE),
1498			 errmsg("invalid locale name \"%s\" for builtin provider",
1499					locale)));
1500
1501	return 0;					/* keep compiler quiet */
1502}

1505/*
1506 * Validate the locale and encoding combination, and return the canonical form
1507 * of the locale name.
1508 */
1509const char *
1510builtin_validate_locale(int encoding, const char *locale)
1511{
1512	const char *canonical_name = NULL;
1513	int			required_encoding;
1514
1515	if (strcmp(locale, "C") == 0)
1516		canonical_name = "C";
1517	else if (strcmp(locale, "C.UTF-8") == 0 || strcmp(locale, "C.UTF8") == 0)
1518		canonical_name = "C.UTF-8";
1519	else if (strcmp(locale, "PG_UNICODE_FAST") == 0)
1520		canonical_name = "PG_UNICODE_FAST";
1521
1522	if (!canonical_name)
1523		ereport(ERROR,
1524				(errcode(ERRCODE_WRONG_OBJECT_TYPE),
1525				 errmsg("invalid locale name \"%s\" for builtin provider",
1526						locale)));
1527
1528	required_encoding = builtin_locale_encoding(canonical_name);
1529	if (required_encoding >= 0 && encoding != required_encoding)
1530		ereport(ERROR,
1531				(errcode(ERRCODE_WRONG_OBJECT_TYPE),
1532				 errmsg("encoding \"%s\" does not match locale \"%s\"",
1533						pg_encoding_to_char(encoding), locale)));
1534
1535	return canonical_name;
1536}

1540/*
1541 * Return the BCP47 language tag representation of the requested locale.
1542 *
1543 * This function should be called before passing the string to ucol_open(),
1544 * because conversion to a language tag also performs "level 2
1545 * canonicalization". In addition to producing a consistent format, level 2
1546 * canonicalization is able to more accurately interpret different input
1547 * locale string formats, such as POSIX and .NET IDs.
1548 */
1549char *
1550icu_language_tag(const char *loc_str, int elevel)
1551{
1552#ifdef USE_ICU
1553	UErrorCode	status;
1554	char	   *langtag;
1555	size_t		buflen = 32;	/* arbitrary starting buffer size */
1556	const bool	strict = true;
1557
1558	/*
1559	 * A BCP47 language tag doesn't have a clearly-defined upper limit (cf.
1560	 * RFC5646 section 4.4). Additionally, in older ICU versions,
1561	 * uloc_toLanguageTag() doesn't always return the ultimate length on the
1562	 * first call, necessitating a loop.
1563	 */
1564	langtag = palloc(buflen);
1565	while (true)
1566	{
1567		status = U_ZERO_ERROR;
1568		uloc_toLanguageTag(loc_str, langtag, buflen, strict, &status);
1569
1570		/* try again if the buffer is not large enough */
1571		if ((status == U_BUFFER_OVERFLOW_ERROR ||
1572			 status == U_STRING_NOT_TERMINATED_WARNING) &&
1573			buflen < MaxAllocSize)
1574		{
1575			buflen = Min(buflen * 2, MaxAllocSize);
1576			langtag = repalloc(langtag, buflen);
1577			continue;
1578		}
1579
1580		break;
1581	}
1582
1583	if (U_FAILURE(status))
1584	{
1585		pfree(langtag);
1586
1587		if (elevel > 0)
1588			ereport(elevel,
1589					(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1590					 errmsg("could not convert locale name \"%s\" to language tag: %s",
1591							loc_str, u_errorName(status))));
1592		return NULL;
1593	}
1594
1595	return langtag;
1596#else							/* not USE_ICU */
1597	ereport(ERROR,
1598			(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
1599			 errmsg("ICU is not supported in this build")));
1600	return NULL;				/* keep compiler quiet */
1601#endif							/* not USE_ICU */
1602}

1604/*
1605 * Perform best-effort check that the locale is a valid one.
1606 */
1607void
1608icu_validate_locale(const char *loc_str)
1609{
1610#ifdef USE_ICU
1611	UCollator  *collator;
1612	UErrorCode	status;
1613	char		lang[ULOC_LANG_CAPACITY];
1614	bool		found = false;
1615	int			elevel = icu_validation_level;
1616
1617	/* no validation */
1618	if (elevel < 0)
1619		return;
1620
1621	/* downgrade to WARNING during pg_upgrade */
1622	if (IsBinaryUpgrade && elevel > WARNING)
1623		elevel = WARNING;
1624
1625	/* validate that we can extract the language */
1626	status = U_ZERO_ERROR;
1627	uloc_getLanguage(loc_str, lang, ULOC_LANG_CAPACITY, &status);
1628	if (U_FAILURE(status) || status == U_STRING_NOT_TERMINATED_WARNING)
1629	{
1630		ereport(elevel,
1631				(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1632				 errmsg("could not get language from ICU locale \"%s\": %s",
1633						loc_str, u_errorName(status)),
1634				 errhint("To disable ICU locale validation, set the parameter \"%s\" to \"%s\".",
1635						 "icu_validation_level", "disabled")));
1636		return;
1637	}
1638
1639	/* check for special language name */
1640	if (strcmp(lang, "") == 0 ||
1641		strcmp(lang, "root") == 0 || strcmp(lang, "und") == 0)
1642		found = true;
1643
1644	/* search for matching language within ICU */
1645	for (int32_t i = 0; !found && i < uloc_countAvailable(); i++)
1646	{
1647		const char *otherloc = uloc_getAvailable(i);
1648		char		otherlang[ULOC_LANG_CAPACITY];
1649
1650		status = U_ZERO_ERROR;
1651		uloc_getLanguage(otherloc, otherlang, ULOC_LANG_CAPACITY, &status);
1652		if (U_FAILURE(status) || status == U_STRING_NOT_TERMINATED_WARNING)
1653			continue;
1654
1655		if (strcmp(lang, otherlang) == 0)
1656			found = true;
1657	}
1658
1659	if (!found)
1660		ereport(elevel,
1661				(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
1662				 errmsg("ICU locale \"%s\" has unknown language \"%s\"",
1663						loc_str, lang),
1664				 errhint("To disable ICU locale validation, set the parameter \"%s\" to \"%s\".",
1665						 "icu_validation_level", "disabled")));
1666
1667	/* check that it can be opened */
1668	collator = pg_ucol_open(loc_str);
1669	ucol_close(collator);
1670#else							/* not USE_ICU */
1671	/* could get here if a collation was created by a build with ICU */
1672	ereport(ERROR,
1673			(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
1674			 errmsg("ICU is not supported in this build")));
1675#endif							/* not USE_ICU */
1676}
