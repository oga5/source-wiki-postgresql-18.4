34/* can be used in either frontend or backend */
35#include "postgres_fe.h"
36
37#include "mb/pg_wchar.h"
38
39int
40pg_wchar_strncmp(const pg_wchar *s1, const pg_wchar *s2, size_t n)
41{
42	if (n == 0)
43		return 0;
44	do
45	{
46		if (*s1 != *s2++)
47			return (*s1 - *(s2 - 1));
48		if (*s1++ == 0)
49			break;
50	} while (--n != 0);
51	return 0;
52}
53
54int
55pg_char_and_wchar_strncmp(const char *s1, const pg_wchar *s2, size_t n)
56{
57	if (n == 0)
58		return 0;
59	do
60	{
61		if ((pg_wchar) ((unsigned char) *s1) != *s2++)
62			return ((pg_wchar) ((unsigned char) *s1) - *(s2 - 1));
63		if (*s1++ == 0)
64			break;
65	} while (--n != 0);
66	return 0;
67}
68
69size_t
70pg_wchar_strlen(const pg_wchar *str)
71{
72	const pg_wchar *s;
73
74	for (s = str; *s; ++s)
75		;
76	return (s - str);
77}
