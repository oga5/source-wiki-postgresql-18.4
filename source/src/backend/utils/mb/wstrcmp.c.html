35/* can be used in either frontend or backend */
36#include "postgres_fe.h"
37
38#include "mb/pg_wchar.h"
39
40int
41pg_char_and_wchar_strcmp(const char *s1, const pg_wchar *s2)
42{
43	while ((pg_wchar) *s1 == *s2++)
44		if (*s1++ == 0)
45			return 0;
46	return *(const unsigned char *) s1 - *(const pg_wchar *) (s2 - 1);
47}
