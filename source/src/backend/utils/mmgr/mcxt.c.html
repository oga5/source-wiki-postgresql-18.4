191/*
192 * GetMemoryChunkMethodID
193 *		Return the MemoryContextMethodID from the uint64 chunk header which
194 *		directly precedes 'pointer'.
195 */
196static inline MemoryContextMethodID
197GetMemoryChunkMethodID(const void *pointer)
198{
199	uint64		header;
200
201	/*
202	 * Try to detect bogus pointers handed to us, poorly though we can.
203	 * Presumably, a pointer that isn't MAXALIGNED isn't pointing at an
204	 * allocated chunk.
205	 */
206	Assert(pointer == (const void *) MAXALIGN(pointer));
207
208	/* Allow access to the uint64 header */
209	VALGRIND_MAKE_MEM_DEFINED((char *) pointer - sizeof(uint64), sizeof(uint64));
210
211	header = *((const uint64 *) ((const char *) pointer - sizeof(uint64)));
212
213	/* Disallow access to the uint64 header */
214	VALGRIND_MAKE_MEM_NOACCESS((char *) pointer - sizeof(uint64), sizeof(uint64));
215
216	return (MemoryContextMethodID) (header & MEMORY_CONTEXT_METHODID_MASK);
217}

222 *
223 * This is only used after GetMemoryChunkMethodID, so no need for error checks.

226GetMemoryChunkHeader(const void *pointer)
227{
228	uint64		header;
229
230	/* Allow access to the uint64 header */
231	VALGRIND_MAKE_MEM_DEFINED((char *) pointer - sizeof(uint64), sizeof(uint64));

236	VALGRIND_MAKE_MEM_NOACCESS((char *) pointer - sizeof(uint64), sizeof(uint64));

241/*
242 * MemoryContextTraverseNext
243 *		Helper function to traverse all descendants of a memory context
244 *		without recursion.
245 *
246 * Recursion could lead to out-of-stack errors with deep context hierarchies,
247 * which would be unpleasant in error cleanup code paths.
248 *
249 * To process 'context' and all its descendants, use a loop like this:
250 *
251 *     <process 'context'>
252 *     for (MemoryContext curr = context->firstchild;
253 *          curr != NULL;
254 *          curr = MemoryContextTraverseNext(curr, context))
255 *     {
256 *         <process 'curr'>
257 *     }
258 *
259 * This visits all the contexts in pre-order, that is a node is visited
260 * before its children.
261 */
262static MemoryContext
263MemoryContextTraverseNext(MemoryContext curr, MemoryContext top)
264{
265	/* After processing a node, traverse to its first child if any */
266	if (curr->firstchild != NULL)
267		return curr->firstchild;
268
269	/*
270	 * After processing a childless node, traverse to its next sibling if
271	 * there is one.  If there isn't, traverse back up to the parent (which
272	 * has already been visited, and now so have all its descendants).  We're
273	 * done if that is "top", otherwise traverse to its next sibling if any,
274	 * otherwise repeat moving up.
275	 */
276	while (curr->nextchild == NULL)
277	{
278		curr = curr->parent;
279		if (curr == top)
280			return NULL;
281	}
282	return curr->nextchild;
283}

328/*
329 * MemoryContextInit
330 *		Start up the memory-context subsystem.
331 *
332 * This must be called before creating contexts or allocating memory in
333 * contexts.  TopMemoryContext and ErrorContext are initialized here;
334 * other contexts must be created afterwards.
335 *
336 * In normal multi-backend operation, this is called once during
337 * postmaster startup, and not at all by individual backend startup
338 * (since the backends inherit an already-initialized context subsystem
339 * by virtue of being forked off the postmaster).  But in an EXEC_BACKEND
340 * build, each process must do this for itself.
341 *
342 * In a standalone backend this must be called during backend startup.
343 */
344void
345MemoryContextInit(void)
346{
347	Assert(TopMemoryContext == NULL);
348
349	/*
350	 * First, initialize TopMemoryContext, which is the parent of all others.
351	 */
352	TopMemoryContext = AllocSetContextCreate((MemoryContext) NULL,
353											 "TopMemoryContext",
354											 ALLOCSET_DEFAULT_SIZES);
355
356	/*
357	 * Not having any other place to point CurrentMemoryContext, make it point
358	 * to TopMemoryContext.  Caller should change this soon!
359	 */
360	CurrentMemoryContext = TopMemoryContext;
361
362	/*
363	 * Initialize ErrorContext as an AllocSetContext with slow growth rate ---
364	 * we don't really expect much to be allocated in it. More to the point,
365	 * require it to contain at least 8K at all times. This is the only case
366	 * where retained memory in a context is *essential* --- we want to be
367	 * sure ErrorContext still has some memory even if we've run out
368	 * elsewhere! Also, allow allocations in ErrorContext within a critical
369	 * section. Otherwise a PANIC will cause an assertion failure in the error
370	 * reporting code, before printing out the real cause of the failure.
371	 *
372	 * This should be the last step in this function, as elog.c assumes memory
373	 * management works once ErrorContext is non-null.
374	 */
375	ErrorContext = AllocSetContextCreate(TopMemoryContext,
376										 "ErrorContext",
377										 8 * 1024,
378										 8 * 1024,
379										 8 * 1024);
380	MemoryContextAllowInCriticalSection(ErrorContext, true);
381}

383/*
384 * MemoryContextReset
385 *		Release all space allocated within a context and delete all its
386 *		descendant contexts (but not the named context itself).
387 */
388void
389MemoryContextReset(MemoryContext context)
390{
391	Assert(MemoryContextIsValid(context));
392
393	/* save a function call in common case where there are no children */
394	if (context->firstchild != NULL)
395		MemoryContextDeleteChildren(context);
396
397	/* save a function call if no pallocs since startup or last reset */
398	if (!context->isReset)
399		MemoryContextResetOnly(context);
400}
401
402/*
403 * MemoryContextResetOnly
404 *		Release all space allocated within a context.
405 *		Nothing is done to the context's descendant contexts.
406 */
407void
408MemoryContextResetOnly(MemoryContext context)
409{
410	Assert(MemoryContextIsValid(context));
411
412	/* Nothing to do if no pallocs since startup or last reset */
413	if (!context->isReset)
414	{
415		MemoryContextCallResetCallbacks(context);
416
417		/*
418		 * If context->ident points into the context's memory, it will become
419		 * a dangling pointer.  We could prevent that by setting it to NULL
420		 * here, but that would break valid coding patterns that keep the
421		 * ident elsewhere, e.g. in a parent context.  So for now we assume
422		 * the programmer got it right.
423		 */
424
425		context->methods->reset(context);
426		context->isReset = true;
427		VALGRIND_DESTROY_MEMPOOL(context);
428		VALGRIND_CREATE_MEMPOOL(context, 0, false);
429	}
430}

432/*
433 * MemoryContextResetChildren
434 *		Release all space allocated within a context's descendants,
435 *		but don't delete the contexts themselves.  The named context
436 *		itself is not touched.
437 */
438void
439MemoryContextResetChildren(MemoryContext context)
440{
441	Assert(MemoryContextIsValid(context));
442
443	for (MemoryContext curr = context->firstchild;
444		 curr != NULL;
445		 curr = MemoryContextTraverseNext(curr, context))
446	{
447		MemoryContextResetOnly(curr);
448	}
449}

451/*
452 * MemoryContextDelete
453 *		Delete a context and its descendants, and release all space
454 *		allocated therein.
455 *
456 * The type-specific delete routine removes all storage for the context,
457 * but we have to deal with descendant nodes here.
458 */
459void
460MemoryContextDelete(MemoryContext context)
461{
462	MemoryContext curr;
463
464	Assert(MemoryContextIsValid(context));
465
466	/*
467	 * Delete subcontexts from the bottom up.
468	 *
469	 * Note: Do not use recursion here.  A "stack depth limit exceeded" error
470	 * would be unpleasant if we're already in the process of cleaning up from
471	 * transaction abort.  We also cannot use MemoryContextTraverseNext() here
472	 * because we modify the tree as we go.
473	 */
474	curr = context;
475	for (;;)
476	{
477		MemoryContext parent;
478
479		/* Descend down until we find a leaf context with no children */
480		while (curr->firstchild != NULL)
481			curr = curr->firstchild;
482
483		/*
484		 * We're now at a leaf with no children. Free it and continue from the
485		 * parent.  Or if this was the original node, we're all done.
486		 */
487		parent = curr->parent;
488		MemoryContextDeleteOnly(curr);
489
490		if (curr == context)
491			break;
492		curr = parent;
493	}
494}

496/*
497 * Subroutine of MemoryContextDelete,
498 * to delete a context that has no children.
499 * We must also delink the context from its parent, if it has one.
500 */
501static void
502MemoryContextDeleteOnly(MemoryContext context)
503{
504	Assert(MemoryContextIsValid(context));
505	/* We had better not be deleting TopMemoryContext ... */
506	Assert(context != TopMemoryContext);
507	/* And not CurrentMemoryContext, either */
508	Assert(context != CurrentMemoryContext);
509	/* All the children should've been deleted already */
510	Assert(context->firstchild == NULL);
511
512	/*
513	 * It's not entirely clear whether 'tis better to do this before or after
514	 * delinking the context; but an error in a callback will likely result in
515	 * leaking the whole context (if it's not a root context) if we do it
516	 * after, so let's do it before.
517	 */
518	MemoryContextCallResetCallbacks(context);
519
520	/*
521	 * We delink the context from its parent before deleting it, so that if
522	 * there's an error we won't have deleted/busted contexts still attached
523	 * to the context tree.  Better a leak than a crash.
524	 */
525	MemoryContextSetParent(context, NULL);
526
527	/*
528	 * Also reset the context's ident pointer, in case it points into the
529	 * context.  This would only matter if someone tries to get stats on the
530	 * (already unlinked) context, which is unlikely, but let's be safe.
531	 */
532	context->ident = NULL;
533
534	context->methods->delete_context(context);
535
536	VALGRIND_DESTROY_MEMPOOL(context);
537}

539/*
540 * MemoryContextDeleteChildren
541 *		Delete all the descendants of the named context and release all
542 *		space allocated therein.  The named context itself is not touched.
543 */
544void
545MemoryContextDeleteChildren(MemoryContext context)
546{
547	Assert(MemoryContextIsValid(context));
548
549	/*
550	 * MemoryContextDelete will delink the child from me, so just iterate as
551	 * long as there is a child.
552	 */
553	while (context->firstchild != NULL)
554		MemoryContextDelete(context->firstchild);
555}

557/*
558 * MemoryContextRegisterResetCallback
559 *		Register a function to be called before next context reset/delete.
560 *		Such callbacks will be called in reverse order of registration.
561 *
562 * The caller is responsible for allocating a MemoryContextCallback struct
563 * to hold the info about this callback request, and for filling in the
564 * "func" and "arg" fields in the struct to show what function to call with
565 * what argument.  Typically the callback struct should be allocated within
566 * the specified context, since that means it will automatically be freed
567 * when no longer needed.
568 *
569 * There is no API for deregistering a callback once registered.  If you
570 * want it to not do anything anymore, adjust the state pointed to by its
571 * "arg" to indicate that.
572 */
573void
574MemoryContextRegisterResetCallback(MemoryContext context,
575								   MemoryContextCallback *cb)
576{
577	Assert(MemoryContextIsValid(context));
578
579	/* Push onto head so this will be called before older registrants. */
580	cb->next = context->reset_cbs;
581	context->reset_cbs = cb;
582	/* Mark the context as non-reset (it probably is already). */
583	context->isReset = false;
584}
585
586/*
587 * MemoryContextCallResetCallbacks
588 *		Internal function to call all registered callbacks for context.
589 */
590static void
591MemoryContextCallResetCallbacks(MemoryContext context)
592{
593	MemoryContextCallback *cb;
594
595	/*
596	 * We pop each callback from the list before calling.  That way, if an
597	 * error occurs inside the callback, we won't try to call it a second time
598	 * in the likely event that we reset or delete the context later.
599	 */
600	while ((cb = context->reset_cbs) != NULL)
601	{
602		context->reset_cbs = cb->next;
603		cb->func(cb->arg);
604	}
605}

624/*
625 * MemoryContextSetParent
626 *		Change a context to belong to a new parent (or no parent).
627 *
628 * We provide this as an API function because it is sometimes useful to
629 * change a context's lifespan after creation.  For example, a context
630 * might be created underneath a transient context, filled with data,
631 * and then reparented underneath CacheMemoryContext to make it long-lived.
632 * In this way no special effort is needed to get rid of the context in case
633 * a failure occurs before its contents are completely set up.
634 *
635 * Callers often assume that this function cannot fail, so don't put any
636 * elog(ERROR) calls in it.
637 *
638 * A possible caller error is to reparent a context under itself, creating
639 * a loop in the context graph.  We assert here that context != new_parent,
640 * but checking for multi-level loops seems more trouble than it's worth.
641 */
642void
643MemoryContextSetParent(MemoryContext context, MemoryContext new_parent)
644{
645	Assert(MemoryContextIsValid(context));
646	Assert(context != new_parent);
647
648	/* Fast path if it's got correct parent already */
649	if (new_parent == context->parent)
650		return;
651
652	/* Delink from existing parent, if any */
653	if (context->parent)
654	{
655		MemoryContext parent = context->parent;
656
657		if (context->prevchild != NULL)
658			context->prevchild->nextchild = context->nextchild;
659		else
660		{
661			Assert(parent->firstchild == context);
662			parent->firstchild = context->nextchild;
663		}
664
665		if (context->nextchild != NULL)
666			context->nextchild->prevchild = context->prevchild;
667	}
668
669	/* And relink */
670	if (new_parent)
671	{
672		Assert(MemoryContextIsValid(new_parent));
673		context->parent = new_parent;
674		context->prevchild = NULL;
675		context->nextchild = new_parent->firstchild;
676		if (new_parent->firstchild != NULL)
677			new_parent->firstchild->prevchild = context;
678		new_parent->firstchild = context;
679	}
680	else
681	{
682		context->parent = NULL;
683		context->prevchild = NULL;
684		context->nextchild = NULL;
685	}
686}

688/*
689 * MemoryContextAllowInCriticalSection
690 *		Allow/disallow allocations in this memory context within a critical

692 *
693 * Normally, memory allocations are not allowed within a critical section,

874/*
875 * MemoryContextStatsInternal
876 *		One recursion level for MemoryContextStats
877 *
878 * Print stats for this context if possible, but in any case accumulate counts
879 * into *totals (if not NULL).
880 */
881static void
882MemoryContextStatsInternal(MemoryContext context, int level,
883						   int max_level, int max_children,
884						   MemoryContextCounters *totals,
885						   bool print_to_stderr)
886{
887	MemoryContext child;
888	int			ichild;
889
890	Assert(MemoryContextIsValid(context));
891
892	/* Examine the context itself */
893	context->methods->stats(context,
894							MemoryContextStatsPrint,
895							&level,
896							totals, print_to_stderr);
897
898	/*
899	 * Examine children.
900	 *
901	 * If we are past the recursion depth limit or already running low on
902	 * stack, do not print them explicitly but just summarize them. Similarly,
903	 * if there are more than max_children of them, we do not print the rest
904	 * explicitly, but just summarize them.
905	 */
906	child = context->firstchild;
907	ichild = 0;
908	if (level <= max_level && !stack_is_too_deep())
909	{
910		for (; child != NULL && ichild < max_children;
911			 child = child->nextchild, ichild++)
912		{
913			MemoryContextStatsInternal(child, level + 1,
914									   max_level, max_children,
915									   totals,
916									   print_to_stderr);
917		}
918	}
919
920	if (child != NULL)
921	{
922		/* Summarize the rest of the children, avoiding recursion. */
923		MemoryContextCounters local_totals;
924
925		memset(&local_totals, 0, sizeof(local_totals));
926
927		ichild = 0;
928		while (child != NULL)
929		{
930			child->methods->stats(child, NULL, NULL, &local_totals, false);
931			ichild++;
932			child = MemoryContextTraverseNext(child, context);
933		}
934
935		if (print_to_stderr)
936		{
937			for (int i = 0; i < level; i++)
938				fprintf(stderr, "  ");
939			fprintf(stderr,
940					"%d more child contexts containing %zu total in %zu blocks; %zu free (%zu chunks); %zu used\n",
941					ichild,
942					local_totals.totalspace,
943					local_totals.nblocks,
944					local_totals.freespace,
945					local_totals.freechunks,
946					local_totals.totalspace - local_totals.freespace);
947		}
948		else
949			ereport(LOG_SERVER_ONLY,
950					(errhidestmt(true),
951					 errhidecontext(true),
952					 errmsg_internal("level: %d; %d more child contexts containing %zu total in %zu blocks; %zu free (%zu chunks); %zu used",
953									 level,
954									 ichild,
955									 local_totals.totalspace,
956									 local_totals.nblocks,
957									 local_totals.freespace,
958									 local_totals.freechunks,
959									 local_totals.totalspace - local_totals.freespace)));
960
961		if (totals)
962		{
963			totals->nblocks += local_totals.nblocks;
964			totals->freechunks += local_totals.freechunks;
965			totals->totalspace += local_totals.totalspace;
966			totals->freespace += local_totals.freespace;
967		}
968	}
969}

971/*
972 * MemoryContextStatsPrint
973 *		Print callback used by MemoryContextStatsInternal
974 *
975 * For now, the passthru pointer just points to "int level"; later we might
976 * make that more complicated.
977 */
978static void
979MemoryContextStatsPrint(MemoryContext context, void *passthru,
980						const char *stats_string,
981						bool print_to_stderr)
982{
983	int			level = *(int *) passthru;
984	const char *name = context->name;
985	const char *ident = context->ident;
986	char		truncated_ident[110];
987	int			i;
988
989	/*
990	 * It seems preferable to label dynahash contexts with just the hash table
991	 * name.  Those are already unique enough, so the "dynahash" part isn't
992	 * very helpful, and this way is more consistent with pre-v11 practice.
993	 */
994	if (ident && strcmp(name, "dynahash") == 0)
995	{
996		name = ident;
997		ident = NULL;
998	}
999
1000	truncated_ident[0] = '\0';
1001
1002	if (ident)
1003	{
1004		/*
1005		 * Some contexts may have very long identifiers (e.g., SQL queries).
1006		 * Arbitrarily truncate at 100 bytes, but be careful not to break
1007		 * multibyte characters.  Also, replace ASCII control characters, such
1008		 * as newlines, with spaces.
1009		 */
1010		int			idlen = strlen(ident);
1011		bool		truncated = false;
1012
1013		strcpy(truncated_ident, ": ");
1014		i = strlen(truncated_ident);
1015
1016		if (idlen > 100)
1017		{
1018			idlen = pg_mbcliplen(ident, idlen, 100);
1019			truncated = true;
1020		}
1021
1022		while (idlen-- > 0)
1023		{
1024			unsigned char c = *ident++;
1025
1026			if (c < ' ')
1027				c = ' ';
1028			truncated_ident[i++] = c;
1029		}
1030		truncated_ident[i] = '\0';
1031
1032		if (truncated)
1033			strcat(truncated_ident, "...");
1034	}
1035
1036	if (print_to_stderr)
1037	{
1038		for (i = 1; i < level; i++)
1039			fprintf(stderr, "  ");
1040		fprintf(stderr, "%s: %s%s\n", name, stats_string, truncated_ident);
1041	}
1042	else
1043		ereport(LOG_SERVER_ONLY,
1044				(errhidestmt(true),
1045				 errhidecontext(true),
1046				 errmsg_internal("level: %d; %s: %s%s",
1047								 level, name, stats_string, truncated_ident)));
1048}

1073/*
1074 * MemoryContextCreate
1075 *		Context-type-independent part of context creation.
1076 *
1077 * This is only intended to be called by context-type-specific
1078 * context creation routines, not by the unwashed masses.
1079 *
1080 * The memory context creation procedure goes like this:
1081 *	1.  Context-type-specific routine makes some initial space allocation,
1082 *		including enough space for the context header.  If it fails,
1083 *		it can ereport() with no damage done.
1084 *	2.	Context-type-specific routine sets up all type-specific fields of
1085 *		the header (those beyond MemoryContextData proper), as well as any
1086 *		other management fields it needs to have a fully valid context.
1087 *		Usually, failure in this step is impossible, but if it's possible
1088 *		the initial space allocation should be freed before ereport'ing.
1089 *	3.	Context-type-specific routine calls MemoryContextCreate() to fill in
1090 *		the generic header fields and link the context into the context tree.
1091 *	4.  We return to the context-type-specific routine, which finishes
1092 *		up type-specific initialization.  This routine can now do things
1093 *		that might fail (like allocate more memory), so long as it's
1094 *		sure the node is left in a state that delete will handle.
1095 *
1096 * node: the as-yet-uninitialized common part of the context header node.
1097 * tag: NodeTag code identifying the memory context type.
1098 * method_id: MemoryContextMethodID of the context-type being created.
1099 * parent: parent context, or NULL if this will be a top-level context.
1100 * name: name of context (must be statically allocated).
1101 *
1102 * Context routines generally assume that MemoryContextCreate can't fail,
1103 * so this can contain Assert but not elog/ereport.
1104 */
1105void
1106MemoryContextCreate(MemoryContext node,
1107					NodeTag tag,
1108					MemoryContextMethodID method_id,
1109					MemoryContext parent,
1110					const char *name)
1111{
1112	/* Creating new memory contexts is not allowed in a critical section */
1113	Assert(CritSectionCount == 0);
1114
1115	/* Validate parent, to help prevent crazy context linkages */
1116	Assert(parent == NULL || MemoryContextIsValid(parent));
1117	Assert(node != parent);
1118
1119	/* Initialize all standard fields of memory context header */
1120	node->type = tag;
1121	node->isReset = true;
1122	node->methods = &mcxt_methods[method_id];
1123	node->parent = parent;
1124	node->firstchild = NULL;
1125	node->mem_allocated = 0;
1126	node->prevchild = NULL;
1127	node->name = name;
1128	node->ident = NULL;
1129	node->reset_cbs = NULL;
1130
1131	/* OK to link node into context tree */
1132	if (parent)
1133	{
1134		node->nextchild = parent->firstchild;
1135		if (parent->firstchild != NULL)
1136			parent->firstchild->prevchild = node;
1137		parent->firstchild = node;
1138		/* inherit allowInCritSection flag from parent */
1139		node->allowInCritSection = parent->allowInCritSection;
1140	}
1141	else
1142	{
1143		node->nextchild = NULL;
1144		node->allowInCritSection = false;
1145	}
1146
1147	VALGRIND_CREATE_MEMPOOL(node, 0, false);
1148}

1289/*
1290 * ProcessLogMemoryContextInterrupt
1291 * 		Perform logging of memory contexts of this backend process.
1292 *
1293 * Any backend that participates in ProcSignal signaling must arrange
1294 * to call this function if we see LogMemoryContextPending set.
1295 * It is called from CHECK_FOR_INTERRUPTS(), which is enough because
1296 * the target process for logging of memory contexts is a backend.
1297 */
1298void
1299ProcessLogMemoryContextInterrupt(void)
1300{
1301	LogMemoryContextPending = false;
1302
1303	/*
1304	 * Exit immediately if memory context logging is already in progress. This
1305	 * prevents recursive calls, which could occur if logging is requested
1306	 * repeatedly and rapidly, potentially leading to infinite recursion and a
1307	 * crash.
1308	 */
1309	if (LogMemoryContextInProgress)
1310		return;
1311	LogMemoryContextInProgress = true;
1312
1313	PG_TRY();
1314	{
1315		/*
1316		 * Use LOG_SERVER_ONLY to prevent this message from being sent to the
1317		 * connected client.
1318		 */
1319		ereport(LOG_SERVER_ONLY,
1320				(errhidestmt(true),
1321				 errhidecontext(true),
1322				 errmsg("logging memory contexts of PID %d", MyProcPid)));
1323
1324		/*
1325		 * When a backend process is consuming huge memory, logging all its
1326		 * memory contexts might overrun available disk space. To prevent
1327		 * this, we limit the depth of the hierarchy, as well as the number of
1328		 * child contexts to log per parent to 100.
1329		 *
1330		 * As with MemoryContextStats(), we suppose that practical cases where
1331		 * the dump gets long will typically be huge numbers of siblings under
1332		 * the same parent context; while the additional debugging value from
1333		 * seeing details about individual siblings beyond 100 will not be
1334		 * large.
1335		 */
1336		MemoryContextStatsDetail(TopMemoryContext, 100, 100, false);
1337	}
1338	PG_FINALLY();
1339	{
1340		LogMemoryContextInProgress = false;
1341	}
1342	PG_END_TRY();
1343}

1429 *
1430 * Currently, we align addresses by requesting additional bytes from the
1431 * MemoryContext's standard allocator function and then aligning the returned
1432 * address by the required alignment.  This means that the given MemoryContext
1433 * must support providing us with a chunk of memory that's larger than 'size'.
1434 * For allocators such as Slab, that's not going to work, as slab only allows
1435 * chunks of the size that's specified when the context is created.
1436 *
1437 * 'alignto' must be a power of 2.
1438 * 'flags' may be 0 or set the same as MemoryContextAllocExtended().
1439 */
1440void *
1441MemoryContextAllocAligned(MemoryContext context,
1442						  Size size, Size alignto, int flags)
1443{
1444	MemoryChunk *alignedchunk;
1445	Size		alloc_size;
1446	void	   *unaligned;
1447	void	   *aligned;
1448
1449	/* wouldn't make much sense to waste that much space */
1450	Assert(alignto < (128 * 1024 * 1024));
1451
1452	/* ensure alignto is a power of 2 */
1453	Assert((alignto & (alignto - 1)) == 0);
1454
1455	/*
1456	 * If the alignment requirements are less than what we already guarantee
1457	 * then just use the standard allocation function.
1458	 */
1459	if (unlikely(alignto <= MAXIMUM_ALIGNOF))
1460		return MemoryContextAllocExtended(context, size, flags);
1461
1462	/*
1463	 * We implement aligned pointers by simply allocating enough memory for
1464	 * the requested size plus the alignment and an additional "redirection"
1465	 * MemoryChunk.  This additional MemoryChunk is required for operations
1466	 * such as pfree when used on the pointer returned by this function.  We
1467	 * use this redirection MemoryChunk in order to find the pointer to the
1468	 * memory that was returned by the MemoryContextAllocExtended call below.
1469	 * We do that by "borrowing" the block offset field and instead of using
1470	 * that to find the offset into the owning block, we use it to find the
1471	 * original allocated address.
1472	 *
1473	 * Here we must allocate enough extra memory so that we can still align
1474	 * the pointer returned by MemoryContextAllocExtended and also have enough
1475	 * space for the redirection MemoryChunk.  Since allocations will already
1476	 * be at least aligned by MAXIMUM_ALIGNOF, we can subtract that amount
1477	 * from the allocation size to save a little memory.
1478	 */
1479	alloc_size = size + PallocAlignedExtraBytes(alignto);
1480
1481#ifdef MEMORY_CONTEXT_CHECKING
1482	/* ensure there's space for a sentinel byte */
1483	alloc_size += 1;
1484#endif
1485
1486	/* perform the actual allocation */
1487	unaligned = MemoryContextAllocExtended(context, alloc_size, flags);
1488
1489	/* set the aligned pointer */
1490	aligned = (void *) TYPEALIGN(alignto, (char *) unaligned +
1491								 sizeof(MemoryChunk));
1492
1493	alignedchunk = PointerGetMemoryChunk(aligned);
1494
1495	/*
1496	 * We set the redirect MemoryChunk so that the block offset calculation is
1497	 * used to point back to the 'unaligned' allocated chunk.  This allows us
1498	 * to use MemoryChunkGetBlock() to find the unaligned chunk when we need
1499	 * to perform operations such as pfree() and repalloc().
1500	 *
1501	 * We store 'alignto' in the MemoryChunk's 'value' so that we know what
1502	 * the alignment was set to should we ever be asked to realloc this
1503	 * pointer.
1504	 */
1505	MemoryChunkSetHdrMask(alignedchunk, unaligned, alignto,
1506						  MCTX_ALIGNED_REDIRECT_ID);
1507
1508	/* double check we produced a correctly aligned pointer */
1509	Assert((void *) TYPEALIGN(alignto, aligned) == aligned);
1510
1511#ifdef MEMORY_CONTEXT_CHECKING
1512	alignedchunk->requested_size = size;
1513	/* set mark to catch clobber of "unused" space */
1514	set_sentinel(aligned, size);
1515#endif
1516
1517	/* Mark the bytes before the redirection header as noaccess */
1518	VALGRIND_MAKE_MEM_NOACCESS(unaligned,
1519							   (char *) alignedchunk - (char *) unaligned);
1520
1521	/* Disallow access to the redirection chunk header. */
1522	VALGRIND_MAKE_MEM_NOACCESS(alignedchunk, sizeof(MemoryChunk));
1523
1524	return aligned;
1525}

1569 * repalloc
1570 *		Adjust the size of a previously allocated chunk.
1571 */
1572void *
1573repalloc(void *pointer, Size size)
1574{
1575#ifdef USE_VALGRIND
1576	MemoryContextMethodID method = GetMemoryChunkMethodID(pointer);
1577#endif
1578#if defined(USE_ASSERT_CHECKING) || defined(USE_VALGRIND)
1579	MemoryContext context = GetMemoryChunkContext(pointer);
1580#endif
1581	void	   *ret;
1582
1583	AssertNotInCriticalSection(context);
1584
1585	/* isReset must be false already */
1586	Assert(!context->isReset);
1587
1588	/*
1589	 * For efficiency reasons, we purposefully offload the handling of
1590	 * allocation failures to the MemoryContextMethods implementation as this
1591	 * allows these checks to be performed only when an actual malloc needs to
1592	 * be done to request more memory from the OS.  Additionally, not having
1593	 * to execute any instructions after this call allows the compiler to use
1594	 * the sibling call optimization.  If you're considering adding code after
1595	 * this call, consider making it the responsibility of the 'realloc'
1596	 * function instead.
1597	 */
1598	ret = MCXT_METHOD(pointer, realloc) (pointer, size, 0);
1599
1600#ifdef USE_VALGRIND
1601	if (method != MCTX_ALIGNED_REDIRECT_ID)
1602		VALGRIND_MEMPOOL_CHANGE(context, pointer, ret, size);
1603#endif
1604
1605	return ret;
1606}
