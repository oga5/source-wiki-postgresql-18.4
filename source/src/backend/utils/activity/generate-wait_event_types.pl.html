src/backend/utils/activity/generate-wait_event_types.pl
1#!/usr/bin/perl
2#----------------------------------------------------------------------
3#
4# Generate wait events support files from wait_event_names.txt:
5# - wait_event_types.h (if --code is passed)
6# - pgstat_wait_event.c (if --code is passed)
7# - wait_event_funcs_data.c (if --code is passed)
8# - wait_event_types.sgml (if --docs is passed)
9#
10# Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
11# Portions Copyright (c) 1994, Regents of the University of California
12#
13# src/backend/utils/activity/generate-wait_event_types.pl
14#
15#----------------------------------------------------------------------
16
17use strict;
18use warnings FATAL => 'all';
19use Getopt::Long;
20
21my $output_path = '.';
22my $gen_docs = 0;
23my $gen_code = 0;
24
25my $continue = "\n";
26my %hashwe;
27
28GetOptions(
29	'outdir:s' => \$output_path,
30	'docs' => \$gen_docs,
31	'code' => \$gen_code) || usage();
32
33die "Needs to specify --docs or --code"
34  if (!$gen_docs && !$gen_code);
35
36die "Not possible to specify --docs and --code simultaneously"
37  if ($gen_docs && $gen_code);
38
39open my $wait_event_names, '<', $ARGV[0] or die;
40
41my @abi_compatibility_lines;
42my @lines;
43my $abi_compatibility = 0;
44my $section_name;
45
46# Remove comments and empty lines and add waitclassname based on the section
47while (<$wait_event_names>)
48{
49	chomp;
50
51	# Skip comments
52	next if /^#/;
53
54	# Skip empty lines
55	next if /^\s*$/;
56
57	# Get waitclassname based on the section
58	if (/^Section: ClassName(.*)/)
59	{
60		$section_name = $_;
61		$section_name =~ s/^.*- //;
62		$abi_compatibility = 0;
63		next;
64	}
65
66	# ABI_compatibility region, preserving ABI compatibility of the code
67	# generated.  Any wait events listed in this part of the file will
68	# not be sorted during the code generation.
69	if (/^ABI_compatibility:$/)
70	{
71		$abi_compatibility = 1;
72		next;
73	}
74
75	if ($gen_code && $abi_compatibility)
76	{
77		push(@abi_compatibility_lines, $section_name . "\t" . $_);
78	}
79	else
80	{
81		push(@lines, $section_name . "\t" . $_);
82	}
83}
84
85# Sort the lines based on the second column.
86# uc() is being used to force the comparison to be case-insensitive.
87my @lines_sorted =
88  sort { uc((split(/\t/, $a))[1]) cmp uc((split(/\t/, $b))[1]) } @lines;
89
90# If we are generating code, concat @lines_sorted and then
91# @abi_compatibility_lines.
92if ($gen_code)
93{
94	push(@lines_sorted, @abi_compatibility_lines);
95}
96
97# Read the sorted lines and populate the hash table
98foreach my $line (@lines_sorted)
99{
100	die "unable to parse wait_event_names.txt for line $line\n"
101	  unless $line =~ /^(\w+)\t+(\w+)\t+("\w.*\.")$/;
102
103	(my $waitclassname, my $waiteventname, my $waitevendocsentence) =
104	  split(/\t/, $line);
105
106	# Generate the element name for the enums based on the
107	# description.  The C symbols are prefixed with "WAIT_EVENT_".
108	my $waiteventenumname = "WAIT_EVENT_$waiteventname";
109
110	# Build the descriptions.  These are in camel-case.
111	# LWLock and Lock classes do not need any modifications.
112	my $waiteventdescription = '';
113	if (   $waitclassname eq 'WaitEventLWLock'
114		|| $waitclassname eq 'WaitEventLock')
115	{
116		$waiteventdescription = $waiteventname;
117	}
118	else
119	{
120		my @waiteventparts = split("_", $waiteventname);
121		foreach my $waiteventpart (@waiteventparts)
122		{
123			$waiteventdescription .= substr($waiteventpart, 0, 1)
124			  . lc(substr($waiteventpart, 1, length($waiteventpart)));
125		}
126	}
127
128	# Store the event into the list for each class.
129	my @waiteventlist =
130	  [ $waiteventenumname, $waiteventdescription, $waitevendocsentence ];
131	push(@{ $hashwe{$waitclassname} }, @waiteventlist);
132}
133
134
135# Generate the .c and .h files.
136if ($gen_code)
137{
138	# Include PID in suffix in case parallel make runs this script
139	# multiple times.
140	my $htmp = "$output_path/wait_event_types.h.tmp$$";
141	my $ctmp = "$output_path/pgstat_wait_event.c.tmp$$";
142	my $wctmp = "$output_path/wait_event_funcs_data.c.tmp$$";
143	open my $h, '>', $htmp or die "Could not open $htmp: $!";
144	open my $c, '>', $ctmp or die "Could not open $ctmp: $!";
145	open my $wc, '>', $wctmp or die "Could not open $wctmp: $!";
146
147	my $header_comment =
148	  '/*-------------------------------------------------------------------------
149 *
150 * %s
151 *    Generated wait events infrastructure code
152 *
153 * Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
154 * Portions Copyright (c) 1994, Regents of the University of California
155 *
156 * NOTES
157 *  ******************************
158 *  *** DO NOT EDIT THIS FILE! ***
159 *  ******************************
160 *
161 *  It has been GENERATED by src/backend/utils/activity/generate-wait_event_types.pl
162 *
163 *-------------------------------------------------------------------------
164 */
165
166';
167
168	printf $h $header_comment, 'wait_event_types.h';
169	printf $h "#ifndef WAIT_EVENT_TYPES_H\n";
170	printf $h "#define WAIT_EVENT_TYPES_H\n\n";
171	printf $h "#include \"utils/wait_classes.h\"\n\n";
172
173	printf $c $header_comment, 'pgstat_wait_event.c';
174
175	printf $wc $header_comment, 'wait_event_funcs_data.c';
176
177	# Generate the pgstat_wait_event.c and wait_event_types.h files
178	# uc() is being used to force the comparison to be case-insensitive.
179	foreach my $waitclass (sort { uc($a) cmp uc($b) } keys %hashwe)
180	{
181		# Don't generate the pgstat_wait_event.c and wait_event_types.h files
182		# for types handled independently.
183		next
184		  if ( $waitclass eq 'WaitEventExtension'
185			|| $waitclass eq 'WaitEventInjectionPoint'
186			|| $waitclass eq 'WaitEventLWLock'
187			|| $waitclass eq 'WaitEventLock');
188
189		my $last = $waitclass;
190		$last =~ s/^WaitEvent//;
191		my $lastuc = uc $last;
192		my $lastlc = lc $last;
193		my $firstpass = 1;
194		my $pg_wait_class;
195
196		printf $c
197		  "static const char *\npgstat_get_wait_$lastlc($waitclass w)\n{\n";
198		printf $c "\tconst char *event_name = \"unknown wait event\";\n\n";
199		printf $c "\tswitch (w)\n\t{\n";
200
201		foreach my $wev (@{ $hashwe{$waitclass} })
202		{
203			if ($firstpass)
204			{
205				printf $h "typedef enum\n{\n";
206				$pg_wait_class = "PG_WAIT_" . $lastuc;
207				printf $h "\t%s = %s", $wev->[0], $pg_wait_class;
208				$continue = ",\n";
209			}
210			else
211			{
212				printf $h "%s\t%s", $continue, $wev->[0];
213				$continue = ",\n";
214			}
215			$firstpass = 0;
216
217			printf $c "\t\t case %s:\n", $wev->[0];
218			# Apply quotes to the wait event name string.
219			printf $c "\t\t\t event_name = \"%s\";\n\t\t\t break;\n",
220			  $wev->[1];
221		}
222
223		printf $h "\n} $waitclass;\n\n";
224
225		printf $c
226		  "\t\t\t /* no default case, so that compiler will warn */\n";
227		printf $c "\t}\n\n";
228		printf $c "\treturn event_name;\n";
229		printf $c "}\n\n";
230	}
231
232	# Generate wait_event_funcs_data.c, building the contents of a static
233	# C structure holding all the information about the wait events.
234	# uc() is being used to force the comparison to be case-insensitive,
235	# even though it is not required here.
236	foreach my $waitclass (sort { uc($a) cmp uc($b) } keys %hashwe)
237	{
238		my $last = $waitclass;
239		$last =~ s/^WaitEvent//;
240
241		foreach my $wev (@{ $hashwe{$waitclass} })
242		{
243			my $new_desc = substr $wev->[2], 1, -2;
244			# Escape single quotes.
245			$new_desc =~ s/'/\\'/g;
246
247			# Replace the "quote" markups by real ones.
248			$new_desc =~ s/<quote>(.*?)<\/quote>/\\"$1\\"/g;
249
250			# Remove SGML markups.
251			$new_desc =~ s/<.*?>(.*?)<.*?>/$1/g;
252
253			# Tweak contents about links <xref linkend="text"/>
254			# on GUCs,
255			while (my ($capture) =
256				$new_desc =~ m/<xref linkend="guc-(.*?)"\/>/g)
257			{
258				$capture =~ s/-/_/g;
259				$new_desc =~ s/<xref linkend="guc-.*?"\/>/$capture/g;
260			}
261			# Then remove any reference to
262			# "see <xref linkend="text"/>".
263			$new_desc =~ s/; see.*$//;
264
265			# Build one element of the C structure holding the
266			# wait event info, as of (type, name, description).
267			printf $wc "\t{\"%s\", \"%s\", \"%s\"},\n", $last, $wev->[1],
268			  $new_desc;
269		}
270	}
271
272	printf $h "#endif                          /* WAIT_EVENT_TYPES_H */\n";
273	close $h;
274	close $c;
275	close $wc;
276
277	rename($htmp, "$output_path/wait_event_types.h")
278	  || die "rename: $htmp to $output_path/wait_event_types.h: $!";
279	rename($ctmp, "$output_path/pgstat_wait_event.c")
280	  || die "rename: $ctmp to $output_path/pgstat_wait_event.c: $!";
281	rename($wctmp, "$output_path/wait_event_funcs_data.c")
282	  || die "rename: $wctmp to $output_path/wait_event_funcs_data.c: $!";
283}
284# Generate the .sgml file.
285elsif ($gen_docs)
286{
287	# Include PID in suffix in case parallel make runs this multiple times.
288	my $stmp = "$output_path/wait_event_names.s.tmp$$";
289	open my $s, '>', $stmp or die "Could not open $stmp: $!";
290
291	# uc() is being used to force the comparison to be case-insensitive.
292	foreach my $waitclass (sort { uc($a) cmp uc($b) } keys %hashwe)
293	{
294		my $last = $waitclass;
295		$last =~ s/^WaitEvent//;
296		my $lastlc = lc $last;
297
298		printf $s "  <table id=\"wait-event-%s-table\">\n", $lastlc;
299		printf $s
300		  "   <title>Wait Events of Type <literal>%s</literal></title>\n",
301		  ucfirst($lastlc);
302		printf $s "   <tgroup cols=\"2\">\n";
303		printf $s "    <thead>\n";
304		printf $s "     <row>\n";
305		printf $s
306		  "      <entry><literal>$last</literal> Wait Event</entry>\n";
307		printf $s "      <entry>Description</entry>\n";
308		printf $s "     </row>\n";
309		printf $s "    </thead>\n\n";
310		printf $s "    <tbody>\n";
311
312		foreach my $wev (@{ $hashwe{$waitclass} })
313		{
314			printf $s "     <row>\n";
315			printf $s "      <entry><literal>%s</literal></entry>\n",
316			  $wev->[1];
317			printf $s "      <entry>%s</entry>\n", substr $wev->[2], 1, -1;
318			printf $s "     </row>\n";
319		}
320
321		printf $s "    </tbody>\n";
322		printf $s "   </tgroup>\n";
323		printf $s "  </table>\n\n";
324	}
325
326	close $s;
327
328	rename($stmp, "$output_path/wait_event_types.sgml")
329	  || die "rename: $stmp to $output_path/wait_event_types.sgml: $!";
330}
331
332close $wait_event_names;
333
334sub usage
335{
336	die <<EOM;
337Usage: perl  [--output <path>] [--code ] [ --docs ] input_file
338
339Options:
340    --outdir         Output directory (default '.')
341    --code           Generate C and header files.
342    --docs           Generate wait_event_types.sgml.
343
344generate-wait_event_types.pl generates the SGML documentation and code
345related to wait events.  This should use wait_event_names.txt in input, or
346an input file with a compatible format.
347
348Report bugs to <pgsql-bugs\@lists.postgresql.org>.
349EOM
350}
