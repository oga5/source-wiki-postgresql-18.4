72/*
73 * Backend local references to shared stats entries. If there are pending
74 * updates to a stats entry, the PgStat_EntryRef is added to the pgStatPending
75 * list.
76 *
77 * When a stats entry is dropped each backend needs to release its reference
78 * to it before the memory can be released. To trigger that
79 * pgStatLocal.shmem->gc_request_count is incremented - which each backend
80 * compares to their copy of pgStatSharedRefAge on a regular basis.
81 */
82static pgstat_entry_ref_hash_hash *pgStatEntryRefHash = NULL;
83static int	pgStatSharedRefAge = 0; /* cache age of pgStatLocal.shmem */
84
85/*
86 * Memory contexts containing the pgStatEntryRefHash table and the
87 * pgStatSharedRef entries respectively. Kept separate to make it easier to
88 * track / attribute memory usage.
89 */
90static MemoryContext pgStatSharedRefContext = NULL;
91static MemoryContext pgStatEntryRefHashContext = NULL;

99/*
100 * The size of the shared memory allocation for stats stored in the shared
101 * stats hash table. This allocation will be done as part of the main shared
102 * memory, rather than dynamic shared memory, allowing it to be initialized in
103 * postmaster.
104 */
105static Size
106pgstat_dsa_init_size(void)
107{
108	Size		sz;
109
110	/*
111	 * The dshash header / initial buckets array needs to fit into "plain"
112	 * shared memory, but it's beneficial to not need dsm segments
113	 * immediately. A size of 256kB seems works well and is not
114	 * disproportional compared to other constant sized shared memory
115	 * allocations. NB: To avoid DSMs further, the user can configure
116	 * min_dynamic_shared_memory.
117	 */
118	sz = 256 * 1024;
119	Assert(dsa_minimum_size() <= sz);
120	return MAXALIGN(sz);
121}

123/*
124 * Compute shared memory space needed for cumulative statistics
125 */
126Size
127StatsShmemSize(void)
128{
129	Size		sz;
130
131	sz = MAXALIGN(sizeof(PgStat_ShmemControl));
132	sz = add_size(sz, pgstat_dsa_init_size());
133
134	/* Add shared memory for all the custom fixed-numbered statistics */
135	for (PgStat_Kind kind = PGSTAT_KIND_CUSTOM_MIN; kind <= PGSTAT_KIND_CUSTOM_MAX; kind++)
136	{
137		const PgStat_KindInfo *kind_info = pgstat_get_kind_info(kind);
138
139		if (!kind_info)
140			continue;
141		if (!kind_info->fixed_amount)
142			continue;
143
144		Assert(kind_info->shared_size != 0);
145		sz = add_size(sz, MAXALIGN(kind_info->shared_size));
146	}
147
148	return sz;
149}

151/*
152 * Initialize cumulative statistics system during startup
153 */
154void
155StatsShmemInit(void)
156{
157	bool		found;
158	Size		sz;
159
160	sz = StatsShmemSize();
161	pgStatLocal.shmem = (PgStat_ShmemControl *)
162		ShmemInitStruct("Shared Memory Stats", sz, &found);
163
164	if (!IsUnderPostmaster)
165	{
166		dsa_area   *dsa;
167		dshash_table *dsh;
168		PgStat_ShmemControl *ctl = pgStatLocal.shmem;
169		char	   *p = (char *) ctl;
170
171		Assert(!found);
172
173		/* the allocation of pgStatLocal.shmem itself */
174		p += MAXALIGN(sizeof(PgStat_ShmemControl));
175
176		/*
177		 * Create a small dsa allocation in plain shared memory. This is
178		 * required because postmaster cannot use dsm segments. It also
179		 * provides a small efficiency win.
180		 */
181		ctl->raw_dsa_area = p;
182		p += MAXALIGN(pgstat_dsa_init_size());
183		dsa = dsa_create_in_place(ctl->raw_dsa_area,
184								  pgstat_dsa_init_size(),
185								  LWTRANCHE_PGSTATS_DSA, NULL);
186		dsa_pin(dsa);
187
188		/*
189		 * To ensure dshash is created in "plain" shared memory, temporarily
190		 * limit size of dsa to the initial size of the dsa.
191		 */
192		dsa_set_size_limit(dsa, pgstat_dsa_init_size());
193
194		/*
195		 * With the limit in place, create the dshash table. XXX: It'd be nice
196		 * if there were dshash_create_in_place().
197		 */
198		dsh = dshash_create(dsa, &dsh_params, NULL);
199		ctl->hash_handle = dshash_get_hash_table_handle(dsh);
200
201		/* lift limit set above */
202		dsa_set_size_limit(dsa, -1);
203
204		/*
205		 * Postmaster will never access these again, thus free the local
206		 * dsa/dshash references.
207		 */
208		dshash_detach(dsh);
209		dsa_detach(dsa);
210
211		pg_atomic_init_u64(&ctl->gc_request_count, 1);
212
213		/* initialize fixed-numbered stats */
214		for (PgStat_Kind kind = PGSTAT_KIND_MIN; kind <= PGSTAT_KIND_MAX; kind++)
215		{
216			const PgStat_KindInfo *kind_info = pgstat_get_kind_info(kind);
217			char	   *ptr;
218
219			if (!kind_info || !kind_info->fixed_amount)
220				continue;
221
222			if (pgstat_is_kind_builtin(kind))
223				ptr = ((char *) ctl) + kind_info->shared_ctl_off;
224			else
225			{
226				int			idx = kind - PGSTAT_KIND_CUSTOM_MIN;
227
228				Assert(kind_info->shared_size != 0);
229				ctl->custom_data[idx] = p;
230				p += MAXALIGN(kind_info->shared_size);
231				ptr = ctl->custom_data[idx];
232			}
233
234			kind_info->init_shmem_cb(ptr);
235		}
236	}
237	else
238	{
239		Assert(found);
240	}
241}

243void
244pgstat_attach_shmem(void)
245{
246	MemoryContext oldcontext;
247
248	Assert(pgStatLocal.dsa == NULL);
249
250	/* stats shared memory persists for the backend lifetime */
251	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
252
253	pgStatLocal.dsa = dsa_attach_in_place(pgStatLocal.shmem->raw_dsa_area,
254										  NULL);
255	dsa_pin_mapping(pgStatLocal.dsa);
256
257	pgStatLocal.shared_hash = dshash_attach(pgStatLocal.dsa, &dsh_params,
258											pgStatLocal.shmem->hash_handle,
259											NULL);
260
261	MemoryContextSwitchTo(oldcontext);
262}
263
264void
265pgstat_detach_shmem(void)
266{
267	Assert(pgStatLocal.dsa);
268
269	/* we shouldn't leave references to shared stats */
270	pgstat_release_all_entry_refs(false);
271
272	dshash_detach(pgStatLocal.shared_hash);
273	pgStatLocal.shared_hash = NULL;
274
275	dsa_detach(pgStatLocal.dsa);
276
277	/*
278	 * dsa_detach() does not decrement the DSA reference count as no segment
279	 * was provided to dsa_attach_in_place(), causing no cleanup callbacks to
280	 * be registered.  Hence, release it manually now.
281	 */
282	dsa_release_in_place(pgStatLocal.shmem->raw_dsa_area);
283
284	pgStatLocal.dsa = NULL;
285}

293/*
294 * Initialize entry newly-created.
295 *
296 * Returns NULL in the event of an allocation failure, so as callers can
297 * take cleanup actions as the entry initialized is already inserted in the
298 * shared hashtable.
299 */
300PgStatShared_Common *
301pgstat_init_entry(PgStat_Kind kind,
302				  PgStatShared_HashEntry *shhashent)
303{
304	/* Create new stats entry. */
305	dsa_pointer chunk;
306	PgStatShared_Common *shheader;
307
308	/*
309	 * Initialize refcount to 1, marking it as valid / not dropped. The entry
310	 * can't be freed before the initialization because it can't be found as
311	 * long as we hold the dshash partition lock. Caller needs to increase
312	 * further if a longer lived reference is needed.
313	 */
314	pg_atomic_init_u32(&shhashent->refcount, 1);
315
316	/*
317	 * Initialize "generation" to 0, as freshly created.
318	 */
319	pg_atomic_init_u32(&shhashent->generation, 0);
320	shhashent->dropped = false;
321
322	chunk = dsa_allocate_extended(pgStatLocal.dsa,
323								  pgstat_get_kind_info(kind)->shared_size,
324								  DSA_ALLOC_ZERO | DSA_ALLOC_NO_OOM);
325	if (chunk == InvalidDsaPointer)
326		return NULL;
327
328	shheader = dsa_get_address(pgStatLocal.dsa, chunk);
329	shheader->magic = 0xdeadbeef;
330
331	/* Link the new entry from the hash entry. */
332	shhashent->body = chunk;
333
334	LWLockInitialize(&shheader->lock, LWTRANCHE_PGSTATS_DATA);
335
336	return shheader;
337}
338
339static PgStatShared_Common *
340pgstat_reinit_entry(PgStat_Kind kind, PgStatShared_HashEntry *shhashent)
341{
342	PgStatShared_Common *shheader;
343
344	shheader = dsa_get_address(pgStatLocal.dsa, shhashent->body);
345
346	/* mark as not dropped anymore */
347	pg_atomic_fetch_add_u32(&shhashent->refcount, 1);
348
349	/*
350	 * Increment "generation", to let any backend with local references know
351	 * that what they point to is outdated.
352	 */
353	pg_atomic_fetch_add_u32(&shhashent->generation, 1);
354	shhashent->dropped = false;
355
356	/* reinitialize content */
357	Assert(shheader->magic == 0xdeadbeef);
358	memset(pgstat_get_entry_data(kind, shheader), 0,
359		   pgstat_get_entry_len(kind));
360
361	return shheader;
362}

364static void
365pgstat_setup_shared_refs(void)
366{
367	if (likely(pgStatEntryRefHash != NULL))
368		return;
369
370	pgStatEntryRefHash =
371		pgstat_entry_ref_hash_create(pgStatEntryRefHashContext,
372									 PGSTAT_ENTRY_REF_HASH_SIZE, NULL);
373	pgStatSharedRefAge = pg_atomic_read_u64(&pgStatLocal.shmem->gc_request_count);
374	Assert(pgStatSharedRefAge != 0);
375}

377/*
378 * Helper function for pgstat_get_entry_ref().
379 */
380static void
381pgstat_acquire_entry_ref(PgStat_EntryRef *entry_ref,
382						 PgStatShared_HashEntry *shhashent,
383						 PgStatShared_Common *shheader)
384{
385	Assert(shheader->magic == 0xdeadbeef);
386	Assert(pg_atomic_read_u32(&shhashent->refcount) > 0);
387
388	pg_atomic_fetch_add_u32(&shhashent->refcount, 1);
389
390	dshash_release_lock(pgStatLocal.shared_hash, shhashent);
391
392	entry_ref->shared_stats = shheader;
393	entry_ref->shared_entry = shhashent;
394	entry_ref->generation = pg_atomic_read_u32(&shhashent->generation);
395}

397/*
398 * Helper function for pgstat_get_entry_ref().
399 */
400static bool
401pgstat_get_entry_ref_cached(PgStat_HashKey key, PgStat_EntryRef **entry_ref_p)
402{
403	bool		found;
404	PgStat_EntryRefHashEntry *cache_entry;
405
406	/*
407	 * We immediately insert a cache entry, because it avoids 1) multiple
408	 * hashtable lookups in case of a cache miss 2) having to deal with
409	 * out-of-memory errors after incrementing PgStatShared_Common->refcount.
410	 */
411
412	cache_entry = pgstat_entry_ref_hash_insert(pgStatEntryRefHash, key, &found);
413
414	if (!found || !cache_entry->entry_ref)
415	{
416		PgStat_EntryRef *entry_ref;
417
418		cache_entry->entry_ref = entry_ref =
419			MemoryContextAlloc(pgStatSharedRefContext,
420							   sizeof(PgStat_EntryRef));
421		entry_ref->shared_stats = NULL;
422		entry_ref->shared_entry = NULL;
423		entry_ref->pending = NULL;
424
425		found = false;
426	}
427	else if (cache_entry->entry_ref->shared_stats == NULL)
428	{
429		Assert(cache_entry->entry_ref->pending == NULL);
430		found = false;
431	}
432	else
433	{
434		PgStat_EntryRef *entry_ref PG_USED_FOR_ASSERTS_ONLY;
435
436		entry_ref = cache_entry->entry_ref;
437		Assert(entry_ref->shared_entry != NULL);
438		Assert(entry_ref->shared_stats != NULL);
439
440		Assert(entry_ref->shared_stats->magic == 0xdeadbeef);
441		/* should have at least our reference */
442		Assert(pg_atomic_read_u32(&entry_ref->shared_entry->refcount) > 0);
443	}
444
445	*entry_ref_p = cache_entry->entry_ref;
446	return found;
447}

449/*
450 * Get a shared stats reference. If create is true, the shared stats object is
451 * created if it does not exist.
452 *
453 * When create is true, and created_entry is non-NULL, it'll be set to true
454 * if the entry is newly created, false otherwise.
455 */
456PgStat_EntryRef *
457pgstat_get_entry_ref(PgStat_Kind kind, Oid dboid, uint64 objid, bool create,
458					 bool *created_entry)
459{
460	PgStat_HashKey key;
461	PgStatShared_HashEntry *shhashent;
462	PgStatShared_Common *shheader = NULL;
463	PgStat_EntryRef *entry_ref;
464
465	/* clear padding */
466	memset(&key, 0, sizeof(struct PgStat_HashKey));
467
468	key.kind = kind;
469	key.dboid = dboid;
470	key.objid = objid;
471
472	/*
473	 * passing in created_entry only makes sense if we possibly could create
474	 * entry.
475	 */
476	Assert(create || created_entry == NULL);
477	pgstat_assert_is_up();
478	Assert(pgStatLocal.shared_hash != NULL);
479	Assert(!pgStatLocal.shmem->is_shutdown);
480
481	pgstat_setup_memcxt();
482	pgstat_setup_shared_refs();
483
484	if (created_entry != NULL)
485		*created_entry = false;
486
487	/*
488	 * Check if other backends dropped stats that could not be deleted because
489	 * somebody held references to it. If so, check this backend's references.
490	 * This is not expected to happen often. The location of the check is a
491	 * bit random, but this is a relatively frequently called path, so better
492	 * than most.
493	 */
494	if (pgstat_need_entry_refs_gc())
495		pgstat_gc_entry_refs();
496
497	/*
498	 * First check the lookup cache hashtable in local memory. If we find a
499	 * match here we can avoid taking locks / causing contention.
500	 */
501	if (pgstat_get_entry_ref_cached(key, &entry_ref))
502		return entry_ref;
503
504	Assert(entry_ref != NULL);
505
506	/*
507	 * Do a lookup in the hash table first - it's quite likely that the entry
508	 * already exists, and that way we only need a shared lock.
509	 */
510	shhashent = dshash_find(pgStatLocal.shared_hash, &key, false);
511
512	if (create && !shhashent)
513	{
514		bool		shfound;
515
516		/*
517		 * It's possible that somebody created the entry since the above
518		 * lookup. If so, fall through to the same path as if we'd have if it
519		 * already had been created before the dshash_find() calls.
520		 */
521		shhashent = dshash_find_or_insert(pgStatLocal.shared_hash, &key, &shfound);
522		if (!shfound)
523		{
524			shheader = pgstat_init_entry(kind, shhashent);
525			if (shheader == NULL)
526			{
527				/*
528				 * Failed the allocation of a new entry, so clean up the
529				 * shared hashtable before giving up.
530				 */
531				dshash_delete_entry(pgStatLocal.shared_hash, shhashent);
532
533				ereport(ERROR,
534						(errcode(ERRCODE_OUT_OF_MEMORY),
535						 errmsg("out of memory"),
536						 errdetail("Failed while allocating entry %u/%u/%" PRIu64 ".",
537								   key.kind, key.dboid, key.objid)));
538			}
539			pgstat_acquire_entry_ref(entry_ref, shhashent, shheader);
540
541			if (created_entry != NULL)
542				*created_entry = true;
543
544			return entry_ref;
545		}
546	}
547
548	if (!shhashent)
549	{
550		/*
551		 * If we're not creating, delete the reference again. In all
552		 * likelihood it's just a stats lookup - no point wasting memory for a
553		 * shared ref to nothing...
554		 */
555		pgstat_release_entry_ref(key, entry_ref, false);
556
557		return NULL;
558	}
559	else
560	{
561		/*
562		 * Can get here either because dshash_find() found a match, or if
563		 * dshash_find_or_insert() found a concurrently inserted entry.
564		 */
565
566		if (shhashent->dropped && create)
567		{
568			/*
569			 * There are legitimate cases where the old stats entry might not
570			 * yet have been dropped by the time it's reused. The most obvious
571			 * case are replication slot stats, where a new slot can be
572			 * created with the same index just after dropping. But oid
573			 * wraparound can lead to other cases as well. We just reset the
574			 * stats to their plain state, while incrementing its "generation"
575			 * in the shared entry for any remaining local references.
576			 */
577			shheader = pgstat_reinit_entry(kind, shhashent);
578			pgstat_acquire_entry_ref(entry_ref, shhashent, shheader);
579
580			if (created_entry != NULL)
581				*created_entry = true;
582
583			return entry_ref;
584		}
585		else if (shhashent->dropped)
586		{
587			dshash_release_lock(pgStatLocal.shared_hash, shhashent);
588			pgstat_release_entry_ref(key, entry_ref, false);
589
590			return NULL;
591		}
592		else
593		{
594			shheader = dsa_get_address(pgStatLocal.dsa, shhashent->body);
595			pgstat_acquire_entry_ref(entry_ref, shhashent, shheader);
596
597			return entry_ref;
598		}
599	}
600}

602static void
603pgstat_release_entry_ref(PgStat_HashKey key, PgStat_EntryRef *entry_ref,
604						 bool discard_pending)
605{
606	if (entry_ref && entry_ref->pending)
607	{
608		if (discard_pending)
609			pgstat_delete_pending_entry(entry_ref);
610		else
611			elog(ERROR, "releasing ref with pending data");
612	}
613
614	if (entry_ref && entry_ref->shared_stats)
615	{
616		Assert(entry_ref->shared_stats->magic == 0xdeadbeef);
617		Assert(entry_ref->pending == NULL);
618
619		/*
620		 * This can't race with another backend looking up the stats entry and
621		 * increasing the refcount because it is not "legal" to create
622		 * additional references to dropped entries.
623		 */
624		if (pg_atomic_fetch_sub_u32(&entry_ref->shared_entry->refcount, 1) == 1)
625		{
626			PgStatShared_HashEntry *shent;
627
628			/*
629			 * We're the last referrer to this entry, try to drop the shared
630			 * entry.
631			 */
632
633			/* only dropped entries can reach a 0 refcount */
634			Assert(entry_ref->shared_entry->dropped);
635
636			shent = dshash_find(pgStatLocal.shared_hash,
637								&entry_ref->shared_entry->key,
638								true);
639			if (!shent)
640				elog(ERROR, "could not find just referenced shared stats entry");
641
642			/*
643			 * This entry may have been reinitialized while trying to release
644			 * it, so double-check that it has not been reused while holding a
645			 * lock on its shared entry.
646			 */
647			if (pg_atomic_read_u32(&entry_ref->shared_entry->generation) ==
648				entry_ref->generation)
649			{
650				/* Same "generation", so we're OK with the removal */
651				Assert(pg_atomic_read_u32(&entry_ref->shared_entry->refcount) == 0);
652				Assert(entry_ref->shared_entry == shent);
653				pgstat_free_entry(shent, NULL);
654			}
655			else
656			{
657				/*
658				 * Shared stats entry has been reinitialized, so do not drop
659				 * its shared entry, only release its lock.
660				 */
661				dshash_release_lock(pgStatLocal.shared_hash, shent);
662			}
663		}
664	}
665
666	if (!pgstat_entry_ref_hash_delete(pgStatEntryRefHash, key))
667		elog(ERROR, "entry ref vanished before deletion");
668
669	if (entry_ref)
670		pfree(entry_ref);
671}

673/*
674 * Acquire exclusive lock on the entry.
675 *
676 * If nowait is true, it's just a conditional acquire, and the result
677 * *must* be checked to verify success.
678 * If nowait is false, waits as necessary, always returning true.
679 */
680bool
681pgstat_lock_entry(PgStat_EntryRef *entry_ref, bool nowait)
682{
683	LWLock	   *lock = &entry_ref->shared_stats->lock;
684
685	if (nowait)
686		return LWLockConditionalAcquire(lock, LW_EXCLUSIVE);
687
688	LWLockAcquire(lock, LW_EXCLUSIVE);
689	return true;
690}

692/*
693 * Acquire shared lock on the entry.
694 *
695 * Separate from pgstat_lock_entry() as most callers will need to lock
696 * exclusively.  The wait semantics are identical.
697 */
698bool
699pgstat_lock_entry_shared(PgStat_EntryRef *entry_ref, bool nowait)
700{
701	LWLock	   *lock = &entry_ref->shared_stats->lock;
702
703	if (nowait)
704		return LWLockConditionalAcquire(lock, LW_SHARED);
705
706	LWLockAcquire(lock, LW_SHARED);
707	return true;
708}

710void
711pgstat_unlock_entry(PgStat_EntryRef *entry_ref)
712{
713	LWLockRelease(&entry_ref->shared_stats->lock);
714}

716/*
717 * Helper function to fetch and lock shared stats.
718 */
719PgStat_EntryRef *
720pgstat_get_entry_ref_locked(PgStat_Kind kind, Oid dboid, uint64 objid,
721							bool nowait)
722{
723	PgStat_EntryRef *entry_ref;
724
725	/* find shared table stats entry corresponding to the local entry */
726	entry_ref = pgstat_get_entry_ref(kind, dboid, objid, true, NULL);
727
728	/* lock the shared entry to protect the content, skip if failed */
729	if (!pgstat_lock_entry(entry_ref, nowait))
730		return NULL;
731
732	return entry_ref;
733}

735void
736pgstat_request_entry_refs_gc(void)
737{
738	pg_atomic_fetch_add_u64(&pgStatLocal.shmem->gc_request_count, 1);
739}

741static bool
742pgstat_need_entry_refs_gc(void)
743{
744	uint64		curage;
745
746	if (!pgStatEntryRefHash)
747		return false;
748
749	/* should have been initialized when creating pgStatEntryRefHash */
750	Assert(pgStatSharedRefAge != 0);
751
752	curage = pg_atomic_read_u64(&pgStatLocal.shmem->gc_request_count);
753
754	return pgStatSharedRefAge != curage;
755}
756
757static void
758pgstat_gc_entry_refs(void)
759{
760	pgstat_entry_ref_hash_iterator i;
761	PgStat_EntryRefHashEntry *ent;
762	uint64		curage;
763
764	curage = pg_atomic_read_u64(&pgStatLocal.shmem->gc_request_count);
765	Assert(curage != 0);
766
767	/*
768	 * Some entries have been dropped or reinitialized.  Invalidate cache
769	 * pointer to them.
770	 */
771	pgstat_entry_ref_hash_start_iterate(pgStatEntryRefHash, &i);
772	while ((ent = pgstat_entry_ref_hash_iterate(pgStatEntryRefHash, &i)) != NULL)
773	{
774		PgStat_EntryRef *entry_ref = ent->entry_ref;
775
776		Assert(!entry_ref->shared_stats ||
777			   entry_ref->shared_stats->magic == 0xdeadbeef);
778
779		/*
780		 * "generation" checks for the case of entries being reinitialized,
781		 * and "dropped" for the case where these are..  dropped.
782		 */
783		if (!entry_ref->shared_entry->dropped &&
784			pg_atomic_read_u32(&entry_ref->shared_entry->generation) ==
785			entry_ref->generation)
786			continue;
787
788		/* cannot gc shared ref that has pending data */
789		if (entry_ref->pending != NULL)
790			continue;
791
792		pgstat_release_entry_ref(ent->key, entry_ref, false);
793	}
794
795	pgStatSharedRefAge = curage;
796}
797
798static void
799pgstat_release_matching_entry_refs(bool discard_pending, ReleaseMatchCB match,
800								   Datum match_data)
801{
802	pgstat_entry_ref_hash_iterator i;
803	PgStat_EntryRefHashEntry *ent;
804
805	if (pgStatEntryRefHash == NULL)
806		return;
807
808	pgstat_entry_ref_hash_start_iterate(pgStatEntryRefHash, &i);
809
810	while ((ent = pgstat_entry_ref_hash_iterate(pgStatEntryRefHash, &i))
811		   != NULL)
812	{
813		Assert(ent->entry_ref != NULL);
814
815		if (match && !match(ent, match_data))
816			continue;
817
818		pgstat_release_entry_ref(ent->key, ent->entry_ref, discard_pending);
819	}
820}

822/*
823 * Release all local references to shared stats entries.
824 *
825 * When a process exits it cannot do so while still holding references onto
826 * stats entries, otherwise the shared stats entries could never be freed.
827 */
828static void
829pgstat_release_all_entry_refs(bool discard_pending)
830{
831	if (pgStatEntryRefHash == NULL)
832		return;
833
834	pgstat_release_matching_entry_refs(discard_pending, NULL, 0);
835	Assert(pgStatEntryRefHash->members == 0);
836	pgstat_entry_ref_hash_destroy(pgStatEntryRefHash);
837	pgStatEntryRefHash = NULL;
838}

857/* ------------------------------------------------------------
858 * Dropping and resetting of stats entries
859 * ------------------------------------------------------------
860 */
861
862static void
863pgstat_free_entry(PgStatShared_HashEntry *shent, dshash_seq_status *hstat)
864{
865	dsa_pointer pdsa;
866
867	/*
868	 * Fetch dsa pointer before deleting entry - that way we can free the
869	 * memory after releasing the lock.
870	 */
871	pdsa = shent->body;
872
873	if (!hstat)
874		dshash_delete_entry(pgStatLocal.shared_hash, shent);
875	else
876		dshash_delete_current(hstat);
877
878	dsa_free(pgStatLocal.dsa, pdsa);
879}
880
881/*
882 * Helper for both pgstat_drop_database_and_contents() and
883 * pgstat_drop_entry(). If hstat is non-null delete the shared entry using
884 * dshash_delete_current(), otherwise use dshash_delete_entry(). In either
885 * case the entry needs to be already locked.
886 */
887static bool
888pgstat_drop_entry_internal(PgStatShared_HashEntry *shent,
889						   dshash_seq_status *hstat)
890{
891	Assert(shent->body != InvalidDsaPointer);
892
893	/* should already have released local reference */
894	if (pgStatEntryRefHash)
895		Assert(!pgstat_entry_ref_hash_lookup(pgStatEntryRefHash, shent->key));
896
897	/*
898	 * Signal that the entry is dropped - this will eventually cause other
899	 * backends to release their references.
900	 */
901	if (shent->dropped)
902		elog(ERROR,
903			 "trying to drop stats entry already dropped: kind=%s dboid=%u objid=%" PRIu64 " refcount=%u generation=%u",
904			 pgstat_get_kind_info(shent->key.kind)->name,
905			 shent->key.dboid,
906			 shent->key.objid,
907			 pg_atomic_read_u32(&shent->refcount),
908			 pg_atomic_read_u32(&shent->generation));
909	shent->dropped = true;
910
911	/* release refcount marking entry as not dropped */
912	if (pg_atomic_sub_fetch_u32(&shent->refcount, 1) == 0)
913	{
914		pgstat_free_entry(shent, hstat);
915		return true;
916	}
917	else
918	{
919		if (!hstat)
920			dshash_release_lock(pgStatLocal.shared_hash, shent);
921		return false;
922	}
923}
924
925/*
926 * Drop stats for the database and all the objects inside that database.
927 */
928static void
929pgstat_drop_database_and_contents(Oid dboid)
930{
931	dshash_seq_status hstat;
932	PgStatShared_HashEntry *p;
933	uint64		not_freed_count = 0;
934
935	Assert(OidIsValid(dboid));
936
937	Assert(pgStatLocal.shared_hash != NULL);
938
939	/*
940	 * This backend might very well be the only backend holding a reference to
941	 * about-to-be-dropped entries. Ensure that we're not preventing it from
942	 * being cleaned up till later.
943	 *
944	 * Doing this separately from the dshash iteration below avoids having to
945	 * do so while holding a partition lock on the shared hashtable.
946	 */
947	pgstat_release_db_entry_refs(dboid);
948
949	/* some of the dshash entries are to be removed, take exclusive lock. */
950	dshash_seq_init(&hstat, pgStatLocal.shared_hash, true);
951	while ((p = dshash_seq_next(&hstat)) != NULL)
952	{
953		if (p->dropped)
954			continue;
955
956		if (p->key.dboid != dboid)
957			continue;
958
959		if (!pgstat_drop_entry_internal(p, &hstat))
960		{
961			/*
962			 * Even statistics for a dropped database might currently be
963			 * accessed (consider e.g. database stats for pg_stat_database).
964			 */
965			not_freed_count++;
966		}
967	}
968	dshash_seq_term(&hstat);
969
970	/*
971	 * If some of the stats data could not be freed, signal the reference
972	 * holders to run garbage collection of their cached pgStatLocal.shmem.
973	 */
974	if (not_freed_count > 0)
975		pgstat_request_entry_refs_gc();
976}
977
978/*
979 * Drop a single stats entry.
980 *
981 * This routine returns false if the stats entry of the dropped object could
982 * not be freed, true otherwise.
983 *
984 * The callers of this function should call pgstat_request_entry_refs_gc()
985 * if the stats entry could not be freed, to ensure that this entry's memory
986 * can be reclaimed later by a different backend calling
987 * pgstat_gc_entry_refs().
988 */
989bool
990pgstat_drop_entry(PgStat_Kind kind, Oid dboid, uint64 objid)
991{
992	PgStat_HashKey key;
993	PgStatShared_HashEntry *shent;
994	bool		freed = true;
995
996	/* clear padding */
997	memset(&key, 0, sizeof(struct PgStat_HashKey));
998
999	key.kind = kind;
1000	key.dboid = dboid;
1001	key.objid = objid;
1002
1003	/* delete local reference */
1004	if (pgStatEntryRefHash)
1005	{
1006		PgStat_EntryRefHashEntry *lohashent =
1007			pgstat_entry_ref_hash_lookup(pgStatEntryRefHash, key);
1008
1009		if (lohashent)
1010			pgstat_release_entry_ref(lohashent->key, lohashent->entry_ref,
1011									 true);
1012	}
1013
1014	/* mark entry in shared hashtable as deleted, drop if possible */
1015	shent = dshash_find(pgStatLocal.shared_hash, &key, true);
1016	if (shent)
1017	{
1018		freed = pgstat_drop_entry_internal(shent, NULL);
1019
1020		/*
1021		 * Database stats contain other stats. Drop those as well when
1022		 * dropping the database. XXX: Perhaps this should be done in a
1023		 * slightly more principled way? But not obvious what that'd look
1024		 * like, and so far this is the only case...
1025		 */
1026		if (key.kind == PGSTAT_KIND_DATABASE)
1027			pgstat_drop_database_and_contents(key.dboid);
1028	}
1029
1030	return freed;
1031}

1033/*
1034 * Scan through the shared hashtable of stats, dropping statistics if
1035 * approved by the optional do_drop() function.
1036 */
1037void
1038pgstat_drop_matching_entries(bool (*do_drop) (PgStatShared_HashEntry *, Datum),
1039							 Datum match_data)
1040{
1041	dshash_seq_status hstat;
1042	PgStatShared_HashEntry *ps;
1043	uint64		not_freed_count = 0;
1044
1045	/* entries are removed, take an exclusive lock */
1046	dshash_seq_init(&hstat, pgStatLocal.shared_hash, true);
1047	while ((ps = dshash_seq_next(&hstat)) != NULL)
1048	{
1049		if (ps->dropped)
1050			continue;
1051
1052		if (do_drop != NULL && !do_drop(ps, match_data))
1053			continue;
1054
1055		/* delete local reference */
1056		if (pgStatEntryRefHash)
1057		{
1058			PgStat_EntryRefHashEntry *lohashent =
1059				pgstat_entry_ref_hash_lookup(pgStatEntryRefHash, ps->key);
1060
1061			if (lohashent)
1062				pgstat_release_entry_ref(lohashent->key, lohashent->entry_ref,
1063										 true);
1064		}
1065
1066		if (!pgstat_drop_entry_internal(ps, &hstat))
1067			not_freed_count++;
1068	}
1069	dshash_seq_term(&hstat);
1070
1071	if (not_freed_count > 0)
1072		pgstat_request_entry_refs_gc();
1073}

1078void
1079pgstat_drop_all_entries(void)
1080{
1081	pgstat_drop_matching_entries(NULL, 0);

1084static void
1085shared_stat_reset_contents(PgStat_Kind kind, PgStatShared_Common *header,
1086						   TimestampTz ts)
1087{
1088	const PgStat_KindInfo *kind_info = pgstat_get_kind_info(kind);
1089
1090	memset(pgstat_get_entry_data(kind, header), 0,
1091		   pgstat_get_entry_len(kind));
1092
1093	if (kind_info->reset_timestamp_cb)
1094		kind_info->reset_timestamp_cb(header, ts);
1095}

1097/*
1098 * Reset one variable-numbered stats entry.
1099 */
1100void
1101pgstat_reset_entry(PgStat_Kind kind, Oid dboid, uint64 objid, TimestampTz ts)
1102{
1103	PgStat_EntryRef *entry_ref;
1104
1105	Assert(!pgstat_get_kind_info(kind)->fixed_amount);
1106
1107	entry_ref = pgstat_get_entry_ref(kind, dboid, objid, false, NULL);
1108	if (!entry_ref || entry_ref->shared_entry->dropped)
1109		return;
1110
1111	(void) pgstat_lock_entry(entry_ref, false);
1112	shared_stat_reset_contents(kind, entry_ref->shared_stats, ts);
1113	pgstat_unlock_entry(entry_ref);
1114}

1116/*
1117 * Scan through the shared hashtable of stats, resetting statistics if
1118 * approved by the provided do_reset() function.
1119 */
1120void
1121pgstat_reset_matching_entries(bool (*do_reset) (PgStatShared_HashEntry *, Datum),
1122							  Datum match_data, TimestampTz ts)
1123{
1124	dshash_seq_status hstat;
1125	PgStatShared_HashEntry *p;
1126
1127	/* dshash entry is not modified, take shared lock */
1128	dshash_seq_init(&hstat, pgStatLocal.shared_hash, false);
1129	while ((p = dshash_seq_next(&hstat)) != NULL)
1130	{
1131		PgStatShared_Common *header;
1132
1133		if (p->dropped)
1134			continue;
1135
1136		if (!do_reset(p, match_data))
1137			continue;
1138
1139		header = dsa_get_address(pgStatLocal.dsa, p->body);
1140
1141		LWLockAcquire(&header->lock, LW_EXCLUSIVE);
1142
1143		shared_stat_reset_contents(p->key.kind, header, ts);
1144
1145		LWLockRelease(&header->lock);
1146	}
1147	dshash_seq_term(&hstat);
1148}
