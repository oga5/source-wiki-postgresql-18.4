69/*
70 * Entry point for a new backend process.
71 *
72 * Initialize the connection, read the startup packet, authenticate the
73 * client, and start the main processing loop.
74 */
75void
76BackendMain(const void *startup_data, size_t startup_data_len)
77{
78	const BackendStartupData *bsdata = startup_data;
79
80	Assert(startup_data_len == sizeof(BackendStartupData));
81	Assert(MyClientSocket != NULL);
82
83#ifdef EXEC_BACKEND
84
85	/*
86	 * Need to reinitialize the SSL library in the backend, since the context
87	 * structures contain function pointers and cannot be passed through the
88	 * parameter file.
89	 *
90	 * If for some reason reload fails (maybe the user installed broken key
91	 * files), soldier on without SSL; that's better than all connections
92	 * becoming impossible.
93	 *
94	 * XXX should we do this in all child processes?  For the moment it's
95	 * enough to do it in backend children.
96	 */
97#ifdef USE_SSL
98	if (EnableSSL)
99	{
100		if (secure_initialize(false) == 0)
101			LoadedSSL = true;
102		else
103			ereport(LOG,
104					(errmsg("SSL configuration could not be loaded in child process")));
105	}
106#endif
107#endif
108
109	/* Perform additional initialization and collect startup packet */
110	BackendInitialize(MyClientSocket, bsdata->canAcceptConnections);
111
112	/*
113	 * Create a per-backend PGPROC struct in shared memory.  We must do this
114	 * before we can use LWLocks or access any shared memory.
115	 */
116	InitProcess();
117
118	/*
119	 * Make sure we aren't in PostmasterContext anymore.  (We can't delete it
120	 * just yet, though, because InitPostgres will need the HBA data.)
121	 */
122	MemoryContextSwitchTo(TopMemoryContext);
123
124	PostgresMain(MyProcPort->database_name, MyProcPort->user_name);
125}

128/*
129 * BackendInitialize -- initialize an interactive (postmaster-child)
130 *				backend process, and collect the client's startup packet.
131 *
132 * returns: nothing.  Will not return at all if there's any failure.
133 *
134 * Note: this code does not depend on having any access to shared memory.
135 * Indeed, our approach to SIGTERM/timeout handling *requires* that
136 * shared memory not have been touched yet; see comments within.
137 * In the EXEC_BACKEND case, we are physically attached to shared memory
138 * but have not yet set up most of our local pointers to shmem structures.
139 */
140static void
141BackendInitialize(ClientSocket *client_sock, CAC_state cac)
142{
143	int			status;
144	int			ret;
145	Port	   *port;
146	char		remote_host[NI_MAXHOST];
147	char		remote_port[NI_MAXSERV];
148	StringInfoData ps_data;
149	MemoryContext oldcontext;
150
151	/* Tell fd.c about the long-lived FD associated with the client_sock */
152	ReserveExternalFD();
153
154	/*
155	 * PreAuthDelay is a debugging aid for investigating problems in the
156	 * authentication cycle: it can be set in postgresql.conf to allow time to
157	 * attach to the newly-forked backend with a debugger.  (See also
158	 * PostAuthDelay, which we allow clients to pass through PGOPTIONS, but it
159	 * is not honored until after authentication.)
160	 */
161	if (PreAuthDelay > 0)
162		pg_usleep(PreAuthDelay * 1000000L);
163
164	/* This flag will remain set until InitPostgres finishes authentication */
165	ClientAuthInProgress = true;	/* limit visibility of log messages */
166
167	/*
168	 * Initialize libpq and enable reporting of ereport errors to the client.
169	 * Must do this now because authentication uses libpq to send messages.
170	 *
171	 * The Port structure and all data structures attached to it are allocated
172	 * in TopMemoryContext, so that they survive into PostgresMain execution.
173	 * We need not worry about leaking this storage on failure, since we
174	 * aren't in the postmaster process anymore.
175	 */
176	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
177	port = MyProcPort = pq_init(client_sock);
178	MemoryContextSwitchTo(oldcontext);
179
180	whereToSendOutput = DestRemote; /* now safe to ereport to client */
181
182	/* set these to empty in case they are needed before we set them up */
183	port->remote_host = "";
184	port->remote_port = "";
185
186	/*
187	 * We arrange to do _exit(1) if we receive SIGTERM or timeout while trying
188	 * to collect the startup packet; while SIGQUIT results in _exit(2).
189	 * Otherwise the postmaster cannot shutdown the database FAST or IMMED
190	 * cleanly if a buggy client fails to send the packet promptly.
191	 *
192	 * Exiting with _exit(1) is only possible because we have not yet touched
193	 * shared memory; therefore no outside-the-process state needs to get
194	 * cleaned up.
195	 */
196	pqsignal(SIGTERM, process_startup_packet_die);
197	/* SIGQUIT handler was already set up by InitPostmasterChild */
198	InitializeTimeouts();		/* establishes SIGALRM handler */
199	sigprocmask(SIG_SETMASK, &StartupBlockSig, NULL);
200
201	/*
202	 * Get the remote host name and port for logging and status display.
203	 */
204	remote_host[0] = '\0';
205	remote_port[0] = '\0';
206	if ((ret = pg_getnameinfo_all(&port->raddr.addr, port->raddr.salen,
207								  remote_host, sizeof(remote_host),
208								  remote_port, sizeof(remote_port),
209								  (log_hostname ? 0 : NI_NUMERICHOST) | NI_NUMERICSERV)) != 0)
210		ereport(WARNING,
211				(errmsg_internal("pg_getnameinfo_all() failed: %s",
212								 gai_strerror(ret))));
213
214	/*
215	 * Save remote_host and remote_port in port structure (after this, they
216	 * will appear in log_line_prefix data for log messages).
217	 */
218	port->remote_host = MemoryContextStrdup(TopMemoryContext, remote_host);
219	port->remote_port = MemoryContextStrdup(TopMemoryContext, remote_port);
220
221	/* And now we can log that the connection was received, if enabled */
222	if (log_connections & LOG_CONNECTION_RECEIPT)
223	{
224		if (remote_port[0])
225			ereport(LOG,
226					(errmsg("connection received: host=%s port=%s",
227							remote_host,
228							remote_port)));
229		else
230			ereport(LOG,
231					(errmsg("connection received: host=%s",
232							remote_host)));
233	}
234
235	/* For testing client error handling */
236#ifdef USE_INJECTION_POINTS
237	INJECTION_POINT("backend-initialize", NULL);
238	if (IS_INJECTION_POINT_ATTACHED("backend-initialize-v2-error"))
239	{
240		/*
241		 * This simulates an early error from a pre-v14 server, which used the
242		 * version 2 protocol for any errors that occurred before processing
243		 * the startup packet.
244		 */
245		FrontendProtocol = PG_PROTOCOL(2, 0);
246		elog(FATAL, "protocol version 2 error triggered");
247	}
248#endif
249
250	/*
251	 * If we did a reverse lookup to name, we might as well save the results
252	 * rather than possibly repeating the lookup during authentication.
253	 *
254	 * Note that we don't want to specify NI_NAMEREQD above, because then we'd
255	 * get nothing useful for a client without an rDNS entry.  Therefore, we
256	 * must check whether we got a numeric IPv4 or IPv6 address, and not save
257	 * it into remote_hostname if so.  (This test is conservative and might
258	 * sometimes classify a hostname as numeric, but an error in that
259	 * direction is safe; it only results in a possible extra lookup.)
260	 */
261	if (log_hostname &&
262		ret == 0 &&
263		strspn(remote_host, "0123456789.") < strlen(remote_host) &&
264		strspn(remote_host, "0123456789ABCDEFabcdef:") < strlen(remote_host))
265	{
266		port->remote_hostname = MemoryContextStrdup(TopMemoryContext, remote_host);
267	}
268
269	/*
270	 * Ready to begin client interaction.  We will give up and _exit(1) after
271	 * a time delay, so that a broken client can't hog a connection
272	 * indefinitely.  PreAuthDelay and any DNS interactions above don't count
273	 * against the time limit.
274	 *
275	 * Note: AuthenticationTimeout is applied here while waiting for the
276	 * startup packet, and then again in InitPostgres for the duration of any
277	 * authentication operations.  So a hostile client could tie up the
278	 * process for nearly twice AuthenticationTimeout before we kick him off.
279	 *
280	 * Note: because PostgresMain will call InitializeTimeouts again, the
281	 * registration of STARTUP_PACKET_TIMEOUT will be lost.  This is okay
282	 * since we never use it again after this function.
283	 */
284	RegisterTimeout(STARTUP_PACKET_TIMEOUT, StartupPacketTimeoutHandler);
285	enable_timeout_after(STARTUP_PACKET_TIMEOUT, AuthenticationTimeout * 1000);
286
287	/* Handle direct SSL handshake */
288	status = ProcessSSLStartup(port);
289
290	/*
291	 * Receive the startup packet (which might turn out to be a cancel request
292	 * packet).
293	 */
294	if (status == STATUS_OK)
295		status = ProcessStartupPacket(port, false, false);
296
297	/*
298	 * If we're going to reject the connection due to database state, say so
299	 * now instead of wasting cycles on an authentication exchange. (This also
300	 * allows a pg_ping utility to be written.)
301	 */
302	if (status == STATUS_OK)
303	{
304		switch (cac)
305		{
306			case CAC_STARTUP:
307				ereport(FATAL,
308						(errcode(ERRCODE_CANNOT_CONNECT_NOW),
309						 errmsg("the database system is starting up")));
310				break;
311			case CAC_NOTHOTSTANDBY:
312				if (!EnableHotStandby)
313					ereport(FATAL,
314							(errcode(ERRCODE_CANNOT_CONNECT_NOW),
315							 errmsg("the database system is not accepting connections"),
316							 errdetail("Hot standby mode is disabled.")));
317				else if (reachedConsistency)
318					ereport(FATAL,
319							(errcode(ERRCODE_CANNOT_CONNECT_NOW),
320							 errmsg("the database system is not yet accepting connections"),
321							 errdetail("Recovery snapshot is not yet ready for hot standby."),
322							 errhint("To enable hot standby, close write transactions with more than %d subtransactions on the primary server.",
323									 PGPROC_MAX_CACHED_SUBXIDS)));
324				else
325					ereport(FATAL,
326							(errcode(ERRCODE_CANNOT_CONNECT_NOW),
327							 errmsg("the database system is not yet accepting connections"),
328							 errdetail("Consistent recovery state has not been yet reached.")));
329				break;
330			case CAC_SHUTDOWN:
331				ereport(FATAL,
332						(errcode(ERRCODE_CANNOT_CONNECT_NOW),
333						 errmsg("the database system is shutting down")));
334				break;
335			case CAC_RECOVERY:
336				ereport(FATAL,
337						(errcode(ERRCODE_CANNOT_CONNECT_NOW),
338						 errmsg("the database system is in recovery mode")));
339				break;
340			case CAC_TOOMANY:
341				ereport(FATAL,
342						(errcode(ERRCODE_TOO_MANY_CONNECTIONS),
343						 errmsg("sorry, too many clients already")));
344				break;
345			case CAC_OK:
346				break;
347		}
348	}
349
350	/*
351	 * Disable the timeout, and prevent SIGTERM again.
352	 */
353	disable_timeout(STARTUP_PACKET_TIMEOUT, false);
354	sigprocmask(SIG_SETMASK, &BlockSig, NULL);
355
356	/*
357	 * As a safety check that nothing in startup has yet performed
358	 * shared-memory modifications that would need to be undone if we had
359	 * exited through SIGTERM or timeout above, check that no on_shmem_exit
360	 * handlers have been registered yet.  (This isn't terribly bulletproof,
361	 * since someone might misuse an on_proc_exit handler for shmem cleanup,
362	 * but it's a cheap and helpful check.  We cannot disallow on_proc_exit
363	 * handlers unfortunately, since pq_init() already registered one.)
364	 */
365	check_on_shmem_exit_lists_are_empty();
366
367	/*
368	 * Stop here if it was bad or a cancel packet.  ProcessStartupPacket
369	 * already did any appropriate error reporting.
370	 */
371	if (status != STATUS_OK)
372		proc_exit(0);
373
374	/*
375	 * Now that we have the user and database name, we can set the process
376	 * title for ps.  It's good to do this as early as possible in startup.
377	 */
378	initStringInfo(&ps_data);
379	if (am_walsender)
380		appendStringInfo(&ps_data, "%s ", GetBackendTypeDesc(B_WAL_SENDER));
381	appendStringInfo(&ps_data, "%s ", port->user_name);
382	if (port->database_name[0] != '\0')
383		appendStringInfo(&ps_data, "%s ", port->database_name);
384	appendStringInfoString(&ps_data, port->remote_host);
385	if (port->remote_port[0] != '\0')
386		appendStringInfo(&ps_data, "(%s)", port->remote_port);
387
388	init_ps_display(ps_data.data);
389	pfree(ps_data.data);
390
391	set_ps_display("initializing");
392}

394/*
395 * Check for a direct SSL connection.
396 *
397 * This happens before the startup packet so we are careful not to actually
398 * read any bytes from the stream if it's not a direct SSL connection.
399 */
400static int
401ProcessSSLStartup(Port *port)
402{
403	int			firstbyte;
404
405	Assert(!port->ssl_in_use);
406
407	pq_startmsgread();
408	firstbyte = pq_peekbyte();
409	pq_endmsgread();
410	if (firstbyte == EOF)
411	{
412		/*
413		 * Like in ProcessStartupPacket, if we get no data at all, don't
414		 * clutter the log with a complaint.
415		 */
416		return STATUS_ERROR;
417	}
418
419	if (firstbyte != 0x16)
420	{
421		/* Not an SSL handshake message */
422		return STATUS_OK;
423	}
424
425	/*
426	 * First byte indicates standard SSL handshake message
427	 *
428	 * (It can't be a Postgres startup length because in network byte order
429	 * that would be a startup packet hundreds of megabytes long)
430	 */
431
432#ifdef USE_SSL
433	if (!LoadedSSL || port->laddr.addr.ss_family == AF_UNIX)
434	{
435		/* SSL not supported */
436		goto reject;
437	}
438
439	if (secure_open_server(port) == -1)
440	{
441		/*
442		 * we assume secure_open_server() sent an appropriate TLS alert
443		 * already
444		 */
445		goto reject;
446	}
447	Assert(port->ssl_in_use);
448
449	if (!port->alpn_used)
450	{
451		ereport(COMMERROR,
452				(errcode(ERRCODE_PROTOCOL_VIOLATION),
453				 errmsg("received direct SSL connection request without ALPN protocol negotiation extension")));
454		goto reject;
455	}
456
457	if (Trace_connection_negotiation)
458		ereport(LOG,
459				(errmsg("direct SSL connection accepted")));
460	return STATUS_OK;
461#else
462	/* SSL not supported by this build */
463	goto reject;
464#endif
465
466reject:
467	if (Trace_connection_negotiation)
468		ereport(LOG,
469				(errmsg("direct SSL connection rejected")));
470	return STATUS_ERROR;
471}

473/*
474 * Read a client's startup packet and do something according to it.
475 *
476 * Returns STATUS_OK or STATUS_ERROR, or might call ereport(FATAL) and
477 * not return at all.
478 *
479 * (Note that ereport(FATAL) stuff is sent to the client, so only use it
480 * if that's what you want.  Return STATUS_ERROR if you don't want to
481 * send anything to the client, which would typically be appropriate
482 * if we detect a communications failure.)
483 *
484 * Set ssl_done and/or gss_done when negotiation of an encrypted layer
485 * (currently, TLS or GSSAPI) is completed. A successful negotiation of either
486 * encryption layer sets both flags, but a rejected negotiation sets only the
487 * flag for that layer, since the client may wish to try the other one. We
488 * should make no assumption here about the order in which the client may make
489 * requests.
490 */
491static int
492ProcessStartupPacket(Port *port, bool ssl_done, bool gss_done)
493{
494	int32		len;
495	char	   *buf;
496	ProtocolVersion proto;
497	MemoryContext oldcontext;
498
499retry:
500	pq_startmsgread();
501
502	/*
503	 * Grab the first byte of the length word separately, so that we can tell
504	 * whether we have no data at all or an incomplete packet.  (This might
505	 * sound inefficient, but it's not really, because of buffering in
506	 * pqcomm.c.)
507	 */
508	if (pq_getbytes(&len, 1) == EOF)
509	{
510		/*
511		 * If we get no data at all, don't clutter the log with a complaint;
512		 * such cases often occur for legitimate reasons.  An example is that
513		 * we might be here after responding to NEGOTIATE_SSL_CODE, and if the
514		 * client didn't like our response, it'll probably just drop the
515		 * connection.  Service-monitoring software also often just opens and
516		 * closes a connection without sending anything.  (So do port
517		 * scanners, which may be less benign, but it's not really our job to
518		 * notice those.)
519		 */
520		return STATUS_ERROR;
521	}
522
523	if (pq_getbytes(((char *) &len) + 1, 3) == EOF)
524	{
525		/* Got a partial length word, so bleat about that */
526		if (!ssl_done && !gss_done)
527			ereport(COMMERROR,
528					(errcode(ERRCODE_PROTOCOL_VIOLATION),
529					 errmsg("incomplete startup packet")));
530		return STATUS_ERROR;
531	}
532
533	len = pg_ntoh32(len);
534	len -= 4;
535
536	if (len < (int32) sizeof(ProtocolVersion) ||
537		len > MAX_STARTUP_PACKET_LENGTH)
538	{
539		ereport(COMMERROR,
540				(errcode(ERRCODE_PROTOCOL_VIOLATION),
541				 errmsg("invalid length of startup packet")));
542		return STATUS_ERROR;
543	}
544
545	/*
546	 * Allocate space to hold the startup packet, plus one extra byte that's
547	 * initialized to be zero.  This ensures we will have null termination of
548	 * all strings inside the packet.
549	 */
550	buf = palloc(len + 1);
551	buf[len] = '\0';
552
553	if (pq_getbytes(buf, len) == EOF)
554	{
555		ereport(COMMERROR,
556				(errcode(ERRCODE_PROTOCOL_VIOLATION),
557				 errmsg("incomplete startup packet")));
558		return STATUS_ERROR;
559	}
560	pq_endmsgread();
561
562	/*
563	 * The first field is either a protocol version number or a special
564	 * request code.
565	 */
566	port->proto = proto = pg_ntoh32(*((ProtocolVersion *) buf));
567
568	if (proto == CANCEL_REQUEST_CODE)
569	{
570		ProcessCancelRequestPacket(port, buf, len);
571		/* Not really an error, but we don't want to proceed further */
572		return STATUS_ERROR;
573	}
574
575	if (proto == NEGOTIATE_SSL_CODE && !ssl_done)
576	{
577		char		SSLok;
578
579#ifdef USE_SSL
580
581		/*
582		 * No SSL when disabled or on Unix sockets.
583		 *
584		 * Also no SSL negotiation if we already have a direct SSL connection
585		 */
586		if (!LoadedSSL || port->laddr.addr.ss_family == AF_UNIX || port->ssl_in_use)
587			SSLok = 'N';
588		else
589			SSLok = 'S';		/* Support for SSL */
590#else
591		SSLok = 'N';			/* No support for SSL */
592#endif
593
594		if (Trace_connection_negotiation)
595		{
596			if (SSLok == 'S')
597				ereport(LOG,
598						(errmsg("SSLRequest accepted")));
599			else
600				ereport(LOG,
601						(errmsg("SSLRequest rejected")));
602		}
603
604		while (secure_write(port, &SSLok, 1) != 1)
605		{
606			if (errno == EINTR)
607				continue;		/* if interrupted, just retry */
608			ereport(COMMERROR,
609					(errcode_for_socket_access(),
610					 errmsg("failed to send SSL negotiation response: %m")));
611			return STATUS_ERROR;	/* close the connection */
612		}
613
614#ifdef USE_SSL
615		if (SSLok == 'S' && secure_open_server(port) == -1)
616			return STATUS_ERROR;
617#endif
618
619		/*
620		 * At this point we should have no data already buffered.  If we do,
621		 * it was received before we performed the SSL handshake, so it wasn't
622		 * encrypted and indeed may have been injected by a man-in-the-middle.
623		 * We report this case to the client.
624		 */
625		if (pq_buffer_remaining_data() > 0)
626			ereport(FATAL,
627					(errcode(ERRCODE_PROTOCOL_VIOLATION),
628					 errmsg("received unencrypted data after SSL request"),
629					 errdetail("This could be either a client-software bug or evidence of an attempted man-in-the-middle attack.")));
630
631		/*
632		 * regular startup packet, cancel, etc packet should follow, but not
633		 * another SSL negotiation request, and a GSS request should only
634		 * follow if SSL was rejected (client may negotiate in either order)
635		 */
636		ssl_done = true;
637		if (SSLok == 'S')
638		{
639			/*
640			 * We are done with SSL and negotiated correctly, so consider the
641			 * same for GSS.
642			 */
643			gss_done = true;
644		}
645		goto retry;
646	}
647	else if (proto == NEGOTIATE_GSS_CODE && !gss_done)
648	{
649		char		GSSok = 'N';
650
651#ifdef ENABLE_GSS
652		/* No GSSAPI encryption when on Unix socket */
653		if (port->laddr.addr.ss_family != AF_UNIX)
654			GSSok = 'G';
655#endif
656
657		if (Trace_connection_negotiation)
658		{
659			if (GSSok == 'G')
660				ereport(LOG,
661						(errmsg("GSSENCRequest accepted")));
662			else
663				ereport(LOG,
664						(errmsg("GSSENCRequest rejected")));
665		}
666
667		while (secure_write(port, &GSSok, 1) != 1)
668		{
669			if (errno == EINTR)
670				continue;
671			ereport(COMMERROR,
672					(errcode_for_socket_access(),
673					 errmsg("failed to send GSSAPI negotiation response: %m")));
674			return STATUS_ERROR;	/* close the connection */
675		}
676
677#ifdef ENABLE_GSS
678		if (GSSok == 'G' && secure_open_gssapi(port) == -1)
679			return STATUS_ERROR;
680#endif
681
682		/*
683		 * At this point we should have no data already buffered.  If we do,
684		 * it was received before we performed the GSS handshake, so it wasn't
685		 * encrypted and indeed may have been injected by a man-in-the-middle.
686		 * We report this case to the client.
687		 */
688		if (pq_buffer_remaining_data() > 0)
689			ereport(FATAL,
690					(errcode(ERRCODE_PROTOCOL_VIOLATION),
691					 errmsg("received unencrypted data after GSSAPI encryption request"),
692					 errdetail("This could be either a client-software bug or evidence of an attempted man-in-the-middle attack.")));
693
694		/*
695		 * regular startup packet, cancel, etc packet should follow, but not
696		 * another GSS negotiation request, and an SSL request should only
697		 * follow if GSS was rejected (client may negotiate in either order)
698		 */
699		gss_done = true;
700		if (GSSok == 'G')
701		{
702			/*
703			 * We are done with GSS and negotiated correctly, so consider the
704			 * same for SSL.
705			 */
706			ssl_done = true;
707		}
708		goto retry;
709	}
710
711	/* Could add additional special packet types here */
712
713	/*
714	 * Set FrontendProtocol now so that ereport() knows what format to send if
715	 * we fail during startup. We use the protocol version requested by the
716	 * client unless it's higher than the latest version we support. It's
717	 * possible that error message fields might look different in newer
718	 * protocol versions, but that's something those new clients should be
719	 * able to deal with.
720	 */
721	FrontendProtocol = Min(proto, PG_PROTOCOL_LATEST);
722
723	/* Check that the major protocol version is in range. */
724	if (PG_PROTOCOL_MAJOR(proto) < PG_PROTOCOL_MAJOR(PG_PROTOCOL_EARLIEST) ||
725		PG_PROTOCOL_MAJOR(proto) > PG_PROTOCOL_MAJOR(PG_PROTOCOL_LATEST))
726		ereport(FATAL,
727				(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
728				 errmsg("unsupported frontend protocol %u.%u: server supports %u.0 to %u.%u",
729						PG_PROTOCOL_MAJOR(proto), PG_PROTOCOL_MINOR(proto),
730						PG_PROTOCOL_MAJOR(PG_PROTOCOL_EARLIEST),
731						PG_PROTOCOL_MAJOR(PG_PROTOCOL_LATEST),
732						PG_PROTOCOL_MINOR(PG_PROTOCOL_LATEST))));
733
734	/*
735	 * Now fetch parameters out of startup packet and save them into the Port
736	 * structure.
737	 */
738	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
739
740	/* Handle protocol version 3 startup packet */
741	{
742		int32		offset = sizeof(ProtocolVersion);
743		List	   *unrecognized_protocol_options = NIL;
744
745		/*
746		 * Scan packet body for name/option pairs.  We can assume any string
747		 * beginning within the packet body is null-terminated, thanks to
748		 * zeroing extra byte above.
749		 */
750		port->guc_options = NIL;
751
752		while (offset < len)
753		{
754			char	   *nameptr = buf + offset;
755			int32		valoffset;
756			char	   *valptr;
757
758			if (*nameptr == '\0')
759				break;			/* found packet terminator */
760			valoffset = offset + strlen(nameptr) + 1;
761			if (valoffset >= len)
762				break;			/* missing value, will complain below */
763			valptr = buf + valoffset;
764
765			if (strcmp(nameptr, "database") == 0)
766				port->database_name = pstrdup(valptr);
767			else if (strcmp(nameptr, "user") == 0)
768				port->user_name = pstrdup(valptr);
769			else if (strcmp(nameptr, "options") == 0)
770				port->cmdline_options = pstrdup(valptr);
771			else if (strcmp(nameptr, "replication") == 0)
772			{
773				/*
774				 * Due to backward compatibility concerns the replication
775				 * parameter is a hybrid beast which allows the value to be
776				 * either boolean or the string 'database'. The latter
777				 * connects to a specific database which is e.g. required for
778				 * logical decoding while.
779				 */
780				if (strcmp(valptr, "database") == 0)
781				{
782					am_walsender = true;
783					am_db_walsender = true;
784				}
785				else if (!parse_bool(valptr, &am_walsender))
786					ereport(FATAL,
787							(errcode(ERRCODE_INVALID_PARAMETER_VALUE),
788							 errmsg("invalid value for parameter \"%s\": \"%s\"",
789									"replication",
790									valptr),
791							 errhint("Valid values are: \"false\", 0, \"true\", 1, \"database\".")));
792			}
793			else if (strncmp(nameptr, "_pq_.", 5) == 0)
794			{
795				/*
796				 * Any option beginning with _pq_. is reserved for use as a
797				 * protocol-level option, but at present no such options are
798				 * defined.
799				 */
800				unrecognized_protocol_options =
801					lappend(unrecognized_protocol_options, pstrdup(nameptr));
802			}
803			else
804			{
805				/* Assume it's a generic GUC option */
806				port->guc_options = lappend(port->guc_options,
807											pstrdup(nameptr));
808				port->guc_options = lappend(port->guc_options,
809											pstrdup(valptr));
810
811				/*
812				 * Copy application_name to port if we come across it.  This
813				 * is done so we can log the application_name in the
814				 * connection authorization message.  Note that the GUC would
815				 * be used but we haven't gone through GUC setup yet.
816				 */
817				if (strcmp(nameptr, "application_name") == 0)
818				{
819					port->application_name = pg_clean_ascii(valptr, 0);
820				}
821			}
822			offset = valoffset + strlen(valptr) + 1;
823		}
824
825		/*
826		 * If we didn't find a packet terminator exactly at the end of the
827		 * given packet length, complain.
828		 */
829		if (offset != len - 1)
830			ereport(FATAL,
831					(errcode(ERRCODE_PROTOCOL_VIOLATION),
832					 errmsg("invalid startup packet layout: expected terminator as last byte")));
833
834		/*
835		 * If the client requested a newer protocol version or if the client
836		 * requested any protocol options we didn't recognize, let them know
837		 * the newest minor protocol version we do support and the names of
838		 * any unrecognized options.
839		 */
840		if (PG_PROTOCOL_MINOR(proto) > PG_PROTOCOL_MINOR(PG_PROTOCOL_LATEST) ||
841			unrecognized_protocol_options != NIL)
842			SendNegotiateProtocolVersion(unrecognized_protocol_options);
843	}
844
845	/* Check a user name was given. */
846	if (port->user_name == NULL || port->user_name[0] == '\0')
847		ereport(FATAL,
848				(errcode(ERRCODE_INVALID_AUTHORIZATION_SPECIFICATION),
849				 errmsg("no PostgreSQL user name specified in startup packet")));
850
851	/* The database defaults to the user name. */
852	if (port->database_name == NULL || port->database_name[0] == '\0')
853		port->database_name = pstrdup(port->user_name);
854
855	/*
856	 * Truncate given database and user names to length of a Postgres name.
857	 * This avoids lookup failures when overlength names are given.
858	 */
859	if (strlen(port->database_name) >= NAMEDATALEN)
860		port->database_name[NAMEDATALEN - 1] = '\0';
861	if (strlen(port->user_name) >= NAMEDATALEN)
862		port->user_name[NAMEDATALEN - 1] = '\0';
863
864	if (am_walsender)
865		MyBackendType = B_WAL_SENDER;
866	else
867		MyBackendType = B_BACKEND;
868
869	/*
870	 * Normal walsender backends, e.g. for streaming replication, are not
871	 * connected to a particular database. But walsenders used for logical
872	 * replication need to connect to a specific database. We allow streaming
873	 * replication commands to be issued even if connected to a database as it
874	 * can make sense to first make a basebackup and then stream changes
875	 * starting from that.
876	 */
877	if (am_walsender && !am_db_walsender)
878		port->database_name[0] = '\0';
879
880	/*
881	 * Done filling the Port structure
882	 */
883	MemoryContextSwitchTo(oldcontext);
884
885	return STATUS_OK;
886}

888/*
889 * The client has sent a cancel request packet, not a normal
890 * start-a-new-connection packet.  Perform the necessary processing.  Nothing
891 * is sent back to the client.
892 */
893static void
894ProcessCancelRequestPacket(Port *port, void *pkt, int pktlen)
895{
896	CancelRequestPacket *canc;
897	int			len;
898
899	if (pktlen < offsetof(CancelRequestPacket, cancelAuthCode))
900	{
901		ereport(COMMERROR,
902				(errcode(ERRCODE_PROTOCOL_VIOLATION),
903				 errmsg("invalid length of cancel request packet")));
904		return;
905	}
906	len = pktlen - offsetof(CancelRequestPacket, cancelAuthCode);
907	if (len == 0 || len > 256)
908	{
909		ereport(COMMERROR,
910				(errcode(ERRCODE_PROTOCOL_VIOLATION),
911				 errmsg("invalid length of cancel key in cancel request packet")));
912		return;
913	}
914
915	canc = (CancelRequestPacket *) pkt;
916	SendCancelRequest(pg_ntoh32(canc->backendPID), canc->cancelAuthCode, len);
917}

919/*
920 * Send a NegotiateProtocolVersion to the client.  This lets the client know
921 * that they have either requested a newer minor protocol version than we are
922 * able to speak, or at least one protocol option that we don't understand, or
923 * possibly both. FrontendProtocol has already been set to the version
924 * requested by the client or the highest version we know how to speak,
925 * whichever is older. If the highest version that we know how to speak is too
926 * old for the client, it can abandon the connection.
927 *
928 * We also include in the response a list of protocol options we didn't
929 * understand.  This allows clients to include optional parameters that might
930 * be present either in newer protocol versions or third-party protocol
931 * extensions without fear of having to reconnect if those options are not
932 * understood, while at the same time making certain that the client is aware
933 * of which options were actually accepted.
934 */
935static void
936SendNegotiateProtocolVersion(List *unrecognized_protocol_options)
937{
938	StringInfoData buf;
939	ListCell   *lc;
940
941	pq_beginmessage(&buf, PqMsg_NegotiateProtocolVersion);
942	pq_sendint32(&buf, FrontendProtocol);
943	pq_sendint32(&buf, list_length(unrecognized_protocol_options));
944	foreach(lc, unrecognized_protocol_options)
945		pq_sendstring(&buf, lfirst(lc));
946	pq_endmessage(&buf);
947
948	/* no need to flush, some other message will follow */
949}

953 * SIGTERM while processing startup packet.
954 *
955 * Running proc_exit() from a signal handler would be quite unsafe.
956 * However, since we have not yet touched shared memory, we can just
957 * pull the plug and exit without running any atexit handlers.
958 *
959 * One might be tempted to try to send a message, or log one, indicating
960 * why we are disconnecting.  However, that would be quite unsafe in itself.
961 * Also, it seems undesirable to provide clues about the database's state
962 * to a client that has not yet completed authentication, or even sent us
963 * a startup packet.
964 */
965static void
966process_startup_packet_die(SIGNAL_ARGS)
967{
968	_exit(1);
969}

981/*
982 * Helper for the log_connections GUC check hook.
983 *
984 * `elemlist` is a listified version of the string input passed to the
985 * log_connections GUC check hook, check_log_connections().
986 * check_log_connections() is responsible for cleaning up `elemlist`.
987 *
988 * validate_log_connections_options() returns false if an error was
989 * encountered and the GUC input could not be validated and true otherwise.
990 *
991 * `flags` returns the flags that should be stored in the log_connections GUC
992 * by its assign hook.
993 */
994static bool
995validate_log_connections_options(List *elemlist, uint32 *flags)
996{
997	ListCell   *l;
998	char	   *item;
999
1000	/*
1001	 * For backwards compatibility, we accept these tokens by themselves.
1002	 *
1003	 * Prior to PostgreSQL 18, log_connections was a boolean GUC that accepted
1004	 * any unambiguous substring of 'true', 'false', 'yes', 'no', 'on', and
1005	 * 'off'. Since log_connections became a list of strings in 18, we only
1006	 * accept complete option strings.
1007	 */
1008	static const struct config_enum_entry compat_options[] = {
1009		{"off", 0},
1010		{"false", 0},
1011		{"no", 0},
1012		{"0", 0},
1013		{"on", LOG_CONNECTION_ON},
1014		{"true", LOG_CONNECTION_ON},
1015		{"yes", LOG_CONNECTION_ON},
1016		{"1", LOG_CONNECTION_ON},
1017	};
1018
1019	*flags = 0;
1020
1021	/* If an empty string was passed, we're done */
1022	if (list_length(elemlist) == 0)
1023		return true;
1024
1025	/*
1026	 * Now check for the backwards compatibility options. They must always be
1027	 * specified on their own, so we error out if the first option is a
1028	 * backwards compatibility option and other options are also specified.
1029	 */
1030	item = linitial(elemlist);
1031
1032	for (size_t i = 0; i < lengthof(compat_options); i++)
1033	{
1034		struct config_enum_entry option = compat_options[i];
1035
1036		if (pg_strcasecmp(item, option.name) != 0)
1037			continue;
1038
1039		if (list_length(elemlist) > 1)
1040		{
1041			GUC_check_errdetail("Cannot specify log_connections option \"%s\" in a list with other options.",
1042								item);
1043			return false;
1044		}
1045
1046		*flags = option.val;
1047		return true;
1048	}
1049
1050	/* Now check the aspect options. The empty string was already handled */
1051	foreach(l, elemlist)
1052	{
1053		static const struct config_enum_entry options[] = {
1054			{"receipt", LOG_CONNECTION_RECEIPT},
1055			{"authentication", LOG_CONNECTION_AUTHENTICATION},
1056			{"authorization", LOG_CONNECTION_AUTHORIZATION},
1057			{"setup_durations", LOG_CONNECTION_SETUP_DURATIONS},
1058			{"all", LOG_CONNECTION_ALL},
1059		};
1060
1061		item = lfirst(l);
1062		for (size_t i = 0; i < lengthof(options); i++)
1063		{
1064			struct config_enum_entry option = options[i];
1065
1066			if (pg_strcasecmp(item, option.name) == 0)
1067			{
1068				*flags |= option.val;
1069				goto next;
1070			}
1071		}
1072
1073		GUC_check_errdetail("Invalid option \"%s\".", item);
1074		return false;
1075
1076next:	;
1077	}
1078
1079	return true;
1080}

1083/*
1084 * GUC check hook for log_connections
1085 */
1086bool
1087check_log_connections(char **newval, void **extra, GucSource source)
1088{
1089	uint32		flags;
1090	char	   *rawstring;
1091	List	   *elemlist;
1092	bool		success;
1093
1094	/* Need a modifiable copy of string */
1095	rawstring = pstrdup(*newval);
1096
1097	if (!SplitIdentifierString(rawstring, ',', &elemlist))
1098	{
1099		GUC_check_errdetail("Invalid list syntax in parameter \"%s\".", "log_connections");
1100		pfree(rawstring);
1101		list_free(elemlist);
1102		return false;
1103	}
1104
1105	/* Validation logic is all in the helper */
1106	success = validate_log_connections_options(elemlist, &flags);
1107
1108	/* Time for cleanup */
1109	pfree(rawstring);
1110	list_free(elemlist);
1111
1112	if (!success)
1113		return false;
1114
1115	/*
1116	 * We succeeded, so allocate `extra` and save the flags there for use by
1117	 * assign_log_connections().
1118	 */
1119	*extra = guc_malloc(LOG, sizeof(int));
1120	if (!*extra)
1121		return false;
1122	*((int *) *extra) = flags;
1123
1124	return true;
1125}
