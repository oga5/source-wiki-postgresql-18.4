54/*
55 * Data in shared memory related to WAL summarization.
56 */
57typedef struct
58{
59	/*
60	 * These fields are protected by WALSummarizerLock.
61	 *
62	 * Until we've discovered what summary files already exist on disk and
63	 * stored that information in shared memory, initialized is false and the
64	 * other fields here contain no meaningful information. After that has
65	 * been done, initialized is true.
66	 *
67	 * summarized_tli and summarized_lsn indicate the last LSN and TLI at
68	 * which the next summary file will start. Normally, these are the LSN and
69	 * TLI at which the last file ended; in such case, lsn_is_exact is true.
70	 * If, however, the LSN is just an approximation, then lsn_is_exact is
71	 * false. This can happen if, for example, there are no existing WAL
72	 * summary files at startup. In that case, we have to derive the position
73	 * at which to start summarizing from the WAL files that exist on disk,
74	 * and so the LSN might point to the start of the next file even though
75	 * that might happen to be in the middle of a WAL record.
76	 *
77	 * summarizer_pgprocno is the proc number of the summarizer process, if
78	 * one is running, or else INVALID_PROC_NUMBER.
79	 *
80	 * pending_lsn is used by the summarizer to advertise the ending LSN of a
81	 * record it has recently read. It shouldn't ever be less than
82	 * summarized_lsn, but might be greater, because the summarizer buffers
83	 * data for a range of LSNs in memory before writing out a new file.
84	 */
85	bool		initialized;
86	TimeLineID	summarized_tli;
87	XLogRecPtr	summarized_lsn;
88	bool		lsn_is_exact;
89	ProcNumber	summarizer_pgprocno;
90	XLogRecPtr	pending_lsn;
91
92	/*
93	 * This field handles its own synchronization.
94	 */
95	ConditionVariable summary_file_cv;
96} WalSummarizerData;

112/*
113 * When we reach end of WAL and need to read more, we sleep for a number of
114 * milliseconds that is an integer multiple of MS_PER_SLEEP_QUANTUM. This is
115 * the multiplier. It should vary between 1 and MAX_SLEEP_QUANTA, depending
116 * on system activity. See summarizer_wait_for_wal() for how we adjust this.
117 */
118static long sleep_quanta = 1;
119
120/*
121 * The sleep time will always be a multiple of 200ms and will not exceed
122 * thirty seconds (150 * 200 = 30 * 1000). Note that the timeout here needs
123 * to be substantially less than the maximum amount of time for which an
124 * incremental backup will wait for this process to catch up. Otherwise, an
125 * incremental backup might time out on an idle system just because we sleep
126 * for too long.
127 */
128#define MAX_SLEEP_QUANTA		150
129#define MS_PER_SLEEP_QUANTUM	200

210/*
211 * Entry point for walsummarizer process.
212 */
213void
214WalSummarizerMain(const void *startup_data, size_t startup_data_len)
215{
216	sigjmp_buf	local_sigjmp_buf;
217	MemoryContext context;
218
219	/*
220	 * Within this function, 'current_lsn' and 'current_tli' refer to the
221	 * point from which the next WAL summary file should start. 'exact' is
222	 * true if 'current_lsn' is known to be the start of a WAL record or WAL
223	 * segment, and false if it might be in the middle of a record someplace.
224	 *
225	 * 'switch_lsn' and 'switch_tli', if set, are the LSN at which we need to
226	 * switch to a new timeline and the timeline to which we need to switch.
227	 * If not set, we either haven't figured out the answers yet or we're
228	 * already on the latest timeline.
229	 */
230	XLogRecPtr	current_lsn;
231	TimeLineID	current_tli;
232	bool		exact;
233	XLogRecPtr	switch_lsn = InvalidXLogRecPtr;
234	TimeLineID	switch_tli = 0;
235
236	Assert(startup_data_len == 0);
237
238	MyBackendType = B_WAL_SUMMARIZER;
239	AuxiliaryProcessMainCommon();
240
241	ereport(DEBUG1,
242			(errmsg_internal("WAL summarizer started")));
243
244	/*
245	 * Properly accept or ignore signals the postmaster might send us
246	 *
247	 * We have no particular use for SIGINT at the moment, but seems
248	 * reasonable to treat like SIGTERM.
249	 */
250	pqsignal(SIGHUP, SignalHandlerForConfigReload);
251	pqsignal(SIGINT, SignalHandlerForShutdownRequest);
252	pqsignal(SIGTERM, SignalHandlerForShutdownRequest);
253	/* SIGQUIT handler was already set up by InitPostmasterChild */
254	pqsignal(SIGALRM, SIG_IGN);
255	pqsignal(SIGPIPE, SIG_IGN);
256	pqsignal(SIGUSR1, procsignal_sigusr1_handler);
257	pqsignal(SIGUSR2, SIG_IGN); /* not used */
258
259	/* Advertise ourselves. */
260	on_shmem_exit(WalSummarizerShutdown, (Datum) 0);
261	LWLockAcquire(WALSummarizerLock, LW_EXCLUSIVE);
262	WalSummarizerCtl->summarizer_pgprocno = MyProcNumber;
263	LWLockRelease(WALSummarizerLock);
264
265	/* Create and switch to a memory context that we can reset on error. */
266	context = AllocSetContextCreate(TopMemoryContext,
267									"Wal Summarizer",
268									ALLOCSET_DEFAULT_SIZES);
269	MemoryContextSwitchTo(context);
270
271	/*
272	 * Reset some signals that are accepted by postmaster but not here
273	 */
274	pqsignal(SIGCHLD, SIG_DFL);
275
276	/*
277	 * If an exception is encountered, processing resumes here.
278	 */
279	if (sigsetjmp(local_sigjmp_buf, 1) != 0)
280	{
281		/* Since not using PG_TRY, must reset error stack by hand */
282		error_context_stack = NULL;
283
284		/* Prevent interrupts while cleaning up */
285		HOLD_INTERRUPTS();
286
287		/* Report the error to the server log */
288		EmitErrorReport();
289
290		/* Release resources we might have acquired. */
291		LWLockReleaseAll();
292		ConditionVariableCancelSleep();
293		pgstat_report_wait_end();
294		pgaio_error_cleanup();
295		ReleaseAuxProcessResources(false);
296		AtEOXact_Files(false);
297		AtEOXact_HashTables(false);
298
299		/*
300		 * Now return to normal top-level context and clear ErrorContext for
301		 * next time.
302		 */
303		MemoryContextSwitchTo(context);
304		FlushErrorState();
305
306		/* Flush any leaked data in the top-level context */
307		MemoryContextReset(context);
308
309		/* Now we can allow interrupts again */
310		RESUME_INTERRUPTS();
311
312		/*
313		 * Sleep for 10 seconds before attempting to resume operations in
314		 * order to avoid excessive logging.
315		 *
316		 * Many of the likely error conditions are things that will repeat
317		 * every time. For example, if the WAL can't be read or the summary
318		 * can't be written, only administrator action will cure the problem.
319		 * So a really fast retry time doesn't seem to be especially
320		 * beneficial, and it will clutter the logs.
321		 */
322		(void) WaitLatch(NULL,
323						 WL_TIMEOUT | WL_EXIT_ON_PM_DEATH,
324						 10000,
325						 WAIT_EVENT_WAL_SUMMARIZER_ERROR);
326	}
327
328	/* We can now handle ereport(ERROR) */
329	PG_exception_stack = &local_sigjmp_buf;
330
331	/*
332	 * Unblock signals (they were blocked when the postmaster forked us)
333	 */
334	sigprocmask(SIG_SETMASK, &UnBlockSig, NULL);
335
336	/*
337	 * Fetch information about previous progress from shared memory, and ask
338	 * GetOldestUnsummarizedLSN to reset pending_lsn to summarized_lsn. We
339	 * might be recovering from an error, and if so, pending_lsn might have
340	 * advanced past summarized_lsn, but any WAL we read previously has been
341	 * lost and will need to be reread.
342	 *
343	 * If we discover that WAL summarization is not enabled, just exit.
344	 */
345	current_lsn = GetOldestUnsummarizedLSN(&current_tli, &exact);
346	if (XLogRecPtrIsInvalid(current_lsn))
347		proc_exit(0);
348
349	/*
350	 * Loop forever
351	 */
352	for (;;)
353	{
354		XLogRecPtr	latest_lsn;
355		TimeLineID	latest_tli;
356		XLogRecPtr	end_of_summary_lsn;
357
358		/* Flush any leaked data in the top-level context */
359		MemoryContextReset(context);
360
361		/* Process any signals received recently. */
362		ProcessWalSummarizerInterrupts();
363
364		/* If it's time to remove any old WAL summaries, do that now. */
365		MaybeRemoveOldWalSummaries();
366
367		/* Find the LSN and TLI up to which we can safely summarize. */
368		latest_lsn = GetLatestLSN(&latest_tli);
369
370		/*
371		 * If we're summarizing a historic timeline and we haven't yet
372		 * computed the point at which to switch to the next timeline, do that
373		 * now.
374		 *
375		 * Note that if this is a standby, what was previously the current
376		 * timeline could become historic at any time.
377		 *
378		 * We could try to make this more efficient by caching the results of
379		 * readTimeLineHistory when latest_tli has not changed, but since we
380		 * only have to do this once per timeline switch, we probably wouldn't
381		 * save any significant amount of work in practice.
382		 */
383		if (current_tli != latest_tli && XLogRecPtrIsInvalid(switch_lsn))
384		{
385			List	   *tles = readTimeLineHistory(latest_tli);
386
387			switch_lsn = tliSwitchPoint(current_tli, tles, &switch_tli);
388			ereport(DEBUG1,
389					errmsg_internal("switch point from TLI %u to TLI %u is at %X/%X",
390									current_tli, switch_tli, LSN_FORMAT_ARGS(switch_lsn)));
391		}
392
393		/*
394		 * If we've reached the switch LSN, we can't summarize anything else
395		 * on this timeline. Switch to the next timeline and go around again,
396		 * backing up to the exact switch point if we passed it.
397		 */
398		if (!XLogRecPtrIsInvalid(switch_lsn) && current_lsn >= switch_lsn)
399		{
400			/* Restart summarization from switch point. */
401			current_tli = switch_tli;
402			current_lsn = switch_lsn;
403
404			/* Next timeline and switch point, if any, not yet known. */
405			switch_lsn = InvalidXLogRecPtr;
406			switch_tli = 0;
407
408			/* Update (really, rewind, if needed) state in shared memory. */
409			LWLockAcquire(WALSummarizerLock, LW_EXCLUSIVE);
410			WalSummarizerCtl->summarized_lsn = current_lsn;
411			WalSummarizerCtl->summarized_tli = current_tli;
412			WalSummarizerCtl->lsn_is_exact = true;
413			WalSummarizerCtl->pending_lsn = current_lsn;
414			LWLockRelease(WALSummarizerLock);
415
416			continue;
417		}
418
419		/* Summarize WAL. */
420		end_of_summary_lsn = SummarizeWAL(current_tli,
421										  current_lsn, exact,
422										  switch_lsn, latest_lsn);
423		Assert(!XLogRecPtrIsInvalid(end_of_summary_lsn));
424		Assert(end_of_summary_lsn >= current_lsn);
425
426		/*
427		 * Update state for next loop iteration.
428		 *
429		 * Next summary file should start from exactly where this one ended.
430		 */
431		current_lsn = end_of_summary_lsn;
432		exact = true;
433
434		/* Update state in shared memory. */
435		LWLockAcquire(WALSummarizerLock, LW_EXCLUSIVE);
436		WalSummarizerCtl->summarized_lsn = end_of_summary_lsn;
437		WalSummarizerCtl->summarized_tli = current_tli;
438		WalSummarizerCtl->lsn_is_exact = true;
439		WalSummarizerCtl->pending_lsn = end_of_summary_lsn;
440		LWLockRelease(WALSummarizerLock);
441
442		/* Wake up anyone waiting for more summary files to be written. */
443		ConditionVariableBroadcast(&WalSummarizerCtl->summary_file_cv);
444	}
445}

447/*
448 * Get information about the state of the WAL summarizer.
449 */
450void
451GetWalSummarizerState(TimeLineID *summarized_tli, XLogRecPtr *summarized_lsn,
452					  XLogRecPtr *pending_lsn, int *summarizer_pid)
453{
454	LWLockAcquire(WALSummarizerLock, LW_SHARED);
455	if (!WalSummarizerCtl->initialized)
456	{
457		/*
458		 * If initialized is false, the rest of the structure contents are
459		 * undefined.
460		 */
461		*summarized_tli = 0;
462		*summarized_lsn = InvalidXLogRecPtr;
463		*pending_lsn = InvalidXLogRecPtr;
464		*summarizer_pid = -1;
465	}
466	else
467	{
468		int			summarizer_pgprocno = WalSummarizerCtl->summarizer_pgprocno;
469
470		*summarized_tli = WalSummarizerCtl->summarized_tli;
471		*summarized_lsn = WalSummarizerCtl->summarized_lsn;
472		if (summarizer_pgprocno == INVALID_PROC_NUMBER)
473		{
474			/*
475			 * If the summarizer has exited, the fact that it had processed
476			 * beyond summarized_lsn is irrelevant now.
477			 */
478			*pending_lsn = WalSummarizerCtl->summarized_lsn;
479			*summarizer_pid = -1;
480		}
481		else
482		{
483			*pending_lsn = WalSummarizerCtl->pending_lsn;
484
485			/*
486			 * We're not fussed about inexact answers here, since they could
487			 * become stale instantly, so we don't bother taking the lock, but
488			 * make sure that invalid PID values are normalized to -1.
489			 */
490			*summarizer_pid = GetPGProcByNumber(summarizer_pgprocno)->pid;
491			if (*summarizer_pid <= 0)
492				*summarizer_pid = -1;
493		}
494	}
495	LWLockRelease(WALSummarizerLock);
496}

498/*
499 * Get the oldest LSN in this server's timeline history that has not yet been
500 * summarized, and update shared memory state as appropriate.
501 *
502 * If *tli != NULL, it will be set to the TLI for the LSN that is returned.
503 *
504 * If *lsn_is_exact != NULL, it will be set to true if the returned LSN is
505 * necessarily the start of a WAL record and false if it's just the beginning
506 * of a WAL segment.
507 */
508XLogRecPtr
509GetOldestUnsummarizedLSN(TimeLineID *tli, bool *lsn_is_exact)
510{
511	TimeLineID	latest_tli;
512	int			n;
513	List	   *tles;
514	XLogRecPtr	unsummarized_lsn = InvalidXLogRecPtr;
515	TimeLineID	unsummarized_tli = 0;
516	bool		should_make_exact = false;
517	List	   *existing_summaries;
518	ListCell   *lc;
519	bool		am_wal_summarizer = AmWalSummarizerProcess();
520
521	/* If not summarizing WAL, do nothing. */
522	if (!summarize_wal)
523		return InvalidXLogRecPtr;
524
525	/*
526	 * If we are not the WAL summarizer process, then we normally just want to
527	 * read the values from shared memory. However, as an exception, if shared
528	 * memory hasn't been initialized yet, then we need to do that so that we
529	 * can read legal values and not remove any WAL too early.
530	 */
531	if (!am_wal_summarizer)
532	{
533		LWLockAcquire(WALSummarizerLock, LW_SHARED);
534
535		if (WalSummarizerCtl->initialized)
536		{
537			unsummarized_lsn = WalSummarizerCtl->summarized_lsn;
538			if (tli != NULL)
539				*tli = WalSummarizerCtl->summarized_tli;
540			if (lsn_is_exact != NULL)
541				*lsn_is_exact = WalSummarizerCtl->lsn_is_exact;
542			LWLockRelease(WALSummarizerLock);
543			return unsummarized_lsn;
544		}
545
546		LWLockRelease(WALSummarizerLock);
547	}
548
549	/*
550	 * Find the oldest timeline on which WAL still exists, and the earliest
551	 * segment for which it exists.
552	 *
553	 * Note that we do this every time the WAL summarizer process restarts or
554	 * recovers from an error, in case the contents of pg_wal have changed
555	 * under us e.g. if some files were removed, either manually - which
556	 * shouldn't really happen, but might - or by postgres itself, if
557	 * summarize_wal was turned off and then back on again.
558	 */
559	(void) GetLatestLSN(&latest_tli);
560	tles = readTimeLineHistory(latest_tli);
561	for (n = list_length(tles) - 1; n >= 0; --n)
562	{
563		TimeLineHistoryEntry *tle = list_nth(tles, n);
564		XLogSegNo	oldest_segno;
565
566		oldest_segno = XLogGetOldestSegno(tle->tli);
567		if (oldest_segno != 0)
568		{
569			/* Compute oldest LSN that still exists on disk. */
570			XLogSegNoOffsetToRecPtr(oldest_segno, 0, wal_segment_size,
571									unsummarized_lsn);
572
573			unsummarized_tli = tle->tli;
574			break;
575		}
576	}
577
578	/*
579	 * Don't try to summarize anything older than the end LSN of the newest
580	 * summary file that exists for this timeline.
581	 */
582	existing_summaries =
583		GetWalSummaries(unsummarized_tli,
584						InvalidXLogRecPtr, InvalidXLogRecPtr);
585	foreach(lc, existing_summaries)
586	{
587		WalSummaryFile *ws = lfirst(lc);
588
589		if (ws->end_lsn > unsummarized_lsn)
590		{
591			unsummarized_lsn = ws->end_lsn;
592			should_make_exact = true;
593		}
594	}
595
596	/* It really should not be possible for us to find no WAL. */
597	if (unsummarized_tli == 0)
598		ereport(ERROR,
599				errcode(ERRCODE_INTERNAL_ERROR),
600				errmsg_internal("no WAL found on timeline %u", latest_tli));
601
602	/*
603	 * If we're the WAL summarizer, we always want to store the values we just
604	 * computed into shared memory, because those are the values we're going
605	 * to use to drive our operation, and so they are the authoritative
606	 * values. Otherwise, we only store values into shared memory if shared
607	 * memory is uninitialized. Our values are not canonical in such a case,
608	 * but it's better to have something than nothing, to guide WAL retention.
609	 */
610	LWLockAcquire(WALSummarizerLock, LW_EXCLUSIVE);
611	if (am_wal_summarizer || !WalSummarizerCtl->initialized)
612	{
613		WalSummarizerCtl->initialized = true;
614		WalSummarizerCtl->summarized_lsn = unsummarized_lsn;
615		WalSummarizerCtl->summarized_tli = unsummarized_tli;
616		WalSummarizerCtl->lsn_is_exact = should_make_exact;
617		WalSummarizerCtl->pending_lsn = unsummarized_lsn;
618	}
619	else
620		unsummarized_lsn = WalSummarizerCtl->summarized_lsn;
621
622	/* Also return the to the caller as required. */
623	if (tli != NULL)
624		*tli = WalSummarizerCtl->summarized_tli;
625	if (lsn_is_exact != NULL)
626		*lsn_is_exact = WalSummarizerCtl->lsn_is_exact;
627	LWLockRelease(WALSummarizerLock);
628
629	return unsummarized_lsn;

656/*
657 * Wait until WAL summarization reaches the given LSN, but time out with an
658 * error if the summarizer seems to be stick.
659 *
660 * Returns immediately if summarize_wal is turned off while we wait. Caller
661 * is expected to handle this case, if necessary.
662 */
663void
664WaitForWalSummarization(XLogRecPtr lsn)
665{
666	TimestampTz initial_time,
667				cycle_time,
668				current_time;
669	XLogRecPtr	prior_pending_lsn = InvalidXLogRecPtr;
670	int			deadcycles = 0;
671
672	initial_time = cycle_time = GetCurrentTimestamp();
673
674	while (1)
675	{
676		long		timeout_in_ms = 10000;
677		XLogRecPtr	summarized_lsn;
678		XLogRecPtr	pending_lsn;
679
680		CHECK_FOR_INTERRUPTS();
681
682		/* If WAL summarization is disabled while we're waiting, give up. */
683		if (!summarize_wal)
684			return;
685
686		/*
687		 * If the LSN summarized on disk has reached the target value, stop.
688		 */
689		LWLockAcquire(WALSummarizerLock, LW_SHARED);
690		summarized_lsn = WalSummarizerCtl->summarized_lsn;
691		pending_lsn = WalSummarizerCtl->pending_lsn;
692		LWLockRelease(WALSummarizerLock);
693
694		/* If WAL summarization has progressed sufficiently, stop waiting. */
695		if (summarized_lsn >= lsn)
696			break;
697
698		/* Recheck current time. */
699		current_time = GetCurrentTimestamp();
700
701		/* Have we finished the current cycle of waiting? */
702		if (TimestampDifferenceMilliseconds(cycle_time,
703											current_time) >= timeout_in_ms)
704		{
705			long		elapsed_seconds;
706
707			/* Begin new wait cycle. */
708			cycle_time = TimestampTzPlusMilliseconds(cycle_time,
709													 timeout_in_ms);
710
711			/*
712			 * Keep track of the number of cycles during which there has been
713			 * no progression of pending_lsn. If pending_lsn is not advancing,
714			 * that means that not only are no new files appearing on disk,
715			 * but we're not even incorporating new records into the in-memory
716			 * state.
717			 */
718			if (pending_lsn > prior_pending_lsn)
719			{
720				prior_pending_lsn = pending_lsn;
721				deadcycles = 0;
722			}
723			else
724				++deadcycles;
725
726			/*
727			 * If we've managed to wait for an entire minute without the WAL
728			 * summarizer absorbing a single WAL record, error out; probably
729			 * something is wrong.
730			 *
731			 * We could consider also erroring out if the summarizer is taking
732			 * too long to catch up, but it's not clear what rate of progress
733			 * would be acceptable and what would be too slow. So instead, we
734			 * just try to error out in the case where there's no progress at
735			 * all. That seems likely to catch a reasonable number of the
736			 * things that can go wrong in practice (e.g. the summarizer
737			 * process is completely hung, say because somebody hooked up a
738			 * debugger to it or something) without giving up too quickly when
739			 * the system is just slow.
740			 */
741			if (deadcycles >= 6)
742				ereport(ERROR,
743						(errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
744						 errmsg("WAL summarization is not progressing"),
745						 errdetail("Summarization is needed through %X/%X, but is stuck at %X/%X on disk and %X/%X in memory.",
746								   LSN_FORMAT_ARGS(lsn),
747								   LSN_FORMAT_ARGS(summarized_lsn),
748								   LSN_FORMAT_ARGS(pending_lsn))));
749
750
751			/*
752			 * Otherwise, just let the user know what's happening.
753			 */
754			elapsed_seconds =
755				TimestampDifferenceMilliseconds(initial_time,
756												current_time) / 1000;
757			ereport(WARNING,
758					(errcode(ERRCODE_OBJECT_NOT_IN_PREREQUISITE_STATE),
759					 errmsg_plural("still waiting for WAL summarization through %X/%X after %ld second",
760								   "still waiting for WAL summarization through %X/%X after %ld seconds",
761								   elapsed_seconds,
762								   LSN_FORMAT_ARGS(lsn),
763								   elapsed_seconds),
764					 errdetail("Summarization has reached %X/%X on disk and %X/%X in memory.",
765							   LSN_FORMAT_ARGS(summarized_lsn),
766							   LSN_FORMAT_ARGS(pending_lsn))));
767		}
768
769		/*
770		 * Align the wait time to prevent drift. This doesn't really matter,
771		 * but we'd like the warnings about how long we've been waiting to say
772		 * 10 seconds, 20 seconds, 30 seconds, 40 seconds ... without ever
773		 * drifting to something that is not a multiple of ten.
774		 */
775		timeout_in_ms -=
776			TimestampDifferenceMilliseconds(cycle_time, current_time);
777
778		/* Wait and see. */
779		ConditionVariableTimedSleep(&WalSummarizerCtl->summary_file_cv,
780									timeout_in_ms,
781									WAIT_EVENT_WAL_SUMMARY_READY);
782	}
783
784	ConditionVariableCancelSleep();
785}

799/*
800 * Get the latest LSN that is eligible to be summarized, and set *tli to the
801 * corresponding timeline.
802 */
803static XLogRecPtr
804GetLatestLSN(TimeLineID *tli)
805{
806	if (!RecoveryInProgress())
807	{
808		/* Don't summarize WAL before it's flushed. */
809		return GetFlushRecPtr(tli);
810	}
811	else
812	{
813		XLogRecPtr	flush_lsn;
814		TimeLineID	flush_tli;
815		XLogRecPtr	replay_lsn;
816		TimeLineID	replay_tli;
817		TimeLineID	insert_tli;
818
819		/*
820		 * After the insert TLI has been set and before the control file has
821		 * been updated to show the DB in production, RecoveryInProgress()
822		 * will return true, because it's not yet safe for all backends to
823		 * begin writing WAL. However, replay has already ceased, so from our
824		 * point of view, recovery is already over. We should summarize up to
825		 * where replay stopped and then prepare to resume at the start of the
826		 * insert timeline.
827		 */
828		if ((insert_tli = GetWALInsertionTimeLineIfSet()) != 0)
829		{
830			*tli = insert_tli;
831			return GetXLogReplayRecPtr(NULL);
832		}
833
834		/*
835		 * What we really want to know is how much WAL has been flushed to
836		 * disk, but the only flush position available is the one provided by
837		 * the walreceiver, which may not be running, because this could be
838		 * crash recovery or recovery via restore_command. So use either the
839		 * WAL receiver's flush position or the replay position, whichever is
840		 * further ahead, on the theory that if the WAL has been replayed then
841		 * it must also have been flushed to disk.
842		 */
843		flush_lsn = GetWalRcvFlushRecPtr(NULL, &flush_tli);
844		replay_lsn = GetXLogReplayRecPtr(&replay_tli);
845		if (flush_lsn > replay_lsn)
846		{
847			*tli = flush_tli;
848			return flush_lsn;
849		}
850		else
851		{
852			*tli = replay_tli;
853			return replay_lsn;
854		}
855	}
856}

885/*
886 * Summarize a range of WAL records on a single timeline.
887 *
888 * 'tli' is the timeline to be summarized.
889 *
890 * 'start_lsn' is the point at which we should start summarizing. If this
891 * value comes from the end LSN of the previous record as returned by the
892 * xlogreader machinery, 'exact' should be true; otherwise, 'exact' should
893 * be false, and this function will search forward for the start of a valid
894 * WAL record.
895 *
896 * 'switch_lsn' is the point at which we should switch to a later timeline,
897 * if we're summarizing a historic timeline.
898 *
899 * 'maximum_lsn' identifies the point beyond which we can't count on being
900 * able to read any more WAL. It should be the switch point when reading a
901 * historic timeline, or the most-recently-measured end of WAL when reading
902 * the current timeline.
903 *
904 * The return value is the LSN at which the WAL summary actually ends. Most
905 * often, a summary file ends because we notice that a checkpoint has
906 * occurred and reach the redo pointer of that checkpoint, but sometimes
907 * we stop for other reasons, such as a timeline switch.
908 */
909static XLogRecPtr
910SummarizeWAL(TimeLineID tli, XLogRecPtr start_lsn, bool exact,
911			 XLogRecPtr switch_lsn, XLogRecPtr maximum_lsn)
912{
913	SummarizerReadLocalXLogPrivate *private_data;
914	XLogReaderState *xlogreader;
915	XLogRecPtr	summary_start_lsn;
916	XLogRecPtr	summary_end_lsn = switch_lsn;
917	char		temp_path[MAXPGPATH];
918	char		final_path[MAXPGPATH];
919	WalSummaryIO io;
920	BlockRefTable *brtab = CreateEmptyBlockRefTable();
921	bool		fast_forward = true;
922
923	/* Initialize private data for xlogreader. */
924	private_data = (SummarizerReadLocalXLogPrivate *)
925		palloc0(sizeof(SummarizerReadLocalXLogPrivate));
926	private_data->tli = tli;
927	private_data->historic = !XLogRecPtrIsInvalid(switch_lsn);
928	private_data->read_upto = maximum_lsn;
929
930	/* Create xlogreader. */
931	xlogreader = XLogReaderAllocate(wal_segment_size, NULL,
932									XL_ROUTINE(.page_read = &summarizer_read_local_xlog_page,
933											   .segment_open = &wal_segment_open,
934											   .segment_close = &wal_segment_close),
935									private_data);
936	if (xlogreader == NULL)
937		ereport(ERROR,
938				(errcode(ERRCODE_OUT_OF_MEMORY),
939				 errmsg("out of memory"),
940				 errdetail("Failed while allocating a WAL reading processor.")));
941
942	/*
943	 * When exact = false, we're starting from an arbitrary point in the WAL
944	 * and must search forward for the start of the next record.
945	 *
946	 * When exact = true, start_lsn should be either the LSN where a record
947	 * begins, or the LSN of a page where the page header is immediately
948	 * followed by the start of a new record. XLogBeginRead should tolerate
949	 * either case.
950	 *
951	 * We need to allow for both cases because the behavior of xlogreader
952	 * varies. When a record spans two or more xlog pages, the ending LSN
953	 * reported by xlogreader will be the starting LSN of the following
954	 * record, but when an xlog page boundary falls between two records, the
955	 * end LSN for the first will be reported as the first byte of the
956	 * following page. We can't know until we read that page how large the
957	 * header will be, but we'll have to skip over it to find the next record.
958	 */
959	if (exact)
960	{
961		/*
962		 * Even if start_lsn is the beginning of a page rather than the
963		 * beginning of the first record on that page, we should still use it
964		 * as the start LSN for the summary file. That's because we detect
965		 * missing summary files by looking for cases where the end LSN of one
966		 * file is less than the start LSN of the next file. When only a page
967		 * header is skipped, nothing has been missed.
968		 */
969		XLogBeginRead(xlogreader, start_lsn);
970		summary_start_lsn = start_lsn;
971	}
972	else
973	{
974		summary_start_lsn = XLogFindNextRecord(xlogreader, start_lsn);
975		if (XLogRecPtrIsInvalid(summary_start_lsn))
976		{
977			/*
978			 * If we hit end-of-WAL while trying to find the next valid
979			 * record, we must be on a historic timeline that has no valid
980			 * records that begin after start_lsn and before end of WAL.
981			 */
982			if (private_data->end_of_wal)
983			{
984				ereport(DEBUG1,
985						errmsg_internal("could not read WAL from timeline %u at %X/%X: end of WAL at %X/%X",
986										tli,
987										LSN_FORMAT_ARGS(start_lsn),
988										LSN_FORMAT_ARGS(private_data->read_upto)));
989
990				/*
991				 * The timeline ends at or after start_lsn, without containing
992				 * any records. Thus, we must make sure the main loop does not
993				 * iterate. If start_lsn is the end of the timeline, then we
994				 * won't actually emit an empty summary file, but otherwise,
995				 * we must, to capture the fact that the LSN range in question
996				 * contains no interesting WAL records.
997				 */
998				summary_start_lsn = start_lsn;
999				summary_end_lsn = private_data->read_upto;
1000				switch_lsn = xlogreader->EndRecPtr;
1001			}
1002			else
1003				ereport(ERROR,
1004						(errmsg("could not find a valid record after %X/%X",
1005								LSN_FORMAT_ARGS(start_lsn))));
1006		}
1007
1008		/* We shouldn't go backward. */
1009		Assert(summary_start_lsn >= start_lsn);
1010	}
1011
1012	/*
1013	 * Main loop: read xlog records one by one.
1014	 */
1015	while (1)
1016	{
1017		int			block_id;
1018		char	   *errormsg;
1019		XLogRecord *record;
1020		uint8		rmid;
1021
1022		ProcessWalSummarizerInterrupts();
1023
1024		/* We shouldn't go backward. */
1025		Assert(summary_start_lsn <= xlogreader->EndRecPtr);
1026
1027		/* Now read the next record. */
1028		record = XLogReadRecord(xlogreader, &errormsg);
1029		if (record == NULL)
1030		{
1031			if (private_data->end_of_wal)
1032			{
1033				/*
1034				 * This timeline must be historic and must end before we were
1035				 * able to read a complete record.
1036				 */
1037				ereport(DEBUG1,
1038						errmsg_internal("could not read WAL from timeline %u at %X/%X: end of WAL at %X/%X",
1039										tli,
1040										LSN_FORMAT_ARGS(xlogreader->EndRecPtr),
1041										LSN_FORMAT_ARGS(private_data->read_upto)));
1042				/* Summary ends at end of WAL. */
1043				summary_end_lsn = private_data->read_upto;
1044				break;
1045			}
1046			if (errormsg)
1047				ereport(ERROR,
1048						(errcode_for_file_access(),
1049						 errmsg("could not read WAL from timeline %u at %X/%X: %s",
1050								tli, LSN_FORMAT_ARGS(xlogreader->EndRecPtr),
1051								errormsg)));
1052			else
1053				ereport(ERROR,
1054						(errcode_for_file_access(),
1055						 errmsg("could not read WAL from timeline %u at %X/%X",
1056								tli, LSN_FORMAT_ARGS(xlogreader->EndRecPtr))));
1057		}
1058
1059		/* We shouldn't go backward. */
1060		Assert(summary_start_lsn <= xlogreader->EndRecPtr);
1061
1062		if (!XLogRecPtrIsInvalid(switch_lsn) &&
1063			xlogreader->ReadRecPtr >= switch_lsn)
1064		{
1065			/*
1066			 * Whoops! We've read a record that *starts* after the switch LSN,
1067			 * contrary to our goal of reading only until we hit the first
1068			 * record that ends at or after the switch LSN. Pretend we didn't
1069			 * read it after all by bailing out of this loop right here,
1070			 * before we do anything with this record.
1071			 *
1072			 * This can happen because the last record before the switch LSN
1073			 * might be continued across multiple pages, and then we might
1074			 * come to a page with XLP_FIRST_IS_OVERWRITE_CONTRECORD set. In
1075			 * that case, the record that was continued across multiple pages
1076			 * is incomplete and will be disregarded, and the read will
1077			 * restart from the beginning of the page that is flagged
1078			 * XLP_FIRST_IS_OVERWRITE_CONTRECORD.
1079			 *
1080			 * If this case occurs, we can fairly say that the current summary
1081			 * file ends at the switch LSN exactly. The first record on the
1082			 * page marked XLP_FIRST_IS_OVERWRITE_CONTRECORD will be
1083			 * discovered when generating the next summary file.
1084			 */
1085			summary_end_lsn = switch_lsn;
1086			break;
1087		}
1088
1089		/*
1090		 * Certain types of records require special handling. Redo points and
1091		 * shutdown checkpoints trigger creation of new summary files and can
1092		 * also cause us to enter or exit "fast forward" mode. Other types of
1093		 * records can require special updates to the block reference table.
1094		 */
1095		rmid = XLogRecGetRmid(xlogreader);
1096		if (rmid == RM_XLOG_ID)
1097		{
1098			bool		new_fast_forward;
1099
1100			/*
1101			 * If we've already processed some WAL records when we hit a redo
1102			 * point or shutdown checkpoint, then we stop summarization before
1103			 * including this record in the current file, so that it will be
1104			 * the first record in the next file.
1105			 *
1106			 * When we hit one of those record types as the first record in a
1107			 * file, we adjust our notion of whether we're fast-forwarding.
1108			 * Any WAL generated with wal_level=minimal must be skipped
1109			 * without actually generating any summary file, because an
1110			 * incremental backup that crosses such WAL would be unsafe.
1111			 */
1112			if (SummarizeXlogRecord(xlogreader, &new_fast_forward))
1113			{
1114				if (xlogreader->ReadRecPtr > summary_start_lsn)
1115				{
1116					summary_end_lsn = xlogreader->ReadRecPtr;
1117					break;
1118				}
1119				else
1120					fast_forward = new_fast_forward;
1121			}
1122		}
1123		else if (!fast_forward)
1124		{
1125			/*
1126			 * This switch handles record types that require extra updates to
1127			 * the contents of the block reference table.
1128			 */
1129			switch (rmid)
1130			{
1131				case RM_DBASE_ID:
1132					SummarizeDbaseRecord(xlogreader, brtab);
1133					break;
1134				case RM_SMGR_ID:
1135					SummarizeSmgrRecord(xlogreader, brtab);
1136					break;
1137				case RM_XACT_ID:
1138					SummarizeXactRecord(xlogreader, brtab);
1139					break;
1140			}
1141		}
1142
1143		/*
1144		 * If we're in fast-forward mode, we don't really need to do anything.
1145		 * Otherwise, feed block references from xlog record to block
1146		 * reference table.
1147		 */
1148		if (!fast_forward)
1149		{
1150			for (block_id = 0; block_id <= XLogRecMaxBlockId(xlogreader);
1151				 block_id++)
1152			{
1153				RelFileLocator rlocator;
1154				ForkNumber	forknum;
1155				BlockNumber blocknum;
1156
1157				if (!XLogRecGetBlockTagExtended(xlogreader, block_id, &rlocator,
1158												&forknum, &blocknum, NULL))
1159					continue;
1160
1161				/*
1162				 * As we do elsewhere, ignore the FSM fork, because it's not
1163				 * fully WAL-logged.
1164				 */
1165				if (forknum != FSM_FORKNUM)
1166					BlockRefTableMarkBlockModified(brtab, &rlocator, forknum,
1167												   blocknum);
1168			}
1169		}
1170
1171		/* Update our notion of where this summary file ends. */
1172		summary_end_lsn = xlogreader->EndRecPtr;
1173
1174		/* Also update shared memory. */
1175		LWLockAcquire(WALSummarizerLock, LW_EXCLUSIVE);
1176		Assert(summary_end_lsn >= WalSummarizerCtl->summarized_lsn);
1177		WalSummarizerCtl->pending_lsn = summary_end_lsn;
1178		LWLockRelease(WALSummarizerLock);
1179
1180		/*
1181		 * If we have a switch LSN and have reached it, stop before reading
1182		 * the next record.
1183		 */
1184		if (!XLogRecPtrIsInvalid(switch_lsn) &&
1185			xlogreader->EndRecPtr >= switch_lsn)
1186			break;
1187	}
1188
1189	/* Destroy xlogreader. */
1190	pfree(xlogreader->private_data);
1191	XLogReaderFree(xlogreader);
1192
1193	/*
1194	 * If a timeline switch occurs, we may fail to make any progress at all
1195	 * before exiting the loop above. If that happens, we don't write a WAL
1196	 * summary file at all. We can also skip writing a file if we're in
1197	 * fast-forward mode.
1198	 */
1199	if (summary_end_lsn > summary_start_lsn && !fast_forward)
1200	{
1201		/* Generate temporary and final path name. */
1202		snprintf(temp_path, MAXPGPATH,
1203				 XLOGDIR "/summaries/temp.summary");
1204		snprintf(final_path, MAXPGPATH,
1205				 XLOGDIR "/summaries/%08X%08X%08X%08X%08X.summary",
1206				 tli,
1207				 LSN_FORMAT_ARGS(summary_start_lsn),
1208				 LSN_FORMAT_ARGS(summary_end_lsn));
1209
1210		/* Open the temporary file for writing. */
1211		io.filepos = 0;
1212		io.file = PathNameOpenFile(temp_path, O_WRONLY | O_CREAT | O_TRUNC);
1213		if (io.file < 0)
1214			ereport(ERROR,
1215					(errcode_for_file_access(),
1216					 errmsg("could not create file \"%s\": %m", temp_path)));
1217
1218		/* Write the data. */
1219		WriteBlockRefTable(brtab, WriteWalSummary, &io);
1220
1221		/* Close temporary file and shut down xlogreader. */
1222		FileClose(io.file);
1223
1224		/* Tell the user what we did. */
1225		ereport(DEBUG1,
1226				errmsg_internal("summarized WAL on TLI %u from %X/%X to %X/%X",
1227								tli,
1228								LSN_FORMAT_ARGS(summary_start_lsn),
1229								LSN_FORMAT_ARGS(summary_end_lsn)));
1230
1231		/* Durably rename the new summary into place. */
1232		durable_rename(temp_path, final_path, ERROR);
1233	}
1234
1235	/* If we skipped a non-zero amount of WAL, log a debug message. */
1236	if (summary_end_lsn > summary_start_lsn && fast_forward)
1237		ereport(DEBUG1,
1238				errmsg_internal("skipped summarizing WAL on TLI %u from %X/%X to %X/%X",
1239								tli,
1240								LSN_FORMAT_ARGS(summary_start_lsn),
1241								LSN_FORMAT_ARGS(summary_end_lsn)));
1242
1243	return summary_end_lsn;
1244}

1247 * Special handling for WAL records with RM_DBASE_ID.
1248 */
1249static void
1250SummarizeDbaseRecord(XLogReaderState *xlogreader, BlockRefTable *brtab)
1251{
1252	uint8		info = XLogRecGetInfo(xlogreader) & ~XLR_INFO_MASK;
1253
1254	/*
1255	 * We use relfilenode zero for a given database OID and tablespace OID to
1256	 * indicate that all relations with that pair of IDs have been recreated
1257	 * if they exist at all. Effectively, we're setting a limit block of 0 for
1258	 * all such relfilenodes.
1259	 *
1260	 * Technically, this special handling is only needed in the case of
1261	 * XLOG_DBASE_CREATE_FILE_COPY, because that can create a whole bunch of
1262	 * relation files in a directory without logging anything specific to each
1263	 * one. If we didn't mark the whole DB OID/TS OID combination in some way,
1264	 * then a tablespace that was dropped after the reference backup and
1265	 * recreated using the FILE_COPY method prior to the incremental backup
1266	 * would look just like one that was never touched at all, which would be
1267	 * catastrophic.
1268	 *
1269	 * But it seems best to adopt this treatment for all records that drop or
1270	 * create a DB OID/TS OID combination. That's similar to how we treat the
1271	 * limit block for individual relations, and it's an extra layer of safety
1272	 * here. We can never lose data by marking more stuff as needing to be
1273	 * backed up in full.
1274	 */
1275	if (info == XLOG_DBASE_CREATE_FILE_COPY)
1276	{
1277		xl_dbase_create_file_copy_rec *xlrec;
1278		RelFileLocator rlocator;
1279
1280		xlrec =
1281			(xl_dbase_create_file_copy_rec *) XLogRecGetData(xlogreader);
1282		rlocator.spcOid = xlrec->tablespace_id;
1283		rlocator.dbOid = xlrec->db_id;
1284		rlocator.relNumber = 0;
1285		BlockRefTableSetLimitBlock(brtab, &rlocator, MAIN_FORKNUM, 0);
1286	}
1287	else if (info == XLOG_DBASE_CREATE_WAL_LOG)
1288	{
1289		xl_dbase_create_wal_log_rec *xlrec;
1290		RelFileLocator rlocator;
1291
1292		xlrec = (xl_dbase_create_wal_log_rec *) XLogRecGetData(xlogreader);
1293		rlocator.spcOid = xlrec->tablespace_id;
1294		rlocator.dbOid = xlrec->db_id;
1295		rlocator.relNumber = 0;
1296		BlockRefTableSetLimitBlock(brtab, &rlocator, MAIN_FORKNUM, 0);
1297	}
1298	else if (info == XLOG_DBASE_DROP)
1299	{
1300		xl_dbase_drop_rec *xlrec;
1301		RelFileLocator rlocator;
1302		int			i;
1303
1304		xlrec = (xl_dbase_drop_rec *) XLogRecGetData(xlogreader);
1305		rlocator.dbOid = xlrec->db_id;
1306		rlocator.relNumber = 0;
1307		for (i = 0; i < xlrec->ntablespaces; ++i)
1308		{
1309			rlocator.spcOid = xlrec->tablespace_ids[i];
1310			BlockRefTableSetLimitBlock(brtab, &rlocator, MAIN_FORKNUM, 0);
1311		}
1312	}
1313}

1315/*
1316 * Special handling for WAL records with RM_SMGR_ID.
1317 */
1318static void
1319SummarizeSmgrRecord(XLogReaderState *xlogreader, BlockRefTable *brtab)
1320{
1321	uint8		info = XLogRecGetInfo(xlogreader) & ~XLR_INFO_MASK;
1322
1323	if (info == XLOG_SMGR_CREATE)
1324	{
1325		xl_smgr_create *xlrec;
1326
1327		/*
1328		 * If a new relation fork is created on disk, there is no point
1329		 * tracking anything about which blocks have been modified, because
1330		 * the whole thing will be new. Hence, set the limit block for this
1331		 * fork to 0.
1332		 *
1333		 * Ignore the FSM fork, which is not fully WAL-logged.
1334		 */
1335		xlrec = (xl_smgr_create *) XLogRecGetData(xlogreader);
1336
1337		if (xlrec->forkNum != FSM_FORKNUM)
1338			BlockRefTableSetLimitBlock(brtab, &xlrec->rlocator,
1339									   xlrec->forkNum, 0);
1340	}
1341	else if (info == XLOG_SMGR_TRUNCATE)
1342	{
1343		xl_smgr_truncate *xlrec;
1344
1345		xlrec = (xl_smgr_truncate *) XLogRecGetData(xlogreader);
1346
1347		/*
1348		 * If a relation fork is truncated on disk, there is no point in
1349		 * tracking anything about block modifications beyond the truncation
1350		 * point.
1351		 *
1352		 * We ignore SMGR_TRUNCATE_FSM here because the FSM isn't fully
1353		 * WAL-logged and thus we can't track modified blocks for it anyway.
1354		 */
1355		if ((xlrec->flags & SMGR_TRUNCATE_HEAP) != 0)
1356			BlockRefTableSetLimitBlock(brtab, &xlrec->rlocator,
1357									   MAIN_FORKNUM, xlrec->blkno);
1358		if ((xlrec->flags & SMGR_TRUNCATE_VM) != 0)
1359			BlockRefTableSetLimitBlock(brtab, &xlrec->rlocator,
1360									   VISIBILITYMAP_FORKNUM,
1361									   visibilitymap_truncation_length(xlrec->blkno));
1362	}
1363}

1420/*
1421 * Special handling for WAL records with RM_XLOG_ID.
1422 *
1423 * The return value is true if WAL summarization should stop before this
1424 * record and false otherwise. When the return value is true,
1425 * *new_fast_forward indicates whether future processing should be done
1426 * in fast forward mode (i.e. read WAL without emitting summaries) or not.
1427 */
1428static bool
1429SummarizeXlogRecord(XLogReaderState *xlogreader, bool *new_fast_forward)
1430{
1431	uint8		info = XLogRecGetInfo(xlogreader) & ~XLR_INFO_MASK;
1432	int			record_wal_level;
1433
1434	if (info == XLOG_CHECKPOINT_REDO)
1435	{
1436		/* Payload is wal_level at the time record was written. */
1437		memcpy(&record_wal_level, XLogRecGetData(xlogreader), sizeof(int));
1438	}
1439	else if (info == XLOG_CHECKPOINT_SHUTDOWN)
1440	{
1441		CheckPoint	rec_ckpt;
1442
1443		/* Extract wal_level at time record was written from payload. */
1444		memcpy(&rec_ckpt, XLogRecGetData(xlogreader), sizeof(CheckPoint));
1445		record_wal_level = rec_ckpt.wal_level;
1446	}
1447	else if (info == XLOG_PARAMETER_CHANGE)
1448	{
1449		xl_parameter_change xlrec;
1450
1451		/* Extract wal_level at time record was written from payload. */
1452		memcpy(&xlrec, XLogRecGetData(xlogreader),
1453			   sizeof(xl_parameter_change));
1454		record_wal_level = xlrec.wal_level;
1455	}
1456	else if (info == XLOG_END_OF_RECOVERY)
1457	{
1458		xl_end_of_recovery xlrec;
1459
1460		/* Extract wal_level at time record was written from payload. */
1461		memcpy(&xlrec, XLogRecGetData(xlogreader), sizeof(xl_end_of_recovery));
1462		record_wal_level = xlrec.wal_level;
1463	}
1464	else
1465	{
1466		/* No special handling required. Return false. */
1467		return false;
1468	}
1469
1470	/*
1471	 * Redo can only begin at an XLOG_CHECKPOINT_REDO or
1472	 * XLOG_CHECKPOINT_SHUTDOWN record, so we want WAL summarization to begin
1473	 * at those points. Hence, when those records are encountered, return
1474	 * true, so that we stop just before summarizing either of those records.
1475	 *
1476	 * We also reach here if we just saw XLOG_END_OF_RECOVERY or
1477	 * XLOG_PARAMETER_CHANGE. These are not places where recovery can start,
1478	 * but they're still relevant here. A new timeline can begin with
1479	 * XLOG_END_OF_RECOVERY, so we need to confirm the WAL level at that
1480	 * point; and a restart can provoke XLOG_PARAMETER_CHANGE after an
1481	 * intervening change to postgresql.conf, which might force us to stop
1482	 * summarizing.
1483	 */
1484	*new_fast_forward = (record_wal_level == WAL_LEVEL_MINIMAL);
1485	return true;
1486}

1488/*
1489 * Similar to read_local_xlog_page, but limited to read from one particular
1490 * timeline. If the end of WAL is reached, it will wait for more if reading
1491 * from the current timeline, or give up if reading from a historic timeline.
1492 * In the latter case, it will also set private_data->end_of_wal = true.
1493 *
1494 * Caller must set private_data->tli to the TLI of interest,
1495 * private_data->read_upto to the lowest LSN that is not known to be safe
1496 * to read on that timeline, and private_data->historic to true if and only
1497 * if the timeline is not the current timeline. This function will update
1498 * private_data->read_upto and private_data->historic if more WAL appears
1499 * on the current timeline or if the current timeline becomes historic.
1500 */
1501static int
1502summarizer_read_local_xlog_page(XLogReaderState *state,
1503								XLogRecPtr targetPagePtr, int reqLen,
1504								XLogRecPtr targetRecPtr, char *cur_page)
1505{
1506	int			count;
1507	WALReadError errinfo;
1508	SummarizerReadLocalXLogPrivate *private_data;
1509
1510	ProcessWalSummarizerInterrupts();
1511
1512	private_data = (SummarizerReadLocalXLogPrivate *)
1513		state->private_data;
1514
1515	while (1)
1516	{
1517		if (targetPagePtr + XLOG_BLCKSZ <= private_data->read_upto)
1518		{
1519			/*
1520			 * more than one block available; read only that block, have
1521			 * caller come back if they need more.
1522			 */
1523			count = XLOG_BLCKSZ;
1524			break;
1525		}
1526		else if (targetPagePtr + reqLen > private_data->read_upto)
1527		{
1528			/* We don't seem to have enough data. */
1529			if (private_data->historic)
1530			{
1531				/*
1532				 * This is a historic timeline, so there will never be any
1533				 * more data than we have currently.
1534				 */
1535				private_data->end_of_wal = true;
1536				return -1;
1537			}
1538			else
1539			{
1540				XLogRecPtr	latest_lsn;
1541				TimeLineID	latest_tli;
1542
1543				/*
1544				 * This is - or at least was up until very recently - the
1545				 * current timeline, so more data might show up.  Delay here
1546				 * so we don't tight-loop.
1547				 */
1548				ProcessWalSummarizerInterrupts();
1549				summarizer_wait_for_wal();
1550
1551				/* Recheck end-of-WAL. */
1552				latest_lsn = GetLatestLSN(&latest_tli);
1553				if (private_data->tli == latest_tli)
1554				{
1555					/* Still the current timeline, update max LSN. */
1556					Assert(latest_lsn >= private_data->read_upto);
1557					private_data->read_upto = latest_lsn;
1558				}
1559				else
1560				{
1561					List	   *tles = readTimeLineHistory(latest_tli);
1562					XLogRecPtr	switchpoint;
1563
1564					/*
1565					 * The timeline we're scanning is no longer the latest
1566					 * one. Figure out when it ended.
1567					 */
1568					private_data->historic = true;
1569					switchpoint = tliSwitchPoint(private_data->tli, tles,
1570												 NULL);
1571
1572					/*
1573					 * Allow reads up to exactly the switch point.
1574					 *
1575					 * It's possible that this will cause read_upto to move
1576					 * backwards, because we might have been promoted before
1577					 * reaching the end of the previous timeline. In that
1578					 * case, the next loop iteration will likely conclude that
1579					 * we've reached end of WAL.
1580					 */
1581					private_data->read_upto = switchpoint;
1582
1583					/* Debugging output. */
1584					ereport(DEBUG1,
1585							errmsg_internal("timeline %u became historic, can read up to %X/%X",
1586											private_data->tli, LSN_FORMAT_ARGS(private_data->read_upto)));
1587				}
1588
1589				/* Go around and try again. */
1590			}
1591		}
1592		else
1593		{
1594			/* enough bytes available to satisfy the request */
1595			count = private_data->read_upto - targetPagePtr;
1596			break;
1597		}
1598	}
1599
1600	if (!WALRead(state, cur_page, targetPagePtr, count,
1601				 private_data->tli, &errinfo))
1602		WALReadRaiseError(&errinfo);
1603
1604	/* Track that we read a page, for sleep time calculation. */
1605	++pages_read_since_last_sleep;
1606
1607	/* number of valid bytes in the buffer */
1608	return count;
1609}

1611/*
1612 * Sleep for long enough that we believe it's likely that more WAL will
1613 * be available afterwards.
1614 */
1615static void
1616summarizer_wait_for_wal(void)
1617{
1618	if (pages_read_since_last_sleep == 0)
1619	{
1620		/*
1621		 * No pages were read since the last sleep, so double the sleep time,
1622		 * but not beyond the maximum allowable value.
1623		 */
1624		sleep_quanta = Min(sleep_quanta * 2, MAX_SLEEP_QUANTA);
1625	}
1626	else if (pages_read_since_last_sleep > 1)
1627	{
1628		/*
1629		 * Multiple pages were read since the last sleep, so reduce the sleep
1630		 * time.
1631		 *
1632		 * A large burst of activity should be able to quickly reduce the
1633		 * sleep time to the minimum, but we don't want a handful of extra WAL
1634		 * records to provoke a strong reaction. We choose to reduce the sleep
1635		 * time by 1 quantum for each page read beyond the first, which is a
1636		 * fairly arbitrary way of trying to be reactive without overreacting.
1637		 */
1638		if (pages_read_since_last_sleep > sleep_quanta - 1)
1639			sleep_quanta = 1;
1640		else
1641			sleep_quanta -= pages_read_since_last_sleep;
1642	}
1643
1644	/* Report pending statistics to the cumulative stats system. */
1645	pgstat_report_wal(false);
1646
1647	/* OK, now sleep. */
1648	(void) WaitLatch(MyLatch,
1649					 WL_LATCH_SET | WL_TIMEOUT | WL_EXIT_ON_PM_DEATH,
1650					 sleep_quanta * MS_PER_SLEEP_QUANTUM,
1651					 WAIT_EVENT_WAL_SUMMARIZER_WAL);
1652	ResetLatch(MyLatch);
1653
1654	/* Reset count of pages read. */
1655	pages_read_since_last_sleep = 0;
1656}

1658/*
1659 * Remove WAL summaries whose mtimes are older than wal_summary_keep_time.
1660 */
1661static void
1662MaybeRemoveOldWalSummaries(void)
1663{
1664	XLogRecPtr	redo_pointer = GetRedoRecPtr();
1665	List	   *wslist;
1666	time_t		cutoff_time;
1667
1668	/* If WAL summary removal is disabled, don't do anything. */
1669	if (wal_summary_keep_time == 0)
1670		return;
1671
1672	/*
1673	 * If the redo pointer has not advanced, don't do anything.
1674	 *
1675	 * This has the effect that we only try to remove old WAL summary files
1676	 * once per checkpoint cycle.
1677	 */
1678	if (redo_pointer == redo_pointer_at_last_summary_removal)
1679		return;
1680	redo_pointer_at_last_summary_removal = redo_pointer;
1681
1682	/*
1683	 * Files should only be removed if the last modification time precedes the
1684	 * cutoff time we compute here.
1685	 */
1686	cutoff_time = time(NULL) - wal_summary_keep_time * SECS_PER_MINUTE;
1687
1688	/* Get all the summaries that currently exist. */
1689	wslist = GetWalSummaries(0, InvalidXLogRecPtr, InvalidXLogRecPtr);
1690
1691	/* Loop until all summaries have been considered for removal. */
1692	while (wslist != NIL)
1693	{
1694		ListCell   *lc;
1695		XLogSegNo	oldest_segno;
1696		XLogRecPtr	oldest_lsn = InvalidXLogRecPtr;
1697		TimeLineID	selected_tli;
1698
1699		ProcessWalSummarizerInterrupts();
1700
1701		/*
1702		 * Pick a timeline for which some summary files still exist on disk,
1703		 * and find the oldest LSN that still exists on disk for that
1704		 * timeline.
1705		 */
1706		selected_tli = ((WalSummaryFile *) linitial(wslist))->tli;
1707		oldest_segno = XLogGetOldestSegno(selected_tli);
1708		if (oldest_segno != 0)
1709			XLogSegNoOffsetToRecPtr(oldest_segno, 0, wal_segment_size,
1710									oldest_lsn);
1711
1712
1713		/* Consider each WAL file on the selected timeline in turn. */
1714		foreach(lc, wslist)
1715		{
1716			WalSummaryFile *ws = lfirst(lc);
1717
1718			ProcessWalSummarizerInterrupts();
1719
1720			/* If it's not on this timeline, it's not time to consider it. */
1721			if (selected_tli != ws->tli)
1722				continue;
1723
1724			/*
1725			 * If the WAL doesn't exist any more, we can remove it if the file
1726			 * modification time is old enough.
1727			 */
1728			if (XLogRecPtrIsInvalid(oldest_lsn) || ws->end_lsn <= oldest_lsn)
1729				RemoveWalSummaryIfOlderThan(ws, cutoff_time);
1730
1731			/*
1732			 * Whether we removed the file or not, we need not consider it
1733			 * again.
1734			 */
1735			wslist = foreach_delete_current(wslist, lc);
1736			pfree(ws);
1737		}
1738	}
