159/*
160 * Every now and then create a new LLVMContextRef. Unfortunately, during every
161 * round of inlining, types may "leak" (they can still be found/used via the
162 * context, but new types will be created the next time in inlining is
163 * performed). To prevent that from slowly accumulating problematic amounts of
164 * memory, recreate the LLVMContextRef we use. We don't want to do so too
165 * often, as that implies some overhead (particularly re-loading the module
166 * summaries / modules is fairly expensive). A future TODO would be to make
167 * this more finegrained and only drop/recreate the LLVMContextRef when we know
168 * there has been inlining. If we can get the size of the context from LLVM
169 * then that might be a better way to determine when to drop/recreate rather
170 * then the usagecount heuristic currently employed.
171 */
172static void
173llvm_recreate_llvm_context(void)
174{
175	if (!llvm_context)
176		elog(ERROR, "Trying to recreate a non-existing context");
177
178	/*
179	 * We can only safely recreate the LLVM context if no other code is being
180	 * JITed, otherwise we'd release the types in use for that.
181	 */
182	if (llvm_jit_context_in_use_count > 0)
183	{
184		llvm_llvm_context_reuse_count++;
185		return;
186	}
187
188	if (llvm_llvm_context_reuse_count <= LLVMJIT_LLVM_CONTEXT_REUSE_MAX)
189	{
190		llvm_llvm_context_reuse_count++;
191		return;
192	}
193
194	/*
195	 * Need to reset the modules that the inlining code caches before
196	 * disposing of the context. LLVM modules exist within a specific LLVM
197	 * context, therefore disposing of the context before resetting the cache
198	 * would lead to dangling pointers to modules.
199	 */
200	llvm_inline_reset_caches();
201
202	LLVMContextDispose(llvm_context);
203	llvm_context = LLVMContextCreate();
204	llvm_llvm_context_reuse_count = 0;
205
206	/*
207	 * Re-build cached type information, so code generation code can rely on
208	 * that information to be present (also prevents the variables to be
209	 * dangling references).
210	 */
211	llvm_create_types();
212}

215/*
216 * Create a context for JITing work.
217 *
218 * The context, including subsidiary resources, will be cleaned up either when
219 * the context is explicitly released, or when the lifetime of
220 * CurrentResourceOwner ends (usually the end of the current [sub]xact).
221 */
222LLVMJitContext *
223llvm_create_context(int jitFlags)
224{
225	LLVMJitContext *context;
226
227	llvm_assert_in_fatal_section();
228
229	llvm_session_initialize();
230
231	llvm_recreate_llvm_context();
232
233	ResourceOwnerEnlarge(CurrentResourceOwner);
234
235	context = MemoryContextAllocZero(TopMemoryContext,
236									 sizeof(LLVMJitContext));
237	context->base.flags = jitFlags;
238
239	/* ensure cleanup */
240	context->resowner = CurrentResourceOwner;
241	ResourceOwnerRememberJIT(CurrentResourceOwner, context);
242
243	llvm_jit_context_in_use_count++;
244
245	return context;
246}

248/*
249 * Release resources required by one llvm context.
250 */
251static void
252llvm_release_context(JitContext *context)
253{
254	LLVMJitContext *llvm_jit_context = (LLVMJitContext *) context;
255	ListCell   *lc;
256
257	/*
258	 * Consider as cleaned up even if we skip doing so below, that way we can
259	 * verify the tracking is correct (see llvm_shutdown()).
260	 */
261	llvm_jit_context_in_use_count--;
262
263	/*
264	 * When this backend is exiting, don't clean up LLVM. As an error might
265	 * have occurred from within LLVM, we do not want to risk reentering. All
266	 * resource cleanup is going to happen through process exit.
267	 */
268	if (proc_exit_inprogress)
269		return;
270
271	llvm_enter_fatal_on_oom();
272
273	if (llvm_jit_context->module)
274	{
275		LLVMDisposeModule(llvm_jit_context->module);
276		llvm_jit_context->module = NULL;
277	}
278
279	foreach(lc, llvm_jit_context->handles)
280	{
281		LLVMJitHandle *jit_handle = (LLVMJitHandle *) lfirst(lc);
282
283		{
284			LLVMOrcExecutionSessionRef ee;
285			LLVMOrcSymbolStringPoolRef sp;
286
287			LLVMOrcResourceTrackerRemove(jit_handle->resource_tracker);
288			LLVMOrcReleaseResourceTracker(jit_handle->resource_tracker);
289
290			/*
291			 * Without triggering cleanup of the string pool, we'd leak
292			 * memory. It'd be sufficient to do this far less often, but in
293			 * experiments the required time was small enough to just always
294			 * do it.
295			 */
296			ee = LLVMOrcLLJITGetExecutionSession(jit_handle->lljit);
297			sp = LLVMOrcExecutionSessionGetSymbolStringPool(ee);
298			LLVMOrcSymbolStringPoolClearDeadEntries(sp);
299		}
300
301		pfree(jit_handle);
302	}
303	list_free(llvm_jit_context->handles);
304	llvm_jit_context->handles = NIL;
305
306	llvm_leave_fatal_on_oom();
307
308	if (llvm_jit_context->resowner)
309		ResourceOwnerForgetJIT(llvm_jit_context->resowner, llvm_jit_context);
310}

312/*
313 * Return module which may be modified, e.g. by creating new functions.
314 */
315LLVMModuleRef
316llvm_mutable_module(LLVMJitContext *context)
317{
318	llvm_assert_in_fatal_section();
319
320	/*
321	 * If there's no in-progress module, create a new one.
322	 */
323	if (!context->module)
324	{
325		context->compiled = false;
326		context->module_generation = llvm_generation++;
327		context->module = LLVMModuleCreateWithNameInContext("pg", llvm_context);
328		LLVMSetTarget(context->module, llvm_triple);
329		LLVMSetDataLayout(context->module, llvm_layout);
330	}
331
332	return context->module;
333}

335/*
336 * Expand function name to be non-conflicting. This should be used by code
337 * generating code, when adding new externally visible function definitions to
338 * a Module.
339 */
340char *
341llvm_expand_funcname(struct LLVMJitContext *context, const char *basename)
342{
343	Assert(context->module != NULL);
344
345	context->base.instr.created_functions++;
346
347	/*
348	 * Previously we used dots to separate, but turns out some tools, e.g.
349	 * GDB, don't like that and truncate name.
350	 */
351	return psprintf("%s_%zu_%d",
352					basename,
353					context->module_generation,
354					context->counter++);
355}

357/*
358 * Return pointer to function funcname, which has to exist. If there's pending
359 * code to be optimized and emitted, do so first.
360 */
361void *
362llvm_get_function(LLVMJitContext *context, const char *funcname)
363{
364	ListCell   *lc;
365
366	llvm_assert_in_fatal_section();
367
368	/*
369	 * If there is a pending / not emitted module, compile and emit now.
370	 * Otherwise we might not find the [correct] function.
371	 */
372	if (!context->compiled)
373	{
374		llvm_compile_module(context);
375	}
376
377	/*
378	 * ORC's symbol table is of *unmangled* symbols. Therefore we don't need
379	 * to mangle here.
380	 */
381
382	foreach(lc, context->handles)
383	{
384		LLVMJitHandle *handle = (LLVMJitHandle *) lfirst(lc);
385		instr_time	starttime;
386		instr_time	endtime;
387		LLVMErrorRef error;
388		LLVMOrcJITTargetAddress addr;
389
390		INSTR_TIME_SET_CURRENT(starttime);
391
392		addr = 0;
393		error = LLVMOrcLLJITLookup(handle->lljit, &addr, funcname);
394		if (error)
395			elog(ERROR, "failed to look up symbol \"%s\": %s",
396				 funcname, llvm_error_message(error));
397
398		/*
399		 * LLJIT only actually emits code the first time a symbol is
400		 * referenced. Thus add lookup time to emission time. That's counting
401		 * a bit more than with older LLVM versions, but unlikely to ever
402		 * matter.
403		 */
404		INSTR_TIME_SET_CURRENT(endtime);
405		INSTR_TIME_ACCUM_DIFF(context->base.instr.emission_counter,
406							  endtime, starttime);
407
408		if (addr)
409			return (void *) (uintptr_t) addr;
410	}
411
412	elog(ERROR, "failed to JIT: %s", funcname);
413
414	return NULL;
415}

417/*
418 * Return type of a variable in llvmjit_types.c. This is useful to keep types
419 * in sync between plain C and JIT related code.
420 */
421LLVMTypeRef
422llvm_pg_var_type(const char *varname)
423{
424	LLVMValueRef v_srcvar;
425	LLVMTypeRef typ;
426
427	/* this'll return a *pointer* to the global */
428	v_srcvar = LLVMGetNamedGlobal(llvm_types_module, varname);
429	if (!v_srcvar)
430		elog(ERROR, "variable %s not in llvmjit_types.c", varname);
431
432	typ = LLVMGlobalGetValueType(v_srcvar);
433
434	return typ;
435}

456/*
457 * Return declaration for a function referenced in llvmjit_types.c, adding it
458 * to the module if necessary.
459 *
460 * This is used to make functions discovered via llvm_create_types() known to
461 * the module that's currently being worked on.
462 */
463LLVMValueRef
464llvm_pg_func(LLVMModuleRef mod, const char *funcname)
465{
466	LLVMValueRef v_srcfn;
467	LLVMValueRef v_fn;
468
469	/* don't repeatedly add function */
470	v_fn = LLVMGetNamedFunction(mod, funcname);
471	if (v_fn)
472		return v_fn;
473
474	v_srcfn = LLVMGetNamedFunction(llvm_types_module, funcname);
475
476	if (!v_srcfn)
477		elog(ERROR, "function %s not in llvmjit_types.c", funcname);
478
479	v_fn = LLVMAddFunction(mod,
480						   funcname,
481						   LLVMGetFunctionType(v_srcfn));
482	llvm_copy_attributes(v_srcfn, v_fn);
483
484	return v_fn;
485}

487/*
488 * Copy attributes from one function to another, for a specific index (an
489 * index can reference return value, function and parameter attributes).
490 */
491static void
492llvm_copy_attributes_at_index(LLVMValueRef v_from, LLVMValueRef v_to, uint32 index)
493{
494	int			num_attributes;
495	LLVMAttributeRef *attrs;
496
497	num_attributes = LLVMGetAttributeCountAtIndex(v_from, index);
498
499	if (num_attributes == 0)
500		return;
501
502	attrs = palloc(sizeof(LLVMAttributeRef) * num_attributes);
503	LLVMGetAttributesAtIndex(v_from, index, attrs);
504
505	for (int attno = 0; attno < num_attributes; attno++)
506		LLVMAddAttributeAtIndex(v_to, index, attrs[attno]);
507
508	pfree(attrs);
509}
510
511/*
512 * Copy all attributes from one function to another. I.e. function, return and
513 * parameters will be copied.
514 */
515void
516llvm_copy_attributes(LLVMValueRef v_from, LLVMValueRef v_to)
517{
518	uint32		param_count;
519
520	/* copy function attributes */
521	llvm_copy_attributes_at_index(v_from, v_to, LLVMAttributeFunctionIndex);
522
523	if (LLVMGetTypeKind(LLVMGetFunctionReturnType(v_to)) != LLVMVoidTypeKind)
524	{
525		/* and the return value attributes */
526		llvm_copy_attributes_at_index(v_from, v_to, LLVMAttributeReturnIndex);
527	}
528
529	/* and each function parameter's attribute */
530	param_count = LLVMCountParams(v_from);
531
532	for (int paramidx = 1; paramidx <= param_count; paramidx++)
533		llvm_copy_attributes_at_index(v_from, v_to, paramidx);
534}

536/*
537 * Return a callable LLVMValueRef for fcinfo.
538 */
539LLVMValueRef
540llvm_function_reference(LLVMJitContext *context,
541						LLVMBuilderRef builder,
542						LLVMModuleRef mod,
543						FunctionCallInfo fcinfo)
544{
545	char	   *modname;
546	char	   *basename;
547	char	   *funcname;
548
549	LLVMValueRef v_fn;
550
551	fmgr_symbol(fcinfo->flinfo->fn_oid, &modname, &basename);
552
553	if (modname != NULL && basename != NULL)
554	{
555		/* external function in loadable library */
556		funcname = psprintf("pgextern.%s.%s", modname, basename);
557	}
558	else if (basename != NULL)
559	{
560		/* internal function */
561		funcname = pstrdup(basename);
562	}
563	else
564	{
565		/*
566		 * Function we don't know to handle, return pointer. We do so by
567		 * creating a global constant containing a pointer to the function.
568		 * Makes IR more readable.
569		 */
570		LLVMValueRef v_fn_addr;
571
572		funcname = psprintf("pgoidextern.%u",
573							fcinfo->flinfo->fn_oid);
574		v_fn = LLVMGetNamedGlobal(mod, funcname);
575		if (v_fn != 0)
576			return l_load(builder, TypePGFunction, v_fn, "");
577
578		v_fn_addr = l_ptr_const(fcinfo->flinfo->fn_addr, TypePGFunction);
579
580		v_fn = LLVMAddGlobal(mod, TypePGFunction, funcname);
581		LLVMSetInitializer(v_fn, v_fn_addr);
582		LLVMSetGlobalConstant(v_fn, true);
583		LLVMSetLinkage(v_fn, LLVMPrivateLinkage);
584		LLVMSetUnnamedAddr(v_fn, true);
585
586		return l_load(builder, TypePGFunction, v_fn, "");
587	}
588
589	/* check if function already has been added */
590	v_fn = LLVMGetNamedFunction(mod, funcname);
591	if (v_fn != 0)
592		return v_fn;
593
594	v_fn = LLVMAddFunction(mod, funcname, LLVMGetFunctionType(AttributeTemplate));
595
596	return v_fn;
597}

599/*
600 * Optimize code in module using the flags set in context.
601 */
602static void
603llvm_optimize_module(LLVMJitContext *context, LLVMModuleRef module)
604{
605#if LLVM_VERSION_MAJOR < 17
606	LLVMPassManagerBuilderRef llvm_pmb;
607	LLVMPassManagerRef llvm_mpm;
608	LLVMPassManagerRef llvm_fpm;
609	LLVMValueRef func;
610	int			compile_optlevel;
611
612	if (context->base.flags & PGJIT_OPT3)
613		compile_optlevel = 3;
614	else
615		compile_optlevel = 0;
616
617	/*
618	 * Have to create a new pass manager builder every pass through, as the
619	 * inliner has some per-builder state. Otherwise one ends up only inlining
620	 * a function the first time though.
621	 */
622	llvm_pmb = LLVMPassManagerBuilderCreate();
623	LLVMPassManagerBuilderSetOptLevel(llvm_pmb, compile_optlevel);
624	llvm_fpm = LLVMCreateFunctionPassManagerForModule(module);
625
626	if (context->base.flags & PGJIT_OPT3)
627	{
628		/* TODO: Unscientifically determined threshold */
629		LLVMPassManagerBuilderUseInlinerWithThreshold(llvm_pmb, 512);
630	}
631	else
632	{
633		/* we rely on mem2reg heavily, so emit even in the O0 case */
634		LLVMAddPromoteMemoryToRegisterPass(llvm_fpm);
635	}
636
637	LLVMPassManagerBuilderPopulateFunctionPassManager(llvm_pmb, llvm_fpm);
638
639	/*
640	 * Do function level optimization. This could be moved to the point where
641	 * functions are emitted, to reduce memory usage a bit.
642	 */
643	LLVMInitializeFunctionPassManager(llvm_fpm);
644	for (func = LLVMGetFirstFunction(context->module);
645		 func != NULL;
646		 func = LLVMGetNextFunction(func))
647		LLVMRunFunctionPassManager(llvm_fpm, func);
648	LLVMFinalizeFunctionPassManager(llvm_fpm);
649	LLVMDisposePassManager(llvm_fpm);
650
651	/*
652	 * Perform module level optimization. We do so even in the non-optimized
653	 * case, so always-inline functions etc get inlined. It's cheap enough.
654	 */
655	llvm_mpm = LLVMCreatePassManager();
656	LLVMPassManagerBuilderPopulateModulePassManager(llvm_pmb,
657													llvm_mpm);
658	/* always use always-inliner pass */
659	if (!(context->base.flags & PGJIT_OPT3))
660		LLVMAddAlwaysInlinerPass(llvm_mpm);
661	/* if doing inlining, but no expensive optimization, add inlining pass */
662	if (context->base.flags & PGJIT_INLINE
663		&& !(context->base.flags & PGJIT_OPT3))
664		LLVMAddFunctionInliningPass(llvm_mpm);
665	LLVMRunPassManager(llvm_mpm, context->module);
666	LLVMDisposePassManager(llvm_mpm);
667
668	LLVMPassManagerBuilderDispose(llvm_pmb);
669#else
670	LLVMPassBuilderOptionsRef options;
671	LLVMErrorRef err;
672	const char *passes;
673
674	if (context->base.flags & PGJIT_OPT3)
675		passes = "default<O3>";
676	else if (context->base.flags & PGJIT_INLINE)
677		/* if doing inlining, but no expensive optimization, add inline pass */
678		passes = "default<O0>,mem2reg,inline";
679	else
680		/* default<O0> includes always-inline pass */
681		passes = "default<O0>,mem2reg";
682
683	options = LLVMCreatePassBuilderOptions();
684
685#ifdef LLVM_PASS_DEBUG
686	LLVMPassBuilderOptionsSetDebugLogging(options, 1);
687#endif
688
689	/* In assertion builds, run the LLVM verify pass. */
690#ifdef USE_ASSERT_CHECKING
691	LLVMPassBuilderOptionsSetVerifyEach(options, true);
692#endif
693
694	LLVMPassBuilderOptionsSetInlinerThreshold(options, 512);
695
696	err = LLVMRunPasses(module, passes, NULL, options);
697
698	if (err)
699		elog(ERROR, "failed to JIT module: %s", llvm_error_message(err));
700
701	LLVMDisposePassBuilderOptions(options);
702#endif
703}

705/*
706 * Emit code for the currently pending module.
707 */
708static void
709llvm_compile_module(LLVMJitContext *context)
710{
711	LLVMJitHandle *handle;
712	MemoryContext oldcontext;
713	instr_time	starttime;
714	instr_time	endtime;
715	LLVMOrcLLJITRef compile_orc;
716
717	if (context->base.flags & PGJIT_OPT3)
718		compile_orc = llvm_opt3_orc;
719	else
720		compile_orc = llvm_opt0_orc;
721
722	/* perform inlining */
723	if (context->base.flags & PGJIT_INLINE)
724	{
725		INSTR_TIME_SET_CURRENT(starttime);
726		llvm_inline(context->module);
727		INSTR_TIME_SET_CURRENT(endtime);
728		INSTR_TIME_ACCUM_DIFF(context->base.instr.inlining_counter,
729							  endtime, starttime);
730	}
731
732	if (jit_dump_bitcode)
733	{
734		char	   *filename;
735
736		filename = psprintf("%d.%zu.bc",
737							MyProcPid,
738							context->module_generation);
739		LLVMWriteBitcodeToFile(context->module, filename);
740		pfree(filename);
741	}
742
743
744	/* optimize according to the chosen optimization settings */
745	INSTR_TIME_SET_CURRENT(starttime);
746	llvm_optimize_module(context, context->module);
747	INSTR_TIME_SET_CURRENT(endtime);
748	INSTR_TIME_ACCUM_DIFF(context->base.instr.optimization_counter,
749						  endtime, starttime);
750
751	if (jit_dump_bitcode)
752	{
753		char	   *filename;
754
755		filename = psprintf("%d.%zu.optimized.bc",
756							MyProcPid,
757							context->module_generation);
758		LLVMWriteBitcodeToFile(context->module, filename);
759		pfree(filename);
760	}
761
762	handle = (LLVMJitHandle *)
763		MemoryContextAlloc(TopMemoryContext, sizeof(LLVMJitHandle));
764
765	/*
766	 * Emit the code. Note that this can, depending on the optimization
767	 * settings, take noticeable resources as code emission executes low-level
768	 * instruction combining/selection passes etc. Without optimization a
769	 * faster instruction selection mechanism is used.
770	 */
771	INSTR_TIME_SET_CURRENT(starttime);
772	{
773		LLVMOrcThreadSafeModuleRef ts_module;
774		LLVMErrorRef error;
775		LLVMOrcJITDylibRef jd = LLVMOrcLLJITGetMainJITDylib(compile_orc);
776
777		ts_module = LLVMOrcCreateNewThreadSafeModule(context->module, llvm_ts_context);
778
779		handle->lljit = compile_orc;
780		handle->resource_tracker = LLVMOrcJITDylibCreateResourceTracker(jd);
781
782		/*
783		 * NB: This doesn't actually emit code. That happens lazily the first
784		 * time a symbol defined in the module is requested. Due to that
785		 * llvm_get_function() also accounts for emission time.
786		 */
787
788		context->module = NULL; /* will be owned by LLJIT */
789		error = LLVMOrcLLJITAddLLVMIRModuleWithRT(compile_orc,
790												  handle->resource_tracker,
791												  ts_module);
792
793		if (error)
794			elog(ERROR, "failed to JIT module: %s",
795				 llvm_error_message(error));
796
797		/* LLVMOrcLLJITAddLLVMIRModuleWithRT takes ownership of the module */
798	}
799
800	INSTR_TIME_SET_CURRENT(endtime);
801	INSTR_TIME_ACCUM_DIFF(context->base.instr.emission_counter,
802						  endtime, starttime);
803
804	context->module = NULL;
805	context->compiled = true;
806
807	/* remember emitted code for cleanup and lookups */
808	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
809	context->handles = lappend(context->handles, handle);
810	MemoryContextSwitchTo(oldcontext);
811
812	ereport(DEBUG1,
813			(errmsg_internal("time to inline: %.3fs, opt: %.3fs, emit: %.3fs",
814							 INSTR_TIME_GET_DOUBLE(context->base.instr.inlining_counter),
815							 INSTR_TIME_GET_DOUBLE(context->base.instr.optimization_counter),
816							 INSTR_TIME_GET_DOUBLE(context->base.instr.emission_counter)),
817			 errhidestmt(true),
818			 errhidecontext(true)));
819}

821/*
822 * Per session initialization.
823 */
824static void
825llvm_session_initialize(void)
826{
827	MemoryContext oldcontext;
828	char	   *error = NULL;
829	char	   *cpu = NULL;
830	char	   *features = NULL;
831	LLVMTargetMachineRef opt0_tm;
832	LLVMTargetMachineRef opt3_tm;
833
834	if (llvm_session_initialized)
835		return;
836
837	oldcontext = MemoryContextSwitchTo(TopMemoryContext);
838
839	LLVMInitializeNativeTarget();
840	LLVMInitializeNativeAsmPrinter();
841	LLVMInitializeNativeAsmParser();
842
843	if (llvm_context == NULL)
844	{
845		llvm_context = LLVMContextCreate();
846
847		llvm_jit_context_in_use_count = 0;
848		llvm_llvm_context_reuse_count = 0;
849	}
850
851	/*
852	 * Synchronize types early, as that also includes inferring the target
853	 * triple.
854	 */
855	llvm_create_types();
856
857	/*
858	 * Extract target information from loaded module.
859	 */
860	llvm_set_target();
861
862	if (LLVMGetTargetFromTriple(llvm_triple, &llvm_targetref, &error) != 0)
863	{
864		elog(FATAL, "failed to query triple %s", error);
865	}
866
867	/*
868	 * We want the generated code to use all available features. Therefore
869	 * grab the host CPU string and detect features of the current CPU. The
870	 * latter is needed because some CPU architectures default to enabling
871	 * features not all CPUs have (weird, huh).
872	 */
873	cpu = LLVMGetHostCPUName();
874	features = LLVMGetHostCPUFeatures();
875	elog(DEBUG2, "LLVMJIT detected CPU \"%s\", with features \"%s\"",
876		 cpu, features);
877
878	opt0_tm =
879		LLVMCreateTargetMachine(llvm_targetref, llvm_triple, cpu, features,
880								LLVMCodeGenLevelNone,
881								LLVMRelocDefault,
882								LLVMCodeModelJITDefault);
883	opt3_tm =
884		LLVMCreateTargetMachine(llvm_targetref, llvm_triple, cpu, features,
885								LLVMCodeGenLevelAggressive,
886								LLVMRelocDefault,
887								LLVMCodeModelJITDefault);
888
889	LLVMDisposeMessage(cpu);
890	cpu = NULL;
891	LLVMDisposeMessage(features);
892	features = NULL;
893
894	/* force symbols in main binary to be loaded */
895	LLVMLoadLibraryPermanently(NULL);
896
897	{
898		llvm_ts_context = LLVMOrcCreateNewThreadSafeContext();
899
900		llvm_opt0_orc = llvm_create_jit_instance(opt0_tm);
901		opt0_tm = 0;
902
903		llvm_opt3_orc = llvm_create_jit_instance(opt3_tm);
904		opt3_tm = 0;
905	}
906
907	on_proc_exit(llvm_shutdown, 0);
908
909	llvm_session_initialized = true;
910
911	MemoryContextSwitchTo(oldcontext);
912}

914static void
915llvm_shutdown(int code, Datum arg)
916{
917	/*
918	 * If llvm_shutdown() is reached while in a fatal-on-oom section an error
919	 * has occurred in the middle of LLVM code. It is not safe to call back
920	 * into LLVM (which is why a FATAL error was thrown).
921	 *
922	 * We do need to shutdown LLVM in other shutdown cases, otherwise e.g.
923	 * profiling data won't be written out.
924	 */
925	if (llvm_in_fatal_on_oom())
926	{
927		Assert(proc_exit_inprogress);
928		return;
929	}
930
931	if (llvm_jit_context_in_use_count != 0)
932		elog(PANIC, "LLVMJitContext in use count not 0 at exit (is %zu)",
933			 llvm_jit_context_in_use_count);
934
935	{
936		if (llvm_opt3_orc)
937		{
938			LLVMOrcDisposeLLJIT(llvm_opt3_orc);
939			llvm_opt3_orc = NULL;
940		}
941		if (llvm_opt0_orc)
942		{
943			LLVMOrcDisposeLLJIT(llvm_opt0_orc);
944			llvm_opt0_orc = NULL;
945		}
946		if (llvm_ts_context)
947		{
948			LLVMOrcDisposeThreadSafeContext(llvm_ts_context);
949			llvm_ts_context = NULL;
950		}
951	}
952}

955static LLVMTypeRef
956load_return_type(LLVMModuleRef mod, const char *name)
957{
958	LLVMValueRef value;
959	LLVMTypeRef typ;

962	value = LLVMGetNamedFunction(mod, name);

1044/*
1045 * Split a symbol into module / function parts.  If the function is in the
1046 * main binary (or an external library) *modname will be NULL.
1047 */
1048void
1049llvm_split_symbol_name(const char *name, char **modname, char **funcname)
1050{
1051	*modname = NULL;
1052	*funcname = NULL;
1053
1054	/*
1055	 * Module function names are pgextern.$module.$funcname
1056	 */
1057	if (strncmp(name, "pgextern.", strlen("pgextern.")) == 0)
1058	{
1059		/*
1060		 * Symbol names cannot contain a ., therefore we can split based on
1061		 * first and last occurrence of one.
1062		 */
1063		*funcname = rindex(name, '.');
1064		(*funcname)++;			/* jump over . */
1065
1066		*modname = pnstrdup(name + strlen("pgextern."),
1067							*funcname - name - strlen("pgextern.") - 1);
1068		Assert(funcname);
1069
1070		*funcname = pstrdup(*funcname);
1071	}
1072	else
1073	{
1074		*modname = NULL;
1075		*funcname = pstrdup(name);
1076	}
1077}

1079/*
1080 * Attempt to resolve symbol, so LLVM can emit a reference to it.
1081 */
1082static uint64_t
1083llvm_resolve_symbol(const char *symname, void *ctx)
1084{
1085	uintptr_t	addr;
1086	char	   *funcname;
1087	char	   *modname;
1088
1089	/*
1090	 * macOS prefixes all object level symbols with an underscore. But neither
1091	 * dlsym() nor PG's inliner expect that. So undo.
1092	 */
1093#if defined(__darwin__)
1094	if (symname[0] != '_')
1095		elog(ERROR, "expected prefixed symbol name, but got \"%s\"", symname);
1096	symname++;
1097#endif
1098
1099	llvm_split_symbol_name(symname, &modname, &funcname);
1100
1101	/* functions that aren't resolved to names shouldn't ever get here */
1102	Assert(funcname);
1103
1104	if (modname)
1105		addr = (uintptr_t) load_external_function(modname, funcname,
1106												  true, NULL);
1107	else
1108		addr = (uintptr_t) LLVMSearchForAddressOfSymbol(symname);
1109
1110	pfree(funcname);
1111	if (modname)
1112		pfree(modname);
1113
1114	/* let LLVM will error out - should never happen */
1115	if (!addr)
1116		elog(WARNING, "failed to resolve name %s", symname);
1117
1118	return (uint64_t) addr;
1119}

1121static LLVMErrorRef
1122llvm_resolve_symbols(LLVMOrcDefinitionGeneratorRef GeneratorObj, void *Ctx,
1123					 LLVMOrcLookupStateRef *LookupState, LLVMOrcLookupKind Kind,
1124					 LLVMOrcJITDylibRef JD, LLVMOrcJITDylibLookupFlags JDLookupFlags,
1125					 LLVMOrcCLookupSet LookupSet, size_t LookupSetSize)
1126{
1127#if LLVM_VERSION_MAJOR > 14
1128	LLVMOrcCSymbolMapPairs symbols = palloc0(sizeof(LLVMOrcCSymbolMapPair) * LookupSetSize);
1129#else
1130	LLVMOrcCSymbolMapPairs symbols = palloc0(sizeof(LLVMJITCSymbolMapPair) * LookupSetSize);
1131#endif
1132	LLVMErrorRef error;
1133	LLVMOrcMaterializationUnitRef mu;
1134
1135	for (int i = 0; i < LookupSetSize; i++)
1136	{
1137		const char *name = LLVMOrcSymbolStringPoolEntryStr(LookupSet[i].Name);
1138
1139		LLVMOrcRetainSymbolStringPoolEntry(LookupSet[i].Name);
1140		symbols[i].Name = LookupSet[i].Name;
1141		symbols[i].Sym.Address = llvm_resolve_symbol(name, NULL);
1142		symbols[i].Sym.Flags.GenericFlags = LLVMJITSymbolGenericFlagsExported;
1143	}
1144
1145	mu = LLVMOrcAbsoluteSymbols(symbols, LookupSetSize);
1146	error = LLVMOrcJITDylibDefine(JD, mu);
1147	if (error != LLVMErrorSuccess)
1148		LLVMOrcDisposeMaterializationUnit(mu);
1149
1150	pfree(symbols);
1151
1152	return error;
1153}

1157 * just use WARNING here. That's OK anyway, as the error is also reported at
1158 * the top level action (with less detail) and there might be multiple
1159 * invocations of errors with details.
1160 *
1161 * This doesn't really happen during normal operation, but in cases like
1162 * symbol resolution breakage. So just using elog(WARNING) is fine.

1211/*
1212 * Create LLJIT instance, using the passed in target machine. Note that the
1213 * target machine afterwards is owned by the LLJIT instance.
1214 */
1215static LLVMOrcLLJITRef
1216llvm_create_jit_instance(LLVMTargetMachineRef tm)
1217{
1218	LLVMOrcLLJITRef lljit;
1219	LLVMOrcJITTargetMachineBuilderRef tm_builder;
1220	LLVMOrcLLJITBuilderRef lljit_builder;
1221	LLVMErrorRef error;
1222	LLVMOrcDefinitionGeneratorRef main_gen;
1223	LLVMOrcDefinitionGeneratorRef ref_gen;
1224
1225	lljit_builder = LLVMOrcCreateLLJITBuilder();
1226	tm_builder = LLVMOrcJITTargetMachineBuilderCreateFromTargetMachine(tm);
1227	LLVMOrcLLJITBuilderSetJITTargetMachineBuilder(lljit_builder, tm_builder);
1228
1229	LLVMOrcLLJITBuilderSetObjectLinkingLayerCreator(lljit_builder,
1230													llvm_create_object_layer,
1231													NULL);
1232
1233	error = LLVMOrcCreateLLJIT(&lljit, lljit_builder);
1234	if (error)
1235		elog(ERROR, "failed to create lljit instance: %s",
1236			 llvm_error_message(error));
1237
1238	LLVMOrcExecutionSessionSetErrorReporter(LLVMOrcLLJITGetExecutionSession(lljit),
1239											llvm_log_jit_error, NULL);
1240
1241	/*
1242	 * Symbol resolution support for symbols in the postgres binary /
1243	 * libraries already loaded.
1244	 */
1245	error = LLVMOrcCreateDynamicLibrarySearchGeneratorForProcess(&main_gen,
1246																 LLVMOrcLLJITGetGlobalPrefix(lljit),
1247																 0, NULL);
1248	if (error)
1249		elog(ERROR, "failed to create generator: %s",
1250			 llvm_error_message(error));
1251	LLVMOrcJITDylibAddGenerator(LLVMOrcLLJITGetMainJITDylib(lljit), main_gen);
1252
1253	/*
1254	 * Symbol resolution support for "special" functions, e.g. a call into an
1255	 * SQL callable function.
1256	 */
1257#if LLVM_VERSION_MAJOR > 14
1258	ref_gen = LLVMOrcCreateCustomCAPIDefinitionGenerator(llvm_resolve_symbols, NULL, NULL);
1259#else
1260	ref_gen = LLVMOrcCreateCustomCAPIDefinitionGenerator(llvm_resolve_symbols, NULL);
1261#endif
1262	LLVMOrcJITDylibAddGenerator(LLVMOrcLLJITGetMainJITDylib(lljit), ref_gen);
1263
1264	return lljit;
1265}
