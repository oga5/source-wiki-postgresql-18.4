203/*
204 * Adds array of item pointers to tuple's posting list, or
205 * creates posting tree and tuple pointing to tree in case
206 * of not enough space.  Max size of tuple is defined in
207 * GinFormTuple().  Returns a new, modified index tuple.
208 * items[] must be in sorted order with no duplicates.
209 */ view japanese
210static IndexTuple
211addItemPointersToLeafTuple(GinState *ginstate,
212						   IndexTuple old,
213						   ItemPointerData *items, uint32 nitem,
214						   GinStatsData *buildStats, Buffer buffer)
215{
216	OffsetNumber attnum;
217	Datum		key;
218	GinNullCategory category;
219	IndexTuple	res;
220	ItemPointerData *newItems,
221			   *oldItems;
222	int			oldNPosting,
223				newNPosting;
224	GinPostingList *compressedList;
225
226	Assert(!GinIsPostingTree(old));
227
228	attnum = gintuple_get_attrnum(ginstate, old);
229	key = gintuple_get_key(ginstate, old, &category);
230
231	/* merge the old and new posting lists */ view japanese
232	oldItems = ginReadTuple(ginstate, attnum, old, &oldNPosting);
233
234	newItems = ginMergeItemPointers(items, nitem,
235									oldItems, oldNPosting,
236									&newNPosting);
237
238	/* Compress the posting list, and try to a build tuple with room for it */ view japanese
239	res = NULL;
240	compressedList = ginCompressPostingList(newItems, newNPosting, GinMaxItemSize,
241											NULL);
242	pfree(newItems);
243	if (compressedList)
244	{
245		res = GinFormTuple(ginstate, attnum, key, category,
246						   (char *) compressedList,
247						   SizeOfGinPostingList(compressedList),
248						   newNPosting,
249						   false);
250		pfree(compressedList);
251	}
252	if (!res)
253	{
254		/* posting list would be too big, convert to posting tree */ view japanese
255		BlockNumber postingRoot;
256
257		/*
258		 * Initialize posting tree with the old tuple's posting list.  It's
259		 * surely small enough to fit on one posting-tree page, and should
260		 * already be in order with no duplicates.
261		 */ view japanese
262		postingRoot = createPostingTree(ginstate->index,
263										oldItems,
264										oldNPosting,
265										buildStats,
266										buffer);
267
268		/* Now insert the TIDs-to-be-added into the posting tree */ view japanese
269		ginInsertItemPointers(ginstate->index, postingRoot,
270							  items, nitem,
271							  buildStats);
272
273		/* And build a new posting-tree-only result tuple */ view japanese
274		res = GinFormTuple(ginstate, attnum, key, category, NULL, 0, 0, true);
275		GinSetPostingTree(res, postingRoot);
276	}
277	pfree(oldItems);
278
279	return res;
280}

282/*
283 * Build a fresh leaf tuple, either posting-list or posting-tree format
284 * depending on whether the given items list will fit.
285 * items[] must be in sorted order with no duplicates.
286 *
287 * This is basically the same logic as in addItemPointersToLeafTuple,
288 * but working from slightly different input.
289 */ view japanese
290static IndexTuple
291buildFreshLeafTuple(GinState *ginstate,
292					OffsetNumber attnum, Datum key, GinNullCategory category,
293					ItemPointerData *items, uint32 nitem,
294					GinStatsData *buildStats, Buffer buffer)
295{
296	IndexTuple	res = NULL;
297	GinPostingList *compressedList;
298
299	/* try to build a posting list tuple with all the items */ view japanese
300	compressedList = ginCompressPostingList(items, nitem, GinMaxItemSize, NULL);
301	if (compressedList)
302	{
303		res = GinFormTuple(ginstate, attnum, key, category,
304						   (char *) compressedList,
305						   SizeOfGinPostingList(compressedList),
306						   nitem, false);
307		pfree(compressedList);
308	}
309	if (!res)
310	{
311		/* posting list would be too big, build posting tree */ view japanese
312		BlockNumber postingRoot;
313
314		/*
315		 * Build posting-tree-only result tuple.  We do this first so as to
316		 * fail quickly if the key is too big.
317		 */ view japanese
318		res = GinFormTuple(ginstate, attnum, key, category, NULL, 0, 0, true);
319
320		/*
321		 * Initialize a new posting tree with the TIDs.
322		 */ view japanese
323		postingRoot = createPostingTree(ginstate->index, items, nitem,
324										buildStats, buffer);
325
326		/* And save the root link in the result tuple */ view japanese
327		GinSetPostingTree(res, postingRoot);
328	}
329
330	return res;
331}

333/*
334 * Insert one or more heap TIDs associated with the given key value.
335 * This will either add a single key entry, or enlarge a pre-existing entry.
336 *
337 * During an index build, buildStats is non-null and the counters
338 * it contains should be incremented as needed.
339 */ view japanese
340void
341ginEntryInsert(GinState *ginstate,
342			   OffsetNumber attnum, Datum key, GinNullCategory category,
343			   ItemPointerData *items, uint32 nitem,
344			   GinStatsData *buildStats)
345{
346	GinBtreeData btree;
347	GinBtreeEntryInsertData insertdata;
348	GinBtreeStack *stack;
349	IndexTuple	itup;
350	Page		page;
351
352	insertdata.isDelete = false;
353
354	ginPrepareEntryScan(&btree, attnum, key, category, ginstate);
355	btree.isBuild = (buildStats != NULL);
356
357	stack = ginFindLeafPage(&btree, false, false);
358	page = BufferGetPage(stack->buffer);
359
360	if (btree.findItem(&btree, stack))
361	{
362		/* found pre-existing entry */ view japanese
363		itup = (IndexTuple) PageGetItem(page, PageGetItemId(page, stack->off));
364
365		if (GinIsPostingTree(itup))
366		{
367			/* add entries to existing posting tree */ view japanese
368			BlockNumber rootPostingTree = GinGetPostingTree(itup);
369
370			/* release all stack */ view japanese
371			LockBuffer(stack->buffer, GIN_UNLOCK);
372			freeGinBtreeStack(stack);
373
374			/* insert into posting tree */ view japanese
375			ginInsertItemPointers(ginstate->index, rootPostingTree,
376								  items, nitem,
377								  buildStats);
378			return;
379		}
380
381		CheckForSerializableConflictIn(ginstate->index, NULL,
382									   BufferGetBlockNumber(stack->buffer));
383		/* modify an existing leaf entry */ view japanese
384		itup = addItemPointersToLeafTuple(ginstate, itup,
385										  items, nitem, buildStats, stack->buffer);
386
387		insertdata.isDelete = true;
388	}
389	else
390	{
391		CheckForSerializableConflictIn(ginstate->index, NULL,
392									   BufferGetBlockNumber(stack->buffer));
393		/* no match, so construct a new leaf entry */ view japanese
394		itup = buildFreshLeafTuple(ginstate, attnum, key, category,
395								   items, nitem, buildStats, stack->buffer);
396
397		/*
398		 * nEntries counts leaf tuples, so increment it only when we make a
399		 * new one.
400		 */ view japanese
401		if (buildStats)
402			buildStats->nEntries++;
403	}
404
405	/* Insert the new or modified leaf tuple */ view japanese
406	insertdata.entry = itup;
407	ginInsertValue(&btree, stack, &insertdata, buildStats);
408	pfree(itup);
409}

411/*
412 * Extract index entries for a single indexable item, and add them to the
413 * BuildAccumulator's state.
414 *
415 * This function is used only during initial index creation.
416 */ view japanese
417static void
418ginHeapTupleBulkInsert(GinBuildState *buildstate, OffsetNumber attnum,
419					   Datum value, bool isNull,
420					   ItemPointer heapptr)
421{
422	Datum	   *entries;
423	GinNullCategory *categories;
424	int32		nentries;
425	MemoryContext oldCtx;
426
427	oldCtx = MemoryContextSwitchTo(buildstate->funcCtx);
428	entries = ginExtractEntries(buildstate->accum.ginstate, attnum,
429								value, isNull,
430								&nentries, &categories);
431	MemoryContextSwitchTo(oldCtx);
432
433	ginInsertBAEntries(&buildstate->accum, heapptr, attnum,
434					   entries, categories, nentries);
435
436	buildstate->indtuples += nentries;
437
438	MemoryContextReset(buildstate->funcCtx);
439}

441static void
442ginBuildCallback(Relation index, ItemPointer tid, Datum *values,
443				 bool *isnull, bool tupleIsAlive, void *state)
444{
445	GinBuildState *buildstate = (GinBuildState *) state;
446	MemoryContext oldCtx;
447	int			i;
448
449	oldCtx = MemoryContextSwitchTo(buildstate->tmpCtx);
450
451	for (i = 0; i < buildstate->ginstate.origTupdesc->natts; i++)
452		ginHeapTupleBulkInsert(buildstate, (OffsetNumber) (i + 1),
453							   values[i], isnull[i], tid);
454
455	/* If we've maxed out our available memory, dump everything to the index */ view japanese
456	if (buildstate->accum.allocatedMemory >= maintenance_work_mem * (Size) 1024)
457	{
458		ItemPointerData *list;
459		Datum		key;
460		GinNullCategory category;
461		uint32		nlist;
462		OffsetNumber attnum;
463
464		ginBeginBAScan(&buildstate->accum);
465		while ((list = ginGetBAEntry(&buildstate->accum,
466									 &attnum, &key, &category, &nlist)) != NULL)
467		{
468			/* there could be many entries, so be willing to abort here */
469			CHECK_FOR_INTERRUPTS();
470			ginEntryInsert(&buildstate->ginstate, attnum, key, category,
471						   list, nlist, &buildstate->buildStats);
472		}
473
474		MemoryContextReset(buildstate->tmpCtx);
475		ginInitBA(&buildstate->accum);
476	}
477
478	MemoryContextSwitchTo(oldCtx);
479}

481/*
482 * ginFlushBuildState
483 *		Write all data from BuildAccumulator into the tuplesort.
484 *
485 * The number of TIDs written to the tuplesort at once is limited, to reduce
486 * the amount of memory needed when merging the intermediate results later.
487 * The leader will see up to two chunks per worker, so calculate the limit to
488 * not need more than MaxAllocSize overall.
489 *
490 * We don't need to worry about overflowing maintenance_work_mem. We can't
491 * build chunks larger than work_mem, and that limit was set so that workers
492 * produce sufficiently small chunks.
493 */ view japanese
494static void
495ginFlushBuildState(GinBuildState *buildstate, Relation index)
496{
497	ItemPointerData *list;
498	Datum		key;
499	GinNullCategory category;
500	uint32		nlist;
501	OffsetNumber attnum;
502	TupleDesc	tdesc = RelationGetDescr(index);
503	uint32		maxlen;
504
505	/* maximum number of TIDs per chunk (two chunks per worker) */ view japanese
506	maxlen = MaxAllocSize / sizeof(ItemPointerData);
507	maxlen /= (2 * buildstate->bs_num_workers);
508
509	ginBeginBAScan(&buildstate->accum);
510	while ((list = ginGetBAEntry(&buildstate->accum,
511								 &attnum, &key, &category, &nlist)) != NULL)
512	{
513		/* information about the key */ view japanese
514		Form_pg_attribute attr = TupleDescAttr(tdesc, (attnum - 1));
515
516		/* start of the chunk */ view japanese
517		uint32		offset = 0;
518
519		/* split the entry into smaller chunk with up to maxlen items */ view japanese
520		while (offset < nlist)
521		{
522			/* GIN tuple and tuple length */ view japanese
523			GinTuple   *tup;
524			Size		tuplen;
525			uint32		len = Min(maxlen, nlist - offset);
526
527			/* there could be many entries, so be willing to abort here */
528			CHECK_FOR_INTERRUPTS();
529
530			tup = _gin_build_tuple(attnum, category,
531								   key, attr->attlen, attr->attbyval,
532								   &list[offset], len,
533								   &tuplen);
534
535			offset += len;
536
537			tuplesort_putgintuple(buildstate->bs_worker_sort, tup, tuplen);
538
539			pfree(tup);
540		}
541	}
542
543	MemoryContextReset(buildstate->tmpCtx);
544	ginInitBA(&buildstate->accum);
545}

547/*
548 * ginBuildCallbackParallel
549 *		Callback for the parallel index build.
550 *
551 * This is similar to the serial build callback ginBuildCallback, but
552 * instead of writing the accumulated entries into the index, each worker
553 * writes them into a (local) tuplesort.
554 *
555 * The worker then sorts and combines these entries, before writing them
556 * into a shared tuplesort for the leader (see _gin_parallel_scan_and_build
557 * for the whole process).
558 */ view japanese
559static void
560ginBuildCallbackParallel(Relation index, ItemPointer tid, Datum *values,
561						 bool *isnull, bool tupleIsAlive, void *state)
562{
563	GinBuildState *buildstate = (GinBuildState *) state;
564	MemoryContext oldCtx;
565	int			i;
566
567	oldCtx = MemoryContextSwitchTo(buildstate->tmpCtx);
568
569	/*
570	 * if scan wrapped around - flush accumulated entries and start anew
571	 *
572	 * With parallel scans, we don't have a guarantee the scan does not start
573	 * half-way through the relation (serial builds disable sync scans and
574	 * always start from block 0, parallel scans require allow_sync=true).
575	 *
576	 * Building the posting lists assumes the TIDs are monotonic and never go
577	 * back, and the wrap around would break that. We handle that by detecting
578	 * the wraparound, and flushing all entries. This means we'll later see
579	 * two separate entries with non-overlapping TID lists (which can be
580	 * combined by merge sort).
581	 *
582	 * To detect a wraparound, we remember the last TID seen by each worker
583	 * (for any key). If the next TID seen by the worker is lower, the scan
584	 * must have wrapped around.
585	 */ view japanese
586	if (ItemPointerCompare(tid, &buildstate->tid) < 0)
587		ginFlushBuildState(buildstate, index);
588
589	/* remember the TID we're about to process */ view japanese
590	buildstate->tid = *tid;
591
592	for (i = 0; i < buildstate->ginstate.origTupdesc->natts; i++)
593		ginHeapTupleBulkInsert(buildstate, (OffsetNumber) (i + 1),
594							   values[i], isnull[i], tid);
595
596	/*
597	 * If we've maxed out our available memory, dump everything to the
598	 * tuplesort. We use half the per-worker fraction of maintenance_work_mem,
599	 * the other half is used for the tuplesort.
600	 */ view japanese
601	if (buildstate->accum.allocatedMemory >= buildstate->work_mem * (Size) 1024)
602		ginFlushBuildState(buildstate, index);
603
604	MemoryContextSwitchTo(oldCtx);
605}

607IndexBuildResult *
608ginbuild(Relation heap, Relation index, IndexInfo *indexInfo)
609{
610	IndexBuildResult *result;
611	double		reltuples;
612	GinBuildState buildstate;
613	GinBuildState *state = &buildstate;
614	Buffer		RootBuffer,
615				MetaBuffer;
616	ItemPointerData *list;
617	Datum		key;
618	GinNullCategory category;
619	uint32		nlist;
620	MemoryContext oldCtx;
621	OffsetNumber attnum;
622
623	if (RelationGetNumberOfBlocks(index) != 0)
624		elog(ERROR, "index \"%s\" already contains data",
625			 RelationGetRelationName(index));
626
627	initGinState(&buildstate.ginstate, index);
628	buildstate.indtuples = 0;
629	memset(&buildstate.buildStats, 0, sizeof(GinStatsData));
630
631	/* Initialize fields for parallel build too. */ view japanese
632	buildstate.bs_numtuples = 0;
633	buildstate.bs_reltuples = 0;
634	buildstate.bs_leader = NULL;
635	memset(&buildstate.tid, 0, sizeof(ItemPointerData));
636
637	/* initialize the meta page */ view japanese
638	MetaBuffer = GinNewBuffer(index);
639
640	/* initialize the root page */ view japanese
641	RootBuffer = GinNewBuffer(index);
642
643	START_CRIT_SECTION();
644	GinInitMetabuffer(MetaBuffer);
645	MarkBufferDirty(MetaBuffer);
646	GinInitBuffer(RootBuffer, GIN_LEAF);
647	MarkBufferDirty(RootBuffer);
648
649
650	UnlockReleaseBuffer(MetaBuffer);
651	UnlockReleaseBuffer(RootBuffer);
652	END_CRIT_SECTION();
653
654	/* count the root as first entry page */ view japanese
655	buildstate.buildStats.nEntryPages++;
656
657	/*
658	 * create a temporary memory context that is used to hold data not yet
659	 * dumped out to the index
660	 */
661	buildstate.tmpCtx = AllocSetContextCreate(CurrentMemoryContext,
662											  "Gin build temporary context",
663											  ALLOCSET_DEFAULT_SIZES);
664
665	/*
666	 * create a temporary memory context that is used for calling
667	 * ginExtractEntries(), and can be reset after each tuple
668	 */
669	buildstate.funcCtx = AllocSetContextCreate(CurrentMemoryContext,
670											   "Gin build temporary context for user-defined function",
671											   ALLOCSET_DEFAULT_SIZES);
672
673	buildstate.accum.ginstate = &buildstate.ginstate;
674	ginInitBA(&buildstate.accum);
675
676	/* Report table scan phase started */ view japanese
677	pgstat_progress_update_param(PROGRESS_CREATEIDX_SUBPHASE,
678								 PROGRESS_GIN_PHASE_INDEXBUILD_TABLESCAN);
679
680	/*
681	 * Attempt to launch parallel worker scan when required
682	 *
683	 * XXX plan_create_index_workers makes the number of workers dependent on
684	 * maintenance_work_mem, requiring 32MB for each worker. For GIN that's
685	 * reasonable too, because we sort the data just like btree. It does
686	 * ignore the memory used to accumulate data in memory (set by work_mem),
687	 * but there is no way to communicate that to plan_create_index_workers.
688	 */ view japanese
689	if (indexInfo->ii_ParallelWorkers > 0)
690		_gin_begin_parallel(state, heap, index, indexInfo->ii_Concurrent,
691							indexInfo->ii_ParallelWorkers);
692
693	/*
694	 * If parallel build requested and at least one worker process was
695	 * successfully launched, set up coordination state, wait for workers to
696	 * complete. Then read all tuples from the shared tuplesort and insert
697	 * them into the index.
698	 *
699	 * In serial mode, simply scan the table and build the index one index
700	 * tuple at a time.
701	 */ view japanese
702	if (state->bs_leader)
703	{
704		SortCoordinate coordinate;
705
706		coordinate = (SortCoordinate) palloc0(sizeof(SortCoordinateData));
707		coordinate->isWorker = false;
708		coordinate->nParticipants =
709			state->bs_leader->nparticipanttuplesorts;
710		coordinate->sharedsort = state->bs_leader->sharedsort;
711
712		/*
713		 * Begin leader tuplesort.
714		 *
715		 * In cases where parallelism is involved, the leader receives the
716		 * same share of maintenance_work_mem as a serial sort (it is
717		 * generally treated in the same way as a serial sort once we return).
718		 * Parallel worker Tuplesortstates will have received only a fraction
719		 * of maintenance_work_mem, though.
720		 *
721		 * We rely on the lifetime of the Leader Tuplesortstate almost not
722		 * overlapping with any worker Tuplesortstate's lifetime.  There may
723		 * be some small overlap, but that's okay because we rely on leader
724		 * Tuplesortstate only allocating a small, fixed amount of memory
725		 * here. When its tuplesort_performsort() is called (by our caller),
726		 * and significant amounts of memory are likely to be used, all
727		 * workers must have already freed almost all memory held by their
728		 * Tuplesortstates (they are about to go away completely, too).  The
729		 * overall effect is that maintenance_work_mem always represents an
730		 * absolute high watermark on the amount of memory used by a CREATE
731		 * INDEX operation, regardless of the use of parallelism or any other
732		 * factor.
733		 */ view japanese
734		state->bs_sortstate =
735			tuplesort_begin_index_gin(heap, index,
736									  maintenance_work_mem, coordinate,
737									  TUPLESORT_NONE);
738
739		/* scan the relation in parallel and merge per-worker results */ view japanese
740		reltuples = _gin_parallel_merge(state);
741
742		_gin_end_parallel(state->bs_leader, state);
743	}
744	else						/* no parallel index build */ view japanese
745	{
746		/*
747		 * Do the heap scan.  We disallow sync scan here because
748		 * dataPlaceToPage prefers to receive tuples in TID order.
749		 */ view japanese
750		reltuples = table_index_build_scan(heap, index, indexInfo, false, true,
751										   ginBuildCallback, &buildstate, NULL);
752
753		/* dump remaining entries to the index */ view japanese
754		oldCtx = MemoryContextSwitchTo(buildstate.tmpCtx);
755		ginBeginBAScan(&buildstate.accum);
756		while ((list = ginGetBAEntry(&buildstate.accum,
757									 &attnum, &key, &category, &nlist)) != NULL)
758		{
759			/* there could be many entries, so be willing to abort here */ view japanese
760			CHECK_FOR_INTERRUPTS();
761			ginEntryInsert(&buildstate.ginstate, attnum, key, category,
762						   list, nlist, &buildstate.buildStats);
763		}
764		MemoryContextSwitchTo(oldCtx);
765	}
766
767	MemoryContextDelete(buildstate.funcCtx);
768	MemoryContextDelete(buildstate.tmpCtx);
769
770	/*
771	 * Update metapage stats
772	 */ view japanese
773	buildstate.buildStats.nTotalPages = RelationGetNumberOfBlocks(index);
774	ginUpdateStats(index, &buildstate.buildStats, true);
775
776	/*
777	 * We didn't write WAL records as we built the index, so if WAL-logging is
778	 * required, write all pages to the WAL now.
779	 */ view japanese
780	if (RelationNeedsWAL(index))
781	{
782		log_newpage_range(index, MAIN_FORKNUM,
783						  0, RelationGetNumberOfBlocks(index),
784						  true);
785	}
786
787	/*
788	 * Return statistics
789	 */ view japanese
790	result = (IndexBuildResult *) palloc(sizeof(IndexBuildResult));
791
792	result->heap_tuples = reltuples;
793	result->index_tuples = buildstate.indtuples;
794
795	return result;
796}

906/*
907 * Create parallel context, and launch workers for leader.
908 *
909 * buildstate argument should be initialized (with the exception of the
910 * tuplesort states, which may later be created based on shared
911 * state initially set up here).
912 *
913 * isconcurrent indicates if operation is CREATE INDEX CONCURRENTLY.
914 *
915 * request is the target number of parallel worker processes to launch.
916 *
917 * Sets buildstate's GinLeader, which caller must use to shut down parallel
918 * mode by passing it to _gin_end_parallel() at the very end of its index
919 * build.  If not even a single worker process can be launched, this is
920 * never set, and caller should proceed with a serial index build.
921 */ view japanese
922static void
923_gin_begin_parallel(GinBuildState *buildstate, Relation heap, Relation index,
924					bool isconcurrent, int request)
925{
926	ParallelContext *pcxt;
927	int			scantuplesortstates;
928	Snapshot	snapshot;
929	Size		estginshared;
930	Size		estsort;
931	GinBuildShared *ginshared;
932	Sharedsort *sharedsort;
933	GinLeader  *ginleader = (GinLeader *) palloc0(sizeof(GinLeader));
934	WalUsage   *walusage;
935	BufferUsage *bufferusage;
936	bool		leaderparticipates = true;
937	int			querylen;
938
939#ifdef DISABLE_LEADER_PARTICIPATION
940	leaderparticipates = false;
941#endif
942
943	/*
944	 * Enter parallel mode, and create context for parallel build of gin index
945	 */ view japanese
946	EnterParallelMode();
947	Assert(request > 0);
948	pcxt = CreateParallelContext("postgres", "_gin_parallel_build_main",
949								 request);
950
951	scantuplesortstates = leaderparticipates ? request + 1 : request;
952
953	/*
954	 * Prepare for scan of the base relation.  In a normal index build, we use
955	 * SnapshotAny because we must retrieve all tuples and do our own time
956	 * qual checks (because we have to index RECENTLY_DEAD tuples).  In a
957	 * concurrent build, we take a regular MVCC snapshot and index whatever's
958	 * live according to that.
959	 */ view japanese
960	if (!isconcurrent)
961		snapshot = SnapshotAny;
962	else
963		snapshot = RegisterSnapshot(GetTransactionSnapshot());
964
965	/*
966	 * Estimate size for our own PARALLEL_KEY_GIN_SHARED workspace.
967	 */ view japanese
968	estginshared = _gin_parallel_estimate_shared(heap, snapshot);
969	shm_toc_estimate_chunk(&pcxt->estimator, estginshared);
970	estsort = tuplesort_estimate_shared(scantuplesortstates);
971	shm_toc_estimate_chunk(&pcxt->estimator, estsort);
972
973	shm_toc_estimate_keys(&pcxt->estimator, 2);
974
975	/*
976	 * Estimate space for WalUsage and BufferUsage -- PARALLEL_KEY_WAL_USAGE
977	 * and PARALLEL_KEY_BUFFER_USAGE.
978	 *
979	 * If there are no extensions loaded that care, we could skip this.  We
980	 * have no way of knowing whether anyone's looking at pgWalUsage or
981	 * pgBufferUsage, so do it unconditionally.
982	 */ view japanese
983	shm_toc_estimate_chunk(&pcxt->estimator,
984						   mul_size(sizeof(WalUsage), pcxt->nworkers));
985	shm_toc_estimate_keys(&pcxt->estimator, 1);
986	shm_toc_estimate_chunk(&pcxt->estimator,
987						   mul_size(sizeof(BufferUsage), pcxt->nworkers));
988	shm_toc_estimate_keys(&pcxt->estimator, 1);
989
990	/* Finally, estimate PARALLEL_KEY_QUERY_TEXT space */ view japanese
991	if (debug_query_string)
992	{
993		querylen = strlen(debug_query_string);
994		shm_toc_estimate_chunk(&pcxt->estimator, querylen + 1);
995		shm_toc_estimate_keys(&pcxt->estimator, 1);
996	}
997	else
998		querylen = 0;			/* keep compiler quiet */ view japanese
999
1000	/* Everyone's had a chance to ask for space, so now create the DSM */ view japanese
1001	InitializeParallelDSM(pcxt);
1002
1003	/* If no DSM segment was available, back out (do serial build) */ view japanese
1004	if (pcxt->seg == NULL)
1005	{
1006		if (IsMVCCSnapshot(snapshot))
1007			UnregisterSnapshot(snapshot);
1008		DestroyParallelContext(pcxt);
1009		ExitParallelMode();
1010		return;
1011	}
1012
1013	/* Store shared build state, for which we reserved space */ view japanese
1014	ginshared = (GinBuildShared *) shm_toc_allocate(pcxt->toc, estginshared);
1015	/* Initialize immutable state */ view japanese
1016	ginshared->heaprelid = RelationGetRelid(heap);
1017	ginshared->indexrelid = RelationGetRelid(index);
1018	ginshared->isconcurrent = isconcurrent;
1019	ginshared->scantuplesortstates = scantuplesortstates;
1020
1021	ConditionVariableInit(&ginshared->workersdonecv);
1022	SpinLockInit(&ginshared->mutex);
1023
1024	/* Initialize mutable state */ view japanese
1025	ginshared->nparticipantsdone = 0;
1026	ginshared->reltuples = 0.0;
1027	ginshared->indtuples = 0.0;
1028
1029	table_parallelscan_initialize(heap,
1030								  ParallelTableScanFromGinBuildShared(ginshared),
1031								  snapshot);
1032
1033	/*
1034	 * Store shared tuplesort-private state, for which we reserved space.
1035	 * Then, initialize opaque state using tuplesort routine.
1036	 */ view japanese
1037	sharedsort = (Sharedsort *) shm_toc_allocate(pcxt->toc, estsort);
1038	tuplesort_initialize_shared(sharedsort, scantuplesortstates,
1039								pcxt->seg);
1040
1041	shm_toc_insert(pcxt->toc, PARALLEL_KEY_GIN_SHARED, ginshared);
1042	shm_toc_insert(pcxt->toc, PARALLEL_KEY_TUPLESORT, sharedsort);
1043
1044	/* Store query string for workers */ view japanese
1045	if (debug_query_string)
1046	{
1047		char	   *sharedquery;
1048
1049		sharedquery = (char *) shm_toc_allocate(pcxt->toc, querylen + 1);
1050		memcpy(sharedquery, debug_query_string, querylen + 1);
1051		shm_toc_insert(pcxt->toc, PARALLEL_KEY_QUERY_TEXT, sharedquery);
1052	}
1053
1054	/*
1055	 * Allocate space for each worker's WalUsage and BufferUsage; no need to
1056	 * initialize.
1057	 */ view japanese
1058	walusage = shm_toc_allocate(pcxt->toc,
1059								mul_size(sizeof(WalUsage), pcxt->nworkers));
1060	shm_toc_insert(pcxt->toc, PARALLEL_KEY_WAL_USAGE, walusage);
1061	bufferusage = shm_toc_allocate(pcxt->toc,
1062								   mul_size(sizeof(BufferUsage), pcxt->nworkers));
1063	shm_toc_insert(pcxt->toc, PARALLEL_KEY_BUFFER_USAGE, bufferusage);
1064
1065	/* Launch workers, saving status for leader/caller */ view japanese
1066	LaunchParallelWorkers(pcxt);
1067	ginleader->pcxt = pcxt;
1068	ginleader->nparticipanttuplesorts = pcxt->nworkers_launched;
1069	if (leaderparticipates)
1070		ginleader->nparticipanttuplesorts++;
1071	ginleader->ginshared = ginshared;
1072	ginleader->sharedsort = sharedsort;
1073	ginleader->snapshot = snapshot;
1074	ginleader->walusage = walusage;
1075	ginleader->bufferusage = bufferusage;
1076
1077	/* If no workers were successfully launched, back out (do serial build) */ view japanese
1078	if (pcxt->nworkers_launched == 0)
1079	{
1080		_gin_end_parallel(ginleader, NULL);
1081		return;
1082	}
1083
1084	/* Save leader state now that it's clear build will be parallel */ view japanese
1085	buildstate->bs_leader = ginleader;
1086
1087	/* Join heap scan ourselves */ view japanese
1088	if (leaderparticipates)
1089		_gin_leader_participate_as_worker(buildstate, heap, index);
1090
1091	/*
1092	 * Caller needs to wait for all launched workers when we return.  Make
1093	 * sure that the failure-to-start case will not hang forever.
1094	 */ view japanese
1095	WaitForParallelWorkersToAttach(pcxt);
1096}

1099 * Shut down workers, destroy parallel context, and end parallel mode.
1100 */ view japanese
1101static void
1102_gin_end_parallel(GinLeader *ginleader, GinBuildState *state)
1103{

1105
1106	/* Shutdown worker processes */ view japanese
1107	WaitForParallelWorkersToFinish(ginleader->pcxt);
1108

1110	 * Next, accumulate WAL usage.  (This must wait for the workers to finish,

1123/*
1124 * Within leader, wait for end of heap scan.
1125 *
1126 * When called, parallel heap scan started by _gin_begin_parallel() will
1127 * already be underway within worker processes (when leader participates
1128 * as a worker, we should end up here just as workers are finishing).
1129 *
1130 * Returns the total number of heap tuples scanned.
1131 */ view japanese
1132static double
1133_gin_parallel_heapscan(GinBuildState *state)
1134{
1135	GinBuildShared *ginshared = state->bs_leader->ginshared;
1136	int			nparticipanttuplesorts;
1137
1138	nparticipanttuplesorts = state->bs_leader->nparticipanttuplesorts;
1139	for (;;)
1140	{
1141		SpinLockAcquire(&ginshared->mutex);
1142		if (ginshared->nparticipantsdone == nparticipanttuplesorts)
1143		{
1144			/* copy the data into leader state */ view japanese
1145			state->bs_reltuples = ginshared->reltuples;
1146			state->bs_numtuples = ginshared->indtuples;
1147
1148			SpinLockRelease(&ginshared->mutex);
1149			break;
1150		}
1151		SpinLockRelease(&ginshared->mutex);
1152
1153		ConditionVariableSleep(&ginshared->workersdonecv,
1154							   WAIT_EVENT_PARALLEL_CREATE_INDEX_SCAN);
1155	}
1156
1157	ConditionVariableCancelSleep();
1158
1159	return state->bs_reltuples;
1160}

1195/*
1196 * Check that TID array contains valid values, and that it's sorted (if we
1197 * expect it to be).
1198 */ view japanese
1199static void
1200AssertCheckItemPointers(GinBuffer *buffer)
1201{
1202#ifdef USE_ASSERT_CHECKING
1203	/* we should not have a buffer with no TIDs to sort */ view japanese
1204	Assert(buffer->items != NULL);
1205	Assert(buffer->nitems > 0);

1208	{
1209		Assert(ItemPointerIsValid(&buffer->items[i]));
1210
1211		/* don't check ordering for the first TID item */ view japanese
1212		if (i == 0)
1213			continue;
1214
1215		Assert(ItemPointerCompare(&buffer->items[i - 1], &buffer->items[i]) < 0);
1216	}
1217#endif
1218}

1224 * or not empty, the fields need to agree). If there are items, check ordering.

1245/*
1246 * GinBufferInit
1247 *		Initialize buffer to store tuples for a GIN index.
1248 *
1249 * Initialize the buffer used to accumulate TID for a single key at a time
1250 * (we process the data sorted), so we know when we received all data for
1251 * a given key.
1252 *
1253 * Initializes sort support procedures for all index attributes.
1254 */ view japanese
1255static GinBuffer *
1256GinBufferInit(Relation index)
1257{
1258	GinBuffer  *buffer = palloc0(sizeof(GinBuffer));
1259	int			i,
1260				nKeys;
1261	TupleDesc	desc = RelationGetDescr(index);
1262
1263	/*
1264	 * How many items can we fit into the memory limit? We don't want to end
1265	 * with too many TIDs. and 64kB seems more than enough. But maybe this
1266	 * should be tied to maintenance_work_mem or something like that?
1267	 */ view japanese
1268	buffer->maxitems = (64 * 1024L) / sizeof(ItemPointerData);
1269
1270	nKeys = IndexRelationGetNumberOfKeyAttributes(index);
1271
1272	buffer->ssup = palloc0(sizeof(SortSupportData) * nKeys);
1273
1274	/*
1275	 * Lookup ordering operator for the index key data type, and initialize
1276	 * the sort support function.
1277	 */ view japanese
1278	for (i = 0; i < nKeys; i++)
1279	{
1280		Oid			cmpFunc;
1281		SortSupport sortKey = &buffer->ssup[i];
1282		Form_pg_attribute att = TupleDescAttr(desc, i);
1283
1284		sortKey->ssup_cxt = CurrentMemoryContext;
1285		sortKey->ssup_collation = index->rd_indcollation[i];
1286
1287		if (!OidIsValid(sortKey->ssup_collation))
1288			sortKey->ssup_collation = DEFAULT_COLLATION_OID;
1289
1290		sortKey->ssup_nulls_first = false;
1291		sortKey->ssup_attno = i + 1;
1292		sortKey->abbreviate = false;
1293
1294		Assert(sortKey->ssup_attno != 0);
1295
1296		/*
1297		 * If the compare proc isn't specified in the opclass definition, look
1298		 * up the index key type's default btree comparator.
1299		 */ view japanese
1300		cmpFunc = index_getprocid(index, i + 1, GIN_COMPARE_PROC);
1301		if (cmpFunc == InvalidOid)
1302		{
1303			TypeCacheEntry *typentry;
1304

1331 *		Can the buffer store TIDs for the provided GIN tuple (same key)?
1332 *
1333 * Compare if the tuple matches the already accumulated data in the GIN

1335 *
1336 * Returns true if the key matches, and the TID belongs to the buffer, or
1337 * false if the key does not match.
1338 */ view japanese

1370	r = ApplySortComparator(buffer->key, false,
1371							tupkey, false,
1372							&buffer->ssup[buffer->attnum - 1]);

1377/*
1378 * GinBufferShouldTrim
1379 *		Should we trim the list of item pointers?
1380 *
1381 * By trimming we understand writing out and removing the tuple IDs that
1382 * we know can't change by future merges. We can deduce the TID up to which
1383 * this is guaranteed from the "first" TID in each GIN tuple, which provides
1384 * a "horizon" (for a given key) thanks to the sort.
1385 *
1386 * We don't want to do this too often - compressing longer TID lists is more
1387 * efficient. But we also don't want to accumulate too many TIDs, for two
1388 * reasons. First, it consumes memory and we might exceed maintenance_work_mem
1389 * (or whatever limit applies), even if that's unlikely because TIDs are very
1390 * small so we can fit a lot of them. Second, and more importantly, long TID
1391 * lists are an issue if the scan wraps around, because a key may get a very
1392 * wide list (with min/max TID for that key), forcing "full" mergesorts for
1393 * every list merged into it (instead of the efficient append).
1394 *
1395 * So we look at two things when deciding if to trim - if the resulting list
1396 * (after adding TIDs from the new tuple) would be too long, and if there is
1397 * enough TIDs to trim (with values less than "first" TID from the new tuple),
1398 * we do the trim. By enough we mean at least 128 TIDs (mostly an arbitrary
1399 * number).
1400 */ view japanese
1401static bool
1402GinBufferShouldTrim(GinBuffer *buffer, GinTuple *tup)
1403{
1404	/* not enough TIDs to trim (1024 is somewhat arbitrary number) */ view japanese
1405	if (buffer->nfrozen < 1024)
1406		return false;
1407
1408	/* no need to trim if we have not hit the memory limit yet */ view japanese
1409	if ((buffer->nitems + tup->nitems) < buffer->maxitems)
1410		return false;
1411
1412	/*
1413	 * OK, we have enough frozen TIDs to flush, and we have hit the memory
1414	 * limit, so it's time to write it out.
1415	 */ view japanese
1416	return true;
1417}

1419/*
1420 * GinBufferStoreTuple
1421 *		Add data (especially TID list) from a GIN tuple to the buffer.
1422 *
1423 * The buffer is expected to be empty (in which case it's initialized), or
1424 * having the same key. The TID values from the tuple are combined with the
1425 * stored values using a merge sort.
1426 *
1427 * The tuples (for the same key) are expected to be sorted by first TID. But
1428 * this does not guarantee the lists do not overlap, especially in the leader,
1429 * because the workers process interleaving data. There should be no overlaps
1430 * in a single worker - it could happen when the parallel scan wraps around,
1431 * but we detect that and flush the data (see ginBuildCallbackParallel).
1432 *
1433 * By sorting the GinTuple not only by key, but also by the first TID, we make
1434 * it more less likely the lists will overlap during merge. We merge them using
1435 * mergesort, but it's cheaper to just append one list to the other.
1436 *
1437 * How often can the lists overlap? There should be no overlaps in workers,
1438 * and in the leader we can see overlaps between lists built by different
1439 * workers. But the workers merge the items as much as possible, so there
1440 * should not be too many.
1441 */ view japanese
1442static void
1443GinBufferStoreTuple(GinBuffer *buffer, GinTuple *tup)
1444{
1445	ItemPointerData *items;
1446	Datum		key;
1447
1448	AssertCheckGinBuffer(buffer);
1449
1450	key = _gin_parse_tuple_key(tup);
1451	items = _gin_parse_tuple_items(tup);
1452
1453	/* if the buffer is empty, set the fields (and copy the key) */ view japanese
1454	if (GinBufferIsEmpty(buffer))
1455	{
1456		buffer->category = tup->category;
1457		buffer->keylen = tup->keylen;
1458		buffer->attnum = tup->attrnum;
1459
1460		buffer->typlen = tup->typlen;
1461		buffer->typbyval = tup->typbyval;
1462
1463		if (tup->category == GIN_CAT_NORM_KEY)
1464			buffer->key = datumCopy(key, buffer->typbyval, buffer->typlen);
1465		else
1466			buffer->key = (Datum) 0;
1467	}
1468
1469	/*
1470	 * Try freeze TIDs at the beginning of the list, i.e. exclude them from
1471	 * the mergesort. We can do that with TIDs before the first TID in the new
1472	 * tuple we're about to add into the buffer.
1473	 *
1474	 * We do this incrementally when adding data into the in-memory buffer,
1475	 * and not later (e.g. when hitting a memory limit), because it allows us
1476	 * to skip the frozen data during the mergesort, making it cheaper.
1477	 */ view japanese
1478
1479	/*
1480	 * Check if the last TID in the current list is frozen. This is the case
1481	 * when merging non-overlapping lists, e.g. in each parallel worker.
1482	 */ view japanese
1483	if ((buffer->nitems > 0) &&
1484		(ItemPointerCompare(&buffer->items[buffer->nitems - 1],
1485							GinTupleGetFirst(tup)) == 0))
1486		buffer->nfrozen = buffer->nitems;
1487
1488	/*
1489	 * Now find the last TID we know to be frozen, i.e. the last TID right
1490	 * before the new GIN tuple.
1491	 *
1492	 * Start with the first not-yet-frozen tuple, and walk until we find the
1493	 * first TID that's higher. If we already know the whole list is frozen
1494	 * (i.e. nfrozen == nitems), this does nothing.
1495	 *
1496	 * XXX This might do a binary search for sufficiently long lists, but it
1497	 * does not seem worth the complexity. Overlapping lists should be rare
1498	 * common, TID comparisons are cheap, and we should quickly freeze most of
1499	 * the list.
1500	 */ view japanese
1501	for (int i = buffer->nfrozen; i < buffer->nitems; i++)
1502	{
1503		/* Is the TID after the first TID of the new tuple? Can't freeze. */ view japanese
1504		if (ItemPointerCompare(&buffer->items[i],
1505							   GinTupleGetFirst(tup)) > 0)
1506			break;
1507
1508		buffer->nfrozen++;
1509	}
1510
1511	/* add the new TIDs into the buffer, combine using merge-sort */ view japanese
1512	{
1513		int			nnew;
1514		ItemPointer new;
1515
1516		/*
1517		 * Resize the array - we do this first, because we'll dereference the
1518		 * first unfrozen TID, which would fail if the array is NULL. We'll
1519		 * still pass 0 as number of elements in that array though.
1520		 */ view japanese
1521		if (buffer->items == NULL)
1522			buffer->items = palloc((buffer->nitems + tup->nitems) * sizeof(ItemPointerData));
1523		else
1524			buffer->items = repalloc(buffer->items,
1525									 (buffer->nitems + tup->nitems) * sizeof(ItemPointerData));
1526
1527		new = ginMergeItemPointers(&buffer->items[buffer->nfrozen], /* first unfrozen */ view japanese
1528								   (buffer->nitems - buffer->nfrozen),	/* num of unfrozen */ view japanese
1529								   items, tup->nitems, &nnew);
1530
1531		Assert(nnew == (tup->nitems + (buffer->nitems - buffer->nfrozen)));
1532
1533		memcpy(&buffer->items[buffer->nfrozen], new,
1534			   nnew * sizeof(ItemPointerData));
1535
1536		pfree(new);
1537
1538		buffer->nitems += tup->nitems;
1539
1540		AssertCheckItemPointers(buffer);
1541	}
1542
1543	/* free the decompressed TID list */ view japanese
1544	pfree(items);
1545}

1628/*
1629 * Within leader, wait for end of heap scan and merge per-worker results.
1630 *
1631 * After waiting for all workers to finish, merge the per-worker results into
1632 * the complete index. The results from each worker are sorted by block number
1633 * (start of the page range). While combining the per-worker results we merge
1634 * summaries for the same page range, and also fill-in empty summaries for
1635 * ranges without any tuples.
1636 *
1637 * Returns the total number of heap tuples scanned.
1638 */ view japanese
1639static double
1640_gin_parallel_merge(GinBuildState *state)
1641{
1642	GinTuple   *tup;
1643	Size		tuplen;
1644	double		reltuples = 0;
1645	GinBuffer  *buffer;
1646
1647	/* GIN tuples from workers, merged by leader */ view japanese
1648	double		numtuples = 0;
1649
1650	/* wait for workers to scan table and produce partial results */ view japanese
1651	reltuples = _gin_parallel_heapscan(state);
1652
1653	/* Execute the sort */ view japanese
1654	pgstat_progress_update_param(PROGRESS_CREATEIDX_SUBPHASE,
1655								 PROGRESS_GIN_PHASE_PERFORMSORT_2);
1656
1657	/* do the actual sort in the leader */ view japanese
1658	tuplesort_performsort(state->bs_sortstate);
1659
1660	/*
1661	 * Initialize buffer to combine entries for the same key.
1662	 *
1663	 * The leader is allowed to use the whole maintenance_work_mem buffer to
1664	 * combine data. The parallel workers already completed.
1665	 */ view japanese
1666	buffer = GinBufferInit(state->ginstate.index);
1667
1668	/*
1669	 * Set the progress target for the next phase.  Reset the block number
1670	 * values set by table_index_build_scan
1671	 */ view japanese
1672	{
1673		const int	progress_index[] = {
1674			PROGRESS_CREATEIDX_SUBPHASE,
1675			PROGRESS_CREATEIDX_TUPLES_TOTAL,
1676			PROGRESS_SCAN_BLOCKS_TOTAL,
1677			PROGRESS_SCAN_BLOCKS_DONE
1678		};
1679		const int64 progress_vals[] = {
1680			PROGRESS_GIN_PHASE_MERGE_2,
1681			state->bs_numtuples,
1682			0, 0
1683		};
1684
1685		pgstat_progress_update_multi_param(4, progress_index, progress_vals);
1686	}
1687
1688	/*
1689	 * Read the GIN tuples from the shared tuplesort, sorted by category and
1690	 * key. That probably gives us order matching how data is organized in the
1691	 * index.
1692	 *
1693	 * We don't insert the GIN tuples right away, but instead accumulate as
1694	 * many TIDs for the same key as possible, and then insert that at once.
1695	 * This way we don't need to decompress/recompress the posting lists, etc.
1696	 */ view japanese
1697	while ((tup = tuplesort_getgintuple(state->bs_sortstate, &tuplen, true)) != NULL)
1698	{
1699		MemoryContext oldCtx;
1700
1701		CHECK_FOR_INTERRUPTS();
1702
1703		/*
1704		 * If the buffer can accept the new GIN tuple, just store it there and
1705		 * we're done. If it's a different key (or maybe too much data) flush
1706		 * the current contents into the index first.
1707		 */
1708		if (!GinBufferCanAddKey(buffer, tup))
1709		{
1710			/*
1711			 * Buffer is not empty and it's storing a different key - flush
1712			 * the data into the insert, and start a new entry for current
1713			 * GinTuple.
1714			 */
1715			AssertCheckItemPointers(buffer);
1716
1717			oldCtx = MemoryContextSwitchTo(state->tmpCtx);
1718
1719			ginEntryInsert(&state->ginstate,
1720						   buffer->attnum, buffer->key, buffer->category,
1721						   buffer->items, buffer->nitems, &state->buildStats);
1722
1723			MemoryContextSwitchTo(oldCtx);
1724			MemoryContextReset(state->tmpCtx);
1725
1726			/* discard the existing data */
1727			GinBufferReset(buffer);
1728		}
1729
1730		/*
1731		 * We're about to add a GIN tuple to the buffer - check the memory
1732		 * limit first, and maybe write out some of the data into the index
1733		 * first, if needed (and possible). We only flush the part of the TID
1734		 * list that we know won't change, and only if there's enough data for
1735		 * compression to work well.
1736		 */
1737		if (GinBufferShouldTrim(buffer, tup))
1738		{
1739			Assert(buffer->nfrozen > 0);
1740
1741			/*
1742			 * Buffer is not empty and it's storing a different key - flush
1743			 * the data into the insert, and start a new entry for current
1744			 * GinTuple.
1745			 */
1746			AssertCheckItemPointers(buffer);
1747
1748			oldCtx = MemoryContextSwitchTo(state->tmpCtx);
1749
1750			ginEntryInsert(&state->ginstate,
1751						   buffer->attnum, buffer->key, buffer->category,
1752						   buffer->items, buffer->nfrozen, &state->buildStats);
1753
1754			MemoryContextSwitchTo(oldCtx);
1755			MemoryContextReset(state->tmpCtx);
1756
1757			/* truncate the data we've just discarded */
1758			GinBufferTrim(buffer);
1759		}
1760
1761		/*
1762		 * Remember data for the current tuple (either remember the new key,
1763		 * or append if to the existing data).
1764		 */
1765		GinBufferStoreTuple(buffer, tup);
1766
1767		/* Report progress */
1768		pgstat_progress_update_param(PROGRESS_CREATEIDX_TUPLES_DONE,
1769									 ++numtuples);
1770	}
1771
1772	/* flush data remaining in the buffer (for the last key) */
1773	if (!GinBufferIsEmpty(buffer))
1774	{
1775		AssertCheckItemPointers(buffer);
1776
1777		ginEntryInsert(&state->ginstate,
1778					   buffer->attnum, buffer->key, buffer->category,
1779					   buffer->items, buffer->nitems, &state->buildStats);
1780
1781		/* discard the existing data */
1782		GinBufferReset(buffer);
1783
1784		/* Report progress */ view japanese
1785		pgstat_progress_update_param(PROGRESS_CREATEIDX_TUPLES_DONE,
1786									 ++numtuples);
1787	}
1788
1789	/* relase all the memory */
1790	GinBufferFree(buffer);
1791
1792	tuplesort_end(state->bs_sortstate);
1793
1794	return reltuples;
1795}

1831/*
1832 * _gin_process_worker_data
1833 *		First phase of the key merging, happening in the worker.
1834 *
1835 * Depending on the number of distinct keys, the TID lists produced by the
1836 * callback may be very short (due to frequent evictions in the callback).
1837 * But combining many tiny lists is expensive, so we try to do as much as
1838 * possible in the workers and only then pass the results to the leader.
1839 *
1840 * We read the tuples sorted by the key, and merge them into larger lists.
1841 * At the moment there's no memory limit, so this will just produce one
1842 * huge (sorted) list per key in each worker. Which means the leader will
1843 * do a very limited number of mergesorts, which is good.
1844 */ view japanese
1845static void
1846_gin_process_worker_data(GinBuildState *state, Tuplesortstate *worker_sort,
1847						 bool progress)
1848{
1849	GinTuple   *tup;
1850	Size		tuplen;
1851
1852	GinBuffer  *buffer;
1853
1854	/*
1855	 * Initialize buffer to combine entries for the same key.
1856	 *
1857	 * The workers are limited to the same amount of memory as during the sort
1858	 * in ginBuildCallbackParallel. But this probably should be the 32MB used
1859	 * during planning, just like there.
1860	 */ view japanese
1861	buffer = GinBufferInit(state->ginstate.index);
1862
1863	/* sort the raw per-worker data */ view japanese
1864	if (progress)
1865		pgstat_progress_update_param(PROGRESS_CREATEIDX_SUBPHASE,
1866									 PROGRESS_GIN_PHASE_PERFORMSORT_1);
1867
1868	tuplesort_performsort(state->bs_worker_sort);
1869
1870	/* reset the number of GIN tuples produced by this worker */ view japanese
1871	state->bs_numtuples = 0;
1872
1873	if (progress)
1874		pgstat_progress_update_param(PROGRESS_CREATEIDX_SUBPHASE,
1875									 PROGRESS_GIN_PHASE_MERGE_1);
1876
1877	/*
1878	 * Read the GIN tuples from the shared tuplesort, sorted by the key, and
1879	 * merge them into larger chunks for the leader to combine.
1880	 */ view japanese
1881	while ((tup = tuplesort_getgintuple(worker_sort, &tuplen, true)) != NULL)
1882	{
1883
1884		CHECK_FOR_INTERRUPTS();
1885
1886		/*
1887		 * If the buffer can accept the new GIN tuple, just store it there and
1888		 * we're done. If it's a different key (or maybe too much data) flush
1889		 * the current contents into the index first.
1890		 */ view japanese
1891		if (!GinBufferCanAddKey(buffer, tup))
1892		{
1893			GinTuple   *ntup;
1894			Size		ntuplen;
1895
1896			/*
1897			 * Buffer is not empty and it's storing a different key - flush
1898			 * the data into the insert, and start a new entry for current
1899			 * GinTuple.
1900			 */
1901			AssertCheckItemPointers(buffer);
1902
1903			ntup = _gin_build_tuple(buffer->attnum, buffer->category,
1904									buffer->key, buffer->typlen, buffer->typbyval,
1905									buffer->items, buffer->nitems, &ntuplen);
1906
1907			tuplesort_putgintuple(state->bs_sortstate, ntup, ntuplen);
1908			state->bs_numtuples++;
1909
1910			pfree(ntup);
1911
1912			/* discard the existing data */
1913			GinBufferReset(buffer);
1914		}
1915
1916		/*
1917		 * We're about to add a GIN tuple to the buffer - check the memory
1918		 * limit first, and maybe write out some of the data into the index
1919		 * first, if needed (and possible). We only flush the part of the TID
1920		 * list that we know won't change, and only if there's enough data for
1921		 * compression to work well.
1922		 */ view japanese
1923		if (GinBufferShouldTrim(buffer, tup))

1987 *

1989 * expressed in KBs.
1990 *

1992 *
1993 * Before feeding data into a shared tuplesort (for the leader process),
1994 * the workers process data in two phases.
1995 *
1996 * 1) A worker reads a portion of rows from the table, accumulates entries
1997 * in memory, and flushes them into a private tuplesort (e.g. because of
1998 * using too much memory).
1999 *
2000 * 2) The private tuplesort gets sorted (by key and TID), the worker reads
2001 * the data again, and combines the entries as much as possible. This has
2002 * to happen eventually, and this way it's done in workers in parallel.
2003 *

2005 * that the leader can process them.

2016 * many tiny ones.
