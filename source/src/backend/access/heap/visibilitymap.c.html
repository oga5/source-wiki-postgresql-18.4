1/*-------------------------------------------------------------------------
2 *
3 * visibilitymap.c
4 *	  bitmap for tracking visibility of heap tuples
5 *
6 * Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
7 * Portions Copyright (c) 1994, Regents of the University of California
8 *
9 *
10 * IDENTIFICATION
11 *	  src/backend/access/heap/visibilitymap.c
12 *
13 * INTERFACE ROUTINES
14 *		visibilitymap_clear  - clear bits for one page in the visibility map
15 *		visibilitymap_pin	 - pin a map page for setting a bit
16 *		visibilitymap_pin_ok - check whether correct map page is already pinned
17 *		visibilitymap_set	 - set a bit in a previously pinned page
18 *		visibilitymap_get_status - get status of bits
19 *		visibilitymap_count  - count number of bits set in visibility map
20 *		visibilitymap_prepare_truncate -
21 *			prepare for truncation of the visibility map
22 *
23 * NOTES
24 *
25 * The visibility map is a bitmap with two bits (all-visible and all-frozen)
26 * per heap page. A set all-visible bit means that all tuples on the page are
27 * known visible to all transactions, and therefore the page doesn't need to
28 * be vacuumed. A set all-frozen bit means that all tuples on the page are
29 * completely frozen, and therefore the page doesn't need to be vacuumed even
30 * if whole table scanning vacuum is required (e.g. anti-wraparound vacuum).
31 * The all-frozen bit must be set only when the page is already all-visible.
32 *
33 * The map is conservative in the sense that we make sure that whenever a bit
34 * is set, we know the condition is true, but if a bit is not set, it might or
35 * might not be true.
36 *
37 * Clearing visibility map bits is not separately WAL-logged.  The callers
38 * must make sure that whenever a bit is cleared, the bit is cleared on WAL
39 * replay of the updating operation as well.
40 *
41 * When we *set* a visibility map during VACUUM, we must write WAL.  This may
42 * seem counterintuitive, since the bit is basically a hint: if it is clear,
43 * it may still be the case that every tuple on the page is visible to all
44 * transactions; we just don't know that for certain.  The difficulty is that
45 * there are two bits which are typically set together: the PD_ALL_VISIBLE bit
46 * on the page itself, and the visibility map bit.  If a crash occurs after the
47 * visibility map page makes it to disk and before the updated heap page makes
48 * it to disk, redo must set the bit on the heap page.  Otherwise, the next
49 * insert, update, or delete on the heap page will fail to realize that the
50 * visibility map bit must be cleared, possibly causing index-only scans to
51 * return wrong answers.
52 *
53 * VACUUM will normally skip pages for which the visibility map bit is set;
54 * such pages can't contain any dead tuples and therefore don't need vacuuming.
55 *
56 * LOCKING
57 *
58 * In heapam.c, whenever a page is modified so that not all tuples on the
59 * page are visible to everyone anymore, the corresponding bit in the
60 * visibility map is cleared. In order to be crash-safe, we need to do this
61 * while still holding a lock on the heap page and in the same critical
62 * section that logs the page modification. However, we don't want to hold
63 * the buffer lock over any I/O that may be required to read in the visibility
64 * map page.  To avoid this, we examine the heap page before locking it;
65 * if the page-level PD_ALL_VISIBLE bit is set, we pin the visibility map
66 * bit.  Then, we lock the buffer.  But this creates a race condition: there
67 * is a possibility that in the time it takes to lock the buffer, the
68 * PD_ALL_VISIBLE bit gets set.  If that happens, we have to unlock the
69 * buffer, pin the visibility map page, and relock the buffer.  This shouldn't
70 * happen often, because only VACUUM currently sets visibility map bits,
71 * and the race will only occur if VACUUM processes a given page at almost
72 * exactly the same time that someone tries to further modify it.
73 *
74 * To set a bit, you need to hold a lock on the heap page. That prevents
75 * the race condition where VACUUM sees that all tuples on the page are
76 * visible to everyone, but another backend modifies the page before VACUUM
77 * sets the bit in the visibility map.
78 *
79 * When a bit is set, the LSN of the visibility map page is updated to make
80 * sure that the visibility map update doesn't get written to disk before the
81 * WAL record of the changes that made it possible to set the bit is flushed.
82 * But when a bit is cleared, we don't have to do that because it's always
83 * safe to clear a bit in the map from correctness point of view.
84 *
85 *-------------------------------------------------------------------------
86 */

103/*
104 * Size of the bitmap on each visibility map page, in bytes. There's no
105 * extra headers, so the whole page minus the standard page header is
106 * used for the bitmap.
107 */ view japanese
108#define MAPSIZE (BLCKSZ - MAXALIGN(SizeOfPageHeaderData))
109
110/* Number of heap blocks we can represent in one byte */ view japanese
111#define HEAPBLOCKS_PER_BYTE (BITS_PER_BYTE / BITS_PER_HEAPBLOCK)
112
113/* Number of heap blocks we can represent in one visibility map page. */ view japanese
114#define HEAPBLOCKS_PER_PAGE (MAPSIZE * HEAPBLOCKS_PER_BYTE)
115
116/* Mapping from heap block number to the right bit in the visibility map */ view japanese
117#define HEAPBLK_TO_MAPBLOCK(x) ((x) / HEAPBLOCKS_PER_PAGE)
118#define HEAPBLK_TO_MAPBLOCK_LIMIT(x) \
119	(((x) + HEAPBLOCKS_PER_PAGE - 1) / HEAPBLOCKS_PER_PAGE)
120#define HEAPBLK_TO_MAPBYTE(x) (((x) % HEAPBLOCKS_PER_PAGE) / HEAPBLOCKS_PER_BYTE)
121#define HEAPBLK_TO_OFFSET(x) (((x) % HEAPBLOCKS_PER_BYTE) * BITS_PER_HEAPBLOCK)
122
123/* Masks for counting subsets of bits in the visibility map. */ view japanese
124#define VISIBLE_MASK8	(0x55)	/* The lower bit of each bit pair */ view japanese
125#define FROZEN_MASK8	(0xaa)	/* The upper bit of each bit pair */ view japanese
126
127/* prototypes for internal routines */ view japanese
128static Buffer vm_readbuf(Relation rel, BlockNumber blkno, bool extend);
129static Buffer vm_extend(Relation rel, BlockNumber vm_nblocks);

132/*
133 *	visibilitymap_clear - clear specified bits for one page in visibility map
134 *
135 * You must pass a buffer containing the correct map page to this function.
136 * Call visibilitymap_pin first to pin the right one. This function doesn't do
137 * any I/O.  Returns true if any bits have been cleared and false otherwise.
138 */ view japanese
139bool
140visibilitymap_clear(Relation rel, BlockNumber heapBlk, Buffer vmbuf, uint8 flags)
141{
142	BlockNumber mapBlock = HEAPBLK_TO_MAPBLOCK(heapBlk);
143	int			mapByte = HEAPBLK_TO_MAPBYTE(heapBlk);
144	int			mapOffset = HEAPBLK_TO_OFFSET(heapBlk);
145	uint8		mask = flags << mapOffset;
146	char	   *map;
147	bool		cleared = false;
148
149	/* Must never clear all_visible bit while leaving all_frozen bit set */ view japanese
150	Assert(flags & VISIBILITYMAP_VALID_BITS);
151	Assert(flags != VISIBILITYMAP_ALL_VISIBLE);
152
153#ifdef TRACE_VISIBILITYMAP
154	elog(DEBUG1, "vm_clear %s %d", RelationGetRelationName(rel), heapBlk);
155#endif
156
157	if (!BufferIsValid(vmbuf) || BufferGetBlockNumber(vmbuf) != mapBlock)
158		elog(ERROR, "wrong buffer passed to visibilitymap_clear");
159
160	LockBuffer(vmbuf, BUFFER_LOCK_EXCLUSIVE);
161	map = PageGetContents(BufferGetPage(vmbuf));
162
163	if (map[mapByte] & mask)
164	{
165		map[mapByte] &= ~mask;
166
167		MarkBufferDirty(vmbuf);
168		cleared = true;
169	}
170
171	LockBuffer(vmbuf, BUFFER_LOCK_UNLOCK);
172
173	return cleared;
174}

176/*
177 *	visibilitymap_pin - pin a map page for setting a bit
178 *
179 * Setting a bit in the visibility map is a two-phase operation. First, call
180 * visibilitymap_pin, to pin the visibility map page containing the bit for
181 * the heap page. Because that can require I/O to read the map page, you
182 * shouldn't hold a lock on the heap page while doing that. Then, call
183 * visibilitymap_set to actually set the bit.
184 *
185 * On entry, *vmbuf should be InvalidBuffer or a valid buffer returned by
186 * an earlier call to visibilitymap_pin or visibilitymap_get_status on the same
187 * relation. On return, *vmbuf is a valid buffer with the map page containing
188 * the bit for heapBlk.
189 *
190 * If the page doesn't exist in the map file yet, it is extended.
191 */ view japanese
192void
193visibilitymap_pin(Relation rel, BlockNumber heapBlk, Buffer *vmbuf)
194{
195	BlockNumber mapBlock = HEAPBLK_TO_MAPBLOCK(heapBlk);
196
197	/* Reuse the old pinned buffer if possible */
198	if (BufferIsValid(*vmbuf))
199	{
200		if (BufferGetBlockNumber(*vmbuf) == mapBlock)
201			return;
202
203		ReleaseBuffer(*vmbuf);
204	}
205	*vmbuf = vm_readbuf(rel, mapBlock, true);
206}

208/*
209 *	visibilitymap_pin_ok - do we already have the correct page pinned?
210 *
211 * On entry, vmbuf should be InvalidBuffer or a valid buffer returned by
212 * an earlier call to visibilitymap_pin or visibilitymap_get_status on the same
213 * relation.  The return value indicates whether the buffer covers the
214 * given heapBlk.
215 */ view japanese
216bool
217visibilitymap_pin_ok(BlockNumber heapBlk, Buffer vmbuf)
218{
219	BlockNumber mapBlock = HEAPBLK_TO_MAPBLOCK(heapBlk);
220
221	return BufferIsValid(vmbuf) && BufferGetBlockNumber(vmbuf) == mapBlock;
222}

224/*
225 *	visibilitymap_set - set bit(s) on a previously pinned page
226 *
227 * recptr is the LSN of the XLOG record we're replaying, if we're in recovery,
228 * or InvalidXLogRecPtr in normal running.  The VM page LSN is advanced to the
229 * one provided; in normal running, we generate a new XLOG record and set the
230 * page LSN to that value (though the heap page's LSN may *not* be updated;
231 * see below).  cutoff_xid is the largest xmin on the page being marked
232 * all-visible; it is needed for Hot Standby, and can be InvalidTransactionId
233 * if the page contains no tuples.  It can also be set to InvalidTransactionId
234 * when a page that is already all-visible is being marked all-frozen.
235 *
236 * Caller is expected to set the heap page's PD_ALL_VISIBLE bit before calling
237 * this function. Except in recovery, caller should also pass the heap
238 * buffer. When checksums are enabled and we're not in recovery, we must add
239 * the heap buffer to the WAL chain to protect it from being torn.
240 *
241 * You must pass a buffer containing the correct map page to this function.
242 * Call visibilitymap_pin first to pin the right one. This function doesn't do
243 * any I/O.
244 *
245 * Returns the state of the page's VM bits before setting flags.
246 */ view japanese
247uint8
248visibilitymap_set(Relation rel, BlockNumber heapBlk, Buffer heapBuf,
249				  XLogRecPtr recptr, Buffer vmBuf, TransactionId cutoff_xid,
250				  uint8 flags)
251{
252	BlockNumber mapBlock = HEAPBLK_TO_MAPBLOCK(heapBlk);
253	uint32		mapByte = HEAPBLK_TO_MAPBYTE(heapBlk);
254	uint8		mapOffset = HEAPBLK_TO_OFFSET(heapBlk);
255	Page		page;
256	uint8	   *map;
257	uint8		status;
258
259#ifdef TRACE_VISIBILITYMAP
260	elog(DEBUG1, "vm_set %s %d", RelationGetRelationName(rel), heapBlk);
261#endif
262
263	Assert(InRecovery || XLogRecPtrIsInvalid(recptr));
264	Assert(InRecovery || PageIsAllVisible((Page) BufferGetPage(heapBuf)));
265	Assert((flags & VISIBILITYMAP_VALID_BITS) == flags);
266
267	/* Must never set all_frozen bit without also setting all_visible bit */ view japanese
268	Assert(flags != VISIBILITYMAP_ALL_FROZEN);
269
270	/* Check that we have the right heap page pinned, if present */ view japanese
271	if (BufferIsValid(heapBuf) && BufferGetBlockNumber(heapBuf) != heapBlk)
272		elog(ERROR, "wrong heap buffer passed to visibilitymap_set");
273
274	/* Check that we have the right VM page pinned */ view japanese
275	if (!BufferIsValid(vmBuf) || BufferGetBlockNumber(vmBuf) != mapBlock)
276		elog(ERROR, "wrong VM buffer passed to visibilitymap_set");
277
278	page = BufferGetPage(vmBuf);
279	map = (uint8 *) PageGetContents(page);
280	LockBuffer(vmBuf, BUFFER_LOCK_EXCLUSIVE);
281
282	status = (map[mapByte] >> mapOffset) & VISIBILITYMAP_VALID_BITS;
283	if (flags != status)
284	{
285		START_CRIT_SECTION();
286
287		map[mapByte] |= (flags << mapOffset);
288		MarkBufferDirty(vmBuf);
289
290		if (RelationNeedsWAL(rel))
291		{
292			if (XLogRecPtrIsInvalid(recptr))
293			{
294				Assert(!InRecovery);
295				recptr = log_heap_visible(rel, heapBuf, vmBuf, cutoff_xid, flags);
296
297				/*
298				 * If data checksums are enabled (or wal_log_hints=on), we
299				 * need to protect the heap page from being torn.
300				 *
301				 * If not, then we must *not* update the heap page's LSN. In
302				 * this case, the FPI for the heap page was omitted from the
303				 * WAL record inserted above, so it would be incorrect to
304				 * update the heap page's LSN.
305				 */ view japanese
306				if (XLogHintBitIsNeeded())
307				{
308					Page		heapPage = BufferGetPage(heapBuf);
309
310					PageSetLSN(heapPage, recptr);
311				}
312			}
313			PageSetLSN(page, recptr);
314		}
315
316		END_CRIT_SECTION();
317	}
318
319	LockBuffer(vmBuf, BUFFER_LOCK_UNLOCK);
320	return status;
321}

323/*
324 *	visibilitymap_get_status - get status of bits
325 *
326 * Are all tuples on heapBlk visible to all or are marked frozen, according
327 * to the visibility map?
328 *
329 * On entry, *vmbuf should be InvalidBuffer or a valid buffer returned by an
330 * earlier call to visibilitymap_pin or visibilitymap_get_status on the same
331 * relation. On return, *vmbuf is a valid buffer with the map page containing
332 * the bit for heapBlk, or InvalidBuffer. The caller is responsible for
333 * releasing *vmbuf after it's done testing and setting bits.
334 *
335 * NOTE: This function is typically called without a lock on the heap page,
336 * so somebody else could change the bit just after we look at it.  In fact,
337 * since we don't lock the visibility map page either, it's even possible that
338 * someone else could have changed the bit just before we look at it, but yet
339 * we might see the old value.  It is the caller's responsibility to deal with
340 * all concurrency issues!
341 */ view japanese
342uint8
343visibilitymap_get_status(Relation rel, BlockNumber heapBlk, Buffer *vmbuf)
344{
345	BlockNumber mapBlock = HEAPBLK_TO_MAPBLOCK(heapBlk);
346	uint32		mapByte = HEAPBLK_TO_MAPBYTE(heapBlk);
347	uint8		mapOffset = HEAPBLK_TO_OFFSET(heapBlk);
348	char	   *map;
349	uint8		result;
350
351#ifdef TRACE_VISIBILITYMAP
352	elog(DEBUG1, "vm_get_status %s %d", RelationGetRelationName(rel), heapBlk);
353#endif
354
355	/* Reuse the old pinned buffer if possible */ view japanese
356	if (BufferIsValid(*vmbuf))
357	{
358		if (BufferGetBlockNumber(*vmbuf) != mapBlock)
359		{
360			ReleaseBuffer(*vmbuf);
361			*vmbuf = InvalidBuffer;
362		}
363	}
364
365	if (!BufferIsValid(*vmbuf))
366	{
367		*vmbuf = vm_readbuf(rel, mapBlock, false);
368		if (!BufferIsValid(*vmbuf))
369			return false;
370	}
371
372	map = PageGetContents(BufferGetPage(*vmbuf));
373
374	/*
375	 * A single byte read is atomic.  There could be memory-ordering effects
376	 * here, but for performance reasons we make it the caller's job to worry
377	 * about that.
378	 */ view japanese
379	result = ((map[mapByte] >> mapOffset) & VISIBILITYMAP_VALID_BITS);
380	return result;
381}

383/*
384 *	visibilitymap_count  - count number of bits set in visibility map
385 *
386 * Note: we ignore the possibility of race conditions when the table is being
387 * extended concurrently with the call.  New pages added to the table aren't
388 * going to be marked all-visible or all-frozen, so they won't affect the result.
389 */ view japanese
390void
391visibilitymap_count(Relation rel, BlockNumber *all_visible, BlockNumber *all_frozen)
392{
393	BlockNumber mapBlock;
394	BlockNumber nvisible = 0;
395	BlockNumber nfrozen = 0;
396
397	/* all_visible must be specified */ view japanese
398	Assert(all_visible);
399
400	for (mapBlock = 0;; mapBlock++)
401	{
402		Buffer		mapBuffer;
403		uint64	   *map;
404
405		/*
406		 * Read till we fall off the end of the map.  We assume that any extra
407		 * bytes in the last page are zeroed, so we don't bother excluding
408		 * them from the count.
409		 */ view japanese
410		mapBuffer = vm_readbuf(rel, mapBlock, false);
411		if (!BufferIsValid(mapBuffer))
412			break;
413
414		/*
415		 * We choose not to lock the page, since the result is going to be
416		 * immediately stale anyway if anyone is concurrently setting or
417		 * clearing bits, and we only really need an approximate value.
418		 */ view japanese
419		map = (uint64 *) PageGetContents(BufferGetPage(mapBuffer));
420
421		nvisible += pg_popcount_masked((const char *) map, MAPSIZE, VISIBLE_MASK8);
422		if (all_frozen)
423			nfrozen += pg_popcount_masked((const char *) map, MAPSIZE, FROZEN_MASK8);
424
425		ReleaseBuffer(mapBuffer);
426	}
427
428	*all_visible = nvisible;
429	if (all_frozen)
430		*all_frozen = nfrozen;
431}

433/*
434 *	visibilitymap_prepare_truncate -
435 *			prepare for truncation of the visibility map
436 *
437 * nheapblocks is the new size of the heap.
438 *
439 * Return the number of blocks of new visibility map.
440 * If it's InvalidBlockNumber, there is nothing to truncate;
441 * otherwise the caller is responsible for calling smgrtruncate()
442 * to truncate the visibility map pages.
443 */ view japanese
444BlockNumber
445visibilitymap_prepare_truncate(Relation rel, BlockNumber nheapblocks)
446{
447	BlockNumber newnblocks;
448
449	/* last remaining block, byte, and bit */ view japanese
450	BlockNumber truncBlock = HEAPBLK_TO_MAPBLOCK(nheapblocks);
451	uint32		truncByte = HEAPBLK_TO_MAPBYTE(nheapblocks);
452	uint8		truncOffset = HEAPBLK_TO_OFFSET(nheapblocks);
453
454#ifdef TRACE_VISIBILITYMAP
455	elog(DEBUG1, "vm_truncate %s %d", RelationGetRelationName(rel), nheapblocks);
456#endif
457
458	/*
459	 * If no visibility map has been created yet for this relation, there's
460	 * nothing to truncate.
461	 */ view japanese
462	if (!smgrexists(RelationGetSmgr(rel), VISIBILITYMAP_FORKNUM))
463		return InvalidBlockNumber;
464
465	/*
466	 * Unless the new size is exactly at a visibility map page boundary, the
467	 * tail bits in the last remaining map page, representing truncated heap
468	 * blocks, need to be cleared. This is not only tidy, but also necessary
469	 * because we don't get a chance to clear the bits if the heap is extended
470	 * again.
471	 */ view japanese
472	if (truncByte != 0 || truncOffset != 0)
473	{
474		Buffer		mapBuffer;
475		Page		page;
476		char	   *map;
477
478		newnblocks = truncBlock + 1;
479
480		mapBuffer = vm_readbuf(rel, truncBlock, false);
481		if (!BufferIsValid(mapBuffer))
482		{
483			/* nothing to do, the file was already smaller */ view japanese
484			return InvalidBlockNumber;
485		}
486
487		page = BufferGetPage(mapBuffer);
488		map = PageGetContents(page);
489
490		LockBuffer(mapBuffer, BUFFER_LOCK_EXCLUSIVE);
491
492		/* NO EREPORT(ERROR) from here till changes are logged */ view japanese
493		START_CRIT_SECTION();
494
495		/* Clear out the unwanted bytes. */ view japanese
496		MemSet(&map[truncByte + 1], 0, MAPSIZE - (truncByte + 1));
497
498		/*----
499		 * Mask out the unwanted bits of the last remaining byte.
500		 *
501		 * ((1 << 0) - 1) = 00000000
502		 * ((1 << 1) - 1) = 00000001
503		 * ...
504		 * ((1 << 6) - 1) = 00111111
505		 * ((1 << 7) - 1) = 01111111
506		 *----
507		 */ view japanese
508		map[truncByte] &= (1 << truncOffset) - 1;
509
510		/*
511		 * Truncation of a relation is WAL-logged at a higher-level, and we
512		 * will be called at WAL replay. But if checksums are enabled, we need
513		 * to still write a WAL record to protect against a torn page, if the
514		 * page is flushed to disk before the truncation WAL record. We cannot
515		 * use MarkBufferDirtyHint here, because that will not dirty the page
516		 * during recovery.
517		 */ view japanese
518		MarkBufferDirty(mapBuffer);
519		if (!InRecovery && RelationNeedsWAL(rel) && XLogHintBitIsNeeded())
520			log_newpage_buffer(mapBuffer, false);
521
522		END_CRIT_SECTION();
523
524		UnlockReleaseBuffer(mapBuffer);
525	}
526	else
527		newnblocks = truncBlock;
528
529	if (smgrnblocks(RelationGetSmgr(rel), VISIBILITYMAP_FORKNUM) <= newnblocks)
530	{
531		/* nothing to do, the file was already smaller than requested size */ view japanese
532		return InvalidBlockNumber;
533	}
534
535	return newnblocks;
536}

553/*
554 * Read a visibility map page.
555 *
556 * If the page doesn't exist, InvalidBuffer is returned, or if 'extend' is
557 * true, the visibility map file is extended.
558 */ view japanese
559static Buffer
560vm_readbuf(Relation rel, BlockNumber blkno, bool extend)
561{
562	Buffer		buf;
563	SMgrRelation reln;
564
565	/*
566	 * Caution: re-using this smgr pointer could fail if the relcache entry
567	 * gets closed.  It's safe as long as we only do smgr-level operations
568	 * between here and the last use of the pointer.
569	 */ view japanese
570	reln = RelationGetSmgr(rel);
571
572	/*
573	 * If we haven't cached the size of the visibility map fork yet, check it
574	 * first.
575	 */ view japanese
576	if (reln->smgr_cached_nblocks[VISIBILITYMAP_FORKNUM] == InvalidBlockNumber)
577	{
578		if (smgrexists(reln, VISIBILITYMAP_FORKNUM))
579			smgrnblocks(reln, VISIBILITYMAP_FORKNUM);
580		else
581			reln->smgr_cached_nblocks[VISIBILITYMAP_FORKNUM] = 0;
582	}
583
584	/*
585	 * For reading we use ZERO_ON_ERROR mode, and initialize the page if
586	 * necessary. It's always safe to clear bits, so it's better to clear
587	 * corrupt pages than error out.
588	 *
589	 * We use the same path below to initialize pages when extending the
590	 * relation, as a concurrent extension can end up with vm_extend()
591	 * returning an already-initialized page.
592	 */ view japanese
593	if (blkno >= reln->smgr_cached_nblocks[VISIBILITYMAP_FORKNUM])
594	{
595		if (extend)
596			buf = vm_extend(rel, blkno + 1);
597		else
598			return InvalidBuffer;
599	}
600	else
601		buf = ReadBufferExtended(rel, VISIBILITYMAP_FORKNUM, blkno,
602								 RBM_ZERO_ON_ERROR, NULL);
603
604	/*
605	 * Initializing the page when needed is trickier than it looks, because of
606	 * the possibility of multiple backends doing this concurrently, and our
607	 * desire to not uselessly take the buffer lock in the normal path where
608	 * the page is OK.  We must take the lock to initialize the page, so
609	 * recheck page newness after we have the lock, in case someone else
610	 * already did it.  Also, because we initially check PageIsNew with no
611	 * lock, it's possible to fall through and return the buffer while someone
612	 * else is still initializing the page (i.e., we might see pd_upper as set
613	 * but other page header fields are still zeroes).  This is harmless for
614	 * callers that will take a buffer lock themselves, but some callers
615	 * inspect the page without any lock at all.  The latter is OK only so

/*
	 * 必要に応じてページを初期化することは、見た目よりも困難である。なぜなら、複数のバックエンドが同時にこれを行う可能性があることと、ページが正常な場合にバッファロックを無駄に取得したくないという我々の要望があるためである。ページを初期化するにはロックを取得しなければならないため、他の誰かが既に初期化を完了している場合に備えて、ロック取得後にページが新規であるかを再確認する必要がある。また、最初はロックなしでPageIsNewをチェックするため、他の誰かがまだページを初期化している最中に処理が進み、バッファが返される可能性がある（つまり、pd_upperは設定されているが、他のページヘッダーフィールドはまだゼロである可能性がある）。これは、自身でバッファロックを取得する呼び出し元にとっては無害であるが、一部の呼び出し元はロックなしでページを検査する。後者は、ページヘッダーが正しい内容を持っていることに依存しない限り問題ない。現在の使用方法は、PageGetContents()がそれを必要としないため安全である。
	 */

Original comment

/*
	 * Initializing the page when needed is trickier than it looks, because of
	 * the possibility of multiple backends doing this concurrently, and our
	 * desire to not uselessly take the buffer lock in the normal path where
	 * the page is OK.  We must take the lock to initialize the page, so
	 * recheck page newness after we have the lock, in case someone else
	 * already did it.  Also, because we initially check PageIsNew with no
	 * lock, it's possible to fall through and return the buffer while someone
	 * else is still initializing the page (i.e., we might see pd_upper as set
	 * but other page header fields are still zeroes).  This is harmless for
	 * callers that will take a buffer lock themselves, but some callers
	 * inspect the page without any lock at all.  The latter is OK only so
	 * long as it doesn't depend on the page header having correct contents.
	 * Current usage is safe because PageGetContents() does not require that.
	 */
