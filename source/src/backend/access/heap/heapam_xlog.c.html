26/*
27 * Replay XLOG_HEAP2_PRUNE_* records.
28 */
29static void
30heap_xlog_prune_freeze(XLogReaderState *record)
31{
32	XLogRecPtr	lsn = record->EndRecPtr;
33	char	   *maindataptr = XLogRecGetData(record);
34	xl_heap_prune xlrec;
35	Buffer		buffer;
36	RelFileLocator rlocator;
37	BlockNumber blkno;
38	XLogRedoAction action;
39
40	XLogRecGetBlockTag(record, 0, &rlocator, NULL, &blkno);
41	memcpy(&xlrec, maindataptr, SizeOfHeapPrune);
42	maindataptr += SizeOfHeapPrune;
43
44	/*
45	 * We will take an ordinary exclusive lock or a cleanup lock depending on
46	 * whether the XLHP_CLEANUP_LOCK flag is set.  With an ordinary exclusive
47	 * lock, we better not be doing anything that requires moving existing
48	 * tuple data.
49	 */
50	Assert((xlrec.flags & XLHP_CLEANUP_LOCK) != 0 ||
51		   (xlrec.flags & (XLHP_HAS_REDIRECTIONS | XLHP_HAS_DEAD_ITEMS)) == 0);
52
53	/*
54	 * We are about to remove and/or freeze tuples.  In Hot Standby mode,
55	 * ensure that there are no queries running for which the removed tuples
56	 * are still visible or which still consider the frozen xids as running.
57	 * The conflict horizon XID comes after xl_heap_prune.
58	 */
59	if ((xlrec.flags & XLHP_HAS_CONFLICT_HORIZON) != 0)
60	{
61		TransactionId snapshot_conflict_horizon;
62
63		/* memcpy() because snapshot_conflict_horizon is stored unaligned */
64		memcpy(&snapshot_conflict_horizon, maindataptr, sizeof(TransactionId));
65		maindataptr += sizeof(TransactionId);
66
67		if (InHotStandby)
68			ResolveRecoveryConflictWithSnapshot(snapshot_conflict_horizon,
69												(xlrec.flags & XLHP_IS_CATALOG_REL) != 0,
70												rlocator);
71	}
72
73	/*
74	 * If we have a full-page image, restore it and we're done.
75	 */
76	action = XLogReadBufferForRedoExtended(record, 0, RBM_NORMAL,
77										   (xlrec.flags & XLHP_CLEANUP_LOCK) != 0,
78										   &buffer);
79	if (action == BLK_NEEDS_REDO)
80	{
81		Page		page = (Page) BufferGetPage(buffer);
82		OffsetNumber *redirected;
83		OffsetNumber *nowdead;
84		OffsetNumber *nowunused;
85		int			nredirected;
86		int			ndead;
87		int			nunused;
88		int			nplans;
89		Size		datalen;
90		xlhp_freeze_plan *plans;
91		OffsetNumber *frz_offsets;
92		char	   *dataptr = XLogRecGetBlockData(record, 0, &datalen);
93
94		heap_xlog_deserialize_prune_and_freeze(dataptr, xlrec.flags,
95											   &nplans, &plans, &frz_offsets,
96											   &nredirected, &redirected,
97											   &ndead, &nowdead,
98											   &nunused, &nowunused);
99
100		/*
101		 * Update all line pointers per the record, and repair fragmentation
102		 * if needed.
103		 */
104		if (nredirected > 0 || ndead > 0 || nunused > 0)
105			heap_page_prune_execute(buffer,
106									(xlrec.flags & XLHP_CLEANUP_LOCK) == 0,
107									redirected, nredirected,
108									nowdead, ndead,
109									nowunused, nunused);
110
111		/* Freeze tuples */
112		for (int p = 0; p < nplans; p++)
113		{
114			HeapTupleFreeze frz;
115
116			/*
117			 * Convert freeze plan representation from WAL record into
118			 * per-tuple format used by heap_execute_freeze_tuple
119			 */
120			frz.xmax = plans[p].xmax;
121			frz.t_infomask2 = plans[p].t_infomask2;
122			frz.t_infomask = plans[p].t_infomask;
123			frz.frzflags = plans[p].frzflags;
124			frz.offset = InvalidOffsetNumber;	/* unused, but be tidy */
125
126			for (int i = 0; i < plans[p].ntuples; i++)
127			{
128				OffsetNumber offset = *(frz_offsets++);
129				ItemId		lp;
130				HeapTupleHeader tuple;
131
132				lp = PageGetItemId(page, offset);
133				tuple = (HeapTupleHeader) PageGetItem(page, lp);
134				heap_execute_freeze_tuple(tuple, &frz);
135			}
136		}
137
138		/* There should be no more data */
139		Assert((char *) frz_offsets == dataptr + datalen);
140
141		/*
142		 * Note: we don't worry about updating the page's prunability hints.
143		 * At worst this will cause an extra prune cycle to occur soon.
144		 */
145
146		PageSetLSN(page, lsn);
147		MarkBufferDirty(buffer);
148	}
149
150	/*
151	 * If we released any space or line pointers, update the free space map.
152	 *
153	 * Do this regardless of a full-page image being applied, since the FSM
154	 * data is not in the page anyway.
155	 */
156	if (BufferIsValid(buffer))
157	{
158		if (xlrec.flags & (XLHP_HAS_REDIRECTIONS |
159						   XLHP_HAS_DEAD_ITEMS |
160						   XLHP_HAS_NOW_UNUSED_ITEMS))
161		{
162			Size		freespace = PageGetHeapFreeSpace(BufferGetPage(buffer));
163
164			UnlockReleaseBuffer(buffer);
165
166			XLogRecordPageWithFreeSpace(rlocator, blkno, freespace);
167		}
168		else
169			UnlockReleaseBuffer(buffer);
170	}
171}

173/*
174 * Replay XLOG_HEAP2_VISIBLE records.
175 *
176 * The critical integrity requirement here is that we must never end up with
177 * a situation where the visibility map bit is set, and the page-level
178 * PD_ALL_VISIBLE bit is clear.  If that were to occur, then a subsequent
179 * page modification would fail to clear the visibility map bit.
180 */
181static void
182heap_xlog_visible(XLogReaderState *record)
183{
184	XLogRecPtr	lsn = record->EndRecPtr;
185	xl_heap_visible *xlrec = (xl_heap_visible *) XLogRecGetData(record);
186	Buffer		vmbuffer = InvalidBuffer;
187	Buffer		buffer;
188	Page		page;
189	RelFileLocator rlocator;
190	BlockNumber blkno;
191	XLogRedoAction action;
192
193	Assert((xlrec->flags & VISIBILITYMAP_XLOG_VALID_BITS) == xlrec->flags);
194
195	XLogRecGetBlockTag(record, 1, &rlocator, NULL, &blkno);
196
197	/*
198	 * If there are any Hot Standby transactions running that have an xmin
199	 * horizon old enough that this page isn't all-visible for them, they
200	 * might incorrectly decide that an index-only scan can skip a heap fetch.
201	 *
202	 * NB: It might be better to throw some kind of "soft" conflict here that
203	 * forces any index-only scan that is in flight to perform heap fetches,
204	 * rather than killing the transaction outright.
205	 */
206	if (InHotStandby)
207		ResolveRecoveryConflictWithSnapshot(xlrec->snapshotConflictHorizon,
208											xlrec->flags & VISIBILITYMAP_XLOG_CATALOG_REL,
209											rlocator);
210
211	/*
212	 * Read the heap page, if it still exists. If the heap file has dropped or
213	 * truncated later in recovery, we don't need to update the page, but we'd
214	 * better still update the visibility map.
215	 */
216	action = XLogReadBufferForRedo(record, 1, &buffer);
217	if (action == BLK_NEEDS_REDO)
218	{
219		/*
220		 * We don't bump the LSN of the heap page when setting the visibility
221		 * map bit (unless checksums or wal_hint_bits is enabled, in which
222		 * case we must). This exposes us to torn page hazards, but since
223		 * we're not inspecting the existing page contents in any way, we
224		 * don't care.
225		 */
226		page = BufferGetPage(buffer);
227
228		PageSetAllVisible(page);
229
230		if (XLogHintBitIsNeeded())
231			PageSetLSN(page, lsn);
232
233		MarkBufferDirty(buffer);
234	}
235	else if (action == BLK_RESTORED)
236	{
237		/*
238		 * If heap block was backed up, we already restored it and there's
239		 * nothing more to do. (This can only happen with checksums or
240		 * wal_log_hints enabled.)
241		 */
242	}
243
244	if (BufferIsValid(buffer))
245	{
246		Size		space = PageGetFreeSpace(BufferGetPage(buffer));
247
248		UnlockReleaseBuffer(buffer);
249
250		/*
251		 * Since FSM is not WAL-logged and only updated heuristically, it
252		 * easily becomes stale in standbys.  If the standby is later promoted
253		 * and runs VACUUM, it will skip updating individual free space
254		 * figures for pages that became all-visible (or all-frozen, depending
255		 * on the vacuum mode,) which is troublesome when FreeSpaceMapVacuum
256		 * propagates too optimistic free space values to upper FSM layers;
257		 * later inserters try to use such pages only to find out that they
258		 * are unusable.  This can cause long stalls when there are many such
259		 * pages.
260		 *
261		 * Forestall those problems by updating FSM's idea about a page that
262		 * is becoming all-visible or all-frozen.
263		 *
264		 * Do this regardless of a full-page image being applied, since the
265		 * FSM data is not in the page anyway.
266		 */
267		if (xlrec->flags & VISIBILITYMAP_VALID_BITS)
268			XLogRecordPageWithFreeSpace(rlocator, blkno, space);
269	}
270
271	/*
272	 * Even if we skipped the heap page update due to the LSN interlock, it's
273	 * still safe to update the visibility map.  Any WAL record that clears
274	 * the visibility map bit does so before checking the page LSN, so any
275	 * bits that need to be cleared will still be cleared.
276	 */
277	if (XLogReadBufferForRedoExtended(record, 0, RBM_ZERO_ON_ERROR, false,
278									  &vmbuffer) == BLK_NEEDS_REDO)
279	{
280		Page		vmpage = BufferGetPage(vmbuffer);
281		Relation	reln;
282		uint8		vmbits;
283
284		/* initialize the page if it was read as zeros */
285		if (PageIsNew(vmpage))
286			PageInit(vmpage, BLCKSZ, 0);
287
288		/* remove VISIBILITYMAP_XLOG_* */
289		vmbits = xlrec->flags & VISIBILITYMAP_VALID_BITS;
290
291		/*
292		 * XLogReadBufferForRedoExtended locked the buffer. But
293		 * visibilitymap_set will handle locking itself.
294		 */
295		LockBuffer(vmbuffer, BUFFER_LOCK_UNLOCK);
296
297		reln = CreateFakeRelcacheEntry(rlocator);
298		visibilitymap_pin(reln, blkno, &vmbuffer);
299
300		visibilitymap_set(reln, blkno, InvalidBuffer, lsn, vmbuffer,
301						  xlrec->snapshotConflictHorizon, vmbits);
302
303		ReleaseBuffer(vmbuffer);
304		FreeFakeRelcacheEntry(reln);
305	}
306	else if (BufferIsValid(vmbuffer))
307		UnlockReleaseBuffer(vmbuffer);
308}

310/*
311 * Given an "infobits" field from an XLog record, set the correct bits in the
312 * given infomask and infomask2 for the tuple touched by the record.
313 *
314 * (This is the reverse of compute_infobits).
315 */
316static void
317fix_infomask_from_infobits(uint8 infobits, uint16 *infomask, uint16 *infomask2)
318{
319	*infomask &= ~(HEAP_XMAX_IS_MULTI | HEAP_XMAX_LOCK_ONLY |
320				   HEAP_XMAX_KEYSHR_LOCK | HEAP_XMAX_EXCL_LOCK);
321	*infomask2 &= ~HEAP_KEYS_UPDATED;
322
323	if (infobits & XLHL_XMAX_IS_MULTI)
324		*infomask |= HEAP_XMAX_IS_MULTI;
325	if (infobits & XLHL_XMAX_LOCK_ONLY)
326		*infomask |= HEAP_XMAX_LOCK_ONLY;
327	if (infobits & XLHL_XMAX_EXCL_LOCK)
328		*infomask |= HEAP_XMAX_EXCL_LOCK;
329	/* note HEAP_XMAX_SHR_LOCK isn't considered here */
330	if (infobits & XLHL_XMAX_KEYSHR_LOCK)
331		*infomask |= HEAP_XMAX_KEYSHR_LOCK;
332
333	if (infobits & XLHL_KEYS_UPDATED)
334		*infomask2 |= HEAP_KEYS_UPDATED;
335}

337/*
338 * Replay XLOG_HEAP_DELETE records.
339 */
340static void
341heap_xlog_delete(XLogReaderState *record)
342{
343	XLogRecPtr	lsn = record->EndRecPtr;
344	xl_heap_delete *xlrec = (xl_heap_delete *) XLogRecGetData(record);
345	Buffer		buffer;
346	Page		page;
347	ItemId		lp = NULL;
348	HeapTupleHeader htup;
349	BlockNumber blkno;
350	RelFileLocator target_locator;
351	ItemPointerData target_tid;
352
353	XLogRecGetBlockTag(record, 0, &target_locator, NULL, &blkno);
354	ItemPointerSetBlockNumber(&target_tid, blkno);
355	ItemPointerSetOffsetNumber(&target_tid, xlrec->offnum);
356
357	/*
358	 * The visibility map may need to be fixed even if the heap page is
359	 * already up-to-date.
360	 */
361	if (xlrec->flags & XLH_DELETE_ALL_VISIBLE_CLEARED)
362	{
363		Relation	reln = CreateFakeRelcacheEntry(target_locator);
364		Buffer		vmbuffer = InvalidBuffer;
365
366		visibilitymap_pin(reln, blkno, &vmbuffer);
367		visibilitymap_clear(reln, blkno, vmbuffer, VISIBILITYMAP_VALID_BITS);
368		ReleaseBuffer(vmbuffer);
369		FreeFakeRelcacheEntry(reln);
370	}
371
372	if (XLogReadBufferForRedo(record, 0, &buffer) == BLK_NEEDS_REDO)
373	{
374		page = BufferGetPage(buffer);
375
376		if (PageGetMaxOffsetNumber(page) >= xlrec->offnum)
377			lp = PageGetItemId(page, xlrec->offnum);
378
379		if (PageGetMaxOffsetNumber(page) < xlrec->offnum || !ItemIdIsNormal(lp))
380			elog(PANIC, "invalid lp");
381
382		htup = (HeapTupleHeader) PageGetItem(page, lp);
383
384		htup->t_infomask &= ~(HEAP_XMAX_BITS | HEAP_MOVED);
385		htup->t_infomask2 &= ~HEAP_KEYS_UPDATED;
386		HeapTupleHeaderClearHotUpdated(htup);
387		fix_infomask_from_infobits(xlrec->infobits_set,
388								   &htup->t_infomask, &htup->t_infomask2);
389		if (!(xlrec->flags & XLH_DELETE_IS_SUPER))
390			HeapTupleHeaderSetXmax(htup, xlrec->xmax);
391		else
392			HeapTupleHeaderSetXmin(htup, InvalidTransactionId);
393		HeapTupleHeaderSetCmax(htup, FirstCommandId, false);
394
395		/* Mark the page as a candidate for pruning */
396		PageSetPrunable(page, XLogRecGetXid(record));
397
398		if (xlrec->flags & XLH_DELETE_ALL_VISIBLE_CLEARED)
399			PageClearAllVisible(page);
400
401		/* Make sure t_ctid is set correctly */
402		if (xlrec->flags & XLH_DELETE_IS_PARTITION_MOVE)
403			HeapTupleHeaderSetMovedPartitions(htup);
404		else
405			htup->t_ctid = target_tid;
406		PageSetLSN(page, lsn);
407		MarkBufferDirty(buffer);
408	}
409	if (BufferIsValid(buffer))
410		UnlockReleaseBuffer(buffer);
411}

413/*
414 * Replay XLOG_HEAP_INSERT records.
415 */
416static void
417heap_xlog_insert(XLogReaderState *record)
418{
419	XLogRecPtr	lsn = record->EndRecPtr;
420	xl_heap_insert *xlrec = (xl_heap_insert *) XLogRecGetData(record);
421	Buffer		buffer;
422	Page		page;
423	union
424	{
425		HeapTupleHeaderData hdr;
426		char		data[MaxHeapTupleSize];
427	}			tbuf;
428	HeapTupleHeader htup;
429	xl_heap_header xlhdr;
430	uint32		newlen;
431	Size		freespace = 0;
432	RelFileLocator target_locator;
433	BlockNumber blkno;
434	ItemPointerData target_tid;
435	XLogRedoAction action;
436
437	XLogRecGetBlockTag(record, 0, &target_locator, NULL, &blkno);
438	ItemPointerSetBlockNumber(&target_tid, blkno);
439	ItemPointerSetOffsetNumber(&target_tid, xlrec->offnum);
440
441	/* No freezing in the heap_insert() code path */
442	Assert(!(xlrec->flags & XLH_INSERT_ALL_FROZEN_SET));
443
444	/*
445	 * The visibility map may need to be fixed even if the heap page is
446	 * already up-to-date.
447	 */
448	if (xlrec->flags & XLH_INSERT_ALL_VISIBLE_CLEARED)
449	{
450		Relation	reln = CreateFakeRelcacheEntry(target_locator);
451		Buffer		vmbuffer = InvalidBuffer;
452
453		visibilitymap_pin(reln, blkno, &vmbuffer);
454		visibilitymap_clear(reln, blkno, vmbuffer, VISIBILITYMAP_VALID_BITS);
455		ReleaseBuffer(vmbuffer);
456		FreeFakeRelcacheEntry(reln);
457	}
458
459	/*
460	 * If we inserted the first and only tuple on the page, re-initialize the
461	 * page from scratch.
462	 */
463	if (XLogRecGetInfo(record) & XLOG_HEAP_INIT_PAGE)
464	{
465		buffer = XLogInitBufferForRedo(record, 0);
466		page = BufferGetPage(buffer);
467		PageInit(page, BufferGetPageSize(buffer), 0);
468		action = BLK_NEEDS_REDO;
469	}
470	else
471		action = XLogReadBufferForRedo(record, 0, &buffer);
472	if (action == BLK_NEEDS_REDO)
473	{
474		Size		datalen;
475		char	   *data;
476
477		page = BufferGetPage(buffer);
478
479		if (PageGetMaxOffsetNumber(page) + 1 < xlrec->offnum)
480			elog(PANIC, "invalid max offset number");
481
482		data = XLogRecGetBlockData(record, 0, &datalen);
483
484		newlen = datalen - SizeOfHeapHeader;
485		Assert(datalen > SizeOfHeapHeader && newlen <= MaxHeapTupleSize);
486		memcpy(&xlhdr, data, SizeOfHeapHeader);
487		data += SizeOfHeapHeader;
488
489		htup = &tbuf.hdr;
490		MemSet(htup, 0, SizeofHeapTupleHeader);
491		/* PG73FORMAT: get bitmap [+ padding] [+ oid] + data */
492		memcpy((char *) htup + SizeofHeapTupleHeader,
493			   data,
494			   newlen);
495		newlen += SizeofHeapTupleHeader;
496		htup->t_infomask2 = xlhdr.t_infomask2;
497		htup->t_infomask = xlhdr.t_infomask;
498		htup->t_hoff = xlhdr.t_hoff;
499		HeapTupleHeaderSetXmin(htup, XLogRecGetXid(record));
500		HeapTupleHeaderSetCmin(htup, FirstCommandId);
501		htup->t_ctid = target_tid;
502
503		if (PageAddItem(page, (Item) htup, newlen, xlrec->offnum,
504						true, true) == InvalidOffsetNumber)
505			elog(PANIC, "failed to add tuple");
506
507		freespace = PageGetHeapFreeSpace(page); /* needed to update FSM below */
508
509		PageSetLSN(page, lsn);
510
511		if (xlrec->flags & XLH_INSERT_ALL_VISIBLE_CLEARED)
512			PageClearAllVisible(page);
513
514		MarkBufferDirty(buffer);
515	}
516	if (BufferIsValid(buffer))
517		UnlockReleaseBuffer(buffer);
518
519	/*
520	 * If the page is running low on free space, update the FSM as well.
521	 * Arbitrarily, our definition of "low" is less than 20%. We can't do much
522	 * better than that without knowing the fill-factor for the table.
523	 *
524	 * XXX: Don't do this if the page was restored from full page image. We
525	 * don't bother to update the FSM in that case, it doesn't need to be
526	 * totally accurate anyway.
527	 */
528	if (action == BLK_NEEDS_REDO && freespace < BLCKSZ / 5)
529		XLogRecordPageWithFreeSpace(target_locator, blkno, freespace);
530}

532/*
533 * Replay XLOG_HEAP2_MULTI_INSERT records.
534 */
535static void
536heap_xlog_multi_insert(XLogReaderState *record)
537{
538	XLogRecPtr	lsn = record->EndRecPtr;
539	xl_heap_multi_insert *xlrec;
540	RelFileLocator rlocator;
541	BlockNumber blkno;
542	Buffer		buffer;
543	Page		page;
544	union
545	{
546		HeapTupleHeaderData hdr;
547		char		data[MaxHeapTupleSize];
548	}			tbuf;
549	HeapTupleHeader htup;
550	uint32		newlen;
551	Size		freespace = 0;
552	int			i;
553	bool		isinit = (XLogRecGetInfo(record) & XLOG_HEAP_INIT_PAGE) != 0;
554	XLogRedoAction action;
555
556	/*
557	 * Insertion doesn't overwrite MVCC data, so no conflict processing is
558	 * required.
559	 */
560	xlrec = (xl_heap_multi_insert *) XLogRecGetData(record);
561
562	XLogRecGetBlockTag(record, 0, &rlocator, NULL, &blkno);
563
564	/* check that the mutually exclusive flags are not both set */
565	Assert(!((xlrec->flags & XLH_INSERT_ALL_VISIBLE_CLEARED) &&
566			 (xlrec->flags & XLH_INSERT_ALL_FROZEN_SET)));
567
568	/*
569	 * The visibility map may need to be fixed even if the heap page is
570	 * already up-to-date.
571	 */
572	if (xlrec->flags & XLH_INSERT_ALL_VISIBLE_CLEARED)
573	{
574		Relation	reln = CreateFakeRelcacheEntry(rlocator);
575		Buffer		vmbuffer = InvalidBuffer;
576
577		visibilitymap_pin(reln, blkno, &vmbuffer);
578		visibilitymap_clear(reln, blkno, vmbuffer, VISIBILITYMAP_VALID_BITS);
579		ReleaseBuffer(vmbuffer);
580		FreeFakeRelcacheEntry(reln);
581	}
582
583	if (isinit)
584	{
585		buffer = XLogInitBufferForRedo(record, 0);
586		page = BufferGetPage(buffer);
587		PageInit(page, BufferGetPageSize(buffer), 0);
588		action = BLK_NEEDS_REDO;
589	}
590	else
591		action = XLogReadBufferForRedo(record, 0, &buffer);
592	if (action == BLK_NEEDS_REDO)
593	{
594		char	   *tupdata;
595		char	   *endptr;
596		Size		len;
597
598		/* Tuples are stored as block data */
599		tupdata = XLogRecGetBlockData(record, 0, &len);
600		endptr = tupdata + len;
601
602		page = (Page) BufferGetPage(buffer);
603
604		for (i = 0; i < xlrec->ntuples; i++)
605		{
606			OffsetNumber offnum;
607			xl_multi_insert_tuple *xlhdr;
608
609			/*
610			 * If we're reinitializing the page, the tuples are stored in
611			 * order from FirstOffsetNumber. Otherwise there's an array of
612			 * offsets in the WAL record, and the tuples come after that.
613			 */
614			if (isinit)
615				offnum = FirstOffsetNumber + i;
616			else
617				offnum = xlrec->offsets[i];
618			if (PageGetMaxOffsetNumber(page) + 1 < offnum)
619				elog(PANIC, "invalid max offset number");
620
621			xlhdr = (xl_multi_insert_tuple *) SHORTALIGN(tupdata);
622			tupdata = ((char *) xlhdr) + SizeOfMultiInsertTuple;
623
624			newlen = xlhdr->datalen;
625			Assert(newlen <= MaxHeapTupleSize);
626			htup = &tbuf.hdr;
627			MemSet(htup, 0, SizeofHeapTupleHeader);
628			/* PG73FORMAT: get bitmap [+ padding] [+ oid] + data */
629			memcpy((char *) htup + SizeofHeapTupleHeader,
630				   tupdata,
631				   newlen);
632			tupdata += newlen;
633
634			newlen += SizeofHeapTupleHeader;
635			htup->t_infomask2 = xlhdr->t_infomask2;
636			htup->t_infomask = xlhdr->t_infomask;
637			htup->t_hoff = xlhdr->t_hoff;
638			HeapTupleHeaderSetXmin(htup, XLogRecGetXid(record));
639			HeapTupleHeaderSetCmin(htup, FirstCommandId);
640			ItemPointerSetBlockNumber(&htup->t_ctid, blkno);
641			ItemPointerSetOffsetNumber(&htup->t_ctid, offnum);
642
643			offnum = PageAddItem(page, (Item) htup, newlen, offnum, true, true);
644			if (offnum == InvalidOffsetNumber)
645				elog(PANIC, "failed to add tuple");
646		}
647		if (tupdata != endptr)
648			elog(PANIC, "total tuple length mismatch");
649
650		freespace = PageGetHeapFreeSpace(page); /* needed to update FSM below */
651
652		PageSetLSN(page, lsn);
653
654		if (xlrec->flags & XLH_INSERT_ALL_VISIBLE_CLEARED)
655			PageClearAllVisible(page);
656
657		/* XLH_INSERT_ALL_FROZEN_SET implies that all tuples are visible */
658		if (xlrec->flags & XLH_INSERT_ALL_FROZEN_SET)
659			PageSetAllVisible(page);
660
661		MarkBufferDirty(buffer);
662	}
663	if (BufferIsValid(buffer))
664		UnlockReleaseBuffer(buffer);
665
666	/*
667	 * If the page is running low on free space, update the FSM as well.
668	 * Arbitrarily, our definition of "low" is less than 20%. We can't do much
669	 * better than that without knowing the fill-factor for the table.
670	 *
671	 * XXX: Don't do this if the page was restored from full page image. We
672	 * don't bother to update the FSM in that case, it doesn't need to be
673	 * totally accurate anyway.
674	 */
675	if (action == BLK_NEEDS_REDO && freespace < BLCKSZ / 5)
676		XLogRecordPageWithFreeSpace(rlocator, blkno, freespace);
677}

679/*
680 * Replay XLOG_HEAP_UPDATE and XLOG_HEAP_HOT_UPDATE records.
681 */
682static void
683heap_xlog_update(XLogReaderState *record, bool hot_update)
684{
685	XLogRecPtr	lsn = record->EndRecPtr;
686	xl_heap_update *xlrec = (xl_heap_update *) XLogRecGetData(record);
687	RelFileLocator rlocator;
688	BlockNumber oldblk;
689	BlockNumber newblk;
690	ItemPointerData newtid;
691	Buffer		obuffer,
692				nbuffer;
693	Page		page;
694	OffsetNumber offnum;
695	ItemId		lp = NULL;
696	HeapTupleData oldtup;
697	HeapTupleHeader htup;
698	uint16		prefixlen = 0,
699				suffixlen = 0;
700	char	   *newp;
701	union
702	{
703		HeapTupleHeaderData hdr;
704		char		data[MaxHeapTupleSize];
705	}			tbuf;
706	xl_heap_header xlhdr;
707	uint32		newlen;
708	Size		freespace = 0;
709	XLogRedoAction oldaction;
710	XLogRedoAction newaction;
711
712	/* initialize to keep the compiler quiet */
713	oldtup.t_data = NULL;
714	oldtup.t_len = 0;
715
716	XLogRecGetBlockTag(record, 0, &rlocator, NULL, &newblk);
717	if (XLogRecGetBlockTagExtended(record, 1, NULL, NULL, &oldblk, NULL))
718	{
719		/* HOT updates are never done across pages */
720		Assert(!hot_update);
721	}
722	else
723		oldblk = newblk;
724
725	ItemPointerSet(&newtid, newblk, xlrec->new_offnum);
726
727	/*
728	 * The visibility map may need to be fixed even if the heap page is
729	 * already up-to-date.
730	 */
731	if (xlrec->flags & XLH_UPDATE_OLD_ALL_VISIBLE_CLEARED)
732	{
733		Relation	reln = CreateFakeRelcacheEntry(rlocator);
734		Buffer		vmbuffer = InvalidBuffer;
735
736		visibilitymap_pin(reln, oldblk, &vmbuffer);
737		visibilitymap_clear(reln, oldblk, vmbuffer, VISIBILITYMAP_VALID_BITS);
738		ReleaseBuffer(vmbuffer);
739		FreeFakeRelcacheEntry(reln);
740	}
741
742	/*
743	 * In normal operation, it is important to lock the two pages in
744	 * page-number order, to avoid possible deadlocks against other update
745	 * operations going the other way.  However, during WAL replay there can
746	 * be no other update happening, so we don't need to worry about that. But
747	 * we *do* need to worry that we don't expose an inconsistent state to Hot
748	 * Standby queries --- so the original page can't be unlocked before we've
749	 * added the new tuple to the new page.
750	 */
751
752	/* Deal with old tuple version */
753	oldaction = XLogReadBufferForRedo(record, (oldblk == newblk) ? 0 : 1,
754									  &obuffer);
755	if (oldaction == BLK_NEEDS_REDO)
756	{
757		page = BufferGetPage(obuffer);
758		offnum = xlrec->old_offnum;
759		if (PageGetMaxOffsetNumber(page) >= offnum)
760			lp = PageGetItemId(page, offnum);
761
762		if (PageGetMaxOffsetNumber(page) < offnum || !ItemIdIsNormal(lp))
763			elog(PANIC, "invalid lp");
764
765		htup = (HeapTupleHeader) PageGetItem(page, lp);
766
767		oldtup.t_data = htup;
768		oldtup.t_len = ItemIdGetLength(lp);
769
770		htup->t_infomask &= ~(HEAP_XMAX_BITS | HEAP_MOVED);
771		htup->t_infomask2 &= ~HEAP_KEYS_UPDATED;
772		if (hot_update)
773			HeapTupleHeaderSetHotUpdated(htup);
774		else
775			HeapTupleHeaderClearHotUpdated(htup);
776		fix_infomask_from_infobits(xlrec->old_infobits_set, &htup->t_infomask,
777								   &htup->t_infomask2);
778		HeapTupleHeaderSetXmax(htup, xlrec->old_xmax);
779		HeapTupleHeaderSetCmax(htup, FirstCommandId, false);
780		/* Set forward chain link in t_ctid */
781		htup->t_ctid = newtid;
782
783		/* Mark the page as a candidate for pruning */
784		PageSetPrunable(page, XLogRecGetXid(record));
785
786		if (xlrec->flags & XLH_UPDATE_OLD_ALL_VISIBLE_CLEARED)
787			PageClearAllVisible(page);
788
789		PageSetLSN(page, lsn);
790		MarkBufferDirty(obuffer);
791	}
792
793	/*
794	 * Read the page the new tuple goes into, if different from old.
795	 */
796	if (oldblk == newblk)
797	{
798		nbuffer = obuffer;
799		newaction = oldaction;
800	}
801	else if (XLogRecGetInfo(record) & XLOG_HEAP_INIT_PAGE)
802	{
803		nbuffer = XLogInitBufferForRedo(record, 0);
804		page = (Page) BufferGetPage(nbuffer);
805		PageInit(page, BufferGetPageSize(nbuffer), 0);
806		newaction = BLK_NEEDS_REDO;
807	}
808	else
809		newaction = XLogReadBufferForRedo(record, 0, &nbuffer);
810
811	/*
812	 * The visibility map may need to be fixed even if the heap page is
813	 * already up-to-date.
814	 */
815	if (xlrec->flags & XLH_UPDATE_NEW_ALL_VISIBLE_CLEARED)
816	{
817		Relation	reln = CreateFakeRelcacheEntry(rlocator);
818		Buffer		vmbuffer = InvalidBuffer;
819
820		visibilitymap_pin(reln, newblk, &vmbuffer);
821		visibilitymap_clear(reln, newblk, vmbuffer, VISIBILITYMAP_VALID_BITS);
822		ReleaseBuffer(vmbuffer);
823		FreeFakeRelcacheEntry(reln);
824	}
825
826	/* Deal with new tuple */
827	if (newaction == BLK_NEEDS_REDO)
828	{
829		char	   *recdata;
830		char	   *recdata_end;
831		Size		datalen;
832		Size		tuplen;
833
834		recdata = XLogRecGetBlockData(record, 0, &datalen);
835		recdata_end = recdata + datalen;
836
837		page = BufferGetPage(nbuffer);
838
839		offnum = xlrec->new_offnum;
840		if (PageGetMaxOffsetNumber(page) + 1 < offnum)
841			elog(PANIC, "invalid max offset number");
842
843		if (xlrec->flags & XLH_UPDATE_PREFIX_FROM_OLD)
844		{
845			Assert(newblk == oldblk);
846			memcpy(&prefixlen, recdata, sizeof(uint16));
847			recdata += sizeof(uint16);
848		}
849		if (xlrec->flags & XLH_UPDATE_SUFFIX_FROM_OLD)
850		{
851			Assert(newblk == oldblk);
852			memcpy(&suffixlen, recdata, sizeof(uint16));
853			recdata += sizeof(uint16);
854		}
855
856		memcpy(&xlhdr, recdata, SizeOfHeapHeader);
857		recdata += SizeOfHeapHeader;
858
859		tuplen = recdata_end - recdata;
860		Assert(tuplen <= MaxHeapTupleSize);
861
862		htup = &tbuf.hdr;
863		MemSet(htup, 0, SizeofHeapTupleHeader);
864
865		/*
866		 * Reconstruct the new tuple using the prefix and/or suffix from the
867		 * old tuple, and the data stored in the WAL record.
868		 */
869		newp = (char *) htup + SizeofHeapTupleHeader;
870		if (prefixlen > 0)
871		{
872			int			len;
873
874			/* copy bitmap [+ padding] [+ oid] from WAL record */
875			len = xlhdr.t_hoff - SizeofHeapTupleHeader;
876			memcpy(newp, recdata, len);
877			recdata += len;
878			newp += len;
879
880			/* copy prefix from old tuple */
881			memcpy(newp, (char *) oldtup.t_data + oldtup.t_data->t_hoff, prefixlen);
882			newp += prefixlen;
883
884			/* copy new tuple data from WAL record */
885			len = tuplen - (xlhdr.t_hoff - SizeofHeapTupleHeader);
886			memcpy(newp, recdata, len);
887			recdata += len;
888			newp += len;
889		}
890		else
891		{
892			/*
893			 * copy bitmap [+ padding] [+ oid] + data from record, all in one
894			 * go
895			 */
896			memcpy(newp, recdata, tuplen);
897			recdata += tuplen;
898			newp += tuplen;
899		}
900		Assert(recdata == recdata_end);
901
902		/* copy suffix from old tuple */
903		if (suffixlen > 0)
904			memcpy(newp, (char *) oldtup.t_data + oldtup.t_len - suffixlen, suffixlen);
905
906		newlen = SizeofHeapTupleHeader + tuplen + prefixlen + suffixlen;
907		htup->t_infomask2 = xlhdr.t_infomask2;
908		htup->t_infomask = xlhdr.t_infomask;
909		htup->t_hoff = xlhdr.t_hoff;
910
911		HeapTupleHeaderSetXmin(htup, XLogRecGetXid(record));
912		HeapTupleHeaderSetCmin(htup, FirstCommandId);
913		HeapTupleHeaderSetXmax(htup, xlrec->new_xmax);
914		/* Make sure there is no forward chain link in t_ctid */
915		htup->t_ctid = newtid;
916
917		offnum = PageAddItem(page, (Item) htup, newlen, offnum, true, true);
918		if (offnum == InvalidOffsetNumber)
919			elog(PANIC, "failed to add tuple");
920
921		if (xlrec->flags & XLH_UPDATE_NEW_ALL_VISIBLE_CLEARED)
922			PageClearAllVisible(page);
923
924		freespace = PageGetHeapFreeSpace(page); /* needed to update FSM below */
925
926		PageSetLSN(page, lsn);
927		MarkBufferDirty(nbuffer);
928	}
929
930	if (BufferIsValid(nbuffer) && nbuffer != obuffer)
931		UnlockReleaseBuffer(nbuffer);
932	if (BufferIsValid(obuffer))
933		UnlockReleaseBuffer(obuffer);
934
935	/*
936	 * If the new page is running low on free space, update the FSM as well.
937	 * Arbitrarily, our definition of "low" is less than 20%. We can't do much
938	 * better than that without knowing the fill-factor for the table.
939	 *
940	 * However, don't update the FSM on HOT updates, because after crash
941	 * recovery, either the old or the new tuple will certainly be dead and
942	 * prunable. After pruning, the page will have roughly as much free space
943	 * as it did before the update, assuming the new tuple is about the same
944	 * size as the old one.
945	 *
946	 * XXX: Don't do this if the page was restored from full page image. We
947	 * don't bother to update the FSM in that case, it doesn't need to be
948	 * totally accurate anyway.
949	 */
950	if (newaction == BLK_NEEDS_REDO && !hot_update && freespace < BLCKSZ / 5)
951		XLogRecordPageWithFreeSpace(rlocator, newblk, freespace);
952}

1180void
1181heap_redo(XLogReaderState *record)
1182{
1183	uint8		info = XLogRecGetInfo(record) & ~XLR_INFO_MASK;
1184
1185	/*
1186	 * These operations don't overwrite MVCC data so no conflict processing is
1187	 * required. The ones in heap2 rmgr do.
1188	 */
1189
1190	switch (info & XLOG_HEAP_OPMASK)
1191	{
1192		case XLOG_HEAP_INSERT:
1193			heap_xlog_insert(record);
1194			break;
1195		case XLOG_HEAP_DELETE:
1196			heap_xlog_delete(record);
1197			break;
1198		case XLOG_HEAP_UPDATE:
1199			heap_xlog_update(record, false);
1200			break;
1201		case XLOG_HEAP_TRUNCATE:
1202
1203			/*
1204			 * TRUNCATE is a no-op because the actions are already logged as
1205			 * SMGR WAL records.  TRUNCATE WAL record only exists for logical
1206			 * decoding.
1207			 */
1208			break;
1209		case XLOG_HEAP_HOT_UPDATE:
1210			heap_xlog_update(record, true);
1211			break;
1212		case XLOG_HEAP_CONFIRM:
1213			heap_xlog_confirm(record);
1214			break;
1215		case XLOG_HEAP_LOCK:
1216			heap_xlog_lock(record);
1217			break;
1218		case XLOG_HEAP_INPLACE:
1219			heap_xlog_inplace(record);
1220			break;
1221		default:
1222			elog(PANIC, "heap_redo: unknown op code %u", info);
1223	}
1224}

1226void
1227heap2_redo(XLogReaderState *record)
1228{
1229	uint8		info = XLogRecGetInfo(record) & ~XLR_INFO_MASK;
1230
1231	switch (info & XLOG_HEAP_OPMASK)
1232	{
1233		case XLOG_HEAP2_PRUNE_ON_ACCESS:
1234		case XLOG_HEAP2_PRUNE_VACUUM_SCAN:
1235		case XLOG_HEAP2_PRUNE_VACUUM_CLEANUP:
1236			heap_xlog_prune_freeze(record);
1237			break;
1238		case XLOG_HEAP2_VISIBLE:
1239			heap_xlog_visible(record);
1240			break;
1241		case XLOG_HEAP2_MULTI_INSERT:
1242			heap_xlog_multi_insert(record);
1243			break;
1244		case XLOG_HEAP2_LOCK_UPDATED:
1245			heap_xlog_lock_updated(record);
1246			break;
1247		case XLOG_HEAP2_NEW_CID:
1248
1249			/*
1250			 * Nothing to do on a real replay, only used during logical
1251			 * decoding.
1252			 */
1253			break;
1254		case XLOG_HEAP2_REWRITE:
1255			heap_xlog_logical_rewrite(record);
1256			break;
1257		default:
1258			elog(PANIC, "heap2_redo: unknown op code %u", info);
1259	}
1260}
1261
1262/*
1263 * Mask a heap page before performing consistency checks on it.
1264 */
1265void
1266heap_mask(char *pagedata, BlockNumber blkno)
1267{
1268	Page		page = (Page) pagedata;
1269	OffsetNumber off;
1270
1271	mask_page_lsn_and_checksum(page);
1272
1273	mask_page_hint_bits(page);
1274	mask_unused_space(page);
1275
1276	for (off = 1; off <= PageGetMaxOffsetNumber(page); off++)
1277	{
1278		ItemId		iid = PageGetItemId(page, off);
1279		char	   *page_item;
1280
1281		page_item = (char *) (page + ItemIdGetOffset(iid));
1282
1283		if (ItemIdIsNormal(iid))
1284		{
1285			HeapTupleHeader page_htup = (HeapTupleHeader) page_item;
1286
1287			/*
1288			 * If xmin of a tuple is not yet frozen, we should ignore
1289			 * differences in hint bits, since they can be set without
1290			 * emitting WAL.
1291			 */
1292			if (!HeapTupleHeaderXminFrozen(page_htup))
1293				page_htup->t_infomask &= ~HEAP_XACT_MASK;
1294			else
1295			{
1296				/* Still we need to mask xmax hint bits. */
1297				page_htup->t_infomask &= ~HEAP_XMAX_INVALID;
1298				page_htup->t_infomask &= ~HEAP_XMAX_COMMITTED;
1299			}
1300
1301			/*
1302			 * During replay, we set Command Id to FirstCommandId. Hence, mask
1303			 * it. See heap_xlog_insert() for details.
1304			 */
1305			page_htup->t_choice.t_heap.t_field3.t_cid = MASK_MARKER;
1306
1307			/*
1308			 * For a speculative tuple, heap_insert() does not set ctid in the
1309			 * caller-passed heap tuple itself, leaving the ctid field to
1310			 * contain a speculative token value - a per-backend monotonically
1311			 * increasing identifier. Besides, it does not WAL-log ctid under
1312			 * any circumstances.
1313			 *
1314			 * During redo, heap_xlog_insert() sets t_ctid to current block
1315			 * number and self offset number. It doesn't care about any
1316			 * speculative insertions on the primary. Hence, we set t_ctid to
1317			 * current block number and self offset number to ignore any
1318			 * inconsistency.
1319			 */
1320			if (HeapTupleHeaderIsSpeculative(page_htup))
1321				ItemPointerSet(&page_htup->t_ctid, blkno, off);
1322
1323			/*
1324			 * NB: Not ignoring ctid changes due to the tuple having moved
1325			 * (i.e. HeapTupleHeaderIndicatesMovedPartitions), because that's
1326			 * important information that needs to be in-sync between primary
1327			 * and standby, and thus is WAL logged.
1328			 */
1329		}
1330
1331		/*
1332		 * Ignore any padding bytes after the tuple, when the length of the
1333		 * item is not MAXALIGNed.
1334		 */
1335		if (ItemIdHasStorage(iid))
1336		{
1337			int			len = ItemIdGetLength(iid);
1338			int			padlen = MAXALIGN(len) - len;
1339
1340			if (padlen > 0)
1341				memset(page_item + len, MASK_MARKER, padlen);
1342		}
1343	}
1344}
