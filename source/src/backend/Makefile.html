101# The following targets are specified in make commands that appear in
102# the make files in our subdirectories. Note that it's important we
103# match the dependencies shown in the subdirectory makefiles!
104# Also, in cases where a subdirectory makefile generates two files in
105# what's really one step, such as bison producing both gram.h and gram.c,
106# we must request making the one that is shown as the secondary (dependent)
107# output, else the timestamp on it might be wrong.  By project convention,
108# the .h file is the dependent one for bison output, so we need only request
109# that; but in other cases, request both for safety.
110
111parser/gram.h: parser/gram.y
112	$(MAKE) -C parser gram.h
113
114storage/lmgr/lwlocknames.h: storage/lmgr/generate-lwlocknames.pl ../include/storage/lwlocklist.h utils/activity/wait_event_names.txt
115	$(MAKE) -C storage/lmgr lwlocknames.h
116
117# run this unconditionally to avoid needing to know its dependencies here:
118submake-catalog-headers:
119	$(MAKE) -C ../include/catalog generated-headers
120
121# run this unconditionally to avoid needing to know its dependencies here:
122submake-nodes-headers:
123	$(MAKE) -C nodes generated-header-symlinks
124
125# run this unconditionally to avoid needing to know its dependencies here:
126submake-utils-headers:
127	$(MAKE) -C utils generated-header-symlinks
128
129.PHONY: submake-catalog-headers submake-nodes-headers submake-utils-headers
130
131# Make symlinks for these headers in the include directory. That way
132# we can cut down on the -I options. Also, a symlink is automatically
133# up to date when we update the base file.
134#
135# The point of the prereqdir incantation in some of the rules below is to
136# force the symlink to use an absolute path rather than a relative path.
137# This is needed to support platforms without ln -s.
138
139.PHONY: generated-headers
140
141generated-headers: $(top_builddir)/src/include/storage/lwlocknames.h submake-catalog-headers submake-nodes-headers submake-utils-headers parser/gram.h
142
143$(top_builddir)/src/include/storage/lwlocknames.h: storage/lmgr/lwlocknames.h
144	prereqdir=`cd '$(dir $<)' >/dev/null && pwd` && \
145	  cd '$(dir $@)' && rm -f $(notdir $@) && \
146	  $(LN_S) "$$prereqdir/$(notdir $<)" .
147
148utils/probes.o: utils/probes.d $(SUBDIROBJS)
149	$(DTRACE) $(DTRACEFLAGS) -C -G -s $(call expand_subsys,$^) -o $@

221	rm -f '$(DESTDIR)$(bindir)/postgres$(X)'
222ifeq ($(MAKE_EXPORTS), true)

225endif

227ifeq ($(MAKE_DLL), true)
228	rm -f '$(DESTDIR)$(libdir)/libpostgres.a'
229endif
230endif
231ifeq ($(PORTNAME), win32)
232ifeq ($(MAKE_DLL), true)
233	rm -f '$(DESTDIR)$(libdir)/libpostgres.a'
234endif

236	$(MAKE) -C catalog uninstall-data
237	$(MAKE) -C tsearch uninstall-data
238	$(MAKE) -C utils uninstall-data

240	      '$(DESTDIR)$(datadir)/pg_ident.conf.sample' \
241	      '$(DESTDIR)$(datadir)/postgresql.conf.sample'
