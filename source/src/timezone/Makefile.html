11PGFILEDESC = "zic - time zone compiler"
12PGAPPICON = win32
13
14subdir = src/timezone
15top_builddir = ../..
16include $(top_builddir)/src/Makefile.global
17
18# files to build into backend
19OBJS = \
20	localtime.o \
21	pgtz.o \
22	strftime.o
23
24# files needed to build zic utility program
25ZICOBJS = \
26	$(WIN32RES) \
27	zic.o
28
29# we now distribute the timezone data as a single file
30TZDATAFILES = $(srcdir)/data/tzdata.zi
31
32# any custom options you might want to pass to zic while installing data files
33ZIC_OPTIONS =
34
35# use system timezone data?
36ifneq (,$(with_system_tzdata))
37override CPPFLAGS += '-DSYSTEMTZDIR="$(with_system_tzdata)"'
38endif
39
40include $(top_srcdir)/src/backend/common.mk
41
42ifeq (,$(with_system_tzdata))
43all: zic
44endif
45
46# We could do this test in the action section:
47#	$(if $(ZIC),$(ZIC),./zic)
48# but GNU make versions <= 3.78.1 or perhaps later have a bug
49# that causes a segfault;  GNU make 3.81 or later fixes this.
50ifeq (,$(ZIC))
51ZIC= ./zic
52endif
53
54zic: $(ZICOBJS) | submake-libpgport
55	$(CC) $(CFLAGS) $(ZICOBJS) $(LDFLAGS) $(LDFLAGS_EX) $(LIBS) -o $@$(X)
56
57install: all installdirs
58ifeq (,$(with_system_tzdata))
59	$(ZIC) -d '$(DESTDIR)$(datadir)/timezone' $(ZIC_OPTIONS) $(TZDATAFILES)
60endif
61	$(MAKE) -C tznames $@
62
63# Note: -P code currently depends on '-b fat'.  Not worth fixing right now.
64abbrevs.txt: zic $(TZDATAFILES)
65	mkdir junkdir
66	$(ZIC) -P -b fat -d junkdir $(TZDATAFILES) | LANG=C sort | uniq >abbrevs.txt
67	rm -rf junkdir
68
69installdirs:
70	$(MKDIR_P) '$(DESTDIR)$(datadir)'
71
72uninstall:
73ifeq (,$(with_system_tzdata))
74	rm -rf '$(DESTDIR)$(datadir)/timezone'
75endif
76	$(MAKE) -C tznames $@
77
78clean distclean:
79	rm -f zic$(X) $(ZICOBJS) abbrevs.txt
