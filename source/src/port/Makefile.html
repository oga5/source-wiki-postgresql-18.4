3# Makefile
4#    Makefile for src/port
5#
6# These files are used by the Postgres backend, and also by frontend
7# programs.  Primarily, they are meant to provide portability on systems
8# with broken/missing library files.
9#
10# This makefile generates three outputs:
11#
12#	libpgport.a - contains object files with FRONTEND defined,
13#		for use by client applications
14#
15#	libpgport_shlib.a - contains object files with FRONTEND defined,
16#		built suitably for use in shared libraries; for use
17#		by frontend libraries
18#
19#	libpgport_srv.a - contains object files without FRONTEND defined,
20#		for use only by the backend
21#
22# LIBOBJS is set by configure (via Makefile.global) to be the list of object
23# files that are conditionally needed as determined by configure's probing.
24# OBJS adds additional object files that are always compiled.
25#
26# IDENTIFICATION
27#    src/port/Makefile
28#
29#-------------------------------------------------------------------------

65# libpgport.a, libpgport_shlib.a, and libpgport_srv.a contain the same files
66# foo.o, foo_shlib.o, and foo_srv.o are all built from foo.c
67OBJS_SHLIB = $(OBJS:%.o=%_shlib.o)
68OBJS_SRV = $(OBJS:%.o=%_srv.o)
69
70all: libpgport.a libpgport_shlib.a libpgport_srv.a
71
72# libpgport is needed by some contrib
73install: all installdirs
74	$(INSTALL_STLIB) libpgport.a '$(DESTDIR)$(libdir)/libpgport.a'
75	$(INSTALL_STLIB) libpgport_shlib.a '$(DESTDIR)$(libdir)/libpgport_shlib.a'
76
77installdirs:
78	$(MKDIR_P) '$(DESTDIR)$(libdir)'
79
80uninstall:
81	rm -f '$(DESTDIR)$(libdir)/libpgport.a'
82	rm -f '$(DESTDIR)$(libdir)/libpgport_shlib.a'
83
84libpgport.a: $(OBJS)
85	rm -f $@
86	$(AR) $(AROPT) $@ $^
87
88# all versions of pg_crc32c_armv8.o need CFLAGS_CRC
89pg_crc32c_armv8.o: CFLAGS+=$(CFLAGS_CRC)
90pg_crc32c_armv8_shlib.o: CFLAGS+=$(CFLAGS_CRC)
91pg_crc32c_armv8_srv.o: CFLAGS+=$(CFLAGS_CRC)
92
93#
94# Shared library versions of object files
95#
96
97libpgport_shlib.a: $(OBJS_SHLIB)
98	rm -f $@
99	$(AR) $(AROPT) $@ $^
100
101# Because this uses its own compilation rule, it doesn't use the
102# dependency tracking logic from Makefile.global.  To make sure that
103# dependency tracking works anyway for the *_shlib.o files, depend on
104# their *.o siblings as well, which do have proper dependencies.  It's
105# a hack that might fail someday if there is a *_shlib.o without a
106# corresponding *.o, but there seems little reason for that.
107%_shlib.o: %.c %.o
108	$(CC) $(CFLAGS) $(CFLAGS_SL) $(CPPFLAGS) -c $< -o $@
109
110#
111# Server versions of object files
112#
113
114libpgport_srv.a: $(OBJS_SRV)
115	rm -f $@
116	$(AR) $(AROPT) $@ $^
117
118# Because this uses its own compilation rule, it doesn't use the
119# dependency tracking logic from Makefile.global.  To make sure that
120# dependency tracking works anyway for the *_srv.o files, depend on
121# their *.o siblings as well, which do have proper dependencies.  It's
122# a hack that might fail someday if there is a *_srv.o without a
123# corresponding *.o, but it works for now (and those would probably go
124# into src/backend/port/ anyway).
125%_srv.o: %.c %.o
126	$(CC) $(CFLAGS) $(subst -DFRONTEND,, $(CPPFLAGS)) -c $< -o $@

136# We create a separate file rather than put these in pg_config.h
137# because many of these values come from makefiles and are not
138# available to configure.
139pg_config_paths.h: $(top_builddir)/src/Makefile.global
140	echo "#define PGBINDIR \"$(bindir)\"" >$@
141	echo "#define PGSHAREDIR \"$(datadir)\"" >>$@
142	echo "#define SYSCONFDIR \"$(sysconfdir)\"" >>$@
143	echo "#define INCLUDEDIR \"$(includedir)\"" >>$@
144	echo "#define PKGINCLUDEDIR \"$(pkgincludedir)\"" >>$@
145	echo "#define INCLUDEDIRSERVER \"$(includedir_server)\"" >>$@
146	echo "#define LIBDIR \"$(libdir)\"" >>$@
147	echo "#define PKGLIBDIR \"$(pkglibdir)\"" >>$@
148	echo "#define LOCALEDIR \"$(localedir)\"" >>$@
149	echo "#define DOCDIR \"$(docdir)\"" >>$@
150	echo "#define HTMLDIR \"$(htmldir)\"" >>$@
151	echo "#define MANDIR \"$(mandir)\"" >>$@

153clean distclean:

155	rm -f $(OBJS) $(OBJS_SHLIB) $(OBJS_SRV) pg_config_paths.h
