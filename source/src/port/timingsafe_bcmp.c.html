23#include "c.h"
24
25#ifdef USE_SSL
26#include <openssl/crypto.h>
27#endif
28
29int
30timingsafe_bcmp(const void *b1, const void *b2, size_t n)
31{
32#ifdef USE_SSL
33	return CRYPTO_memcmp(b1, b2, n);
34#else
35	const unsigned char *p1 = b1,
36			   *p2 = b2;
37	int			ret = 0;
38
39	for (; n > 0; n--)
40		ret |= *p1++ ^ *p2++;
41	return (ret != 0);
42#endif
43}
