152# Build pgport once for backend, once for use in frontend binaries, and once
153# for use in shared libraries
154pgport = {}
155pgport_variants = {
156  '_srv': internal_lib_args + {
157    'dependencies': [backend_port_code],
158  },
159  '': default_lib_args + {
160    'dependencies': [frontend_port_code],
161  },
162  '_shlib': default_lib_args + {
163    'pic': true,
164    'dependencies': [frontend_port_code],
165  },
166}
167
168foreach name, opts : pgport_variants
169
170  # Build internal static libraries for sets of files that need to be built
171  # with different cflags
172  cflag_libs = []
173  foreach cflagname, sources : pgport_sources_cflags
174    if sources.length() == 0
175      continue
176    endif
177    c_args = opts.get('c_args', []) + pgport_cflags[cflagname]
178    cflag_libs += static_library('libpgport@0@_@1@'.format(name, cflagname),
179      sources,
180      c_pch: pch_c_h,
181      kwargs: opts + {
182        'c_args': c_args,
183        'build_by_default': false,
184        'install': false,
185      },
186    )
187  endforeach
188
189  lib = static_library('libpgport@0@'.format(name),
190      pgport_sources,
191      link_with: cflag_libs,
192      link_whole: cflag_libs,
193      c_pch: pch_c_h,
194      kwargs: opts + {
195        'dependencies': opts['dependencies'] + [ssl],
196      }
197    )
198  pgport += {name: lib}
199endforeach

202pgport_static = pgport['']
203pgport_shlib = pgport['_shlib']
