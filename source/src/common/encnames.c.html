21/* ----------
22 * All encoding names, sorted:		 *** A L P H A B E T I C ***
23 *
24 * All names must be without irrelevant chars, search routines use
25 * isalnum() chars only. It means ISO-8859-1, iso_8859-1 and Iso8859_1
26 * are always converted to 'iso88591'. All must be lower case.
27 *
28 * The table doesn't contain 'cs' aliases (like csISOLatin1). It's needed?
29 *
30 * Karel Zak, Aug 2001
31 * ----------
32 */
33typedef struct pg_encname
34{
35	const char *name;
36	pg_enc		encoding;
37} pg_encname;

460/*
461 * Is this encoding supported by ICU?
462 */
463bool
464is_encoding_supported_by_icu(int encoding)
465{
466	if (!PG_VALID_BE_ENCODING(encoding))
467		return false;
468	return (pg_enc2icu_tbl[encoding] != NULL);
469}
470
471/*
472 * Returns ICU's name for encoding, or NULL if not supported
473 */
474const char *
475get_encoding_name_for_icu(int encoding)
476{
477	if (!PG_VALID_BE_ENCODING(encoding))
478		return NULL;
479	return pg_enc2icu_tbl[encoding];
480}

483/* ----------
484 * Encoding checks, for error returns -1 else encoding id
485 * ----------
486 */
487int
488pg_valid_client_encoding(const char *name)
489{
490	int			enc;
491
492	if ((enc = pg_char_to_encoding(name)) < 0)
493		return -1;
494
495	if (!PG_VALID_FE_ENCODING(enc))
496		return -1;
497
498	return enc;
499}

501int
502pg_valid_server_encoding(const char *name)
503{
504	int			enc;
505
506	if ((enc = pg_char_to_encoding(name)) < 0)
507		return -1;
508
509	if (!PG_VALID_BE_ENCODING(enc))
510		return -1;
511
512	return enc;
513}

515int
516pg_valid_server_encoding_id(int encoding)
517{
518	return PG_VALID_BE_ENCODING(encoding);

521/*
522 * Remove irrelevant chars from encoding name, store at *newkey
523 *
524 * (Caller's responsibility to provide a large enough buffer)
525 */
526static char *
527clean_encoding_name(const char *key, char *newkey)
528{
529	const char *p;
530	char	   *np;
531
532	for (p = key, np = newkey; *p != '\0'; p++)
533	{
534		if (isalnum((unsigned char) *p))
535		{
536			if (*p >= 'A' && *p <= 'Z')
537				*np++ = *p + 'a' - 'A';
538			else
539				*np++ = *p;
540		}
541	}
542	*np = '\0';
543	return newkey;
544}

546/*
547 * Search encoding by encoding name
548 *
549 * Returns encoding ID, or -1 if not recognized
550 */
551int
552pg_char_to_encoding(const char *name)
553{
554	unsigned int nel = lengthof(pg_encname_tbl);
555	const pg_encname *base = pg_encname_tbl,
556			   *last = base + nel - 1,
557			   *position;
558	int			result;
559	char		buff[NAMEDATALEN],
560			   *key;
561
562	if (name == NULL || *name == '\0')
563		return -1;
564
565	if (strlen(name) >= NAMEDATALEN)
566		return -1;				/* it's certainly not in the table */
567
568	key = clean_encoding_name(name, buff);
569
570	while (last >= base)
571	{
572		position = base + ((last - base) >> 1);
573		result = key[0] - position->name[0];
574
575		if (result == 0)
576		{
577			result = strcmp(key, position->name);
578			if (result == 0)
579				return position->encoding;
580		}
581		if (result < 0)
582			last = position - 1;
583		else
584			base = position + 1;
585	}
586	return -1;
587}

589const char *
590pg_encoding_to_char(int encoding)
591{
592	if (PG_VALID_ENCODING(encoding))
593	{
594		const pg_enc2name *p = &pg_enc2name_tbl[encoding];
595
596		Assert(encoding == p->encoding);
597		return p->name;
598	}
599	return "";
