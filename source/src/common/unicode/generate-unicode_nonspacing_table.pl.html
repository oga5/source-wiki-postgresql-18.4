src/common/unicode/generate-unicode_nonspacing_table.pl
1#!/usr/bin/perl
2#
3# Generate sorted list of non-overlapping intervals of non-spacing
4# characters, using Unicode data files as input.  Pass UnicodeData.txt
5# as argument.  The output is on stdout.
6#
7# Copyright (c) 2019-2025, PostgreSQL Global Development Group
8
9use strict;
10use warnings FATAL => 'all';
11
12my $range_start = undef;
13my $codepoint;
14my $prev_codepoint;
15my $count = 0;
16
17print
18  "/* generated by src/common/unicode/generate-unicode_nonspacing_table.pl, do not edit */\n\n";
19
20print "static const struct mbinterval nonspacing[] = {\n";
21
22foreach my $line (<ARGV>)
23{
24	chomp $line;
25	my @fields = split ';', $line;
26	$codepoint = hex $fields[0];
27
28	# Me and Mn refer to combining characters
29	# Cf refers to format characters
30	if ($fields[2] eq 'Me' || $fields[2] eq 'Mn' || $fields[2] eq 'Cf')
31	{
32		# non-spacing character, save for start of range
33		if (!defined($range_start))
34		{
35			$range_start = $codepoint;
36		}
37	}
38	else
39	{
40		# not a non-spacing character, print out previous range if any
41		if (defined($range_start))
42		{
43			printf "\t{0x%04X, 0x%04X},\n", $range_start, $prev_codepoint;
44			$range_start = undef;
45		}
46	}
47}
48continue
49{
50	$prev_codepoint = $codepoint;
51}
52
53print "};\n";
