529{
530	my ($range, $from, $to, $indent) = @_;

547				$space, $entry->{Start}, $entry->{End}, $space);
548		}
549		else

558			$space, $entry->{Start}, $entry->{Offset});

597# Group numbers into ranges where the difference between neighboring
598# elements does not exceed $limit. If the difference is greater, a new
599# range is created. This is used to break the sequence into intervals
600# where the gaps between numbers are greater than limit.
601#
602# For example, if there are numbers 1, 2, 3, 5, 6 and limit = 1, then
603# there is a difference of 2 between 3 and 5, which is greater than 1,
604# so there will be ranges 1-3 and 5-6.
605sub make_ranges
606{
607	my ($nums, $limit) = @_;
608	my ($prev, $start, $total, @sorted, @range);
609
610	@sorted = sort { $a <=> $b } @$nums;
611
612	die "expecting at least 2 codepoints" if (scalar @sorted < 2);
613
614	$start = shift @sorted;
615
616	die "expecting first codepoint to start at 0" unless $start == 0;
617
618	$prev = $start;
619	$total = 0;
620
621	# append final 'undef' to signal final iteration
622	push @sorted, undef;
623
624	foreach my $curr (@sorted)
625	{
626		# if last iteration always append the range
627		if (!defined($curr) || ($curr - $prev > $limit))
628		{
629			push @range,
630			  {
631				Start => $start,
632				End => $prev + 1,
633				Offset => $total
634			  };
635			$total += $prev + 1 - $start;
636			$start = $curr;
637		}
638
639		$prev = $curr;
640	}
641
642	return \@range;
643}
