1#!/usr/bin/perl
2#
3# Generate a composition table and its lookup utilities, using Unicode data
4# files as input.
5#
6# Input: UnicodeData.txt and CompositionExclusions.txt
7# Output: unicode_norm_table.h and unicode_norm_hashfunc.h
8#
9# Copyright (c) 2000-2025, PostgreSQL Global Development Group
10
11use strict;
12use warnings FATAL => 'all';
13use Getopt::Long;
14
15use FindBin;
16use lib "$FindBin::RealBin/../../tools/";
17use PerfectHash;
18
19my $output_path = '.';
20
21GetOptions('outdir:s' => \$output_path);
22
23my $output_table_file = "$output_path/unicode_norm_table.h";
24my $output_func_file = "$output_path/unicode_norm_hashfunc.h";
25
26
27my $FH;
28
29# Read list of codes that should be excluded from re-composition.
30my @composition_exclusion_codes = ();
31open($FH, '<', "$output_path/CompositionExclusions.txt")
32  or die "Could not open $output_path/CompositionExclusions.txt: $!.";
33while (my $line = <$FH>)
34{
35	if ($line =~ /^([[:xdigit:]]+)/)
36	{
37		push @composition_exclusion_codes, $1;
38	}
39}
40close $FH;
41
42# Read entries from UnicodeData.txt into a list, and a hash table. We need
43# three fields from each row: the codepoint, canonical combining class,
44# and character decomposition mapping
45my @characters = ();
46my %character_hash = ();
47open($FH, '<', "$output_path/UnicodeData.txt")
48  or die "Could not open $output_path/UnicodeData.txt: $!.";
49while (my $line = <$FH>)
50{
51
52	# Split the line wanted and get the fields needed:
53	# - Unicode code value
54	# - Canonical Combining Class
55	# - Character Decomposition Mapping
56	my @elts = split(';', $line);
57	my $code = $elts[0];
58	my $class = $elts[3];
59	my $decomp = $elts[5];
60
61	# Skip codepoints above U+10FFFF. They cannot be represented in 4 bytes
62	# in UTF-8, and PostgreSQL doesn't support UTF-8 characters longer than
63	# 4 bytes. (This is just pro forma, as there aren't any such entries in
64	# the data file, currently.)
65	next if hex($code) > 0x10FFFF;
66
67	# Skip characters with no decompositions and a class of 0, to reduce the
68	# table size.
69	next if $class eq '0' && $decomp eq '';
70
71	my %char_entry = (code => $code, class => $class, decomp => $decomp);
72	push(@characters, \%char_entry);
73	$character_hash{$code} = \%char_entry;
74}
75close $FH;
76
77my $num_characters = scalar @characters;
78
79# Start writing out the output files
80open my $OT, '>', $output_table_file
81  or die "Could not open output file $output_table_file: $!\n";
82open my $OF, '>', $output_func_file
83  or die "Could not open output file $output_func_file: $!\n";
84
85print $OT <<HEADER;
86/*-------------------------------------------------------------------------
87 *
88 * unicode_norm_table.h
89 *	  Composition table used for Unicode normalization
90 *
91 * Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
92 * Portions Copyright (c) 1994, Regents of the University of California
93 *
94 * src/include/common/unicode_norm_table.h
95 *
96 *-------------------------------------------------------------------------
97 */
98
99/*
100 * File auto-generated by src/common/unicode/generate-unicode_norm_table.pl,
101 * do not edit. There is deliberately not an #ifndef PG_UNICODE_NORM_TABLE_H
102 * here.
103 */
104typedef struct
105{
106	uint32		codepoint;		/* Unicode codepoint */
107	uint8		comb_class;		/* combining class of character */
108	uint8		dec_size_flags; /* size and flags of decomposition code list */
109	uint16		dec_index;		/* index into UnicodeDecomp_codepoints, or the
110								 * decomposition itself if DECOMP_INLINE */
111} pg_unicode_decomposition;
112
113#define DECOMP_NO_COMPOSE	0x80	/* don't use for re-composition */
114#define DECOMP_INLINE		0x40	/* decomposition is stored inline in
115									 * dec_index */
116#define DECOMP_COMPAT		0x20	/* compatibility mapping */
117
118#define DECOMPOSITION_SIZE(x) ((x)->dec_size_flags & 0x1F)
119#define DECOMPOSITION_NO_COMPOSE(x) (((x)->dec_size_flags & (DECOMP_NO_COMPOSE | DECOMP_COMPAT)) != 0)
120#define DECOMPOSITION_IS_INLINE(x) (((x)->dec_size_flags & DECOMP_INLINE) != 0)
121#define DECOMPOSITION_IS_COMPAT(x) (((x)->dec_size_flags & DECOMP_COMPAT) != 0)

136 *
137 * src/include/common/unicode_norm_hashfunc.h
138 *
139 *-------------------------------------------------------------------------

176my $last_code = $characters[-1]->{code};

178{
179	my $code = $char->{code};
180	my $class = $char->{class};
181	my $decomp = $char->{decomp};
182
183	# Save the code point bytes as a string in network order.
184	push @dec_cp_packed, pack('N', hex($char->{code}));
185
186	# The character decomposition mapping field in UnicodeData.txt is a list
187	# of unicode codepoints, separated by space. But it can be prefixed with
188	# so-called compatibility formatting tag, like "<compat>", or "<font>".
189	# The entries with compatibility formatting tags should not be used for
190	# re-composing characters during normalization, so flag them in the table.
191	# (The tag doesn't matter, only whether there is a tag or not)
192	my $compat = 0;
193	if ($decomp =~ /\<.*\>/)
194	{
195		$compat = 1;
196		$decomp =~ s/\<[^][]*\>//g;
197	}
198	my @decomp_elts = split(" ", $decomp);
199
200	# Decomposition size
201	# Print size of decomposition
202	my $decomp_size = scalar(@decomp_elts);
203	die if $decomp_size > 0x1F;    # to not overrun bitmask
204
205	my $first_decomp = shift @decomp_elts;
206
207	my $flags = "";
208	my $comment = "";
209
210	if ($compat)
211	{
212		$flags .= " | DECOMP_COMPAT";
213	}
214
215	if ($decomp_size == 2)
216	{
217		# Should this be used for recomposition?
218		if (   $character_hash{$first_decomp}
219			&& $character_hash{$first_decomp}->{class} != 0)
220		{
221			$flags .= " | DECOMP_NO_COMPOSE";
222			$comment = "non-starter decomposition";
223		}
224		else
225		{
226			foreach my $lcode (@composition_exclusion_codes)
227			{
228				if ($lcode eq $code)
229				{
230					$flags .= " | DECOMP_NO_COMPOSE";
231					$comment = "in exclusion list";
232					last;
233				}
234			}
235		}
236
237		# Save info for recomposeable codepoints.
238		# Note that this MUST match the macro DECOMPOSITION_NO_COMPOSE in C
239		# above!  See also the inverse lookup in recompose_code() found in
240		# src/common/unicode_norm.c.
241		if (!($flags =~ /DECOMP_COMPAT/ || $flags =~ /DECOMP_NO_COMPOSE/))
242		{
243			push @rec_info,
244			  {
245				code => $code,
246				main_index => $main_index,
247				first => $first_decomp,
248				second => $decomp_elts[0]
249			  };
250		}
251	}
252
253	if ($decomp_size == 0)
254	{
255		print $OT "\t{0x$code, $class, 0$flags, 0}";
256	}
257	elsif ($decomp_size == 1 && length($first_decomp) <= 4)
258	{
259
260		# The decomposition consists of a single codepoint, and it fits
261		# in a uint16, so we can store it "inline" in the main table.
262		$flags .= " | DECOMP_INLINE";
263		print $OT "\t{0x$code, $class, 1$flags, 0x$first_decomp}";
264	}
265	else
266	{
267		print $OT "\t{0x$code, $class, $decomp_size$flags, $decomp_index}";
268
269		# Now save the decompositions into a dedicated area that will
270		# be written afterwards.  First build the entry dedicated to
271		# a sub-table with the code and decomposition.
272		$decomp_string .= ",\n" if ($decomp_string ne "");
273
274		$decomp_string .= "\t /* $decomp_index */ 0x$first_decomp";
275		foreach (@decomp_elts)
276		{
277			$decomp_string .= ", 0x$_";
278		}
279
280		$decomp_index = $decomp_index + $decomp_size;
281	}
282
283	# Print a comma after all items except the last one.
284	print $OT "," unless ($code eq $last_code);
285
286	print $OT "\t/* $comment */" if ($comment ne "");
287	print $OT "\n";
288
289	$main_index++;
290}
291print $OT "\n};\n\n";

294print $OT <<HEADER;

299};

307print $OF "static $dec_func\n";

322# Find the lowest codepoint that decomposes to each recomposeable
323# code pair and create a mapping to it.
324my $recomp_string = "";
325my @rec_cp_packed;
326my %seenit;
327my $firstentry = 1;
328foreach my $rec (sort recomp_sort @rec_info)
329{
330	# The hash key is formed by concatenating the bytes of the two
331	# codepoints. See also recompose_code() in common/unicode_norm.c.
332	my $hashkey = (hex($rec->{first}) << 32) | hex($rec->{second});
333
334	# We are only interested in the lowest code point that decomposes
335	# to the given code pair.
336	next if $seenit{$hashkey};
337
338	# Save the hash key bytes in network order
339	push @rec_cp_packed, pack('Q>', $hashkey);
340
341	# Append inverse lookup element
342	$recomp_string .= ",\n" if !$firstentry;
343	$recomp_string .= sprintf "\t/* U+%s+%s -> U+%s */ %s",
344	  $rec->{first},
345	  $rec->{second},
346	  $rec->{code},
347	  $rec->{main_index};
348
349	$seenit{$hashkey} = 1;
350	$firstentry = 0;
351}
352

354my $num_recomps = scalar @rec_cp_packed;
355print $OF <<HEADER;

369print $OF "/* Perfect hash function for recomposition */\n";
370print $OF "static $rec_func\n";

373# one variable.

387sub recomp_sort
388{
389	my $a1 = hex($a->{first});
390	my $b1 = hex($b->{first});
391
392	my $a2 = hex($a->{second});
393	my $b2 = hex($b->{second});
394
395	# First sort by the first code point
396	return -1 if $a1 < $b1;
397	return 1 if $a1 > $b1;
398
399	# Then sort by the second code point
400	return -1 if $a2 < $b2;
401	return 1 if $a2 > $b2;
402
403	# Finally sort by the code point that decomposes into first and
404	# second ones.
405	my $acode = hex($a->{code});
406	my $bcode = hex($b->{code});
407
408	return -1 if $acode < $bcode;
409	return 1 if $acode > $bcode;
410
411	die "found duplicate entries of recomposeable code pairs";
412}
