src/common/unicode/generate-unicode_version.pl
1#!/usr/bin/perl
2#
3# Generate header file with Unicode version used by Postgres.
4#
5# Output: unicode_version.h
6#
7# Copyright (c) 2000-2025, PostgreSQL Global Development Group
8
9use strict;
10use warnings FATAL => 'all';
11use Getopt::Long;
12
13use FindBin;
14use lib "$FindBin::RealBin/../../tools/";
15
16my $output_path = '.';
17my $version_str = undef;
18
19GetOptions('outdir:s' => \$output_path, 'version:s' => \$version_str);
20
21my @version_parts = split /\./, $version_str;
22
23my $unicode_version_str = sprintf "%d.%d", $version_parts[0],
24  $version_parts[1];
25
26my $output_file = "$output_path/unicode_version.h";
27
28# Start writing out the output files
29open my $OT, '>', $output_file
30  or die "Could not open output file $output_file: $!\n";
31
32print $OT <<HEADER;
33/*-------------------------------------------------------------------------
34 *
35 * unicode_version.h
36 *	  Unicode version used by Postgres.
37 *
38 * Portions Copyright (c) 1996-2025, PostgreSQL Global Development Group
39 * Portions Copyright (c) 1994, Regents of the University of California
40 *
41 * src/include/common/unicode_version.h
42 *
43 *-------------------------------------------------------------------------
44 */
45
46#define PG_UNICODE_VERSION		"$unicode_version_str"
47HEADER
