1#!/usr/bin/perl
2#
3# Generate a code point category table and its lookup utilities, using
4# Unicode data files as input.
5#
6# Input: UnicodeData.txt
7# Output: unicode_category_table.h
8#
9# Copyright (c) 2000-2025, PostgreSQL Global Development Group
10
11use strict;
12use warnings FATAL => 'all';
13use Getopt::Long;
14
15use FindBin;
16use lib "$FindBin::RealBin/../../tools/";
17
18my $CATEGORY_UNASSIGNED = 'Cn';
19
20my $output_path = '.';
21
22GetOptions('outdir:s' => \$output_path);
23
24my $output_table_file = "$output_path/unicode_category_table.h";
25
26my $FH;
27
28# create a table of all codepoints < 0x80 and their associated
29# categories and properties for fast lookups
30my %opt_ascii = ();
31
32# Read entries from UnicodeData.txt into a list of codepoint ranges
33# and their general category.
34my @category_ranges = ();
35my $range_start = undef;
36my $range_end = undef;
37my $range_category = undef;
38
39# If between a "<..., First>" entry and a "<..., Last>" entry, the gap in
40# codepoints represents a range, and $gap_category is equal to the
41# category for both (which must match). Otherwise, the gap represents
42# unassigned code points.
43my $gap_category = undef;
44
45open($FH, '<', "$output_path/UnicodeData.txt")
46  or die "Could not open $output_path/UnicodeData.txt: $!.";
47while (my $line = <$FH>)
48{
49	my @elts = split(';', $line);
50	my $code = hex($elts[0]);
51	my $name = $elts[1];
52	my $category = $elts[2];
53
54	die "codepoint out of range" if $code > 0x10FFFF;
55	die "unassigned codepoint in UnicodeData.txt"
56	  if $category eq $CATEGORY_UNASSIGNED;
57
58	if ($code < 0x80)
59	{
60		my @properties = ();
61		# No ASCII characters have category Titlecase_Letter,
62		# but include here for completeness.
63		push @properties, "PG_U_PROP_CASED" if ($category eq 'Lt');
64		$opt_ascii{$code} = {
65			Category => $category,
66			Properties => \@properties
67		};
68	}
69
70	if (!defined($range_start))
71	{
72		my $code_str = sprintf "0x%06x", $code;
73		die
74		  if defined($range_end)
75		  || defined($range_category)
76		  || defined($gap_category);
77		die "unexpected first entry <..., Last>" if ($name =~ /Last>/);
78		die "expected 0x000000 for first entry, got $code_str"
79		  if $code != 0x000000;
80
81		# initialize
82		$range_start = $code;
83		$range_end = $code;
84		$range_category = $category;
85		if ($name =~ /<.*, First>$/)
86		{
87			$gap_category = $category;
88		}
89		else
90		{
91			$gap_category = $CATEGORY_UNASSIGNED;
92		}
93		next;
94	}
95
96	# Gap in codepoints detected. If it's a different category than
97	# the current range, emit the current range and initialize a new
98	# range representing the gap.
99	if ($range_end + 1 != $code && $range_category ne $gap_category)
100	{
101		if ($range_category ne $CATEGORY_UNASSIGNED)
102		{
103			push(
104				@category_ranges,
105				{
106					start => $range_start,
107					end => $range_end,
108					category => $range_category
109				});
110		}
111		$range_start = $range_end + 1;
112		$range_end = $code - 1;
113		$range_category = $gap_category;
114	}
115
116	# different category; new range
117	if ($range_category ne $category)
118	{
119		if ($range_category ne $CATEGORY_UNASSIGNED)
120		{
121			push(
122				@category_ranges,
123				{
124					start => $range_start,
125					end => $range_end,
126					category => $range_category
127				});
128		}
129		$range_start = $code;
130		$range_end = $code;
131		$range_category = $category;
132	}
133
134	if ($name =~ /<.*, First>$/)
135	{
136		die
137		  "<..., First> entry unexpectedly follows another <..., First> entry"
138		  if $gap_category ne $CATEGORY_UNASSIGNED;
139		$gap_category = $category;
140	}
141	elsif ($name =~ /<.*, Last>$/)
142	{
143		die
144		  "<..., First> and <..., Last> entries have mismatching general category"
145		  if $gap_category ne $category;
146		$gap_category = $CATEGORY_UNASSIGNED;
147	}
148	else
149	{
150		die "unexpected entry found between <..., First> and <..., Last>"
151		  if $gap_category ne $CATEGORY_UNASSIGNED;
152	}
153
154	$range_end = $code;
155}
156close $FH;
157
158die "<..., First> entry with no corresponding <..., Last> entry"
159  if $gap_category ne $CATEGORY_UNASSIGNED;
160
161# emit final range
162if ($range_category ne $CATEGORY_UNASSIGNED)
163{
164	push(
165		@category_ranges,
166		{
167			start => $range_start,
168			end => $range_end,
169			category => $range_category
170		});
171}
172
173# See: https://www.unicode.org/reports/tr44/#General_Category_Values
174my $categories = {
175	Cn => 'PG_U_UNASSIGNED',
176	Lu => 'PG_U_UPPERCASE_LETTER',
177	Ll => 'PG_U_LOWERCASE_LETTER',
178	Lt => 'PG_U_TITLECASE_LETTER',
179	Lm => 'PG_U_MODIFIER_LETTER',
180	Lo => 'PG_U_OTHER_LETTER',
181	Mn => 'PG_U_NONSPACING_MARK',
182	Me => 'PG_U_ENCLOSING_MARK',
183	Mc => 'PG_U_SPACING_MARK',
184	Nd => 'PG_U_DECIMAL_NUMBER',
185	Nl => 'PG_U_LETTER_NUMBER',
186	No => 'PG_U_OTHER_NUMBER',
187	Zs => 'PG_U_SPACE_SEPARATOR',
188	Zl => 'PG_U_LINE_SEPARATOR',
189	Zp => 'PG_U_PARAGRAPH_SEPARATOR',
190	Cc => 'PG_U_CONTROL',
191	Cf => 'PG_U_FORMAT',
192	Co => 'PG_U_PRIVATE_USE',
193	Cs => 'PG_U_SURROGATE',
194	Pd => 'PG_U_DASH_PUNCTUATION',
195	Ps => 'PG_U_OPEN_PUNCTUATION',
196	Pe => 'PG_U_CLOSE_PUNCTUATION',
197	Pc => 'PG_U_CONNECTOR_PUNCTUATION',
198	Po => 'PG_U_OTHER_PUNCTUATION',
199	Sm => 'PG_U_MATH_SYMBOL',
200	Sc => 'PG_U_CURRENCY_SYMBOL',
201	Sk => 'PG_U_MODIFIER_SYMBOL',
202	So => 'PG_U_OTHER_SYMBOL',
203	Pi => 'PG_U_INITIAL_PUNCTUATION',
204	Pf => 'PG_U_FINAL_PUNCTUATION'
205};
206
207# Find White_Space and Hex_Digit characters
208my @white_space = ();
209my @hex_digits = ();
210my @join_control = ();
211open($FH, '<', "$output_path/PropList.txt")
212  or die "Could not open $output_path/PropList.txt: $!.";
213while (my $line = <$FH>)
214{
215	my $pattern = qr/([0-9A-F\.]+)\s*;\s*(\w+)\s*#.*/s;
216	next unless $line =~ $pattern;
217
218	my $code = $line =~ s/$pattern/$1/rg;
219	my $property = $line =~ s/$pattern/$2/rg;
220	my $start;
221	my $end;
222
223	if ($code =~ /\.\./)
224	{
225		# code range
226		my @sp = split /\.\./, $code;
227		$start = hex($sp[0]);
228		$end = hex($sp[1]);
229	}
230	else
231	{
232		# single code point
233		$start = hex($code);
234		$end = hex($code);
235	}
236
237	if ($property eq "White_Space")
238	{
239		push @white_space, { start => $start, end => $end };
240		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
241		{
242			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_WHITE_SPACE";
243		}
244	}
245	elsif ($property eq "Hex_Digit")
246	{
247		push @hex_digits, { start => $start, end => $end };
248		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
249		{
250			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_HEX_DIGIT";
251		}
252	}
253	elsif ($property eq "Join_Control")
254	{
255		push @join_control, { start => $start, end => $end };
256		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
257		{
258			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_JOIN_CONTROL";
259		}
260	}
261}
262
263# Find Alphabetic, Lowercase, and Uppercase characters
264my @alphabetic = ();
265my @lowercase = ();
266my @uppercase = ();
267my @case_ignorable = ();
268open($FH, '<', "$output_path/DerivedCoreProperties.txt")
269  or die "Could not open $output_path/DerivedCoreProperties.txt: $!.";
270while (my $line = <$FH>)
271{
272	my $pattern = qr/^([0-9A-F\.]+)\s*;\s*(\w+)\s*#.*$/s;
273	next unless $line =~ $pattern;
274
275	my $code = $line =~ s/$pattern/$1/rg;
276	my $property = $line =~ s/$pattern/$2/rg;
277	my $start;
278	my $end;
279
280	if ($code =~ /\.\./)
281	{
282		# code range
283		my @sp = split /\.\./, $code;
284		die "line: {$line} code: {$code} sp[0] {$sp[0]} sp[1] {$sp[1]}"
285		  unless $sp[0] =~ /^[0-9A-F]+$/ && $sp[1] =~ /^[0-9A-F]+$/;
286		$start = hex($sp[0]);
287		$end = hex($sp[1]);
288	}
289	else
290	{
291		die "line: {$line} code: {$code}" unless $code =~ /^[0-9A-F]+$/;
292		# single code point
293		$start = hex($code);
294		$end = hex($code);
295	}
296
297	if ($property eq "Alphabetic")
298	{
299		push @alphabetic, { start => $start, end => $end };
300		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
301		{
302			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_ALPHABETIC";
303		}
304	}
305	elsif ($property eq "Lowercase")
306	{
307		push @lowercase, { start => $start, end => $end };
308		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
309		{
310			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_LOWERCASE";
311			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_CASED";
312		}
313	}
314	elsif ($property eq "Uppercase")
315	{
316		push @uppercase, { start => $start, end => $end };
317		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
318		{
319			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_UPPERCASE";
320			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_CASED";
321		}
322	}
323	elsif ($property eq "Case_Ignorable")
324	{
325		push @case_ignorable, { start => $start, end => $end };
326		for (my $i = $start; $i <= $end && $i < 0x80; $i++)
327		{
328			push @{ $opt_ascii{$i}{Properties} }, "PG_U_PROP_CASE_IGNORABLE";
329		}
330	}
331}
332
333my $num_category_ranges = scalar @category_ranges;
334my $num_alphabetic_ranges = scalar @alphabetic;
335my $num_lowercase_ranges = scalar @lowercase;
336my $num_uppercase_ranges = scalar @uppercase;
337my $num_case_ignorable_ranges = scalar @case_ignorable;
338my $num_white_space_ranges = scalar @white_space;
339my $num_hex_digit_ranges = scalar @hex_digits;
340my $num_join_control_ranges = scalar @join_control;
341
342# Start writing out the output file
343open my $OT, '>', $output_table_file
344  or die "Could not open output file $output_table_file: $!\n";
345

402/* table for fast lookup of ASCII codepoints */
403static const pg_unicode_properties unicode_opt_ascii[128] =
404{
405EOS
406
407for (my $i = 0; $i < 128; $i++)
408{
409	my $category_str = $categories->{ $opt_ascii{$i}->{Category} };
410	my $props_str = (join ' | ', @{ $opt_ascii{$i}{Properties} }) || "0";
411	printf $OT
412	  "\t{\n\t\t/* 0x%06x */\n\t\t.category = %s,\n\t\t.properties = %s\n\t},\n",
413	  $i, $category_str, $props_str;
414}
