11subdir = src/common/unicode

13include $(top_builddir)/src/Makefile.global

15override CPPFLAGS := -DFRONTEND -I. $(CPPFLAGS)
16LIBS += $(PTHREAD_LIBS)
17
18LDFLAGS_INTERNAL += $(ICU_LIBS)
19CPPFLAGS += $(ICU_CFLAGS)
20
21# By default, do nothing.
22all:
23
24update-unicode: unicode_case_table.h unicode_category_table.h unicode_east_asian_fw_table.h unicode_nonspacing_table.h unicode_norm_hashfunc.h unicode_norm_table.h unicode_normprops_table.h unicode_version.h
25	mv $^ $(top_srcdir)/src/include/common/
26	$(MAKE) case-check
27	$(MAKE) category-check
28	$(MAKE) normalization-check
29
30# These files are part of the Unicode Character Database. Download
31# them on demand.  The dependency on Makefile.global is for
32# UNICODE_VERSION.
33CompositionExclusions.txt CaseFolding.txt DerivedCoreProperties.txt DerivedNormalizationProps.txt EastAsianWidth.txt NormalizationTest.txt PropList.txt SpecialCasing.txt UnicodeData.txt: $(top_builddir)/src/Makefile.global
34	$(DOWNLOAD) https://www.unicode.org/Public/$(UNICODE_VERSION)/ucd/$(@F)
35
36unicode_version.h: generate-unicode_version.pl
37	$(PERL) $< --version $(UNICODE_VERSION)
38
39unicode_case_table.h: generate-unicode_case_table.pl CaseFolding.txt SpecialCasing.txt UnicodeData.txt
40	$(PERL) $<
41
42unicode_category_table.h: generate-unicode_category_table.pl DerivedCoreProperties.txt PropList.txt UnicodeData.txt
43	$(PERL) $<
44
45# Generation of conversion tables used for string normalization with
46# UTF-8 strings.
47unicode_norm_hashfunc.h: unicode_norm_table.h
48
49unicode_norm_table.h: generate-unicode_norm_table.pl UnicodeData.txt CompositionExclusions.txt
50	$(PERL) $<
51
52unicode_nonspacing_table.h: generate-unicode_nonspacing_table.pl UnicodeData.txt
53	$(PERL) $^ >$@
54
55unicode_east_asian_fw_table.h: generate-unicode_east_asian_fw_table.pl EastAsianWidth.txt
56	$(PERL) $^ >$@
57
58unicode_normprops_table.h: generate-unicode_normprops_table.pl DerivedNormalizationProps.txt
59	$(PERL) $^ >$@
