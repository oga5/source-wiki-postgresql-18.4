43override CPPFLAGS += -DFD_SETSIZE=1024
44endif
45
46all: libpgfeutils.a
47
48libpgfeutils.a: $(OBJS)
49	rm -f $@
50	$(AR) $(AROPT) $@ $^
51
52psqlscan.c: FLEXFLAGS = -Cfe -p -p
53psqlscan.c: FLEX_NO_BACKUP=yes
54
55# libpgfeutils could be useful to contrib, so install it
56install: all installdirs
57	$(INSTALL_STLIB) libpgfeutils.a '$(DESTDIR)$(libdir)/libpgfeutils.a'
58
59installdirs:
60	$(MKDIR_P) '$(DESTDIR)$(libdir)'
61
62uninstall:
63	rm -f '$(DESTDIR)$(libdir)/libpgfeutils.a'
64
65clean distclean:
66	rm -f libpgfeutils.a $(OBJS) lex.backup
67	rm -f psqlscan.c
