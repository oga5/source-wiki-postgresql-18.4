91	 * If the strnxfrm method is not trusted to return the correct results,
92	 * set strxfrm_is_safe to false. It set to false, the method will not be

94	 * estimation purposes (where incorrect results are acceptable).
95	 */
96	bool		strxfrm_is_safe;
97};

99/*
100 * We use a discriminated union to hold either a locale_t or an ICU collator.
101 * pg_locale_t is occasionally checked for truth, so make it a pointer.
102 *
103 * Also, hold two flags: whether the collation's LC_COLLATE or LC_CTYPE is C
104 * (or POSIX), so we can optimize a few code paths in various places.  For the
105 * built-in C and POSIX collations, we can know that without even doing a
106 * cache lookup, but we want to support aliases for C/POSIX too.  For the
107 * "default" collation, there are separate static cache variables, since
108 * consulting the pg_collation catalog doesn't tell us what we need.
109 *
110 * Note that some code relies on the flags not reporting false negatives
111 * (that is, saying it's not C when it is).  For example, char2wchar()
112 * could fail if the locale is C, so str_tolower() shouldn't call it
113 * in that case.
114 */
115struct pg_locale_struct
116{
117	char		provider;
118	bool		deterministic;
119	bool		collate_is_c;
120	bool		ctype_is_c;
121	bool		is_default;
122
123	const struct collate_methods *collate;	/* NULL if collate_is_c */
124
125	union
126	{
127		struct
128		{
129			const char *locale;
130			bool		casemap_full;
131		}			builtin;
132		locale_t	lt;
133#ifdef USE_ICU
134		struct
135		{
136			const char *locale;
137			UCollator  *ucol;
138		}			icu;
139#endif
140	}			info;
141};
