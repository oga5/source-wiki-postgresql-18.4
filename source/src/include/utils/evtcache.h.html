20typedef enum
21{
22	EVT_DDLCommandStart,
23	EVT_DDLCommandEnd,
24	EVT_SQLDrop,
25	EVT_TableRewrite,
26	EVT_Login,
27} EventTriggerEvent;
28
29typedef struct
30{
31	Oid			fnoid;			/* function to be called */
32	char		enabled;		/* as SESSION_REPLICATION_ROLE_* */

34} EventTriggerCacheItem;
