14#ifndef UUID_H
15#define UUID_H
16
17/* uuid size in bytes */
18#define UUID_LEN 16
19
20typedef struct pg_uuid_t
21{
22	unsigned char data[UUID_LEN];
23} pg_uuid_t;
24
25/* fmgr interface macros */
26static inline Datum
27UUIDPGetDatum(const pg_uuid_t *X)
28{
29	return PointerGetDatum(X);
30}
31
32#define PG_RETURN_UUID_P(X)		return UUIDPGetDatum(X)
33
34static inline pg_uuid_t *
35DatumGetUUIDP(Datum X)
36{
37	return (pg_uuid_t *) DatumGetPointer(X);
38}
39
40#define PG_GETARG_UUID_P(X)		DatumGetUUIDP(PG_GETARG_DATUM(X))
41
42#endif							/* UUID_H */
