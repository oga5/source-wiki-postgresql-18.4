21/*
22 * TriggerData is the node type that is passed as fmgr "context" info
23 * when a function is called by the trigger manager.
24 */
25
26#define CALLED_AS_TRIGGER(fcinfo) \
27	((fcinfo)->context != NULL && IsA((fcinfo)->context, TriggerData))

46/*
47 * The state for capturing old and new tuples into transition tables for a
48 * single ModifyTable node (or other operation source, e.g. copyfrom.c).
49 *
50 * This is per-caller to avoid conflicts in setting
51 * tcs_original_insert_tuple.  Note, however, that the pointed-to
52 * private data may be shared across multiple callers.
53 */
54struct AfterTriggersTableData;	/* private in trigger.c */
55
56typedef struct TransitionCaptureState
57{
58	/*
59	 * Is there at least one trigger specifying each transition relation on
60	 * the relation explicitly named in the DML statement or COPY command?
61	 * Note: in current usage, these flags could be part of the private state,
62	 * but it seems possibly useful to let callers see them.
63	 */
64	bool		tcs_delete_old_table;
65	bool		tcs_update_old_table;
66	bool		tcs_update_new_table;
67	bool		tcs_insert_new_table;
68
69	/*
70	 * For INSERT and COPY, it would be wasteful to convert tuples from child
71	 * format to parent format after they have already been converted in the
72	 * opposite direction during routing.  In that case we bypass conversion
73	 * and allow the inserting code (copyfrom.c and nodeModifyTable.c) to
74	 * provide a slot containing the original tuple directly.
75	 */
76	TupleTableSlot *tcs_original_insert_tuple;
77
78	/*
79	 * Private data including the tuplestore(s) into which to insert tuples.
80	 */
81	struct AfterTriggersTableData *tcs_insert_private;
82	struct AfterTriggersTableData *tcs_update_private;
83	struct AfterTriggersTableData *tcs_delete_private;
84} TransitionCaptureState;

86/*
87 * TriggerEvent bit flags
88 *
89 * Note that we assume different event types (INSERT/DELETE/UPDATE/TRUNCATE)
90 * can't be OR'd together in a single TriggerEvent.  This is unlike the
91 * situation for pg_trigger rows, so pg_trigger.tgtype uses a different
92 * representation!
93 */
94#define TRIGGER_EVENT_INSERT			0x00000000
95#define TRIGGER_EVENT_DELETE			0x00000001
96#define TRIGGER_EVENT_UPDATE			0x00000002
97#define TRIGGER_EVENT_TRUNCATE			0x00000003
98#define TRIGGER_EVENT_OPMASK			0x00000003
99
100#define TRIGGER_EVENT_ROW				0x00000004
101
102#define TRIGGER_EVENT_BEFORE			0x00000008
103#define TRIGGER_EVENT_AFTER				0x00000000
104#define TRIGGER_EVENT_INSTEAD			0x00000010
105#define TRIGGER_EVENT_TIMINGMASK		0x00000018
106
107/* More TriggerEvent flags, used only within trigger.c */
108
109#define AFTER_TRIGGER_DEFERRABLE		0x00000020
110#define AFTER_TRIGGER_INITDEFERRED		0x00000040
111
112#define TRIGGER_FIRED_BY_INSERT(event) \
113	(((event) & TRIGGER_EVENT_OPMASK) == TRIGGER_EVENT_INSERT)
114
115#define TRIGGER_FIRED_BY_DELETE(event) \
116	(((event) & TRIGGER_EVENT_OPMASK) == TRIGGER_EVENT_DELETE)
117
118#define TRIGGER_FIRED_BY_UPDATE(event) \
119	(((event) & TRIGGER_EVENT_OPMASK) == TRIGGER_EVENT_UPDATE)
120
121#define TRIGGER_FIRED_BY_TRUNCATE(event) \
122	(((event) & TRIGGER_EVENT_OPMASK) == TRIGGER_EVENT_TRUNCATE)
123
124#define TRIGGER_FIRED_FOR_ROW(event) \
125	((event) & TRIGGER_EVENT_ROW)
126
127#define TRIGGER_FIRED_FOR_STATEMENT(event) \
128	(!TRIGGER_FIRED_FOR_ROW(event))
129
130#define TRIGGER_FIRED_BEFORE(event) \
131	(((event) & TRIGGER_EVENT_TIMINGMASK) == TRIGGER_EVENT_BEFORE)
132
133#define TRIGGER_FIRED_AFTER(event) \
134	(((event) & TRIGGER_EVENT_TIMINGMASK) == TRIGGER_EVENT_AFTER)
135
136#define TRIGGER_FIRED_INSTEAD(event) \
137	(((event) & TRIGGER_EVENT_TIMINGMASK) == TRIGGER_EVENT_INSTEAD)
