33/// This is a simple memory manager which implements the methods called by
34/// the RuntimeDyld class to allocate memory for section-based loading of
35/// objects, usually those generated by the MCJIT execution engine.
36///
37/// This memory manager allocates all section memory as read-write.  The
38/// RuntimeDyld will copy JITed section memory into these allocated blocks
39/// and perform any necessary linking and relocations.
40///
41/// Any client using this memory manager MUST ensure that section-specific
42/// page permissions have been applied before attempting to execute functions
43/// in the JITed object.  Permissions can be applied either by calling
44/// MCJIT::finalizeObject or by calling SectionMemoryManager::finalizeMemory
45/// directly.  Clients of MCJIT should call MCJIT::finalizeObject.
46class SectionMemoryManager : public RTDyldMemoryManager {
47public:
48  /// This enum describes the various reasons to allocate pages from
49  /// allocateMappedMemory.
50  enum class AllocationPurpose {
51    Code,
52    ROData,
53    RWData,
54  };
55
56  /// Implementations of this interface are used by SectionMemoryManager to
57  /// request pages from the operating system.
58  class MemoryMapper {
59  public:
60    /// This method attempts to allocate \p NumBytes bytes of virtual memory for
61    /// \p Purpose.  \p NearBlock may point to an existing allocation, in which
62    /// case an attempt is made to allocate more memory near the existing block.
63    /// The actual allocated address is not guaranteed to be near the requested
64    /// address.  \p Flags is used to set the initial protection flags for the
65    /// block of the memory.  \p EC [out] returns an object describing any error
66    /// that occurs.
67    ///
68    /// This method may allocate more than the number of bytes requested.  The
69    /// actual number of bytes allocated is indicated in the returned
70    /// MemoryBlock.
71    ///
72    /// The start of the allocated block must be aligned with the system
73    /// allocation granularity (64K on Windows, page size on Linux).  If the
74    /// address following \p NearBlock is not so aligned, it will be rounded up
75    /// to the next allocation granularity boundary.
76    ///
77    /// \r a non-null MemoryBlock if the function was successful, otherwise a
78    /// null MemoryBlock with \p EC describing the error.
79    virtual sys::MemoryBlock
80    allocateMappedMemory(AllocationPurpose Purpose, size_t NumBytes,
81                         const sys::MemoryBlock *const NearBlock,
82                         unsigned Flags, std::error_code &EC) = 0;
83
84    /// This method sets the protection flags for a block of memory to the state
85    /// specified by \p Flags.  The behavior is not specified if the memory was
86    /// not allocated using the allocateMappedMemory method.
87    /// \p Block describes the memory block to be protected.
88    /// \p Flags specifies the new protection state to be assigned to the block.
89    ///
90    /// If \p Flags is MF_WRITE, the actual behavior varies with the operating
91    /// system (i.e. MF_READ | MF_WRITE on Windows) and the target architecture
92    /// (i.e. MF_WRITE -> MF_READ | MF_WRITE on i386).
93    ///
94    /// \r error_success if the function was successful, or an error_code
95    /// describing the failure if an error occurred.
96    virtual std::error_code protectMappedMemory(const sys::MemoryBlock &Block,
97                                                unsigned Flags) = 0;
98
99    /// This method releases a block of memory that was allocated with the
100    /// allocateMappedMemory method. It should not be used to release any memory
101    /// block allocated any other way.
102    /// \p Block describes the memory to be released.
103    ///
104    /// \r error_success if the function was successful, or an error_code
105    /// describing the failure if an error occurred.
106    virtual std::error_code releaseMappedMemory(sys::MemoryBlock &M) = 0;
107
108    virtual ~MemoryMapper();
109  };
110
111  /// Creates a SectionMemoryManager instance with \p MM as the associated
112  /// memory mapper.  If \p MM is nullptr then a default memory mapper is used
113  /// that directly calls into the operating system.
114  ///
115  /// If \p ReserveAlloc is true all memory will be pre-allocated, and any
116  /// attempts to allocate beyond pre-allocated memory will fail.
117  SectionMemoryManager(MemoryMapper *MM = nullptr, bool ReserveAlloc = false);
118  SectionMemoryManager(const SectionMemoryManager &) = delete;
119  void operator=(const SectionMemoryManager &) = delete;
120  ~SectionMemoryManager() override;
121
122  /// Enable reserveAllocationSpace when requested.
123  bool needsToReserveAllocationSpace() override { return ReserveAllocation; }
124
125  /// Implements allocating all memory in a single block. This is required to
126  /// limit memory offsets to fit the ARM ABI; large memory systems may
127  /// otherwise allocate separate sections too far apart.
128#if LLVM_VERSION_MAJOR < 16
129  virtual void reserveAllocationSpace(uintptr_t CodeSize, uint32_t CodeAlign,
130                                      uintptr_t RODataSize,
131                                      uint32_t RODataAlign,
132                                      uintptr_t RWDataSize,
133                                      uint32_t RWDataAlign) override;
134#else
135  void reserveAllocationSpace(uintptr_t CodeSize, Align CodeAlign,
136                              uintptr_t RODataSize, Align RODataAlign,
137                              uintptr_t RWDataSize, Align RWDataAlign) override;
138#endif
139
140  /// Allocates a memory block of (at least) the given size suitable for
141  /// executable code.
142  ///
143  /// The value of \p Alignment must be a power of two.  If \p Alignment is zero
144  /// a default alignment of 16 will be used.
145  uint8_t *allocateCodeSection(uintptr_t Size, unsigned Alignment,
146                               unsigned SectionID,
147                               StringRef SectionName) override;
148
149  /// Allocates a memory block of (at least) the given size suitable for
150  /// executable code.
151  ///
152  /// The value of \p Alignment must be a power of two.  If \p Alignment is zero
153  /// a default alignment of 16 will be used.
154  uint8_t *allocateDataSection(uintptr_t Size, unsigned Alignment,
155                               unsigned SectionID, StringRef SectionName,
156                               bool IsReadOnly) override;
157
158  /// Update section-specific memory permissions and other attributes.
159  ///
160  /// This method is called when object loading is complete and section page
161  /// permissions can be applied.  It is up to the memory manager implementation
162  /// to decide whether or not to act on this method.  The memory manager will
163  /// typically allocate all sections as read-write and then apply specific
164  /// permissions when this method is called.  Code sections cannot be executed
165  /// until this function has been called.  In addition, any cache coherency
166  /// operations needed to reliably use the memory are also performed.
167  ///
168  /// \returns true if an error occurred, false otherwise.
169  bool finalizeMemory(std::string *ErrMsg = nullptr) override;
170
171  /// Invalidate instruction cache for code sections.
172  ///
173  /// Some platforms with separate data cache and instruction cache require
174  /// explicit cache flush, otherwise JIT code manipulations (like resolved
175  /// relocations) will get to the data cache but not to the instruction cache.
176  ///
177  /// This method is called from finalizeMemory.
178  virtual void invalidateInstructionCache();
179
180private:
181  struct FreeMemBlock {
182    // The actual block of free memory
183    sys::MemoryBlock Free;
184    // If there is a pending allocation from the same reservation right before
185    // this block, store its index in PendingMem, to be able to update the
186    // pending region if part of this block is allocated, rather than having to
187    // create a new one
188    unsigned PendingPrefixIndex;
189  };
190
191  struct MemoryGroup {
192    // PendingMem contains all blocks of memory (subblocks of AllocatedMem)
193    // which have not yet had their permissions applied, but have been given
194    // out to the user. FreeMem contains all block of memory, which have
195    // neither had their permissions applied, nor been given out to the user.
196    SmallVector<sys::MemoryBlock, 16> PendingMem;
197    SmallVector<FreeMemBlock, 16> FreeMem;
198
199    // All memory blocks that have been requested from the system
200    SmallVector<sys::MemoryBlock, 16> AllocatedMem;
201
202    sys::MemoryBlock Near;
203  };
204
205  uint8_t *allocateSection(AllocationPurpose Purpose, uintptr_t Size,
206                           unsigned Alignment);
207
208  std::error_code applyMemoryGroupPermissions(MemoryGroup &MemGroup,
209                                              unsigned Permissions);
210
211  bool hasSpace(const MemoryGroup &MemGroup, uintptr_t Size) const;
212
213  void anchor() override;
214
215  MemoryGroup CodeMem;
216  MemoryGroup RWDataMem;
217  MemoryGroup RODataMem;
218  MemoryMapper *MMapper;
219  std::unique_ptr<MemoryMapper> OwnedMMapper;
220  bool ReserveAllocation;
221};
