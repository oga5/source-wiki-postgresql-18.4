22/*
23 *	This structure holds a list of possible functions or operators
24 *	found by namespace lookup.  Each function/operator is identified
25 *	by OID and by argument types; the list must be pruned by type
26 *	resolution rules that are embodied in the parser, not here.
27 *	See FuncnameGetCandidates's comments for more info.
28 */
29typedef struct _FuncCandidateList
30{
31	struct _FuncCandidateList *next;
32	int			pathpos;		/* for internal use of namespace lookup */
33	Oid			oid;			/* the function or operator's OID */
34	int			nominalnargs;	/* either pronargs or length(proallargtypes) */
35	int			nargs;			/* number of arg types returned */
36	int			nvargs;			/* number of args to become variadic array */
37	int			ndargs;			/* number of defaulted args */
38	int		   *argnumbers;		/* args' positional indexes, if named call */
39	Oid			args[FLEXIBLE_ARRAY_MEMBER];	/* arg types */
40}		   *FuncCandidateList;

42/*
43 * Result of checkTempNamespaceStatus
44 */
45typedef enum TempNamespaceStatus
46{
47	TEMP_NAMESPACE_NOT_TEMP,	/* nonexistent, or non-temp namespace */
48	TEMP_NAMESPACE_IDLE,		/* exists, belongs to no active session */
49	TEMP_NAMESPACE_IN_USE,		/* belongs to some active session */
50} TempNamespaceStatus;

52/*
53 *	Structure for xxxSearchPathMatcher functions
54 *
55 * The generation counter is private to namespace.c and shouldn't be touched
56 * by other code.  It can be initialized to zero if necessary (that means
57 * "not known equal to the current active path").
58 */
59typedef struct SearchPathMatcher
60{
61	List	   *schemas;		/* OIDs of explicitly named schemas */
62	bool		addCatalog;		/* implicitly prepend pg_catalog? */
63	bool		addTemp;		/* implicitly prepend temp schema? */
64	uint64		generation;		/* for quick detection of equality to active */
65} SearchPathMatcher;
