149 * pg_noreturn corresponds to the C11 noreturn/_Noreturn function specifier.
150 * We can't use the standard name "noreturn" because some third-party code
151 * uses __attribute__((noreturn)) in headers, which would get confused if

376/*
377 * VA_ARGS_NARGS
378 *		Returns the number of macro arguments it is passed.
379 *
380 * An empty argument still counts as an argument, so effectively, this is
381 * "one more than the number of commas in the argument list".
382 *
383 * This works for up to 63 arguments.  Internally, VA_ARGS_NARGS_() is passed
384 * 64+N arguments, and the C99 standard only requires macros to allow up to
385 * 127 arguments, so we can't portably go higher.  The implementation is
386 * pretty trivial: VA_ARGS_NARGS_() returns its 64th argument, and we set up
387 * the call so that that is the appropriate one of the list of constants.
388 * This idea is due to Laurent Deniau.
389 *
390 * MSVC has an implementation of __VA_ARGS__ that doesn't conform to the
391 * standard unless you use the /Zc:preprocessor compiler flag, but that
392 * isn't available before Visual Studio 2019.  For now, use a different
393 * definition that also works on older compilers.
394 */
395#ifdef _MSC_VER
396#define EXPAND(args) args
397#define VA_ARGS_NARGS(...) \
398	VA_ARGS_NARGS_ EXPAND((__VA_ARGS__, \
399				   63,62,61,60,                   \
400				   59,58,57,56,55,54,53,52,51,50, \
401				   49,48,47,46,45,44,43,42,41,40, \
402				   39,38,37,36,35,34,33,32,31,30, \
403				   29,28,27,26,25,24,23,22,21,20, \
404				   19,18,17,16,15,14,13,12,11,10, \
405				   9, 8, 7, 6, 5, 4, 3, 2, 1, 0))
406#else
407
408#define VA_ARGS_NARGS(...) \
409	VA_ARGS_NARGS_(__VA_ARGS__, \
410				   63,62,61,60,                   \
411				   59,58,57,56,55,54,53,52,51,50, \
412				   49,48,47,46,45,44,43,42,41,40, \
413				   39,38,37,36,35,34,33,32,31,30, \
414				   29,28,27,26,25,24,23,22,21,20, \
415				   19,18,17,16,15,14,13,12,11,10, \
416				   9, 8, 7, 6, 5, 4, 3, 2, 1, 0)
417#endif
418
419#define VA_ARGS_NARGS_( \
420	_01,_02,_03,_04,_05,_06,_07,_08,_09,_10, \
421	_11,_12,_13,_14,_15,_16,_17,_18,_19,_20, \
422	_21,_22,_23,_24,_25,_26,_27,_28,_29,_30, \
423	_31,_32,_33,_34,_35,_36,_37,_38,_39,_40, \

446/*
447 * Does the compiler support #pragma GCC system_header? We optionally use it
448 * to avoid warnings that we can't fix (e.g. in the perl headers).
449 * See https://gcc.gnu.org/onlinedocs/cpp/System-Headers.html
450 *
451 * Headers for which we do not want to show compiler warnings can,
452 * conditionally, use #pragma GCC system_header to avoid warnings. Obviously
453 * this should only be used for external headers over which we do not have
454 * control.
455 *
456 * Support for the pragma is tested here, instead of during configure, as gcc
457 * also warns about the pragma being used in a .c file. It's surprisingly hard
458 * to get autoconf to use .h as the file-ending. Looks like gcc has
459 * implemented the pragma since the 2000, so this test should suffice.
460 *
461 *
462 * Alternatively, we could add the include paths for problematic headers with
463 * -isystem, but that is a larger hammer and is harder to search for.
464 *
465 * A more granular alternative would be to use #pragma GCC diagnostic
466 * push/ignored/pop, but gcc warns about unknown warnings being ignored, so
467 * every to-be-ignored-temporarily compiler warning would require its own
468 * pg_config.h symbol and #ifdef.
469 */

471#define HAVE_PRAGMA_GCC_SYSTEM_HEADER	1
472#endif

651/* ----------------
652 *		Variable-length datatypes all share the 'struct varlena' header.
653 *
654 * NOTE: for TOASTable types, this is an oversimplification, since the value
655 * may be compressed or moved out-of-line.  However datatype-specific routines
656 * are mostly content to deal with de-TOASTed values only, and of course
657 * client-side routines should never see a TOASTed value.  But even in a
658 * de-TOASTed value, beware of touching vl_len_ directly, as its
659 * representation is no longer convenient.  It's recommended that code always
660 * use macros VARDATA_ANY, VARSIZE_ANY, VARSIZE_ANY_EXHDR, VARDATA, VARSIZE,
661 * and SET_VARSIZE instead of relying on direct mentions of the struct fields.
662 * See postgres.h for details of the TOASTed form.
663 * ----------------
664 */
665struct varlena
666{
667	char		vl_len_[4];		/* Do not touch this field directly! */
668	char		vl_dat[FLEXIBLE_ARRAY_MEMBER];	/* Data content is here */
669};

674 * These widely-used datatypes are just a varlena header and the data bytes.

769/* ----------------
770 * Alignment macros: align a length or address appropriately for a given type.
771 * The fooALIGN() macros round up to a multiple of the required alignment,
772 * while the fooALIGN_DOWN() macros round down.  The latter are more useful
773 * for problems like "how many X-sized structures will fit in a page?".
774 *
775 * NOTE: TYPEALIGN[_DOWN] will not work if ALIGNVAL is not a power of 2.
776 * That case seems extremely unlikely to be needed in practice, however.
777 *
778 * NOTE: MAXIMUM_ALIGNOF, and hence MAXALIGN(), intentionally exclude any
779 * larger-than-8-byte types the compiler might have.
780 * ----------------
781 */
782
783#define TYPEALIGN(ALIGNVAL,LEN)  \
784	(((uintptr_t) (LEN) + ((ALIGNVAL) - 1)) & ~((uintptr_t) ((ALIGNVAL) - 1)))
785
786#define SHORTALIGN(LEN)			TYPEALIGN(ALIGNOF_SHORT, (LEN))
787#define INTALIGN(LEN)			TYPEALIGN(ALIGNOF_INT, (LEN))
788#define LONGALIGN(LEN)			TYPEALIGN(ALIGNOF_LONG, (LEN))
789#define DOUBLEALIGN(LEN)		TYPEALIGN(ALIGNOF_DOUBLE, (LEN))
790#define MAXALIGN(LEN)			TYPEALIGN(MAXIMUM_ALIGNOF, (LEN))
791/* MAXALIGN covers only built-in types, not buffers */
792#define BUFFERALIGN(LEN)		TYPEALIGN(ALIGNOF_BUFFER, (LEN))
793#define CACHELINEALIGN(LEN)		TYPEALIGN(PG_CACHE_LINE_SIZE, (LEN))
794
795#define TYPEALIGN_DOWN(ALIGNVAL,LEN)  \
796	(((uintptr_t) (LEN)) & ~((uintptr_t) ((ALIGNVAL) - 1)))
797
798#define SHORTALIGN_DOWN(LEN)	TYPEALIGN_DOWN(ALIGNOF_SHORT, (LEN))
799#define INTALIGN_DOWN(LEN)		TYPEALIGN_DOWN(ALIGNOF_INT, (LEN))
800#define LONGALIGN_DOWN(LEN)		TYPEALIGN_DOWN(ALIGNOF_LONG, (LEN))
801#define DOUBLEALIGN_DOWN(LEN)	TYPEALIGN_DOWN(ALIGNOF_DOUBLE, (LEN))
802#define MAXALIGN_DOWN(LEN)		TYPEALIGN_DOWN(MAXIMUM_ALIGNOF, (LEN))
803#define BUFFERALIGN_DOWN(LEN)	TYPEALIGN_DOWN(ALIGNOF_BUFFER, (LEN))
804
805/*
806 * The above macros will not work with types wider than uintptr_t, like with
807 * uint64 on 32-bit platforms.  That's not problem for the usual use where a
808 * pointer or a length is aligned, but for the odd case that you need to
809 * align something (potentially) wider, use TYPEALIGN64.
810 */
811#define TYPEALIGN64(ALIGNVAL,LEN)  \
812	(((uint64) (LEN) + ((ALIGNVAL) - 1)) & ~((uint64) ((ALIGNVAL) - 1)))
813
814/* we don't currently need wider versions of the other ALIGN macros */
815#define MAXALIGN64(LEN)			TYPEALIGN64(MAXIMUM_ALIGNOF, (LEN))

887/*
888 * Macros to support compile-time assertion checks.
889 *
890 * If the "condition" (a compile-time-constant expression) evaluates to false,
891 * throw a compile error using the "errmessage" (a string literal).
892 *
893 * C11 has _Static_assert(), and most C99 compilers already support that.  For
894 * portability, we wrap it into StaticAssertDecl().  _Static_assert() is a
895 * "declaration", and so it must be placed where for example a variable
896 * declaration would be valid.  As long as we compile with
897 * -Wno-declaration-after-statement, that also means it cannot be placed after
898 * statements in a function.  Macros StaticAssertStmt() and StaticAssertExpr()
899 * make it safe to use as a statement or in an expression, respectively.
900 *
901 * For compilers without _Static_assert(), we fall back on a kluge that
902 * assumes the compiler will complain about a negative width for a struct
903 * bit-field.  This will not include a helpful error message, but it beats not
904 * getting an error at all.
905 */
906#ifndef __cplusplus
907#ifdef HAVE__STATIC_ASSERT
908#define StaticAssertDecl(condition, errmessage) \
909	_Static_assert(condition, errmessage)
910#define StaticAssertStmt(condition, errmessage) \
911	do { _Static_assert(condition, errmessage); } while(0)
912#define StaticAssertExpr(condition, errmessage) \
913	((void) ({ StaticAssertStmt(condition, errmessage); true; }))
914#else							/* !HAVE__STATIC_ASSERT */
915#define StaticAssertDecl(condition, errmessage) \
916	extern void static_assert_func(int static_assert_failure[(condition) ? 1 : -1])
917#define StaticAssertStmt(condition, errmessage) \
918	((void) sizeof(struct { int static_assert_failure : (condition) ? 1 : -1; }))
919#define StaticAssertExpr(condition, errmessage) \
920	StaticAssertStmt(condition, errmessage)
921#endif							/* HAVE__STATIC_ASSERT */
922#else							/* C++ */
923#if defined(__cpp_static_assert) && __cpp_static_assert >= 200410
924#define StaticAssertDecl(condition, errmessage) \
925	static_assert(condition, errmessage)
926#define StaticAssertStmt(condition, errmessage) \
927	static_assert(condition, errmessage)
928#define StaticAssertExpr(condition, errmessage) \
929	({ static_assert(condition, errmessage); })
930#else							/* !__cpp_static_assert */
931#define StaticAssertDecl(condition, errmessage) \
932	extern void static_assert_func(int static_assert_failure[(condition) ? 1 : -1])
933#define StaticAssertStmt(condition, errmessage) \
934	do { struct static_assert_struct { int static_assert_failure : (condition) ? 1 : -1; }; } while(0)
935#define StaticAssertExpr(condition, errmessage) \
936	((void) ({ StaticAssertStmt(condition, errmessage); }))
937#endif							/* __cpp_static_assert */
938#endif							/* C++ */

989/*
990 * MemSet
991 *	Exactly the same as standard library function memset(), but considerably
992 *	faster for zeroing small word-aligned structures (such as parsetree nodes).
993 *	This has to be a macro because the main point is to avoid function-call
994 *	overhead.   However, we have also found that the loop is faster than
995 *	native libc memset() on some platforms, even those with assembler
996 *	memset() functions.  More research needs to be done, perhaps with
997 *	MEMSET_LOOP_LIMIT tests in configure.
998 */
999#define MemSet(start, val, len) \
1000	do \
1001	{ \
1002		/* must be void* because we don't know if it is integer aligned yet */ \
1003		void   *_vstart = (void *) (start); \
1004		int		_val = (val); \
1005		Size	_len = (len); \
1006\
1007		if ((((uintptr_t) _vstart) & LONG_ALIGN_MASK) == 0 && \
1008			(_len & LONG_ALIGN_MASK) == 0 && \
1009			_val == 0 && \
1010			_len <= MEMSET_LOOP_LIMIT && \
1011			/* \
1012			 *	If MEMSET_LOOP_LIMIT == 0, optimizer should find \
1013			 *	the whole "if" false at compile time. \
1014			 */ \
1015			MEMSET_LOOP_LIMIT != 0) \
1016		{ \
1017			long *_start = (long *) _vstart; \
1018			long *_stop = (long *) ((char *) _start + _len); \
1019			while (_start < _stop) \
1020				*_start++ = 0; \
1021		} \
1022		else \
1023			memset(_vstart, _val, _len); \
1024	} while (0)
1025
1026/*
1027 * MemSetAligned is the same as MemSet except it omits the test to see if
1028 * "start" is word-aligned.  This is okay to use if the caller knows a-priori
1029 * that the pointer is suitably aligned (typically, because he just got it
1030 * from palloc(), which always delivers a max-aligned pointer).
1031 */
1032#define MemSetAligned(start, val, len) \
1033	do \
1034	{ \
1035		long   *_start = (long *) (start); \
1036		int		_val = (val); \
1037		Size	_len = (len); \
1038\
1039		if ((_len & LONG_ALIGN_MASK) == 0 && \
1040			_val == 0 && \
1041			_len <= MEMSET_LOOP_LIMIT && \
1042			MEMSET_LOOP_LIMIT != 0) \
1043		{ \
1044			long *_stop = (long *) ((char *) _start + _len); \
1045			while (_start < _stop) \
1046				*_start++ = 0; \
1047		} \
1048		else \
1049			memset(_start, _val, _len); \
1050	} while (0)

1270/*
1271 * Thin wrappers that convert strings to exactly 64-bit integers, matching our
1272 * definition of int64.  (For the naming, compare that POSIX has
1273 * strtoimax()/strtoumax() which return intmax_t/uintmax_t.)
1274 */
1275#if SIZEOF_LONG == 8
1276#define strtoi64(str, endptr, base) ((int64) strtol(str, endptr, base))
1277#define strtou64(str, endptr, base) ((uint64) strtoul(str, endptr, base))
1278#elif SIZEOF_LONG_LONG == 8
1279#define strtoi64(str, endptr, base) ((int64) strtoll(str, endptr, base))
1280#define strtou64(str, endptr, base) ((uint64) strtoull(str, endptr, base))
1281#else
1282#error "cannot find integer type of the same size as int64_t"
1283#endif
