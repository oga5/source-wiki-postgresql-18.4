29/*
30 * Generic descriptor for table scans. This is the base-class for table scans,
31 * which needs to be embedded in the scans of individual AMs.
32 */
33typedef struct TableScanDescData
34{
35	/* scan parameters */
36	Relation	rs_rd;			/* heap relation descriptor */
37	struct SnapshotData *rs_snapshot;	/* snapshot to see */
38	int			rs_nkeys;		/* number of scan keys */
39	struct ScanKeyData *rs_key; /* array of scan key descriptors */
40
41	/*
42	 * Scan type-specific members
43	 */
44	union
45	{
46		/* Iterator for Bitmap Table Scans */
47		TBMIterator rs_tbmiterator;
48
49		/*
50		 * Range of ItemPointers for table_scan_getnextslot_tidrange() to
51		 * scan.
52		 */
53		struct
54		{
55			ItemPointerData rs_mintid;
56			ItemPointerData rs_maxtid;
57		}			tidrange;
58	}			st;
59
60	/*
61	 * Information about type and behaviour of the scan, a bitmask of members
62	 * of the ScanOptions enum (see tableam.h).
63	 */
64	uint32		rs_flags;
65
66	struct ParallelTableScanDescData *rs_parallel;	/* parallel scan
67													 * information */
68} TableScanDescData;
69typedef struct TableScanDescData *TableScanDesc;
70
71/*
72 * Shared state for parallel table scan.
73 *
74 * Each backend participating in a parallel table scan has its own
75 * TableScanDesc in backend-private memory, and those objects all contain a
76 * pointer to this structure.  The information here must be sufficient to
77 * properly initialize each new TableScanDesc as workers join the scan, and it
78 * must act as a information what to scan for those workers.
79 */
80typedef struct ParallelTableScanDescData
81{
82	RelFileLocator phs_locator; /* physical relation to scan */
83	bool		phs_syncscan;	/* report location to syncscan logic? */
84	bool		phs_snapshot_any;	/* SnapshotAny, not phs_snapshot_data? */
85	Size		phs_snapshot_off;	/* data for snapshot */
86} ParallelTableScanDescData;
87typedef struct ParallelTableScanDescData *ParallelTableScanDesc;
88
89/*
90 * Shared state for parallel table scans, for block oriented storage.
91 */
92typedef struct ParallelBlockTableScanDescData
93{
94	ParallelTableScanDescData base;
95
96	BlockNumber phs_nblocks;	/* # blocks in relation at start of scan */
97	slock_t		phs_mutex;		/* mutual exclusion for setting startblock */
98	BlockNumber phs_startblock; /* starting block number */
99	pg_atomic_uint64 phs_nallocated;	/* number of blocks allocated to
100										 * workers so far. */
101}			ParallelBlockTableScanDescData;
102typedef struct ParallelBlockTableScanDescData *ParallelBlockTableScanDesc;
103
104/*
105 * Per backend state for parallel table scan, for block-oriented storage.
106 */
107typedef struct ParallelBlockTableScanWorkerData
108{
109	uint64		phsw_nallocated;	/* Current # of blocks into the scan */
110	uint32		phsw_chunk_remaining;	/* # blocks left in this chunk */
111	uint32		phsw_chunk_size;	/* The number of blocks to allocate in
112									 * each I/O chunk for the scan */
113} ParallelBlockTableScanWorkerData;
114typedef struct ParallelBlockTableScanWorkerData *ParallelBlockTableScanWorker;

126struct IndexScanInstrumentation;
127
128/*
129 * We use the same IndexScanDescData structure for both amgettuple-based
130 * and amgetbitmap-based index scans.  Some fields are only relevant in
131 * amgettuple-based scans.
132 */
133typedef struct IndexScanDescData
134{
135	/* scan parameters */
136	Relation	heapRelation;	/* heap relation descriptor, or NULL */
137	Relation	indexRelation;	/* index relation descriptor */
138	struct SnapshotData *xs_snapshot;	/* snapshot to see */
139	int			numberOfKeys;	/* number of index qualifier conditions */
140	int			numberOfOrderBys;	/* number of ordering operators */
141	struct ScanKeyData *keyData;	/* array of index qualifier descriptors */
142	struct ScanKeyData *orderByData;	/* array of ordering op descriptors */
143	bool		xs_want_itup;	/* caller requests index tuples */
144	bool		xs_temp_snap;	/* unregister snapshot at scan end? */
145
146	/* signaling to index AM about killing index tuples */
147	bool		kill_prior_tuple;	/* last-returned tuple is dead */
148	bool		ignore_killed_tuples;	/* do not return killed entries */
149	bool		xactStartedInRecovery;	/* prevents killing/seeing killed
150										 * tuples */
151
152	/* index access method's private state */
153	void	   *opaque;			/* access-method-specific info */
154
155	/*
156	 * Instrumentation counters maintained by all index AMs during both
157	 * amgettuple calls and amgetbitmap calls (unless field remains NULL)
158	 */
159	struct IndexScanInstrumentation *instrument;
160
161	/*
162	 * In an index-only scan, a successful amgettuple call must fill either
163	 * xs_itup (and xs_itupdesc) or xs_hitup (and xs_hitupdesc) to provide the
164	 * data returned by the scan.  It can fill both, in which case the heap
165	 * format will be used.
166	 */
167	IndexTuple	xs_itup;		/* index tuple returned by AM */
168	struct TupleDescData *xs_itupdesc;	/* rowtype descriptor of xs_itup */
169	HeapTuple	xs_hitup;		/* index data returned by AM, as HeapTuple */
170	struct TupleDescData *xs_hitupdesc; /* rowtype descriptor of xs_hitup */
171
172	ItemPointerData xs_heaptid; /* result */
173	bool		xs_heap_continue;	/* T if must keep walking, potential
174									 * further results */
175	IndexFetchTableData *xs_heapfetch;
176
177	bool		xs_recheck;		/* T means scan keys must be rechecked */
178
179	/*
180	 * When fetching with an ordering operator, the values of the ORDER BY
181	 * expressions of the last returned tuple, according to the index.  If
182	 * xs_recheckorderby is true, these need to be rechecked just like the
183	 * scan keys, and the values returned here are a lower-bound on the actual
184	 * values.
185	 */
186	Datum	   *xs_orderbyvals;
187	bool	   *xs_orderbynulls;
188	bool		xs_recheckorderby;
189
190	/* parallel index scan information, in shared memory */
191	struct ParallelIndexScanDescData *parallel_scan;
192}			IndexScanDescData;
193
194/* Generic structure for parallel scans */
195typedef struct ParallelIndexScanDescData
196{
197	RelFileLocator ps_locator;	/* physical table relation to scan */
198	RelFileLocator ps_indexlocator; /* physical index relation to scan */
199	Size		ps_offset_ins;	/* Offset to SharedIndexScanInstrumentation */
200	Size		ps_offset_am;	/* Offset to am-specific structure */
201	char		ps_snapshot_data[FLEXIBLE_ARRAY_MEMBER];
202}			ParallelIndexScanDescData;
