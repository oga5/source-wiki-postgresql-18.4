13-----------------------------
14-- Window Functions
15-----------------------------
16
17-- a sample table
18CREATE TABLE empsalary (
19    depname text,
20    empno bigint,
21    salary int,
22    enroll_date date
23);
24
25INSERT INTO empsalary VALUES
26('develop', 10, 5200, '2007-08-01'),
27('sales', 1, 5000, '2006-10-01'),
28('personnel', 5, 3500, '2007-12-10'),
29('sales', 4, 4800, '2007-08-08'),
30('personnel', 2, 3900, '2006-12-23'),
31('develop', 7, 4200, '2008-01-01'),
32('develop', 9, 4500, '2008-01-01'),
33('sales', 3, 4800, '2007-08-01'),
34('develop', 8, 6000, '2006-10-01'),
35('develop', 11, 5200, '2007-08-15');
36
37SELECT depname, empno, salary, avg(salary) OVER (PARTITION BY depname)
38  FROM empsalary;
39
40SELECT depname, empno, salary,
41       row_number() OVER (PARTITION BY depname ORDER BY salary DESC)
42FROM empsalary;
43
44SELECT salary, sum(salary) OVER () FROM empsalary;
45
46SELECT salary, sum(salary) OVER (ORDER BY salary) FROM empsalary;
47
48SELECT depname, empno, salary, enroll_date
49FROM
50  (SELECT depname, empno, salary, enroll_date,
51     row_number() OVER (PARTITION BY depname ORDER BY salary DESC, empno) AS pos
52     FROM empsalary
53  ) AS ss
54WHERE pos < 3;
55
56SELECT sum(salary) OVER w, avg(salary) OVER w
57  FROM empsalary
58  WINDOW w AS (PARTITION BY depname ORDER BY salary DESC);
59
60DROP TABLE empsalary;

62-----------------------------
63-- Inheritance:
64--	A table can inherit from zero or more tables.  A query can reference
65--	either all rows of a table or all rows of a table plus all of its
66--	descendants.
67-----------------------------
68
69-- For example, the capitals table inherits from cities table. (It inherits
70-- all data fields from cities.)
71
72CREATE TABLE cities (
73	name		text,
74	population	float8,
75	elevation	int		-- (in ft)
76);
77
78CREATE TABLE capitals (
79	state		char(2)
80) INHERITS (cities);
81
82-- Now, let's populate the tables.
83INSERT INTO cities VALUES ('San Francisco', 7.24E+5, 63);
84INSERT INTO cities VALUES ('Las Vegas', 2.583E+5, 2174);
85INSERT INTO cities VALUES ('Mariposa', 1200, 1953);
86
87INSERT INTO capitals VALUES ('Sacramento', 3.694E+5, 30, 'CA');
88INSERT INTO capitals VALUES ('Madison', 1.913E+5, 845, 'WI');
89
90SELECT * FROM cities;
91SELECT * FROM capitals;
92
93-- You can find all cities, including capitals, that
94-- are located at an elevation of 500 ft or higher by:
95
96SELECT c.name, c.elevation
97FROM cities c
98WHERE c.elevation > 500;
99
100-- To scan rows of the parent table only, use ONLY:
101
102SELECT name, elevation
103FROM ONLY cities
104WHERE elevation > 500;
105
106
107-- clean up (you must remove the children first)
108DROP TABLE capitals;
109DROP TABLE cities;
