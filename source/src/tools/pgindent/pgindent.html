6# Exit codes:

12use strict;
13use warnings FATAL => 'all';
14
15use Cwd qw(abs_path getcwd);
16use File::Find;
17use File::Spec;
18use File::Temp;
19use IO::Handle;
20use Getopt::Long;

22# Update for pg_bsd_indent version
23my $INDENT_VERSION = "2.1.2";
24
25# Our standard indent settings
26my $indent_opts =
27  "-bad -bap -bbb -bc -bl -cli1 -cp33 -cdb -nce -d0 -di12 -nfc1 -i4 -l79 -lp -lpl -nip -npro -sac -tpg -ts4";
28
29my $devnull = File::Spec->devnull;
30
31my ($typedefs_file, $typedef_str, @excludes, $indent, $diff,
32	$check, $help, @commits,);
33
34$help = 0;
35
36my %options = (
37	"help" => \$help,
38	"commit=s" => \@commits,
39	"typedefs=s" => \$typedefs_file,
40	"list-of-typedefs=s" => \$typedef_str,
41	"excludes=s" => \@excludes,
42	"indent=s" => \$indent,
43	"diff" => \$diff,
44	"check" => \$check,);
45GetOptions(%options) || usage("bad command line argument");
46
47usage() if $help;
48
49usage("Cannot use --commit with command line file list")
50  if (@commits && @ARGV);
51
52# command line option wins, then environment, then locations based on current
53# dir, then default location
54$typedefs_file ||= $ENV{PGTYPEDEFS};
55
56# get indent location for environment or default
57$indent ||= $ENV{PGINDENT} || $ENV{INDENT} || "pg_bsd_indent";
58
59my $sourcedir = locate_sourcedir();
60
61# if it's the base of a postgres tree, we will exclude the files
62# postgres wants excluded
63if ($sourcedir)
64{
65	my $exclude_candidate = "$sourcedir/exclude_file_patterns";
66	push(@excludes, $exclude_candidate) if -f $exclude_candidate;
67}
68
69# The typedef list that's mechanically extracted by the buildfarm may omit
70# some names we want to treat like typedefs, e.g. "bool" (which is a macro
71# according to <stdbool.h>), and may include some names we don't want
72# treated as typedefs, although various headers that some builds include
73# might make them so.  For the moment we just hardwire a list of names
74# to add and a list of names to exclude; eventually this may need to be
75# easier to configure.  Note that the typedefs need trailing newlines.
76my @additional = map { "$_\n" } qw(
77  bool regex_t regmatch_t regoff
78);
79
80my %excluded = map { +"$_\n" => 1 } qw(
81  FD_SET LookupSet boolean date duration
82  element_type inquiry iterator other
83  pointer reference rep string timestamp type wrap
84);
85
86# globals
87my @files;
88my $filtered_typedefs_fh;
89
90sub check_indent
91{
92	if (system("$indent -? < $devnull > $devnull 2>&1") != 0)
93	{
94		if ($? >> 8 != 1)
95		{
96			print STDERR
97			  "You do not appear to have $indent installed on your system.\n";
98			exit 1;
99		}
100	}
101
102	if (`$indent --version` !~ m/ $INDENT_VERSION /)
103	{
104		print STDERR
105		  "You do not appear to have $indent version $INDENT_VERSION installed on your system.\n";
106		exit 1;
107	}
108
109	if (system("$indent -gnu < $devnull > $devnull 2>&1") == 0)
110	{
111		print STDERR
112		  "You appear to have GNU indent rather than BSD indent.\n";
113		exit 1;
114	}
115
116	return;
117}
118
119sub locate_sourcedir
120{
121	# try fairly hard to locate the sourcedir
122	my $sub = "./src/tools/pgindent";
123	return $sub if -d $sub;
124	# try to find it from an ancestor directory
125	$sub = "../src/tools/pgindent";
126	foreach (1 .. 4)
127	{
128		return $sub if -d $sub;
129		$sub = "../$sub";
130	}
131	return;    # undef if nothing found
132}
133
134sub load_typedefs
135{
136	# try fairly hard to find the typedefs file if it's not set
137
138	foreach my $try ('.', $sourcedir, '/usr/local/etc')
139	{
140		last if $typedefs_file;
141		next unless defined $try;
142		$typedefs_file = "$try/typedefs.list"
143		  if (-f "$try/typedefs.list");
144	}
145
146	die "cannot locate typedefs file \"$typedefs_file\"\n"
147	  unless $typedefs_file && -f $typedefs_file;
148
149	open(my $typedefs_fh, '<', $typedefs_file)
150	  || die "cannot open typedefs file \"$typedefs_file\": $!\n";
151	my @typedefs = <$typedefs_fh>;
152	close($typedefs_fh);
153
154	# add command-line-supplied typedefs?
155	if (defined($typedef_str))
156	{
157		foreach my $typedef (split(m/[, \t\n]+/, $typedef_str))
158		{
159			push(@typedefs, $typedef . "\n");
160		}
161	}
162
163	# add additional entries
164	push(@typedefs, @additional);
165
166	# remove excluded entries
167	@typedefs = grep { !$excluded{$_} } @typedefs;
168
169	# write filtered typedefs
170	my $filter_typedefs_fh = new File::Temp(TEMPLATE => "pgtypedefXXXXX");
171	print $filter_typedefs_fh @typedefs;
172	$filter_typedefs_fh->close();
173
174	# temp file remains because we return a file handle reference
175	return $filter_typedefs_fh;
176}
177
178sub process_exclude
179{
180	foreach my $excl (@excludes)
181	{
182		last unless @files;
183		open(my $eh, '<', $excl)
184		  || die "cannot open exclude file \"$excl\"\n";
185		while (my $line = <$eh>)
186		{
187			chomp $line;
188			next if $line =~ m/^#/ || $line eq "";
189			my $rgx = qr!$line!;
190			@files = grep { $_ !~ /$rgx/ } @files if $rgx;
191		}
192		close($eh);
193	}
194	return;
195}
196
197sub read_source
198{
199	my $source_filename = shift;
200	my $source;
201
202	open(my $src_fd, '<', $source_filename)
203	  || die "cannot open file \"$source_filename\": $!\n";
204	local ($/) = undef;
205	$source = <$src_fd>;
206	close($src_fd);
207
208	return $source;
209}
210
211sub write_source
212{
213	my $source = shift;
214	my $source_filename = shift;
215
216	open(my $src_fh, '>', $source_filename)
217	  || die "cannot open file \"$source_filename\": $!\n";
218	print $src_fh $source;
219	close($src_fh);
220	return;
221}
222
223sub pre_indent
224{
225	my $source = shift;
226
227	## Comments
228
229	# Convert // comments to /* */
230	$source =~ s!^([ \t]*)//(.*)$!$1/* $2 */!gm;
231
232	# Adjust dash-protected block comments so indent won't change them
233	$source =~ s!/\* +---!/*---X_X!g;
234
235	## Other
236
237	# Prevent indenting of code in 'extern "C"' blocks.
238	# we replace the braces with comments which we'll reverse later
239	my $extern_c_start = '/* Open extern "C" */';
240	my $extern_c_stop = '/* Close extern "C" */';
241	$source =~
242	  s!(^#ifdef[ \t]+__cplusplus.*\nextern[ \t]+"C"[ \t]*\n)\{[ \t]*$!$1$extern_c_start!gm;
243	$source =~ s!(^#ifdef[ \t]+__cplusplus.*\n)\}[ \t]*$!$1$extern_c_stop!gm;
244
245	# Protect wrapping in CATALOG()
246	$source =~ s!^(CATALOG\(.*)$!/*$1*/!gm;
247
248	return $source;
249}
250
251sub post_indent
252{
253	my $source = shift;
254
255	# Restore CATALOG lines
256	$source =~ s!^/\*(CATALOG\(.*)\*/$!$1!gm;
257
258	# Put back braces for extern "C"
259	$source =~ s!^/\* Open extern "C" \*/$!{!gm;
260	$source =~ s!^/\* Close extern "C" \*/$!}!gm;
261
262	## Comments
263
264	# Undo change of dash-protected block comments
265	$source =~ s!/\*---X_X!/* ---!g;
266
267	# Fix run-together comments to have a tab between them
268	$source =~ s!\*/(/\*.*\*/)$!*/\t$1!gm;
269
270	## Functions
271
272	# Use a single space before '*' in function return types
273	$source =~ s!^([A-Za-z_]\S*)[ \t]+\*$!$1 *!gm;
274
275	return $source;
276}
277
278sub run_indent
279{
280	my $source = shift;
281	my $error_message = shift;
282
283	my $cmd = "$indent $indent_opts -U" . $filtered_typedefs_fh->filename;
284
285	my $tmp_fh = new File::Temp(TEMPLATE => "pgsrcXXXXX");
286	my $filename = $tmp_fh->filename;
287	print $tmp_fh $source;
288	$tmp_fh->close();
289
290	$$error_message = `$cmd $filename 2>&1`;
291
292	return "" if ($? || length($$error_message) > 0);
293
294	unlink "$filename.BAK";
295
296	open(my $src_out, '<', $filename) || die $!;
297	local ($/) = undef;
298	$source = <$src_out>;
299	close($src_out);
300
301	return $source;
302}
303
304sub diff
305{
306	my $indented = shift;
307	my $source_filename = shift;
308
309	my $post_fh = new File::Temp(TEMPLATE => "pgdiffXXXXX");
310	my $post_fh_filename = $post_fh->filename;
311
312	print $post_fh $indented;
313
314	$post_fh->close();
315
316	my $diff = `diff -upd "$source_filename" "$post_fh_filename"  2>&1`;
317	return $diff;
318}

320sub usage
321{
322	my $message = shift;
323	my $helptext = <<'EOF';
324Usage:
325pgindent [OPTION]... [FILE|DIR]...
326Options:
327	--help                  show this message and quit
328	--commit=gitref         use files modified by the named commit
329	--typedefs=FILE         file containing a list of typedefs
330	--list-of-typedefs=STR  string containing typedefs, space separated
331	--excludes=PATH         file containing list of filename patterns to ignore
332	--indent=PATH           path to pg_bsd_indent program
333	--diff                  show the changes that would be made
334	--check                 exit with status 2 if any changes would be made
335The --excludes and --commit options can be given more than once.
336EOF
337	if ($help)
338	{
339		print $helptext;
340		exit 0;
341	}
342	else
343	{
344		print STDERR "$message\n", $helptext;
345		exit 1;
346	}
347}

349# main
350
351$filtered_typedefs_fh = load_typedefs();
352
353check_indent();
354
355my $wanted = sub {
356	my ($dev, $ino, $mode, $nlink, $uid, $gid);
357	(($dev, $ino, $mode, $nlink, $uid, $gid) = lstat($_))
358	  && -f _
359	  && /^.*\.[ch]\z/s
360	  && push(@files, $File::Find::name);
361};
362
363# any non-option arguments are files or directories to be processed
364File::Find::find({ wanted => $wanted }, @ARGV) if @ARGV;
365
366# commit file locations are relative to the source root
367chdir "$sourcedir/../../.." if @commits && $sourcedir;
368
369# process named commits by comparing each with their immediate ancestor
370foreach my $commit (@commits)
371{
372	my $prev = "$commit~";
373	my @affected = `git diff --diff-filter=ACMR --name-only $prev $commit`;
374	die "git error" if $?;
375	chomp(@affected);
376	push(@files, @affected);
377}
378
379warn "No files to process" unless @files;
380
381# remove excluded files from the file list
382process_exclude();
383
384my %processed;
385my $status = 0;
386
387foreach my $source_filename (@files)
388{
389	# skip duplicates
390	next if $processed{$source_filename};
391	$processed{$source_filename} = 1;
392
393	# ignore anything that's not a .c or .h file
394	next unless $source_filename =~ /\.[ch]$/;
395
396	# don't try to indent a file that doesn't exist
397	unless (-f $source_filename)
398	{
399		warn "Could not find $source_filename";
400		next;
401	}
402	# Automatically ignore .c and .h files that correspond to a .y or .l
403	# file.  indent tends to get badly confused by Bison/flex output,
404	# and there's no value in indenting derived files anyway.
405	my $otherfile = $source_filename;
406	$otherfile =~ s/\.[ch]$/.y/;
407	next if $otherfile ne $source_filename && -f $otherfile;
408	$otherfile =~ s/\.y$/.l/;
409	next if $otherfile ne $source_filename && -f $otherfile;
410
411	my $source = read_source($source_filename);
412	my $orig_source = $source;
413	my $error_message = '';
414
415	$source = pre_indent($source);
416
417	$source = run_indent($source, \$error_message);
418	if ($source eq "")
419	{
420		print STDERR "Failure in $source_filename: " . $error_message . "\n";
421		$status = 3;
422		next;
423	}
424
425	$source = post_indent($source);
426
427	if ($source ne $orig_source)
428	{
429		if (!$diff && !$check)
430		{
431			write_source($source, $source_filename);
432		}
433		else
434		{
435			if ($diff)
436			{
437				print diff($source, $source_filename);
438			}
439
440			if ($check)
441			{
442				$status ||= 2;
443				last unless $diff;
444			}
445		}
446	}
447}
448
449exit $status;
