src/interfaces/ecpg/include/Makefile
1subdir = src/interfaces/ecpg/include
2top_builddir = ../../../..
3include $(top_builddir)/src/Makefile.global
4
5override CPPFLAGS := -I../include -I$(top_srcdir)/src/interfaces/ecpg/include \
6	-I$(libpq_srcdir) -I$(top_builddir)/src/port $(CPPFLAGS)
7
8informix_esql_dir = $(pkgincludedir)/informix/esql
9ecpg_config_h = $(top_builddir)/src/interfaces/ecpg/include/ecpg_config.h
10
11all: $(ecpg_config_h)
12
13install: all installdirs install-headers
14
15.PHONY: install-headers
16ecpg_headers = ecpgerrno.h ecpglib.h ecpgtype.h sqlca.h sql3types.h ecpg_informix.h \
17	pgtypes_error.h pgtypes_numeric.h pgtypes_timestamp.h pgtypes_date.h pgtypes_interval.h pgtypes.h \
18	sqlda.h sqlda-compat.h sqlda-native.h
19informix_headers = datetime.h decimal.h sqltypes.h
20
21install-headers: $(ecpg_headers) $(informix_headers) installdirs
22	$(INSTALL_DATA) $(addprefix $(srcdir)/,$(ecpg_headers)) '$(DESTDIR)$(includedir)/'
23	$(INSTALL_DATA) $(addprefix $(srcdir)/,$(informix_headers)) '$(DESTDIR)$(informix_esql_dir)/'
24	$(INSTALL_DATA) $(ecpg_config_h) '$(DESTDIR)$(includedir)'
25
26installdirs:
27	$(MKDIR_P) '$(DESTDIR)$(includedir)' '$(DESTDIR)$(informix_esql_dir)'
28
29uninstall:
30	rm -f $(addprefix '$(DESTDIR)$(includedir)'/, $(ecpg_headers))
31	rm -f $(addprefix '$(DESTDIR)$(informix_esql_dir)'/, $(informix_headers))
32	rm -f '$(DESTDIR)$(includedir)'/$(notdir $(ecpg_config_h))
33
34distclean:
35	rm -f ecpg_config.h stamp-h
