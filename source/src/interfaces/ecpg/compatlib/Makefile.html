12subdir = src/interfaces/ecpg/compatlib
13top_builddir = ../../../..
14include $(top_builddir)/src/Makefile.global
15
16PGFILEDESC = "ECPG compat - compatibility library for ECPG"
17NAME= ecpg_compat
18SO_MAJOR_VERSION= 3
19SO_MINOR_VERSION= $(MAJORVERSION)
20
21override CPPFLAGS := -I../include -I$(top_srcdir)/src/interfaces/ecpg/include \
22	-I$(libpq_srcdir) $(CPPFLAGS)
23override CFLAGS += $(PTHREAD_CFLAGS)
24
25SHLIB_LINK_INTERNAL = -L../ecpglib -lecpg -L../pgtypeslib -lpgtypes $(libpq_pgport_shlib)
26SHLIB_LINK = $(filter -lintl -lm, $(LIBS)) $(PTHREAD_LIBS)
27SHLIB_PREREQS = submake-ecpglib submake-pgtypeslib
28
29SHLIB_EXPORTS = exports.txt
30
31OBJS = \
32	$(WIN32RES) \
33	informix.o
34
35PKG_CONFIG_REQUIRES_PRIVATE = libecpg, libpgtypes
36
37all: all-lib
38
39.PHONY: submake-ecpglib submake-pgtypeslib
40
41submake-ecpglib:
42	$(MAKE) -C $(top_builddir)/src/interfaces/ecpg/ecpglib all
43
44submake-pgtypeslib:
45	$(MAKE) -C $(top_builddir)/src/interfaces/ecpg/pgtypeslib all
46
47# Shared library stuff
48include $(top_srcdir)/src/Makefile.shlib
49
50install: all installdirs install-lib
51
52installdirs: installdirs-lib
53
54uninstall: uninstall-lib
55
56clean distclean: clean-lib
57	rm -f $(OBJS)
