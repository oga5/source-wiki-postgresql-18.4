9static void
10dump_binary(char *buf, int len, int ind)
11{
12	int i;
13
14	printf("len=%d, ind=%d, data=0x", len, ind);
15	for (i = 0; i < len; ++i)
16		printf("%02x", 0xff & buf[i]);
17	printf("\n");
18}
