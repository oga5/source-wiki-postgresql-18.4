15static void
16fmtlong(long lng, const char *fmt)
17{
18	static int i;
19	int r;
20	char buf[30];
21
22	r = rfmtlong(lng, fmt, buf);
23	printf("r: %d ", r);
24	if (r == 0)
25	{
26		printf("%d: %s (fmt was: %s)\n", i++, buf, fmt);
27	}
28	else
29		check_return(r);
30}

51static void
52check_return(int ret)
53{
54	switch(ret)
55	{
56		case ECPG_INFORMIX_ENOTDMY:
57			printf("(ECPG_INFORMIX_ENOTDMY)");
58			break;
59		case ECPG_INFORMIX_ENOSHORTDATE:
60			printf("(ECPG_INFORMIX_ENOSHORTDATE)");
61			break;
62		case ECPG_INFORMIX_BAD_DAY:
63			printf("(ECPG_INFORMIX_BAD_DAY)");
64			break;
65		case ECPG_INFORMIX_BAD_MONTH:
66			printf("(ECPG_INFORMIX_BAD_MONTH)");
67			break;
68		default:
69			printf("(unknown ret: %d)", ret);
70			break;
71	}
72	printf("\n");
73}
