26int
27main(void)
28{
29	char *text="error\n";
30	char *endptr;
31	numeric *num, *nin;
32	decimal *dec;
33	long l;
34	int i, j, k, q, r, count = 0;
35	double d;
36	numeric **numarr = (numeric **) calloc(1, sizeof(numeric));
37
38	ECPGdebug(1, stderr);
39
40	for (i = 0; nums[i]; i++)
41	{
42		num = PGTYPESnumeric_from_asc(nums[i], &endptr);
43		if (!num) check_errno();
44		if (endptr != NULL)
45		{
46			printf("endptr of %d is not NULL\n", i);
47			if (*endptr != '\0')
48				printf("*endptr of %d is not \\0\n", i);
49		}
50		if (!num) continue;
51
52		numarr = realloc(numarr, sizeof(numeric *) * (count + 1));
53		numarr[count++] = num;
54
55		text = PGTYPESnumeric_to_asc(num, -1);
56		if (!text) check_errno();
57		printf("num[%d,1]: %s\n", i, text); PGTYPESchar_free(text);
58		text = PGTYPESnumeric_to_asc(num, 0);
59		if (!text) check_errno();
60		printf("num[%d,2]: %s\n", i, text); PGTYPESchar_free(text);
61		text = PGTYPESnumeric_to_asc(num, 1);
62		if (!text) check_errno();
63		printf("num[%d,3]: %s\n", i, text); PGTYPESchar_free(text);
64		text = PGTYPESnumeric_to_asc(num, 2);
65		if (!text) check_errno();
66		printf("num[%d,4]: %s\n", i, text); PGTYPESchar_free(text);
67
68		nin = PGTYPESnumeric_new();
69		text = PGTYPESnumeric_to_asc(nin, 2);
70		if (!text) check_errno();
71		printf("num[%d,5]: %s\n", i, text); PGTYPESchar_free(text);
72
73		r = PGTYPESnumeric_to_long(num, &l);
74		if (r) check_errno();
75		printf("num[%d,6]: %ld (r: %d)\n", i, r?0L:l, r);
76		if (r == 0)
77		{
78			r = PGTYPESnumeric_from_long(l, nin);
79			if (r) check_errno();
80			text = PGTYPESnumeric_to_asc(nin, 2);
81			q = PGTYPESnumeric_cmp(num, nin);
82			printf("num[%d,7]: %s (r: %d - cmp: %d)\n", i, text, r, q);
83			PGTYPESchar_free(text);
84		}
85
86		r = PGTYPESnumeric_to_int(num, &k);
87		if (r) check_errno();
88		printf("num[%d,8]: %d (r: %d)\n", i, r?0:k, r);
89		if (r == 0)
90		{
91			r = PGTYPESnumeric_from_int(k, nin);
92			if (r) check_errno();
93			text = PGTYPESnumeric_to_asc(nin, 2);
94			q = PGTYPESnumeric_cmp(num, nin);
95			printf("num[%d,9]: %s (r: %d - cmp: %d)\n", i, text, r, q);
96			PGTYPESchar_free(text);
97		}
98
99		if (i != 6)
100		{
101			/* underflow does not work reliable on several archs, so not testing it here */
102			/* this is a libc problem since we only call strtod() */
103
104			r = PGTYPESnumeric_to_double(num, &d);
105			if (r) check_errno();
106			printf("num[%d,10]: ", i);
107			print_double(r ? 0.0 : d);
108			printf(" (r: %d)\n", r);
109		}
110
111		/* do not test double to numeric because
112		 * - extra digits are different on different architectures
113		 * - PGTYPESnumeric_from_double internally calls PGTYPESnumeric_from_asc anyway
114		 */
115
116		dec = PGTYPESdecimal_new();
117		r = PGTYPESnumeric_to_decimal(num, dec);
118		if (r) check_errno();
119		/* we have no special routine for outputting decimal, it would
120		 * convert to a numeric anyway */
121		printf("num[%d,11]: - (r: %d)\n", i, r);
122		if (r == 0)
123		{
124			r = PGTYPESnumeric_from_decimal(dec, nin);
125			if (r) check_errno();
126			text = PGTYPESnumeric_to_asc(nin, 2);
127			q = PGTYPESnumeric_cmp(num, nin);
128			printf("num[%d,12]: %s (r: %d - cmp: %d)\n", i, text, r, q);
129			PGTYPESchar_free(text);
130		}
131
132		PGTYPESdecimal_free(dec);
133		PGTYPESnumeric_free(nin);
134		printf("\n");
135	}
136
137	for (i = 0; i < count; i++)
138	{
139		for (j = 0; j < count; j++)
140		{
141			numeric* a = PGTYPESnumeric_new();
142			numeric* s = PGTYPESnumeric_new();
143			numeric* m = PGTYPESnumeric_new();
144			numeric* d = PGTYPESnumeric_new();
145			r = PGTYPESnumeric_add(numarr[i], numarr[j], a);
146			if (r)
147			{
148				check_errno();
149				printf("r: %d\n", r);
150			}
151			else
152			{
153				text = PGTYPESnumeric_to_asc(a, 10);
154				printf("num[a,%d,%d]: %s\n", i, j, text);
155				PGTYPESchar_free(text);
156			}
157			r = PGTYPESnumeric_sub(numarr[i], numarr[j], s);
158			if (r)
159			{
160				check_errno();
161				printf("r: %d\n", r);
162			}
163			else
164			{
165				text = PGTYPESnumeric_to_asc(s, 10);
166				printf("num[s,%d,%d]: %s\n", i, j, text);
167				PGTYPESchar_free(text);
168			}
169			r = PGTYPESnumeric_mul(numarr[i], numarr[j], m);
170			if (r)
171			{
172				check_errno();
173				printf("r: %d\n", r);
174			}
175			else
176			{
177				text = PGTYPESnumeric_to_asc(m, 10);
178				printf("num[m,%d,%d]: %s\n", i, j, text);
179				PGTYPESchar_free(text);
180			}
181			r = PGTYPESnumeric_div(numarr[i], numarr[j], d);
182			if (r)
183			{
184				check_errno();
185				printf("r: %d\n", r);
186			}
187			else
188			{
189				text = PGTYPESnumeric_to_asc(d, 10);
190				printf("num[d,%d,%d]: %s\n", i, j, text);
191				PGTYPESchar_free(text);
192			}
193
194			PGTYPESnumeric_free(a);
195			PGTYPESnumeric_free(s);
196			PGTYPESnumeric_free(m);
197			PGTYPESnumeric_free(d);
198		}
199	}
200
201	for (i = 0; i < count; i++)
202	{
203		text = PGTYPESnumeric_to_asc(numarr[i], -1);
204		printf("%d: %s\n", i, text);
205		PGTYPESchar_free(text);
206		PGTYPESnumeric_free(numarr[i]);
207	}
208	free(numarr);
209
210	return 0;
211}

213static void
214check_errno(void)
215{
216	switch(errno)
217	{
218		case 0:
219			printf("(no errno set) - ");
220			break;
221		case PGTYPES_NUM_OVERFLOW:
222			printf("(errno == PGTYPES_NUM_OVERFLOW) - ");
223			break;
224		case PGTYPES_NUM_UNDERFLOW:
225			printf("(errno == PGTYPES_NUM_UNDERFLOW) - ");
226			break;
227		case PGTYPES_NUM_BAD_NUMERIC:
228			printf("(errno == PGTYPES_NUM_BAD_NUMERIC) - ");
229			break;
230		case PGTYPES_NUM_DIVIDE_ZERO:
231			printf("(errno == PGTYPES_NUM_DIVIDE_ZERO) - ");
232			break;
233		default:
234			printf("(unknown errno (%d))\n", errno);
235			printf("(libc: (%s)) ", strerror(errno));
236			break;
237	}
