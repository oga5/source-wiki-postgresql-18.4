399static void
400check_errno(void)
401{
402	switch(errno)
403	{
404		case 0:
405			printf("(no errno set) - ");
406			break;
407		case PGTYPES_TS_BAD_TIMESTAMP:
408			printf("(errno == PGTYPES_TS_BAD_TIMESTAMP) - ");
409			break;
410		default:
411			printf("(unknown errno (%d))\n", errno);
412			printf("(libc: (%s)) ", strerror(errno));
413			break;
414	}
415}
