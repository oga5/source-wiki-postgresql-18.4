11static void warn(void)
12{
13  fprintf(stderr, "Warning: At least one column was truncated\n");
14}
15
16/* Compatible handling of char array to retrieve varchar field to char array
17   should be fixed-length, blank-padded, then null-terminated.
18   Conforms to the ANSI Fixed Character type. */
19
20int main() {
21
22  EXEC SQL WHENEVER SQLWARNING do warn();
23  EXEC SQL WHENEVER SQLERROR STOP;
24
25  const char *ppppp = "XXXXX";
26  int loopcount;
27  sqlda_t *sqlda = NULL;
28
29  EXEC SQL BEGIN DECLARE SECTION;
30  char shortstr[5];
31  char bigstr[11];
32  short shstr_ind = 0;
33  short bigstr_ind = 0;
34  EXEC SQL END DECLARE SECTION;
35
36  ECPGdebug(1, stderr);
37  EXEC SQL CONNECT TO REGRESSDB1;
38
39  EXEC SQL CREATE TABLE strdbase (strval varchar(10));
40  EXEC SQL INSERT INTO strdbase values ('');
41  EXEC SQL INSERT INTO strdbase values ('AB');
42  EXEC SQL INSERT INTO strdbase values ('ABCD');
43  EXEC SQL INSERT INTO strdbase values ('ABCDE');
44  EXEC SQL INSERT INTO strdbase values ('ABCDEF');
45  EXEC SQL INSERT INTO strdbase values ('ABCDEFGHIJ');
46
47  EXEC SQL declare C cursor for select strval, strval from strdbase;
48  EXEC SQL OPEN C;
49
50  EXEC SQL WHENEVER NOT FOUND DO BREAK;
51
52  printf("Full Str.  :  Short  Ind.\n");
53  for (loopcount = 0; loopcount < 100; loopcount++) {
54    strncpy(shortstr, ppppp, sizeof shortstr);
55    memset(bigstr, 0, sizeof bigstr);
56    EXEC SQL FETCH C into :bigstr :bigstr_ind, :shortstr :shstr_ind;
57    printf("\"%s\": \"%s\"  %d\n", bigstr, shortstr, shstr_ind);
58  }
59
60  EXEC SQL CLOSE C;
61  EXEC SQL DROP TABLE strdbase;
62  EXEC SQL COMMIT WORK;
63
64  /* SQLDA handling */
65  EXEC SQL WHENEVER SQLWARNING SQLPRINT;
66  EXEC SQL WHENEVER NOT FOUND STOP;
67  EXEC SQL PREPARE stmt1 FROM "SELECT 123::numeric(3,0), 't'::varchar(2)";
68  EXEC SQL DECLARE cur1 CURSOR FOR stmt1;
69  EXEC SQL OPEN cur1;
70  EXEC SQL FETCH NEXT FROM cur1 INTO DESCRIPTOR sqlda;
71
72  printf("\n-----------------\ntype    : data\n");
73  for (int i = 0 ; i < sqlda->sqld ; i++)
74  {
75	  sqlvar_t v = sqlda->sqlvar[i];
76	  char *sqldata = v.sqldata;
77
78	  if (v.sqltype == ECPGt_numeric)
79		  sqldata =
80			  PGTYPESnumeric_to_asc((numeric*) sqlda->sqlvar[i].sqldata, -1);
81
82	  printf("%-8s: \"%s\"\n", v.sqlname.data, sqldata);
83  }
84
85  EXEC SQL CLOSE cur1;
86  EXEC SQL COMMIT WORK;
87
88  printf("\nGOOD-BYE!!\n\n");
89
90  EXEC SQL DISCONNECT ALL;
91
92  return 0;
93}
