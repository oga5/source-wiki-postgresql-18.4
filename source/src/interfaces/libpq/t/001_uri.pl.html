254# test to run for each of the above test definitions
255sub test_uri
256{
257	local $Test::Builder::Level = $Test::Builder::Level + 1;
258	local %ENV = %ENV;
259
260	my $uri;
261	my %expect;
262	my %envvars;
263	my %result;
264
265	($uri, $expect{stdout}, $expect{stderr}, %envvars) = @$_;
266
267	$expect{'exit'} = $expect{stderr} eq '';
268	%ENV = (%ENV, %envvars);
269
270	my $cmd = [ 'libpq_uri_regress', $uri ];
271	$result{exit} = IPC::Run::run $cmd,
272	  '>' => \$result{stdout},
273	  '2>' => \$result{stderr};
274
275	chomp($result{stdout});
276	chomp($result{stderr});
277
278	# use is_deeply so there's one test result for each test above, without
279	# losing the information whether stdout/stderr mismatched.
280	is_deeply(\%result, \%expect, $uri);
281}
282
283foreach (@tests)
284{
285	test_uri($_);
286}
287
288done_testing();
