132# Check for functions that libpq must not call, currently just exit().
133# (Ideally we'd reject abort() too, but there are various scenarios where
134# build toolchains insert abort() calls, e.g. to implement assert().)
135# If nm doesn't exist or doesn't work on shlibs, this test will do nothing,
136# which is fine.  The exclusion of __cxa_atexit is necessary on OpenBSD,
137# which seems to insert references to that even in pure C code. Excluding
138# __tsan_func_exit is necessary when using ThreadSanitizer data race detector
139# which use this function for instrumentation of function exit.
140# Skip the test when profiling, as gcc may insert exit() calls for that.
141# Also skip the test on platforms where libpq infrastructure may be provided
142# by statically-linked libraries, as we can't expect them to honor this
143# coding rule.
144libpq-refs-stamp: $(shlib)
145ifneq ($(enable_coverage), yes)
146ifeq (,$(filter solaris,$(PORTNAME)))
147	@if nm -A -u $< 2>/dev/null | grep -v -e __cxa_atexit -e __tsan_func_exit | grep exit; then \
148		echo 'libpq must not be calling any function which invokes exit'; exit 1; \
149	fi
150endif
151endif
152	touch $@
