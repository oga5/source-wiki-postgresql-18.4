20libpq_so_sources = [] # for shared lib, in addition to the above
21
22if host_system == 'windows'
23  libpq_sources += files('pthread-win32.c', 'win32.c')
24  libpq_so_sources += rc_lib_gen.process(win32ver_rc, extra_args: [
25    '--NAME', 'libpq',
26    '--FILEDESC', 'PostgreSQL Access Library',])
27endif
28
29if ssl.found()
30  libpq_sources += files('fe-secure-common.c')
31  libpq_sources += files('fe-secure-openssl.c')
32endif
33
34if gssapi.found()
35  libpq_sources += files(
36    'fe-gssapi-common.c',
37    'fe-secure-gssapi.c',
38  )
39endif
40
41export_file = custom_target('libpq.exports',
42  kwargs: gen_export_kwargs,
43)
44
45# port needs to be in include path due to pthread-win32.h
46libpq_inc = include_directories('.', '../../port')
47libpq_c_args = ['-DSO_MAJOR_VERSION=5']
48
49# The OAuth implementation differs depending on the type of library being built.
50libpq_so_c_args = ['-DUSE_DYNAMIC_OAUTH']
51
52# Not using both_libraries() here as
53# 1) resource files should only be in the shared library
54# 2) we want the .pc file to include a dependency to {pgport,common}_static for
55#    libpq_st, and {pgport,common}_shlib for libpq_sh
56#
57# We could try to avoid building the source files twice, but it probably adds
58# more complexity than its worth (reusing object files requires also linking
59# to the library on windows or breaks precompiled headers).
60libpq_st = static_library('libpq',
61  libpq_sources,
62  include_directories: [libpq_inc],
63  c_args: libpq_c_args,
64  c_pch: pch_postgres_fe_h,
65  dependencies: [frontend_stlib_code, libpq_deps],
66  kwargs: default_lib_args,
67)
68
69libpq_so = shared_library('libpq',
70  libpq_sources + libpq_so_sources,
71  include_directories: [libpq_inc, postgres_inc],
72  c_args: libpq_c_args + libpq_so_c_args,
73  c_pch: pch_postgres_fe_h,
74  version: '5.' + pg_version_major.to_string(),
75  soversion: host_system != 'windows' ? '5' : '',
76  darwin_versions: ['5', '5.' + pg_version_major.to_string()],
77  dependencies: [frontend_shlib_code, libpq_deps],
78  link_depends: export_file,
79  link_args: export_fmt.format(export_file.full_path()),
80  kwargs: default_lib_args,
81)
82
83libpq = declare_dependency(
84  link_with: [libpq_so],
85  include_directories: [include_directories('.')]
86)
87
88private_deps = [
89  frontend_stlib_code,
90  libpq_deps,
91]
92
93if oauth_flow_supported
94  # libpq.so doesn't link against libcurl, but libpq.a needs libpq-oauth, and
95  # libpq-oauth needs libcurl. Put both into *.private.
96  private_deps += [
97    libpq_oauth_deps,
98    '-lpq-oauth',
99  ]
100endif
101
102pkgconfig.generate(
103  name: 'libpq',
104  description: 'PostgreSQL libpq library',
105  url: pg_url,
106  libraries: libpq,
107  libraries_private: private_deps,
108)
