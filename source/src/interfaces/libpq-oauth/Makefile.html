12subdir = src/interfaces/libpq-oauth
13top_builddir = ../../..
14include $(top_builddir)/src/Makefile.global
15
16PGFILEDESC = "libpq-oauth - device authorization OAuth support"
17
18# This is an internal module; we don't want an SONAME and therefore do not set
19# SO_MAJOR_VERSION.
20NAME = pq-oauth-$(MAJORVERSION)
21
22# Force the name "libpq-oauth" for both the static and shared libraries. The
23# staticlib doesn't need version information in its name.
24override shlib := lib$(NAME)$(DLSUFFIX)
25override stlib := libpq-oauth.a
26
27override CPPFLAGS := -I$(libpq_srcdir) -I$(top_builddir)/src/port $(CPPFLAGS) $(LIBCURL_CPPFLAGS)
28override CFLAGS += $(PTHREAD_CFLAGS)
29
30OBJS = \
31	$(WIN32RES)
32
33OBJS_STATIC = oauth-curl.o
34
35# The shared library needs additional glue symbols.
36OBJS_SHLIB = \
37	oauth-curl_shlib.o \
38	oauth-utils.o \
39
40oauth-utils.o: override CPPFLAGS += -DUSE_DYNAMIC_OAUTH
41oauth-curl_shlib.o: override CPPFLAGS_SHLIB += -DUSE_DYNAMIC_OAUTH
42
43# Add shlib-/stlib-specific objects.
44$(shlib): override OBJS += $(OBJS_SHLIB)
45$(shlib): $(OBJS_SHLIB)
46
47$(stlib): override OBJS += $(OBJS_STATIC)
48$(stlib): $(OBJS_STATIC)
49
50SHLIB_LINK_INTERNAL = $(libpq_pgport_shlib)
51SHLIB_LINK = $(LIBCURL_LDFLAGS) $(LIBCURL_LDLIBS) $(filter -lintl -lm $(PTHREAD_LIBS), $(LIBS))
52SHLIB_PREREQS = submake-libpq
53SHLIB_EXPORTS = exports.txt
54
55# Disable -bundle_loader on macOS.
56BE_DLLLIBS =
57
58# By default, a library without an SONAME doesn't get a static library, so we
59# add it to the build explicitly.
60all: all-lib all-static-lib
61
62# Shared library stuff
63include $(top_srcdir)/src/Makefile.shlib
64
65# Use src/common/Makefile's trick for tracking dependencies of shlib-specific
66# objects.
67%_shlib.o: %.c %.o
68	$(CC) $(CFLAGS) $(CFLAGS_SL) $(CPPFLAGS) $(CPPFLAGS_SHLIB) -c $< -o $@
69
70# Ignore the standard rules for SONAME-less installation; we want both the
71# static and shared libraries to go into libdir.
72install: all installdirs $(stlib) $(shlib)
73	$(INSTALL_SHLIB) $(shlib) '$(DESTDIR)$(libdir)/$(shlib)'
74	$(INSTALL_STLIB) $(stlib) '$(DESTDIR)$(libdir)/$(stlib)'
75
76installdirs:
77	$(MKDIR_P) '$(DESTDIR)$(libdir)'
78
79uninstall:
80	rm -f '$(DESTDIR)$(libdir)/$(stlib)'
81	rm -f '$(DESTDIR)$(libdir)/$(shlib)'
82
83clean distclean: clean-lib
84	rm -f $(OBJS) $(OBJS_STATIC) $(OBJS_SHLIB)
