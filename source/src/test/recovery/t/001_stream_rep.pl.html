120# Attempt to connect to $node1, then $node2, using target_session_attrs=$mode.
121# Expect to connect to $target_node (undef for failure) with given $status.
122sub test_target_session_attrs
123{
124	local $Test::Builder::Level = $Test::Builder::Level + 1;
125
126	my $node1 = shift;
127	my $node2 = shift;
128	my $target_node = shift;
129	my $mode = shift;
130	my $status = shift;
131
132	my $node1_host = $node1->host;
133	my $node1_port = $node1->port;
134	my $node1_name = $node1->name;
135	my $node2_host = $node2->host;
136	my $node2_port = $node2->port;
137	my $node2_name = $node2->name;
138	my $target_port = undef;
139	$target_port = $target_node->port if (defined $target_node);
140	my $target_name = undef;
141	$target_name = $target_node->name if (defined $target_node);
142
143	# Build connection string for connection attempt.
144	my $connstr = "host=$node1_host,$node2_host ";
145	$connstr .= "port=$node1_port,$node2_port ";
146	$connstr .= "target_session_attrs=$mode";
147
148	# Attempt to connect, and if successful, get the server port number
149	# we connected to.  Note we must pass the SQL command via the command
150	# line not stdin, else Perl may spit up trying to write to stdin of
151	# an already-failed psql process.
152	my ($ret, $stdout, $stderr) = $node1->psql(
153		'postgres',
154		undef,
155		extra_params => [
156			'--dbname' => $connstr,
157			'--command' => 'SHOW port;',
158		]);
159	if ($status == 0)
160	{
161		is( $status == $ret && $stdout eq $target_port,
162			1,
163			"connect to node $target_name with mode \"$mode\" and $node1_name,$node2_name listed"
164		);
165	}
166	else
167	{
168		print "status = $status\n";
169		print "ret = $ret\n";
170		print "stdout = $stdout\n";
171		print "stderr = $stderr\n";
172
173		is( $status == $ret && !defined $target_node,
174			1,
175			"fail to connect with mode \"$mode\" and $node1_name,$node2_name listed"
176		);
177	}
178
179	return;
180}

383# Fetch xmin columns from slot's pg_replication_slots row, after waiting for
384# given boolean condition to be true to ensure we've reached a quiescent state
385sub get_slot_xmins
386{
387	my ($node, $slotname, $check_expr) = @_;
388
389	$node->poll_query_until(
390		'postgres', qq[
391		SELECT $check_expr
392		FROM pg_catalog.pg_replication_slots
393		WHERE slot_name = '$slotname';
394	]) or die "Timed out waiting for slot xmins to advance";
395
396	my $slotinfo = $node->slot($slotname);
397	return ($slotinfo->{'xmin'}, $slotinfo->{'catalog_xmin'});
398}

403	"xmin IS NULL AND catalog_xmin IS NULL");

405is($catalog_xmin, '',

417sub replay_check
418{
419	my $newval = $node_primary->safe_psql('postgres',
420		'INSERT INTO replayed(val) SELECT coalesce(max(val),0) + 1 AS newval FROM replayed RETURNING val'
421	);

423	$node_standby_1->wait_for_replay_catchup($node_standby_2, $node_primary);
424
425	$node_standby_1->safe_psql('postgres',
426		qq[SELECT 1 FROM replayed WHERE val = $newval])
427	  or die "standby_1 didn't replay primary value $newval";
428	$node_standby_2->safe_psql('postgres',
429		qq[SELECT 1 FROM replayed WHERE val = $newval])
430	  or die "standby_2 didn't replay standby_1 value $newval";
431	return;
432}

551# Test physical slot advancing and its durability.  Create a new slot on
552# the primary, not used by any of the standbys. This reserves WAL at creation.
553my $phys_slot = 'phys_slot';
554$node_primary->safe_psql('postgres',
555	"SELECT pg_create_physical_replication_slot('$phys_slot', true);");
556# Generate some WAL, and switch to a new segment, used to check that
557# the previous segment is correctly getting recycled as the slot advancing
558# would recompute the minimum LSN calculated across all slots.
559my $segment_removed = $node_primary->safe_psql('postgres',
560	'SELECT pg_walfile_name(pg_current_wal_lsn())');
561chomp($segment_removed);
562$node_primary->advance_wal(1);
563my $current_lsn =
564  $node_primary->safe_psql('postgres', "SELECT pg_current_wal_lsn();");
565chomp($current_lsn);
566my $psql_rc = $node_primary->psql('postgres',
567	"SELECT pg_replication_slot_advance('$phys_slot', '$current_lsn'::pg_lsn);"
568);
569is($psql_rc, '0', 'slot advancing with physical slot');
570my $phys_restart_lsn_pre = $node_primary->safe_psql('postgres',
571	"SELECT restart_lsn from pg_replication_slots WHERE slot_name = '$phys_slot';"
572);
573chomp($phys_restart_lsn_pre);
574# Slot advance should persist across clean restarts.
575$node_primary->restart;
576my $phys_restart_lsn_post = $node_primary->safe_psql('postgres',
577	"SELECT restart_lsn from pg_replication_slots WHERE slot_name = '$phys_slot';"
578);
579chomp($phys_restart_lsn_post);
580ok( ($phys_restart_lsn_pre cmp $phys_restart_lsn_post) == 0,
581	"physical slot advance persists across restarts");
582
583# Check if the previous segment gets correctly recycled after the
584# server stopped cleanly, causing a shutdown checkpoint to be generated.
585my $primary_data = $node_primary->data_dir;
586ok(!-f "$primary_data/pg_wal/$segment_removed",
587	"WAL segment $segment_removed recycled after physical slot advancing");

605note "testing BASE_BACKUP cancellation";
606
607my $sigchld_bb_timeout =
608  IPC::Run::timer($PostgreSQL::Test::Utils::timeout_default);
609
610# This test requires a replication connection with a database, as it mixes
611# a replication command and a SQL command.  The first BASE_BACKUP is throttled
612# to give enough room for the cancellation running below.  The second command
613# for pg_backup_stop() should fail.
614my ($sigchld_bb_stdin, $sigchld_bb_stdout, $sigchld_bb_stderr) = ('', '', '');
615my $sigchld_bb = IPC::Run::start(
616	[
617		'psql', '--no-psqlrc',
618		'--command' => "BASE_BACKUP (CHECKPOINT 'fast', MAX_RATE 32);",
619		'--command' => 'SELECT pg_backup_stop()',
620		'--dbname' => $connstr
621	],
622	'<' => \$sigchld_bb_stdin,
623	'>' => \$sigchld_bb_stdout,
624	'2>' => \$sigchld_bb_stderr,
625	$sigchld_bb_timeout);
626
627# The cancellation is issued once the database files are streamed and
628# the checkpoint issued at backup start completes.
629is( $node_primary->poll_query_until(
630		'postgres',
631		"SELECT pg_cancel_backend(a.pid) FROM "
632		  . "pg_stat_activity a, pg_stat_progress_basebackup b WHERE "
633		  . "a.pid = b.pid AND a.query ~ 'BASE_BACKUP' AND "
634		  . "b.phase = 'streaming database files';"),
635	"1",
636	"WAL sender sending base backup killed");
637
638# The psql command should fail on pg_backup_stop().
639ok( pump_until(
640		$sigchld_bb, $sigchld_bb_timeout,
641		\$sigchld_bb_stderr, qr/backup is not in progress/),
642	'base backup cleanly canceled');
643$sigchld_bb->finish();
