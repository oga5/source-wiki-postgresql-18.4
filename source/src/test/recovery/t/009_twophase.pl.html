15sub configure_and_reload
16{
17	local $Test::Builder::Level = $Test::Builder::Level + 1;
18
19	my ($node, $parameter) = @_;
20	my $name = $node->name;
21
22	$node->append_conf(
23		'postgresql.conf', qq(
24		$parameter
25	));
26	$node->psql('postgres', "SELECT pg_reload_conf()", stdout => \$psql_out);
27	is($psql_out, 't', "reload node $name with $parameter");
28	return;
29}

31# Set up two nodes, which will alternately be primary and replication standby.
32
33# Setup london node
34my $node_london = PostgreSQL::Test::Cluster->new("london");
35$node_london->init(allows_streaming => 1);
36$node_london->append_conf(
37	'postgresql.conf', qq(
38	max_prepared_transactions = 10
39	log_checkpoints = true
40));
41$node_london->start;
42$node_london->backup('london_backup');
43
44# Setup paris node
45my $node_paris = PostgreSQL::Test::Cluster->new('paris');
46$node_paris->init_from_backup($node_london, 'london_backup',
47	has_streaming => 1);
48$node_paris->append_conf(
49	'postgresql.conf', qq(
50	subtransaction_buffers = 32
51));

54# Switch to synchronous replication in both directions

56configure_and_reload($node_paris, "synchronous_standby_names = 'london'");

58# Set up nonce names for current primary and standby nodes
59note "Initially, london is primary and paris is standby";

66###############################################################################
67# Check that we can commit and abort transaction after soft restart.
68# Here checkpoint happens before shutdown and no WAL replay will occur at next
69# startup. In this case postgres re-creates shared-memory state from twophase
70# files.
71###############################################################################
72
73$cur_primary->psql(
74	'postgres', "
75	BEGIN;
76	INSERT INTO t_009_tbl VALUES (1, 'issued to ${cur_primary_name}');
77	SAVEPOINT s1;
78	INSERT INTO t_009_tbl VALUES (2, 'issued to ${cur_primary_name}');
79	PREPARE TRANSACTION 'xact_009_1';
80	BEGIN;
81	INSERT INTO t_009_tbl VALUES (3, 'issued to ${cur_primary_name}');
82	SAVEPOINT s1;
83	INSERT INTO t_009_tbl VALUES (4, 'issued to ${cur_primary_name}');
84	PREPARE TRANSACTION 'xact_009_2';");
85$cur_primary->stop;
86$cur_primary->start;
87
88$psql_rc = $cur_primary->psql('postgres', "COMMIT PREPARED 'xact_009_1'");
89is($psql_rc, '0', 'Commit prepared transaction after restart');
90
91$psql_rc = $cur_primary->psql('postgres', "ROLLBACK PREPARED 'xact_009_2'");
92is($psql_rc, '0', 'Rollback prepared transaction after restart');
93
94###############################################################################
95# Check that we can commit and abort after a hard restart.
96# At next startup, WAL replay will re-create shared memory state for prepared
97# transaction using dedicated WAL records.

103	BEGIN;
104	INSERT INTO t_009_tbl VALUES (5, 'issued to ${cur_primary_name}');
105	SAVEPOINT s1;
106	INSERT INTO t_009_tbl VALUES (6, 'issued to ${cur_primary_name}');
107	PREPARE TRANSACTION 'xact_009_3';
108	BEGIN;
109	INSERT INTO t_009_tbl VALUES (7, 'issued to ${cur_primary_name}');
110	SAVEPOINT s1;
111	INSERT INTO t_009_tbl VALUES (8, 'issued to ${cur_primary_name}');
112	PREPARE TRANSACTION 'xact_009_4';");
113$cur_primary->teardown_node;
114$cur_primary->start;
115
116$psql_rc = $cur_primary->psql('postgres', "COMMIT PREPARED 'xact_009_3'");

315###############################################################################
316# Check visibility of prepared transactions in standby after a restart while
317# primary is down.
318###############################################################################
319
320$cur_primary->psql(
321	'postgres', "
322	SET synchronous_commit='remote_apply'; -- To ensure the standby is caught up
323	CREATE TABLE t_009_tbl_standby_mvcc (id int, msg text);
324	BEGIN;
325	INSERT INTO t_009_tbl_standby_mvcc VALUES (1, 'issued to ${cur_primary_name}');
326	SAVEPOINT s1;
327	INSERT INTO t_009_tbl_standby_mvcc VALUES (2, 'issued to ${cur_primary_name}');
328	PREPARE TRANSACTION 'xact_009_standby_mvcc';
329	");
330$cur_primary->stop;
331$cur_standby->restart;
332
333# Acquire a snapshot in standby, before we commit the prepared transaction
334my $standby_session =
335  $cur_standby->background_psql('postgres', on_error_die => 1);
336$standby_session->query_safe("BEGIN ISOLATION LEVEL REPEATABLE READ");
337$psql_out =
338  $standby_session->query_safe("SELECT count(*) FROM t_009_tbl_standby_mvcc");
339is($psql_out, '0',
340	"Prepared transaction not visible in standby before commit");
341
342# Commit the transaction in primary
343$cur_primary->start;
344$cur_primary->psql(
345	'postgres', "
346SET synchronous_commit='remote_apply'; -- To ensure the standby is caught up
347COMMIT PREPARED 'xact_009_standby_mvcc';
348");
349
350# Still not visible to the old snapshot
351$psql_out =
352  $standby_session->query_safe("SELECT count(*) FROM t_009_tbl_standby_mvcc");
353is($psql_out, '0',
354	"Committed prepared transaction not visible to old snapshot in standby");
355
356# Is visible to a new snapshot
357$standby_session->query_safe("COMMIT");
358$psql_out =
359  $standby_session->query_safe("SELECT count(*) FROM t_009_tbl_standby_mvcc");
360is($psql_out, '2',
361	"Committed prepared transaction is visible to new snapshot in standby");
362$standby_session->quit;

539# ensuring that enough pg_subtrans pages exist on disk to cover the range of

552my $osubtrans = $cur_primary->safe_psql('postgres',
553	"select 'pg_subtrans/'||f, s.size from pg_ls_dir('pg_subtrans') f, pg_stat_file('pg_subtrans/'||f) s"

555$cur_primary->pgbench(
556	'--no-vacuum --client=5 --transactions=1000',

562		'009_twophase.pgb' => 'insert into test default values'
563	});
564# StartupSUBTRANS is exercised with a wide range of visible XIDs in this
565# stop/start sequence, because we left a prepared transaction open above.
566# Also, setting subtransaction_buffers to 32 above causes to switch SLRU
567# bank, for additional code coverage.
568$cur_primary->stop;
569$cur_primary->start;
570my $nsubtrans = $cur_primary->safe_psql('postgres',
571	"select 'pg_subtrans/'||f, s.size from pg_ls_dir('pg_subtrans') f, pg_stat_file('pg_subtrans/'||f) s"
