10##################################################
11# Test that when a subscription with failover enabled is created, it will alter
12# the failover property of the corresponding slot on the publisher.
13##################################################
14
15# Create publisher
16my $publisher = PostgreSQL::Test::Cluster->new('publisher');
17# Make sure pg_hba.conf is set up to allow connections from repl_role.
18# This is only needed on Windows machines that don't use UNIX sockets.
19$publisher->init(
20	allows_streaming => 'logical',
21	auth_extra => [ '--create-role' => 'repl_role' ]);
22# Disable autovacuum to avoid generating xid during stats update as otherwise
23# the new XID could then be replicated to standby at some random point making
24# slots at primary lag behind standby during slot sync.
25$publisher->append_conf(
26	'postgresql.conf', qq{
27autovacuum = off
28max_prepared_transactions = 1
29});
30$publisher->start;
31
32$publisher->safe_psql('postgres',
33	"CREATE PUBLICATION regress_mypub FOR ALL TABLES;");
34
35my $publisher_connstr = $publisher->connstr . ' dbname=postgres';
36
37# Create a subscriber node, wait for sync to complete
38my $subscriber1 = PostgreSQL::Test::Cluster->new('subscriber1');
39$subscriber1->init;
40$subscriber1->append_conf('postgresql.conf', 'max_prepared_transactions = 1');
41$subscriber1->start;
42
43# Capture the time before the logical failover slot is created on the
44# primary. We later call this publisher as primary anyway.
45my $slot_creation_time_on_primary = $publisher->safe_psql(
46	'postgres', qq[
47    SELECT current_timestamp;
48]);
49
50# Create a subscription that enables failover.
51$subscriber1->safe_psql('postgres',
52	"CREATE SUBSCRIPTION regress_mysub1 CONNECTION '$publisher_connstr' PUBLICATION regress_mypub WITH (slot_name = lsub1_slot, copy_data = false, failover = true, enabled = false);"
53);
54
55# Confirm that the failover flag on the slot is turned on
56is( $publisher->safe_psql(
57		'postgres',
58		q{SELECT failover from pg_replication_slots WHERE slot_name = 'lsub1_slot';}
59	),
60	"t",
61	'logical slot has failover true on the publisher');
62
63##################################################
64# Test that changing the failover property of a subscription updates the
65# corresponding failover property of the slot.
66##################################################
67
68# Disable failover
69$subscriber1->safe_psql('postgres',
70	"ALTER SUBSCRIPTION regress_mysub1 SET (failover = false)");
71
72# Confirm that the failover flag on the slot has now been turned off
73is( $publisher->safe_psql(
74		'postgres',
75		q{SELECT failover from pg_replication_slots WHERE slot_name = 'lsub1_slot';}
76	),
77	"f",
78	'logical slot has failover false on the publisher');
79
80# Enable failover
81$subscriber1->safe_psql('postgres',
82	"ALTER SUBSCRIPTION regress_mysub1 SET (failover = true)");
83
84# Confirm that the failover flag on the slot has now been turned on
85is( $publisher->safe_psql(
86		'postgres',
87		q{SELECT failover from pg_replication_slots WHERE slot_name = 'lsub1_slot';}
88	),
89	"t",
90	'logical slot has failover true on the publisher');
91
92##################################################
93# Test that the failover option cannot be changed for enabled subscriptions.
94##################################################
95
96# Enable subscription
97$subscriber1->safe_psql('postgres',
98	"ALTER SUBSCRIPTION regress_mysub1 ENABLE");
99
100# Disable failover for enabled subscription
101my ($result, $stdout, $stderr) = $subscriber1->psql('postgres',

103ok( $stderr =~
104	  /ERROR:  cannot set option "failover" for enabled subscription/,
105	"altering failover is not allowed for enabled subscription");

118# Test logical failover slots corresponding to different plugins can be
119# synced to the standby.
120#
121# Configure standby1 to replicate and synchronize logical slots configured
122# for failover on the primary
123#
124#              failover slot lsub1_slot   |       output_plugin: pgoutput
125#              failover slot lsub2_slot   |       output_plugin: test_decoding
126# primary --->                            |
127#              physical slot sb1_slot --->| ----> standby1 (connected via streaming replication)
128#                                         |                 lsub1_slot, lsub2_slot (synced_slot)

132my $backup_name = 'backup';
133$primary->backup($backup_name);

135# Create a standby
136my $standby1 = PostgreSQL::Test::Cluster->new('standby1');
137$standby1->init_from_backup(
138	$primary, $backup_name,
139	has_streaming => 1,
140	has_restoring => 1);
141
142# Increase the log_min_messages setting to DEBUG2 on both the standby and
143# primary to debug test failures, if any.
144my $connstr_1 = $primary->connstr;
145$standby1->append_conf(
146	'postgresql.conf', qq(
147hot_standby_feedback = on
148primary_slot_name = 'sb1_slot'
149primary_conninfo = '$connstr_1 dbname=postgres'
150log_min_messages = 'debug2'
151));
152
153$primary->append_conf('postgresql.conf', "log_min_messages = 'debug2'");
154$primary->reload;

156# Drop the subscription to prevent further advancement of the restart_lsn for
157# the lsub1_slot.
158$subscriber1->safe_psql('postgres', "DROP SUBSCRIPTION regress_mysub1;");
159
160# To ensure that restart_lsn has moved to a recent WAL position, we re-create
161# the lsub1_slot.
162$primary->psql('postgres',
163	q{SELECT pg_create_logical_replication_slot('lsub1_slot', 'pgoutput', false, false, true);}
164);

166$primary->psql('postgres',
167	q{SELECT pg_create_logical_replication_slot('lsub2_slot', 'test_decoding', false, false, true);}
168);

170$primary->psql('postgres',
171	q{SELECT pg_create_physical_replication_slot('sb1_slot');});
172
173# Start the standby so that slot syncing can begin
174$standby1->start;
175
176# Capture the inactive_since of the slot from the primary. Note that the slot
177# will be inactive since the corresponding subscription was dropped.
178my $inactive_since_on_primary =
179  $primary->validate_slot_inactive_since('lsub1_slot',
180	$slot_creation_time_on_primary);
181
182# Wait for the standby to catch up so that the standby is not lagging behind
183# the failover slots.
184$primary->wait_for_replay_catchup($standby1);
185
186# Synchronize the primary server slots to the standby.
187$standby1->safe_psql('postgres', "SELECT pg_sync_replication_slots();");
188
189# Confirm that the logical failover slots are created on the standby and are
190# flagged as 'synced'
191is( $standby1->safe_psql(
192		'postgres',
193		q{SELECT count(*) = 2 FROM pg_replication_slots WHERE slot_name IN ('lsub1_slot', 'lsub2_slot') AND synced AND NOT temporary;}
194	),
195	"t",
196	'logical slots have synced as true on standby');
197
198# Capture the inactive_since of the synced slot on the standby
199my $inactive_since_on_standby =
200  $standby1->validate_slot_inactive_since('lsub1_slot',
201	$slot_creation_time_on_primary);
202
203# Synced slot on the standby must get its own inactive_since
204is( $standby1->safe_psql(
205		'postgres',
206		"SELECT '$inactive_since_on_primary'::timestamptz < '$inactive_since_on_standby'::timestamptz;"
207	),
208	"t",
209	'synchronized slot has got its own inactive_since');

244$standby1->safe_psql('postgres', "CHECKPOINT");

247is( $standby1->safe_psql(
248		'postgres',

251	"t",
252	'synchronized slot has been invalidated');

255$standby1->append_conf('postgresql.conf', 'max_slot_wal_keep_size = -1');

261    SELECT current_timestamp;

299	'logical slot is re-synced');

388##################################################
389# Create a failover slot and advance the restart_lsn to a position where a
390# running transaction exists. This setup is for testing that the synced slots
391# can achieve the consistent snapshot state starting from the restart_lsn
392# after promotion without losing any data that otherwise would have been
393# received from the primary.
394##################################################
395
396$primary->safe_psql('postgres',
397	"SELECT pg_create_logical_replication_slot('snap_test_slot', 'test_decoding', false, false, true);"
398);
399
400# Wait for the standby to catch up so that the standby is not lagging behind
401# the failover slots.
402$primary->wait_for_replay_catchup($standby1);
403
404$standby1->safe_psql('postgres', "SELECT pg_sync_replication_slots();");
405
406# Two xl_running_xacts logs are generated here. When decoding the first log, it
407# only serializes the snapshot, without advancing the restart_lsn to the latest
408# position. This is because if a transaction is running, the restart_lsn can
409# only move to a position before that transaction. Hence, the second
410# xl_running_xacts log is needed, the decoding for which allows the restart_lsn
411# to advance to the last serialized snapshot's position (the first log).
412$primary->safe_psql(
413	'postgres', qq(
414		BEGIN;
415		SELECT txid_current();
416		SELECT pg_log_standby_snapshot();
417		COMMIT;
418		BEGIN;
419		SELECT txid_current();
420		SELECT pg_log_standby_snapshot();
421		COMMIT;
422));
423
424# Advance the restart_lsn to the position of the first xl_running_xacts log
425# generated above. Note that there might be concurrent xl_running_xacts logs
426# written by the bgwriter, which could cause the position to be advanced to an
427# unexpected point, but that would be a rare scenario and doesn't affect the
428# test results.
429$primary->safe_psql('postgres',
430	"SELECT pg_replication_slot_advance('snap_test_slot', pg_current_wal_lsn());"
431);
432
433# Wait for the standby to catch up so that the standby is not lagging behind
434# the failover slots.
435$primary->wait_for_replay_catchup($standby1);
436
437# Log a message that will be consumed on the standby after promotion using the
438# synced slot. See the test where we promote standby (Promote the standby1 to
439# primary.)
440$primary->safe_psql('postgres',
441	"SELECT pg_logical_emit_message(false, 'test', 'test');");
442
443# Get the confirmed_flush_lsn for the logical slot snap_test_slot on the primary
444my $confirmed_flush_lsn = $primary->safe_psql('postgres',
445	"SELECT confirmed_flush_lsn from pg_replication_slots WHERE slot_name = 'snap_test_slot';"
446);
447
448$standby1->safe_psql('postgres', "SELECT pg_sync_replication_slots();");
449
450# Verify that confirmed_flush_lsn of snap_test_slot slot is synced to the standby
451ok( $standby1->poll_query_until(
452		'postgres',
453		"SELECT '$confirmed_flush_lsn' = confirmed_flush_lsn from pg_replication_slots WHERE slot_name = 'snap_test_slot' AND synced AND NOT temporary;"
454	),
455	'confirmed_flush_lsn of slot snap_test_slot synced to standby');

595##################################################
596# Test that logical failover replication slots wait for the specified
597# physical replication slots to receive the changes first. It uses the
598# following set up:
599#
600#				(physical standbys)
601#				| ----> standby1 (primary_slot_name = sb1_slot)
602#				| ----> standby2 (primary_slot_name = sb2_slot)
603# primary -----	|
604#				(logical replication)
605#				| ----> subscriber1 (failover = true, slot_name = lsub1_slot)
606#				| ----> subscriber2 (failover = false, slot_name = lsub2_slot)
607#
608# synchronized_standby_slots = 'sb1_slot'
609#
610# The setup is configured in such a way that the logical slot of subscriber1 is
611# enabled for failover, and thus the subscriber1 will wait for the physical
612# slot of standby1(sb1_slot) to catch up before receiving the decoded changes.
613##################################################
614
615$backup_name = 'backup3';
616
617$primary->psql('postgres',
618	q{SELECT pg_create_physical_replication_slot('sb2_slot');});
619
620$primary->backup($backup_name);
621
622# Create another standby
623my $standby2 = PostgreSQL::Test::Cluster->new('standby2');
624$standby2->init_from_backup(
625	$primary, $backup_name,
626	has_streaming => 1,
627	has_restoring => 1);
628$standby2->append_conf(
629	'postgresql.conf', qq(
630primary_slot_name = 'sb2_slot'
631));
632$standby2->start;
633$primary->wait_for_replay_catchup($standby2);
634
635# Configure primary to disallow any logical slots that have enabled failover
636# from getting ahead of the specified physical replication slot (sb1_slot).
637$primary->append_conf(
638	'postgresql.conf', qq(
639synchronized_standby_slots = 'sb1_slot'
640));
641$primary->reload;
642
643# Create another subscriber node without enabling failover, wait for sync to
644# complete
645my $subscriber2 = PostgreSQL::Test::Cluster->new('subscriber2');
646$subscriber2->init;
647$subscriber2->start;
648$subscriber2->safe_psql(
649	'postgres', qq[
650	CREATE TABLE tab_int (a int PRIMARY KEY);
651	CREATE SUBSCRIPTION regress_mysub2 CONNECTION '$publisher_connstr' PUBLICATION regress_mypub WITH (slot_name = lsub2_slot);
652]);
653
654$subscriber2->wait_for_subscription_sync;
655
656$subscriber1->safe_psql('postgres',
657	"ALTER SUBSCRIPTION regress_mysub1 ENABLE");
658
659my $offset = -s $primary->logfile;
660
661# Stop the standby associated with the specified physical replication slot
662# (sb1_slot) so that the logical replication slot (lsub1_slot) won't receive
663# changes until the standby comes up.
664$standby1->stop;
665
666# Create some data on the primary
667my $primary_row_count = 20;
668$primary->safe_psql('postgres',
669	"INSERT INTO tab_int SELECT generate_series(11, $primary_row_count);");
670
671# Wait until the standby2 that's still running gets the data from the primary
672$primary->wait_for_replay_catchup($standby2);
673$result = $standby2->safe_psql('postgres',
674	"SELECT count(*) = $primary_row_count FROM tab_int;");
675is($result, 't', "standby2 gets data from primary");
676
677# Wait for regress_mysub2 to get the data from the primary. This subscription
678# was not enabled for failover so it gets the data without waiting for any
679# standbys.
680$primary->wait_for_catchup('regress_mysub2');
681$result = $subscriber2->safe_psql('postgres',
682	"SELECT count(*) = $primary_row_count FROM tab_int;");
683is($result, 't', "subscriber2 gets data from primary");
684
685# Wait until the primary server logs a warning indicating that it is waiting
686# for the sb1_slot to catch up.
687$primary->wait_for_log(
688	qr/replication slot \"sb1_slot\" specified in parameter "synchronized_standby_slots" does not have active_pid/,
689	$offset);
690
691# The regress_mysub1 was enabled for failover so it doesn't get the data from
692# primary and keeps waiting for the standby specified in synchronized_standby_slots
693# (sb1_slot aka standby1).
694$result =
695  $subscriber1->safe_psql('postgres',
696	"SELECT count(*) <> $primary_row_count FROM tab_int;");
697is($result, 't',
698	"subscriber1 doesn't get data from primary until standby1 acknowledges changes"
699);
700
701# Start the standby specified in synchronized_standby_slots (sb1_slot aka standby1) and
702# wait for it to catch up with the primary.
703$standby1->start;
704$primary->wait_for_replay_catchup($standby1);
705$result = $standby1->safe_psql('postgres',
706	"SELECT count(*) = $primary_row_count FROM tab_int;");
707is($result, 't', "standby1 gets data from primary");
708
709# Now that the standby specified in synchronized_standby_slots is up and running, the
710# primary can send the decoded changes to the subscription enabled for failover
711# (i.e. regress_mysub1). While the standby was down, regress_mysub1 didn't
712# receive any data from the primary. i.e. the primary didn't allow it to go
713# ahead of standby.
714$primary->wait_for_catchup('regress_mysub1');
715$result = $subscriber1->safe_psql('postgres',
716	"SELECT count(*) = $primary_row_count FROM tab_int;");
717is($result, 't',
718	"subscriber1 gets data from primary after standby1 acknowledges changes");

722# logical failover slot, it will also wait for the slots specified in

726# Stop the standby associated with the specified physical replication slot so

731$primary->wait_for_catchup('regress_mysub1');

735# more warnings.

741	'postgres',

743	1);

745# Create a logical 'test_decoding' replication slot with failover enabled
746$primary->safe_psql('postgres',

748);

750my $back_q = $primary->background_psql(
751	'postgres',

753	timeout => $PostgreSQL::Test::Utils::timeout_default);

760   \echo logical_slot_get_changes
761   SELECT pg_logical_slot_get_changes('test_slot', NULL, NULL);
762));

764# Wait until the primary server logs a warning indicating that it is waiting

773$primary->reload;
774
775# Since there are no slots in synchronized_standby_slots, the function
776# pg_logical_slot_get_changes should now return, and the session can be
777# stopped.
778$back_q->quit;

780$primary->safe_psql('postgres',
781	"SELECT pg_drop_replication_slot('test_slot');");

784# tests.

787$primary->reload;

838##################################################
839# Test the synchronization of the two_phase setting for a subscription with the
840# standby. Additionally, prepare a transaction before enabling the two_phase
841# option; subsequent tests will verify if it can be correctly replicated to the

847# Prepare a transaction
848$primary->safe_psql(
849	'postgres', qq[
850	BEGIN;
851	INSERT INTO tab_int values(0);
852	PREPARE TRANSACTION 'test_twophase_slotsync';
853]);
854
855$primary->wait_for_replay_catchup($standby1);
856$primary->wait_for_catchup('regress_mysub1');
857
858# Disable the subscription to allow changing the two_phase option.
859$subscriber1->safe_psql('postgres',
860	"ALTER SUBSCRIPTION regress_mysub1 DISABLE");
861
862# Wait for the replication slot to become inactive on the publisher
863$primary->poll_query_until(
864	'postgres',
865	"SELECT COUNT(*) FROM pg_catalog.pg_replication_slots WHERE slot_name = 'lsub1_slot' AND active='f'",
866	1);

873]);

882ok( $standby1->poll_query_until(

884		"SELECT two_phase AND '$two_phase_at' = two_phase_at from pg_replication_slots WHERE slot_name = 'lsub1_slot' AND synced AND NOT temporary;"
885	),
886	'two_phase setting of slot lsub1_slot synced to standby');

888# Confirm that the prepared transaction is not yet replicated to the
889# subscriber.

895##################################################
896# Promote the standby1 to primary. Confirm that:
897# a) the slot 'lsub1_slot' and 'snap_test_slot' are retained on the new primary
898# b) logical replication for regress_mysub1 is resumed successfully after failover
899# c) changes from the transaction prepared 'test_twophase_slotsync' can be

901#    primary.

905
906# Capture the time before the standby is promoted
907my $promotion_time_on_primary = $standby1->safe_psql(
908	'postgres', qq[
909    SELECT current_timestamp;
910]);

912$standby1->promote;
913
914# Capture the inactive_since of the synced slot after the promotion.
915# The expectation here is that the slot gets its inactive_since as part of the
916# promotion. We do this check before the slot is enabled on the new primary
917# below, otherwise, the slot gets active setting inactive_since to NULL.
918my $inactive_since_on_new_primary =
919  $standby1->validate_slot_inactive_since('lsub1_slot',
920	$promotion_time_on_primary);
921
922is( $standby1->safe_psql(
923		'postgres',
924		"SELECT '$inactive_since_on_new_primary'::timestamptz > '$inactive_since_on_primary'::timestamptz"
925	),
926	"t",
927	'synchronized slot has got its own inactive_since on the new primary after promotion'
928);

930# Update subscription with the new primary's connection info
931my $standby1_conninfo = $standby1->connstr . ' dbname=postgres';

933	"ALTER SUBSCRIPTION regress_mysub1 CONNECTION '$standby1_conninfo';");

936is( $standby1->safe_psql(
937		'postgres',
938		q{SELECT count(*) = 2 FROM pg_replication_slots WHERE slot_name IN ('lsub1_slot', 'snap_test_slot') AND synced AND NOT temporary;}
939	),
940	't',
941	'synced slot retained on the new primary');

943# Commit the prepared transaction

945$standby1->wait_for_catchup('regress_mysub1');

947# Confirm that the prepared transaction is replicated to the subscriber
948is($subscriber1->safe_psql('postgres', q{SELECT count(*) FROM tab_int;}),
949	"11", 'prepared data replicated from the new primary');

951# Insert data on the new primary
952$standby1->safe_psql('postgres',
953	"INSERT INTO tab_int SELECT generate_series(11, 20);");
954$standby1->wait_for_catchup('regress_mysub1');

956# Confirm that data in tab_int replicated on the subscriber
957is($subscriber1->safe_psql('postgres', q{SELECT count(*) FROM tab_int;}),
958	"21", 'data replicated from the new primary');

960# Consume the data from the snap_test_slot. The synced slot should reach a
961# consistent point by restoring the snapshot at the restart_lsn serialized
962# during slot synchronization.
963$result = $standby1->safe_psql('postgres',
964	"SELECT count(*) FROM pg_logical_slot_get_changes('snap_test_slot', NULL, NULL) WHERE data ~ 'message*';"
965);
