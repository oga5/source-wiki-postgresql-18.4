4# Test for timeline switch
5use strict;
6use warnings FATAL => 'all';
7use PostgreSQL::Test::Cluster;
8use PostgreSQL::Test::Utils;
9use Test::More;
10
11# Ensure that a cascading standby is able to follow a newly-promoted standby
12# on a new timeline.
13
14# Initialize primary node
15my $node_primary = PostgreSQL::Test::Cluster->new('primary');
16$node_primary->init(allows_streaming => 1);
17$node_primary->start;
18
19# Take backup
20my $backup_name = 'my_backup';
21$node_primary->backup($backup_name);
22
23# Create two standbys linking to it
24my $node_standby_1 = PostgreSQL::Test::Cluster->new('standby_1');
25$node_standby_1->init_from_backup($node_primary, $backup_name,
26	has_streaming => 1);
27$node_standby_1->start;
28my $node_standby_2 = PostgreSQL::Test::Cluster->new('standby_2');
29$node_standby_2->init_from_backup($node_primary, $backup_name,
30	has_streaming => 1);
31$node_standby_2->start;
32
33# Create some content on primary
34$node_primary->safe_psql('postgres',
35	"CREATE TABLE tab_int AS SELECT generate_series(1,1000) AS a");
36
37# Cleanly stop and remove primary.  A clean stop is required so as all
38# the records generated on the primary are received and flushed by the two
39# standbys.
40$node_primary->stop;
41
42# promote standby 1 using "pg_promote", switching it to a new timeline
43my $psql_out = '';
44$node_standby_1->psql(
45	'postgres',
46	"SELECT pg_promote(wait_seconds => 300)",
47	stdout => \$psql_out);
48is($psql_out, 't', "promotion of standby with pg_promote");
49
50# Switch standby 2 to replay from standby 1
51my $connstr_1 = $node_standby_1->connstr;
52$node_standby_2->append_conf(
53	'postgresql.conf', qq(
54primary_conninfo='$connstr_1'
55));
56
57# Rotate logfile before restarting, for the log checks done below.
58$node_standby_2->rotate_logfile;
59$node_standby_2->restart;
60
61# Wait for walreceiver to reconnect after the restart.  We want to
62# verify that after reconnection, the walreceiver stays alive during
63# the timeline switch.
64$node_standby_2->poll_query_until('postgres',
65	"SELECT EXISTS(SELECT 1 FROM pg_stat_wal_receiver)");
66my $wr_pid_before_switch = $node_standby_2->safe_psql('postgres',
67	"SELECT pid FROM pg_stat_wal_receiver");
68
69# Insert some data in standby 1 and check its presence in standby 2
70# to ensure that the timeline switch has been done.
71$node_standby_1->safe_psql('postgres',
72	"INSERT INTO tab_int VALUES (generate_series(1001,2000))");
73$node_standby_1->wait_for_catchup($node_standby_2);
74
75my $result =
76  $node_standby_2->safe_psql('postgres', "SELECT count(*) FROM tab_int");
77is($result, qq(2000), 'check content of standby 2');
78
79# Check the logs, WAL receiver should not have been stopped while
80# transitioning to its new timeline.  There is no need to rely on an
81# offset in this check of the server logs: a new log file is used on
82# node restart when primary_conninfo is updated above.
83ok( !$node_standby_2->log_contains(
84		"FATAL: .* terminating walreceiver process due to administrator command"
85	),
86	'WAL receiver should not be stopped across timeline jumps');
87
88# Verify that the walreceiver process stayed alive across the timeline
89# switch, check its PID.
90my $wr_pid_after_switch = $node_standby_2->safe_psql('postgres',
91	"SELECT pid FROM pg_stat_wal_receiver");
92
93is($wr_pid_before_switch, $wr_pid_after_switch,
94	'WAL receiver PID matches across timeline jumps');
95
96# Ensure that a standby is able to follow a primary on a newer timeline
97# when WAL archiving is enabled.
98
99# Initialize primary node
100my $node_primary_2 = PostgreSQL::Test::Cluster->new('primary_2');
101$node_primary_2->init(allows_streaming => 1, has_archiving => 1);
102$node_primary_2->append_conf(
103	'postgresql.conf', qq(
104wal_keep_size = 512MB
105));
106$node_primary_2->start;
107
108# Take backup
109$node_primary_2->backup($backup_name);
110
111# Create standby node
112my $node_standby_3 = PostgreSQL::Test::Cluster->new('standby_3');
113$node_standby_3->init_from_backup($node_primary_2, $backup_name,
114	has_streaming => 1);
115
116# Restart primary node in standby mode and promote it, switching it
117# to a new timeline.
118$node_primary_2->set_standby_mode;
119$node_primary_2->restart;
120$node_primary_2->promote;
121
122# Start standby node, create some content on primary and check its presence
123# in standby, to ensure that the timeline switch has been done.
124$node_standby_3->start;
125$node_primary_2->safe_psql('postgres',
126	"CREATE TABLE tab_int AS SELECT 1 AS a");
127$node_primary_2->wait_for_catchup($node_standby_3);
128
129my $result_2 =
130  $node_standby_3->safe_psql('postgres', "SELECT count(*) FROM tab_int");
131is($result_2, qq(1), 'check content of standby 3');
132
133done_testing();
