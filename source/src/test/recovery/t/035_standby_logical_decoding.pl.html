33# Fetch xmin columns from slot's pg_replication_slots row, after waiting for
34# given boolean condition to be true to ensure we've reached a quiescent state.
35sub wait_for_xmins
36{
37	my ($node, $slotname, $check_expr) = @_;
38
39	$node->poll_query_until(
40		'postgres', qq[
41		SELECT $check_expr
42		FROM pg_catalog.pg_replication_slots
43		WHERE slot_name = '$slotname';
44	]) or die "Timed out waiting for slot xmins to advance";
45}

60# Drop the logical slots on standby.
61sub drop_logical_slots
62{
63	my ($slot_prefix) = @_;
64	my $active_slot = $slot_prefix . 'activeslot';
65	my $inactive_slot = $slot_prefix . 'inactiveslot';
66
67	$node_standby->psql('postgres',
68		qq[SELECT pg_drop_replication_slot('$inactive_slot')]);
69	$node_standby->psql('postgres',
70		qq[SELECT pg_drop_replication_slot('$active_slot')]);
71}

73# Acquire one of the standby logical slots created by create_logical_slots().
74# In case wait is true we are waiting for an active pid on the 'activeslot' slot.
75# If wait is not true it means we are testing a known failure scenario.
76sub make_slot_active
77{
78	my ($node, $slot_prefix, $wait, $to_stdout, $to_stderr) = @_;
79	my $slot_user_handle;
80
81	my $active_slot = $slot_prefix . 'activeslot';
82	$slot_user_handle = IPC::Run::start(
83		[
84			'pg_recvlogical',
85			'--dbname' => $node->connstr('testdb'),
86			'--slot' => $active_slot,
87			'--option' => 'include-xids=0',
88			'--option' => 'skip-empty-xacts=1',
89			'--file' => '-',
90			'--no-loop',
91			'--start',
92		],
93		'>' => $to_stdout,
94		'2>' => $to_stderr,
95		IPC::Run::timeout($default_timeout));
96
97	if ($wait)
98	{
99		# make sure activeslot is in use
100		$node->poll_query_until('testdb',
101			qq[SELECT EXISTS (SELECT 1 FROM pg_replication_slots WHERE slot_name = '$active_slot' AND active_pid IS NOT NULL)]
102		) or die "slot never became active";
103	}
104	return $slot_user_handle;
105}

107# Check pg_recvlogical stderr
108sub check_pg_recvlogical_stderr
109{
110	my ($slot_user_handle, $check_stderr) = @_;
111	my $return;
112
113	# our client should've terminated in response to the walsender error
114	$slot_user_handle->finish;
115	$return = $?;
116	cmp_ok($return, "!=", 0, "pg_recvlogical exited non-zero");
117	if ($return)
118	{
119		like($stderr, qr/$check_stderr/, 'slot has been invalidated');
120	}
121
122	return 0;
123}

139# Change hot_standby_feedback and check xmin and catalog_xmin values.
140sub change_hot_standby_feedback_and_wait_for_xmins
141{
142	my ($hsf, $invalidated) = @_;
143
144	$node_standby->append_conf(
145		'postgresql.conf', qq[
146	hot_standby_feedback = $hsf
147	]);
148
149	$node_standby->reload;
150
151	if ($hsf && $invalidated)
152	{
153		# With hot_standby_feedback on, xmin should advance,
154		# but catalog_xmin should still remain NULL since there is no logical slot.
155		wait_for_xmins($node_primary, $primary_slotname,
156			"xmin IS NOT NULL AND catalog_xmin IS NULL");
157	}
158	elsif ($hsf)
159	{
160		# With hot_standby_feedback on, xmin and catalog_xmin should advance.
161		wait_for_xmins($node_primary, $primary_slotname,
162			"xmin IS NOT NULL AND catalog_xmin IS NOT NULL");
163	}
164	else
165	{
166		# Both should be NULL since hs_feedback is off
167		wait_for_xmins($node_primary, $primary_slotname,
168			"xmin IS NULL AND catalog_xmin IS NULL");
169
170	}
171}

173# Check reason for conflict in pg_replication_slots.
174sub check_slots_conflict_reason
175{
176	my ($slot_prefix, $reason) = @_;
177
178	my $active_slot = $slot_prefix . 'activeslot';
179	my $inactive_slot = $slot_prefix . 'inactiveslot';
180
181	$res = $node_standby->safe_psql(
182		'postgres', qq(
183			 select invalidation_reason from pg_replication_slots where slot_name = '$active_slot' and conflicting;)
184	);
185
186	is($res, "$reason", "$active_slot reason for conflict is $reason");
187
188	$res = $node_standby->safe_psql(
189		'postgres', qq(
190			 select invalidation_reason from pg_replication_slots where slot_name = '$inactive_slot' and conflicting;)
191	);
192
193	is($res, "$reason", "$inactive_slot reason for conflict is $reason");
194}

196# Drop the slots, re-create them, change hot_standby_feedback,
197# check xmin and catalog_xmin values, make slot active and reset stat.
198sub reactive_slots_change_hfs_and_wait_for_xmins
199{
200	my ($previous_slot_prefix, $slot_prefix, $hsf, $invalidated) = @_;
201
202	# drop the logical slots
203	drop_logical_slots($previous_slot_prefix);
204
205	# create the logical slots
206	create_logical_slots($node_standby, $slot_prefix);
207
208	change_hot_standby_feedback_and_wait_for_xmins($hsf, $invalidated);
209
210	$handle =
211	  make_slot_active($node_standby, $slot_prefix, 1, \$stdout, \$stderr);
212
213	# reset stat: easier to check for confl_active_logicalslot in pg_stat_database_conflicts
214	$node_standby->psql('testdb', q[select pg_stat_reset();]);
215}

245# Launch $sql query, wait for a new snapshot that has a newer horizon and
246# launch a VACUUM.  $vac_option is the set of options to be passed to the
247# VACUUM command, $sql the sql to launch before triggering the vacuum and
248# $to_vac the relation to vacuum.
249#
250# Note that the injection_point avoids seeing a xl_running_xacts that could
251# advance an active replication slot's catalog_xmin. Advancing the active
252# replication slot's catalog_xmin would break some tests that expect the
253# active slot to conflict with the catalog xmin horizon.
254sub wait_until_vacuum_can_remove
255{
256	my ($vac_option, $sql, $to_vac) = @_;
257
258	# Note that from this point the checkpointer and bgwriter will skip writing
259	# xl_running_xacts record.
260	$node_primary->safe_psql('testdb',
261		"SELECT injection_points_attach('skip-log-running-xacts', 'error');");
262
263	# Get the current xid horizon,
264	my $xid_horizon = $node_primary->safe_psql('testdb',
265		qq[select pg_snapshot_xmin(pg_current_snapshot());]);
266
267	# Launch our sql.
268	$node_primary->safe_psql('testdb', qq[$sql]);
269
270	# Wait until we get a newer horizon.
271	$node_primary->poll_query_until('testdb',
272		"SELECT (select pg_snapshot_xmin(pg_current_snapshot())::text::int - $xid_horizon) > 0"
273	) or die "new snapshot does not have a newer horizon";
274
275	# Launch the vacuum command and insert into flush_wal (see CREATE
276	# TABLE flush_wal for the reason).
277	$node_primary->safe_psql(
278		'testdb', qq[VACUUM $vac_option verbose $to_vac;
279										  INSERT INTO flush_wal DEFAULT VALUES;]);
280
281	$node_primary->wait_for_replay_catchup($node_standby);
282
283	# Resume generating the xl_running_xacts record
284	$node_primary->safe_psql('testdb',
285		"SELECT injection_points_detach('skip-log-running-xacts');");
286}

557##################################################
558# Recovery conflict: Invalidate conflicting slots, including in-use slots
559# Scenario 1: hot_standby_feedback off and vacuum FULL
560#
561# In passing, ensure that replication slot stats are not removed when the
562# active slot is invalidated, and check that an error occurs when
563# attempting to alter the invalid slot.
564##################################################
565
566# One way to produce recovery conflict is to create/drop a relation and
567# launch a vacuum full on pg_class with hot_standby_feedback turned off on
568# the standby.
569reactive_slots_change_hfs_and_wait_for_xmins('behaves_ok_', 'vacuum_full_',
570	0, 1);

572# Ensure that replication slot stats are not empty before triggering the
573# conflict.
574$node_primary->safe_psql('testdb',
575	qq[INSERT INTO decoding_test(x,y) SELECT 100,'100';]);
576
577$node_standby->poll_query_until('testdb',
578	qq[SELECT total_txns > 0 FROM pg_stat_replication_slots WHERE slot_name = 'vacuum_full_activeslot']
579) or die "replication slot stats of vacuum_full_activeslot not updated";
580
581# This should trigger the conflict
582wait_until_vacuum_can_remove(
583	'full', 'CREATE TABLE conflict_test(x integer, y text);
584								 DROP TABLE conflict_test;', 'pg_class');
585
586# Check invalidation in the logfile and in pg_stat_database_conflicts
587check_for_invalidation('vacuum_full_', 1, 'with vacuum FULL on pg_class');
588
589# Verify reason for conflict is 'rows_removed' in pg_replication_slots
590check_slots_conflict_reason('vacuum_full_', 'rows_removed');
591
592# Attempting to alter an invalidated slot should result in an error
593($result, $stdout, $stderr) = $node_standby->psql(
594	'postgres',
595	qq[ALTER_REPLICATION_SLOT vacuum_full_inactiveslot (failover);],
596	replication => 'database');
597ok( $stderr =~
598	  /ERROR:  can no longer access replication slot "vacuum_full_inactiveslot"/
599	  && $stderr =~
600	  /DETAIL:  This replication slot has been invalidated due to "rows_removed"./,
601	"invalidated slot cannot be altered");
602
603# Ensure that replication slot stats are not removed after invalidation.
