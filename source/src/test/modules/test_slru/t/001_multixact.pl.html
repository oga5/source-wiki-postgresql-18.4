3# Test multixid corner cases.
4
5use strict;
6use warnings FATAL => 'all';
7
8use PostgreSQL::Test::Cluster;
9use PostgreSQL::Test::Utils;
10
11use Test::More;
12
13if ($ENV{enable_injection_points} ne 'yes')
14{
15	plan skip_all => 'Injection points not supported by this build';
16}
17
18my $node = PostgreSQL::Test::Cluster->new('main');
19$node->init;
20$node->append_conf('postgresql.conf',
21	"shared_preload_libraries = 'test_slru,injection_points'");
22$node->start;
23$node->safe_psql('postgres', q(CREATE EXTENSION injection_points));
24$node->safe_psql('postgres', q(CREATE EXTENSION test_slru));
25
26# This test creates three multixacts. The middle one is never
27# WAL-logged or recorded on the offsets page, because we pause the
28# backend and crash the server before that. After restart, verify that
29# the other multixacts are readable, despite the middle one being
30# lost.
31
32# Create the first multixact
33my $bg_psql = $node->background_psql('postgres');
34my $multi1 = $bg_psql->query_safe(q(SELECT test_create_multixact();));
35
36# Assign the middle multixact. Use an injection point to prevent it
37# from being fully recorded.
38$node->safe_psql('postgres',
39	q{SELECT injection_points_attach('multixact-create-from-members','wait');}
40);
41
42$bg_psql->query_until(
43	qr/assigning lost multi/, q(
44\echo assigning lost multi
45	SELECT test_create_multixact();
46));
47
48$node->wait_for_event('client backend', 'multixact-create-from-members');
49$node->safe_psql('postgres',
50	q{SELECT injection_points_detach('multixact-create-from-members')});
51
52# Create the third multixid
53my $multi2 = $node->safe_psql('postgres', q{SELECT test_create_multixact();});
54
55# All set and done, it's time for hard restart
56$node->stop('immediate');
57$node->start;
58$bg_psql->{run}->finish;
59
60# Verify that the recorded multixids are readable
61is( $node->safe_psql('postgres', qq{SELECT test_read_multixact('$multi1');}),
62	'',
63	'first recorded multi is readable');
64
65is( $node->safe_psql('postgres', qq{SELECT test_read_multixact('$multi2');}),
66	'',
67	'second recorded multi is readable');
68
69done_testing();
