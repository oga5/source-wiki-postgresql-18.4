21PG_FUNCTION_INFO_V1(test_create_multixact);
22PG_FUNCTION_INFO_V1(test_read_multixact);
23
24/*
25 * Produces multixact with 2 current xids
26 */
27Datum
28test_create_multixact(PG_FUNCTION_ARGS)
29{
30	MultiXactId id;
31
32	MultiXactIdSetOldestMember();
33	id = MultiXactIdCreate(GetCurrentTransactionId(), MultiXactStatusUpdate,
34						   GetCurrentTransactionId(), MultiXactStatusForShare);
35	PG_RETURN_TRANSACTIONID(id);
36}
37
38/*
39 * Reads given multixact.  Discards local cache to make a real read.
40 */
41Datum
42test_read_multixact(PG_FUNCTION_ARGS)
43{
44	MultiXactId id = PG_GETARG_TRANSACTIONID(0);
45	MultiXactMember *members;
46
47	/* discard caches */
48	AtEOXact_MultiXact();
49
50	if (GetMultiXactIdMembers(id, &members, false, false) == -1)
51		elog(ERROR, "MultiXactId not found");
52
53	PG_RETURN_VOID();
54}
