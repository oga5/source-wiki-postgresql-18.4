24# Targets to generate or remove the ssl certificate and key
25# Normally not needed. Don't run these targets in a vpath build, the results
26# won't be in the right place if you do.
27
28# needs to agree with what's in the test script
29PASS = FooBaR1
30
31.PHONY: ssl-files ssl-files-clean
32
33ssl-files:
34	$(OPENSSL) req -new -x509 -days 10000 -nodes -out server.crt \
35		-keyout server.ckey -subj "/CN=localhost"
36	$(OPENSSL) pkey -aes256 -in server.ckey -out server.key -passout pass:$(PASS)
37	rm server.ckey
38
39ssl-files-clean:
40	rm -f server.crt server.key
