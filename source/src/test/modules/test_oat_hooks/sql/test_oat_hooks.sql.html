src/test/modules/test_oat_hooks/sql/test_oat_hooks.sql
1-- This test script fails if debug_discard_caches is enabled, because cache
2-- flushes cause extra calls of the OAT hook in recomputeNamespacePath,
3-- resulting in more NOTICE messages than are in the expected output.
4SET debug_discard_caches = 0;
5
6-- Creating privileges on a placeholder GUC should create entries in the
7-- pg_parameter_acl catalog which conservatively grant no privileges to public.
8CREATE ROLE regress_role_joe;
9GRANT SET ON PARAMETER test_oat_hooks.user_var1 TO regress_role_joe;
10GRANT SET ON PARAMETER test_oat_hooks.super_var1 TO regress_role_joe;
11
12-- SET commands fire both the ProcessUtility_hook and the
13-- object_access_hook_str.  Since the auditing GUC starts out false, we miss the
14-- initial "attempting" audit message from the ProcessUtility_hook, but we
15-- should thereafter see the audit messages.
16LOAD 'test_oat_hooks';
17SET test_oat_hooks.audit = true;
18
19-- Creating privileges on an existent custom GUC should create precisely the
20-- right privileges, not overly conservative ones.
21GRANT SET ON PARAMETER test_oat_hooks.user_var2 TO regress_role_joe;
22GRANT SET ON PARAMETER test_oat_hooks.super_var2 TO regress_role_joe;
23
24-- Granting multiple privileges on a parameter should be reported correctly to
25-- the OAT hook, but beware that WITH GRANT OPTION is not represented.
26GRANT SET, ALTER SYSTEM ON PARAMETER none.such TO regress_role_joe;
27GRANT SET, ALTER SYSTEM ON PARAMETER another.bogus TO regress_role_joe WITH GRANT OPTION;
28
29-- Check when the hooks fire relative to dependency based abort of a drop
30DROP ROLE regress_role_joe;
31
32-- Check the behavior of the hooks relative to do-nothing grants and revokes
33GRANT SET ON PARAMETER maintenance_work_mem TO PUBLIC;
34REVOKE SET ON PARAMETER maintenance_work_mem FROM PUBLIC;
35REVOKE ALTER SYSTEM ON PARAMETER maintenance_work_mem FROM PUBLIC;
36
37-- Check the behavior of the hooks relative to unrecognized parameters
38GRANT ALL ON PARAMETER "none.such" TO PUBLIC;
39
40-- Check relative to an operation that causes the catalog entry to be deleted
41REVOKE ALL ON PARAMETER "none.such" FROM PUBLIC;
42
43-- Create objects for use in the test
44CREATE USER regress_test_user;
45CREATE TABLE regress_test_table (t text);
46CREATE INDEX regress_test_table_t_idx ON regress_test_table (t);
47GRANT SELECT ON Table regress_test_table TO public;
48CREATE FUNCTION regress_test_func (t text) RETURNS text AS $$
49	SELECT $1;
50$$ LANGUAGE sql;
51GRANT EXECUTE ON FUNCTION regress_test_func (text) TO public;
52
53-- Do a few things as superuser
54SELECT * FROM regress_test_table;
55SELECT regress_test_func('arg');
56SET work_mem = 8192;
57RESET work_mem;
58ALTER SYSTEM SET work_mem = 8192;
59ALTER SYSTEM RESET work_mem;
60
61-- Do those same things as non-superuser
62SET SESSION AUTHORIZATION regress_test_user;
63SELECT * FROM regress_test_table;
64SELECT regress_test_func('arg');
65SET work_mem = 8192;
66RESET work_mem;
67ALTER SYSTEM SET work_mem = 8192;
68ALTER SYSTEM RESET work_mem;
69
70SET test_oat_hooks.user_var1 = true;
71SET test_oat_hooks.super_var1 = true;
72ALTER SYSTEM SET test_oat_hooks.user_var1 = true;
73ALTER SYSTEM SET test_oat_hooks.super_var1 = true;
74SET test_oat_hooks.user_var2 = true;
75SET test_oat_hooks.super_var2 = true;
76ALTER SYSTEM SET test_oat_hooks.user_var2 = true;
77ALTER SYSTEM SET test_oat_hooks.super_var2 = true;
78
79RESET SESSION AUTHORIZATION;
80
81-- Turn off non-superuser permissions
82SET test_oat_hooks.deny_set_variable = true;
83SET test_oat_hooks.deny_alter_system = true;
84SET test_oat_hooks.deny_object_access = true;
85SET test_oat_hooks.deny_exec_perms = true;
86SET test_oat_hooks.deny_utility_commands = true;
87
88-- Try again as non-superuser with permissions denied
89SET SESSION AUTHORIZATION regress_test_user;
90SELECT * FROM regress_test_table;
91SELECT regress_test_func('arg');
92SET work_mem = 8192;
93RESET work_mem;
94ALTER SYSTEM SET work_mem = 8192;
95ALTER SYSTEM RESET work_mem;
96
97-- try labelled drops
98RESET SESSION AUTHORIZATION;
99DROP INDEX CONCURRENTLY regress_test_table_t_idx;
100
101-- Clean up
102SET test_oat_hooks.audit = false;
103DROP ROLE regress_role_joe;  -- fails
104REVOKE ALL PRIVILEGES ON PARAMETER
105	none.such, another.bogus,
106	test_oat_hooks.user_var1, test_oat_hooks.super_var1,
107	test_oat_hooks.user_var2, test_oat_hooks.super_var2
108	FROM regress_role_joe;
109DROP ROLE regress_role_joe;
110DROP ROLE regress_test_user;
