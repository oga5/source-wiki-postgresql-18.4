6use strict;
7use warnings FATAL => 'all';
8use Fcntl;
9use IO::File;
10use PostgreSQL::Test::Cluster;
11use PostgreSQL::Test::Utils;
12use Test::More;
13
14# We know that macOS has F_NOCACHE, and we know that Windows has
15# FILE_FLAG_NO_BUFFERING, and we assume that their typical file systems will
16# accept those flags.  For every other system, we'll probe for O_DIRECT
17# support.
18
19if ($^O ne 'darwin' && $^O ne 'MSWin32')
20{
21	# Perl's Fcntl module knows if this system has O_DIRECT in <fcntl.h>.
22	if (defined &O_DIRECT)
23	{
24		# Can we open a file in O_DIRECT mode in the file system where
25		# tmp_check lives?
26		my $f = IO::File->new(
27			"${PostgreSQL::Test::Utils::tmp_check}/test_o_direct_file",
28			O_RDWR | O_DIRECT | O_CREAT);
29		if (!$f)
30		{
31			plan skip_all =>
32			  "pre-flight test if we can open a file with O_DIRECT failed: $!";
33		}
34		$f->close;
35	}
36	else
37	{
38		plan skip_all => "no O_DIRECT";
39	}
40}
41
42my $node = PostgreSQL::Test::Cluster->new('main');
43$node->init;
44$node->append_conf(
45	'postgresql.conf', qq{
46debug_io_direct = 'data,wal,wal_init'
47shared_buffers = '256kB' # tiny to force I/O
48wal_level = replica # minimal runs out of shared_buffers when set so tiny
49});
50$node->start;
51
52# Do some work that is bound to generate shared and local writes and reads as a
53# simple exercise.
54$node->safe_psql('postgres',
55	'create table t1 as select 1 as i from generate_series(1, 10000)');
56$node->safe_psql('postgres', 'create table t2count (i int)');
57$node->safe_psql(
58	'postgres', qq{
59begin;
60create temporary table t2 as select 1 as i from generate_series(1, 10000);
61update t2 set i = i;
62insert into t2count select count(*) from t2;
63commit;
64});
65$node->safe_psql('postgres', 'update t1 set i = i');
66is( '10000',
67	$node->safe_psql('postgres', 'select count(*) from t1'),
68	"read back from shared");
69is( '10000',
70	$node->safe_psql('postgres', 'select * from t2count'),
71	"read back from local");
72$node->stop('immediate');
73
74$node->start;
75is( '10000',
76	$node->safe_psql('postgres', 'select count(*) from t1'),
77	"read back from shared after crash recovery");
78$node->stop;
79
80done_testing();
