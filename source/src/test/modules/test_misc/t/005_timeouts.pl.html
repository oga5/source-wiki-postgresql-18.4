4use strict;
5use warnings FATAL => 'all';
6use locale;
7
8use PostgreSQL::Test::Cluster;
9use PostgreSQL::Test::Utils;
10use Time::HiRes qw(usleep);
11use Test::More;
12
13# Test timeouts that will cause FATAL errors.  This test relies on injection
14# points to await a timeout occurrence. Relying on sleep proved to be unstable
15# on buildfarm. It's difficult to rely on the NOTICE injection point because
16# the backend under FATAL error can behave differently.
17
18if ($ENV{enable_injection_points} ne 'yes')
19{
20	plan skip_all => 'Injection points not supported by this build';
21}
22
23# Node initialization
24my $node = PostgreSQL::Test::Cluster->new('master');
25$node->init();
26$node->start;
27
28# Check if the extension injection_points is available, as it may be
29# possible that this script is run with installcheck, where the module
30# would not be installed by default.
31if (!$node->check_extension('injection_points'))
32{
33	plan skip_all => 'Extension injection_points not installed';
34}
35
36$node->safe_psql('postgres', 'CREATE EXTENSION injection_points;');
37
38#
39# 1. Test of the transaction timeout
40#
41
42$node->safe_psql('postgres',
43	"SELECT injection_points_attach('transaction-timeout', 'wait');");
44
45my $psql_session = $node->background_psql('postgres');
46
47# The following query will generate a stream of SELECT 1 queries. This is done
48# so to exercise transaction timeout in the presence of short queries.
49# Note: the interval value is parsed with locale-aware strtod()
50$psql_session->query_until(
51	qr/starting_bg_psql/,
52	sprintf(
53		q(\echo starting_bg_psql
54		SET transaction_timeout to '10ms';
55		BEGIN;
56		SELECT 1 \watch %g
57		\q
58), 0.001));
59
60# Wait until the backend enters the timeout injection point. Will get an error
61# here if anything goes wrong.
62$node->wait_for_event('client backend', 'transaction-timeout');
63
64my $log_offset = -s $node->logfile;
65
66# Remove the injection point.
67$node->safe_psql('postgres',
68	"SELECT injection_points_wakeup('transaction-timeout');");
69
70# Check that the timeout was logged.
71$node->wait_for_log('terminating connection due to transaction timeout',
72	$log_offset);
73
74# If we send \q with $psql_session->quit the command can be sent to the session
75# already closed. So \q is in initial script, here we only finish IPC::Run.
76$psql_session->{run}->finish;
77
78
79#
80# 2. Test of the idle in transaction timeout
81#
82
83$node->safe_psql('postgres',
84	"SELECT injection_points_attach('idle-in-transaction-session-timeout', 'wait');"
85);
86
87# We begin a transaction and the hand on the line
88$psql_session = $node->background_psql('postgres');
89$psql_session->query_until(
90	qr/starting_bg_psql/, q(
91   \echo starting_bg_psql
92   SET idle_in_transaction_session_timeout to '10ms';
93   BEGIN;
94));
95
96# Wait until the backend enters the timeout injection point.
97$node->wait_for_event('client backend',
98	'idle-in-transaction-session-timeout');
99
100$log_offset = -s $node->logfile;
101
102# Remove the injection point.
103$node->safe_psql('postgres',
104	"SELECT injection_points_wakeup('idle-in-transaction-session-timeout');");
105
106# Check that the timeout was logged.
107$node->wait_for_log(
108	'terminating connection due to idle-in-transaction timeout', $log_offset);
109
110ok($psql_session->quit);
111
112
113#
114# 3. Test of the idle session timeout
115#
116$node->safe_psql('postgres',
117	"SELECT injection_points_attach('idle-session-timeout', 'wait');");
118
119# We just initialize the GUC and wait. No transaction is required.
120$psql_session = $node->background_psql('postgres');
121$psql_session->query_until(
122	qr/starting_bg_psql/, q(
123   \echo starting_bg_psql
124   SET idle_session_timeout to '10ms';
125));
126
127# Wait until the backend enters the timeout injection point.
128$node->wait_for_event('client backend', 'idle-session-timeout');
129
130$log_offset = -s $node->logfile;
131
132# Remove the injection point.
133$node->safe_psql('postgres',
134	"SELECT injection_points_wakeup('idle-session-timeout');");
135
136# Check that the timeout was logged.
137$node->wait_for_log('terminating connection due to idle-session timeout',
138	$log_offset);
139
140ok($psql_session->quit);
141
142done_testing();
