4# Tests to cross-check the consistency of GUC parameters with
5# postgresql.conf.sample.
6
7use strict;
8use warnings FATAL => 'all';
9use PostgreSQL::Test::Cluster;
10use PostgreSQL::Test::Utils;
11use Test::More;
12
13my $node = PostgreSQL::Test::Cluster->new('main');
14$node->init;
15$node->start;
16
17# Grab the names of all the parameters that can be listed in the
18# configuration sample file.  config_file is an exception, it is not
19# in postgresql.conf.sample but is part of the lists from guc_tables.c.
20# Custom GUCs loaded by extensions are excluded.
21my $all_params = $node->safe_psql(
22	'postgres',
23	"SELECT name
24     FROM pg_settings
25   WHERE NOT 'NOT_IN_SAMPLE' = ANY (pg_settings_get_flags(name)) AND
26       name <> 'config_file' AND category <> 'Customized Options'
27     ORDER BY 1");
28# Note the lower-case conversion, for consistency.
29my @all_params_array = split("\n", lc($all_params));
30
31# Grab the names of all parameters marked as NOT_IN_SAMPLE.
32my $not_in_sample = $node->safe_psql(
33	'postgres',
34	"SELECT name
35     FROM pg_settings
36   WHERE 'NOT_IN_SAMPLE' = ANY (pg_settings_get_flags(name))
37     ORDER BY 1");
38my @not_in_sample_array = split("\n", lc($not_in_sample));
39
40# use the sample file from the temp install
41my $share_dir = $node->config_data('--sharedir');
42my $sample_file = "$share_dir/postgresql.conf.sample";
43
44# List of all the GUCs found in the sample file.
45my @gucs_in_file;
46
47# Read the sample file line-by-line, checking its contents to build a list
48# of everything known as a GUC.
49my $num_tests = 0;
50open(my $contents, '<', $sample_file)
51  || die "Could not open $sample_file: $!";
52while (my $line = <$contents>)
53{
54	# Check if this line matches a GUC parameter:
55	# - Each parameter is preceded by "#", but not "# " in the sample
56	# file.
57	# - Valid configuration options are followed immediately by " = ",
58	# with one space before and after the equal sign.
59	if ($line =~ m/^#?([_[:alnum:]]+) = .*/)
60	{
61		# Lower-case conversion matters for some of the GUCs.
62		my $param_name = lc($1);
63
64		# Ignore some exceptions.
65		next if $param_name eq "include";
66		next if $param_name eq "include_dir";
67		next if $param_name eq "include_if_exists";
68
69		# Update the list of GUCs found in the sample file, for the
70		# follow-up tests.
71		push @gucs_in_file, $param_name;
72	}
73}
74
75close $contents;
76
77# Cross-check that all the GUCs found in the sample file match the ones
78# fetched above.  This maps the arrays to a hash, making the creation of
79# each exclude and intersection list easier.
80my %gucs_in_file_hash = map { $_ => 1 } @gucs_in_file;
81my %all_params_hash = map { $_ => 1 } @all_params_array;
82my %not_in_sample_hash = map { $_ => 1 } @not_in_sample_array;
83
84my @missing_from_file = grep(!$gucs_in_file_hash{$_}, @all_params_array);
85is(scalar(@missing_from_file),
86	0, "no parameters missing from postgresql.conf.sample");
87
88my @missing_from_list = grep(!$all_params_hash{$_}, @gucs_in_file);
89is(scalar(@missing_from_list), 0, "no parameters missing from guc_tables.c");
90
91my @sample_intersect = grep($not_in_sample_hash{$_}, @gucs_in_file);
92is(scalar(@sample_intersect),
93	0, "no parameters marked as NOT_IN_SAMPLE in postgresql.conf.sample");
94
95# These would log some information only on errors.
96foreach my $param (@missing_from_file)
97{
98	print(
99		"found GUC $param in guc_tables.c, missing from postgresql.conf.sample\n"
100	);
101}
102foreach my $param (@missing_from_list)
103{
104	print(
105		"found GUC $param in postgresql.conf.sample, with incorrect info in guc_tables.c\n"
106	);
107}
108foreach my $param (@sample_intersect)
109{
110	print(
111		"found GUC $param in postgresql.conf.sample, marked as NOT_IN_SAMPLE\n"
112	);
113}
114
115done_testing();
