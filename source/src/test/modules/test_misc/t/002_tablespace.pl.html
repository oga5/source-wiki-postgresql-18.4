4# Simple tablespace tests that can't be replicated on the same host
5# due to the use of absolute paths, so we keep them out of the regular
6# regression tests.
7
8use strict;
9use warnings FATAL => 'all';
10use PostgreSQL::Test::Cluster;
11use PostgreSQL::Test::Utils;
12use Test::More;
13
14my $node = PostgreSQL::Test::Cluster->new('main');
15$node->init;
16$node->start;
17
18# Create a couple of directories to use as tablespaces.
19my $basedir = $node->basedir();
20my $TS1_LOCATION = "$basedir/ts1";
21$TS1_LOCATION =~ s/\/\.\//\//g;    # collapse foo/./bar to foo/bar
22mkdir($TS1_LOCATION);
23my $TS2_LOCATION = "$basedir/ts2";
24$TS2_LOCATION =~ s/\/\.\//\//g;
25mkdir($TS2_LOCATION);
26
27my $result;
28
29# Create a tablespace with an absolute path
30$result = $node->psql('postgres',
31	"CREATE TABLESPACE regress_ts1 LOCATION '$TS1_LOCATION'");
32ok($result == 0, 'create tablespace with absolute path');
33
34# Can't create a tablespace where there is one already
35$result = $node->psql('postgres',
36	"CREATE TABLESPACE regress_ts1 LOCATION '$TS1_LOCATION'");
37ok($result != 0, 'clobber tablespace with absolute path');
38
39# Create table in it
40$result = $node->psql('postgres', "CREATE TABLE t () TABLESPACE regress_ts1");
41ok($result == 0, 'create table in tablespace with absolute path');
42
43# Can't drop a tablespace that still has a table in it
44$result = $node->psql('postgres', "DROP TABLESPACE regress_ts1");
45ok($result != 0, 'drop tablespace with absolute path');
46
47# Drop the table
48$result = $node->psql('postgres', "DROP TABLE t");
49ok($result == 0, 'drop table in tablespace with absolute path');
50
51# Drop the tablespace
52$result = $node->psql('postgres', "DROP TABLESPACE regress_ts1");
53ok($result == 0, 'drop tablespace with absolute path');
54
55# Create two absolute tablespaces and two in-place tablespaces, so we can
56# testing various kinds of tablespace moves.
57$result = $node->psql('postgres',
58	"CREATE TABLESPACE regress_ts1 LOCATION '$TS1_LOCATION'");
59ok($result == 0, 'create tablespace 1 with absolute path');
60$result = $node->psql('postgres',
61	"CREATE TABLESPACE regress_ts2 LOCATION '$TS2_LOCATION'");
62ok($result == 0, 'create tablespace 2 with absolute path');
63$result = $node->psql('postgres',
64	"SET allow_in_place_tablespaces=on; CREATE TABLESPACE regress_ts3 LOCATION ''"
65);
66ok($result == 0, 'create tablespace 3 with in-place directory');
67$result = $node->psql('postgres',
68	"SET allow_in_place_tablespaces=on; CREATE TABLESPACE regress_ts4 LOCATION ''"
69);
70ok($result == 0, 'create tablespace 4 with in-place directory');
71
72# Create a table and test moving between absolute and in-place tablespaces
73$result = $node->psql('postgres', "CREATE TABLE t () TABLESPACE regress_ts1");
74ok($result == 0, 'create table in tablespace 1');
75$result = $node->psql('postgres', "ALTER TABLE t SET tablespace regress_ts2");
76ok($result == 0, 'move table abs->abs');
77$result = $node->psql('postgres', "ALTER TABLE t SET tablespace regress_ts3");
78ok($result == 0, 'move table abs->in-place');
79$result = $node->psql('postgres', "ALTER TABLE t SET tablespace regress_ts4");
80ok($result == 0, 'move table in-place->in-place');
81$result = $node->psql('postgres', "ALTER TABLE t SET tablespace regress_ts1");
82ok($result == 0, 'move table in-place->abs');
83
84# Drop everything
85$result = $node->psql('postgres', "DROP TABLE t");
86ok($result == 0, 'create table in tablespace 1');
87$result = $node->psql('postgres', "DROP TABLESPACE regress_ts1");
88ok($result == 0, 'drop tablespace 1');
89$result = $node->psql('postgres', "DROP TABLESPACE regress_ts2");
90ok($result == 0, 'drop tablespace 2');
91$result = $node->psql('postgres', "DROP TABLESPACE regress_ts3");
92ok($result == 0, 'drop tablespace 3');
93$result = $node->psql('postgres', "DROP TABLESPACE regress_ts4");
94ok($result == 0, 'drop tablespace 4');
95
96$node->stop;
97
98done_testing();
