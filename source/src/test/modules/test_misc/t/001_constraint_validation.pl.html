19# Run a SQL command and return psql's stderr (including debug messages)
20sub run_sql_command
21{
22	my $sql = shift;
23	my $stderr;
24
25	$node->psql(
26		'postgres',
27		$sql,
28		stderr => \$stderr,
29		on_error_die => 1,
30		on_error_stop => 1);
31	return $stderr;
32}
33
34# Check whether result of run_sql_command shows that we did a verify pass
35sub is_table_verified
36{
37	my $output = shift;
38	return index($output, 'DEBUG:  verifying table') != -1;
39}

43note "test alter table set not null";

45run_sql_command(
46	'create table atacc1 (test_a int, test_b int);
47	 insert into atacc1 values (1, 2);');
48
49$output = run_sql_command('alter table atacc1 alter test_a set not null;');
50ok(is_table_verified($output),
51	'column test_a without constraint will scan table');
52
53run_sql_command(
54	'alter table atacc1 alter test_a drop not null;
55	 alter table atacc1 add constraint atacc1_constr_a_valid
56	 check(test_a is not null);');
57
58# normal run will verify table data
59$output = run_sql_command('alter table atacc1 alter test_a set not null;');
60ok(!is_table_verified($output), 'with constraint will not scan table');
61ok( $output =~
62	  m/existing constraints on column "atacc1.test_a" are sufficient to prove that it does not contain nulls/,
63	'test_a proved by constraints');
64
65run_sql_command('alter table atacc1 alter test_a drop not null;');
66
67# we have check only for test_a column, so we need verify table for test_b
68$output = run_sql_command(
69	'alter table atacc1 alter test_b set not null, alter test_a set not null;'
70);
71ok(is_table_verified($output), 'table was scanned');
72# we may miss debug message for test_a constraint because we need verify table due test_b
73ok( !(  $output =~
74		m/existing constraints on column "atacc1.test_b" are sufficient to prove that it does not contain nulls/
75	),
76	'test_b not proved by wrong constraints');
77run_sql_command(
78	'alter table atacc1 alter test_a drop not null, alter test_b drop not null;'
79);

81# test with both columns having check constraints
82run_sql_command(
83	'alter table atacc1 add constraint atacc1_constr_b_valid check(test_b is not null);'
84);
85$output = run_sql_command(
86	'alter table atacc1 alter test_b set not null, alter test_a set not null;'

88ok(!is_table_verified($output), 'table was not scanned for both columns');
89ok( $output =~

91	'test_a proved by constraints');
92ok( $output =~
93	  m/existing constraints on column "atacc1.test_b" are sufficient to prove that it does not contain nulls/,
94	'test_b proved by constraints');
95run_sql_command('drop table atacc1;');

99run_sql_command(

266# If a partitioned table being created or an existing table being attached
267# as a partition does not have a constraint that would allow validation scan
268# to be skipped, but an individual partition does, then the partition's
269# validation scan is skipped.
270run_sql_command(
271	'CREATE TABLE quuux (a int, b text) PARTITION BY LIST (a);
272	CREATE TABLE quuux_default PARTITION OF quuux DEFAULT PARTITION BY LIST (b);
273	CREATE TABLE quuux_default1 PARTITION OF quuux_default (
274	CONSTRAINT check_1 CHECK (a IS NOT NULL AND a = 1)
275	) FOR VALUES IN (\'b\');
276	CREATE TABLE quuux1 (a int, b text);');
277
278$output = run_sql_command(
279	'ALTER TABLE quuux ATTACH PARTITION quuux1 FOR VALUES IN (1);');
280ok(is_table_verified($output), 'quuux1 table scanned');
281ok( !(  $output =~
282		m/partition constraint for table "quuux1" is implied by existing constraints/
283	),
284	'quuux1 verified by existing constraints');
285
286run_sql_command('CREATE TABLE quuux2 (a int, b text);');
287$output = run_sql_command(
288	'ALTER TABLE quuux ATTACH PARTITION quuux2 FOR VALUES IN (2);');
289ok(!($output =~ m/verifying table "quuux_default1"/),
290	'quuux_default1 not scanned');
291ok($output =~ m/verifying table "quuux2"/, 'quuux2 scanned');
292ok( $output =~
293	  m/updated partition constraint for default partition "quuux_default1" is implied by existing constraints/,
294	'updated partition constraint for default partition quuux_default1');
