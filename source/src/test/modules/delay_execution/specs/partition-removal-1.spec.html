src/test/modules/delay_execution/specs/partition-removal-1.spec
1# Test removal of a partition with less-than-exclusive locking.
2
3setup
4{
5  CREATE TABLE partrem (a int, b text) PARTITION BY LIST(a);
6  CREATE TABLE partrem1 PARTITION OF partrem FOR VALUES IN (1);
7  CREATE TABLE partrem2 PARTITION OF partrem FOR VALUES IN (2);
8  CREATE TABLE partrem3 PARTITION OF partrem FOR VALUES IN (3);
9  INSERT INTO partrem VALUES (1, 'ABC');
10  INSERT INTO partrem VALUES (2, 'JKL');
11  INSERT INTO partrem VALUES (3, 'DEF');
12}
13
14teardown
15{
16  DROP TABLE IF EXISTS partrem, partrem2;
17}
18
19session "s1"
20setup		{ LOAD 'delay_execution';
21		  SET delay_execution.post_planning_lock_id = 12543; }
22step "s1b"	{ BEGIN; }
23step "s1brr"	{ BEGIN ISOLATION LEVEL REPEATABLE READ; }
24step "s1exec"	{ SELECT * FROM partrem WHERE a <> 1 AND a <> (SELECT 3); }
25step "s1exec2"	{ SELECT * FROM partrem WHERE a <> (SELECT 2) AND a <> 1; }
26step "s1prepare" { PREPARE ins AS INSERT INTO partrem VALUES ($1, 'GHI'); }
27step "s1execprep" { EXECUTE ins(2); }
28step "s1check" { SELECT * FROM partrem WHERE b = 'GHI'; }
29step "s1c"	{ COMMIT; }
30step "s1dealloc" { DEALLOCATE ins; }
31
32session "s2"
33step "s2remp"	{ ALTER TABLE partrem DETACH PARTITION partrem2 CONCURRENTLY; }
34
35session "s3"
36step "s3lock"	{ SELECT pg_advisory_lock(12543); }
37step "s3unlock"	{ SELECT pg_advisory_unlock(12543); }
38step "s3check"	{ SELECT * FROM partrem; }
39
40# The SELECT will be planned with all three partitions shown above,
41# of which we expect partrem1 to be pruned at planning and partrem3 at
42# execution. Then we'll block, and by the time the query is actually
43# executed, detach of partrem2 is already underway (so its row doesn't
44# show up in s3's result), but we expect its row to still appear in the
45# result for s1.
46permutation "s3lock" "s1b" "s1exec" "s2remp" "s3check" "s3unlock" "s3check" "s1c"
47permutation "s3lock" "s1brr" "s1exec" "s2remp" "s3check" "s3unlock" "s3check" "s1c"
48
49# In this case we're testing that after pruning partrem2 at runtime, the
50# query still works correctly.
51permutation "s3lock" "s1b" "s1exec2" "s2remp" "s3unlock" "s1c"
52permutation "s3lock" "s1brr" "s1exec2" "s2remp" "s3unlock" "s1c"
53
54# In this case we test that an insert that's prepared in repeatable read
55# mode still works after detaching.
56permutation "s3lock" "s1brr" "s1prepare" "s2remp" "s1execprep" "s3unlock" "s1check" "s1c" "s1check" "s1dealloc"
57permutation "s1brr" "s1prepare" "s2remp" "s3lock" "s1execprep" "s3unlock" "s1check" "s1c" "s1check" "s1dealloc"
58permutation "s1brr" "s1check" "s3lock" "s2remp" "s1prepare" "s1execprep" "s3unlock" "s1check" "s1c" "s1check" "s1dealloc"
