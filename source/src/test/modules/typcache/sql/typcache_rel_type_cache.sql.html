src/test/modules/typcache/sql/typcache_rel_type_cache.sql
1--
2-- This test checks that lookup_type_cache() can correctly handle an
3-- interruption.  We use the injection point to simulate an error but note
4-- that a similar situation could happen due to user query interruption.
5-- Despite the interruption, a map entry from the relation oid to type cache
6-- entry should be created.  This is validated by subsequent modification of
7-- the table schema, then type casts which use new schema implying
8-- successful type cache invalidation by relation oid.
9--
10
11CREATE EXTENSION injection_points;
12
13-- Make all injection points local to this process, for concurrency.
14SELECT injection_points_set_local();
15
16CREATE TABLE t (i int);
17SELECT injection_points_attach('typecache-before-rel-type-cache-insert', 'error');
18SELECT '(1)'::t;
19SELECT injection_points_detach('typecache-before-rel-type-cache-insert');
20ALTER TABLE t ADD COLUMN j int;
21SELECT '(1,2)'::t;
22
23DROP EXTENSION injection_points;
