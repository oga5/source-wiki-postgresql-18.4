17# Definition of the pg_dump runs to make.
18#
19# Each of these runs are named and those names are used below
20# to define how each test should (or shouldn't) treat a result
21# from a given run.
22#
23# compile_option indicates if the commands run depend on a compilation
24# option, if any.  This can be used to control if tests should be
25# skipped when a build dependency is not satisfied.
26#
27# test_key indicates that a given run should simply use the same
28# set of like/unlike tests as another run, and which run that is.
29#
30# dump_cmd is the pg_dump command to run, which is an array of
31# the full command and arguments to run.  Note that this is run
32# using $node->command_ok(), so the port does not need to be
33# specified and is pulled from $PGPORT, which is set by the
34# PostgreSQL::Test::Cluster system.
35#
36# restore_cmd is the pg_restore command to run, if any.  Note
37# that this should generally be used when the pg_dump goes to
38# a non-text file and that the restore can then be used to
39# generate a text file to run through the tests from the
40# non-text file generated by pg_dump.
41#
42# TODO: Have pg_restore actually restore to an independent
43# database and then pg_dump *that* database (or something along
44# those lines) to validate that part of the process.

306###############################################################
307# Definition of the tests to run.
308#
309# Each test is defined using the log message that will be used.
310#
311# A regexp should be defined for each test which provides the
312# basis for the test.  That regexp will be run against the output
313# file of each of the runs which the test is to be run against
314# and the success of the result will depend on if the regexp
315# result matches the expected 'like' or 'unlike' case.
316# The runs listed as 'like' will be checked if they match the
317# regexp and, if so, the test passes.  All runs which are not
318# listed as 'like' will be checked to ensure they don't match
319# the regexp; if they do, the test will fail.
320#
321# The below hashes provide convenience sets of runs.  Individual
322# runs can be excluded from a general hash by placing that run
323# into the 'unlike' section.
324#
325# There can then be a 'create_sql' and 'create_order' for a
326# given test.  The 'create_sql' commands are collected up in
327# 'create_order' and then run against the database prior to any
328# of the pg_dump runs happening.  This is what "seeds" the
329# system with objects to be dumped out.
330#
331# Building of this hash takes a bit of time as all of the regexps
332# included in it are compiled.  This greatly improves performance
333# as the regexps are used for each run the test applies to.
334
335# Tests which are considered 'full' dumps by pg_dump, but there
336# are flags used to exclude specific items (ACLs, LOs, etc).
337my %full_runs = (

345	no_owner => 1,

347	with_extension => 1,
348	exclude_extension => 1,

350	without_extension => 1);

869# Create a PG instance to test actually dumping from
870

872$node->init(auth_extra => [ '--create-role' => 'regress_dump_login_role' ]);
873$node->start;
874
875my $port = $node->port;
876
877my $supports_gzip = check_pg_config("#define HAVE_LIBZ 1");
878
879#########################################
880# Set up schemas, tables, etc, to be dumped.
881
882# Build up the create statements
883my $create_sql = '';
884
885foreach my $test (
886	sort {
887		if ($tests{$a}->{create_order} and $tests{$b}->{create_order})
888		{
889			$tests{$a}->{create_order} <=> $tests{$b}->{create_order};
890		}
891		elsif ($tests{$a}->{create_order})
892		{
893			-1;
894		}
895		elsif ($tests{$b}->{create_order})
896		{
897			1;
898		}
899		else
900		{
901			0;
902		}
903	} keys %tests)
904{
905	if ($tests{$test}->{create_sql})
906	{
907		$create_sql .= $tests{$test}->{create_sql};
908	}
909}
910
911# Send the combined set of commands to psql
912$node->safe_psql('postgres', $create_sql);

914#########################################
915# Create filter file for exclude_extension_filter test
916
917my $filterfile;
918
919open $filterfile, '>', "$tempdir/exclude_extension_filter.txt"
920  or die "unable to open filter file for writing";
921print $filterfile "exclude extension test_pg_dump\n";
922close $filterfile;

924#########################################
925# Run all runs
926
927foreach my $run (sort keys %pgdump_runs)
928{
929
930	my $test_key = $run;
931
932	# Skip command-level tests for gzip if there is no support for it.
933	if (   defined($pgdump_runs{$run}->{compile_option})
934		&& $pgdump_runs{$run}->{compile_option} eq 'gzip'
935		&& !$supports_gzip)
936	{
937		note "$run: skipped due to no gzip support";
938		next;
939	}
940
941	$node->command_ok(\@{ $pgdump_runs{$run}->{dump_cmd} },
942		"$run: pg_dump runs");
943
944	if ($pgdump_runs{$run}->{restore_cmd})
945	{
946		$node->command_ok(\@{ $pgdump_runs{$run}->{restore_cmd} },
947			"$run: pg_restore runs");
948	}
949
950	if ($pgdump_runs{$run}->{test_key})
951	{
952		$test_key = $pgdump_runs{$run}->{test_key};
953	}
954
955	my $output_file = slurp_file("$tempdir/${run}.sql");
956
957	#########################################
958	# Run all tests where this run is included
959	# as either a 'like' or 'unlike' test.
960
961	foreach my $test (sort keys %tests)
962	{
963		# Check for proper test definitions
964		#
965		# There should be a "like" list, even if it is empty.  (This
966		# makes the test more self-documenting.)
967		if (!defined($tests{$test}->{like}))
968		{
969			die "missing \"like\" in test \"$test\"";
970		}
971		# Check for useless entries in "unlike" list.  Runs that are
972		# not listed in "like" don't need to be excluded in "unlike".
973		if ($tests{$test}->{unlike}->{$test_key}
974			&& !defined($tests{$test}->{like}->{$test_key}))
975		{
976			die "useless \"unlike\" entry \"$test_key\" in test \"$test\"";
977		}
978
979		# Run the test listed as a like, unless it is specifically noted
980		# as an unlike (generally due to an explicit exclusion or similar).
981		if ($tests{$test}->{like}->{$test_key}
982			&& !defined($tests{$test}->{unlike}->{$test_key}))
983		{
984			if (!ok($output_file =~ $tests{$test}->{regexp},
985					"$run: should dump $test"))
986			{
987				diag("Review $run results in $tempdir");
988			}
989		}
990		else
991		{
992			if (!ok($output_file !~ $tests{$test}->{regexp},
993					"$run: should not dump $test"))
994			{
995				diag("Review $run results in $tempdir");
996			}
997		}
998	}
999}
