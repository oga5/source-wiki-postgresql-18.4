3# Consume a lot of XIDs, wrapping around a few times.
4#
5
6use strict;
7use warnings FATAL => 'all';
8use PostgreSQL::Test::Cluster;
9use PostgreSQL::Test::Utils;
10use Test::More;
11use Time::HiRes qw(usleep);
12
13if (!$ENV{PG_TEST_EXTRA} || $ENV{PG_TEST_EXTRA} !~ /\bxid_wraparound\b/)
14{
15	plan skip_all => "test xid_wraparound not enabled in PG_TEST_EXTRA";
16}
17
18# Initialize node
19my $node = PostgreSQL::Test::Cluster->new('wraparound');
20
21$node->init;
22$node->append_conf(
23	'postgresql.conf', qq[
24autovacuum_naptime = 1s
25# so it's easier to verify the order of operations
26autovacuum_max_workers = 1
27log_autovacuum_min_duration = 0
28]);
29$node->start;
30$node->safe_psql('postgres', 'CREATE EXTENSION xid_wraparound');
31
32# Create a test table. We disable autovacuum on the table to run
33# it only to prevent wraparound.
34$node->safe_psql(
35	'postgres', qq[
36CREATE TABLE wraparoundtest(t text) WITH (autovacuum_enabled = off);
37INSERT INTO wraparoundtest VALUES ('beginning');
38]);
39
40# Bump the query timeout to avoid false negatives on slow test systems.
41my $psql_timeout_secs = 4 * $PostgreSQL::Test::Utils::timeout_default;
42
43# Burn through 10 billion transactions in total, in batches of 100 million.
44my $ret;
45for my $i (1 .. 100)
46{
47	$ret = $node->safe_psql(
48		'postgres',
49		qq[SELECT consume_xids(100000000)],
50		timeout => $psql_timeout_secs);
51	$ret = $node->safe_psql('postgres',
52		qq[INSERT INTO wraparoundtest VALUES ('after $i batches')]);
53}
54
55$ret = $node->safe_psql('postgres', qq[SELECT COUNT(*) FROM wraparoundtest]);
56is($ret, "101");
57
58$node->stop;
59
60done_testing();
