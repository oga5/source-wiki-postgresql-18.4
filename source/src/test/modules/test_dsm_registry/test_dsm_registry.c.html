13#include "postgres.h"
14
15#include "fmgr.h"
16#include "storage/dsm_registry.h"
17#include "storage/lwlock.h"
18
19PG_MODULE_MAGIC;
20
21typedef struct TestDSMRegistryStruct
22{
23	int			val;
24	LWLock		lck;
25} TestDSMRegistryStruct;
26
27static TestDSMRegistryStruct *tdr_state;
28
29static void
30tdr_init_shmem(void *ptr)
31{
32	TestDSMRegistryStruct *state = (TestDSMRegistryStruct *) ptr;
33
34	LWLockInitialize(&state->lck, LWLockNewTrancheId());
35	state->val = 0;
36}
37
38static void
39tdr_attach_shmem(void)
40{
41	bool		found;
42
43	tdr_state = GetNamedDSMSegment("test_dsm_registry",
44								   sizeof(TestDSMRegistryStruct),
45								   tdr_init_shmem,
46								   &found);
47	LWLockRegisterTranche(tdr_state->lck.tranche, "test_dsm_registry");
48}
49
50PG_FUNCTION_INFO_V1(set_val_in_shmem);
51Datum
52set_val_in_shmem(PG_FUNCTION_ARGS)
53{
54	tdr_attach_shmem();
55
56	LWLockAcquire(&tdr_state->lck, LW_EXCLUSIVE);
57	tdr_state->val = PG_GETARG_INT32(0);
58	LWLockRelease(&tdr_state->lck);
59
60	PG_RETURN_VOID();
61}
62
63PG_FUNCTION_INFO_V1(get_val_in_shmem);
64Datum
65get_val_in_shmem(PG_FUNCTION_ARGS)
66{
67	int			ret;
68
69	tdr_attach_shmem();
70
71	LWLockAcquire(&tdr_state->lck, LW_SHARED);
72	ret = tdr_state->val;
73	LWLockRelease(&tdr_state->lck);
74
75	PG_RETURN_INT32(ret);
76}
