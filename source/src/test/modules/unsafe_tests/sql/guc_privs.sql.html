91-- Revoke privileges from a non-existent custom GUC.  This should not create
92-- entries in the catalog.
93REVOKE ALL ON PARAMETER "none.such" FROM regress_host_resource_admin;
94SELECT 1 FROM pg_parameter_acl WHERE parname = 'none.such';
95-- Grant and then revoke privileges on the non-existent custom GUC.  Check that
96-- a do-nothing entry is not left in the catalogs after the revoke.
97GRANT ALL ON PARAMETER none.such TO regress_host_resource_admin;
98SELECT 1 FROM pg_parameter_acl WHERE parname = 'none.such';
99REVOKE ALL ON PARAMETER "None.Such" FROM regress_host_resource_admin;
100SELECT 1 FROM pg_parameter_acl WHERE parname = 'none.such';
101
102-- Superuser should be able to ALTER SYSTEM SET a non-existent custom GUC.
103ALTER SYSTEM SET none.such = 'whiz bang';
104-- None of the above should have created a placeholder GUC for none.such.
105SHOW none.such;  -- error
106-- However, if we reload ...
107SELECT pg_reload_conf();
108-- and start a new session to avoid race condition ...
109\c -
110SET SESSION AUTHORIZATION regress_admin;
111-- then it should be there.
112SHOW none.such;

114-- Can't grant on a non-existent core GUC.

231-- Try that again, but use "drop owned by" instead of "revoke"
232CREATE ROLE regress_host_resource_admin NOSUPERUSER;
233SET SESSION AUTHORIZATION regress_host_resource_admin;
234ALTER SYSTEM SET autovacuum_work_mem = 32;  -- fail, privileges not yet granted
235SET SESSION AUTHORIZATION regress_admin;
236GRANT SET, ALTER SYSTEM ON PARAMETER
237    autovacuum_work_mem, hash_mem_multiplier, max_stack_depth,
238    shared_buffers, temp_file_limit, work_mem
239TO regress_host_resource_admin;
240DROP ROLE regress_host_resource_admin;  -- fail, privileges remain
241DROP OWNED BY regress_host_resource_admin RESTRICT; -- cascade should not be needed
242SET SESSION AUTHORIZATION regress_host_resource_admin;
243ALTER SYSTEM SET autovacuum_work_mem = 32;  -- fail, "drop owned" has dropped privileges
244SET SESSION AUTHORIZATION regress_admin;
245DROP ROLE regress_host_resource_admin;  -- ok

247-- Check that "reassign owned" doesn't affect privileges
248CREATE ROLE regress_host_resource_admin NOSUPERUSER;
249CREATE ROLE regress_host_resource_newadmin NOSUPERUSER;
250GRANT SET, ALTER SYSTEM ON PARAMETER
251    autovacuum_work_mem, hash_mem_multiplier, max_stack_depth,
252    shared_buffers, temp_file_limit, work_mem
253TO regress_host_resource_admin;
254REASSIGN OWNED BY regress_host_resource_admin TO regress_host_resource_newadmin;
255SET SESSION AUTHORIZATION regress_host_resource_admin;
256ALTER SYSTEM SET autovacuum_work_mem = 32;  -- ok, "reassign owned" did not change privileges
257ALTER SYSTEM RESET autovacuum_work_mem;  -- ok
258SET SESSION AUTHORIZATION regress_admin;
259DROP ROLE regress_host_resource_admin;  -- fail, privileges remain
260DROP ROLE regress_host_resource_newadmin;  -- ok, nothing was transferred
261-- Use "drop owned by" so we can drop the role
262DROP OWNED BY regress_host_resource_admin;  -- ok
263DROP ROLE regress_host_resource_admin;  -- ok
