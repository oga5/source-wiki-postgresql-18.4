src/test/modules/unsafe_tests/sql/alter_system_table.sql
1--
2-- Tests for things affected by allow_system_table_mods
3--
4-- We run the same set of commands once with allow_system_table_mods
5-- off and then again with on.
6--
7-- The "on" tests should where possible be wrapped in BEGIN/ROLLBACK
8-- blocks so as to not leave a mess around.
9
10CREATE USER regress_user_ast;
11
12SET allow_system_table_mods = off;
13
14-- create new table in pg_catalog
15CREATE TABLE pg_catalog.test (a int);
16
17-- anyarray column
18CREATE TABLE t1x (a int, b anyarray);
19
20-- index on system catalog
21ALTER TABLE pg_namespace ADD CONSTRAINT foo UNIQUE USING INDEX pg_namespace_nspname_index;
22
23-- write to system catalog table as superuser
24-- (allowed even without allow_system_table_mods)
25INSERT INTO pg_description (objoid, classoid, objsubid, description) VALUES (0, 0, 0, 'foo');
26
27-- write to system catalog table as normal user
28GRANT INSERT ON pg_description TO regress_user_ast;
29SET ROLE regress_user_ast;
30INSERT INTO pg_description (objoid, classoid, objsubid, description) VALUES (0, 0, 1, 'foo');
31RESET ROLE;
32
33-- policy on system catalog
34CREATE POLICY foo ON pg_description FOR SELECT USING (description NOT LIKE 'secret%');
35
36-- reserved schema name
37CREATE SCHEMA pg_foo;
38
39-- drop system table
40DROP TABLE pg_description;
41
42-- truncate of system table
43TRUNCATE pg_description;
44
45-- rename column of system table
46ALTER TABLE pg_description RENAME COLUMN description TO comment;
47
48-- ATSimplePermissions()
49ALTER TABLE pg_description ALTER COLUMN description SET NOT NULL;
50
51-- SET STATISTICS
52ALTER TABLE pg_description ALTER COLUMN description SET STATISTICS -1;
53
54-- foreign key referencing catalog
55CREATE TABLE foo (a oid, b oid, c int, FOREIGN KEY (a, b, c) REFERENCES pg_description);
56
57-- RangeVarCallbackOwnsRelation()
58CREATE INDEX pg_description_test_index ON pg_description (description);
59
60-- RangeVarCallbackForAlterRelation()
61ALTER TABLE pg_description RENAME TO pg_comment;
62ALTER TABLE pg_description SET SCHEMA public;
63
64-- reserved tablespace name
65CREATE TABLESPACE pg_foo LOCATION '/no/such/location';
66
67-- triggers
68CREATE FUNCTION tf1() RETURNS trigger
69LANGUAGE plpgsql
70AS $$
71BEGIN
72  RETURN NULL;
73END $$;
74
75CREATE TRIGGER t1 BEFORE INSERT ON pg_description EXECUTE FUNCTION tf1();
76ALTER TRIGGER t1 ON pg_description RENAME TO t2;
77--DROP TRIGGER t2 ON pg_description;
78
79-- rules
80CREATE RULE r1 AS ON INSERT TO pg_description DO INSTEAD NOTHING;
81ALTER RULE r1 ON pg_description RENAME TO r2;
82-- now make one to test dropping:
83SET allow_system_table_mods TO on;
84CREATE RULE r2 AS ON INSERT TO pg_description DO INSTEAD NOTHING;
85RESET allow_system_table_mods;
86DROP RULE r2 ON pg_description;
87-- cleanup:
88SET allow_system_table_mods TO on;
89DROP RULE r2 ON pg_description;
90RESET allow_system_table_mods;
91
92-- Reloptions on TOAST tables
93ALTER TABLE pg_toast.pg_toast_2615 SET (fillfactor = '90');
94
95SET allow_system_table_mods = on;
96
97-- create new table in pg_catalog
98BEGIN;
99CREATE TABLE pg_catalog.test (a int);
100ROLLBACK;
101
102-- anyarray column
103BEGIN;
104CREATE TABLE t1 (a int, b anyarray);
105ROLLBACK;
106
107-- index on system catalog
108BEGIN;
109ALTER TABLE pg_namespace ADD CONSTRAINT foo UNIQUE USING INDEX pg_namespace_nspname_index;
110ROLLBACK;
111
112-- write to system catalog table as superuser
113BEGIN;
114INSERT INTO pg_description (objoid, classoid, objsubid, description) VALUES (0, 0, 2, 'foo');
115ROLLBACK;
116
117-- write to system catalog table as normal user
118-- (not allowed)
119SET ROLE regress_user_ast;
120INSERT INTO pg_description (objoid, classoid, objsubid, description) VALUES (0, 0, 3, 'foo');
121RESET ROLE;
122
123-- policy on system catalog
124BEGIN;
125CREATE POLICY foo ON pg_description FOR SELECT USING (description NOT LIKE 'secret%');
126ROLLBACK;
127
128-- reserved schema name
129BEGIN;
130CREATE SCHEMA pg_foo;
131ROLLBACK;
132
133-- drop system table
134-- (This will fail anyway because it's pinned.)
135BEGIN;
136DROP TABLE pg_description;
137ROLLBACK;
138
139-- truncate of system table
140BEGIN;
141TRUNCATE pg_description;
142ROLLBACK;
143
144-- rename column of system table
145BEGIN;
146ALTER TABLE pg_description RENAME COLUMN description TO comment;
147ROLLBACK;
148
149-- ATSimplePermissions()
150BEGIN;
151ALTER TABLE pg_description ALTER COLUMN description SET NOT NULL;
152ROLLBACK;
153
154-- SET STATISTICS
155BEGIN;
156ALTER TABLE pg_description ALTER COLUMN description SET STATISTICS -1;
157ROLLBACK;
158
159-- foreign key referencing catalog
160BEGIN;
161CREATE TABLE foo (a oid, b oid, c int, FOREIGN KEY (a, b, c) REFERENCES pg_description);
162ROLLBACK;
163
164-- RangeVarCallbackOwnsRelation()
165BEGIN;
166CREATE INDEX pg_description_test_index ON pg_description (description);
167ROLLBACK;
168

175ROLLBACK;

177-- reserved tablespace name
178SET client_min_messages = error;  -- disable ENFORCE_REGRESSION_TEST_NAME_RESTRICTIONS warning
179CREATE TABLESPACE pg_foo LOCATION '/no/such/location';
180RESET client_min_messages;
181
182-- triggers
183CREATE TRIGGER t1 BEFORE INSERT ON pg_description EXECUTE FUNCTION tf1();
184ALTER TRIGGER t1 ON pg_description RENAME TO t2;
185DROP TRIGGER t2 ON pg_description;
186
187-- rules
188CREATE RULE r1 AS ON INSERT TO pg_description DO INSTEAD NOTHING;
189ALTER RULE r1 ON pg_description RENAME TO r2;
190DROP RULE r2 ON pg_description;
191
192-- Reloptions on TOAST tables
193ALTER TABLE pg_toast.pg_toast_2615 SET (fillfactor = '90');
194
195-- cleanup
196REVOKE ALL ON pg_description FROM regress_user_ast;
197DROP USER regress_user_ast;
198DROP FUNCTION tf1;
