8-- all combinations of four boolean values
9create table booleans as
10select
11  case i%3 when 0 then true when 1 then false else null end as x,
12  case (i/3)%3 when 0 then true when 1 then false else null end as y,
13  case (i/9)%3 when 0 then true when 1 then false else null end as z,
14  case (i/27)%3 when 0 then true when 1 then false else null end as w
15from generate_series(0, 3*3*3*3-1) i;
16
17-- all combinations of two integers 0..9, plus null
18create table integers as
19select
20  case i%11 when 10 then null else i%11 end as x,
21  case (i/11)%11 when 10 then null else (i/11)%11 end as y
22from generate_series(0, 11*11-1) i;
23
24-- and a simple strict function that's opaque to the optimizer
25create function strictf(bool, bool) returns bool
26language plpgsql as $$begin return $1 and not $2; end$$ strict;
27
28-- a simple function to make arrays opaque to the optimizer
29create function opaque_array(int[]) returns int[]
30language plpgsql as $$begin return $1; end$$ strict;
