35static TidStore *tidstore = NULL;
36static size_t tidstore_empty_size;

38/* array for verification of some tests */
39typedef struct ItemArray
40{
41	ItemPointerData *insert_tids;
42	ItemPointerData *lookup_tids;
43	ItemPointerData *iter_tids;
44	int			max_tids;
45	int			num_tids;
46} ItemArray;
47
48static ItemArray items;
49
50/* comparator routine for ItemPointer */
51static int
52itemptr_cmp(const void *left, const void *right)
53{
54	BlockNumber lblk,
55				rblk;
56	OffsetNumber loff,
57				roff;
58
59	lblk = ItemPointerGetBlockNumber((ItemPointer) left);
60	rblk = ItemPointerGetBlockNumber((ItemPointer) right);
61
62	if (lblk < rblk)
63		return -1;
64	if (lblk > rblk)
65		return 1;
66
67	loff = ItemPointerGetOffsetNumber((ItemPointer) left);
68	roff = ItemPointerGetOffsetNumber((ItemPointer) right);
69
70	if (loff < roff)
71		return -1;
72	if (loff > roff)
73		return 1;
74
75	return 0;
76}
77
78static int
79offsetnumber_cmp(const void *a, const void *b)
80{
81	OffsetNumber l = *(const OffsetNumber *) a;
82	OffsetNumber r = *(const OffsetNumber *) b;
83
84	if (l < r)
85		return -1;
86	else if (l > r)
87		return 1;
88	return 0;
89}

91/*
92 * Create a TidStore. If shared is false, the tidstore is created
93 * on TopMemoryContext, otherwise on DSA. Although the tidstore
94 * is created on DSA, only the same process can subsequently use
95 * the tidstore. The tidstore handle is not shared anywhere.
96*/
97Datum
98test_create(PG_FUNCTION_ARGS)
99{
100	bool		shared = PG_GETARG_BOOL(0);
101	MemoryContext old_ctx;
102
103	/* doesn't really matter, since it's just a hint */
104	size_t		tidstore_max_size = 2 * 1024 * 1024;
105	size_t		array_init_size = 1024;
106
107	Assert(tidstore == NULL);
108
109	/*
110	 * Create the TidStore on TopMemoryContext so that the same process use it
111	 * for subsequent tests.
112	 */
113	old_ctx = MemoryContextSwitchTo(TopMemoryContext);
114
115	if (shared)
116	{
117		int			tranche_id;
118
119		tranche_id = LWLockNewTrancheId();
120		LWLockRegisterTranche(tranche_id, "test_tidstore");
121
122		tidstore = TidStoreCreateShared(tidstore_max_size, tranche_id);
123
124		/*
125		 * Remain attached until end of backend or explicitly detached so that
126		 * the same process use the tidstore for subsequent tests.
127		 */
128		dsa_pin_mapping(TidStoreGetDSA(tidstore));
129	}
130	else
131		/* VACUUM uses insert only, so we test the other option. */
132		tidstore = TidStoreCreateLocal(tidstore_max_size, false);
133
134	tidstore_empty_size = TidStoreMemoryUsage(tidstore);
135
136	items.num_tids = 0;
137	items.max_tids = array_init_size / sizeof(ItemPointerData);
138	items.insert_tids = (ItemPointerData *) palloc0(array_init_size);
139	items.lookup_tids = (ItemPointerData *) palloc0(array_init_size);
140	items.iter_tids = (ItemPointerData *) palloc0(array_init_size);
141
142	MemoryContextSwitchTo(old_ctx);
143
144	PG_RETURN_VOID();
145}

147static void
148sanity_check_array(ArrayType *ta)
149{
150	if (ARR_HASNULL(ta) && array_contains_nulls(ta))
151		ereport(ERROR,
152				(errcode(ERRCODE_NULL_VALUE_NOT_ALLOWED),
153				 errmsg("array must not contain nulls")));
154
155	if (ARR_NDIM(ta) > 1)
156		ereport(ERROR,
157				(errcode(ERRCODE_DATA_EXCEPTION),
158				 errmsg("argument must be empty or one-dimensional array")));
159}

161static void
162check_tidstore_available(void)
163{
164	if (tidstore == NULL)
165		elog(ERROR, "tidstore is not created");
166}

168static void
169purge_from_verification_array(BlockNumber blkno)
170{
171	int			dst = 0;
172
173	for (int src = 0; src < items.num_tids; src++)
174		if (ItemPointerGetBlockNumber(&items.insert_tids[src]) != blkno)
175			items.insert_tids[dst++] = items.insert_tids[src];
176	items.num_tids = dst;
177}

180/* Set the given block and offsets pairs */
181Datum
182do_set_block_offsets(PG_FUNCTION_ARGS)
183{
184	BlockNumber blkno = PG_GETARG_INT64(0);
185	ArrayType  *ta = PG_GETARG_ARRAYTYPE_P_COPY(1);
186	OffsetNumber *offs;
187	int			noffs;
188
189	check_tidstore_available();
190	sanity_check_array(ta);
191
192	noffs = ArrayGetNItems(ARR_NDIM(ta), ARR_DIMS(ta));
193	offs = ((OffsetNumber *) ARR_DATA_PTR(ta));
194
195	/* TidStoreSetBlockOffsets() requires offsets to be strictly ascending. */
196	qsort(offs, noffs, sizeof(OffsetNumber), offsetnumber_cmp);
197
198	/* Set TIDs in the store */
199	TidStoreLockExclusive(tidstore);
200	TidStoreSetBlockOffsets(tidstore, blkno, offs, noffs);
201	TidStoreUnlock(tidstore);
202
203	/* Remove the existing items of blkno from the verification array */
204	purge_from_verification_array(blkno);
205
206	/* Set TIDs in verification array */
207	for (int i = 0; i < noffs; i++)
208	{
209		ItemPointer tid;
210		int			idx = items.num_tids + i;
211
212		/* Enlarge the TID arrays if necessary */
213		if (idx >= items.max_tids)
214		{
215			items.max_tids *= 2;
216			items.insert_tids = repalloc(items.insert_tids, sizeof(ItemPointerData) * items.max_tids);
217			items.lookup_tids = repalloc(items.lookup_tids, sizeof(ItemPointerData) * items.max_tids);
218			items.iter_tids = repalloc(items.iter_tids, sizeof(ItemPointerData) * items.max_tids);
219		}
220
221		tid = &(items.insert_tids[idx]);
222		ItemPointerSet(tid, blkno, offs[i]);
223	}
224
225	/* Update statistics */
226	items.num_tids += noffs;
227
228	PG_RETURN_INT64(blkno);
229}

231/*
232 * Verify TIDs in store against the array.
233 */
234Datum
235check_set_block_offsets(PG_FUNCTION_ARGS)
236{
237	TidStoreIter *iter;
238	TidStoreIterResult *iter_result;
239	int			num_iter_tids = 0;
240	int			num_lookup_tids = 0;
241	BlockNumber prevblkno = 0;
242
243	check_tidstore_available();
244
245	/* lookup each member in the verification array */
246	for (int i = 0; i < items.num_tids; i++)
247		if (!TidStoreIsMember(tidstore, &items.insert_tids[i]))
248			elog(ERROR, "missing TID with block %u, offset %u",
249				 ItemPointerGetBlockNumber(&items.insert_tids[i]),
250				 ItemPointerGetOffsetNumber(&items.insert_tids[i]));
251
252	/*
253	 * Lookup all possible TIDs for each distinct block in the verification
254	 * array and save successful lookups in the lookup array.
255	 */
256
257	for (int i = 0; i < items.num_tids; i++)
258	{
259		BlockNumber blkno = ItemPointerGetBlockNumber(&items.insert_tids[i]);
260
261		if (i > 0 && blkno == prevblkno)
262			continue;
263
264		for (OffsetNumber offset = FirstOffsetNumber; offset < MaxOffsetNumber; offset++)
265		{
266			ItemPointerData tid;
267
268			ItemPointerSet(&tid, blkno, offset);
269
270			TidStoreLockShare(tidstore);
271			if (TidStoreIsMember(tidstore, &tid))
272				ItemPointerSet(&items.lookup_tids[num_lookup_tids++], blkno, offset);
273			TidStoreUnlock(tidstore);
274		}
275
276		prevblkno = blkno;
277	}
278
279	/* Collect TIDs stored in the tidstore, in order */
280
281	TidStoreLockShare(tidstore);
282	iter = TidStoreBeginIterate(tidstore);
283	while ((iter_result = TidStoreIterateNext(iter)) != NULL)
284	{
285		OffsetNumber offsets[MaxOffsetNumber];
286		int			num_offsets;
287
288		num_offsets = TidStoreGetBlockOffsets(iter_result, offsets, lengthof(offsets));
289		Assert(num_offsets <= lengthof(offsets));
290		for (int i = 0; i < num_offsets; i++)
291			ItemPointerSet(&(items.iter_tids[num_iter_tids++]), iter_result->blkno,
292						   offsets[i]);
293	}
294	TidStoreEndIterate(iter);
295	TidStoreUnlock(tidstore);
296
297	/*
298	 * Sort verification and lookup arrays and test that all arrays are the
299	 * same.
300	 */
301
302	if (num_lookup_tids != items.num_tids)
303		elog(ERROR, "should have %d TIDs, have %d", items.num_tids, num_lookup_tids);
304	if (num_iter_tids != items.num_tids)
305		elog(ERROR, "should have %d TIDs, have %d", items.num_tids, num_iter_tids);
306
307	qsort(items.insert_tids, items.num_tids, sizeof(ItemPointerData), itemptr_cmp);
308	qsort(items.lookup_tids, items.num_tids, sizeof(ItemPointerData), itemptr_cmp);
309	for (int i = 0; i < items.num_tids; i++)
310	{
311		if (itemptr_cmp(&items.insert_tids[i], &items.iter_tids[i]) != 0)
312			elog(ERROR, "TID iter array doesn't match verification array, got (%u,%u) expected (%u,%u)",
313				 ItemPointerGetBlockNumber(&items.iter_tids[i]),
314				 ItemPointerGetOffsetNumber(&items.iter_tids[i]),
315				 ItemPointerGetBlockNumber(&items.insert_tids[i]),
316				 ItemPointerGetOffsetNumber(&items.insert_tids[i]));
317		if (itemptr_cmp(&items.insert_tids[i], &items.lookup_tids[i]) != 0)
318			elog(ERROR, "TID lookup array doesn't match verification array, got (%u,%u) expected (%u,%u)",
319				 ItemPointerGetBlockNumber(&items.lookup_tids[i]),
320				 ItemPointerGetOffsetNumber(&items.lookup_tids[i]),
321				 ItemPointerGetBlockNumber(&items.insert_tids[i]),
322				 ItemPointerGetOffsetNumber(&items.insert_tids[i]));
323	}
324
325	PG_RETURN_VOID();
326}

328/*
329 * In real world use, we care if the memory usage is greater than
330 * some configured limit. Here we just want to verify that
331 * TidStoreMemoryUsage is not broken.
332 */
333Datum
334test_is_full(PG_FUNCTION_ARGS)
335{
336	bool		is_full;
337
338	check_tidstore_available();
339
340	is_full = (TidStoreMemoryUsage(tidstore) > tidstore_empty_size);
341
342	PG_RETURN_BOOL(is_full);
343}
344
345/* Free the tidstore */
346Datum
347test_destroy(PG_FUNCTION_ARGS)
348{
349	check_tidstore_available();
350
351	TidStoreDestroy(tidstore);
352	tidstore = NULL;
353	items.num_tids = 0;
354	pfree(items.insert_tids);
355	pfree(items.lookup_tids);
356	pfree(items.iter_tids);
357
358	PG_RETURN_VOID();
