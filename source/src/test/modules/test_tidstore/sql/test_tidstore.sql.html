src/test/modules/test_tidstore/sql/test_tidstore.sql
1CREATE EXTENSION test_tidstore;
2
3-- To hide the output of do_set_block_offsets()
4CREATE TEMP TABLE hideblocks(blockno bigint);
5
6-- Constant values used in the tests.
7\set maxblkno 4294967295
8-- The maximum number of heap tuples (MaxHeapTuplesPerPage) in 8kB block is 291.
9-- We use a higher number to test tidstore.
10\set maxoffset 512
11
12SELECT test_create(false);
13
14-- Test on empty tidstore.
15SELECT test_is_full();
16SELECT check_set_block_offsets();
17
18-- Add TIDs.
19INSERT INTO hideblocks (blockno)
20SELECT do_set_block_offsets(blk, array_agg(off)::int2[])
21  FROM
22    (VALUES (0), (1), (:maxblkno / 2), (:maxblkno - 1), (:maxblkno)) AS blocks(blk),
23    (VALUES (1), (2), (:maxoffset / 2), (:maxoffset - 1), (:maxoffset)) AS offsets(off)
24  GROUP BY blk;
25
26-- Test offsets embedded in the bitmap header.
27SELECT do_set_block_offsets(501, array[greatest((random() * :maxoffset)::int, 1)]::int2[]);
28SELECT do_set_block_offsets(502, array_agg(DISTINCT greatest((random() * :maxoffset)::int, 1))::int2[])
29  FROM generate_series(1, 3);
30
31-- Add enough TIDs to cause the store to appear "full", compared
32-- to the allocated memory it started out with. This is easier
33-- with memory contexts in local memory.
34INSERT INTO hideblocks (blockno)
35SELECT do_set_block_offsets(blk, ARRAY[1,31,32,63,64,200]::int2[])
36  FROM generate_series(1000, 2000, 1) blk;
37
38-- Zero offset not allowed
39SELECT do_set_block_offsets(1, ARRAY[0]::int2[]);
40
41-- Check TIDs we've added to the store.
42SELECT check_set_block_offsets();
43
44SELECT test_is_full();
45
46-- Re-create the TID store for randommized tests.
47SELECT test_destroy();
48
49
50-- Test replacements crossing RT_CHILDPTR_IS_VALUE in both directions
51SELECT test_create(false);
52SELECT do_set_block_offsets(1, array[1]::int2[]); SELECT check_set_block_offsets();
53SELECT do_set_block_offsets(1, array[1,2]::int2[]); SELECT check_set_block_offsets();
54SELECT do_set_block_offsets(1, array[1,2,3]::int2[]); SELECT check_set_block_offsets();
55SELECT do_set_block_offsets(1, array[1,2,3,4]::int2[]); SELECT check_set_block_offsets();
56SELECT do_set_block_offsets(1, array[1,2,3,4,100]::int2[]); SELECT check_set_block_offsets();
57SELECT do_set_block_offsets(1, array[1,2,3,4]::int2[]); SELECT check_set_block_offsets();
58SELECT do_set_block_offsets(1, array[1,2,3]::int2[]); SELECT check_set_block_offsets();
59SELECT do_set_block_offsets(1, array[1,2]::int2[]); SELECT check_set_block_offsets();
60SELECT do_set_block_offsets(1, array[1]::int2[]); SELECT check_set_block_offsets();
61SELECT test_destroy();
62
63
64-- Use shared memory this time. We can't do that in test_radixtree.sql,
65-- because unused static functions would raise warnings there.
66SELECT test_create(true);
67
68-- Test offsets embedded in the bitmap header.
69SELECT do_set_block_offsets(501, array[greatest((random() * :maxoffset)::int, 1)]::int2[]);
70SELECT do_set_block_offsets(502, array_agg(DISTINCT greatest((random() * :maxoffset)::int, 1))::int2[])
71  FROM generate_series(1, 3);
72
73-- Random TIDs test. The offset numbers are randomized and must be
74-- unique and ordered.
75INSERT INTO hideblocks (blockno)
76SELECT do_set_block_offsets(blkno, array_agg(DISTINCT greatest((random() * :maxoffset)::int, 1))::int2[])
77  FROM generate_series(1, 100) num_offsets,
78  generate_series(1000, 1100, 1) blkno
79GROUP BY blkno;
80
81-- Check TIDs we've added to the store.
82SELECT check_set_block_offsets();
83
84-- cleanup
85SELECT test_destroy();
86DROP TABLE hideblocks;
