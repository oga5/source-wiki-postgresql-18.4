1-- Tests with various custom types
2
3CREATE EXTENSION test_custom_types;
4
5-- Test comparison functions
6SELECT '42'::int_custom = '42'::int_custom AS eq_test;
7SELECT '42'::int_custom <> '42'::int_custom AS nt_test;
8SELECT '42'::int_custom < '100'::int_custom AS lt_test;
9SELECT '100'::int_custom > '42'::int_custom AS gt_test;
10SELECT '42'::int_custom <= '100'::int_custom AS le_test;
11SELECT '100'::int_custom >= '42'::int_custom AS ge_test;
12
13-- Create a table with the int_custom type
14CREATE TABLE test_table (
15    id int,
16    data int_custom
17);
18INSERT INTO test_table VALUES (1, '42'), (2, '100'), (3, '200');
19
20-- Verify data was inserted correctly
21SELECT * FROM test_table ORDER BY id;
22
23-- Dummy function used for expression evaluations.
24-- Note that this function does not use a SQL-standard function body on
25-- purpose, so as external statistics can be loaded from it.
26CREATE OR REPLACE FUNCTION func_int_custom (p_value int_custom)
27  RETURNS int_custom LANGUAGE plpgsql AS $$
28  BEGIN
29    RETURN p_value;
30  END; $$;
31
32-- Switch type to use typanalyze function that always returns false.
33ALTER TYPE int_custom SET (ANALYZE = int_custom_typanalyze_false);
34
35-- Extended statistics with an attribute that cannot be analyzed.
36-- This includes all statistics kinds.
37CREATE STATISTICS test_stats ON data, id FROM test_table;
38-- Computation of the stats fails, no data generated.
39ANALYZE test_table;
40SELECT stxname, stxdexpr IS NULL as expr_stats_is_null
41  FROM pg_statistic_ext s
42  LEFT JOIN pg_statistic_ext_data d ON s.oid = d.stxoid
43  WHERE stxname = 'test_stats';
44DROP STATISTICS test_stats;
45
46-- Extended statistics with an expression that cannot be analyzed.
47CREATE STATISTICS test_stats ON func_int_custom(data), (id) FROM test_table;
48-- Computation of the stats fails, no data generated.
49ANALYZE test_table;
50SELECT stxname, stxdexpr IS NULL as expr_stats_is_null
51  FROM pg_statistic_ext s
52  LEFT JOIN pg_statistic_ext_data d ON s.oid = d.stxoid
53  WHERE stxname = 'test_stats';
54DROP STATISTICS test_stats;
55-- There should be no data stored for the expression.
56SELECT tablename,
57  statistics_name,
58  null_frac,
59  avg_width
60  FROM pg_stats_ext_exprs WHERE statistics_name = 'test_stats' \gx
61
62-- Switch type to use typanalyze function that generates invalid data.
63ALTER TYPE int_custom SET (ANALYZE = int_custom_typanalyze_invalid);
64
65-- Extended statistics with an attribute that generates invalid stats.
66CREATE STATISTICS test_stats ON data, id FROM test_table;
67-- Computation of the stats fails, no data generated.
68ANALYZE test_table;
69SELECT stxname, stxdexpr IS NULL as expr_stats_is_null

74
75-- Extended statistics with an expression that generates invalid data.
76CREATE STATISTICS test_stats ON func_int_custom(data), (id) FROM test_table;
77-- Computation of the stats fails, some data generated.
78ANALYZE test_table;
79SELECT stxname, stxdexpr IS NULL as expr_stats_is_null
80  FROM pg_statistic_ext s
81  LEFT JOIN pg_statistic_ext_data d ON s.oid = d.stxoid
82  WHERE stxname = 'test_stats';
83-- There should be some data stored for the expression, stored as NULL.
84SELECT tablename,
85  statistics_name,
86  null_frac,
87  avg_width,
88  n_distinct,
89  most_common_vals,
90  most_common_freqs,
91  histogram_bounds,
92  correlation,
93  most_common_elems,
94  most_common_elem_freqs,
95  elem_count_histogram
96  FROM pg_stats_ext_exprs WHERE statistics_name = 'test_stats' \gx
97-- Run a query able to load the extended stats, including the NULL data.
98SELECT COUNT(*) FROM test_table GROUP BY (func_int_custom(data));
99DROP STATISTICS test_stats;
