1--
2-- The test uses a GIN index over int[].  The table contains arrays
3-- with integers from 1 to :next_i.  Each integer occurs exactly once,
4-- no gaps or duplicates, although the index does contain some
5-- duplicate elements because some of the inserting transactions are
6-- rolled back during the test. The exact contents of the table depend
7-- on the physical layout of the index, which in turn depends at least
8-- on the block size, so instead of check for the exact contents, we
9-- check those invariants.  :next_i psql variable is maintained at all
10-- times to hold the last inserted integer + 1.
11--
12-- This uses injection points to cause errors that leave some page
13-- splits in "incomplete" state
14create extension injection_points;
15-- Make all injection points local to this process, for concurrency.
16SELECT injection_points_set_local();

31--
32-- First create the test table and some helper functions
33--
34create table gin_incomplete_splits(i int4[]) with (autovacuum_enabled = off);
35create index on gin_incomplete_splits using gin (i) with (fastupdate = off);
36-- Creates an array with all integers from $1 (inclusive) $2 (exclusive)
37create function range_array(int, int) returns int[] language sql immutable as $$
38  select array_agg(g) from generate_series($1, $2 - 1) g
39$$;
40-- Inserts an array with 'n' rows to the test table. Pass :next_i as
41-- the first argument, returns the new value for :next_i.
42create function insert_n(first_i int, n int) returns int language plpgsql as $$
43begin
44  insert into gin_incomplete_splits values (range_array(first_i, first_i+n));
45  return first_i + n;
46end;
47$$;
48-- Inserts to the table until an insert fails. Like insert_n(), returns the
49-- new value for :next_i.
50create function insert_until_fail(next_i int, step int default 1) returns int language plpgsql as $$
51declare
52  i integer;
53begin
54  -- Insert arrays with 'step' elements each, until an error occurs.
55  i := 0;
56  loop
57    begin
58      select insert_n(next_i, step) into next_i;
59    exception when others then
60      raise notice 'failed with: %', sqlerrm;
61      exit;
62    end;
63
64    -- The caller is expected to set an injection point that eventually
65    -- causes an error. But bail out if still no error after 10000
66    -- attempts, so that we don't get stuck in an infinite loop.
67    i := i + 1;
68    if i = 10000 then
69      raise 'no error on inserts after % iterations', i;
70    end if;
71  end loop;
72
73  return next_i;
74end;
75$$;
76-- Check the invariants.
77create function verify(next_i int) returns bool language plpgsql as $$
78declare
79  a integer[];
80  elem integer;
81  c integer;
82begin
83  -- Perform a scan over the trailing part of the index, where the
84  -- possible incomplete splits are. (We don't check the whole table,
85  -- because that'd be pretty slow.)
86  c := 0;
87  -- Find all arrays that overlap with the last 200 inserted integers. Or
88  -- the next 100, which shouldn't exist.
89  for a in select i from gin_incomplete_splits where i && range_array(next_i - 200, next_i + 100)
90  loop
91    -- count all elements in those arrays in the window.
92    foreach elem in ARRAY a loop
93      if elem >= next_i then
94        raise 'unexpected value % in array', elem;
95      end if;
96      if elem >= next_i - 200 then
97        c := c + 1;
98      end if;
99    end loop;
100  end loop;
101  if c <> 200 then
102    raise 'unexpected count % ', c;
103  end if;
104
105  return true;
106end;
107$$;

109select insert_n(1, 1000) as next_i
110\gset
