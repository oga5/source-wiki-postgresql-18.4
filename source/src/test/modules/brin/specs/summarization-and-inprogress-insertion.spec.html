src/test/modules/brin/specs/summarization-and-inprogress-insertion.spec
1# This test verifies that values inserted in transactions still in progress
2# are considered during concurrent range summarization (either using the
3# brin_summarize_new_values function or regular VACUUM).
4
5setup
6{
7    CREATE TABLE brin_iso (
8        value int
9    ) WITH (fillfactor=10);
10    CREATE INDEX brinidx ON brin_iso USING brin (value) WITH (pages_per_range=1);
11    -- this fills the first page
12    INSERT INTO brin_iso VALUES (NULL);
13    DO $$
14    DECLARE curtid tid;
15    BEGIN
16      LOOP
17        INSERT INTO brin_iso VALUES (1) RETURNING ctid INTO curtid;
18        EXIT WHEN curtid > tid '(1, 0)';
19      END LOOP;
20    END;
21    $$;
22    CREATE EXTENSION IF NOT EXISTS pageinspect;
23}
24
25teardown
26{
27    DROP TABLE brin_iso;
28}
29
30session "s1"
31step "s1b"		{ BEGIN ISOLATION LEVEL REPEATABLE READ; }
32step "s1i"		{ INSERT INTO brin_iso VALUES (1000); }
33step "s1c"		{ COMMIT; }
34
35session "s2"
36step "s2b"		{ BEGIN ISOLATION LEVEL REPEATABLE READ; SELECT 1; }
37step "s2summ"	{ SELECT brin_summarize_new_values('brinidx'::regclass); }
38step "s2c"		{ COMMIT; }
39
40step "s2vacuum"	{ VACUUM brin_iso; }
41
42step "s2check"	{ SELECT * FROM brin_page_items(get_raw_page('brinidx', 2), 'brinidx'::regclass); }
43
44permutation "s2check" "s1b" "s2b" "s1i" "s2summ" "s1c" "s2c" "s2check"
45permutation "s2check" "s1b" "s1i" "s2vacuum" "s1c" "s2check"
