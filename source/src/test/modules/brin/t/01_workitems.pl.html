6use strict;
7use warnings FATAL => 'all';
8
9use PostgreSQL::Test::Utils;
10use Test::More;
11use PostgreSQL::Test::Cluster;
12
13my $node = PostgreSQL::Test::Cluster->new('tango');
14$node->init;
15$node->append_conf('postgresql.conf', 'autovacuum_naptime=1s');
16$node->start;
17
18$node->safe_psql('postgres', 'create extension pageinspect');
19
20# Create a table with an autosummarizing BRIN index
21$node->safe_psql(
22	'postgres',
23	'create table brin_wi (a int) with (fillfactor = 10);
24	 create index brin_wi_idx on brin_wi using brin (a) with (pages_per_range=1, autosummarize=on);
25	 '
26);
27# Another table with an index that requires a snapshot to run
28$node->safe_psql(
29	'postgres',
30	'create table journal (d timestamp) with (fillfactor = 10);
31	 create function packdate(d timestamp) returns text language plpgsql
32	   as $$ begin return to_char(d, \'yyyymm\'); end; $$
33	   returns null on null input immutable;
34	 create index brin_packdate_idx on journal using brin (packdate(d))
35	   with (autosummarize = on, pages_per_range = 1);
36	 '
37);
38
39my $count = $node->safe_psql('postgres',
40	"select count(*) from brin_page_items(get_raw_page('brin_wi_idx', 2), 'brin_wi_idx'::regclass)"
41);
42is($count, '1', "initial brin_wi_index index state is correct");
43$count = $node->safe_psql('postgres',
44	"select count(*) from brin_page_items(get_raw_page('brin_packdate_idx', 2), 'brin_packdate_idx'::regclass)"
45);
46is($count, '1', "initial brin_packdate_idx index state is correct");
47
48$node->safe_psql('postgres',
49	'insert into brin_wi select * from generate_series(1, 100)');
50$node->safe_psql('postgres',
51	"insert into journal select * from generate_series(timestamp '1976-08-01', '1976-10-28', '1 day')");
52
53$node->poll_query_until(
54	'postgres',
55	"select count(*) > 1 from brin_page_items(get_raw_page('brin_wi_idx', 2), 'brin_wi_idx'::regclass)",
56	't');
57
58$count = $node->safe_psql('postgres',
59	"select count(*) from brin_page_items(get_raw_page('brin_wi_idx', 2), 'brin_wi_idx'::regclass)
60	 where not placeholder;"
61);
62cmp_ok($count, '>', '1', "$count brin_wi_idx ranges got summarized");
63
64$node->poll_query_until(
65	'postgres',
66	"select count(*) > 1 from brin_page_items(get_raw_page('brin_packdate_idx', 2), 'brin_packdate_idx'::regclass)",
67	't');
68
69$count = $node->safe_psql('postgres',
70	"select count(*) from brin_page_items(get_raw_page('brin_packdate_idx', 2), 'brin_packdate_idx'::regclass)
71	 where not placeholder;"
72);
73cmp_ok($count, '>', '1', "$count brin_packdate_idx ranges got summarized");
74
75$node->stop;
76
77done_testing();
