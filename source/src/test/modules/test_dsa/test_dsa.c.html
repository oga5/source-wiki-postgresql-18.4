24/* Test basic DSA functionality */
25PG_FUNCTION_INFO_V1(test_dsa_basic);
26Datum
27test_dsa_basic(PG_FUNCTION_ARGS)
28{
29	int			tranche_id;
30	dsa_area   *a;
31	dsa_pointer p[100];
32
33	/* XXX: this tranche is leaked */
34	tranche_id = LWLockNewTrancheId();
35	LWLockRegisterTranche(tranche_id, "test_dsa");
36
37	a = dsa_create(tranche_id);
38	for (int i = 0; i < 100; i++)
39	{
40		p[i] = dsa_allocate(a, 1000);
41		snprintf(dsa_get_address(a, p[i]), 1000, "foobar%d", i);
42	}
43
44	for (int i = 0; i < 100; i++)
45	{
46		char		buf[100];
47
48		snprintf(buf, 100, "foobar%d", i);
49		if (strcmp(dsa_get_address(a, p[i]), buf) != 0)
50			elog(ERROR, "no match");
51	}
52
53	for (int i = 0; i < 100; i++)
54	{
55		dsa_free(a, p[i]);
56	}
57
58	dsa_detach(a);
59
60	PG_RETURN_VOID();
61}

63/* Test using DSA across different resource owners */
64PG_FUNCTION_INFO_V1(test_dsa_resowners);
65Datum
66test_dsa_resowners(PG_FUNCTION_ARGS)
67{
68	int			tranche_id;
69	dsa_area   *a;
70	dsa_pointer p[10000];
71	ResourceOwner oldowner;
72	ResourceOwner childowner;
73
74	/* XXX: this tranche is leaked */
75	tranche_id = LWLockNewTrancheId();
76	LWLockRegisterTranche(tranche_id, "test_dsa");
77
78	/* Create DSA in parent resource owner */
79	a = dsa_create(tranche_id);
80
81	/*
82	 * Switch to child resource owner, and do a bunch of allocations in the
83	 * DSA
84	 */
85	oldowner = CurrentResourceOwner;
86	childowner = ResourceOwnerCreate(oldowner, "test_dsa temp owner");
87	CurrentResourceOwner = childowner;
88
89	for (int i = 0; i < 10000; i++)
90	{
91		p[i] = dsa_allocate(a, 1000);
92		snprintf(dsa_get_address(a, p[i]), 1000, "foobar%d", i);
93	}
94
95	/* Also test freeing, by freeing some of the allocations. */
96	for (int i = 0; i < 500; i++)
97		dsa_free(a, p[i]);
98
99	/* Release the child resource owner */
100	CurrentResourceOwner = oldowner;
101	ResourceOwnerRelease(childowner,
102						 RESOURCE_RELEASE_BEFORE_LOCKS,
103						 true, false);
104	ResourceOwnerRelease(childowner,
105						 RESOURCE_RELEASE_LOCKS,
106						 true, false);
107	ResourceOwnerRelease(childowner,
108						 RESOURCE_RELEASE_AFTER_LOCKS,
109						 true, false);
110	ResourceOwnerDelete(childowner);
111
112	dsa_detach(a);
113
114	PG_RETURN_VOID();
115}

117/*
118 * test_dsa_allocate
119 *
120 * Test DSA allocation across a range of sizes to exercise the pagemap
121 * sizing logic in make_new_segment().  A fresh DSA is created for each
122 * iteration so that each allocation triggers a new segment creation,
123 * including the odd-sized segment path.
124 */
125PG_FUNCTION_INFO_V1(test_dsa_allocate);
126Datum
127test_dsa_allocate(PG_FUNCTION_ARGS)
128{
129	int			start_num_pages = PG_GETARG_INT32(0);
130	int			end_num_pages = PG_GETARG_INT32(1);
131	int			step = PG_GETARG_INT32(2);
132	size_t		usable_pages;
133	int			tranche_id;
134	dsa_area   *a;
135	dsa_pointer dp;
136
137	if (start_num_pages > end_num_pages)
138		elog(ERROR, "incorrect start and end parameters");
139
140	/* XXX: this tranche is leaked */
141	tranche_id = LWLockNewTrancheId();
142	LWLockRegisterTranche(tranche_id, "test_dsa");
143
144	for (usable_pages = start_num_pages; usable_pages < end_num_pages; usable_pages += step)
145	{
146		a = dsa_create(tranche_id);
147		dp = dsa_allocate(a, usable_pages * FPM_PAGE_SIZE);
148
149		dsa_free(a, dp);
150		dsa_detach(a);
151	}
152
153	PG_RETURN_VOID();
154}
