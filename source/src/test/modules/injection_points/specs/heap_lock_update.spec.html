src/test/modules/injection_points/specs/heap_lock_update.spec
1# Test race condition in tuple locking
2#
3# This is a reproducer for the bug reported at:
4# https://www.postgresql.org/message-id/CAOG%2BRQ74x0q%3DkgBBQ%3DmezuvOeZBfSxM1qu_o0V28bwDz3dHxLw%40mail.gmail.com
5#
6# The bug was that when following an update chain when locking tuples,
7# we sometimes failed to check that the xmin on the next tuple matched
8# the prior's xmax. If the updated tuple version was vacuumed away and
9# the slot was reused for an unrelated tuple, we'd incorrectly follow
10# and lock the unrelated tuple.
11
12
13# Set up a test table with enough rows to fill a page. We need the
14# UPDATE used in the test to put the new tuple on a different page,
15# because otherwise the VACUUM cannot remove the aborted tuple because
16# we hold a pin on the first page.
17#
18# The exact number of rows inserted doesn't otherwise matter, but we
19# arrange things in a deterministic fashion so that the last inserted
20# tuple goes to (1,1), and the updated and aborted tuple goes to
21# (1,2). That way we can just memorize those ctids in the expected
22# output, to verify that the test exercises the scenario we want.
23setup
24{
25	CREATE EXTENSION injection_points;
26
27	CREATE TABLE t (id int PRIMARY KEY);
28	do $$
29		DECLARE
30			i int;
31			tid tid;
32		BEGIN
33			FOR i IN 1..5000 LOOP
34				INSERT INTO t VALUES (i) RETURNING ctid INTO tid;
35				IF tid = '(1,1)' THEN
36					RETURN;
37				END IF;
38			END LOOP;
39			RAISE 'expected to insert tuple to (1,1)';
40	   END;
41   $$;
42}
43teardown
44{
45	DROP TABLE t;
46	DROP EXTENSION injection_points;
47}
48
49session s1
50step s1begin	{ BEGIN; }
51step s1update	{ UPDATE t SET id = 10000 WHERE id = 1 RETURNING ctid; }
52step s1abort	{ ABORT; }
53step vacuum		{ VACUUM t; }
54
55# Insert a new tuple, and update it.
56step reinsert	{
57	INSERT INTO t VALUES (10001) RETURNING ctid;
58	UPDATE t SET id = 10002 WHERE id = 10001 RETURNING ctid;
59}
60
61# Same as the 'reinsert' step, but for extra confusion, we also stamp
62# the original tuple with the same 'xmax' as the re-inserted one.
63step reinsert_and_lock {
64	BEGIN;
65	INSERT INTO t VALUES (10001) RETURNING ctid;
66	SELECT ctid, * FROM t WHERE id = 1 FOR UPDATE;
67	COMMIT;
68	UPDATE t SET id = 10002 WHERE id = 10001 returning ctid;
69}
70
71step wake	{
72	SELECT FROM injection_points_detach('heap_lock_updated_tuple');
73	SELECT FROM injection_points_wakeup('heap_lock_updated_tuple');
74}
75
76session s2
77setup	{
78	SELECT FROM injection_points_set_local();
79	SELECT FROM injection_points_attach('heap_lock_updated_tuple', 'wait');
80}
81step s2lock	{ select * from t where id = 1 for update; }
82
83permutation
84	# Begin transaction, update a row. Because of how we set up the
85	# test table, the updated tuple lands at (1,2)
86	s1begin
87	s1update
88
89	# While the updating transaction is open, start a new session that
90	# tries to lock the row. This blocks on the open transaction.
91	s2lock
92
93	# Abort the updating transaction. This unblocks session 2, but it
94	# will immediately hit the injection point and block on that.
95	s1abort
96	# Vacuum away the updated, aborted tuple.
97	vacuum
98
99	# Insert a new tuple. It lands at the same location where the
100	# updated tuple was.
101	reinsert
102
103	# Let the locking transaction continue. It should lock the
104	# original tuple, ignoring the re-inserted tuple.
105	wake(s2lock)
106
107# Variant where the re-inserted tuple is also locked by the inserting
108# transaction. This failed an earlier version of the fix during
109# development.
110permutation
111	s1begin
112	s1update
113	s2lock
114	s1abort
115	vacuum
116	reinsert_and_lock
117	wake(s2lock)
