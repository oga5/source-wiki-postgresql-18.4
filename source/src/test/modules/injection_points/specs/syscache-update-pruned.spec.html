src/test/modules/injection_points/specs/syscache-update-pruned.spec
1# Test race conditions involving:
2# - s1: heap_update($FROM_SYSCACHE), without a snapshot or pin
3# - s2: ALTER TABLE making $FROM_SYSCACHE a dead tuple
4# - s3: "VACUUM pg_class" making $FROM_SYSCACHE become LP_UNUSED
5
6# This is a derivative work of inplace.spec, which exercises the corresponding
7# race condition for inplace updates.
8
9# Despite local injection points, this is incompatible with runningcheck.
10# First, removable_cutoff() could move backward, per its header comment.
11# Second, other activity could trigger sinval queue overflow, negating our
12# efforts to delay inval.  Third, this deadlock emerges:
13#
14# - step at2 waits at an injection point, with interrupts held
15# - an unrelated backend waits for at2 to do PROCSIGNAL_BARRIER_SMGRRELEASE
16# - step waitprunable4 waits for the unrelated backend to release its xmin
17
18# The alternative expected output is for -DCATCACHE_FORCE_RELEASE, a setting
19# that thwarts testing the race conditions this spec seeks.
20
21
22# Need s2 to make a non-HOT update.  Otherwise, "VACUUM pg_class" would leave
23# an LP_REDIRECT that persists.  To get non-HOT, make rels so the pg_class row
24# for vactest.orig50 is on a filled page (assuming BLCKSZ=8192).  Just to save
25# on filesystem syscalls, use relkind=c for every other rel.
26setup
27{
28	CREATE EXTENSION injection_points;
29	CREATE SCHEMA vactest;
30	-- Ensure a leader RELOID catcache entry.  PARALLEL RESTRICTED since a
31	-- parallel worker running pg_relation_filenode() would lack that effect.
32	CREATE FUNCTION vactest.reloid_catcache_set(regclass) RETURNS int
33		LANGUAGE sql PARALLEL RESTRICTED
34		AS 'SELECT 0 FROM pg_relation_filenode($1)';
35	CREATE FUNCTION vactest.mkrels(text, int, int) RETURNS void
36		LANGUAGE plpgsql SET search_path = vactest AS $$
37	DECLARE
38		tname text;
39	BEGIN
40		FOR i in $2 .. $3 LOOP
41			tname := $1 || i;
42			EXECUTE FORMAT('CREATE TYPE ' || tname || ' AS ()');
43			RAISE DEBUG '% at %', tname, ctid
44				FROM pg_class WHERE oid = tname::regclass;
45		END LOOP;
46	END
47	$$;
48	CREATE PROCEDURE vactest.wait_prunable() LANGUAGE plpgsql AS $$
49	DECLARE
50		barrier xid8;
51		cutoff xid8;
52	BEGIN
53		barrier := pg_current_xact_id();
54		-- autovacuum worker RelationCacheInitializePhase3() or the
55		-- isolationtester control connection might hold a snapshot that
56		-- limits pruning.  Sleep until that clears.  See comments at
57		-- removable_cutoff() for why we pass a shared catalog rather than
58		-- pg_class, the table we'll prune.
59		LOOP
60			ROLLBACK;  -- release MyProc->xmin, which could be the oldest
61			cutoff := removable_cutoff('pg_database');
62			EXIT WHEN cutoff >= barrier;
63			RAISE LOG 'removable cutoff %; waiting for %', cutoff, barrier;
64			PERFORM pg_sleep(.1);
65		END LOOP;
66	END
67	$$;
68}
69# Eliminate HEAPTUPLE_DEAD and HEAPTUPLE_RECENTLY_DEAD from pg_class.
70# Minimize free space.
71#
72# If we kept HEAPTUPLE_RECENTLY_DEAD, step vac4 could prune what we missed,
73# breaking some permutation assumptions.  Specifically, the next pg_class
74# tuple could end up in free space we failed to liberate here, instead of
75# going in the specific free space vac4 intended to liberate for it.
76setup	{ CALL vactest.wait_prunable();  -- maximize VACUUM FULL }
77setup	{ VACUUM FULL pg_class;  -- reduce free space }
78# Remove the one tuple that VACUUM FULL makes dead, a tuple pertaining to
79# pg_class itself.  Populate the FSM for pg_class.
80#
81# wait_prunable waits for snapshots that would thwart pruning, while FREEZE
82# waits for buffer pins that would thwart pruning.  DISABLE_PAGE_SKIPPING
83# isn't actually needed, but other pruning-dependent tests use it.  If those
84# tests remove it, remove it here.
85setup	{ CALL vactest.wait_prunable();  -- maximize lazy VACUUM }
86setup	{ VACUUM (FREEZE, DISABLE_PAGE_SKIPPING) pg_class;  -- fill fsm etc. }
87setup
88{
89	SELECT FROM vactest.mkrels('orig', 1, 49);
90	CREATE TABLE vactest.orig50 (c int) WITH (autovacuum_enabled = off);
91	CREATE TABLE vactest.child50 (c int) WITH (autovacuum_enabled = off);
92	SELECT FROM vactest.mkrels('orig', 51, 100);
93}
94teardown
95{
96	DROP SCHEMA vactest CASCADE;
97	DROP EXTENSION injection_points;
98}

100# Wait during GRANT.  Disable debug_discard_caches, since we're here to
101# exercise an outcome that happens under permissible cache staleness.
102session s1
103setup	{
104	SET debug_discard_caches = 0;
105	SELECT FROM injection_points_set_local();
106	SELECT FROM injection_points_attach('heap_update-before-pin', 'wait');
107}
108step cachefill1	{ SELECT FROM vactest.reloid_catcache_set('vactest.orig50'); }
109step grant1	{ GRANT SELECT ON vactest.orig50 TO PUBLIC; }
110
111# Update of the tuple that grant1 will update.  Wait before sending invals, so
112# s1 will not get a cache miss.  Choose the commands for making such updates
113# from among those whose heavyweight locking does not conflict with GRANT's
114# heavyweight locking.  (GRANT will see our XID as committed, so observing
115# that XID in the tuple xmax also won't block GRANT.)
116session s2
117setup	{
118	SELECT FROM injection_points_set_local();
119	SELECT FROM
120		injection_points_attach('transaction-end-process-inval', 'wait');
121}
122step at2	{
123	CREATE TRIGGER to_set_relhastriggers BEFORE UPDATE ON vactest.orig50
124		FOR EACH ROW EXECUTE PROCEDURE suppress_redundant_updates_trigger();
125}
126
127# Hold snapshot to block pruning.
128session s3
129step snap3	{ BEGIN ISOLATION LEVEL REPEATABLE READ; SELECT; }
130step r3		{ ROLLBACK; }
131
132# Non-blocking actions.
133session s4
134step waitprunable4	{ CALL vactest.wait_prunable(); }
135# Eliminate HEAPTUPLE_DEAD.  See above discussion of FREEZE.
136step vac4		{ VACUUM (FREEZE, DISABLE_PAGE_SKIPPING) pg_class; }
137# Reuse the lp that s1 is waiting to change.  I've observed reuse at the 1st
138# or 18th CREATE, so create excess.
139step mkrels4	{
140	SELECT FROM vactest.mkrels('intruder', 1, 100);  -- repopulate LP_UNUSED
141}
142step wakegrant4	{
143	SELECT FROM injection_points_detach('heap_update-before-pin');
144	SELECT FROM injection_points_wakeup('heap_update-before-pin');
145}
146step at4	{ ALTER TABLE vactest.child50 INHERIT vactest.orig50; }
147step wakeinval4	{
148	SELECT FROM injection_points_detach('transaction-end-process-inval');
149	SELECT FROM injection_points_wakeup('transaction-end-process-inval');
150}
151# Witness effects of steps at2 and/or at4.
152step inspect4	{
153	SELECT relhastriggers, relhassubclass FROM pg_class
154		WHERE oid = 'vactest.orig50'::regclass;
155}

157# TID from syscache becomes LP_UNUSED.  Before the bug fix, this permutation
158# made s1 fail with "attempted to update invisible tuple" or an assert.
159# However, suppose a pd_lsn value such that (pd_lsn.xlogid, pd_lsn.xrecoff)
160# passed for (xmin, xmax) with xmin known-committed and xmax known-aborted.
161# Persistent page header corruption ensued.  For example, s1 overwrote
162# pd_lower, pd_upper, and pd_special as though they were t_ctid.
163permutation
164	cachefill1			# reads pg_class tuple T0, xmax invalid
165	at2					# T0 dead, T1 live
166	waitprunable4		# T0 prunable
167	vac4				# T0 becomes LP_UNUSED
168	grant1				# pauses at heap_update(T0)
169	wakeinval4(at2)		# at2 sends inval message
170	wakegrant4(grant1)	# s1 wakes: "tuple concurrently deleted"
171

174	cachefill1			# reads pg_class tuple T0, xmax invalid
