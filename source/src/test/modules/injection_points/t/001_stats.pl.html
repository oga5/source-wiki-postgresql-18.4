6use strict;
7use warnings FATAL => 'all';
8use locale;
9
10use PostgreSQL::Test::Cluster;
11use PostgreSQL::Test::Utils;
12use Test::More;
13
14# Test persistency of statistics generated for injection points.
15if ($ENV{enable_injection_points} ne 'yes')
16{
17	plan skip_all => 'Injection points not supported by this build';
18}
19
20# Node initialization
21my $node = PostgreSQL::Test::Cluster->new('master');
22$node->init;
23$node->append_conf(
24	'postgresql.conf', qq(
25shared_preload_libraries = 'injection_points'
26injection_points.stats = true
27));
28$node->start;
29$node->safe_psql('postgres', 'CREATE EXTENSION injection_points;');
30
31# This should count for two calls.
32$node->safe_psql('postgres',
33	"SELECT injection_points_attach('stats-notice', 'notice');");
34$node->safe_psql('postgres', "SELECT injection_points_run('stats-notice');");
35$node->safe_psql('postgres', "SELECT injection_points_run('stats-notice');");
36my $numcalls = $node->safe_psql('postgres',
37	"SELECT injection_points_stats_numcalls('stats-notice');");
38is($numcalls, '2', 'number of stats calls');
39my $fixedstats = $node->safe_psql('postgres',
40	"SELECT * FROM injection_points_stats_fixed();");
41is($fixedstats, '1|0|2|0|0', 'fixed stats after some calls');
42
43# Loading and caching.
44$node->safe_psql(
45	'postgres', "
46SELECT injection_points_load('stats-notice');
47SELECT injection_points_cached('stats-notice');
48");
49$fixedstats = $node->safe_psql('postgres',
50	"SELECT * FROM injection_points_stats_fixed();");
51is($fixedstats, '1|0|2|1|1', 'fixed stats after loading and caching');
52
53# Restart the node cleanly, stats should still be around.
54$node->restart;
55$numcalls = $node->safe_psql('postgres',
56	"SELECT injection_points_stats_numcalls('stats-notice');");
57is($numcalls, '3', 'number of stats after clean restart');
58$fixedstats = $node->safe_psql('postgres',
59	"SELECT * FROM injection_points_stats_fixed();");
60is($fixedstats, '1|0|2|1|1', 'fixed stats after clean restart');
61
62# On crash the stats are gone.
63$node->stop('immediate');
64$node->start;
65$numcalls = $node->safe_psql('postgres',
66	"SELECT injection_points_stats_numcalls('stats-notice');");
67is($numcalls, '', 'number of stats after crash');
68$fixedstats = $node->safe_psql('postgres',
69	"SELECT * FROM injection_points_stats_fixed();");
70is($fixedstats, '0|0|0|0|0', 'fixed stats after crash');
71
72# On drop all stats are gone
73$node->safe_psql('postgres',
74	"SELECT injection_points_attach('stats-notice', 'notice');");
75$node->safe_psql('postgres', "SELECT injection_points_run('stats-notice');");
76$node->safe_psql('postgres', "SELECT injection_points_run('stats-notice');");
77$numcalls = $node->safe_psql('postgres',
78	"SELECT injection_points_stats_numcalls('stats-notice');");
79is($numcalls, '2', 'number of stats calls');
80$node->safe_psql('postgres', "SELECT injection_points_stats_drop();");
81$numcalls = $node->safe_psql('postgres',
82	"SELECT injection_points_stats_numcalls('stats-notice');");
83is($numcalls, '', 'no stats after drop via SQL function');
84
85# Stop the server, disable the module, then restart.  The server
86# should be able to come up.
87$node->stop;
88$node->adjust_conf('postgresql.conf', 'shared_preload_libraries', "''");
89$node->start;
90
91done_testing();
