1/*
2 * This test must be run in a database with UTF-8 encoding,
3 * because other encodings don't support all the characters used.
4 */
5
6SELECT getdatabaseencoding() <> 'UTF8'
7       AS skip_test \gset
8\if :skip_test
9\quit
10\endif
11
12set client_encoding = utf8;
13
14set standard_conforming_strings = on;

17-- Run the Tcl test cases that require Unicode
18
19-- expectMatch	9.44 EMP*	{a[\u00fe-\u0507][\u00ff-\u0300]b} \
20-- 	"a\u0102\u02ffb"	"a\u0102\u02ffb"
21select * from test_regex('a[\u00fe-\u0507][\u00ff-\u0300]b', E'a\u0102\u02ffb', 'EMP*');
