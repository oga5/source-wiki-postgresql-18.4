11subdir = src/test
12top_builddir = ../..
13include $(top_builddir)/src/Makefile.global
14
15SUBDIRS = perl postmaster regress isolation modules authentication recovery subscription
16
17ifeq ($(with_icu),yes)
18SUBDIRS += icu
19endif
20ifeq ($(with_gssapi),yes)
21SUBDIRS += kerberos
22endif
23ifeq ($(with_ldap),yes)
24SUBDIRS += ldap
25endif
26ifeq ($(with_ssl),openssl)
27SUBDIRS += ssl
28endif
29
30# We don't build or execute these by default, but we do want "make
31# clean" etc to recurse into them.  (We must filter out those that we
32# have conditionally included into SUBDIRS above, else there will be
33# make confusion.)
34ALWAYS_SUBDIRS = $(filter-out $(SUBDIRS),examples kerberos icu ldap ssl)
35
36# We want to recurse to all subdirs for all standard targets, except that
37# installcheck and install should not recurse into the subdirectory "modules".
38
39recurse_alldirs_targets := $(filter-out installcheck install, $(standard_targets))
40installable_dirs := $(filter-out modules, $(SUBDIRS))
41
42$(call recurse,$(recurse_alldirs_targets))
43$(call recurse,installcheck, $(installable_dirs))
44$(call recurse,install, $(installable_dirs))
45
46$(recurse_always)
