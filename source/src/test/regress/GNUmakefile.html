13PGFILEDESC = "pg_regress - test driver"
14PGAPPICON = win32
15
16subdir = src/test/regress
17top_builddir = ../../..
18include $(top_builddir)/src/Makefile.global
19
20# maximum simultaneous connections for parallel tests
21MAXCONNOPT =
22ifdef MAX_CONNECTIONS
23MAXCONNOPT += --max-connections=$(MAX_CONNECTIONS)
24endif
25
26# stuff to pass into build of pg_regress
27EXTRADEFS = '-DHOST_TUPLE="$(host_tuple)"' \
28	'-DSHELLPROG="$(SHELL)"'
29
30##
31## Prepare for tests
32##
33
34# Build regression test driver
35
36all: pg_regress$(X)
37
38pg_regress$(X): pg_regress.o pg_regress_main.o $(WIN32RES) | submake-libpgport
39	$(CC) $(CFLAGS) $^ $(libpq_pgport) $(LDFLAGS) $(LDFLAGS_EX) $(LIBS) -o $@
40
41# dependencies ensure that path changes propagate
42pg_regress.o: pg_regress.c $(top_builddir)/src/port/pg_config_paths.h
43pg_regress.o: override CPPFLAGS := -I$(top_builddir)/src/port -I$(libpq_srcdir) $(EXTRADEFS) $(CPPFLAGS)
44
45# note: because of the submake dependency, this rule's action is really a no-op
46$(top_builddir)/src/port/pg_config_paths.h: | submake-libpgport
47	$(MAKE) -C $(top_builddir)/src/port pg_config_paths.h
48
49install: all installdirs
50	$(INSTALL_PROGRAM) pg_regress$(X) '$(DESTDIR)$(pgxsdir)/$(subdir)/pg_regress$(X)'
51
52installdirs:
53	$(MKDIR_P) '$(DESTDIR)$(pgxsdir)/$(subdir)'
54
55uninstall:
56	rm -f '$(DESTDIR)$(pgxsdir)/$(subdir)/pg_regress$(X)'
57
58
59# Build dynamically-loaded object file for CREATE FUNCTION ... LANGUAGE C.
60
61NAME = regress
62OBJS = $(WIN32RES) regress.o
63
64include $(top_srcdir)/src/Makefile.shlib
65
66all: all-lib
67
68# Ensure parallel safety if a build is started in this directory
69$(OBJS): | submake-libpgport submake-generated-headers
70
71
72# not installed by default
73
74regress_data_files = \
75	$(wildcard $(srcdir)/sql/*.sql) \
76	$(wildcard $(srcdir)/expected/*.out) \
77	$(wildcard $(srcdir)/data/*.data) \
78	$(srcdir)/parallel_schedule $(srcdir)/resultmap
79
80install-tests: all install install-lib installdirs-tests
81	$(MAKE) -C $(top_builddir)/contrib/spi install
82	for file in $(subst $(srcdir)/,,$(regress_data_files)); do \
83		$(INSTALL_DATA) $(srcdir)/$$file '$(DESTDIR)$(pkglibdir)/regress/'$$file || exit; \
84	done
85
86installdirs-tests: installdirs
87	$(MKDIR_P)  $(patsubst $(srcdir)/%/,'$(DESTDIR)$(pkglibdir)/regress/%',$(sort $(dir $(regress_data_files))))
88

90##
91## Run tests
92##
93
94REGRESS_OPTS = --dlpath=. --max-concurrent-tests=20 \
95	$(EXTRA_REGRESS_OPTS)
96
97check: all
98	$(pg_regress_check) $(REGRESS_OPTS) --schedule=$(srcdir)/parallel_schedule $(MAXCONNOPT) $(EXTRA_TESTS)
99
100check-tests: all | temp-install
101	$(pg_regress_check) $(REGRESS_OPTS) $(MAXCONNOPT) $(TESTS) $(EXTRA_TESTS)
102
103installcheck: all
104	$(pg_regress_installcheck) $(REGRESS_OPTS) --schedule=$(srcdir)/parallel_schedule --max-connections=1 $(EXTRA_TESTS)
105
106installcheck-parallel: all
107	$(pg_regress_installcheck) $(REGRESS_OPTS) --schedule=$(srcdir)/parallel_schedule $(MAXCONNOPT) $(EXTRA_TESTS)
108
109installcheck-tests: all
110	$(pg_regress_installcheck) $(REGRESS_OPTS) $(TESTS) $(EXTRA_TESTS)
111
112# old interfaces follow...
113
114runcheck: check
115runtest: installcheck
116runtest-parallel: installcheck-parallel
117
118bigtest: all
119	$(pg_regress_installcheck) $(REGRESS_OPTS) --schedule=$(srcdir)/parallel_schedule --max-connections=1 numeric_big
120
121bigcheck: all | temp-install
122	$(pg_regress_check) $(REGRESS_OPTS) --schedule=$(srcdir)/parallel_schedule $(MAXCONNOPT) numeric_big
