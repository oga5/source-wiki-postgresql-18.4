758CREATE OR REPLACE FUNCTION view_trigger() RETURNS trigger
759LANGUAGE plpgsql AS $$
760declare
761    argstr text := '';
762begin
763    for i in 0 .. TG_nargs - 1 loop
764        if i > 0 then
765            argstr := argstr || ', ';
766        end if;
767        argstr := argstr || TG_argv[i];
768    end loop;
769
770    raise notice '% % % % (%)', TG_TABLE_NAME, TG_WHEN, TG_OP, TG_LEVEL, argstr;
771
772    if TG_LEVEL = 'ROW' then
773        if TG_OP = 'INSERT' then
774            raise NOTICE 'NEW: %', NEW;
775            INSERT INTO main_table VALUES (NEW.a, NEW.b);
776            RETURN NEW;
777        end if;
778
779        if TG_OP = 'UPDATE' then
780            raise NOTICE 'OLD: %, NEW: %', OLD, NEW;
781            UPDATE main_table SET a = NEW.a, b = NEW.b WHERE a = OLD.a AND b = OLD.b;
782            if NOT FOUND then RETURN NULL; end if;
783            RETURN NEW;
784        end if;
785
786        if TG_OP = 'DELETE' then
787            raise NOTICE 'OLD: %', OLD;
788            DELETE FROM main_table WHERE a = OLD.a AND b = OLD.b;
789            if NOT FOUND then RETURN NULL; end if;
790            RETURN OLD;
791        end if;
792    end if;
793
794    RETURN NULL;
795end;
796$$;
