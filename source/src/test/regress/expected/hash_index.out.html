272CREATE TABLE hash_split_heap (keycol INT);
273INSERT INTO hash_split_heap SELECT 1 FROM generate_series(1, 500) a;
274CREATE INDEX hash_split_index on hash_split_heap USING HASH (keycol);
275INSERT INTO hash_split_heap SELECT 1 FROM generate_series(1, 5000) a;
276-- Let's do a backward scan.
277BEGIN;
278SET enable_seqscan = OFF;
279SET enable_bitmapscan = OFF;
280DECLARE c CURSOR FOR SELECT * from hash_split_heap WHERE keycol = 1;
281MOVE FORWARD ALL FROM c;
282MOVE BACKWARD 10000 FROM c;
283MOVE BACKWARD ALL FROM c;
284CLOSE c;
285END;
286-- DELETE, INSERT, VACUUM.
287DELETE FROM hash_split_heap WHERE keycol = 1;
288INSERT INTO hash_split_heap SELECT a/2 FROM generate_series(1, 25000) a;
289VACUUM hash_split_heap;
290-- Rebuild the index using a different fillfactor
291ALTER INDEX hash_split_index SET (fillfactor = 10);
292REINDEX INDEX hash_split_index;
293-- Clean up.
294DROP TABLE hash_split_heap;
295-- Testcases for removing overflow pages.
296CREATE TABLE hash_cleanup_heap(keycol INT);
297CREATE INDEX hash_cleanup_index on hash_cleanup_heap USING HASH (keycol);
298-- Insert tuples to both the primary bucket page and overflow pages.
299INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 500) as i;
300-- Fill overflow pages by "dead" tuples.
301BEGIN;
302INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 1000) as i;
303ROLLBACK;
304-- Checkpoint will ensure that all hash buffers are cleaned before we try
305-- to remove overflow pages.
306CHECKPOINT;
307-- This will squeeze the bucket and remove overflow pages.
308VACUUM hash_cleanup_heap;
309TRUNCATE hash_cleanup_heap;
310-- Insert a few tuples so that the primary bucket page doesn't get full and
311-- tuples can be moved to it.
312INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 50) as i;
313-- Fill overflow pages by "dead" tuples.
314BEGIN;
315INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 1500) as i;
316ROLLBACK;
317-- And insert some tuples again. During squeeze operation, these will be moved
318-- to the primary bucket allowing to test freeing intermediate overflow pages.
319INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 500) as i;
320CHECKPOINT;
321VACUUM hash_cleanup_heap;
322TRUNCATE hash_cleanup_heap;
323-- Insert tuples to both the primary bucket page and overflow pages.
324INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 500) as i;
325-- Fill overflow pages by "dead" tuples.
326BEGIN;
327INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 1500) as i;
328ROLLBACK;
329-- And insert some tuples again. During squeeze operation, these will be moved
330-- to other overflow pages and also allow overflow pages filled by dead tuples
331-- to be freed. Note the main purpose of this test is to test the case where
332-- we don't need to move any tuple from the overflow page being freed.
333INSERT INTO hash_cleanup_heap SELECT 1 FROM generate_series(1, 50) as i;
334CHECKPOINT;
335VACUUM hash_cleanup_heap;
336-- Clean up.
337DROP TABLE hash_cleanup_heap;
