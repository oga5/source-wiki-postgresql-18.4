243CREATE OR REPLACE FUNCTION undroppable() RETURNS event_trigger
244LANGUAGE plpgsql AS $$
245DECLARE
246	obj record;
247BEGIN
248	PERFORM 1 FROM pg_tables WHERE tablename = 'undroppable_objs';
249	IF NOT FOUND THEN
250		RAISE NOTICE 'table undroppable_objs not found, skipping';
251		RETURN;
252	END IF;
253	FOR obj IN
254		SELECT * FROM pg_event_trigger_dropped_objects() JOIN
255			undroppable_objs USING (object_type, object_identity)
256	LOOP
257		RAISE EXCEPTION 'object % of type % cannot be dropped',
258			obj.object_identity, obj.object_type;
259	END LOOP;
260END;
261$$;

382CREATE OR REPLACE FUNCTION event_trigger_report_dropped()
383 RETURNS event_trigger
384 LANGUAGE plpgsql
385AS $$
386DECLARE r record;
387BEGIN
388    FOR r IN SELECT * from pg_event_trigger_dropped_objects()
389    LOOP
390    IF NOT r.normal AND NOT r.original THEN
391        CONTINUE;
392    END IF;
393    RAISE NOTICE 'NORMAL: orig=% normal=% istemp=% type=% identity=% schema=% name=% addr=% args=%',
394        r.original, r.normal, r.is_temporary, r.object_type,
395        r.object_identity, r.schema_name, r.object_name,
396        r.address_names, r.address_args;
397    END LOOP;
398END; $$;
