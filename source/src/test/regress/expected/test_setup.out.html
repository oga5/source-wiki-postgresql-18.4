204CREATE FUNCTION binary_coercible(oid, oid)
205    RETURNS bool
206    AS :'regresslib', 'binary_coercible'
207    LANGUAGE C STRICT STABLE PARALLEL SAFE;
208-- Use hand-rolled hash functions and operator classes to get predictable
209-- result on different machines.  The hash function for int4 simply returns
210-- the sum of the values passed to it and the one for text returns the length
211-- of the non-empty string value passed to it or 0.
212create function part_hashint4_noop(value int4, seed int8)
213    returns int8 as $$
214    select value + seed;
215    $$ language sql strict immutable parallel safe;
216create operator class part_test_int4_ops for type int4 using hash as
217    operator 1 =,
218    function 2 part_hashint4_noop(int4, int8);
219create function part_hashtext_length(value text, seed int8)
220    returns int8 as $$
221    select length(coalesce(value, ''))::int8
222    $$ language sql strict immutable parallel safe;
223create operator class part_test_text_ops for type text using hash as
224    operator 1 =,
225    function 2 part_hashtext_length(text, int8);
226--
227-- These functions are used in tests that used to use md5(), which we now
228-- mostly avoid so that the tests will pass in FIPS mode.
229--
230create function fipshash(bytea)
231    returns text
232    strict immutable parallel safe leakproof
233    return substr(encode(sha256($1), 'hex'), 1, 32);
234create function fipshash(text)
235    returns text
236    strict immutable parallel safe leakproof
237    return substr(encode(sha256($1::bytea), 'hex'), 1, 32);
