291-- UPDATE with row movement
292---------------------------
293-- When a partitioned table receives an UPDATE to the partitioned key and the
294-- new values no longer meet the partition's bound, the row must be moved to
295-- the correct partition for the new partition key (if one exists). We must
296-- also ensure that updatable views on partitioned tables properly enforce any
297-- WITH CHECK OPTION that is defined. The situation with triggers in this case
298-- also requires thorough testing as partition key updates causing row
299-- movement convert UPDATEs into DELETE+INSERT.
300CREATE TABLE range_parted (
301	a text,
302	b bigint,
303	c numeric,
304	d int,
305	e varchar
306) PARTITION BY RANGE (a, b);
307-- Create partitions intentionally in descending bound order, so as to test
308-- that update-row-movement works with the leaf partitions not in bound order.
309CREATE TABLE part_b_20_b_30 (e varchar, c numeric, a text, b bigint, d int);
310ALTER TABLE range_parted ATTACH PARTITION part_b_20_b_30 FOR VALUES FROM ('b', 20) TO ('b', 30);
311CREATE TABLE part_b_10_b_20 (e varchar, c numeric, a text, b bigint, d int) PARTITION BY RANGE (c);
312CREATE TABLE part_b_1_b_10 PARTITION OF range_parted FOR VALUES FROM ('b', 1) TO ('b', 10);
313ALTER TABLE range_parted ATTACH PARTITION part_b_10_b_20 FOR VALUES FROM ('b', 10) TO ('b', 20);
314CREATE TABLE part_a_10_a_20 PARTITION OF range_parted FOR VALUES FROM ('a', 10) TO ('a', 20);
315CREATE TABLE part_a_1_a_10 PARTITION OF range_parted FOR VALUES FROM ('a', 1) TO ('a', 10);
316-- Check that partition-key UPDATE works sanely on a partitioned table that
317-- does not have any child partitions.
318UPDATE part_b_10_b_20 set b = b - 6;
319-- Create some more partitions following the above pattern of descending bound
320-- order, but let's make the situation a bit more complex by having the
321-- attribute numbers of the columns vary from their parent partition.

488CREATE FUNCTION trans_updatetrigfunc() RETURNS trigger LANGUAGE plpgsql AS
489$$
490  begin
491    raise notice 'trigger = %, old table = %, new table = %',
492                 TG_NAME,
493                 (select string_agg(old_table::text, ', ' ORDER BY a) FROM old_table),
494                 (select string_agg(new_table::text, ', ' ORDER BY a) FROM new_table);
495    return null;
496  end;
497$$;
498CREATE TRIGGER trans_updatetrig
499  AFTER UPDATE ON range_parted REFERENCING OLD TABLE AS old_table NEW TABLE AS new_table
500  FOR EACH STATEMENT EXECUTE PROCEDURE trans_updatetrigfunc();
