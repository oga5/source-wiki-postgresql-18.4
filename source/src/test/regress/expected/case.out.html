332--
333-- Nested CASE expressions
334--
335-- This test exercises a bug caused by aliasing econtext->caseValue_isNull
336-- with the isNull argument of the inner CASE's CaseExpr evaluation.  After
337-- evaluating the vol(null) expression in the inner CASE's second WHEN-clause,
338-- the isNull flag for the case test value incorrectly became true, causing
339-- the third WHEN-clause not to match.  The volatile function calls are needed
340-- to prevent constant-folding in the planner, which would hide the bug.
341-- Wrap this in a single transaction so the transient '=' operator doesn't
342-- cause problems in concurrent sessions
343BEGIN;
344CREATE FUNCTION vol(text) returns text as
345  'begin return $1; end' language plpgsql volatile;
346SELECT CASE
347  (CASE vol('bar')
348    WHEN 'foo' THEN 'it was foo!'
349    WHEN vol(null) THEN 'null input'
350    WHEN 'bar' THEN 'it was bar!' END
351  )
352  WHEN 'it was foo!' THEN 'foo recognized'
353  WHEN 'it was bar!' THEN 'bar recognized'
354  ELSE 'unrecognized' END;
355      case      
356----------------
357 bar recognized
358(1 row)

360-- In this case, we can't inline the SQL function without confusing things.
361CREATE DOMAIN foodomain AS text;
362CREATE FUNCTION volfoo(text) returns foodomain as
363  'begin return $1::foodomain; end' language plpgsql volatile;
364CREATE FUNCTION inline_eq(foodomain, foodomain) returns boolean as
365  'SELECT CASE $2::text WHEN $1::text THEN true ELSE false END' language sql;
366CREATE OPERATOR = (procedure = inline_eq,
367                   leftarg = foodomain, rightarg = foodomain);
368SELECT CASE volfoo('bar') WHEN 'foo'::foodomain THEN 'is foo' ELSE 'is not foo' END;
369    case    
370------------
371 is not foo
372(1 row)
373
374ROLLBACK;
375-- Test multiple evaluation of a CASE arg that is a read/write object (#14472)
376-- Wrap this in a single transaction so the transient '=' operator doesn't
377-- cause problems in concurrent sessions
378BEGIN;
379CREATE DOMAIN arrdomain AS int[];
380CREATE FUNCTION make_ad(int,int) returns arrdomain as
381  'declare x arrdomain;
382   begin
383     x := array[$1,$2];
384     return x;
385   end' language plpgsql volatile;
386CREATE FUNCTION ad_eq(arrdomain, arrdomain) returns boolean as
387  'begin return array_eq($1, $2); end' language plpgsql;
388CREATE OPERATOR = (procedure = ad_eq,
389                   leftarg = arrdomain, rightarg = arrdomain);
390SELECT CASE make_ad(1,2)
391  WHEN array[2,4]::arrdomain THEN 'wrong'
392  WHEN array[2,5]::arrdomain THEN 'still wrong'
393  WHEN array[1,2]::arrdomain THEN 'right'
394  END;
395 case  
396-------
397 right
