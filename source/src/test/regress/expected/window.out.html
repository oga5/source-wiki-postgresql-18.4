1951-- There is a syntactic ambiguity in the SQL standard.  Since
1952-- UNBOUNDED is a non-reserved word, it could be the name of a
1953-- function parameter and be used as an expression.  There is a
1954-- grammar hack to resolve such cases as the keyword.  The following
1955-- tests record this behavior.
1956CREATE FUNCTION unbounded_syntax_test1a(x int) RETURNS TABLE (a int, b int, c int)
1957LANGUAGE SQL
1958BEGIN ATOMIC
1959  SELECT sum(unique1) over (rows between x preceding and x following),
1960         unique1, four
1961  FROM tenk1 WHERE unique1 < 10;
1962END;
1963CREATE FUNCTION unbounded_syntax_test1b(x int) RETURNS TABLE (a int, b int, c int)
1964LANGUAGE SQL
1965AS $$
1966  SELECT sum(unique1) over (rows between x preceding and x following),
1967         unique1, four
1968  FROM tenk1 WHERE unique1 < 10;
1969$$;
