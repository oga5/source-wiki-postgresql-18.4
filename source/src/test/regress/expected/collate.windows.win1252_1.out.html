1/*
2 * This test is meant to run on Windows systems that has successfully
3 * run pg_import_system_collations().  Also, the database must have
4 * WIN1252 encoding, because of the locales' own encodings.  Because
5 * of this, some test are lost from UTF-8 version, such as Turkish
6 * dotted and undotted 'i'.
7 */
8SELECT getdatabaseencoding() <> 'WIN1252' OR
9       (SELECT count(*) FROM pg_collation WHERE collname IN ('de_DE', 'en_US', 'sv_SE') AND collencoding = pg_char_to_encoding('WIN1252')) <> 3 OR
10       (version() !~ 'Visual C\+\+' AND version() !~ 'mingw32' AND version() !~ 'windows')
11       AS skip_test \gset
12\if :skip_test
13\quit
