79-- Test that GiST indexes provide same behavior as sequential scan

82CREATE INDEX point_gist_tbl_index ON point_gist_tbl USING gist (f1);
83INSERT INTO point_gist_tbl VALUES ('(0.0000009,0.0000009)');
