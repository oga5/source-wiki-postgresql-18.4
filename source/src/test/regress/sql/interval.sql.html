118-- Test multiplication and division with intervals.
119-- Floating point arithmetic rounding errors can lead to unexpected results,
120-- though the code attempts to do the right thing and round up to days and
121-- minutes to avoid results such as '3 days 24:00 hours' or '14:20:60'.
122-- Note that it is expected for some day components to be greater than 29 and
123-- some time components be greater than 23:59:59 due to how intervals are
124-- stored internally.
125
126CREATE TABLE INTERVAL_MULDIV_TBL (span interval);
127COPY INTERVAL_MULDIV_TBL FROM STDIN;
12841 mon 12 days 360:00
129-41 mon -12 days +360:00
130-12 days

132-3 years 482 days 76:54:32.189
1334 mon
13414 mon
135999 mon 999 days
136\.
