222-- Check that clustering sets new relfilenodes:
223CREATE TEMP TABLE old_cluster_info AS SELECT relname, level, relfilenode, relkind FROM pg_partition_tree('clstrpart'::regclass) AS tree JOIN pg_class c ON c.oid=tree.relid ;
224CLUSTER clstrpart USING clstrpart_idx;
225CREATE TEMP TABLE new_cluster_info AS SELECT relname, level, relfilenode, relkind FROM pg_partition_tree('clstrpart'::regclass) AS tree JOIN pg_class c ON c.oid=tree.relid ;
226SELECT relname, old.level, old.relkind, old.relfilenode = new.relfilenode FROM old_cluster_info AS old JOIN new_cluster_info AS new USING (relname) ORDER BY relname COLLATE "C";

244ALTER TABLE ptnowner OWNER TO regress_ptnowner;
245CREATE TEMP TABLE ptnowner_oldnodes AS
246  SELECT oid, relname, relfilenode FROM pg_partition_tree('ptnowner') AS tree
247  JOIN pg_class AS c ON c.oid=tree.relid;

250RESET SESSION AUTHORIZATION;
251SELECT a.relname, a.relfilenode=b.relfilenode FROM pg_class a
252  JOIN ptnowner_oldnodes b USING (oid) ORDER BY a.relname COLLATE "C";

275-- test CLUSTER on expression index
276CREATE TABLE clstr_expression(id serial primary key, a int, b text COLLATE "C");
277INSERT INTO clstr_expression(a, b) SELECT g.i % 42, 'prefix'||g.i FROM generate_series(1, 133) g(i);
278CREATE INDEX clstr_expression_minus_a ON clstr_expression ((-a), b);
279CREATE INDEX clstr_expression_upper_b ON clstr_expression ((upper(b)));

282BEGIN;

284EXPLAIN (COSTS OFF) SELECT * FROM clstr_expression WHERE upper(b) = 'PREFIX3';

287SELECT * FROM clstr_expression WHERE -a = -3 ORDER BY -a, b;
288COMMIT;

294SELECT * FROM rows WHERE la < a;
295BEGIN;
296SET LOCAL enable_seqscan = false;

298SELECT * FROM clstr_expression WHERE upper(b) = 'PREFIX3';

300SELECT * FROM clstr_expression WHERE -a = -3 ORDER BY -a, b;
