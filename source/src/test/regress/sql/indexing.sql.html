249-- Verify that re-attaching an already-attached partition index can
250-- validate the parent index if it was still invalid, including
251-- indirect ancestors in subpartitions.
252create table idxpart (a int, b int) partition by range (a);
253create table idxpart1 partition of idxpart for values from (0) to (1000) partition by range (a);
254create table idxpart11 partition of idxpart1 for values from (0) to (500);
255-- Partitioned table with no partitions
256create table idxpart2 partition of idxpart for values from (1000) to (2000) partition by range (a);
257-- create parent indexes
258create index on only idxpart ((a/b));
259create index on only idxpart1 ((a/b));
260create index on only idxpart2 ((a/b));
261-- fail, leaves behind an invalid index on the leaf partition
262insert into idxpart11 values (1, 0);
263create index concurrently on idxpart11 ((a/b));
264select relname, indisvalid from pg_class join pg_index on indexrelid = oid
265   where relname like 'idxpart%' order by relname;
266-- attach the indexes; parents stay invalid
267alter index idxpart1_expr_idx attach partition idxpart11_expr_idx;
268alter index idxpart_expr_idx attach partition idxpart1_expr_idx;
269alter index idxpart_expr_idx attach partition idxpart2_expr_idx;
270select relname, indisvalid from pg_class join pg_index on indexrelid = oid
271   where relname like 'idxpart%' order by relname;
272-- fix the index on the leaf partition
273delete from idxpart11 where b = 0;
274reindex index concurrently idxpart11_expr_idx;
275-- reattach the leaf partition index; parents should now be valid
276alter index idxpart1_expr_idx attach partition idxpart11_expr_idx;
