46--
47-- Test the planner's ability to reorder the distinctClause Pathkeys to match
48-- the input path's ordering
49--
50
51CREATE TABLE distinct_on_tbl (x int, y int, z int);
52INSERT INTO distinct_on_tbl SELECT i%10, i%10, i%10 FROM generate_series(1, 1000) AS i;
53CREATE INDEX distinct_on_tbl_x_y_idx ON distinct_on_tbl (x, y);
54ANALYZE distinct_on_tbl;
55
56-- Produce results with sorting.
57SET enable_hashagg TO OFF;
58
59-- Ensure we avoid the need to re-sort by reordering the distinctClause
60-- Pathkeys to match the ordering of the input path
61EXPLAIN (COSTS OFF)
62SELECT DISTINCT ON (y, x) x, y FROM distinct_on_tbl;
63SELECT DISTINCT ON (y, x) x, y FROM distinct_on_tbl;
64
65-- Ensure we leverage incremental-sort by reordering the distinctClause
66-- Pathkeys to partially match the ordering of the input path
67EXPLAIN (COSTS OFF)
68SELECT DISTINCT ON (y, x) x, y FROM (SELECT * FROM distinct_on_tbl ORDER BY x) s;
69SELECT DISTINCT ON (y, x) x, y FROM (SELECT * FROM distinct_on_tbl ORDER BY x) s;
70
71-- Ensure we reorder the distinctClause Pathkeys to match the ordering of the
72-- input path even if there is ORDER BY clause
73EXPLAIN (COSTS OFF)
74SELECT DISTINCT ON (y, x) x, y FROM distinct_on_tbl ORDER BY y;
75SELECT DISTINCT ON (y, x) x, y FROM distinct_on_tbl ORDER BY y;

77-- Ensure the resulting pathkey list matches the initial distinctClause Pathkeys
78EXPLAIN (COSTS OFF)
79SELECT DISTINCT ON (y, x) x, y FROM (select * from distinct_on_tbl order by x, z, y) s ORDER BY y, x, z;
80SELECT DISTINCT ON (y, x) x, y FROM (select * from distinct_on_tbl order by x, z, y) s ORDER BY y, x, z;
