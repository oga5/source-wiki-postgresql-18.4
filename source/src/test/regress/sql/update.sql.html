161-- UPDATE with row movement
162---------------------------
163
164-- When a partitioned table receives an UPDATE to the partitioned key and the
165-- new values no longer meet the partition's bound, the row must be moved to
166-- the correct partition for the new partition key (if one exists). We must
167-- also ensure that updatable views on partitioned tables properly enforce any
168-- WITH CHECK OPTION that is defined. The situation with triggers in this case
169-- also requires thorough testing as partition key updates causing row
170-- movement convert UPDATEs into DELETE+INSERT.
171
172CREATE TABLE range_parted (
173	a text,
174	b bigint,
175	c numeric,
176	d int,
177	e varchar
178) PARTITION BY RANGE (a, b);
179
180-- Create partitions intentionally in descending bound order, so as to test
181-- that update-row-movement works with the leaf partitions not in bound order.
182CREATE TABLE part_b_20_b_30 (e varchar, c numeric, a text, b bigint, d int);
183ALTER TABLE range_parted ATTACH PARTITION part_b_20_b_30 FOR VALUES FROM ('b', 20) TO ('b', 30);
184CREATE TABLE part_b_10_b_20 (e varchar, c numeric, a text, b bigint, d int) PARTITION BY RANGE (c);
185CREATE TABLE part_b_1_b_10 PARTITION OF range_parted FOR VALUES FROM ('b', 1) TO ('b', 10);
186ALTER TABLE range_parted ATTACH PARTITION part_b_10_b_20 FOR VALUES FROM ('b', 10) TO ('b', 20);
187CREATE TABLE part_a_10_a_20 PARTITION OF range_parted FOR VALUES FROM ('a', 10) TO ('a', 20);
188CREATE TABLE part_a_1_a_10 PARTITION OF range_parted FOR VALUES FROM ('a', 1) TO ('a', 10);
189
190-- Check that partition-key UPDATE works sanely on a partitioned table that
191-- does not have any child partitions.
192UPDATE part_b_10_b_20 set b = b - 6;
193
194-- Create some more partitions following the above pattern of descending bound
195-- order, but let's make the situation a bit more complex by having the
196-- attribute numbers of the columns vary from their parent partition.
197CREATE TABLE part_c_100_200 (e varchar, c numeric, a text, b bigint, d int) PARTITION BY range (abs(d));
198ALTER TABLE part_c_100_200 DROP COLUMN e, DROP COLUMN c, DROP COLUMN a;
199ALTER TABLE part_c_100_200 ADD COLUMN c numeric, ADD COLUMN e varchar, ADD COLUMN a text;
200ALTER TABLE part_c_100_200 DROP COLUMN b;
201ALTER TABLE part_c_100_200 ADD COLUMN b bigint;
202CREATE TABLE part_d_1_15 PARTITION OF part_c_100_200 FOR VALUES FROM (1) TO (15);
203CREATE TABLE part_d_15_20 PARTITION OF part_c_100_200 FOR VALUES FROM (15) TO (20);
204
205ALTER TABLE part_b_10_b_20 ATTACH PARTITION part_c_100_200 FOR VALUES FROM (100) TO (200);
206
207CREATE TABLE part_c_1_100 (e varchar, d int, c numeric, b bigint, a text);
208ALTER TABLE part_b_10_b_20 ATTACH PARTITION part_c_1_100 FOR VALUES FROM (1) TO (100);

269CREATE FUNCTION trans_updatetrigfunc() RETURNS trigger LANGUAGE plpgsql AS
270$$
271  begin
272    raise notice 'trigger = %, old table = %, new table = %',
273                 TG_NAME,
274                 (select string_agg(old_table::text, ', ' ORDER BY a) FROM old_table),
275                 (select string_agg(new_table::text, ', ' ORDER BY a) FROM new_table);
276    return null;
277  end;
278$$;
279
280CREATE TRIGGER trans_updatetrig
281  AFTER UPDATE ON range_parted REFERENCING OLD TABLE AS old_table NEW TABLE AS new_table
282  FOR EACH STATEMENT EXECUTE PROCEDURE trans_updatetrigfunc();

303-- Test with transition tuple conversion happening for rows moved into the
304-- new partition. This requires a trigger that references transition table
305-- (we already have trans_updatetrig). For inserted rows, the conversion
306-- is not usually needed, because the original tuple is already compatible with
307-- the desired transition tuple format. But conversion happens when there is a
308-- BR trigger because the trigger can change the inserted row. So install a
309-- BR triggers on those child partitions where the rows will be moved.
310CREATE FUNCTION func_parted_mod_b() RETURNS trigger AS $$
311BEGIN
312   NEW.b = NEW.b + 1;
313   return NEW;
314END $$ language plpgsql;
315CREATE TRIGGER trig_c1_100 BEFORE UPDATE OR INSERT ON part_c_1_100
316   FOR EACH ROW EXECUTE PROCEDURE func_parted_mod_b();
317CREATE TRIGGER trig_d1_15 BEFORE UPDATE OR INSERT ON part_d_1_15
318   FOR EACH ROW EXECUTE PROCEDURE func_parted_mod_b();
319CREATE TRIGGER trig_d15_20 BEFORE UPDATE OR INSERT ON part_d_15_20
320   FOR EACH ROW EXECUTE PROCEDURE func_parted_mod_b();
321:init_range_parted;

429CREATE FUNCTION trigfunc() returns trigger language plpgsql as
430$$
431  begin
432    raise notice 'trigger = % fired on table % during %',
433                 TG_NAME, TG_TABLE_NAME, TG_OP;

435  end;
436$$;
