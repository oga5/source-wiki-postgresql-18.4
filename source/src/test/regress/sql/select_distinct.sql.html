227-- the input path's ordering
228--

230CREATE TABLE distinct_tbl (x int, y int);
231INSERT INTO distinct_tbl SELECT i%10, i%10 FROM generate_series(1, 1000) AS i;
232CREATE INDEX distinct_tbl_x_y_idx ON distinct_tbl (x, y);
233ANALYZE distinct_tbl;

235-- Produce results with sorting.
236SET enable_hashagg TO OFF;
237
238-- Ensure we avoid the need to re-sort by reordering the distinctClause
239-- Pathkeys to match the ordering of the input path
240EXPLAIN (COSTS OFF)
241SELECT DISTINCT y, x FROM distinct_tbl;
242SELECT DISTINCT y, x FROM distinct_tbl;
243
244-- Ensure we leverage incremental-sort by reordering the distinctClause
245-- Pathkeys to partially match the ordering of the input path
246EXPLAIN (COSTS OFF)
247SELECT DISTINCT y, x FROM (SELECT * FROM distinct_tbl ORDER BY x) s;
248SELECT DISTINCT y, x FROM (SELECT * FROM distinct_tbl ORDER BY x) s;
