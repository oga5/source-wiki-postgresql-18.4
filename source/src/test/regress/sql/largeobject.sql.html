41-- lo_open(lobjId oid, mode integer) returns integer
42-- The mode parameter to lo_open uses two constants:
43--   INV_WRITE = 0x20000
44--   INV_READ  = 0x40000
45-- The return value is a file descriptor-like value which remains valid for the
46-- transaction.
47UPDATE lotest_stash_values SET fd = lo_open(loid, CAST(x'20000' | x'40000' AS integer));

49-- loread/lowrite names are wonky, different from other functions which are lo_*
50-- lowrite(fd integer, data bytea) returns integer
51-- the integer is the number of bytes written
52SELECT lowrite(fd, '
53I wandered lonely as a cloud
54That floats on high o''er vales and hills,
55When all at once I saw a crowd,
56A host, of golden daffodils;
57Beside the lake, beneath the trees,
58Fluttering and dancing in the breeze.
59
60Continuous as the stars that shine
61And twinkle on the milky way,
62They stretched in never-ending line
63Along the margin of a bay:
64Ten thousand saw I at a glance,
65Tossing their heads in sprightly dance.
66
67The waves beside them danced; but they
68Out-did the sparkling waves in glee:
69A poet could not but be gay,
70In such a jocund company:
71I gazed--and gazed--but little thought
72What wealth the show to me had brought:
73
74For oft, when on my couch I lie
75In vacant or in pensive mood,
76They flash upon that inward eye
77Which is the bliss of solitude;
78And then my heart with pleasure fills,
79And dances with the daffodils.
80
81         -- William Wordsworth
