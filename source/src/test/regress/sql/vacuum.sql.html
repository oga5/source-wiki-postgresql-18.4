356-- Check the number of vacuums done on table vac_option_tab and on its
357-- toast relation, to check that PROCESS_TOAST and PROCESS_MAIN work on
358-- what they should.
359CREATE VIEW vac_option_tab_counts AS
360  SELECT CASE WHEN c.relname IS NULL
361    THEN 'main' ELSE 'toast' END as rel,
362  s.vacuum_count
363  FROM pg_stat_all_tables s
364  LEFT JOIN pg_class c ON s.relid = c.reltoastrelid
365  WHERE c.relname = 'vac_option_tab' OR s.relname = 'vac_option_tab'
366  ORDER BY rel;
367VACUUM (PROCESS_TOAST TRUE) vac_option_tab;
368SELECT * FROM vac_option_tab_counts;
369VACUUM (PROCESS_TOAST FALSE) vac_option_tab;
370SELECT * FROM vac_option_tab_counts;
371VACUUM (PROCESS_TOAST FALSE, FULL) vac_option_tab; -- error
