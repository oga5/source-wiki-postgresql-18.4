234-- Concurrent refresh requires a unique index on the materialized
235-- view. Test what happens if it's dropped during the refresh.
236SET search_path = mvtest_mvschema, public;
237CREATE OR REPLACE FUNCTION mvtest_drop_the_index()
238  RETURNS bool AS $$
239BEGIN
240  EXECUTE 'DROP INDEX IF EXISTS mvtest_mvschema.mvtest_drop_idx';
241  RETURN true;
242END;
243$$ LANGUAGE plpgsql;
244
245CREATE MATERIALIZED VIEW drop_idx_matview AS
246  SELECT 1 as i WHERE mvtest_drop_the_index();
247
248CREATE UNIQUE INDEX mvtest_drop_idx ON drop_idx_matview (i);
249REFRESH MATERIALIZED VIEW CONCURRENTLY drop_idx_matview;
250DROP MATERIALIZED VIEW drop_idx_matview; -- clean up
251RESET search_path;
