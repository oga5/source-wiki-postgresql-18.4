248-- Round-trip non-ASCII data through xpath().
249DO $$
250DECLARE
251  xml_declaration text := '<?xml version="1.0" encoding="ISO-8859-1"?>';
252  degree_symbol text;
253  res xml[];
254BEGIN
255  -- Per the documentation, except when the server encoding is UTF8, xpath()
256  -- may not work on non-ASCII data.  The untranslatable_character and
257  -- undefined_function traps below, currently dead code, will become relevant
258  -- if we remove this limitation.
259  IF current_setting('server_encoding') <> 'UTF8' THEN
260    RAISE LOG 'skip: encoding % unsupported for xpath',
261      current_setting('server_encoding');
262    RETURN;
263  END IF;
264
265  degree_symbol := convert_from('\xc2b0', 'UTF8');
266  res := xpath('text()', (xml_declaration ||
267    '<x>' || degree_symbol || '</x>')::xml);
268  IF degree_symbol <> res[1]::text THEN
269    RAISE 'expected % (%), got % (%)',
270      degree_symbol, convert_to(degree_symbol, 'UTF8'),
271      res[1], convert_to(res[1]::text, 'UTF8');
272  END IF;
273EXCEPTION
274  -- character with byte sequence 0xc2 0xb0 in encoding "UTF8" has no equivalent in encoding "LATIN8"
275  WHEN untranslatable_character
276  -- default conversion function for encoding "UTF8" to "MULE_INTERNAL" does not exist
277  OR undefined_function
278  -- unsupported XML feature
279  OR feature_not_supported THEN
280    RAISE LOG 'skip: %', SQLERRM;
281END
282$$;

339-- Since xpath() deals with namespaces, it's a bit stricter about
340-- what's well-formed and what's not. If we don't obey these rules
341-- (i.e. ignore namespace-related errors from libxml), xpath()
342-- fails in subtle ways. The following would for example produce
343-- the xml value
344--   <invalidns xmlns='<'/>
345-- which is invalid because '<' may not appear un-escaped in
346-- attribute values.
347-- Since different libxml versions emit slightly different
348-- error messages, we suppress the DETAIL in this test.
349\set VERBOSITY terse
350SELECT xpath('/*', '<invalidns xmlns=''&lt;''/>');
351\set VERBOSITY default
352
353-- Again, the XML isn't well-formed for namespace purposes
