11-- Setup
12CREATE SCHEMA lock_schema1;
13SET search_path = lock_schema1;
14CREATE TABLE lock_tbl1 (a BIGINT);
15CREATE TABLE lock_tbl1a (a BIGINT);
16CREATE VIEW lock_view1 AS SELECT * FROM lock_tbl1;
17CREATE VIEW lock_view2(a,b) AS SELECT * FROM lock_tbl1, lock_tbl1a;
18CREATE VIEW lock_view3 AS SELECT * from lock_view2;
19CREATE VIEW lock_view4 AS SELECT (select a from lock_tbl1a limit 1) from lock_tbl1;
20CREATE VIEW lock_view5 AS SELECT * from lock_tbl1 where a in (select * from lock_tbl1a);
21CREATE VIEW lock_view6 AS SELECT * from (select * from lock_tbl1) sub;
22CREATE ROLE regress_rol_lock1;
23ALTER ROLE regress_rol_lock1 SET search_path = lock_schema1;
24GRANT USAGE ON SCHEMA lock_schema1 TO regress_rol_lock1;
25
26-- Try all valid lock options; also try omitting the optional TABLE keyword.
27BEGIN TRANSACTION;
28LOCK TABLE lock_tbl1 IN ACCESS SHARE MODE;
29LOCK lock_tbl1 IN ROW SHARE MODE;
30LOCK TABLE lock_tbl1 IN ROW EXCLUSIVE MODE;
31LOCK TABLE lock_tbl1 IN SHARE UPDATE EXCLUSIVE MODE;
32LOCK TABLE lock_tbl1 IN SHARE MODE;
33LOCK lock_tbl1 IN SHARE ROW EXCLUSIVE MODE;
34LOCK TABLE lock_tbl1 IN EXCLUSIVE MODE;
35LOCK TABLE lock_tbl1 IN ACCESS EXCLUSIVE MODE;
36ROLLBACK;
37
38-- Try using NOWAIT along with valid options.
39BEGIN TRANSACTION;
40LOCK TABLE lock_tbl1 IN ACCESS SHARE MODE NOWAIT;
41LOCK TABLE lock_tbl1 IN ROW SHARE MODE NOWAIT;
42LOCK TABLE lock_tbl1 IN ROW EXCLUSIVE MODE NOWAIT;
43LOCK TABLE lock_tbl1 IN SHARE UPDATE EXCLUSIVE MODE NOWAIT;
44LOCK TABLE lock_tbl1 IN SHARE MODE NOWAIT;
45LOCK TABLE lock_tbl1 IN SHARE ROW EXCLUSIVE MODE NOWAIT;
46LOCK TABLE lock_tbl1 IN EXCLUSIVE MODE NOWAIT;
47LOCK TABLE lock_tbl1 IN ACCESS EXCLUSIVE MODE NOWAIT;
48ROLLBACK;
49
50-- Verify that we can lock views.
51BEGIN TRANSACTION;
52LOCK TABLE lock_view1 IN EXCLUSIVE MODE;
53-- lock_view1 and lock_tbl1 are locked.
54select relname from pg_locks l, pg_class c
55 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
56 order by relname;
57ROLLBACK;
58BEGIN TRANSACTION;
59LOCK TABLE lock_view2 IN EXCLUSIVE MODE;
60-- lock_view1, lock_tbl1, and lock_tbl1a are locked.
61select relname from pg_locks l, pg_class c
62 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
63 order by relname;
64ROLLBACK;
65BEGIN TRANSACTION;
66LOCK TABLE lock_view3 IN EXCLUSIVE MODE;
67-- lock_view3, lock_view2, lock_tbl1, and lock_tbl1a are locked recursively.
68select relname from pg_locks l, pg_class c
69 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
70 order by relname;
71ROLLBACK;
72BEGIN TRANSACTION;
73LOCK TABLE lock_view4 IN EXCLUSIVE MODE;
74-- lock_view4, lock_tbl1, and lock_tbl1a are locked.
75select relname from pg_locks l, pg_class c
76 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
77 order by relname;
78ROLLBACK;
79BEGIN TRANSACTION;
80LOCK TABLE lock_view5 IN EXCLUSIVE MODE;
81-- lock_view5, lock_tbl1, and lock_tbl1a are locked.
82select relname from pg_locks l, pg_class c
83 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
84 order by relname;
85ROLLBACK;
86BEGIN TRANSACTION;
87LOCK TABLE lock_view6 IN EXCLUSIVE MODE;
88-- lock_view6 an lock_tbl1 are locked.
89select relname from pg_locks l, pg_class c
90 where l.relation = c.oid and relname like '%lock_%' and mode = 'ExclusiveLock'
91 order by relname;
92ROLLBACK;
93-- Verify that we cope with infinite recursion in view definitions.
94CREATE OR REPLACE VIEW lock_view2 AS SELECT * from lock_view3;
95BEGIN TRANSACTION;
96LOCK TABLE lock_view2 IN EXCLUSIVE MODE;
97ROLLBACK;
98CREATE VIEW lock_view7 AS SELECT * from lock_view2;
99BEGIN TRANSACTION;
100LOCK TABLE lock_view7 IN EXCLUSIVE MODE;
101ROLLBACK;
102
103-- Verify that we can lock a table with inheritance children.
104CREATE TABLE lock_tbl2 (b BIGINT) INHERITS (lock_tbl1);
105CREATE TABLE lock_tbl3 () INHERITS (lock_tbl2);
106BEGIN TRANSACTION;
107LOCK TABLE lock_tbl1 * IN ACCESS EXCLUSIVE MODE;
108ROLLBACK;
109
110-- Child tables are locked without granting explicit permission to do so as
111-- long as we have permission to lock the parent.
112GRANT UPDATE ON TABLE lock_tbl1 TO regress_rol_lock1;
113SET ROLE regress_rol_lock1;
114-- fail when child locked directly
115BEGIN;
116LOCK TABLE lock_tbl2;
117ROLLBACK;
118BEGIN;
119LOCK TABLE lock_tbl1 * IN ACCESS EXCLUSIVE MODE;
120ROLLBACK;
121BEGIN;
122LOCK TABLE ONLY lock_tbl1;
123ROLLBACK;
124RESET ROLE;
125REVOKE UPDATE ON TABLE lock_tbl1 FROM regress_rol_lock1;
126
127-- Tables referred to by views are locked without explicit permission to do so
128-- as long as we have permission to lock the view itself.
129SET ROLE regress_rol_lock1;
130-- fail without permissions on the view
131BEGIN;
132LOCK TABLE lock_view1;
133ROLLBACK;
134RESET ROLE;
135GRANT UPDATE ON TABLE lock_view1 TO regress_rol_lock1;
136SET ROLE regress_rol_lock1;
137BEGIN;
138LOCK TABLE lock_view1 IN ACCESS EXCLUSIVE MODE;
139-- lock_view1 and lock_tbl1 (plus children lock_tbl2 and lock_tbl3) are locked.
140select relname from pg_locks l, pg_class c
141 where l.relation = c.oid and relname like '%lock_%' and mode = 'AccessExclusiveLock'
142 order by relname;
143ROLLBACK;
144RESET ROLE;
145REVOKE UPDATE ON TABLE lock_view1 FROM regress_rol_lock1;
146
147-- Tables referred to by security invoker views require explicit permission to
148-- be locked.
149CREATE VIEW lock_view8 WITH (security_invoker) AS SELECT * FROM lock_tbl1;
150SET ROLE regress_rol_lock1;
151-- fail without permissions on the view
152BEGIN;
153LOCK TABLE lock_view8;
154ROLLBACK;
155RESET ROLE;
156GRANT UPDATE ON TABLE lock_view8 TO regress_rol_lock1;
157SET ROLE regress_rol_lock1;
158-- fail without permissions on the table referenced by the view
159BEGIN;
160LOCK TABLE lock_view8;
161ROLLBACK;
162RESET ROLE;
163GRANT UPDATE ON TABLE lock_tbl1 TO regress_rol_lock1;
164BEGIN;
165LOCK TABLE lock_view8 IN ACCESS EXCLUSIVE MODE;
166-- lock_view8 and lock_tbl1 (plus children lock_tbl2 and lock_tbl3) are locked.
167select relname from pg_locks l, pg_class c
168 where l.relation = c.oid and relname like '%lock_%' and mode = 'AccessExclusiveLock'
169 order by relname;
170ROLLBACK;
171RESET ROLE;
172REVOKE UPDATE ON TABLE lock_view8 FROM regress_rol_lock1;
