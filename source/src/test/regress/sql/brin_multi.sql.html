244DO $x$
245DECLARE
246	r record;
247	r2 record;
248	cond text;
249	idx_ctids tid[];
250	ss_ctids tid[];
251	count int;
252	plan_ok bool;
253	plan_line text;
254BEGIN
255	FOR r IN SELECT colname, oper, typ, value[ordinality], matches[ordinality] FROM brinopers_multi, unnest(op) WITH ORDINALITY AS oper LOOP
256
257		-- prepare the condition
258		IF r.value IS NULL THEN
259			cond := format('%I %s %L', r.colname, r.oper, r.value);
260		ELSE
261			cond := format('%I %s %L::%s', r.colname, r.oper, r.value, r.typ);
262		END IF;
263
264		-- run the query using the brin index
265		SET enable_seqscan = 0;
266		SET enable_bitmapscan = 1;
267
268		plan_ok := false;
269		FOR plan_line IN EXECUTE format($y$EXPLAIN SELECT array_agg(ctid) FROM brintest_multi WHERE %s $y$, cond) LOOP
270			IF plan_line LIKE '%Bitmap Heap Scan on brintest_multi%' THEN
271				plan_ok := true;
272			END IF;
273		END LOOP;
274		IF NOT plan_ok THEN
275			RAISE WARNING 'did not get bitmap indexscan plan for %', r;
276		END IF;
277
278		EXECUTE format($y$SELECT array_agg(ctid) FROM brintest_multi WHERE %s $y$, cond)
279			INTO idx_ctids;
280
281		-- run the query using a seqscan
282		SET enable_seqscan = 1;
283		SET enable_bitmapscan = 0;
284
285		plan_ok := false;
286		FOR plan_line IN EXECUTE format($y$EXPLAIN SELECT array_agg(ctid) FROM brintest_multi WHERE %s $y$, cond) LOOP
287			IF plan_line LIKE '%Seq Scan on brintest_multi%' THEN
288				plan_ok := true;
289			END IF;
290		END LOOP;
291		IF NOT plan_ok THEN
292			RAISE WARNING 'did not get seqscan plan for %', r;
293		END IF;
294
295		EXECUTE format($y$SELECT array_agg(ctid) FROM brintest_multi WHERE %s $y$, cond)
296			INTO ss_ctids;
297
298		-- make sure both return the same results
299		count := array_length(idx_ctids, 1);
300
301		IF NOT (count = array_length(ss_ctids, 1) AND
302				idx_ctids @> ss_ctids AND
303				idx_ctids <@ ss_ctids) THEN
304			-- report the results of each scan to make the differences obvious
305			RAISE WARNING 'something not right in %: count %', r, count;
306			SET enable_seqscan = 1;
307			SET enable_bitmapscan = 0;
308			FOR r2 IN EXECUTE 'SELECT ' || r.colname || ' FROM brintest_multi WHERE ' || cond LOOP
309				RAISE NOTICE 'seqscan: %', r2;
310			END LOOP;
311
312			SET enable_seqscan = 0;
313			SET enable_bitmapscan = 1;
314			FOR r2 IN EXECUTE 'SELECT ' || r.colname || ' FROM brintest_multi WHERE ' || cond LOOP
315				RAISE NOTICE 'bitmapscan: %', r2;
316			END LOOP;
317		END IF;
318
319		-- make sure we found expected number of matches
320		IF count != r.matches THEN RAISE WARNING 'unexpected number of results % for %', count, r; END IF;
321	END LOOP;
322END;
323$x$;

325RESET enable_seqscan;
326RESET enable_bitmapscan;
