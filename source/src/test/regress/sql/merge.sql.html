607create or replace function merge_trigfunc () returns trigger
608language plpgsql as
609$$
610DECLARE
611	line text;
612BEGIN
613	SELECT INTO line format('%s %s %s trigger%s',
614		TG_WHEN, TG_OP, TG_LEVEL, CASE
615		WHEN TG_OP = 'INSERT' AND TG_LEVEL = 'ROW'
616			THEN format(' row: %s', NEW)
617		WHEN TG_OP = 'UPDATE' AND TG_LEVEL = 'ROW'
618			THEN format(' row: %s -> %s', OLD, NEW)
619		WHEN TG_OP = 'DELETE' AND TG_LEVEL = 'ROW'
620			THEN format(' row: %s', OLD)
621		END);
622
623	RAISE NOTICE '%', line;
624	IF (TG_WHEN = 'BEFORE' AND TG_LEVEL = 'ROW') THEN
625		IF (TG_OP = 'DELETE') THEN
626			RETURN OLD;
627		ELSE
628			RETURN NEW;
629		END IF;
630	ELSE
631		RETURN NULL;
632	END IF;
633END;
634$$;
