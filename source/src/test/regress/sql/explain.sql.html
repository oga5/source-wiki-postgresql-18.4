1--
2-- EXPLAIN
3--
4-- There are many test cases elsewhere that use EXPLAIN as a vehicle for
5-- checking something else (usually planner behavior).  This file is
6-- concerned with testing EXPLAIN in its own right.
7--
8
9-- To produce stable regression test output, it's usually necessary to
10-- ignore details such as exact costs or row counts.  These filter
11-- functions replace changeable output details with fixed strings.
12
13create function explain_filter(text) returns setof text
14language plpgsql as
15$$
16declare
17    ln text;
18begin
19    for ln in execute $1
20    loop
21        -- Replace any numeric word with just 'N'
22        ln := regexp_replace(ln, '-?\m\d+\M', 'N', 'g');
23        -- In sort output, the above won't match units-suffixed numbers
24        ln := regexp_replace(ln, '\m\d+kB', 'NkB', 'g');
25        -- Ignore text-mode buffers output because it varies depending
26        -- on the system state
27        CONTINUE WHEN (ln ~ ' +Buffers: .*');
28        -- Ignore text-mode "Planning:" line because whether it's output
29        -- varies depending on the system state
30        CONTINUE WHEN (ln = 'Planning:');
31        return next ln;
32    end loop;
33end;
34$$;
35
36-- To produce valid JSON output, replace numbers with "0" or "0.0" not "N"
37create function explain_filter_to_json(text) returns jsonb
38language plpgsql as
39$$
40declare
41    data text := '';
42    ln text;
43begin
44    for ln in execute $1
45    loop
46        -- Replace any numeric word with just '0'
47        ln := regexp_replace(ln, '\m\d+\M', '0', 'g');
48        data := data || ln;
49    end loop;
50    return data::jsonb;
51end;
52$$;

110-- Test EXPLAIN (GENERIC_PLAN) with partition pruning
111-- partitions should be pruned at plan time, based on constants,
112-- but there should be no pruning based on parameter placeholders
113create table gen_part (
114  key1 integer not null,
115  key2 integer not null
116) partition by list (key1);
117create table gen_part_1
118  partition of gen_part for values in (1)
119  partition by range (key2);
120create table gen_part_1_1
121  partition of gen_part_1 for values from (1) to (2);
122create table gen_part_1_2
123  partition of gen_part_1 for values from (2) to (3);
124create table gen_part_2
125  partition of gen_part for values in (2);
126-- should scan gen_part_1_1 and gen_part_1_2, but not gen_part_2
127select explain_filter('explain (generic_plan) select key1, key2 from gen_part where key1 = 1 and key2 = $1');
128drop table gen_part;

131-- Test production of per-worker data
132--
133-- Unfortunately, because we don't know how many worker processes we'll
134-- actually get (maybe none at all), we can't examine the "Workers" output
135-- in any detail.  We can check that it parses correctly as JSON, and then
136-- remove it from the displayed results.

142set min_parallel_table_scan_size=0;
143set max_parallel_workers_per_gather=4;
144
145select jsonb_pretty(
146  explain_filter_to_json('explain (analyze, verbose, buffers, format json)
147                         select * from tenk1 order by tenthous')
148  -- remove "Workers" node of the Seq Scan plan node
149  #- '{0,Plan,Plans,0,Plans,0,Workers}'
150  -- remove "Workers" node of the Sort plan node
151  #- '{0,Plan,Plans,0,Workers}'
152  -- Also remove its sort-type fields, as those aren't 100% stable
153  #- '{0,Plan,Plans,0,Sort Method}'
154  #- '{0,Plan,Plans,0,Sort Space Type}'
155);
