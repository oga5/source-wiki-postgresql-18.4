187-- Check behavior of SIMILAR TO, which uses largely the same regexp variant
188SELECT 'abcdefg' SIMILAR TO '_bcd%' AS true;
189SELECT 'abcdefg' SIMILAR TO 'bcd%' AS false;
190SELECT 'abcdefg' SIMILAR TO '_bcd#%' ESCAPE '#' AS false;
191SELECT 'abcd%' SIMILAR TO '_bcd#%' ESCAPE '#' AS true;
192-- Postgres uses '\' as the default escape character, which is not per spec
193SELECT 'abcdefg' SIMILAR TO '_bcd\%' AS false;
194-- and an empty string to mean "no escape", which is also not per spec
195SELECT 'abcd\efg' SIMILAR TO '_bcd\%' ESCAPE '' AS true;
196-- these behaviors are per spec, though:
197SELECT 'abcdefg' SIMILAR TO '_bcd%' ESCAPE NULL AS null;
198SELECT 'abcdefg' SIMILAR TO '_bcd#%' ESCAPE '##' AS error;

200-- Characters that should be left alone in character classes when a
201-- SIMILAR TO regexp pattern is converted to POSIX style.
202-- Underscore "_"
203EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '_[_[:alpha:]_]_';
204-- Percentage "%"
205EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '%[%[:alnum:]%]%';
206-- Dot "."
207EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '.[.[:alnum:].].';
208-- Dollar "$"
209EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '$[$[:alnum:]$]$';
210-- Opening parenthesis "("
211EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '()[([:alnum:](]()';
212-- Caret "^"
213EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '^[^[:alnum:]^[^^][[^^]][\^][[\^]]\^]^';
214-- Closing square bracket "]" at the beginning of character class
215EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '[]%][^]%][^%]%';
216-- Closing square bracket effective after two carets at the beginning
217-- of character class.
218EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '[^^]^';
219-- Closing square bracket after an escape sequence at the beginning of
220-- a character closes the character class
221EXPLAIN (COSTS OFF) SELECT * FROM TEXT_TBL WHERE f1 SIMILAR TO '[|a]%' ESCAPE '|';
