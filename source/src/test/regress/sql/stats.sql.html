118SELECT st.seq_scan >= pr.seq_scan + 1,
119       st.seq_tup_read >= pr.seq_tup_read + cl.reltuples,

127  FROM pg_statio_user_tables AS st, pg_class AS cl, prevstats AS pr
128 WHERE st.relname='tenk2' AND cl.relname='tenk2';

131FROM prevstats AS pr;

300-- Test that last_seq_scan, last_idx_scan are correctly maintained
301--
302-- Perform test using a temporary table. That way autovacuum etc won't
303-- interfere. To be able to check that timestamps increase, we sleep for 100ms
304-- between tests, assuming that there aren't systems with a coarser timestamp
305-- granularity.
306-----
307
308BEGIN;
309CREATE TEMPORARY TABLE test_last_scan(idx_col int primary key, noidx_col int);
310INSERT INTO test_last_scan(idx_col, noidx_col) VALUES(1, 1);
311SELECT pg_stat_force_next_flush();
312SELECT last_seq_scan, last_idx_scan FROM pg_stat_all_tables WHERE relid = 'test_last_scan'::regclass;
313COMMIT;
314
315SELECT pg_stat_reset_single_table_counters('test_last_scan'::regclass);
316SELECT seq_scan, idx_scan FROM pg_stat_all_tables WHERE relid = 'test_last_scan'::regclass;
317
318-- ensure we start out with exactly one index and sequential scan
319BEGIN;
320SET LOCAL enable_seqscan TO on;
321SET LOCAL enable_indexscan TO on;
322SET LOCAL enable_bitmapscan TO off;
323EXPLAIN (COSTS off) SELECT count(*) FROM test_last_scan WHERE noidx_col = 1;
324SELECT count(*) FROM test_last_scan WHERE noidx_col = 1;
325SET LOCAL enable_seqscan TO off;
326EXPLAIN (COSTS off) SELECT count(*) FROM test_last_scan WHERE idx_col = 1;
327SELECT count(*) FROM test_last_scan WHERE idx_col = 1;
328SELECT pg_stat_force_next_flush();
329COMMIT;
330
331-- fetch timestamps from before the next test
332SELECT last_seq_scan AS test_last_seq, last_idx_scan AS test_last_idx
333FROM pg_stat_all_tables WHERE relid = 'test_last_scan'::regclass \gset
334SELECT pg_sleep(0.1); -- assume a minimum timestamp granularity of 100ms
335
336-- cause one sequential scan
337BEGIN;
338SET LOCAL enable_seqscan TO on;
339SET LOCAL enable_indexscan TO off;
340SET LOCAL enable_bitmapscan TO off;
341EXPLAIN (COSTS off) SELECT count(*) FROM test_last_scan WHERE noidx_col = 1;
342SELECT count(*) FROM test_last_scan WHERE noidx_col = 1;
343SELECT pg_stat_force_next_flush();
344COMMIT;
345-- check that just sequential scan stats were incremented
346SELECT seq_scan, :'test_last_seq' < last_seq_scan AS seq_ok, idx_scan, :'test_last_idx' = last_idx_scan AS idx_ok
347FROM pg_stat_all_tables WHERE relid = 'test_last_scan'::regclass;

542BEGIN;

545SELECT pg_stat_get_function_calls(0);

849DECLARE
850  start_time timestamptz := clock_timestamp();
851  updated bool;
852BEGIN
853  -- we don't want to wait forever; loop will exit after 30 seconds
854  FOR i IN 1 .. 300 LOOP

862  END LOOP;
863  -- report time waited in postmaster log (where it won't change test output)
864  RAISE log 'wait_for_hot_stats delayed % seconds',
865    EXTRACT(epoch FROM clock_timestamp() - start_time);
866END
