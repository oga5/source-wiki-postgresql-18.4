91BEGIN;
92SET LOCAL TimeZone TO 'UTC';
93CREATE VIEW timetz_local_view AS
94  SELECT f1 AS dat,
95       timezone(f1) AS dat_func,
96       f1 AT LOCAL AS dat_at_local,
97       f1 AT TIME ZONE current_setting('TimeZone') AS dat_at_tz,
98       f1 AT TIME ZONE INTERVAL '00:00' AS dat_at_int
99  FROM TIMETZ_TBL
100  ORDER BY f1;
101SELECT pg_get_viewdef('timetz_local_view', true);
102TABLE timetz_local_view;
103SELECT f1 AS dat,
104       f1 AT TIME ZONE 'UTC+10' AS dat_at_tz,
105       f1 AT TIME ZONE INTERVAL '-10:00' AS dat_at_int
106  FROM TIMETZ_TBL
