169create table tab_progress_reporting (
170	name text,
171	age int4,
172	location point,
173	salary int4,
174	manager name
175);

177-- Add a trigger to catch and print the contents of the catalog view
178-- pg_stat_progress_copy during data insertion.  This allows to test
179-- the validation of some progress reports for COPY FROM where the trigger
180-- would fire.
181create function notice_after_tab_progress_reporting() returns trigger AS
182$$
183declare report record;
184begin
185  -- The fields ignored here are the ones that may not remain
186  -- consistent across multiple runs.  The sizes reported may differ
187  -- across platforms, so just check if these are strictly positive.
188  with progress_data as (
189    select
190       relid::regclass::text as relname,
191       command,
192       type,
193       bytes_processed > 0 as has_bytes_processed,
194       bytes_total > 0 as has_bytes_total,
195       tuples_processed,
196       tuples_excluded,
197       tuples_skipped
198      from pg_stat_progress_copy
199      where pid = pg_backend_pid())
200  select into report (to_jsonb(r)) as value
201    from progress_data r;
202
203  raise info 'progress: %', report.value::text;
204  return new;
205end;
206$$ language plpgsql;
207
208create trigger check_after_tab_progress_reporting
209	after insert on tab_progress_reporting
210	for each statement
211	execute function notice_after_tab_progress_reporting();

324--
325-- Create partitioned table that does not allow bulk insertions, to test bugs
326-- related to the reuse of BulkInsertState across partitions (only done when
327-- not using bulk insert).  Switching between partitions often makes it more
328-- likely to encounter these bugs, so we just switch on roughly every insert
329-- by having an even/odd number partition and inserting evenly distributed
330-- data.
331--
332CREATE TABLE parted_si (
333  id int not null,
334  data text not null,
335  -- prevent use of bulk insert by having a volatile function
336  rand float8 not null default random()
337)
338PARTITION BY LIST((id % 2));
339
340CREATE TABLE parted_si_p_even PARTITION OF parted_si FOR VALUES IN (0);
341CREATE TABLE parted_si_p_odd PARTITION OF parted_si FOR VALUES IN (1);
342
343-- Test that bulk relation extension handles reusing a single BulkInsertState
344-- across partitions.  Without the fix applied, this reliably reproduces
345-- #18130 unless shared_buffers is extremely small (preventing any use of bulk
346-- relation extension). See
347-- https://postgr.es/m/18130-7a86a7356a75209d%40postgresql.org
348-- https://postgr.es/m/257696.1695670946%40sss.pgh.pa.us
349\set filename :abs_srcdir '/data/desc.data'
350COPY parted_si(id, data) FROM :'filename';
351
352-- An earlier bug (see commit b1ecb9b3fcf) could end up using a buffer from
353-- the wrong partition. This test is *not* guaranteed to trigger that bug, but
354-- does so when shared_buffers is small enough.  To test if we encountered the
355-- bug, check that the partition condition isn't violated.
356SELECT tableoid::regclass, id % 2 = 0 is_even, count(*) from parted_si GROUP BY 1, 2 ORDER BY 1;
357
358DROP TABLE parted_si;
