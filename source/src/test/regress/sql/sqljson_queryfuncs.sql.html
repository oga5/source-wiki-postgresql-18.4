323CREATE TABLE test_jsonb_constraints (
324	js text,
325	i int,
326	x jsonb DEFAULT JSON_QUERY(jsonb '[1,2]', '$[*]' WITH WRAPPER)
327	CONSTRAINT test_jsonb_constraint1
328		CHECK (js IS JSON)
329	CONSTRAINT test_jsonb_constraint2
330		CHECK (JSON_EXISTS(js::jsonb, '$.a' PASSING i + 5 AS int, i::text AS "TXT", array[1,2,3] as arr))
331	CONSTRAINT test_jsonb_constraint3
332		CHECK (JSON_VALUE(js::jsonb, '$.a' RETURNING int DEFAULT '12' ON EMPTY ERROR ON ERROR) > i)
333	CONSTRAINT test_jsonb_constraint4
334		CHECK (JSON_QUERY(js::jsonb, '$.a' WITH CONDITIONAL WRAPPER EMPTY OBJECT ON ERROR) = jsonb '[10]')
335	CONSTRAINT test_jsonb_constraint5
336		CHECK (JSON_QUERY(js::jsonb, '$.a' RETURNING char(5) OMIT QUOTES EMPTY ARRAY ON EMPTY) >  'a' COLLATE "C")
337);
