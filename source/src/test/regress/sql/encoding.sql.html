99-- mb<->wchar conversions
100CREATE FUNCTION test_encoding(encoding text, description text, input bytea)
101RETURNS VOID LANGUAGE plpgsql AS
102$$
103DECLARE
104	prefix text;
105	len int;
106	wchars int[];
107	round_trip bytea;
108	result text;
109BEGIN
110	prefix := rpad(encoding || ' ' || description || ':', 28);
111
112	-- XXX could also test validation, length functions and include client
113	-- only encodings with these test cases
114
115	IF test_valid_server_encoding(encoding) THEN
116		wchars := test_text_to_wchars(encoding, test_bytea_to_text(input));
117		round_trip = test_text_to_bytea(test_wchars_to_text(encoding, wchars));
118		if input = round_trip then
119			result := 'OK';
120		elsif length(input) > length(round_trip) and round_trip = substr(input, 1, length(round_trip)) then
121			result := 'truncated';
122		else
123			result := 'failed';
124		end if;
125		RAISE NOTICE '% % -> % -> % = %', prefix, input, wchars, round_trip, result;
126	END IF;
127END;
128$$;
