211DO $x$
212DECLARE
213	r record;
214	r2 record;
215	cond text;
216	idx_ctids tid[];
217	ss_ctids tid[];
218	count int;
219	plan_ok bool;
220	plan_line text;
221BEGIN
222	FOR r IN SELECT colname, oper, typ, value[ordinality], matches[ordinality] FROM brinopers_bloom, unnest(op) WITH ORDINALITY AS oper LOOP
223
224		-- prepare the condition
225		IF r.value IS NULL THEN
226			cond := format('%I %s %L', r.colname, r.oper, r.value);
227		ELSE
228			cond := format('%I %s %L::%s', r.colname, r.oper, r.value, r.typ);
229		END IF;
230
231		-- run the query using the brin index
232		SET enable_seqscan = 0;
233		SET enable_bitmapscan = 1;
234
235		plan_ok := false;
236		FOR plan_line IN EXECUTE format($y$EXPLAIN SELECT array_agg(ctid) FROM brintest_bloom WHERE %s $y$, cond) LOOP
237			IF plan_line LIKE '%Bitmap Heap Scan on brintest_bloom%' THEN
238				plan_ok := true;
239			END IF;
240		END LOOP;
241		IF NOT plan_ok THEN
242			RAISE WARNING 'did not get bitmap indexscan plan for %', r;
243		END IF;
244
245		EXECUTE format($y$SELECT array_agg(ctid) FROM brintest_bloom WHERE %s $y$, cond)
246			INTO idx_ctids;
247
248		-- run the query using a seqscan
249		SET enable_seqscan = 1;
250		SET enable_bitmapscan = 0;
251
252		plan_ok := false;
253		FOR plan_line IN EXECUTE format($y$EXPLAIN SELECT array_agg(ctid) FROM brintest_bloom WHERE %s $y$, cond) LOOP
254			IF plan_line LIKE '%Seq Scan on brintest_bloom%' THEN
255				plan_ok := true;
256			END IF;
257		END LOOP;
258		IF NOT plan_ok THEN
259			RAISE WARNING 'did not get seqscan plan for %', r;
260		END IF;
261
262		EXECUTE format($y$SELECT array_agg(ctid) FROM brintest_bloom WHERE %s $y$, cond)
263			INTO ss_ctids;
264
265		-- make sure both return the same results
266		count := array_length(idx_ctids, 1);
267
268		IF NOT (count = array_length(ss_ctids, 1) AND
269				idx_ctids @> ss_ctids AND
270				idx_ctids <@ ss_ctids) THEN
271			-- report the results of each scan to make the differences obvious
272			RAISE WARNING 'something not right in %: count %', r, count;
273			SET enable_seqscan = 1;
274			SET enable_bitmapscan = 0;
275			FOR r2 IN EXECUTE 'SELECT ' || r.colname || ' FROM brintest_bloom WHERE ' || cond LOOP
276				RAISE NOTICE 'seqscan: %', r2;
277			END LOOP;
278
279			SET enable_seqscan = 0;
280			SET enable_bitmapscan = 1;
281			FOR r2 IN EXECUTE 'SELECT ' || r.colname || ' FROM brintest_bloom WHERE ' || cond LOOP
282				RAISE NOTICE 'bitmapscan: %', r2;
283			END LOOP;
284		END IF;
285
286		-- make sure we found expected number of matches
287		IF count != r.matches THEN RAISE WARNING 'unexpected number of results % for %', count, r; END IF;
288	END LOOP;
289END;
290$x$;
291
292RESET enable_seqscan;
293RESET enable_bitmapscan;
