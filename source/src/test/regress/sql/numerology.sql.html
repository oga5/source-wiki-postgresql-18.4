149--
150-- Group-by combinations
151--
152
153CREATE TABLE TEMP_GROUP (f1 INT4, f2 INT4, f3 FLOAT8);
154
155INSERT INTO TEMP_GROUP
156  SELECT 1, (- i.f1), (- f.f1)
157  FROM INT4_TBL i, FLOAT8_TBL f;
158
159INSERT INTO TEMP_GROUP
160  SELECT 2, i.f1, f.f1
161  FROM INT4_TBL i, FLOAT8_TBL f;
162
163SELECT DISTINCT f1 AS two FROM TEMP_GROUP ORDER BY 1;
164
165SELECT f1 AS two, max(f3) AS max_float, min(f3) as min_float
166  FROM TEMP_GROUP
167  GROUP BY f1
168  ORDER BY two, max_float, min_float;
169
170-- GROUP BY a result column name is not legal per SQL92, but we accept it
171-- anyway (if the name is not the name of any column exposed by FROM).
172SELECT f1 AS two, max(f3) AS max_float, min(f3) AS min_float
173  FROM TEMP_GROUP
174  GROUP BY two
175  ORDER BY two, max_float, min_float;
176
177SELECT f1 AS two, (max(f3) + 1) AS max_plus_1, (min(f3) - 1) AS min_minus_1
178  FROM TEMP_GROUP
179  GROUP BY f1
180  ORDER BY two, min_minus_1;
181
182SELECT f1 AS two,
183       max(f2) + min(f2) AS max_plus_min,
184       min(f3) - 1 AS min_minus_1
185  FROM TEMP_GROUP
186  GROUP BY f1
187  ORDER BY two, min_minus_1;
