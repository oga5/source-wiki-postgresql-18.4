279-- LIKE ... INCLUDING STATISTICS with dropped columns in the parent,
280-- so stxkeys attnums are not contiguous.
281CREATE TABLE ctl_stats3_parent (a int, b int, c int);
282ALTER TABLE ctl_stats3_parent DROP COLUMN b;
283CREATE STATISTICS ctl_stats3_stat ON a, c FROM ctl_stats3_parent;
284CREATE TABLE ctl_stats3_child (LIKE ctl_stats3_parent INCLUDING STATISTICS);
285CREATE TABLE ctl_stats4_parent (a int, b int, c int, d int);
286ALTER TABLE ctl_stats4_parent DROP COLUMN b;
287CREATE STATISTICS ctl_stats4_stat ON a, c FROM ctl_stats4_parent;
288CREATE TABLE ctl_stats4_child (LIKE ctl_stats4_parent INCLUDING STATISTICS);
289SELECT s.stxrelid::regclass AS relation,
290       array_agg(a.attname ORDER BY u.ord) AS stats_columns
291FROM pg_statistic_ext s
292CROSS JOIN LATERAL
293  unnest(s.stxkeys::int2[]) WITH ORDINALITY AS u(attnum, ord)
294JOIN pg_attribute a
295  ON a.attrelid = s.stxrelid AND a.attnum = u.attnum
296WHERE s.stxrelid IN ('ctl_stats3_child'::regclass,
297                     'ctl_stats4_child'::regclass)
298GROUP BY s.stxrelid
299ORDER BY s.stxrelid::regclass::text;
