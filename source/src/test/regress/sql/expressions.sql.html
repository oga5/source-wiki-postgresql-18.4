82--
83-- Ordinarily, IN/NOT IN can be converted to a ScalarArrayOpExpr
84-- with a suitably-chosen array type.
85--
86explain (verbose, costs off)
87select random() IN (1, 4, 8.0);
88explain (verbose, costs off)
89select random()::int IN (1, 4, 8.0);
90-- However, if there's not a common supertype for the IN elements,
91-- we should instead try to produce "x = v1 OR x = v2 OR ...".
92-- In most cases that'll fail for lack of all the requisite = operators,
93-- but it can succeed sometimes.  So this should complain about lack of
94-- an = operator, not about cast failure.
95select '(0,0)'::point in ('(0,0,0,0)'::box, point(0,0));

138-- We need to create our own type for this.

140begin;
141
142create type myint;
143create function myintin(cstring) returns myint strict immutable language
144  internal as 'int4in';
145create function myintout(myint) returns cstring strict immutable language
146  internal as 'int4out';
147create function myinthash(myint) returns integer strict immutable language
148  internal as 'hashint4';
149
150create type myint (input = myintin, output = myintout, like = int4);
151
152create cast (int4 as myint) without function;
153create cast (myint as int4) without function;
154
155create function myinteq(myint, myint) returns bool as $$
156begin
157  if $1 is null and $2 is null then
158    return true;
159  else
160    return $1::int = $2::int;
161  end if;
162end;
163$$ language plpgsql immutable;
164
165create function myintne(myint, myint) returns bool as $$
166begin
167  return not myinteq($1, $2);
168end;
169$$ language plpgsql immutable;
170
171create operator = (
172  leftarg    = myint,
173  rightarg   = myint,
174  commutator = =,
175  negator    = <>,
176  procedure  = myinteq,
177  restrict   = eqsel,
178  join       = eqjoinsel,
179  merges
180);
181
182create operator <> (
183  leftarg    = myint,
184  rightarg   = myint,
185  commutator = <>,
186  negator    = =,
187  procedure  = myintne,
188  restrict   = eqsel,
189  join       = eqjoinsel,
190  merges
191);
