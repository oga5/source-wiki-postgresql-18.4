39SELECT * FROM CIRCLE_TBL;
40
41SELECT center(f1) AS center
42  FROM CIRCLE_TBL;
43
44SELECT radius(f1) AS radius
45  FROM CIRCLE_TBL;
46
47SELECT diameter(f1) AS diameter
48  FROM CIRCLE_TBL;
49
50SELECT f1 FROM CIRCLE_TBL WHERE radius(f1) < 5;
51
52SELECT f1 FROM CIRCLE_TBL WHERE diameter(f1) >= 10;
53
54SELECT c1.f1 AS one, c2.f1 AS two, (c1.f1 <-> c2.f1) AS distance
55  FROM CIRCLE_TBL c1, CIRCLE_TBL c2
56  WHERE (c1.f1 < c2.f1) AND ((c1.f1 <-> c2.f1) > 0)
57  ORDER BY distance, area(c1.f1), area(c2.f1);
