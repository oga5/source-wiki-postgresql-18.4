343--
344-- test output (and round-trip safety) of various values.
345-- To ensure we're testing what we think we're testing, start with
346-- float values specified by bit patterns (as a useful side effect,
347-- this means we'll fail on non-IEEE platforms).
348
349create type xfloat8;
350create function xfloat8in(cstring) returns xfloat8 immutable strict
351  language internal as 'int8in';
352create function xfloat8out(xfloat8) returns cstring immutable strict
353  language internal as 'int8out';
354create type xfloat8 (input = xfloat8in, output = xfloat8out, like = no_such_type);
355create type xfloat8 (input = xfloat8in, output = xfloat8out, like = float8);
356create cast (xfloat8 as float8) without function;
357create cast (float8 as xfloat8) without function;
358create cast (xfloat8 as bigint) without function;
359create cast (bigint as xfloat8) without function;
