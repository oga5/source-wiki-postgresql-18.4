288-- UPDATE on view with INSTEAD OF trigger
289CREATE FUNCTION joinview_upd_trig_fn() RETURNS trigger
290LANGUAGE plpgsql AS
291$$
292BEGIN
293  RAISE NOTICE 'UPDATE: % -> %', old, new;
294  UPDATE foo SET f1 = new.f1, f3 = new.f3, f4 = new.f4 * 10
295    FROM joinme WHERE f2 = f2j AND f2 = old.f2
296    RETURNING new.f1, new.f4 INTO new.f1, new.f4;  -- should fail
297  RETURN NEW;
298END;
299$$;
300CREATE TRIGGER joinview_upd_trig INSTEAD OF UPDATE ON joinview
301  FOR EACH ROW EXECUTE FUNCTION joinview_upd_trig_fn();
302DROP RULE joinview_u ON joinview;
303UPDATE joinview SET f3 = f3 + 1, f4 = 7 WHERE f3 = 58
304  RETURNING old.*, new.*, *, new.f3 - old.f3 AS delta_f3;  -- should fail

306CREATE OR REPLACE FUNCTION joinview_upd_trig_fn() RETURNS trigger
307LANGUAGE plpgsql AS
308$$
309BEGIN
310  RAISE NOTICE 'UPDATE: % -> %', old, new;
311  UPDATE foo SET f1 = new.f1, f3 = new.f3, f4 = new.f4 * 10
312    FROM joinme WHERE f2 = f2j AND f2 = old.f2
313    RETURNING WITH (new AS n) new.f1, n.f4 INTO new.f1, new.f4;  -- now ok

315END;

387-- Test deparsing
388CREATE FUNCTION foo_update()
389  RETURNS void
390  LANGUAGE sql
391BEGIN ATOMIC
392  WITH u1 AS (
393    UPDATE foo SET f1 = f1 + 1 RETURNING old.*, new.*
394  ), u2 AS (
395    UPDATE foo SET f1 = f1 + 1 RETURNING WITH (OLD AS "old foo") "old foo".*, new.*
396  ), u3 AS (
397    UPDATE foo SET f1 = f1 + 1 RETURNING WITH (NEW AS "new foo") old.*, "new foo".*
398  )
399  UPDATE foo SET f1 = f1 + 1
400    RETURNING WITH (OLD AS o, NEW AS n)
401              o.*, n.*, o, n, o.f1 = n.f1, o = n,
402              (SELECT o.f2 = n.f2),
403              (SELECT count(*) FROM foo WHERE foo.f1 = o.f4),
404              (SELECT count(*) FROM foo WHERE foo.f4 = n.f4),
405              (SELECT count(*) FROM foo WHERE foo = o),
406              (SELECT count(*) FROM foo WHERE foo = n);
407END;
