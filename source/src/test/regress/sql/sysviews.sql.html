1--
2-- Test assorted system views
3--
4-- This test is mainly meant to provide some code coverage for the
5-- set-returning functions that underlie certain system views.
6-- The output of most of these functions is very environment-dependent,
7-- so our ability to test with fixed expected output is pretty limited;
8-- but even a trivial check of count(*) will exercise the normal code path
9-- through the SRF.
10
11select count(*) >= 0 as ok from pg_available_extension_versions;
12
13select count(*) >= 0 as ok from pg_available_extensions;
14
15-- The entire output of pg_backend_memory_contexts is not stable,
16-- we test only the existence and basic condition of TopMemoryContext.
17select type, name, ident, level, total_bytes >= free_bytes
18  from pg_backend_memory_contexts where level = 1;
19
20-- We can exercise some MemoryContext type stats functions.  Most of the
21-- column values are too platform-dependant to display.
22
23-- Ensure stats from the bump allocator look sane.  Bump isn't a commonly
24-- used context, but it is used in tuplesort.c, so open a cursor to keep
25-- the tuplesort alive long enough for us to query the context stats.
26begin;
27declare cur cursor for select left(a,10), b
28  from (values(repeat('a', 512 * 1024),1),(repeat('b', 512),2)) v(a,b)
29  order by v.a desc;
30fetch 1 from cur;
31select type, name, total_bytes > 0, total_nblocks, free_bytes > 0, free_chunks
32from pg_backend_memory_contexts where name = 'Caller tuples';
33rollback;
34
35-- Further sanity checks on pg_backend_memory_contexts.  We expect
36-- CacheMemoryContext to have multiple children.  Ensure that's the case.
37with contexts as (
38  select * from pg_backend_memory_contexts
39)
40select count(*) > 1
41from contexts c1, contexts c2
42where c2.name = 'CacheMemoryContext'
43and c1.path[c2.level] = c2.path[c2.level];
44
45-- At introduction, pg_config had 23 entries; it may grow
46select count(*) > 20 as ok from pg_config;
47
48-- We expect no cursors in this test; see also portals.sql
49select count(*) = 0 as ok from pg_cursors;
50
51select count(*) >= 0 as ok from pg_file_settings;
52
53-- There will surely be at least one rule, with no errors.
54select count(*) > 0 as ok, count(*) FILTER (WHERE error IS NOT NULL) = 0 AS no_err
55  from pg_hba_file_rules;
56
57-- There may be no rules, and there should be no errors.
58select count(*) >= 0 as ok, count(*) FILTER (WHERE error IS NOT NULL) = 0 AS no_err
59  from pg_ident_file_mappings;
60
61-- There will surely be at least one active lock
62select count(*) > 0 as ok from pg_locks;
63
64-- We expect no prepared statements in this test; see also prepare.sql
65select count(*) = 0 as ok from pg_prepared_statements;
66
67-- See also prepared_xacts.sql
68select count(*) >= 0 as ok from pg_prepared_xacts;
69
70-- There will surely be at least one SLRU cache
71select count(*) > 0 as ok from pg_stat_slru;
72
73-- There must be only one record
74select count(*) = 1 as ok from pg_stat_wal;
75
76-- We expect no walreceiver running in this test
77select count(*) = 0 as ok from pg_stat_wal_receiver;
78
79-- This is to record the prevailing planner enable_foo settings during
80-- a regression test run.
81select name, setting from pg_settings where name like 'enable%';
82
83-- There are always wait event descriptions for various types.  InjectionPoint
84-- may be present or absent, depending on history since last postmaster start.
85select type, count(*) > 0 as ok FROM pg_wait_events
86  where type <> 'InjectionPoint' group by type order by type COLLATE "C";
87
88-- Test that the pg_timezone_names and pg_timezone_abbrevs views are
89-- more-or-less working.  We can't test their contents in any great detail
90-- without the outputs changing anytime IANA updates the underlying data,
91-- but it seems reasonable to expect at least one entry per major meridian.
92-- (At the time of writing, the actual counts are around 38 because of
93-- zones using fractional GMT offsets, so this is a pretty loose test.)
94select count(distinct utc_offset) >= 24 as ok from pg_timezone_names;
95select count(distinct utc_offset) >= 24 as ok from pg_timezone_abbrevs;
96-- Let's check the non-default timezone abbreviation sets, too
97set timezone_abbreviations = 'Australia';
98select count(distinct utc_offset) >= 24 as ok from pg_timezone_abbrevs;
99set timezone_abbreviations = 'India';
100select count(distinct utc_offset) >= 24 as ok from pg_timezone_abbrevs;
101-- One specific case we can check without much fear of breakage
102-- is the historical local-mean-time value used for America/Los_Angeles.
103select * from pg_timezone_abbrevs where abbrev = 'LMT';
