3-- Test random() and allies
4--
5-- Tests in this file may have a small probability of failure,
6-- since we are dealing with randomness.  Try to keep the failure
7-- risk for any one test case under 1e-9.
8--
9
10-- There should be no duplicates in 1000 random() values.
11-- (Assuming 52 random bits in the float8 results, we could
12-- take as many as 3000 values and still have less than 1e-9 chance
13-- of failure, per https://en.wikipedia.org/wiki/Birthday_problem)
14SELECT r, count(*)
15FROM (SELECT random() r FROM generate_series(1, 1000)) ss
16GROUP BY r HAVING count(*) > 1;
17
18-- The range should be [0, 1).  We can expect that at least one out of 2000
19-- random values is in the lowest or highest 1% of the range with failure
20-- probability less than about 1e-9.
21
22SELECT count(*) FILTER (WHERE r < 0 OR r >= 1) AS out_of_range,
23       (count(*) FILTER (WHERE r < 0.01)) > 0 AS has_small,
24       (count(*) FILTER (WHERE r > 0.99)) > 0 AS has_large
25FROM (SELECT random() r FROM generate_series(1, 2000)) ss;

27-- Check for uniform distribution using the Kolmogorov-Smirnov test.
28
29CREATE FUNCTION ks_test_uniform_random()
30RETURNS boolean AS
31$$
32DECLARE
33  n int := 1000;        -- Number of samples
34  c float8 := 1.94947;  -- Critical value for 99.9% confidence
35  ok boolean;
36BEGIN
37  ok := (
38    WITH samples AS (
39      SELECT random() r FROM generate_series(1, n) ORDER BY 1
40    ), indexed_samples AS (
41      SELECT (row_number() OVER())-1.0 i, r FROM samples
42    )
43    SELECT max(abs(i/n-r)) < c / sqrt(n) FROM indexed_samples
44  );
45  RETURN ok;
46END
47$$
48LANGUAGE plpgsql;
49
50-- As written, ks_test_uniform_random() returns true about 99.9%
51-- of the time.  To get down to a roughly 1e-9 test failure rate,
52-- just run it 3 times and accept if any one of them passes.
53SELECT ks_test_uniform_random() OR
54       ks_test_uniform_random() OR
55       ks_test_uniform_random() AS uniform;

73-- Check standard normal distribution using the Kolmogorov-Smirnov test.
74
75CREATE FUNCTION ks_test_normal_random()
76RETURNS boolean AS
77$$
78DECLARE
79  n int := 1000;        -- Number of samples
80  c float8 := 1.94947;  -- Critical value for 99.9% confidence
81  ok boolean;
82BEGIN
83  ok := (
84    WITH samples AS (
85      SELECT random_normal() r FROM generate_series(1, n) ORDER BY 1
86    ), indexed_samples AS (
87      SELECT (row_number() OVER())-1.0 i, r FROM samples
88    )
89    SELECT max(abs((1+erf(r/sqrt(2)))/2 - i/n)) < c / sqrt(n)
90    FROM indexed_samples
91  );
92  RETURN ok;
93END
94$$
95LANGUAGE plpgsql;

119-- There should be no triple duplicates in 1000 full-range 32-bit random()
120-- values.  (Each of the C(1000, 3) choices of triplets from the 1000 values
121-- has a probability of 1/(2^32)^2 of being a triple duplicate, so the
122-- average number of triple duplicates is 1000 * 999 * 998 / 6 / 2^64, which
123-- is roughly 9e-12.)

125FROM (SELECT random(-2147483648, 2147483647) r
126      FROM generate_series(1, 1000)) ss
127GROUP BY r HAVING count(*) > 2;

184RETURNS boolean AS
185$$

198  );
199  RETURN ok;
200END
201$$
