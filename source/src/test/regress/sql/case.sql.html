175--
176-- Nested CASE expressions
177--
178
179-- This test exercises a bug caused by aliasing econtext->caseValue_isNull
180-- with the isNull argument of the inner CASE's CaseExpr evaluation.  After
181-- evaluating the vol(null) expression in the inner CASE's second WHEN-clause,
182-- the isNull flag for the case test value incorrectly became true, causing
183-- the third WHEN-clause not to match.  The volatile function calls are needed
184-- to prevent constant-folding in the planner, which would hide the bug.
185
186-- Wrap this in a single transaction so the transient '=' operator doesn't
187-- cause problems in concurrent sessions
188BEGIN;
189
190CREATE FUNCTION vol(text) returns text as
191  'begin return $1; end' language plpgsql volatile;
192
193SELECT CASE
194  (CASE vol('bar')
195    WHEN 'foo' THEN 'it was foo!'
196    WHEN vol(null) THEN 'null input'
197    WHEN 'bar' THEN 'it was bar!' END
198  )
199  WHEN 'it was foo!' THEN 'foo recognized'
200  WHEN 'it was bar!' THEN 'bar recognized'
201  ELSE 'unrecognized' END;
202
203-- In this case, we can't inline the SQL function without confusing things.
204CREATE DOMAIN foodomain AS text;
205
206CREATE FUNCTION volfoo(text) returns foodomain as
207  'begin return $1::foodomain; end' language plpgsql volatile;
208
209CREATE FUNCTION inline_eq(foodomain, foodomain) returns boolean as
210  'SELECT CASE $2::text WHEN $1::text THEN true ELSE false END' language sql;
211
212CREATE OPERATOR = (procedure = inline_eq,
213                   leftarg = foodomain, rightarg = foodomain);
214
215SELECT CASE volfoo('bar') WHEN 'foo'::foodomain THEN 'is foo' ELSE 'is not foo' END;
216
217ROLLBACK;

219-- Test multiple evaluation of a CASE arg that is a read/write object (#14472)
220-- Wrap this in a single transaction so the transient '=' operator doesn't
221-- cause problems in concurrent sessions
222BEGIN;
223
224CREATE DOMAIN arrdomain AS int[];
225
226CREATE FUNCTION make_ad(int,int) returns arrdomain as
227  'declare x arrdomain;
228   begin
229     x := array[$1,$2];
230     return x;
231   end' language plpgsql volatile;
232
233CREATE FUNCTION ad_eq(arrdomain, arrdomain) returns boolean as
234  'begin return array_eq($1, $2); end' language plpgsql;
235
236CREATE OPERATOR = (procedure = ad_eq,
237                   leftarg = arrdomain, rightarg = arrdomain);
238
239SELECT CASE make_ad(1,2)
240  WHEN array[2,4]::arrdomain THEN 'wrong'
241  WHEN array[2,5]::arrdomain THEN 'still wrong'
242  WHEN array[1,2]::arrdomain THEN 'right'
243  END;
244
245-- While we're here, also test handling of a NULLIF arg that is a read/write
246-- object (bug #18722)
247
248SELECT NULLIF(make_ad(1,2), array[2,3]::arrdomain);
249
250ROLLBACK;
