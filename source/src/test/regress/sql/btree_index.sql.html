357--
358-- Test B-tree fast path (cache rightmost leaf page) optimization.
359--
360
361-- First create a tree that's at least three levels deep (i.e. has one level
362-- between the root and leaf levels). The text inserted is long.  It won't be
363-- TOAST compressed because we use plain storage in the table.  Only a few
364-- index tuples fit on each internal page, allowing us to get a tall tree with
365-- few pages.  (A tall tree is required to trigger caching.)
366--
367-- The text column must be the leading column in the index, since suffix
368-- truncation would otherwise truncate tuples on internal pages, leaving us
369-- with a short tree.
370create table btree_tall_tbl(id int4, t text);
371alter table btree_tall_tbl alter COLUMN t set storage plain;
372create index btree_tall_idx on btree_tall_tbl (t, id) with (fillfactor = 10);
373insert into btree_tall_tbl select g, repeat('x', 250)
374from generate_series(1, 130) g;
375insert into btree_tall_tbl select g, NULL
376from generate_series(50, 60) g;
