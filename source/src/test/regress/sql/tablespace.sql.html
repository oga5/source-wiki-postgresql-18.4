7-- as a special developer-only option to allow us to use tablespaces
8-- with streaming replication on the same server, an empty location
9-- can be allowed as a way to say that the tablespace should be created
10-- as a directory in pg_tblspc, rather than being a symlink

111-- REINDEX (TABLESPACE) with partitions
112-- Create a partition tree and check the set of relations reindexed
113-- with their new tablespace.
114CREATE TABLE tbspace_reindex_part (c1 int, c2 int) PARTITION BY RANGE (c1);
115CREATE TABLE tbspace_reindex_part_0 PARTITION OF tbspace_reindex_part
116  FOR VALUES FROM (0) TO (10) PARTITION BY list (c2);
117CREATE TABLE tbspace_reindex_part_0_1 PARTITION OF tbspace_reindex_part_0
118  FOR VALUES IN (1);
119CREATE TABLE tbspace_reindex_part_0_2 PARTITION OF tbspace_reindex_part_0
120  FOR VALUES IN (2);
121-- This partitioned table will have no partitions.
122CREATE TABLE tbspace_reindex_part_10 PARTITION OF tbspace_reindex_part
123   FOR VALUES FROM (10) TO (20) PARTITION BY list (c2);
124-- Create some partitioned indexes
125CREATE INDEX tbspace_reindex_part_index ON ONLY tbspace_reindex_part (c1);
126CREATE INDEX tbspace_reindex_part_index_0 ON ONLY tbspace_reindex_part_0 (c1);
127ALTER INDEX tbspace_reindex_part_index ATTACH PARTITION tbspace_reindex_part_index_0;
128-- This partitioned index will have no partitions.
129CREATE INDEX tbspace_reindex_part_index_10 ON ONLY tbspace_reindex_part_10 (c1);
130ALTER INDEX tbspace_reindex_part_index ATTACH PARTITION tbspace_reindex_part_index_10;
131CREATE INDEX tbspace_reindex_part_index_0_1 ON ONLY tbspace_reindex_part_0_1 (c1);
132ALTER INDEX tbspace_reindex_part_index_0 ATTACH PARTITION tbspace_reindex_part_index_0_1;
133CREATE INDEX tbspace_reindex_part_index_0_2 ON ONLY tbspace_reindex_part_0_2 (c1);
134ALTER INDEX tbspace_reindex_part_index_0 ATTACH PARTITION tbspace_reindex_part_index_0_2;
135SELECT relid, parentrelid, level FROM pg_partition_tree('tbspace_reindex_part_index')
136  ORDER BY relid, level;
137-- Track the original tablespace, relfilenode and OID of each index
138-- in the tree.
139CREATE TEMP TABLE reindex_temp_before AS
140  SELECT oid, relname, relfilenode, reltablespace
141  FROM pg_class
142    WHERE relname ~ 'tbspace_reindex_part_index';
143REINDEX (TABLESPACE regress_tblspace, CONCURRENTLY) TABLE tbspace_reindex_part;
144-- REINDEX CONCURRENTLY changes the OID of the old relation, hence a check
145-- based on the relation name below.
146SELECT b.relname,
147       CASE WHEN a.relfilenode = b.relfilenode THEN 'relfilenode is unchanged'
148       ELSE 'relfilenode has changed' END AS filenode,
149       CASE WHEN a.reltablespace = b.reltablespace THEN 'reltablespace is unchanged'
150       ELSE 'reltablespace has changed' END AS tbspace
151  FROM reindex_temp_before b JOIN pg_class a ON b.relname = a.relname
152  ORDER BY 1;
153DROP TABLE tbspace_reindex_part;
