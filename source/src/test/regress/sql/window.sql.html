504-- There is a syntactic ambiguity in the SQL standard.  Since
505-- UNBOUNDED is a non-reserved word, it could be the name of a
506-- function parameter and be used as an expression.  There is a
507-- grammar hack to resolve such cases as the keyword.  The following
508-- tests record this behavior.
509
510CREATE FUNCTION unbounded_syntax_test1a(x int) RETURNS TABLE (a int, b int, c int)
511LANGUAGE SQL
512BEGIN ATOMIC
513  SELECT sum(unique1) over (rows between x preceding and x following),
514         unique1, four
515  FROM tenk1 WHERE unique1 < 10;
516END;
517
518CREATE FUNCTION unbounded_syntax_test1b(x int) RETURNS TABLE (a int, b int, c int)
519LANGUAGE SQL
520AS $$
521  SELECT sum(unique1) over (rows between x preceding and x following),
522         unique1, four
523  FROM tenk1 WHERE unique1 < 10;
524$$;
525
526-- These will apply the argument to the window specification inside the function.
527SELECT * FROM unbounded_syntax_test1a(2);
528SELECT * FROM unbounded_syntax_test1b(2);

534         unique1, four
