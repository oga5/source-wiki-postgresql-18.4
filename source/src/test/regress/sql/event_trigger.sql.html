216-- This tests errors raised within event triggers; the one in audit_tbls
217-- uses 2nd-level recursive invocation via test_evtrig_dropped_objects().
218CREATE OR REPLACE FUNCTION undroppable() RETURNS event_trigger
219LANGUAGE plpgsql AS $$
220DECLARE
221	obj record;
222BEGIN
223	PERFORM 1 FROM pg_tables WHERE tablename = 'undroppable_objs';
224	IF NOT FOUND THEN
225		RAISE NOTICE 'table undroppable_objs not found, skipping';
226		RETURN;
227	END IF;
228	FOR obj IN
229		SELECT * FROM pg_event_trigger_dropped_objects() JOIN
230			undroppable_objs USING (object_type, object_identity)
231	LOOP
232		RAISE EXCEPTION 'object % of type % cannot be dropped',
233			obj.object_identity, obj.object_type;
234	END LOOP;
235END;
236$$;

261END

289CREATE OR REPLACE FUNCTION event_trigger_report_dropped()
290 RETURNS event_trigger
291 LANGUAGE plpgsql
292AS $$
293DECLARE r record;
294BEGIN
295    FOR r IN SELECT * from pg_event_trigger_dropped_objects()
296    LOOP
297    IF NOT r.normal AND NOT r.original THEN
298        CONTINUE;
299    END IF;
300    RAISE NOTICE 'NORMAL: orig=% normal=% istemp=% type=% identity=% schema=% name=% addr=% args=%',
301        r.original, r.normal, r.is_temporary, r.object_type,
302        r.object_identity, r.schema_name, r.object_name,
303        r.address_names, r.address_args;
304    END LOOP;
305END; $$;
