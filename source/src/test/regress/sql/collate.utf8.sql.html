1/*
2 * This test is for collations and character operations when using the
3 * builtin provider with the C.UTF-8 locale.
4 */
5
6/* skip test if not UTF8 server encoding */
7SELECT getdatabaseencoding() <> 'UTF8' AS skip_test \gset
8\if :skip_test
9\quit
10\endif
11
12SET client_encoding TO UTF8;
13
14--
15-- Test builtin "C"
16--
17CREATE COLLATION regress_builtin_c (
18  provider = builtin, locale = 'C');
19
20-- non-ASCII characters are unchanged
21SELECT LOWER(U&'\00C1' COLLATE regress_builtin_c) = U&'\00C1';
22SELECT UPPER(U&'\00E1' COLLATE regress_builtin_c) = U&'\00E1';
23
24-- non-ASCII characters are not alphabetic
25SELECT U&'\00C1\00E1' !~ '[[:alpha:]]' COLLATE regress_builtin_c;
26
27DROP COLLATION regress_builtin_c;
28
29--
30-- Test PG_C_UTF8
31--
32
33CREATE COLLATION regress_pg_c_utf8 (
34  provider = builtin, locale = 'C_UTF8'); -- fails
35CREATE COLLATION regress_pg_c_utf8 (
36  provider = builtin, locale = 'C.UTF8');
37DROP COLLATION regress_pg_c_utf8;
38CREATE COLLATION regress_pg_c_utf8 (
39  provider = builtin, locale = 'C.UTF-8');
40
41CREATE TABLE test_pg_c_utf8 (
42  t TEXT COLLATE PG_C_UTF8
43);
44INSERT INTO test_pg_c_utf8 VALUES
45  ('abc DEF 123abc'),
46  ('ábc sßs ßss DÉF'),
47  ('ǄxxǄ ǆxxǅ ǅxxǆ'),
48  (U&'Λλ 1a \FF11a'),
49  ('ȺȺȺ'),
50  ('ⱥⱥⱥ'),
51  ('ⱥȺ');
52
53SELECT
54    t, lower(t), initcap(t), upper(t),
55    length(convert_to(t, 'UTF8')) AS t_bytes,
56    length(convert_to(lower(t), 'UTF8')) AS lower_t_bytes,
57    length(convert_to(initcap(t), 'UTF8')) AS initcap_t_bytes,
58    length(convert_to(upper(t), 'UTF8')) AS upper_t_bytes
59  FROM test_pg_c_utf8;
60
61DROP TABLE test_pg_c_utf8;
62
63-- negative test: Final_Sigma not used for builtin locale C.UTF-8
64SELECT lower('ΑΣ' COLLATE PG_C_UTF8);
65SELECT lower('ΑͺΣͺ' COLLATE PG_C_UTF8);
66SELECT lower('Α΄Σ΄' COLLATE PG_C_UTF8);
67
68-- properties
69
70SELECT 'xyz' ~ '[[:alnum:]]' COLLATE PG_C_UTF8;
71SELECT 'xyz' !~ '[[:upper:]]' COLLATE PG_C_UTF8;
72SELECT '@' !~ '[[:alnum:]]' COLLATE PG_C_UTF8;
73SELECT '=' ~ '[[:punct:]]' COLLATE PG_C_UTF8; -- symbols are punctuation in posix
74SELECT 'a8a' ~ '[[:digit:]]' COLLATE PG_C_UTF8;
75SELECT '൧' !~ '\d' COLLATE PG_C_UTF8; -- only 0-9 considered digits in posix
76
77-- case mapping
78
79SELECT 'xYz' ~* 'XyZ' COLLATE PG_C_UTF8;
80SELECT 'xAb' ~* '[W-Y]' COLLATE PG_C_UTF8;
81SELECT 'xAb' !~* '[c-d]' COLLATE PG_C_UTF8;
82SELECT 'Δ' ~* '[γ-λ]' COLLATE PG_C_UTF8;
83SELECT 'δ' ~* '[Γ-Λ]' COLLATE PG_C_UTF8; -- same as above with cases reversed
84
85-- case folding
86select casefold('AbCd 123 #$% ıiIİ ẞ ß Ǆǅǆ Σσς' collate PG_C_UTF8);
