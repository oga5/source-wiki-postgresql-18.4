683CREATE FUNCTION gtest_trigger_func() RETURNS trigger
684  LANGUAGE plpgsql
685AS $$
686BEGIN
687  IF tg_op IN ('DELETE', 'UPDATE') THEN
688    RAISE INFO '%: %: old = %', TG_NAME, TG_WHEN, OLD;
689  END IF;
690  IF tg_op IN ('INSERT', 'UPDATE') THEN
691    RAISE INFO '%: %: new = %', TG_NAME, TG_WHEN, NEW;
692  END IF;
693  IF tg_op = 'DELETE' THEN
694    RETURN OLD;
695  ELSE
696    RETURN NEW;
697  END IF;
698END
699$$;
