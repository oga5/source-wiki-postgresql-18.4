279--
280-- check transactional behaviour of ALTER TYPE ... ADD VALUE
281--
282CREATE TYPE bogus AS ENUM('good');
283
284-- check that we can add new values to existing enums in a transaction
285-- but we can't use them
286BEGIN;
287ALTER TYPE bogus ADD VALUE 'new';
288SAVEPOINT x;
289SELECT 'new'::bogus;  -- unsafe
290ROLLBACK TO x;
291SELECT enum_first(null::bogus);  -- safe
292SELECT enum_last(null::bogus);  -- unsafe
293ROLLBACK TO x;
294SELECT enum_range(null::bogus);  -- unsafe
295ROLLBACK TO x;
296COMMIT;
297SELECT 'new'::bogus;  -- now safe
298SELECT enumlabel, enumsortorder
299FROM pg_enum
300WHERE enumtypid = 'bogus'::regtype
301ORDER BY 2;
302
303-- check that we recognize the case where the enum already existed but was
304-- modified in the current txn; this should not be considered safe
305BEGIN;
306ALTER TYPE bogus RENAME TO bogon;
307ALTER TYPE bogon ADD VALUE 'bad';
308SELECT 'bad'::bogon;
309ROLLBACK;
310
311-- but a renamed value is safe to use later in same transaction
312BEGIN;
313ALTER TYPE bogus RENAME VALUE 'good' to 'bad';
314SELECT 'bad'::bogus;
315ROLLBACK;
316
317DROP TYPE bogus;
318
319-- check that values created during CREATE TYPE can be used in any case
320BEGIN;
321CREATE TYPE bogus AS ENUM('good','bad','ugly');
322ALTER TYPE bogus RENAME TO bogon;
323select enum_range(null::bogon);
324ROLLBACK;
