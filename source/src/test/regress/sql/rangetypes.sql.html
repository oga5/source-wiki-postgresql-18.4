7
8-- negative tests; should fail
9select ''::textrange;
10select '-[a,z)'::textrange;
11select '[a,z) - '::textrange;
12select '(",a)'::textrange;
13select '(,,a)'::textrange;
14select '(),a)'::textrange;
15select '(a,))'::textrange;
16select '(],a)'::textrange;
17select '(a,])'::textrange;
18select '[z,a]'::textrange;
19
20-- should succeed
21select '  empty  '::textrange;
22select ' ( empty, empty )  '::textrange;
23select ' ( " a " " a ", " z " " z " )  '::textrange;
24select '(a,)'::textrange;
25select '[,z]'::textrange;
26select '[a,]'::textrange;
27select '(,)'::textrange;
28select '[ , ]'::textrange;
29select '["",""]'::textrange;
30select '[",",","]'::textrange;
31select '["\\","\\"]'::textrange;
32select '(\\,a)'::textrange;
33select '((,z)'::textrange;
34select '([,z)'::textrange;
35select '(!,()'::textrange;
36select '(!,[)'::textrange;
37select '[a,a]'::textrange;
38-- these are allowed but normalize to empty:
39select '[a,a)'::textrange;
40select '(a,a]'::textrange;
41select '(a,a)'::textrange;
