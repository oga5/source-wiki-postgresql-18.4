274-- propagation of collation in SQL functions (inlined and non-inlined cases)
275-- and plpgsql functions too
276
277CREATE FUNCTION mylt (text, text) RETURNS boolean LANGUAGE sql
278    AS $$ select $1 < $2 $$;
279
280CREATE FUNCTION mylt_noninline (text, text) RETURNS boolean LANGUAGE sql
281    AS $$ select $1 < $2 limit 1 $$;
282
283CREATE FUNCTION mylt_plpgsql (text, text) RETURNS boolean LANGUAGE plpgsql
284    AS $$ begin return $1 < $2; end $$;
285
286SELECT a.b AS a, b.b AS b, a.b < b.b AS lt,
287       mylt(a.b, b.b), mylt_noninline(a.b, b.b), mylt_plpgsql(a.b, b.b)
288FROM collate_test1 a, collate_test1 b
289ORDER BY a.b, b.b;
290
291SELECT a.b AS a, b.b AS b, a.b < b.b COLLATE "C" AS lt,
292       mylt(a.b, b.b COLLATE "C"), mylt_noninline(a.b, b.b COLLATE "C"),
293       mylt_plpgsql(a.b, b.b COLLATE "C")
294FROM collate_test1 a, collate_test1 b
295ORDER BY a.b, b.b;

298-- collation override in plpgsql
299
300CREATE FUNCTION mylt2 (x text, y text) RETURNS boolean LANGUAGE plpgsql AS $$
301declare
302  xx text := x;
303  yy text := y;
304begin
305  return xx < yy;
306end
307$$;
308
309SELECT mylt2('a', 'B' collate "en_US") as t, mylt2('a', 'B' collate "C") as f;
310
311CREATE OR REPLACE FUNCTION
312  mylt2 (x text, y text) RETURNS boolean LANGUAGE plpgsql AS $$
313declare
314  xx text COLLATE "POSIX" := x;
315  yy text := y;
316begin
317  return xx < yy;
318end
319$$;
320
321SELECT mylt2('a', 'B') as f;
322SELECT mylt2('a', 'B' collate "C") as fail; -- conflicting collations
323SELECT mylt2('a', 'B' collate "POSIX") as f;
