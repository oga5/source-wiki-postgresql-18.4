622-- Check that domain constraints on plpgsql function parameters, results,
623-- and local variables are enforced correctly.
624
625create function doubledecrement(p1 pos_int) returns pos_int as $$
626declare v pos_int;
627begin
628    return p1;
629end$$ language plpgsql;
630
631select doubledecrement(3); -- fail because of implicit null assignment
632
633create or replace function doubledecrement(p1 pos_int) returns pos_int as $$
634declare v pos_int := 0;
635begin
636    return p1;
637end$$ language plpgsql;
638
639select doubledecrement(3); -- fail at initialization assignment
640
641create or replace function doubledecrement(p1 pos_int) returns pos_int as $$
642declare v pos_int := 1;
643begin
644    v := p1 - 1;
645    return v - 1;
646end$$ language plpgsql;
647
648select doubledecrement(null); -- fail before call
649select doubledecrement(0); -- fail before call
650select doubledecrement(1); -- fail at assignment to v
651select doubledecrement(2); -- fail at return
652select doubledecrement(3); -- good
