29-- Tests for currtid2() with various relation kinds
30
31-- Materialized view
32CREATE MATERIALIZED VIEW tid_matview AS SELECT a FROM tid_tab;
33SELECT currtid2('tid_matview'::text, '(0,1)'::tid); -- fails
34INSERT INTO tid_tab VALUES (1);
35REFRESH MATERIALIZED VIEW tid_matview;
36SELECT currtid2('tid_matview'::text, '(0,1)'::tid); -- ok
37DROP MATERIALIZED VIEW tid_matview;
38TRUNCATE tid_tab;
39
40-- Sequence
41CREATE SEQUENCE tid_seq;
42SELECT currtid2('tid_seq'::text, '(0,1)'::tid); -- ok
43DROP SEQUENCE tid_seq;
44
45-- Index, fails with incorrect relation type
46CREATE INDEX tid_ind ON tid_tab(a);
47SELECT currtid2('tid_ind'::text, '(0,1)'::tid); -- fails
48DROP INDEX tid_ind;
49
50-- Partitioned table, no storage
51CREATE TABLE tid_part (a int) PARTITION BY RANGE (a);
52SELECT currtid2('tid_part'::text, '(0,1)'::tid); -- fails
53DROP TABLE tid_part;
54
55-- Views
56-- ctid not defined in the view
