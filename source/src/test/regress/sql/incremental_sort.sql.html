16create or replace function explain_analyze_without_memory(query text)
17returns table (out_line text) language plpgsql
18as
19$$
20declare
21  line text;
22begin
23  for line in
24    execute 'explain (analyze, costs off, summary off, timing off, buffers off) ' || query
25  loop
26    out_line := regexp_replace(line, '\d+kB', 'NNkB', 'g');
27    return next;
28  end loop;
29end;
30$$;

32create or replace function explain_analyze_inc_sort_nodes(query text)
33returns jsonb language plpgsql
34as
35$$
36declare
37  elements jsonb;
38  element jsonb;
39  matching_nodes jsonb := '[]'::jsonb;
40begin
41  execute 'explain (analyze, costs off, summary off, timing off, buffers off, format ''json'') ' || query into strict elements;
42  while jsonb_array_length(elements) > 0 loop
43    element := elements->0;
44    elements := elements - 0;
45    case jsonb_typeof(element)
46    when 'array' then
47      if jsonb_array_length(element) > 0 then
48        elements := elements || element;
49      end if;
50    when 'object' then
51      if element ? 'Plan' then
52        elements := elements || jsonb_build_array(element->'Plan');
53        element := element - 'Plan';
54      else
55        if element ? 'Plans' then
56          elements := elements || jsonb_build_array(element->'Plans');
57          element := element - 'Plans';
58        end if;
59        if (element->>'Node Type')::text = 'Incremental Sort' then
60          matching_nodes := matching_nodes || element;
61        end if;
62      end if;
63    end case;
64  end loop;
65  return matching_nodes;
66end;
67$$;

69create or replace function explain_analyze_inc_sort_nodes_without_memory(query text)
70returns jsonb language plpgsql
71as
72$$
73declare
74  nodes jsonb := '[]'::jsonb;
75  node jsonb;
76  group_key text;
77  space_key text;
78begin
79  for node in select * from jsonb_array_elements(explain_analyze_inc_sort_nodes(query)) t loop
80    for group_key in select unnest(array['Full-sort Groups', 'Pre-sorted Groups']::text[]) t loop
81      for space_key in select unnest(array['Sort Space Memory', 'Sort Space Disk']::text[]) t loop
82        node := jsonb_set(node, array[group_key, space_key, 'Average Sort Space Used'], '"NN"', false);
83        node := jsonb_set(node, array[group_key, space_key, 'Peak Sort Space Used'], '"NN"', false);
84      end loop;
85    end loop;
86    nodes := nodes || node;
87  end loop;
88  return nodes;
89end;
90$$;

92create or replace function explain_analyze_inc_sort_nodes_verify_invariants(query text)
93returns bool language plpgsql
94as
95$$
96declare
97  node jsonb;
98  group_stats jsonb;
99  group_key text;
100  space_key text;
101begin
102  for node in select * from jsonb_array_elements(explain_analyze_inc_sort_nodes(query)) t loop
103    for group_key in select unnest(array['Full-sort Groups', 'Pre-sorted Groups']::text[]) t loop
104      group_stats := node->group_key;
105      for space_key in select unnest(array['Sort Space Memory', 'Sort Space Disk']::text[]) t loop
106        if (group_stats->space_key->'Peak Sort Space Used')::bigint < (group_stats->space_key->'Peak Sort Space Used')::bigint then
107          raise exception '% has invalid max space < average space', group_key;
108        end if;
109      end loop;
110    end loop;
111  end loop;
112  return true;
113end;
114$$;
