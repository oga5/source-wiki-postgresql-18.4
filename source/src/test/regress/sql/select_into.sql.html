19--
20-- SELECT INTO and INSERT permission, if owner is not allowed to insert.
21--
22CREATE SCHEMA selinto_schema;
23CREATE USER regress_selinto_user;
24ALTER DEFAULT PRIVILEGES FOR ROLE regress_selinto_user
25	  REVOKE INSERT ON TABLES FROM regress_selinto_user;
26GRANT ALL ON SCHEMA selinto_schema TO public;
27
28SET SESSION AUTHORIZATION regress_selinto_user;
29-- WITH DATA, passes.
30CREATE TABLE selinto_schema.tbl_withdata1 (a)
31  AS SELECT generate_series(1,3) WITH DATA;
32INSERT INTO selinto_schema.tbl_withdata1 VALUES (4);
33EXPLAIN (ANALYZE, COSTS OFF, SUMMARY OFF, TIMING OFF, BUFFERS OFF)
34  CREATE TABLE selinto_schema.tbl_withdata2 (a) AS
35  SELECT generate_series(1,3) WITH DATA;
36-- WITH NO DATA, passes.
37CREATE TABLE selinto_schema.tbl_nodata1 (a) AS
38  SELECT generate_series(1,3) WITH NO DATA;
39EXPLAIN (ANALYZE, COSTS OFF, SUMMARY OFF, TIMING OFF, BUFFERS OFF)
40  CREATE TABLE selinto_schema.tbl_nodata2 (a) AS
41  SELECT generate_series(1,3) WITH NO DATA;
42-- EXECUTE and WITH DATA, passes.
43PREPARE data_sel AS SELECT generate_series(1,3);
44CREATE TABLE selinto_schema.tbl_withdata3 (a) AS
45  EXECUTE data_sel WITH DATA;
46EXPLAIN (ANALYZE, COSTS OFF, SUMMARY OFF, TIMING OFF, BUFFERS OFF)
47  CREATE TABLE selinto_schema.tbl_withdata4 (a) AS
48  EXECUTE data_sel WITH DATA;
49-- EXECUTE and WITH NO DATA, passes.
50CREATE TABLE selinto_schema.tbl_nodata3 (a) AS
51  EXECUTE data_sel WITH NO DATA;
52EXPLAIN (ANALYZE, COSTS OFF, SUMMARY OFF, TIMING OFF, BUFFERS OFF)
53  CREATE TABLE selinto_schema.tbl_nodata4 (a) AS
54  EXECUTE data_sel WITH NO DATA;
55RESET SESSION AUTHORIZATION;

86--
87-- CREATE TABLE AS/SELECT INTO as last command in a SQL function
88-- have been known to cause problems
89--
90CREATE FUNCTION make_table() RETURNS VOID
91AS $$
92  CREATE TABLE created_table AS SELECT * FROM int8_tbl;
93$$ LANGUAGE SQL;
94
95SELECT make_table();
96
97SELECT * FROM created_table;

99-- Try EXPLAIN ANALYZE SELECT INTO and EXPLAIN ANALYZE CREATE TABLE AS
100-- WITH NO DATA, but hide the outputs since they won't be stable.
101DO $$
102BEGIN
103	EXECUTE 'EXPLAIN ANALYZE SELECT * INTO TABLE easi FROM int8_tbl';
104	EXECUTE 'EXPLAIN ANALYZE CREATE TABLE easi2 AS SELECT * FROM int8_tbl WITH NO DATA';
105END$$;
106
107DROP TABLE created_table;
108DROP TABLE easi, easi2;
