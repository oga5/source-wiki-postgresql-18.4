1/*
2 * This test is intended to pass on all platforms supported by Postgres.
3 * We can therefore only assume that the default, C, and POSIX collations
4 * are available --- and since the regression tests are often run in a
5 * C-locale database, these may well all have the same behavior.  But
6 * fortunately, the system doesn't know that and will treat them as
7 * incompatible collations.  It is therefore at least possible to test
8 * parser behaviors such as collation conflict resolution.  This test will,
9 * however, be more revealing when run in a database with non-C locale,
10 * since any departure from C sorting behavior will show as a failure.
11 */

285CREATE COLLATION coll_dup_chk (LC_COLLATE = "POSIX", LC_COLLATE = "NONSENSE", LC_CTYPE = "POSIX");
286-- LC_CTYPE
287CREATE COLLATION coll_dup_chk (LC_CTYPE = "POSIX", LC_CTYPE = "NONSENSE", LC_COLLATE = "POSIX");
288-- PROVIDER
289CREATE COLLATION coll_dup_chk (PROVIDER = icu, PROVIDER = NONSENSE, LC_COLLATE = "POSIX", LC_CTYPE = "POSIX");
290-- LOCALE
291CREATE COLLATION case_sensitive (LOCALE = '', LOCALE = "NONSENSE");
292-- DETERMINISTIC
293CREATE COLLATION coll_dup_chk (DETERMINISTIC = TRUE, DETERMINISTIC = NONSENSE, LOCALE = '');
294-- VERSION
295CREATE COLLATION coll_dup_chk (VERSION = '1', VERSION = "NONSENSE", LOCALE = '');
296-- LOCALE conflicts with LC_COLLATE and LC_CTYPE
297CREATE COLLATION coll_dup_chk (LC_COLLATE = "POSIX", LC_CTYPE = "POSIX", LOCALE = '');
298-- LOCALE conflicts with LC_COLLATE
299CREATE COLLATION coll_dup_chk (LC_COLLATE = "POSIX", LOCALE = '');
300-- LOCALE conflicts with LC_CTYPE
301CREATE COLLATION coll_dup_chk (LC_CTYPE = "POSIX", LOCALE = '');
302-- FROM conflicts with any other option
303CREATE COLLATION coll_dup_chk (FROM = "C", VERSION = "1");
