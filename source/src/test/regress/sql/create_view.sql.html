596-- We used to have a bug that would allow the above to succeed, posing
597-- hazards for later execution of the view.  Check that the internal
598-- defenses for those hazards haven't bit-rotted, in case some other
599-- bug with similar symptoms emerges.
600begin;
601
602-- destroy the dependency entry that prevents the DROP:
603delete from pg_depend where
604  objid = (select oid from pg_rewrite
605           where ev_class = 'tt14v'::regclass and rulename = '_RETURN')
606  and refobjsubid = 3
607returning pg_describe_object(classid, objid, objsubid) as obj,
608          pg_describe_object(refclassid, refobjid, refobjsubid) as ref,
609          deptype;
610
611-- this will now succeed:
612alter table tt14t drop column f3;
613
614-- column f3 is still in the view, sort of ...
615select pg_get_viewdef('tt14v', true);
616-- ... and you can even EXPLAIN it ...
617explain (verbose, costs off) select * from tt14v;
618-- but it will fail at execution
619select f1, f4 from tt14v;
620select * from tt14v;
621
622rollback;
