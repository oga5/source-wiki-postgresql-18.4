2-- CHAR

4
5-- Per SQL standard, CHAR means character(1), that is a varlena type
6-- with a constraint restricting it to one character (not byte)
7
8SELECT char 'c' = char 'c' AS true;
9
10--
11-- Build a table for testing
12-- (This temporarily hides the table created in test_setup.sql)
13--
14
15CREATE TEMP TABLE CHAR_TBL(f1 char);
16
17INSERT INTO CHAR_TBL (f1) VALUES ('a');
18
19INSERT INTO CHAR_TBL (f1) VALUES ('A');
20
21-- any of the following three input formats are acceptable
22INSERT INTO CHAR_TBL (f1) VALUES ('1');
23
24INSERT INTO CHAR_TBL (f1) VALUES (2);
25
26INSERT INTO CHAR_TBL (f1) VALUES ('3');
27
28-- zero-length char
29INSERT INTO CHAR_TBL (f1) VALUES ('');
30
31-- try char's of greater than 1 length
32INSERT INTO CHAR_TBL (f1) VALUES ('cd');
33INSERT INTO CHAR_TBL (f1) VALUES ('c     ');
34
35
36SELECT * FROM CHAR_TBL;
37
38SELECT c.*
39   FROM CHAR_TBL c
40   WHERE c.f1 <> 'a';
41
42SELECT c.*
43   FROM CHAR_TBL c
44   WHERE c.f1 = 'a';
45
46SELECT c.*
47   FROM CHAR_TBL c
48   WHERE c.f1 < 'a';
49
50SELECT c.*
51   FROM CHAR_TBL c
52   WHERE c.f1 <= 'a';
53
54SELECT c.*
55   FROM CHAR_TBL c
56   WHERE c.f1 > 'a';
57
58SELECT c.*
59   FROM CHAR_TBL c
60   WHERE c.f1 >= 'a';

62DROP TABLE CHAR_TBL;

80--
81-- Also test "char", which is an ad-hoc one-byte type.  It can only
82-- really store ASCII characters, but we allow high-bit-set characters
83-- to be accessed via bytea-like escapes.
84--
85
86SELECT 'a'::"char";
87SELECT '\101'::"char";
88SELECT '\377'::"char";
89SELECT 'a'::"char"::text;
90SELECT '\377'::"char"::text;
91SELECT '\000'::"char"::text;
92SELECT 'a'::text::"char";
93SELECT '\377'::text::"char";
94SELECT ''::text::"char";
