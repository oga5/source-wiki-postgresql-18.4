157CREATE FUNCTION test_future_xid_status(xid8)
158RETURNS void
159LANGUAGE plpgsql
160AS
161$$
162BEGIN
163  PERFORM pg_xact_status($1);
164  RAISE EXCEPTION 'didn''t ERROR at xid in the future as expected';
165EXCEPTION
166  WHEN invalid_parameter_value THEN
167    RAISE NOTICE 'Got expected error for xid in the future';
168END;
169$$;
