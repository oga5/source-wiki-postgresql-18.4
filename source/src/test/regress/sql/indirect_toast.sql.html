1--
2-- Tests for external toast datums
3--
4
5-- directory paths and dlsuffix are passed to us in environment variables
6\getenv libdir PG_LIBDIR
7\getenv dlsuffix PG_DLSUFFIX
8
9\set regresslib :libdir '/regress' :dlsuffix
10
11CREATE FUNCTION make_tuple_indirect (record)
12        RETURNS record
13        AS :'regresslib'
14        LANGUAGE C STRICT;
15
16-- Other compression algorithms may cause the compressed data to be stored
17-- inline.  pglz guarantees that the data is externalized, so stick to it.
18SET default_toast_compression = 'pglz';
19
20CREATE TABLE indtoasttest(descr text, cnt int DEFAULT 0, f1 text, f2 text);
21
22INSERT INTO indtoasttest(descr, f1, f2) VALUES('two-compressed', repeat('1234567890',1000), repeat('1234567890',1000));
23INSERT INTO indtoasttest(descr, f1, f2) VALUES('two-toasted', repeat('1234567890',30000), repeat('1234567890',50000));
24INSERT INTO indtoasttest(descr, f1, f2) VALUES('one-compressed,one-null', NULL, repeat('1234567890',1000));
25INSERT INTO indtoasttest(descr, f1, f2) VALUES('one-toasted,one-null', NULL, repeat('1234567890',50000));
26
27-- check whether indirect tuples works on the most basic level
28SELECT descr, substring(make_tuple_indirect(indtoasttest)::text, 1, 200) FROM indtoasttest;
29
30-- modification without changing varlenas
31UPDATE indtoasttest SET cnt = cnt +1 RETURNING substring(indtoasttest::text, 1, 200);
32
33-- modification without modifying assigned value
34UPDATE indtoasttest SET cnt = cnt +1, f1 = f1 RETURNING substring(indtoasttest::text, 1, 200);
35
36-- modification modifying, but effectively not changing
37UPDATE indtoasttest SET cnt = cnt +1, f1 = f1||'' RETURNING substring(indtoasttest::text, 1, 200);
38
39UPDATE indtoasttest SET cnt = cnt +1, f1 = '-'||f1||'-' RETURNING substring(indtoasttest::text, 1, 200);
40
41SELECT substring(indtoasttest::text, 1, 200) FROM indtoasttest;
42-- check we didn't screw with main/toast tuple visibility
43VACUUM FREEZE indtoasttest;
44SELECT substring(indtoasttest::text, 1, 200) FROM indtoasttest;
45
46-- now create a trigger that forces all Datums to be indirect ones
47CREATE FUNCTION update_using_indirect()
48        RETURNS trigger
49        LANGUAGE plpgsql AS $$
50BEGIN
51    NEW := make_tuple_indirect(NEW);
52    RETURN NEW;
53END$$;
54
55CREATE TRIGGER indtoasttest_update_indirect
56        BEFORE INSERT OR UPDATE
57        ON indtoasttest
58        FOR EACH ROW
59        EXECUTE PROCEDURE update_using_indirect();
60
61-- modification without changing varlenas
62UPDATE indtoasttest SET cnt = cnt +1 RETURNING substring(indtoasttest::text, 1, 200);
63
64-- modification without modifying assigned value
65UPDATE indtoasttest SET cnt = cnt +1, f1 = f1 RETURNING substring(indtoasttest::text, 1, 200);
66
67-- modification modifying, but effectively not changing
68UPDATE indtoasttest SET cnt = cnt +1, f1 = f1||'' RETURNING substring(indtoasttest::text, 1, 200);
69
70UPDATE indtoasttest SET cnt = cnt +1, f1 = '-'||f1||'-' RETURNING substring(indtoasttest::text, 1, 200);
71
72INSERT INTO indtoasttest(descr, f1, f2) VALUES('one-toasted,one-null, via indirect', repeat('1234567890',30000), NULL);
73
74SELECT substring(indtoasttest::text, 1, 200) FROM indtoasttest;
75-- check we didn't screw with main/toast tuple visibility
76VACUUM FREEZE indtoasttest;
77SELECT substring(indtoasttest::text, 1, 200) FROM indtoasttest;

79DROP TABLE indtoasttest;
