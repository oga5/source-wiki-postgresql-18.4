1--
2-- encoding-sensitive tests for jsonpath
3--
4
5-- We provide expected-results files for UTF8 (jsonpath_encoding.out)
6-- and for SQL_ASCII (jsonpath_encoding_1.out).  Skip otherwise.
7SELECT getdatabaseencoding() NOT IN ('UTF8', 'SQL_ASCII')
8       AS skip_test \gset
9\if :skip_test
10\quit
11\endif
12
13SELECT getdatabaseencoding();           -- just to label the results files
14
15-- checks for double-quoted values
16
17-- basic unicode input
18SELECT '"\u"'::jsonpath;		-- ERROR, incomplete escape
19SELECT '"\u00"'::jsonpath;		-- ERROR, incomplete escape
20SELECT '"\u000g"'::jsonpath;	-- ERROR, g is not a hex digit
21SELECT '"\u0000"'::jsonpath;	-- OK, legal escape
22SELECT '"\uaBcD"'::jsonpath;	-- OK, uppercase and lower case both OK
23
24-- handling of unicode surrogate pairs
25select '"\ud83d\ude04\ud83d\udc36"'::jsonpath as correct_in_utf8;
26select '"\ud83d\ud83d"'::jsonpath; -- 2 high surrogates in a row
27select '"\ude04\ud83d"'::jsonpath; -- surrogates in wrong order
28select '"\ud83dX"'::jsonpath; -- orphan high surrogate
29select '"\ude04X"'::jsonpath; -- orphan low surrogate
30
31--handling of simple unicode escapes
32select '"the Copyright \u00a9 sign"'::jsonpath as correct_in_utf8;
33select '"dollar \u0024 character"'::jsonpath as correct_everywhere;
34select '"dollar \\u0024 character"'::jsonpath as not_an_escape;
35select '"null \u0000 escape"'::jsonpath as not_unescaped;
36select '"null \\u0000 escape"'::jsonpath as not_an_escape;
37
