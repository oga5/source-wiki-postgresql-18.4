214SELECT (pg_identify_object(addr1.classid, addr1.objid, addr1.objsubid)).*,
215        -- test roundtrip through pg_identify_object_as_address
216        ROW(pg_identify_object(addr1.classid, addr1.objid, addr1.objsubid)) =
217          ROW(pg_identify_object(addr2.classid, addr2.objid, addr2.objsubid)) AS roundtrip
218FROM objects,
219     pg_get_object_address(type, name, args) AS addr1,
220     pg_identify_object_as_address(classid, objid, objsubid) AS ioa (typ, nms, args),
221     pg_get_object_address(typ, nms, ioa.args) AS addr2
