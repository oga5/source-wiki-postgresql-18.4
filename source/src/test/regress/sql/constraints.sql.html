549CREATE TABLE circles (
550  c1 CIRCLE,
551  c2 TEXT,
552  EXCLUDE USING gist
553    (c1 WITH &&, (c2::circle) WITH &&)
554    WHERE (circle_center(c1) <> '(0,0)')
555);
556
557-- these should succeed because they don't match the index predicate
558INSERT INTO circles VALUES('<(0,0), 5>', '<(0,0), 5>');
559INSERT INTO circles VALUES('<(0,0), 5>', '<(0,0), 4>');
560
561-- succeed
562INSERT INTO circles VALUES('<(10,10), 10>', '<(0,0), 5>');
563-- fail, overlaps
564INSERT INTO circles VALUES('<(20,20), 10>', '<(0,0), 4>');
565-- succeed, because violation is ignored
566INSERT INTO circles VALUES('<(20,20), 10>', '<(0,0), 4>')
567  ON CONFLICT ON CONSTRAINT circles_c1_c2_excl DO NOTHING;
568-- fail, because DO UPDATE variant requires unique index
569INSERT INTO circles VALUES('<(20,20), 10>', '<(0,0), 4>')
570  ON CONFLICT ON CONSTRAINT circles_c1_c2_excl DO UPDATE SET c2 = EXCLUDED.c2;
571-- succeed because c1 doesn't overlap
572INSERT INTO circles VALUES('<(20,20), 1>', '<(0,0), 5>');
573-- succeed because c2 doesn't overlap
574INSERT INTO circles VALUES('<(20,20), 10>', '<(10,10), 5>');

816PREPARE get_nnconstraint_info(regclass[]) AS
817SELECT conrelid::regclass as tabname, conname, convalidated, conislocal, coninhcount
818FROM  pg_constraint
819WHERE conrelid = ANY($1)
820ORDER BY conrelid::regclass::text COLLATE "C", conname;
