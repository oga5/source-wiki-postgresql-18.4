76=pod
77
78=item PostgreSQL::Test::Kerberos->new(host, hostaddr, realm, %params)
79
80Sets up a new Kerberos realm and KDC.  This function assigns a free
81port for the KDC.  The KDC will be shut down automatically when the
82test script exits.
83
84=over
85
86=item host => 'auth-test-localhost.postgresql.example.com'
87
88Hostname to use in the service principal.
89
90=item hostaddr => '127.0.0.1'
91
92Network interface the KDC will listen on.
93
94=item realm => 'EXAMPLE.COM'
95
96Name of the Kerberos realm.
97
98=back
99
100=cut
101
102sub new
103{
104	my $class = shift;
105	my ($host, $hostaddr, $realm) = @_;
106
107	my ($stdout, $krb5_version);
108	run_log [ $krb5_config, '--version' ], '>' => \$stdout
109	  or BAIL_OUT("could not execute krb5-config");
110	BAIL_OUT("Heimdal is not supported") if $stdout =~ m/heimdal/;
111	$stdout =~ m/Kerberos 5 release ([0-9]+\.[0-9]+)/
112	  or BAIL_OUT("could not get Kerberos version");
113	$krb5_version = $1;
114
115	# Build the krb5.conf to use.
116	#
117	# Explicitly specify the default (test) realm and the KDC for
118	# that realm to avoid the Kerberos library trying to look up
119	# that information in DNS, and also because we're using a
120	# non-standard KDC port.
121	#
122	# Also explicitly disable DNS lookups since this isn't really
123	# our domain and we shouldn't be causing random DNS requests
124	# to be sent out (not to mention that broken DNS environments
125	# can cause the tests to take an extra long time and timeout).
126	#
127	# Reverse DNS is explicitly disabled to avoid any issue with a
128	# captive portal or other cases where the reverse DNS succeeds
129	# and the Kerberos library uses that as the canonical name of
130	# the host and then tries to acquire a cross-realm ticket.
131	append_to_file(
132		$krb5_conf,
133		qq![logging]
134default = FILE:$krb5_log
135kdc = FILE:$kdc_log
136
137[libdefaults]
138dns_lookup_realm = false
139dns_lookup_kdc = false
140default_realm = $realm
141forwardable = false
142rdns = false
143
144[realms]
145$realm = {
146    kdc = $hostaddr:$kdc_port
147}
148!);
149
150	append_to_file(
151		$kdc_conf,
152		qq![kdcdefaults]
153!);
154
155	# For new-enough versions of krb5, use the _listen settings rather
156	# than the _ports settings so that we can bind to localhost only.
157	if ($krb5_version >= 1.15)
158	{
159		append_to_file(
160			$kdc_conf,
161			qq!kdc_listen = $hostaddr:$kdc_port
162kdc_tcp_listen = $hostaddr:$kdc_port
163!);
164	}
165	else
166	{
167		append_to_file(
168			$kdc_conf,
169			qq!kdc_ports = $kdc_port
170kdc_tcp_ports = $kdc_port
171!);
172	}
173	append_to_file(
174		$kdc_conf,
175		qq!
176[realms]
177$realm = {
178    database_name = $kdc_datadir/principal
179    admin_keytab = FILE:$kdc_datadir/kadm5.keytab
180    acl_file = $kdc_datadir/kadm5.acl
181    key_stash_file = $kdc_datadir/_k5.$realm
182}!);
183
184	mkdir $kdc_datadir
185	  or BAIL_OUT("could not create directory \"$kdc_datadir\"");
186
187	# Ensure that we use test's config and cache files, not global ones.
188	$ENV{'KRB5_CONFIG'} = $krb5_conf;
189	$ENV{'KRB5_KDC_PROFILE'} = $kdc_conf;
190	$ENV{'KRB5CCNAME'} = $krb5_cache;
191
192	my $service_principal = "$ENV{with_krb_srvnam}/$host";
193
194	system_or_bail $kdb5_util, 'create', '-s', '-P', 'secret0';
195
196	system_or_bail $kadmin_local, '-q',
197	  "addprinc -randkey $service_principal";
198	system_or_bail $kadmin_local, '-q', "ktadd -k $keytab $service_principal";
199
200	system_or_bail $krb5kdc, '-P', $kdc_pidfile;
201
202	my $self = {};
203	$self->{keytab} = $keytab;
204
205	bless $self, $class;
206
207	return $self;
208}
209
210sub create_principal
211{
212	my ($self, $principal, $password) = @_;
213
214	system_or_bail $kadmin_local, '-q', "addprinc -pw $password $principal";
215}
216
217sub create_ticket
218{
219	my ($self, $principal, $password, %params) = @_;
220
221	my @cmd = ($kinit, $principal);
222
223	push @cmd, '-f' if ($params{forwardable});
224
225	run_log [@cmd], \$password or BAIL_OUT($?);
226	run_log [ $klist, '-f' ] or BAIL_OUT($?);
227}
228
229END
230{
231	# take care not to change the script's exit value
232	my $exit_code = $?;
233
234	kill 'INT', `cat $kdc_pidfile`
235	  if defined($kdc_pidfile) && -f $kdc_pidfile;
236
237	$? = $exit_code;
238}
