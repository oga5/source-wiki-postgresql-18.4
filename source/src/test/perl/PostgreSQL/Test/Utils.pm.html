6=head1 NAME
7
8PostgreSQL::Test::Utils - helper module for writing PostgreSQL's C<prove> tests.
9
10=head1 SYNOPSIS
11
12  use PostgreSQL::Test::Utils;
13
14  # Test basic output of a command
15  program_help_ok('initdb');
16  program_version_ok('initdb');
17  program_options_handling_ok('initdb');
18
19  # Test option combinations
20  command_fails(['initdb', '--invalid-option'],
21              'command fails with invalid option');
22  my $tempdir = PostgreSQL::Test::Utils::tempdir;
23  command_ok('initdb', '--pgdata' => $tempdir);
24
25  # Miscellanea
26  print "on Windows" if $PostgreSQL::Test::Utils::windows_os;
27  ok(check_mode_recursive($stream_dir, 0700, 0600),
28    "check stream dir permissions");
29  PostgreSQL::Test::Utils::system_log('pg_ctl', 'kill', 'QUIT', $slow_pid);
30
31=head1 DESCRIPTION
32
33C<PostgreSQL::Test::Utils> contains a set of routines dedicated to environment setup for
34a PostgreSQL regression test run and includes some low-level routines
35aimed at controlling command execution, logging and test functions.
36
37=cut

320
321=item has_wal_read_bug()
322
323Returns true if $tmp_check is subject to a sparc64+ext4 bug that causes WAL
324readers to see zeros if another process simultaneously wrote the same offsets.
325Consult this in tests that fail frequently on affected configurations.  The
326bug has made streaming standbys fail to advance, reporting corrupt WAL.  It
327has made COMMIT PREPARED fail with "could not read two-phase state from WAL".
328Non-WAL PostgreSQL reads haven't been affected, likely because those readers
329and writers have buffering systems in common.  See
330https://postgr.es/m/20220116210241.GC756210@rfd.leadboat.com for details.
331
332=cut
333
334sub has_wal_read_bug
335{
336	return
337		 $Config{osname} eq 'linux'
338	  && $Config{archname} =~ /^sparc/
339	  && !run_log([ qw(df -x ext4), $tmp_check ], '&>' => '/dev/null');
340}

357=pod
358
359=item system_or_bail(@cmd)
360
361Run (via C<system()>) the command passed as argument, and returns
362if the command is successful.
363On failure, abandon further tests and exit the program.
364
365=cut
366
367sub system_or_bail
368{
369	if (system_log(@_) != 0)
370	{
371		if ($? == -1)
372		{
373			BAIL_OUT(
374				sprintf(
375					"failed to execute command \"%s\": $!", join(" ", @_)));
376		}
377		elsif ($? & 127)
378		{
379			BAIL_OUT(
380				sprintf(
381					"command \"%s\" died with signal %d",
382					join(" ", @_),
383					$? & 127));
384		}
385		else
386		{
387			BAIL_OUT(
388				sprintf(
389					"command \"%s\" exited with value %d",
390					join(" ", @_),
391					$? >> 8));
392		}
393	}
394}

431=pod
432
433=item pump_until(proc, timeout, stream, until)
434
435Pump until string is matched on the specified stream, or timeout occurs.
436
437=cut
438
439sub pump_until
440{
441	my ($proc, $timeout, $stream, $until) = @_;
442	$proc->pump_nb();
443	while (1)
444	{
445		last if $$stream =~ /$until/;
446		if ($timeout->is_expired)
447		{
448			diag(
449				"pump_until: timeout expired when searching for \"$until\" with stream: \"$$stream\""
450			);
451			return 0;
452		}
453		if (not $proc->pumpable())
454		{
455			diag(
456				"pump_until: process terminated unexpectedly when searching for \"$until\" with stream: \"$$stream\""
457			);
458			return 0;
459		}
460		$proc->pump();
461	}
462	return 1;
463}

503=pod
504
505=item slurp_file(filename [, $offset])
506
507Return the full contents of the specified file, beginning from an
508offset position if specified.
509
510=cut
511
512sub slurp_file
513{
514	my ($filename, $offset) = @_;
515	local $/;
516	my $contents;
517	my $fh;
518
519	# On windows open file using win32 APIs, to allow us to set the
520	# FILE_SHARE_DELETE flag ("d" below), otherwise other accesses to the file
521	# may fail.
522	if ($Config{osname} ne 'MSWin32')
523	{
524		open($fh, '<', $filename)
525		  or croak "could not read \"$filename\": $!";
526	}
527	else
528	{
529		my $fHandle = createFile($filename, "r", "rwd")
530		  or croak "could not open \"$filename\": $^E";
531		OsFHandleOpen($fh = IO::Handle->new(), $fHandle, 'r')
532		  or croak "could not read \"$filename\": $^E\n";
533	}
534
535	if (defined($offset))
536	{
537		seek($fh, $offset, SEEK_SET)
538		  or croak "could not seek \"$filename\": $!";
539	}
540
541	$contents = <$fh>;
542	close $fh;
543
544	return $contents;
545}

574sub string_replace_file
575{
576	my ($filename, $find, $replace) = @_;
577	open(my $in, '<', $filename) or croak $!;
578	my $content = '';
579	while (<$in>)

594=item check_mode_recursive(dir, expected_dir_mode, expected_file_mode, ignore_list)
595
596Check that all file/dir modes in a directory match the expected values,
597ignoring files in C<ignore_list> (basename only).
598
599=cut
600
601sub check_mode_recursive
602{
603	my ($dir, $expected_dir_mode, $expected_file_mode, $ignore_list) = @_;
604
605	# Result defaults to true
606	my $result = 1;
607
608	find(
609		{
610			follow_fast => 1,
611			wanted => sub {
612				# Is file in the ignore list?
613				foreach my $ignore ($ignore_list ? @{$ignore_list} : [])
614				{
615					if ("$dir/$ignore" eq $File::Find::name)
616					{
617						return;
618					}
619				}
620
621				# Allow ENOENT.  A running server can delete files, such as
622				# those in pg_stat.  Other stat() failures are fatal.
623				my $file_stat = stat($File::Find::name);
624				unless (defined($file_stat))
625				{
626					my $is_ENOENT = $!{ENOENT};
627					my $msg = "unable to stat $File::Find::name: $!";
628					if ($is_ENOENT)
629					{
630						warn $msg;
631						return;
632					}
633					else
634					{
635						die $msg;
636					}
637				}
638
639				my $file_mode = S_IMODE($file_stat->mode);
640
641				# Is this a file?
642				if (S_ISREG($file_stat->mode))
643				{
644					if ($file_mode != $expected_file_mode)
645					{
646						print(
647							*STDERR,
648							sprintf("$File::Find::name mode must be %04o\n",
649								$expected_file_mode));
650
651						$result = 0;
652						return;
653					}
654				}
655
656				# Else a directory?
657				elsif (S_ISDIR($file_stat->mode))
658				{
659					if ($file_mode != $expected_dir_mode)
660					{
661						print(
662							*STDERR,
663							sprintf("$File::Find::name mode must be %04o\n",
664								$expected_dir_mode));
665
666						$result = 0;
667						return;
668					}
669				}
670
671				# Else something we can't handle
672				else
673				{
674					die "unknown file type for $File::Find::name";
675				}
676			}
677		},
678		$dir);
679
680	return $result;
681}

694

696		{
697			follow_fast => 1,
698			wanted => sub {
699				my $file_stat = stat($File::Find::name);

702				{

707				}
708			}
709		},
710		$dir);
711	return;

714=pod
715
716=item scan_server_header(header_path, regexp)
717
718Returns an array that stores all the matches of the given regular expression
719within the PostgreSQL installation's C<header_path>.  This can be used to
720retrieve specific value patterns from the installation's header files.
721
722=cut
723
724sub scan_server_header
725{
726	my ($header_path, $regexp) = @_;
727
728	my ($stdout, $stderr);
729	my $result = IPC::Run::run [ 'pg_config', '--includedir-server' ],
730	  '>' => \$stdout,
731	  '2>' => \$stderr
732	  or die "could not execute pg_config";
733	chomp($stdout);
734	$stdout =~ s/\r$//;
735
736	open my $header_h, '<', "$stdout/$header_path" or die "$!";
737
738	my @match = undef;
739	while (<$header_h>)
740	{
741		my $line = $_;
742
743		if (@match = $line =~ /^$regexp/)
744		{
745			last;
746		}
747	}
748
749	close $header_h;
750	die "could not find match in header $header_path\n"
751	  unless @match;
752	return @match;
753}

764sub check_pg_config

781=pod
782
783=item compare_files(file1, file2, testname)
784
785Check that two files match, printing the difference if any.
786
787C<line_comp_function> is an optional CODE reference to a line comparison
788function, passed down as-is to File::Compare::compare_text.
789
790=cut
791
792sub compare_files
793{
794	my ($file1, $file2, $testname, $line_comp_function) = @_;
795
796	# If nothing is given, all lines should be equal.
797	$line_comp_function = sub { $_[0] ne $_[1] }
798	  unless defined $line_comp_function;
799
800	my $compare_res =
801	  File::Compare::compare_text($file1, $file2, $line_comp_function);
802	is($compare_res, 0, $testname);
803
804	# Provide more context if the files do not match.
805	if ($compare_res != 0)
806	{
807		my ($stdout, $stderr) =
808		  run_command([ 'diff', '-u', $file1, $file2 ]);
809		print "=== diff of $file1 and $file2\n";
810		print "=== stdout ===\n";
811		print $stdout;
812		print "=== stderr ===\n";
813		print $stderr;
814		print "=== EOF ===\n";
815	}
816
817	return;
818}

824Portably create a symlink for a directory. On Windows this creates a junction
825point. Elsewhere it just calls perl's builtin symlink.
826
827=cut
828
829sub dir_symlink
830{
831	my $oldname = shift;
832	my $newname = shift;
833	if ($windows_os)
834	{
835		$oldname =~ s,/,\\,g;
836		$newname =~ s,/,\\,g;
837		my $cmd = qq{mklink /j "$newname" "$oldname"};
838		if ($Config{osname} eq 'msys')
839		{
840			# need some indirection on msys
841			$cmd = qq{echo '$cmd' | \$COMSPEC /Q};
842		}
843		system($cmd) == 0 or die;
844	}
845	else
846	{
847		symlink $oldname, $newname or die $!;
848	}
849	die "No $newname" unless -e $newname;
850}

900sub command_exit_is
901{
902	local $Test::Builder::Level = $Test::Builder::Level + 1;
903	my ($cmd, $expected, $test_name) = @_;
904	print("# Running: " . join(" ", @{$cmd}) . "\n");
905	my $h = IPC::Run::start $cmd;
906	$h->finish();
907
908	# Normally, if the child called exit(N), IPC::Run::result() returns N.  On
909	# Windows, with IPC::Run v20220807.0 and earlier, full_results() is the
910	# method that returns N (https://github.com/toddr/IPC-Run/issues/161).
911	my $result =
912	  ($Config{osname} eq "MSWin32" && $IPC::Run::VERSION <= 20220807.0)
913	  ? ($h->full_results)[0]
914	  : $h->result(0);
915	is($result, $expected, $test_name);
916	return;
917}

1021=item command_like_safe(cmd, expected_stdout, test_name)
1022
1023Check that the command runs successfully and the output
1024matches the given regular expression.  Doesn't assume that the
1025output files are closed.
1026
1027=cut

1029sub command_like_safe
1030{
1031	local $Test::Builder::Level = $Test::Builder::Level + 1;
1032
1033	# Doesn't rely on detecting end of file on the file descriptors,
1034	# which can fail, causing the process to hang, notably on Msys
1035	# when used with 'pg_ctl start'
1036	my ($cmd, $expected_stdout, $test_name) = @_;
1037	my ($stdout, $stderr);
1038	my $stdoutfile = File::Temp->new();
1039	my $stderrfile = File::Temp->new();
1040	print("# Running: " . join(" ", @{$cmd}) . "\n");
1041	my $result = IPC::Run::run $cmd, '>' => $stdoutfile, '2>' => $stderrfile;
1042	$stdout = slurp_file($stdoutfile);
1043	$stderr = slurp_file($stderrfile);
1044	ok($result, "$test_name: exit code 0");
1045	is($stderr, '', "$test_name: no stderr");
1046	like($stdout, $expected_stdout, "$test_name: matches");
1047	return;

1119{

1121

1123
1124	# run command
1125	my ($stdout, $stderr);
1126	print("# Running: " . join(" ", @{$cmd}) . "\n");
1127	IPC::Run::run($cmd, '>' => \$stdout, '2>' => \$stderr);
1128

1130	my $ret = $?;

1132	  if $ret & 127;

1137		"$test_name status (got $ret vs expected $expected_ret)");

1140	for my $re (@$out)
1141	{
1142		like($stdout, $re, "$test_name stdout /$re/");
1143	}
1144
1145	# check stderr
1146	for my $re (@$err)
1147	{
1148		like($stderr, $re, "$test_name stderr /$re/");
1149	}
1150
1151	return;

1156=item tar_portability_options(tar)
1157
1158Check for non-default options we need to give to tar to create
1159a tarfile we can decode (i.e., no "pax" extensions).
1160Not needed in tests that only use tar to read tarfiles.
1161
1162Returns options as an array.
1163
1164=cut
1165
1166sub tar_portability_options
1167{
1168	local $Test::Builder::Level = $Test::Builder::Level + 1;
1169
1170	my ($tar) = @_;
1171
1172	my @tar_p_flags = ();
1173
1174	return @tar_p_flags if (!defined $tar || $tar eq '');
1175
1176	# GNU tar typically produces gnu-format archives, which we can read fine.
1177	# But some platforms configure it to default to posix/pax format, and
1178	# apparently they enable --sparse too.  Override that.
1179	#
1180	# ustar format supports UIDs only up to 2^21 - 1 (2097151).  Override
1181	# owner/group to avoid failures on systems where the running user's UID/GID
1182	# exceeds that limit.
1183	my $devnull = File::Spec->devnull();
1184	if (system(
1185			"$tar --format=ustar --owner=0 --group=0 -cf $devnull $devnull 2>$devnull"
1186		) == 0)
1187	{
1188		# GNU tar (Linux), BSD tar (FreeBSD, NetBSD, macOS, Windows)
1189		push(@tar_p_flags, "--format=ustar", "--owner=0", "--group=0");
1190	}
1191	elsif (system("$tar -F ustar -cf $devnull $devnull 2>$devnull") == 0)
1192	{
1193		# OpenBSD tar
1194		push(@tar_p_flags, "-F", "ustar");
1195	}
1196	return @tar_p_flags;
1197}
