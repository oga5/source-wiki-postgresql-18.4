3# Test the negotiation of combined SSL and GSS requests.  This test
4# relies on both SSL and GSS requests to be rejected first, followed
5# by more requests.
6
7use strict;
8use warnings FATAL => 'all';
9use PostgreSQL::Test::Cluster;
10use PostgreSQL::Test::Utils;
11use Test::More;
12use Time::HiRes qw(usleep);
13
14my $node = PostgreSQL::Test::Cluster->new('main');
15$node->init;
16$node->append_conf('postgresql.conf', "log_min_messages = debug2");
17$node->append_conf('postgresql.conf',
18	"log_connections = 'receipt,authentication,authorization'");
19$node->append_conf('postgresql.conf', 'trace_connection_negotiation=on');
20$node->start;
21
22if (!$node->raw_connect_works())
23{
24	plan skip_all => "this test requires working raw_connect()";
25}
26
27my $sock = $node->raw_connect();
28
29# SSLRequest: packet length followed by NEGOTIATE_SSL_CODE.
30my $ssl_request = pack("Nnn", 8, 1234, 5679);
31
32# GSSENCRequest: packet length followed by NEGOTIATE_GSS_CODE.
33my $gss_request = pack("Nnn", 8, 1234, 5680);
34
35# Send SSLRequest, reject or bypass.
36$sock->send($ssl_request);
37my $reply = "";
38$sock->recv($reply, 1);
39if ($reply ne 'N')
40{
41	$sock->close();
42	plan skip_all =>
43	  "server accepted SSL; test requires SSL to be rejected";
44}
45
46# Send GSSENCRequest, reject or bypass test.
47$sock->send($gss_request);
48$reply = "";
49$sock->recv($reply, 1);
50if ($reply ne 'N')
51{
52	$sock->close();
53	plan skip_all =>
54	  "server accepted GSS; test requires GSS to be rejected";
55}
56
57my $log_offset = -s $node->logfile;
58
59# Send a second SSLRequest, now that we know that both SSL and GSS have
60# been rejected for this connection.  We are done with both requests, so
61# extra requests will be rejected and fail with an invalid protocol
62# version, and the connection should be closed by the server.
63$sock->send($ssl_request);
64
65# Try to read a response, there should be nothing, and certainly not an
66# extra 'N' message indicating a rejection.
67$reply = "";
68my $bytes = $sock->recv($reply, 1024);
69isnt($reply, 'N',
70	"server does not re-enter SSL negotiation after SSL+GSS were both tried");
71
72$sock->close();
73$node->wait_for_log(qr/FATAL: .* unsupported frontend protocol 1234.5679/,
74					$log_offset);
75
76# Check extra connection with a simple query.
77my $result = $node->safe_psql('postgres', 'select 1;');
78is($result, '1', 'server able to accept connection');
79
80$node->stop;
81
82done_testing();
