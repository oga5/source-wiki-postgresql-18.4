6use strict;
7use warnings FATAL => 'all';
8use PostgreSQL::Test::Cluster;
9use PostgreSQL::Test::Utils;
10use Test::More;
11
12#
13# Test that dead-end backends don't prevent the server from shutting
14# down.
15#
16# Dead-end backends can linger until they reach
17# authentication_timeout. We use a long authentication_timeout and a
18# much shorter timeout for the "pg_ctl stop" operation, to test that
19# if dead-end backends are killed at fast shut down. If they're not,
20# "pg_ctl stop" will error out before the authentication timeout kicks
21# in and cleans up the dead-end backends.
22my $authentication_timeout = $PostgreSQL::Test::Utils::timeout_default;
23
24# Don't fail due to hitting the max value allowed for authentication_timeout.
25$authentication_timeout = 600 unless $authentication_timeout < 600;
26
27my $stop_timeout = $authentication_timeout / 2;
28
29# Initialize the server with low connection limits, to test dead-end backends
30my $node = PostgreSQL::Test::Cluster->new('main');
31$node->init;
32$node->append_conf('postgresql.conf', "max_connections = 5");
33$node->append_conf('postgresql.conf', "max_wal_senders = 0");
34$node->append_conf('postgresql.conf', "autovacuum_max_workers = 1");
35$node->append_conf('postgresql.conf', "max_worker_processes = 1");
36$node->append_conf('postgresql.conf',
37	"log_connections = 'receipt,authentication,authorization'");
38$node->append_conf('postgresql.conf', "log_min_messages = debug2");
39$node->append_conf('postgresql.conf',
40	"authentication_timeout = '$authentication_timeout s'");
41$node->append_conf('postgresql.conf', 'trace_connection_negotiation=on');
42$node->start;
43
44if (!$node->raw_connect_works())
45{
46	plan skip_all => "this test requires working raw_connect()";
47}
48
49# Restart the server to ensure that the backend launched for
50# raw_connect_works() is gone. Otherwise, it might free up the
51# connection slot later, when we expect all the slots to be in use.
52$node->restart;
53
54my @raw_connections = ();
55
56# Open a lot of TCP (or Unix domain socket) connections to use up all
57# the connection slots. Beyond a certain number (roughly 2x
58# max_connections), they will be "dead-end backends".
59for (my $i = 0; $i <= 20; $i++)
60{
61	my $sock = $node->raw_connect();
62
63	# On a busy system, the server might reject connections if
64	# postmaster cannot accept() them fast enough. The exact limit
65	# and behavior depends on the platform. To make this reliable,
66	# we attempt SSL negotiation on each connection before opening
67	# next one. The server will reject the SSL negotiations, but
68	# when it does so, we know that the backend has been launched
69	# and we should be able to open another connection.
70
71	# SSLRequest packet consists of packet length followed by
72	# NEGOTIATE_SSL_CODE.
73	my $negotiate_ssl_code = pack("Nnn", 8, 1234, 5679);
74	my $sent = $sock->send($negotiate_ssl_code);
75
76	# Read reply. We expect the server to reject it with 'N'
77	my $reply = "";
78	$sock->recv($reply, 1);
79	is($reply, "N", "dead-end connection $i");
80
81	push(@raw_connections, $sock);
82}
83
84# When all the connection slots are in use, new connections will fail
85# before even looking up the user. Hence you now get "sorry, too many
86# clients already" instead of "role does not exist" error. Test that
87# to ensure that we have used up all the slots.
88$node->connect_fails("dbname=postgres user=invalid_user",
89	"connection is rejected when all slots are in use",
90	expected_stderr => qr/FATAL:  sorry, too many clients already/);
91
92# Open one more connection, to really ensure that we have at least one
93# dead-end backend.
94my $sock = $node->raw_connect();
95
96# Test that the dead-end backends don't prevent the server from stopping.
97$node->stop('fast', timeout => $stop_timeout);
98
99$node->start();
100$node->connect_ok("dbname=postgres", "works after restart");
101
102# Clean up
103foreach my $socket (@raw_connections)
104{
105	$socket->close();
106}
107
108done_testing();
