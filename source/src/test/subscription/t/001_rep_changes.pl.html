207# Test behaviour of ALTER PUBLICATION ... DROP TABLE
208#
209# When a publisher drops a table from publication, it should also stop sending
210# its changes to subscribers. We look at the subscriber whether it receives
211# the row that is inserted to the table on the publisher after it is dropped
212# from the publication.
213$result = $node_subscriber->safe_psql('postgres',
214	"SELECT count(*), min(a), max(a) FROM tab_ins");
215is($result, qq(1052|1|1002),
216	'check rows on subscriber before table drop from publication');

220	"ALTER PUBLICATION tap_pub_ins_only DROP TABLE tab_ins");

453# check that change of connection string and/or publication list causes
454# restart of subscription workers. We check the state along with

457# Not all of these are registered as tests as we need to poll for a change
