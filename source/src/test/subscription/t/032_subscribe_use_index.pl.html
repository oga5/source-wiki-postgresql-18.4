10# create publisher node
11my $node_publisher = PostgreSQL::Test::Cluster->new('publisher');
12$node_publisher->init(allows_streaming => 'logical');
13$node_publisher->start;
14
15# create subscriber node
16my $node_subscriber = PostgreSQL::Test::Cluster->new('subscriber');
17$node_subscriber->init;
18$node_subscriber->start;
19
20my $publisher_connstr = $node_publisher->connstr . ' dbname=postgres';
21my $appname = 'tap_sub';

24# =============================================================================
25# Testcase start: Subscription can use index with multiple rows and columns
26#
27
28# create tables pub and sub
29$node_publisher->safe_psql('postgres',
30	"CREATE TABLE test_replica_id_full (x int, y text)");
31$node_publisher->safe_psql('postgres',
32	"ALTER TABLE test_replica_id_full REPLICA IDENTITY FULL");
33$node_subscriber->safe_psql('postgres',
34	"CREATE TABLE test_replica_id_full (x int, y text)");
35$node_subscriber->safe_psql('postgres',
36	"CREATE INDEX test_replica_id_full_idx ON test_replica_id_full(x,y)");
37
38# insert some initial data within the range 0-9 for x and y
39$node_publisher->safe_psql('postgres',
40	"INSERT INTO test_replica_id_full SELECT (i%10), (i%10)::text FROM generate_series(0,10) i"
41);
42
43# create pub/sub
44$node_publisher->safe_psql('postgres',
45	"CREATE PUBLICATION tap_pub_rep_full FOR TABLE test_replica_id_full");
46$node_subscriber->safe_psql('postgres',
47	"CREATE SUBSCRIPTION tap_sub_rep_full CONNECTION '$publisher_connstr application_name=$appname' PUBLICATION tap_pub_rep_full"
48);
49
50# wait for initial table synchronization to finish
51$node_subscriber->wait_for_subscription_sync($node_publisher, $appname);
52
53# delete 2 rows
54$node_publisher->safe_psql('postgres',
55	"DELETE FROM test_replica_id_full WHERE x IN (5, 6)");
56
57# update 2 rows
58$node_publisher->safe_psql('postgres',
59	"UPDATE test_replica_id_full SET x = 100, y = '200' WHERE x IN (1, 2)");
60
61# wait until the index is used on the subscriber
62$node_publisher->wait_for_catchup($appname);
63$node_subscriber->poll_query_until('postgres',
64	q{select (idx_scan = 4) from pg_stat_all_indexes where indexrelname = 'test_replica_id_full_idx';}
65  )
66  or die
67  "Timed out while waiting for check subscriber tap_sub_rep_full updates 4 rows via index";
68
69# make sure that the subscriber has the correct data after the UPDATE
70$result = $node_subscriber->safe_psql('postgres',
71	"select count(*) from test_replica_id_full WHERE (x = 100 and y = '200')"
72);
73is($result, qq(2),
74	'ensure subscriber has the correct data at the end of the test');
75
76# make sure that the subscriber has the correct data after the first DELETE
77$result = $node_subscriber->safe_psql('postgres',
78	"select count(*) from test_replica_id_full where x in (5, 6)");
79is($result, qq(0),
80	'ensure subscriber has the correct data at the end of the test');
81
82# cleanup pub
83$node_publisher->safe_psql('postgres', "DROP PUBLICATION tap_pub_rep_full");
84$node_publisher->safe_psql('postgres', "DROP TABLE test_replica_id_full");
85# cleanup sub
86$node_subscriber->safe_psql('postgres', "DROP SUBSCRIPTION tap_sub_rep_full");
87$node_subscriber->safe_psql('postgres', "DROP TABLE test_replica_id_full");

95# create tables pub and sub
96$node_publisher->safe_psql('postgres',
97	"CREATE TABLE users_table_part(user_id bigint, value_1 int, value_2 int) PARTITION BY RANGE (value_1)"
98);
99$node_publisher->safe_psql('postgres',
100	"CREATE TABLE users_table_part_0 PARTITION OF users_table_part FOR VALUES FROM (0) TO (10)"
101);
102$node_publisher->safe_psql('postgres',
103	"CREATE TABLE users_table_part_1 PARTITION OF users_table_part FOR VALUES FROM (10) TO (20)"
104);
105
106$node_publisher->safe_psql('postgres',
107	"ALTER TABLE users_table_part REPLICA IDENTITY FULL");
108$node_publisher->safe_psql('postgres',
109	"ALTER TABLE users_table_part_0 REPLICA IDENTITY FULL");
110$node_publisher->safe_psql('postgres',
111	"ALTER TABLE users_table_part_1 REPLICA IDENTITY FULL");
112
113$node_subscriber->safe_psql('postgres',
114	"CREATE TABLE users_table_part(user_id bigint, value_1 int, value_2 int) PARTITION BY RANGE (value_1)"
115);
116$node_subscriber->safe_psql('postgres',
117	"CREATE TABLE users_table_part_0 PARTITION OF users_table_part FOR VALUES FROM (0) TO (10)"
118);
119$node_subscriber->safe_psql('postgres',
120	"CREATE TABLE users_table_part_1 PARTITION OF users_table_part FOR VALUES FROM (10) TO (20)"
121);
122$node_subscriber->safe_psql('postgres',
123	"CREATE INDEX users_table_part_idx ON users_table_part(user_id, value_1)"
124);
125
126# insert some initial data
127$node_publisher->safe_psql('postgres',
128	"INSERT INTO users_table_part SELECT (i%100), (i%20), i FROM generate_series(0,100) i"
129);
130
131# create pub/sub
132$node_publisher->safe_psql('postgres',
133	"CREATE PUBLICATION tap_pub_rep_full FOR TABLE users_table_part");
134$node_subscriber->safe_psql('postgres',
135	"CREATE SUBSCRIPTION tap_sub_rep_full CONNECTION '$publisher_connstr application_name=$appname' PUBLICATION tap_pub_rep_full"
136);
137
138# wait for initial table synchronization to finish
139$node_subscriber->wait_for_subscription_sync($node_publisher, $appname);
140
141# update rows, moving them to other partitions
142$node_publisher->safe_psql('postgres',
143	"UPDATE users_table_part SET value_1 = 0 WHERE user_id = 4");
144
145# delete rows from different partitions
146$node_publisher->safe_psql('postgres',
147	"DELETE FROM users_table_part WHERE user_id = 1 and value_1 = 1");
148$node_publisher->safe_psql('postgres',
149	"DELETE FROM users_table_part WHERE user_id = 12 and value_1 = 12");
150
151# wait until the index is used on the subscriber
152$node_publisher->wait_for_catchup($appname);
153$node_subscriber->poll_query_until('postgres',
154	q{select sum(idx_scan)=3 from pg_stat_all_indexes where indexrelname ilike 'users_table_part_%';}
155  )
156  or die
157  "Timed out while waiting for check subscriber tap_sub_rep_full updates partitioned table";
158
159# make sure that the subscriber has the correct data
160$result = $node_subscriber->safe_psql('postgres',
161	"select sum(user_id+value_1+value_2) from users_table_part");
162is($result, qq(10907),
163	'ensure subscriber has the correct data at the end of the test');
164$result = $node_subscriber->safe_psql('postgres',
165	"select count(DISTINCT(user_id,value_1, value_2)) from users_table_part");
166is($result, qq(99),
167	'ensure subscriber has the correct data at the end of the test');
168
169# cleanup pub
170$node_publisher->safe_psql('postgres', "DROP PUBLICATION tap_pub_rep_full");
171$node_publisher->safe_psql('postgres', "DROP TABLE users_table_part");
172
173# cleanup sub
174$node_subscriber->safe_psql('postgres', "DROP SUBSCRIPTION tap_sub_rep_full");
175$node_subscriber->safe_psql('postgres', "DROP TABLE users_table_part");

195);

198$node_subscriber->safe_psql('postgres',
199	"CREATE INDEX people_names_partial ON people(firstname) WHERE (firstname = 'first_name_1')"
200);
201
202# insert some initial data
203$node_publisher->safe_psql('postgres',
204	"INSERT INTO people SELECT 'first_name_' || i::text, 'last_name_' || i::text FROM generate_series(0,200) i"
205);
206
207# create pub/sub
208$node_publisher->safe_psql('postgres',
209	"CREATE PUBLICATION tap_pub_rep_full FOR TABLE people");

211	"CREATE SUBSCRIPTION tap_sub_rep_full CONNECTION '$publisher_connstr application_name=$appname' PUBLICATION tap_pub_rep_full"
212);
213
214# wait for initial table synchronization to finish
215$node_subscriber->wait_for_subscription_sync($node_publisher, $appname);
216
217# update 2 rows
218$node_publisher->safe_psql('postgres',
219	"UPDATE people SET firstname = 'no-name' WHERE firstname = 'first_name_1'"
220);
221$node_publisher->safe_psql('postgres',
222	"UPDATE people SET firstname = 'no-name' WHERE firstname = 'first_name_2' AND lastname = 'last_name_2'"

224

226$node_publisher->wait_for_catchup($appname);

231	'ensure subscriber tap_sub_rep_full updates two rows via seq. scan with index on expressions'
232);
233
234$node_publisher->safe_psql('postgres',
235	"DELETE FROM people WHERE firstname = 'first_name_3'");
236$node_publisher->safe_psql('postgres',
237	"DELETE FROM people WHERE firstname = 'first_name_4' AND lastname = 'last_name_4'"
238);
239
240# make sure the index is not used on the subscriber
241$node_publisher->wait_for_catchup($appname);
242$result = $node_subscriber->safe_psql('postgres',
243	"select sum(idx_scan) from pg_stat_all_indexes where indexrelname IN ('people_names_expr_only', 'people_names_partial')"
244);
245is($result, qq(0),
246	'ensure subscriber tap_sub_rep_full updates two rows via seq. scan with index on expressions'
247);
248
249# make sure that the subscriber has the correct data
250$result =
251  $node_subscriber->safe_psql('postgres', "SELECT count(*) FROM people");

259$node_subscriber->safe_psql('postgres', "DROP SUBSCRIPTION tap_sub_rep_full");

411# Testcase start: Subscription using a unique index when Pub/Sub has different
412# data
413#
414# The subscriber has duplicate tuples that publisher does not have. When
415# publisher updates/deletes 1 row, subscriber uses indexes and updates/deletes
416# exactly 1 row.
417#
418
419# create tables pub and sub
420$node_publisher->safe_psql('postgres',
421	"CREATE TABLE test_replica_id_full (x int, y int)");
422$node_publisher->safe_psql('postgres',
423	"ALTER TABLE test_replica_id_full REPLICA IDENTITY FULL");
424$node_subscriber->safe_psql('postgres',
425	"CREATE TABLE test_replica_id_full (x int, y int)");
426$node_subscriber->safe_psql('postgres',
427	"CREATE UNIQUE INDEX test_replica_id_full_idxy ON test_replica_id_full(x,y)"
428);
429
430# insert some initial data
431$node_publisher->safe_psql('postgres',
432	"INSERT INTO test_replica_id_full SELECT i, i FROM generate_series(0,21) i"
433);
434
435# create pub/sub
436$node_publisher->safe_psql('postgres',
437	"CREATE PUBLICATION tap_pub_rep_full FOR TABLE test_replica_id_full");
438$node_subscriber->safe_psql('postgres',
439	"CREATE SUBSCRIPTION tap_sub_rep_full CONNECTION '$publisher_connstr application_name=$appname' PUBLICATION tap_pub_rep_full"
440);
441
442# wait for initial table synchronization to finish
443$node_subscriber->wait_for_subscription_sync($node_publisher, $appname);
444
445# duplicate the data in subscriber for y column
446$node_subscriber->safe_psql('postgres',
447	"INSERT INTO test_replica_id_full SELECT i+100, i FROM generate_series(0,21) i"
448);
449
450# now, we update only 1 row on the publisher and expect the subscriber to only
451# update 1 row although there are two tuples with y = 15 on the subscriber
452$node_publisher->safe_psql('postgres',
453	"UPDATE test_replica_id_full SET x = 2000 WHERE y = 15");
454
455# wait until the index is used on the subscriber
456$node_publisher->wait_for_catchup($appname);
457$node_subscriber->poll_query_until('postgres',
458	q{select (idx_scan = 1) from pg_stat_all_indexes where indexrelname = 'test_replica_id_full_idxy';}
459  )
460  or die
461  "Timed out while waiting for check subscriber tap_sub_rep_full updates one row via index";

466	"SELECT count(*) FROM test_replica_id_full WHERE x = 2000");
467is($result, qq(1),
468	'ensure subscriber has the correct data at the end of the test');
469
470# cleanup pub
471$node_publisher->safe_psql('postgres', "DROP PUBLICATION tap_pub_rep_full");
472$node_publisher->safe_psql('postgres', "DROP TABLE test_replica_id_full");
473# cleanup sub
474$node_subscriber->safe_psql('postgres', "DROP SUBSCRIPTION tap_sub_rep_full");
475$node_subscriber->safe_psql('postgres', "DROP TABLE test_replica_id_full");
