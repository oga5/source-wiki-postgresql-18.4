4# Test materialized views behavior
5use strict;
6use warnings FATAL => 'all';
7use PostgreSQL::Test::Cluster;
8use PostgreSQL::Test::Utils;
9use Test::More;
10
11my $node_publisher = PostgreSQL::Test::Cluster->new('publisher');
12$node_publisher->init(allows_streaming => 'logical');
13$node_publisher->start;
14
15my $node_subscriber = PostgreSQL::Test::Cluster->new('subscriber');
16$node_subscriber->init;
17$node_subscriber->start;
18
19my $publisher_connstr = $node_publisher->connstr . ' dbname=postgres';
20
21$node_publisher->safe_psql('postgres',
22	"CREATE PUBLICATION mypub FOR ALL TABLES;");
23$node_subscriber->safe_psql('postgres',
24	"CREATE SUBSCRIPTION mysub CONNECTION '$publisher_connstr' PUBLICATION mypub;"
25);
26
27$node_publisher->safe_psql('postgres',
28	q{CREATE TABLE test1 (a int PRIMARY KEY, b text)});
29$node_publisher->safe_psql('postgres',
30	q{INSERT INTO test1 (a, b) VALUES (1, 'one'), (2, 'two');});
31
32$node_subscriber->safe_psql('postgres',
33	q{CREATE TABLE test1 (a int PRIMARY KEY, b text);});
34
35$node_publisher->wait_for_catchup('mysub');
36
37# Materialized views are not supported by logical replication, but
38# logical decoding does produce change information for them, so we
39# need to make sure they are properly ignored. (bug #15044)
40
41# create a MV with some data
42$node_publisher->safe_psql('postgres',
43	q{CREATE MATERIALIZED VIEW testmv1 AS SELECT * FROM test1;});
44$node_publisher->wait_for_catchup('mysub');
45
46# There is no equivalent relation on the subscriber, but MV data is
47# not replicated, so this does not hang.
48
49pass "materialized view data not replicated";
50
51$node_subscriber->stop;
52$node_publisher->stop;
53
54done_testing();
