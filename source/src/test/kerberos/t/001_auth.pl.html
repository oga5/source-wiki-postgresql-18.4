4# Sets up a KDC and then runs a variety of tests to make sure that the
5# GSSAPI/Kerberos authentication and encryption are working properly,
6# that the options in pg_hba.conf and pg_ident.conf are handled correctly,
7# that the server-side pg_stat_gssapi view reports what we expect to
8# see for each test and that SYSTEM_USER returns what we expect to see.
9#
10# Also test that GSSAPI delegation is working properly and that those
11# credentials can be used to make dblink / postgres_fdw connections.
12#
13# Since this requires setting up a full KDC, it doesn't make much sense
14# to have multiple test scripts (since they'd have to also create their
15# own KDC and that could cause race conditions or other problems)- so
16# just add whatever other tests are needed to here.
17#
18# See the README for additional information.
19

117# Test connection success or failure, and if success, that query returns true.
118sub test_access
119{
120	local $Test::Builder::Level = $Test::Builder::Level + 1;
121
122	my ($node, $role, $query, $expected_res, $gssencmode, $test_name,
123		@expect_log_msgs)
124	  = @_;
125
126	# need to connect over TCP/IP for Kerberos
127	my $connstr = $node->connstr('postgres')
128	  . " user=$role host=$host hostaddr=$hostaddr $gssencmode";
129
130	my %params = (sql => $query,);
131
132	if (@expect_log_msgs)
133	{
134		# Match every message literally.
135		my @regexes = map { qr/\Q$_\E/ } @expect_log_msgs;
136
137		$params{log_like} = \@regexes;
138	}
139
140	if ($expected_res eq 0)
141	{
142		# The result is assumed to match "true", or "t", here.
143		$params{expected_stdout} = qr/^t$/;
144
145		$node->connect_ok($connstr, $test_name, %params);
146	}
147	else
148	{
149		$node->connect_fails($connstr, $test_name, %params);
150	}
151}
