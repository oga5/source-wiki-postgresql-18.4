src/test/isolation/specs/referential-integrity.spec
1# Referential Integrity test
2#
3# The assumption here is that the application code issuing the SELECT
4# to test for the presence or absence of a related record would do the
5# right thing -- this script doesn't include that logic.
6#
7# Any overlap between the transactions must cause a serialization failure.
8
9setup
10{
11 CREATE TABLE a (i int PRIMARY KEY);
12 CREATE TABLE b (a_id int);
13 INSERT INTO a VALUES (1);
14}
15
16teardown
17{
18 DROP TABLE a, b;
19}
20
21session s1
22setup		{ BEGIN ISOLATION LEVEL SERIALIZABLE; }
23step rx1	{ SELECT i FROM a WHERE i = 1; }
24step wy1	{ INSERT INTO b VALUES (1); }
25step c1		{ COMMIT; }
26
27session s2
28setup		{ BEGIN ISOLATION LEVEL SERIALIZABLE; }
29step rx2	{ SELECT i FROM a WHERE i = 1; }
30step ry2	{ SELECT a_id FROM b WHERE a_id = 1; }
31step wx2	{ DELETE FROM a WHERE i = 1; }
32step c2		{ COMMIT; }
