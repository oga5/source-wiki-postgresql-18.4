1# Test for page level predicate locking in hash index
2#
3# Test to verify serialization failures and to check reduced false positives
4#
5# To verify serialization failures, queries and permutations are written in such
6# a way that an index scan  (from one transaction) and an index insert (from
7# another transaction) will try to access the same bucket of the index
8# whereas to check reduced false positives, they will try to access different
9# buckets of the index.
10
11setup
12{
13 create table hash_tbl(id int4, p integer);
14 create index hash_idx on hash_tbl using hash(p);
15 insert into hash_tbl (id, p)
16 select g, 10 from generate_series(1, 10) g;
17 insert into hash_tbl (id, p)
18 select g, 20 from generate_series(11, 20) g;
19 insert into hash_tbl (id, p)
20 select g, 30 from generate_series(21, 30) g;
21 insert into hash_tbl (id, p)
22 select g, 40 from generate_series(31, 40) g;
23}
24
25teardown
26{
27 drop table hash_tbl;
28}
29
30session s1
31setup
32{
33 begin isolation level serializable;
34 set enable_seqscan=off;
35 set enable_bitmapscan=off;
36 set enable_indexonlyscan=on;
37}
38step rxy1	{ select sum(p) from hash_tbl where p=20; }
39step wx1	{ insert into hash_tbl (id, p)
40			  select g, 30 from generate_series(41, 50) g; }
41step rxy3	{ select sum(p) from hash_tbl where p=20; }
42step wx3	{ insert into hash_tbl (id, p)
43			  select g, 50 from generate_series(41, 50) g; }
44step c1		{ commit; }
