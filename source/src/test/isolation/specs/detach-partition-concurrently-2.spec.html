src/test/isolation/specs/detach-partition-concurrently-2.spec
1# Test that detach partition concurrently makes the partition safe
2# for foreign keys that reference it.
3
4setup
5{
6  DROP TABLE IF EXISTS d_lp_fk, d_lp_fk_1, d_lp_fk_2, d_lp_fk_r;
7
8  CREATE TABLE d_lp_fk (a int PRIMARY KEY) PARTITION BY LIST(a);
9  CREATE TABLE d_lp_fk_1 PARTITION OF d_lp_fk FOR VALUES IN (1);
10  CREATE TABLE d_lp_fk_2 PARTITION OF d_lp_fk FOR VALUES IN (2);
11  INSERT INTO d_lp_fk VALUES (1), (2);
12
13  CREATE TABLE d_lp_fk_r (a int references d_lp_fk);
14}
15
16teardown { DROP TABLE IF EXISTS d_lp_fk, d_lp_fk_1, d_lp_fk_2, d_lp_fk_r; }
17
18session s1
19step s1b		{ BEGIN; }
20step s1s		{ SELECT * FROM d_lp_fk; }
21step s1c		{ COMMIT; }
22
23session s2
24step s2d		{ ALTER TABLE d_lp_fk DETACH PARTITION d_lp_fk_1 CONCURRENTLY; }
25
26session s3
27step s3b		{ BEGIN; }
28step s3i1		{ INSERT INTO d_lp_fk_r VALUES (1); }
29step s3i2		{ INSERT INTO d_lp_fk_r VALUES (2); }
30step s3c		{ COMMIT; }
31
32# The transaction that detaches hangs until it sees any older transaction
33# terminate.
34permutation s1b s1s s2d s3i1 s1c
35permutation s1b s1s s2d s3i2 s3i2 s1c
36
37permutation s1b s1s s3i1 s2d s1c
38permutation s1b s1s s3i2 s2d s1c
39
40# what if s3 has an uncommitted insertion?
41permutation s1b s1s s3b s2d s3i1 s1c s3c
