3    CREATE TABLE trigtest(key text primary key, data text);
4
5    CREATE FUNCTION noisy_oper(p_comment text, p_a anynonarray, p_op text, p_b anynonarray)
6    RETURNS bool LANGUAGE plpgsql AS $body$
7        DECLARE
8            r bool;
9        BEGIN
10            EXECUTE format('SELECT $1 %s $2', p_op) INTO r USING p_a, p_b;
11            RAISE NOTICE '%: % % % % %: %', p_comment, pg_typeof(p_a), p_a, p_op, pg_typeof(p_b), p_b, r;
12        RETURN r;
13    END;$body$;
14
15    CREATE FUNCTION trig_report() RETURNS TRIGGER LANGUAGE plpgsql AS $body$
16    DECLARE
17	r_new text;
18	r_old text;
19	r_ret record;
20    BEGIN
21	-- In older releases it wasn't allowed to reference OLD/NEW
22        -- when not applicable for TG_WHEN
23	IF TG_OP = 'INSERT' THEN
24	    r_old = NULL;
25	    r_new = NEW;
26	    r_ret = NEW;
27	ELSIF TG_OP = 'DELETE' THEN
28	    r_old = OLD;
29	    r_new = NULL;
30	    r_ret = OLD;
31	ELSIF TG_OP = 'UPDATE' THEN
32	    r_old = OLD;
33	    r_new = NEW;
34	    r_ret = NEW;
35	END IF;
36
37	IF TG_WHEN = 'AFTER' THEN
38	   r_ret = NULL;
39	END IF;
40
41        RAISE NOTICE 'trigger: name %; when: %; lev: %s; op: %; old: % new: %',
42            TG_NAME, TG_WHEN, TG_LEVEL, TG_OP, r_old, r_new;
43
44	RETURN r_ret;
45    END;
46    $body$;
47}
