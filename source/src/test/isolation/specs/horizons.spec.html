src/test/isolation/specs/horizons.spec
1# Test that pruning and vacuuming pay attention to concurrent sessions
2# in the right way. For normal relations that means that rows cannot
3# be pruned away if there's an older snapshot, in contrast to that
4# temporary tables should nearly always be prunable.
5#
6# NB: Think hard before adding a test showing that rows in permanent
7# tables get pruned - it's quite likely that it'd be racy, e.g. due to
8# an autovacuum worker holding a snapshot.
9
10setup {
11    CREATE OR REPLACE FUNCTION explain_json(p_query text)
12    RETURNS json
13    LANGUAGE plpgsql AS $$
14        DECLARE
15            v_ret json;
16        BEGIN
17            EXECUTE p_query INTO STRICT v_ret;
18            RETURN v_ret;
19        END;$$;
20}
21
22teardown {
23    DROP FUNCTION explain_json(text);
24}
25
26session lifeline
27
28# Start a transaction, force a snapshot to be held
29step ll_start
30{
31    BEGIN TRANSACTION ISOLATION LEVEL REPEATABLE READ;
32    SELECT 1;
33}
34
35step ll_commit { COMMIT; }
36
37
38session pruner
39
40setup
41{
42    SET enable_seqscan = false;
43    SET enable_bitmapscan = false;
44}
45
46step pruner_create_temp
47{
48    CREATE TEMPORARY TABLE horizons_tst (data int unique) WITH (autovacuum_enabled = off);
49    INSERT INTO horizons_tst(data) VALUES(1),(2);
50}
51
52step pruner_create_perm
53{
54    CREATE TABLE horizons_tst (data int unique) WITH (autovacuum_enabled = off);
55    INSERT INTO horizons_tst(data) VALUES(1),(2);
56}
57
58# Temp tables cannot be dropped in the teardown, so just always do so
59# as part of the permutation
60step pruner_drop
61{
62    DROP TABLE horizons_tst;
63}
64
65step pruner_delete
66{
67    DELETE FROM horizons_tst;
68}
69
70step pruner_begin { BEGIN; }
71step pruner_commit { COMMIT; }
72
73step pruner_vacuum
74{
75    VACUUM horizons_tst;
76}
77
78# Show the heap fetches of an ordered index-only-scan (other plans
79# have been forbidden above) - that tells us how many non-killed leaf
80# entries there are.
81step pruner_query
82{
83    SELECT explain_json($$
84        EXPLAIN (FORMAT json, BUFFERS, ANALYZE)
85	  SELECT * FROM horizons_tst ORDER BY data;$$)->0->'Plan'->'Heap Fetches';
86}
87
88# Verify that the query plan still is an IOS
89step pruner_query_plan
90{
91    EXPLAIN (COSTS OFF) SELECT * FROM horizons_tst ORDER BY data;
92}

95# Show that with a permanent relation deleted rows cannot be pruned
96# away if there's a concurrent session still seeing the rows.
97permutation
98    pruner_create_perm
99    ll_start
100    pruner_query_plan
101    # Run query that could do pruning twice, first has chance to prune,
102    # second would not perform heap fetches if first query did.
103    pruner_query
104    pruner_query
105    pruner_delete
106    pruner_query
107    pruner_query
