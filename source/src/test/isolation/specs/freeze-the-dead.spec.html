1# Test for interactions of tuple freezing with dead, as well as recently-dead
2# tuples using multixacts via FOR KEY SHARE.
3setup
4{
5  CREATE TABLE tab_freeze (
6    id int PRIMARY KEY,
7    name char(3),
8    x int);
9  INSERT INTO tab_freeze VALUES (1, '111', 0);
10  INSERT INTO tab_freeze VALUES (3, '333', 0);
11}
12
13teardown
14{
15  DROP TABLE tab_freeze;
16}

18session s1
19step s1_begin		{ BEGIN; }
20step s1_update		{ UPDATE tab_freeze SET x = x + 1 WHERE id = 3; }
21step s1_commit		{ COMMIT; }
22step s1_selectone	{
23    BEGIN;
24    SET LOCAL enable_seqscan = false;
25    SET LOCAL enable_bitmapscan = false;
26    SELECT * FROM tab_freeze WHERE id = 3;
27    COMMIT;
28}
29step s1_selectall	{ SELECT * FROM tab_freeze ORDER BY name, id; }
30
31session s2
32step s2_begin		{ BEGIN; }
33step s2_key_share	{ SELECT id FROM tab_freeze WHERE id = 3 FOR KEY SHARE; }
34step s2_commit		{ COMMIT; }
35step s2_vacuum		{ VACUUM FREEZE tab_freeze; }

37session s3

39step s3_key_share	{ SELECT id FROM tab_freeze WHERE id = 3 FOR KEY SHARE; }

42# This permutation verifies that a previous bug
43#     https://postgr.es/m/E5711E62-8FDF-4DCA-A888-C200BF6B5742@amazon.com
44#     https://postgr.es/m/20171102112019.33wb7g5wp4zpjelu@alap3.anarazel.de
45# is not reintroduced. We used to make wrong pruning / freezing
46# decision for multixacts, which could lead to a) broken hot chains b)

48permutation s1_begin s2_begin s3_begin # start transactions
49   s1_update s2_key_share s3_key_share # have xmax be a multi with an updater, updater being oldest xid
50   s1_update # create additional row version that has multis
51   s1_commit s2_commit # commit both updater and share locker
52   s2_vacuum # due to bug in freezing logic, we used to *not* prune updated row, and then froze it
53   s1_selectone # if hot chain is broken, the row can't be found via index scan
54   s3_commit # commit remaining open xact
55   s2_vacuum # pruning / freezing in broken hot chains would unset xmax, reviving rows
