src/test/isolation/specs/merge-update.spec
1# MERGE UPDATE
2#
3# This test exercises atypical cases
4# 1. UPDATEs of PKs that change the join in the ON clause
5# 2. UPDATEs with WHEN conditions that would fail after concurrent update
6# 3. UPDATEs with extra ON conditions that would fail after concurrent update
7# 4. UPDATEs with duplicate source rows
8
9setup
10{
11  CREATE TABLE target (key int primary key, val text);
12  INSERT INTO target VALUES (1, 'setup1');
13
14  CREATE TABLE pa_target (key integer, val text)
15	  PARTITION BY LIST (key);
16  CREATE TABLE part1 (key integer, val text);
17  CREATE TABLE part2 (val text, key integer);
18  CREATE TABLE part3 (key integer, val text);
19
20  ALTER TABLE pa_target ATTACH PARTITION part1 FOR VALUES IN (1,4);
21  ALTER TABLE pa_target ATTACH PARTITION part2 FOR VALUES IN (2,5,6);
22  ALTER TABLE pa_target ATTACH PARTITION part3 DEFAULT;
23
24  INSERT INTO pa_target VALUES (1, 'initial');
25  INSERT INTO pa_target VALUES (2, 'initial');
26
27  CREATE FUNCTION explain_filter(text) RETURNS SETOF text
28  LANGUAGE plpgsql AS
29  $$
30  DECLARE
31    ln text;
32  BEGIN
33    FOR ln IN EXECUTE $1 LOOP
34      -- Ignore hash memory usage because it varies depending on the system
35      CONTINUE WHEN (ln ~ 'Memory Usage');
36      RETURN NEXT ln;
37    END LOOP;
38  END;
39  $$;
40}
41
42teardown
43{
44  DROP TABLE target;
45  DROP TABLE pa_target CASCADE;
46  DROP FUNCTION explain_filter;
47}

103step "merge2a"
104{
105  MERGE INTO target t
106  USING (SELECT 1 as key, 'merge2a' as val) s
107  ON s.key = t.key
108  WHEN NOT MATCHED THEN
109	INSERT VALUES (s.key, s.val)
110  WHEN MATCHED THEN
111	UPDATE set key = t.key + 1, val = t.val || ' updated by ' || s.val
112  WHEN NOT MATCHED BY SOURCE THEN
113	UPDATE set key = t.key + 1, val = t.val || ' source not matched by merge2a'
114  RETURNING merge_action(), old, new, t.*;

181# MERGE proceeds only if 'val' unchanged
182step "pa_merge2b_when"
183{
184  MERGE INTO pa_target t

186  ON s.key = t.key
187  WHEN NOT MATCHED THEN
188	INSERT VALUES (s.key, s.val)
189  WHEN MATCHED AND t.val like 'initial%' THEN
190	UPDATE set key = t.key + 1, val = t.val || ' updated by ' || s.val;
191}
192# Duplicate source row should fail

200}
