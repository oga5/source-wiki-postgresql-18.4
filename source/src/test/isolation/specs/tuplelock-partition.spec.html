src/test/isolation/specs/tuplelock-partition.spec
1# Test tuple locking on INSERT ON CONFLICT UPDATE on a partitioned table.
2
3setup
4{
5   DROP TABLE IF EXISTS parttab;
6   CREATE TABLE parttab (col1 text, key INTEGER PRIMARY KEY, col2 text) PARTITION BY LIST (key);
7   CREATE TABLE parttab1 (key INTEGER PRIMARY KEY, col1 text, col2 text);
8   CREATE TABLE parttab2 (key INTEGER PRIMARY KEY, col1 text, col2 text);
9   ALTER TABLE parttab ATTACH PARTITION parttab1 FOR VALUES IN (1);
10   ALTER TABLE parttab ATTACH PARTITION parttab2 FOR VALUES IN (2);
11   INSERT INTO parttab (key, col1, col2) VALUES (1, 'a', 'b');
12}
13
14teardown
15{
16   DROP TABLE parttab;
17}
18
19session s1
20step s1b               { BEGIN; }
21step s1update_nokey  { INSERT INTO parttab (key, col1, col2) VALUES (1, 'a', 'b') ON CONFLICT (key) DO UPDATE SET col1 = 'x', col2 = 'y'; }
22step s1update_key  { INSERT INTO parttab (key, col1, col2) VALUES (1, 'a', 'b') ON CONFLICT (key) DO UPDATE SET key=1; }
23step s1c               { COMMIT; }
24
25session s2
26step s2locktuple  { SELECT * FROM parttab FOR KEY SHARE; }
27
28# INSERT ON CONFLICT UPDATE, performs an UPDATE on non-key columns
29permutation s1b s1update_nokey s2locktuple s1c
30
31# INSERT ON CONFLICT UPDATE, performs an UPDATE on key column
32permutation s1b s1update_key s2locktuple s1c
