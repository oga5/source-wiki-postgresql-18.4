src/test/isolation/specs/vacuum-skip-locked.spec
1# Test for SKIP_LOCKED option of VACUUM and ANALYZE commands.
2#
3# This also verifies that log messages are not emitted for skipped relations
4# that were not specified in the VACUUM or ANALYZE command.
5
6setup
7{
8	CREATE TABLE parted (a INT) PARTITION BY LIST (a);
9	CREATE TABLE part1 PARTITION OF parted FOR VALUES IN (1);
10	ALTER TABLE part1 SET (autovacuum_enabled = false);
11	CREATE TABLE part2 PARTITION OF parted FOR VALUES IN (2);
12	ALTER TABLE part2 SET (autovacuum_enabled = false);
13}
14
15teardown
16{
17	DROP TABLE IF EXISTS parted;
18}
19
20session s1
21step lock_share
22{
23	BEGIN;
24	LOCK part1 IN SHARE MODE;
25}
26step lock_access_exclusive
27{
28	BEGIN;
29	LOCK part1 IN ACCESS EXCLUSIVE MODE;
30}
31step commit
32{
33	COMMIT;
34}
35
36session s2
37step vac_specified			{ VACUUM (SKIP_LOCKED) part1, part2; }
38step vac_all_parts			{ VACUUM (SKIP_LOCKED) parted; }
39step analyze_specified		{ ANALYZE (SKIP_LOCKED) part1, part2; }
40step analyze_all_parts		{ ANALYZE (SKIP_LOCKED) parted; }
41step vac_analyze_specified	{ VACUUM (ANALYZE, SKIP_LOCKED) part1, part2; }
42step vac_analyze_all_parts	{ VACUUM (ANALYZE, SKIP_LOCKED) parted; }
43step vac_full_specified		{ VACUUM (SKIP_LOCKED, FULL) part1, part2; }
44step vac_full_all_parts		{ VACUUM (SKIP_LOCKED, FULL) parted; }
45
46permutation lock_share vac_specified commit
47permutation lock_share vac_all_parts commit
48permutation lock_share analyze_specified commit
49permutation lock_share analyze_all_parts commit
50permutation lock_share vac_analyze_specified commit
51permutation lock_share vac_analyze_all_parts commit
52permutation lock_share vac_full_specified commit
53permutation lock_share vac_full_all_parts commit
54permutation lock_access_exclusive vac_specified commit
55permutation lock_access_exclusive vac_all_parts commit
56permutation lock_access_exclusive analyze_specified commit
57permutation lock_access_exclusive analyze_all_parts commit
58permutation lock_access_exclusive vac_analyze_specified commit
59permutation lock_access_exclusive vac_analyze_all_parts commit
60permutation lock_access_exclusive vac_full_specified commit
61permutation lock_access_exclusive vac_full_all_parts commit
