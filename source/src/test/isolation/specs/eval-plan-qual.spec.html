50 CREATE FUNCTION noisy_oper(p_comment text, p_a anynonarray, p_op text, p_b anynonarray)
51 RETURNS bool LANGUAGE plpgsql AS $$
52 DECLARE
53  r bool;
54  BEGIN
55  EXECUTE format('SELECT $1 %s $2', p_op) INTO r USING p_a, p_b;
56  RAISE NOTICE '%: % % % % %: %', p_comment, pg_typeof(p_a), p_a, p_op, pg_typeof(p_b), p_b, r;
57  RETURN r;
58  END;$$;
