src/test/isolation/specs/merge-match-recheck.spec
1# MERGE MATCHED recheck
2#
3# This test looks at what happens when we have complex
4# WHEN MATCHED AND conditions and a concurrent UPDATE causes a
5# recheck of the AND condition on the new row
6
7setup
8{
9  CREATE TABLE target (key int primary key, balance integer, status text, val text);
10  INSERT INTO target VALUES (1, 160, 's1', 'setup');
11
12  CREATE TABLE target_pa (key int, balance integer, status text, val text) PARTITION BY RANGE (balance);
13  CREATE TABLE target_pa1 PARTITION OF target_pa FOR VALUES FROM (0) TO (200);
14  CREATE TABLE target_pa2 PARTITION OF target_pa FOR VALUES FROM (200) TO (1000);
15  INSERT INTO target_pa VALUES (1, 160, 's1', 'setup');
16
17  CREATE TABLE target_tg (key int primary key, balance integer, status text, val text);
18  CREATE FUNCTION target_tg_trig_fn() RETURNS trigger LANGUAGE plpgsql AS
19  $$
20  BEGIN
21    IF tg_op = 'INSERT' THEN
22      RAISE NOTICE 'Insert: %', NEW;
23      RETURN NEW;
24    ELSIF tg_op = 'UPDATE' THEN
25      RAISE NOTICE 'Update: % -> %', OLD, NEW;
26      RETURN NEW;
27    ELSE
28      RAISE NOTICE 'Delete: %', OLD;
29      RETURN OLD;
30    END IF;
31  END
32  $$;
33  CREATE TRIGGER target_tg_trig BEFORE INSERT OR UPDATE OR DELETE ON target_tg
34    FOR EACH ROW EXECUTE FUNCTION target_tg_trig_fn();
35  INSERT INTO target_tg VALUES (1, 160, 's1', 'setup');
36}
37
38teardown
39{
40  DROP TABLE target;
41  DROP TABLE target_pa;
42  DROP TABLE target_tg;
43  DROP FUNCTION target_tg_trig_fn;
44}

51step "merge_status"
52{
53  MERGE INTO target t
54  USING (SELECT 1 as key) s
55  ON s.key = t.key
56  WHEN MATCHED AND status = 's1' THEN
57	UPDATE SET status = 's2', val = t.val || ' when1'
58  WHEN MATCHED AND status = 's2' THEN
59	UPDATE SET status = 's3', val = t.val || ' when2'
60  WHEN MATCHED AND status = 's3' THEN
61	UPDATE SET status = 's4', val = t.val || ' when3';
62}

100step "merge_bal_tg"
101{
102  WITH t AS (
103    MERGE INTO target_tg t
104    USING (SELECT 1 as key) s
105    ON s.key = t.key
106    WHEN MATCHED AND balance < 100 THEN
107      UPDATE SET balance = balance * 2, val = t.val || ' when1'
108    WHEN MATCHED AND balance < 200 THEN
109      UPDATE SET balance = balance * 4, val = t.val || ' when2'
110    WHEN MATCHED AND balance < 300 THEN
111      UPDATE SET balance = balance * 8, val = t.val || ' when3'
112    RETURNING t.*
113  )
114  SELECT * FROM t;
115}

117step "merge_delete"
118{
119  MERGE INTO target t
120  USING (SELECT 1 as key) s
121  ON s.key = t.key
122  WHEN MATCHED AND balance < 100 THEN
123	UPDATE SET balance = balance * 2, val = t.val || ' when1'
124  WHEN MATCHED AND balance < 200 THEN
125    DELETE;
126}

134  WHEN MATCHED AND balance < 200 THEN
135    DELETE;
136}
