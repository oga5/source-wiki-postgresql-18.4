1# MERGE DELETE
2#
3# This test looks at the interactions involving concurrent deletes
4# comparing the behavior of MERGE, DELETE and UPDATE
5
6setup
7{
8  CREATE TABLE target (key int primary key, val text);
9  INSERT INTO target VALUES (1, 'setup1');
10
11  CREATE TABLE target_pa (key int primary key, val text) PARTITION BY LIST (key);
12  CREATE TABLE target_pa1 PARTITION OF target_pa FOR VALUES IN (1);
13  CREATE TABLE target_pa2 PARTITION OF target_pa FOR VALUES IN (2);
14  INSERT INTO target_pa VALUES (1, 'setup1');
15
16  CREATE TABLE target_tg (key int primary key, val text);
17  CREATE FUNCTION target_tg_trig_fn() RETURNS trigger LANGUAGE plpgsql AS
18  $$
19  BEGIN
20    IF tg_op = 'INSERT' THEN
21      RAISE NOTICE 'Insert: %', NEW;
22      RETURN NEW;
23    ELSIF tg_op = 'UPDATE' THEN
24      RAISE NOTICE 'Update: % -> %', OLD, NEW;
25      RETURN NEW;
26    ELSE
27      RAISE NOTICE 'Delete: %', OLD;
28      RETURN OLD;
29    END IF;
30  END
31  $$;
32  CREATE TRIGGER target_tg_trig BEFORE INSERT OR UPDATE OR DELETE ON target_tg
33    FOR EACH ROW EXECUTE FUNCTION target_tg_trig_fn();
34  INSERT INTO target_tg VALUES (1, 'setup1');
35}
36
37teardown
38{
39  DROP TABLE target;
40  DROP TABLE target_pa;
41  DROP TABLE target_tg;
42  DROP FUNCTION target_tg_trig_fn;
43}
44
45session "s1"
46setup
47{
48  BEGIN ISOLATION LEVEL READ COMMITTED;
49}
50step "delete" { DELETE FROM target t WHERE t.key = 1; }
51step "delete_pa" { DELETE FROM target_pa t WHERE t.key = 1; }
52step "delete_tg" { DELETE FROM target_tg t WHERE t.key = 1; }
53step "c1" { COMMIT; }
54
55session "s2"
56setup
57{
58  BEGIN ISOLATION LEVEL READ COMMITTED;
59}
60step "update2" { UPDATE target t SET val = t.val || ' updated by update2' WHERE t.key = 1; }
61step "update2_pa" { UPDATE target_pa t SET val = t.val || ' updated by update2_pa' WHERE t.key = 1; }
62step "update2_tg" { UPDATE target_tg t SET val = t.val || ' updated by update2_tg' WHERE t.key = 1; }
63step "merge2" { MERGE INTO target t USING (SELECT 1 as key, 'merge2' as val) s ON s.key = t.key WHEN NOT MATCHED THEN INSERT VALUES (s.key, s.val) WHEN MATCHED THEN UPDATE set key = t.key + 1, val = t.val || ' updated by ' || s.val; }
64step "merge2_pa" { MERGE INTO target_pa t USING (SELECT 1 as key, 'merge2_pa' as val) s ON s.key = t.key WHEN NOT MATCHED THEN INSERT VALUES (s.key, s.val) WHEN MATCHED THEN UPDATE set key = t.key + 1, val = t.val || ' updated by ' || s.val; }
65step "merge2_tg" { MERGE INTO target_tg t USING (SELECT 1 as key, 'merge2_tg' as val) s ON s.key = t.key WHEN NOT MATCHED THEN INSERT VALUES (s.key, s.val) WHEN MATCHED THEN UPDATE set key = t.key + 1, val = t.val || ' updated by ' || s.val; }
66step "merge_delete2" { MERGE INTO target t USING (SELECT 1 as key, 'merge_delete2' as val) s ON s.key = t.key WHEN NOT MATCHED THEN INSERT VALUES (s.key, s.val) WHEN MATCHED THEN DELETE; }
67step "merge_delete2_tg" { MERGE INTO target_tg t USING (SELECT 1 as key, 'merge_delete2_tg' as val) s ON s.key = t.key WHEN NOT MATCHED THEN INSERT VALUES (s.key, s.val) WHEN MATCHED THEN DELETE; }
68step "select2" { SELECT * FROM target; }
69step "select2_pa" { SELECT * FROM target_pa; }
70step "select2_tg" { SELECT * FROM target_tg; }
71step "c2" { COMMIT; }
