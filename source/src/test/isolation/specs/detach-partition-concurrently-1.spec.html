src/test/isolation/specs/detach-partition-concurrently-1.spec
1# Test that detach partition concurrently makes the partition invisible at the
2# correct time.
3
4setup
5{
6  DROP TABLE IF EXISTS d_listp, d_listp1, d_listp2;
7  CREATE TABLE d_listp (a int) PARTITION BY LIST(a);
8  CREATE TABLE d_listp1 PARTITION OF d_listp FOR VALUES IN (1);
9  CREATE TABLE d_listp2 PARTITION OF d_listp FOR VALUES IN (2);
10  INSERT INTO d_listp VALUES (1),(2);
11}
12
13teardown {
14    DROP TABLE IF EXISTS d_listp, d_listp2, d_listp_foobar;
15}
16
17session s1
18step s1b		{ BEGIN; }
19step s1brr		{ BEGIN ISOLATION LEVEL REPEATABLE READ; }
20step s1s		{ SELECT * FROM d_listp; }
21step s1ins		{ INSERT INTO d_listp VALUES (1); }
22step s1ins2		{ INSERT INTO d_listp VALUES (2); }
23step s1prep		{ PREPARE f(int) AS INSERT INTO d_listp VALUES ($1); }
24step s1prep1	{ PREPARE f(int) AS INSERT INTO d_listp VALUES (1); }
25step s1prep2	{ PREPARE f(int) AS INSERT INTO d_listp VALUES (2); }
26step s1exec1	{ EXECUTE f(1); }
27step s1exec2	{ EXECUTE f(2); }
28step s1dealloc	{ DEALLOCATE f; }
29step s1c		{ COMMIT; }
30
31session s2
32step s2detach	{ ALTER TABLE d_listp DETACH PARTITION d_listp2 CONCURRENTLY; }
33step s2drop		{ DROP TABLE d_listp2; }
34
35session s3
36step s3s		{ SELECT * FROM d_listp; }
37step s3i		{ SELECT relpartbound IS NULL FROM pg_class where relname = 'd_listp2'; }
38step s3ins2		{ INSERT INTO d_listp VALUES (2); }
39
40# The transaction that detaches hangs until it sees any older transaction
41# terminate, as does anybody else.
42permutation s1b s1s s2detach s1s s1c s1s
43
44# relpartbound remains set until s1 commits
45# XXX this could be timing dependent :-(
46permutation s1b s1s s2detach s1s s3s s3i s1c s3i s2drop s1s
47
48# In read-committed mode, the partition disappears from view of concurrent
49# transactions immediately.  But if a write lock is held, then the detach
50# has to wait.
51permutation s1b   s1s          s2detach s1ins s1s s1c
52permutation s1b   s1s s1ins2 s2detach s1ins s1s s1c
53
54# In repeatable-read mode, the partition remains visible until commit even
55# if the to-be-detached partition is not locked for write.
56permutation s1brr s1s          s2detach s1ins s1s s1c
57permutation s1brr s1s          s2detach         s1s s1c
58
59# Another process trying to acquire a write lock will be blocked behind the
60# detacher
61permutation s1b s1ins2 s2detach s3ins2 s1c
62
63# a prepared query is not blocked
64permutation s1brr s1prep s1s s2detach s1s s1exec1 s3s s1dealloc s1c
65permutation s1brr s1prep s1exec2 s2detach s1s s1exec2 s3s s1c s1dealloc
66permutation s1brr s1prep s1s s2detach s1s s1exec2 s1c s1dealloc
67permutation s1brr s1prep s2detach s1s s1exec2 s1c s1dealloc
68permutation s1brr s1prep1 s2detach s1s s1exec2 s1c s1dealloc
69permutation s1brr s1prep2 s2detach s1s s1exec2 s1c s1dealloc
