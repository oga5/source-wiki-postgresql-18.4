1# Make sure that FKs referencing partitioned tables actually work.
2setup {
3  drop table if exists ppk, pfk, pfk1;
4  create table ppk (a int primary key) partition by list (a);
5  create table ppk1 partition of ppk for values in (1);
6  insert into ppk values (1);
7  create table pfk (a int references ppk) partition by list (a);
8  create table pfk1 partition of pfk for values in (1);
9}
10
11session s1
