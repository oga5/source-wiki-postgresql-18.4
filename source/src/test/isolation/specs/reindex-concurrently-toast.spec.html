1# REINDEX CONCURRENTLY with toast relations
2#
3# Ensure that concurrent operations work correctly when a REINDEX is performed
4# concurrently on toast relations.  Toast relation names are not deterministic,
5# so this abuses of allow_system_table_mods to change the names of toast
6# tables and its indexes so as they can be executed with REINDEX CONCURRENTLY,
7# which cannot be launched in a transaction context.
8
9# Create a table, with deterministic names for its toast relation and indexes.
10# Fortunately ALTER TABLE is transactional, making the renaming of toast
11# relations possible with allow_system_table_mods.
12setup
13{
14    CREATE TABLE reind_con_wide(id int primary key, data text);
15    INSERT INTO reind_con_wide
16      SELECT 1, repeat('1', 11) || string_agg(g.i::text || random()::text, '') FROM generate_series(1, 500) g(i);
17    INSERT INTO reind_con_wide
18      SELECT 2, repeat('2', 11) || string_agg(g.i::text || random()::text, '') FROM generate_series(1, 500) g(i);
19    SET allow_system_table_mods TO true;
20    DO $$DECLARE r record;
21      BEGIN
22      SELECT INTO r reltoastrelid::regclass::text AS table_name FROM pg_class
23        WHERE oid = 'reind_con_wide'::regclass;
24      EXECUTE 'ALTER TABLE ' || r.table_name || ' RENAME TO reind_con_toast;';
25      SELECT INTO r indexrelid::regclass::text AS index_name FROM pg_index
26        WHERE indrelid = (SELECT oid FROM pg_class where relname = 'reind_con_toast');
27      EXECUTE 'ALTER INDEX ' || r.index_name || ' RENAME TO reind_con_toast_idx;';
28    END$$;
29}
30
31teardown
32{
33    DROP TABLE IF EXISTS reind_con_wide;
34}
35
36session s1
37setup { BEGIN; }
38step lrex1 { lock TABLE reind_con_wide in ROW EXCLUSIVE MODE; }
39step lsha1 { lock TABLE reind_con_wide in SHARE MODE; }
40step lexc1 { lock TABLE reind_con_wide in EXCLUSIVE MODE; }
41step ins1 { INSERT INTO reind_con_wide SELECT 3, repeat('3', 11) || string_agg(g.i::text || random()::text, '') FROM generate_series(1, 500) g(i); }
42step upd1 { UPDATE reind_con_wide SET data = (SELECT repeat('4', 11) || string_agg(g.i::text || random()::text, '') FROM generate_series(1, 500) g(i)) WHERE id = 1; }
43step del1 { DELETE FROM reind_con_wide WHERE id = 2; }
44step dro1 { DROP TABLE reind_con_wide; }
45step end1 { COMMIT; }
46step rol1 { ROLLBACK; }
47
48session s2
49step retab2 { REINDEX TABLE CONCURRENTLY pg_toast.reind_con_toast; }
50step reind2 { REINDEX INDEX CONCURRENTLY pg_toast.reind_con_toast_idx; }
51step sel2 { SELECT id, substr(data, 1, 10) FROM reind_con_wide ORDER BY id; }
