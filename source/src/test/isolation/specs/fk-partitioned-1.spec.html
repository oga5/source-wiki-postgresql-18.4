1# Verify that cloning a foreign key constraint to a partition ensures
2# that referenced values exist, even if they're being concurrently
3# deleted.
4setup {
5drop table if exists ppk, pfk, pfk1;
6  create table ppk (a int primary key) partition by list (a);
7  create table ppk1 partition of ppk for values in (1);
8  insert into ppk values (1);
9  create table pfk (a int references ppk) partition by list (a);
10  create table pfk1 (a int not null);
11  insert into pfk1 values (1);
12}
13
14session s1
15step s1b	{	begin; }
16step s1d	{	delete from ppk1 where a = 1; }
17step s1c	{	commit; }
18
19session s2
20step s2b	{	begin; }
21step s2a	{	alter table pfk attach partition pfk1 for values in (1); }

23
24teardown	{	drop table ppk, pfk, pfk1; }
