src/test/isolation/specs/partial-index.spec
1# Partial Index test
2#
3# Make sure that an update which moves a row out of a partial index
4# is handled correctly.  In early versions, an attempt at optimization
5# broke this behavior, allowing anomalies.
6#
7# Any overlap between the transactions must cause a serialization failure.
8
9setup
10{
11 create table test_t (id integer, val1 text, val2 integer);
12 create index test_idx on test_t(id) where val2 = 1;
13 insert into test_t (select generate_series(0, 10000), 'a', 2);
14 insert into test_t (select generate_series(0, 10), 'a', 1);
15}
16
17teardown
18{
19 DROP TABLE test_t;
20}
21
22session s1
23setup		{ BEGIN ISOLATION LEVEL SERIALIZABLE; }
24step rxy1	{ select * from test_t where val2 = 1; }
25step wx1	{ update test_t set val2 = 2 where val2 = 1 and id = 10; }
26step c1		{ COMMIT; }
27
28session s2
29setup		{ BEGIN ISOLATION LEVEL SERIALIZABLE; }
30step wy2	{ update test_t set val2 = 2 where val2 = 1 and id = 9; }
31step rxy2	{ select * from test_t where val2 = 1; }
32step c2		{ COMMIT; }
