1# RI Trigger test
2#
3# Test trigger-based referential integrity enforcement.
4#
5# Any overlap between the transactions must cause a serialization failure.
6
7setup
8{
9 CREATE TABLE parent (parent_id SERIAL NOT NULL PRIMARY KEY);
10 CREATE TABLE child (child_id SERIAL NOT NULL PRIMARY KEY, parent_id INTEGER NOT NULL);
11 CREATE FUNCTION ri_parent() RETURNS TRIGGER LANGUAGE PLPGSQL AS $body$
12 BEGIN
13  PERFORM TRUE FROM child WHERE parent_id = OLD.parent_id;
14  IF FOUND THEN
15    RAISE SQLSTATE '23503' USING MESSAGE = 'child row exists';
16  END IF;
17  IF TG_OP = 'DELETE' THEN
18    RETURN OLD;
19  END IF;
20  RETURN NEW;
21 END;
22 $body$;
23 CREATE TRIGGER ri_parent BEFORE UPDATE OR DELETE ON parent FOR EACH ROW EXECUTE PROCEDURE ri_parent();
24 CREATE FUNCTION ri_child() RETURNS TRIGGER LANGUAGE PLPGSQL AS $body$
25 BEGIN
26  PERFORM TRUE FROM parent WHERE parent_id = NEW.parent_id;
27  IF NOT FOUND THEN
28    RAISE SQLSTATE '23503' USING MESSAGE = 'parent row missing';
29  END IF;
30  RETURN NEW;
31 END;
32 $body$;
33 CREATE TRIGGER ri_child BEFORE INSERT OR UPDATE ON child FOR EACH ROW EXECUTE PROCEDURE ri_child();
34 INSERT INTO parent VALUES(0);
35}
36
37teardown
38{
39 DROP TABLE parent, child;
40 DROP FUNCTION ri_parent();
41 DROP FUNCTION ri_child();
42}
43
