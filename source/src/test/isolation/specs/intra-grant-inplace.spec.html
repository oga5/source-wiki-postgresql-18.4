1# GRANT's lock is the catalog tuple xmax.  GRANT doesn't acquire a heavyweight
2# lock on the object undergoing an ACL change.  Inplace updates, such as
3# relhasindex=true, need special code to cope.
4
5setup
6{
7	CREATE TABLE intra_grant_inplace (c int);
8}
9
10teardown
11{
12	DROP TABLE IF EXISTS intra_grant_inplace;
13}
14
15# heap_update()
16session s1
17setup	{ SET deadlock_timeout = '100s'; }
18step b1	{ BEGIN; }
19step grant1	{
20	GRANT SELECT ON intra_grant_inplace TO PUBLIC;
21}
22step drop1	{
23	DELETE FROM pg_class WHERE relname = 'intra_grant_inplace';
24}
25step c1	{ COMMIT; }
26
27# inplace update
28session s2
29setup	{ SET deadlock_timeout = '10ms'; }
30step read2	{
31	SELECT relhasindex FROM pg_class
32	WHERE oid = 'intra_grant_inplace'::regclass;
33}
34step b2		{ BEGIN; }
35step addk2	{ ALTER TABLE intra_grant_inplace ADD PRIMARY KEY (c); }
36step sfnku2	{
37	SELECT relhasindex FROM pg_class
38	WHERE oid = 'intra_grant_inplace'::regclass FOR NO KEY UPDATE;
39}
40step c2		{ COMMIT; }
41
42# rowmarks
43session s3
44step b3		{ BEGIN ISOLATION LEVEL READ COMMITTED; }
45step sfnku3	{
46	SELECT relhasindex FROM pg_class
47	WHERE oid = 'intra_grant_inplace'::regclass FOR NO KEY UPDATE;
48}
49step sfu3	{
50	SELECT relhasindex FROM pg_class
51	WHERE oid = 'intra_grant_inplace'::regclass FOR UPDATE;
52}
53step r3	{ ROLLBACK; }
54
55# Additional heap_update()
56session s4
57# swallow error message to keep any OID value out of expected output
58step revoke4	{
59	DO $$
60	BEGIN
61		REVOKE SELECT ON intra_grant_inplace FROM PUBLIC;
62	EXCEPTION WHEN others THEN
63		RAISE WARNING 'got: %', regexp_replace(sqlerrm, '[0-9]+', 'REDACTED');
64	END
65	$$;
66}

68# Additional rowmarks
69session s5
70setup	{ BEGIN; }
71step keyshr5	{
72	SELECT relhasindex FROM pg_class
73	WHERE oid = 'intra_grant_inplace'::regclass FOR KEY SHARE;
74}
75teardown	{ ROLLBACK; }
