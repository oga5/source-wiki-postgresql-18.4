1# Test SKIP LOCKED with tuple locks.
2
3setup
4{
5  CREATE TABLE queue (
6	id	int		PRIMARY KEY,
7	data			text	NOT NULL,
8	status			text	NOT NULL
9  );
10  INSERT INTO queue VALUES (1, 'foo', 'NEW'), (2, 'bar', 'NEW');
11}
12
13teardown
14{
15  DROP TABLE queue;
16}
17
18session s1
19setup		{ BEGIN; }
20step s1a	{ SELECT * FROM queue ORDER BY id FOR UPDATE LIMIT 1; }
21step s1b	{ COMMIT; }
22
23session s2
24setup		{ BEGIN; }
25step s2a	{ SELECT * FROM queue ORDER BY id FOR UPDATE LIMIT 1; }
26step s2b	{ COMMIT; }
27
28session s3
29setup		{ BEGIN; }
30step s3a	{ SELECT * FROM queue ORDER BY id FOR UPDATE SKIP LOCKED LIMIT 1; }
31step s3b	{ COMMIT; }
32
33# s3 skips to the second record because it can't obtain the tuple lock
34# (s2 holds the tuple lock because it is next in line to obtain the
35# row lock, and s1 holds the row lock)
36permutation s1a s2a s3a s1b s2b s3b
