1# Test for page level predicate locking in gist
2#
3# Test to verify serialization failures and to check reduced false positives
4#
5# To verify serialization failures, queries and permutations are written in such
6# a way that an index scan  (from one transaction) and an index insert (from
7# another transaction) will try to access the same part (sub-tree) of the index
8# whereas to check reduced false positives, they will try to access different
9# parts (sub-tree) of the index.
10
11setup
12{
13  create table gist_point_tbl(id int4, p point);
14  create index gist_pointidx on gist_point_tbl using gist(p);
15  insert into gist_point_tbl (id, p)
16  select g, point(g*10, g*10) from generate_series(1, 1000) g;
17}
18
19teardown
20{
21  drop table gist_point_tbl;
22}
23
24session s1
25setup
26{
27  begin isolation level serializable;
28  set enable_seqscan=off;
29  set enable_bitmapscan=off;
30  set enable_indexonlyscan=on;
31}
32
33step rxy1	{ select sum(p[0]) from gist_point_tbl where p << point(2500, 2500); }
34step wx1	{ insert into gist_point_tbl (id, p)
35			  select g, point(g*500, g*500) from generate_series(15, 20) g; }
36step rxy3	{ select sum(p[0]) from gist_point_tbl where p >> point(6000,6000); }
37step wx3	{ insert into gist_point_tbl (id, p)
38			  select g, point(g*500, g*500) from generate_series(12, 18) g; }
39step c1		{ commit; }
40
41
42session s2
43setup
44{
45  begin isolation level serializable;
46  set enable_seqscan=off;
47  set enable_bitmapscan=off;
48  set enable_indexonlyscan=on;
49}
50
51step rxy2	{ select sum(p[0]) from gist_point_tbl where p >> point(7500,7500); }
52step wy2	{ insert into gist_point_tbl (id, p)
53			  select g, point(g*500, g*500) from generate_series(1, 5) g; }
54step rxy4	{ select sum(p[0]) from gist_point_tbl where p << point(1000,1000); }
55step wy4	{ insert into gist_point_tbl (id, p)
56			  select g, point(g*50, g*50) from generate_series(1, 20) g; }
57step c2		{ commit; }
