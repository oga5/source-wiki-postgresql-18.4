14=head1 NAME
15
16LdapServer - class for an LDAP server for testing pg_hba.conf authentication
17
18=head1 SYNOPSIS
19
20  use LdapServer;
21
22  # have we found openldap binaries suitable for setting up a server?
23  my $ldap_binaries_found = $LdapServer::setup;
24
25  # create a server with the given root password and auth type
26  # (users or anonymous)
27  my $server = LdapServer->new($root_password, $auth_type);
28
29  # Add the contents of an LDIF file to the server
30  $server->ldapadd_file ($path_to_ldif_data);
31
32  # set the Ldap password for a user
33  $server->ldapsetpw($user, $password);
34
35  # get details of some settings for the server
36  my @properties = $server->prop($propname1, $propname2, ...);
37
38=head1 DESCRIPTION
39
40  LdapServer tests in its INIT phase for the presence of suitable openldap
41  binaries. Its constructor method sets up and runs an LDAP server, and any
42  servers that are set up are terminated during its END phase.
43
44=cut

63INIT
64{
65	# Find the OpenLDAP server binary and directory containing schema
66	# definition files.  On success, $setup is set to 1. On failure,
67	# it's set to 0, and an error message is set in $setup_error.
68	$setup = 1;
69	if ($^O eq 'darwin')
70	{
71		if (-d '/opt/homebrew/opt/openldap')
72		{
73			# typical paths for Homebrew on ARM
74			$slapd = '/opt/homebrew/opt/openldap/libexec/slapd';
75			$ldap_schema_dir = '/opt/homebrew/etc/openldap/schema';
76		}
77		elsif (-d '/usr/local/opt/openldap')
78		{
79			# typical paths for Homebrew on Intel
80			$slapd = '/usr/local/opt/openldap/libexec/slapd';
81			$ldap_schema_dir = '/usr/local/etc/openldap/schema';
82		}
83		elsif (-d '/opt/local/etc/openldap')
84		{
85			# typical paths for MacPorts
86			$slapd = '/opt/local/libexec/slapd';
87			$ldap_schema_dir = '/opt/local/etc/openldap/schema';
88		}
89		else
90		{
91			$setup_error = "OpenLDAP server installation not found";
92			$setup = 0;
93		}
94	}
95	elsif ($^O eq 'linux')
96	{
97		if (-d '/etc/ldap/schema')
98		{
99			$slapd = '/usr/sbin/slapd';
100			$ldap_schema_dir = '/etc/ldap/schema';
101		}
102		elsif (-d '/etc/openldap/schema')
103		{
104			$slapd = '/usr/sbin/slapd';
105			$ldap_schema_dir = '/etc/openldap/schema';
106		}
107		else
108		{
109			$setup_error = "OpenLDAP server installation not found";
110			$setup = 0;
111		}
112	}
113	elsif ($^O eq 'freebsd')
114	{
115		if (-d '/usr/local/etc/openldap/schema')
116		{
117			$slapd = '/usr/local/libexec/slapd';
118			$ldap_schema_dir = '/usr/local/etc/openldap/schema';
119		}
120		else
121		{
122			$setup_error = "OpenLDAP server installation not found";
123			$setup = 0;
124		}
125	}
126	elsif ($^O eq 'openbsd')
127	{
128		if (-d '/usr/local/share/examples/openldap/schema')
129		{
130			$slapd = '/usr/local/libexec/slapd';
131			$ldap_schema_dir = '/usr/local/share/examples/openldap/schema';
132		}
133		else
134		{
135			$setup_error = "OpenLDAP server installation not found";
136			$setup = 0;
137		}
138	}
139	else
140	{
141		$setup_error = "ldap tests not supported on $^O";
142		$setup = 0;
143	}
144}
145
146END
147{
148	# take care not to change the script's exit value
149	my $exit_code = $?;
150
151	foreach my $server (@servers)
152	{
153		next unless -f $server->{pidfile};
154		my $pid = slurp_file($server->{pidfile});
155		chomp $pid;
156		kill 'INT', $pid;
157	}
158
159	$? = $exit_code;
160}

164=head1 METHODS
165
166=over
167
168=item LdapServer->new($rootpw, $auth_type)
169
170Create a new LDAP server.

172The rootpw can be used when authenticating with the ldapbindpasswd option.

180sub new
181{
182	die "no suitable binaries found" unless $setup;
183
184	my $class = shift;
185	my $rootpw = shift;
186	my $authtype = shift;    # 'users' or 'anonymous'
187	my $testname = basename((caller)[1], '.pl');
188	my $self = {};
189
190	my $test_temp = PostgreSQL::Test::Utils::tempdir("ldap-$testname");
191
192	my $ldap_datadir = "$test_temp/openldap-data";
193	my $slapd_certs = "$test_temp/slapd-certs";
194	my $slapd_pidfile = "$test_temp/slapd.pid";
195	my $slapd_conf = "$test_temp/slapd.conf";
196	my $slapd_logfile =
197	  "${PostgreSQL::Test::Utils::log_path}/slapd-$testname.log";
198	my $ldap_server = 'localhost';
199	my $ldap_port = PostgreSQL::Test::Cluster::get_free_port();
200	my $ldaps_port = PostgreSQL::Test::Cluster::get_free_port();
201	my $ldap_url = "ldap://$ldap_server:$ldap_port";
202	my $ldaps_url = "ldaps://$ldap_server:$ldaps_port";
203	my $ldap_basedn = 'dc=example,dc=net';
204	my $ldap_rootdn = 'cn=Manager,dc=example,dc=net';
205	my $ldap_rootpw = $rootpw;
206	my $ldap_pwfile = "$test_temp/ldappassword";
207
208	(my $conf = <<"EOC") =~ s/^\t\t//gm;
209		include $ldap_schema_dir/core.schema
210		include $ldap_schema_dir/cosine.schema
211		include $ldap_schema_dir/nis.schema
212		include $ldap_schema_dir/inetorgperson.schema
213
214		pidfile $slapd_pidfile
215		logfile $slapd_logfile
216
217		access to *
218		        by * read
219		        by $authtype auth
220
221		database ldif
222		directory $ldap_datadir
223
224		TLSCACertificateFile $slapd_certs/ca.crt
225		TLSCertificateFile $slapd_certs/server.crt
226		TLSCertificateKeyFile $slapd_certs/server.key
227
228		suffix "dc=example,dc=net"
229		rootdn "$ldap_rootdn"
230		rootpw "$ldap_rootpw"
231EOC
232	append_to_file($slapd_conf, $conf);
233
234	mkdir $ldap_datadir or die "making $ldap_datadir: $!";
235	mkdir $slapd_certs or die "making $slapd_certs: $!";
236
237	my $certdir = dirname(__FILE__) . "/../ssl/ssl";
238
239	copy "$certdir/server_ca.crt", "$slapd_certs/ca.crt"
240	  || die "copying ca.crt: $!";
241	# check we actually have the file, as copy() sometimes gives a false success
242	-f "$slapd_certs/ca.crt" || die "copying ca.crt (error unknown)";
243	copy "$certdir/server-cn-only.crt", "$slapd_certs/server.crt"
244	  || die "copying server.crt: $!";
245	copy "$certdir/server-cn-only.key", "$slapd_certs/server.key"
246	  || die "copying server.key: $!";
247
248	append_to_file($ldap_pwfile, $ldap_rootpw);
249	chmod 0600, $ldap_pwfile or die "chmod on $ldap_pwfile";
250
251	# -s0 prevents log messages ending up in syslog
252	system_or_bail $slapd, '-f', $slapd_conf, '-s0', '-h',
253	  "$ldap_url $ldaps_url";
254
255	# wait until slapd accepts requests
256	my $retries = 0;
257	while (1)
258	{
259		last
260		  if (
261			system_log(
262				"ldapsearch", "-sbase",
263				"-H", $ldap_url,
264				"-b", $ldap_basedn,
265				"-D", $ldap_rootdn,
266				"-y", $ldap_pwfile,
267				"-n", "'objectclass=*'") == 0);
268		die "cannot connect to slapd" if ++$retries >= 300;
269		note "waiting for slapd to accept requests...";
270		Time::HiRes::usleep(1000000);
271	}
272
273	$self->{pidfile} = $slapd_pidfile;
274	$self->{pwfile} = $ldap_pwfile;
275	$self->{url} = $ldap_url;
276	$self->{s_url} = $ldaps_url;
277	$self->{server} = $ldap_server;
278	$self->{port} = $ldap_port;
279	$self->{s_port} = $ldaps_port;
280	$self->{basedn} = $ldap_basedn;
281	$self->{rootdn} = $ldap_rootdn;
282
283	bless $self, $class;
284	push @servers, $self;
285	return $self;
286}

288# private routine to set up the environment for methods below
289sub _ldapenv
290{
291	my $self = shift;
292	my %env = %ENV;
293	$env{'LDAPURI'} = $self->{url};
294	$env{'LDAPBINDDN'} = $self->{rootdn};
295	return %env;
296}

311sub ldapadd_file
312{
313	my $self = shift;
314	my $file = shift;
315
316	local %ENV = $self->_ldapenv;
317
318	system_or_bail 'ldapadd', '-x', '-y', $self->{pwfile}, '-f', $file;
319}

333sub ldapsetpw
334{
335	my $self = shift;
336	my $user = shift;
337	my $password = shift;
338
339	local %ENV = $self->_ldapenv;
340
341	system_or_bail 'ldappasswd', '-x', '-y', $self->{pwfile}, '-s', $password,
342	  $user;
343}

358sub prop
