1-- Test non-ASCII error messages
2--
3-- This test case would fail if the database encoding is EUC_CN, EUC_JP,
4-- EUC_KR, or EUC_TW, for lack of any equivalent to U+00A0 (no-break space) in
5-- those encodings.  However, testing with plain ASCII data would be rather
6-- useless, so we must live with that.
7SELECT getdatabaseencoding() IN ('EUC_CN', 'EUC_JP', 'EUC_KR', 'EUC_TW')
8  AS skip_test \gset
9\if :skip_test
10\quit
11\endif
12SET client_encoding TO UTF8;
13create or replace function error_with_nbsp() returns void language plperl as $$
14  elog(ERROR, "this message contains a no-break space");
15$$;
16select error_with_nbsp();
17ERROR:  this message contains a no-break space at line 2.
