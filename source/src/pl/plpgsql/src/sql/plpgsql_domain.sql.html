5CREATE DOMAIN booltrue AS bool CHECK (VALUE IS TRUE OR VALUE IS NULL);
6
7CREATE FUNCTION test_argresult_booltrue(x booltrue, y bool) RETURNS booltrue AS $$
8begin
9return y;
10end
11$$ LANGUAGE plpgsql;
12
13SELECT * FROM test_argresult_booltrue(true, true);
14SELECT * FROM test_argresult_booltrue(false, true);
15SELECT * FROM test_argresult_booltrue(true, false);
16
17CREATE FUNCTION test_assign_booltrue(x bool, y bool) RETURNS booltrue AS $$
18declare v booltrue := x;
19begin
20v := y;
21return v;
22end
23$$ LANGUAGE plpgsql;
24
25SELECT * FROM test_assign_booltrue(true, true);
26SELECT * FROM test_assign_booltrue(false, true);
27SELECT * FROM test_assign_booltrue(true, false);

85CREATE DOMAIN ordered_pair_domain AS integer[] CHECK (array_length(VALUE,1)=2 AND VALUE[1] < VALUE[2]);
86
87CREATE FUNCTION test_argresult_array_domain(x ordered_pair_domain)
88  RETURNS ordered_pair_domain AS $$
89begin
90return x;
91end
92$$ LANGUAGE plpgsql;
93
94SELECT * FROM test_argresult_array_domain(ARRAY[0, 100]::ordered_pair_domain);
95SELECT * FROM test_argresult_array_domain(NULL::ordered_pair_domain);
96
97CREATE FUNCTION test_argresult_array_domain_check_violation()
98  RETURNS ordered_pair_domain AS $$
99begin
100return array[2,1];
101end
102$$ LANGUAGE plpgsql;
103
104SELECT * FROM test_argresult_array_domain_check_violation();

216--
217-- Domains of composite
218--
219
220CREATE TYPE named_pair AS (
221    i integer,
222    j integer
223);
224
225CREATE DOMAIN ordered_named_pair AS named_pair CHECK((VALUE).i <= (VALUE).j);
226
227CREATE FUNCTION read_ordered_named_pair(p ordered_named_pair) RETURNS integer AS $$
228begin
229return p.i + p.j;
230end
231$$ LANGUAGE plpgsql;
232
233SELECT read_ordered_named_pair(row(1, 2));
234SELECT read_ordered_named_pair(row(2, 1));  -- fail
235
236CREATE FUNCTION build_ordered_named_pair(i int, j int) RETURNS ordered_named_pair AS $$
237begin
238return row(i, j);
239end
240$$ LANGUAGE plpgsql;
241
242SELECT build_ordered_named_pair(1,2);
243SELECT build_ordered_named_pair(2,1);  -- fail
244
245CREATE FUNCTION test_assign_ordered_named_pair(x int, y int, z int)
246  RETURNS ordered_named_pair AS $$
247declare v ordered_named_pair := row(x, y);
248begin
249v.j := z;
250return v;
251end
252$$ LANGUAGE plpgsql;
253
254SELECT * FROM test_assign_ordered_named_pair(1,2,3);
255SELECT * FROM test_assign_ordered_named_pair(1,2,0);
256SELECT * FROM test_assign_ordered_named_pair(2,1,3);
