src/pl/plpgsql/src/sql/plpgsql_copy.sql
1-- directory paths are passed to us in environment variables
2\getenv abs_srcdir PG_ABS_SRCDIR
3\getenv abs_builddir PG_ABS_BUILDDIR
4
5-- set up file names to use
6\set srcfilename :abs_srcdir '/data/copy1.data'
7\set destfilename :abs_builddir '/results/copy1.data'
8
9CREATE TABLE copy1 (a int, b float);
10
11-- COPY TO/FROM not authorized from client.
12DO LANGUAGE plpgsql $$
13BEGIN
14  COPY copy1 TO stdout;
15END;
16$$;
17DO LANGUAGE plpgsql $$
18BEGIN
19  COPY copy1 FROM stdin;
20END;
21$$;
22DO LANGUAGE plpgsql $$
23BEGIN
24  EXECUTE 'COPY copy1 TO stdout';
25END;
26$$;
27DO LANGUAGE plpgsql $$
28BEGIN
29  EXECUTE 'COPY copy1 FROM stdin';
30END;
31$$;
32
33-- Valid cases
34-- COPY FROM
35\set dobody 'BEGIN COPY copy1 FROM ' :'srcfilename' '; END'
36DO LANGUAGE plpgsql :'dobody';
37SELECT * FROM copy1 ORDER BY 1;
38TRUNCATE copy1;
39\set cmd 'COPY copy1 FROM ' :'srcfilename'
40\set dobody 'BEGIN EXECUTE ' :'cmd' '; END'
41DO LANGUAGE plpgsql :'dobody';
42SELECT * FROM copy1 ORDER BY 1;
43
44-- COPY TO
45-- Copy the data externally once, then process it back to the table.
46\set dobody 'BEGIN COPY copy1 TO ' :'destfilename' '; END'
47DO LANGUAGE plpgsql :'dobody';
48TRUNCATE copy1;
49\set dobody 'BEGIN COPY copy1 FROM ' :'destfilename' '; END'
50DO LANGUAGE plpgsql :'dobody';
51
52\set cmd 'COPY copy1 FROM ' :'destfilename'
53\set dobody 'BEGIN EXECUTE ' :'cmd' '; END'
54DO LANGUAGE plpgsql :'dobody';
55
56SELECT * FROM copy1 ORDER BY 1;
57
58DROP TABLE copy1;
