95/* ----------
96 * plpgsql_compile		Make an execution tree for a PL/pgSQL function.
97 *
98 * If forValidator is true, we're only compiling for validation purposes,
99 * and so some checks are skipped.
100 *
101 * Note: it's important for this to fall through quickly if the function
102 * has already been compiled.
103 * ----------
104 */
105PLpgSQL_function *
106plpgsql_compile(FunctionCallInfo fcinfo, bool forValidator)
107{
108	PLpgSQL_function *function;
109
110	/*
111	 * funccache.c manages re-use of existing PLpgSQL_function caches.
112	 *
113	 * In PL/pgSQL we use fn_extra directly as the pointer to the long-lived
114	 * function cache entry; we have no need for any query-lifespan cache.
115	 * Also, we don't need to make the cache key depend on composite result
116	 * type (at least for now).
117	 */
118	function = (PLpgSQL_function *)
119		cached_function_compile(fcinfo,
120								fcinfo->flinfo->fn_extra,
121								plpgsql_compile_callback,
122								plpgsql_delete_callback,
123								sizeof(PLpgSQL_function),
124								false,
125								forValidator);
126
127	/*
128	 * Save pointer in FmgrInfo to avoid search on subsequent calls
129	 */
130	fcinfo->flinfo->fn_extra = function;
131
132	/*
133	 * Finally return the compiled function
134	 */
135	return function;
136}

144/*
145 * This is the slow part of plpgsql_compile().
146 *
147 * The passed-in "cfunc" struct is expected to be zeroes, except
148 * for the CachedFunction fields, which we don't touch here.
149 *
150 * While compiling a function, the CurrentMemoryContext is the
151 * per-function memory context of the function we are compiling. That
152 * means a palloc() will allocate storage with the same lifetime as
153 * the function itself.
154 *
155 * Because palloc()'d storage will not be immediately freed, temporary
156 * allocations should either be performed in a short-lived memory
157 * context or explicitly pfree'd. Since not all backend functions are
158 * careful about pfree'ing their allocations, it is also wise to
159 * switch into a short-term context before calling into the
160 * backend. An appropriate context for performing short-term
161 * allocations is the plpgsql_compile_tmp_cxt.
162 *
163 * NB: this code is not re-entrant.  We assume that nothing we do here could
164 * result in the invocation of another plpgsql function.
165 */
166static void
167plpgsql_compile_callback(FunctionCallInfo fcinfo,
168						 HeapTuple procTup,
169						 const CachedFunctionHashKey *hashkey,
170						 CachedFunction *cfunc,
171						 bool forValidator)
172{
173	PLpgSQL_function *function = (PLpgSQL_function *) cfunc;
174	Form_pg_proc procStruct = (Form_pg_proc) GETSTRUCT(procTup);
175	bool		is_dml_trigger = CALLED_AS_TRIGGER(fcinfo);
176	bool		is_event_trigger = CALLED_AS_EVENT_TRIGGER(fcinfo);
177	yyscan_t	scanner;
178	Datum		prosrcdatum;
179	char	   *proc_source;
180	HeapTuple	typeTup;
181	Form_pg_type typeStruct;
182	PLpgSQL_variable *var;
183	PLpgSQL_rec *rec;
184	int			i;
185	struct compile_error_callback_arg cbarg;
186	ErrorContextCallback plerrcontext;
187	int			parse_rc;
188	Oid			rettypeid;
189	int			numargs;
190	int			num_in_args = 0;
191	int			num_out_args = 0;
192	Oid		   *argtypes;
193	char	  **argnames;
194	char	   *argmodes;
195	int		   *in_arg_varnos = NULL;
196	PLpgSQL_variable **out_arg_variables;
197	MemoryContext func_cxt;
198
199	/*
200	 * Setup the scanner input and error info.
201	 */
202	prosrcdatum = SysCacheGetAttrNotNull(PROCOID, procTup, Anum_pg_proc_prosrc);
203	proc_source = TextDatumGetCString(prosrcdatum);
204	scanner = plpgsql_scanner_init(proc_source);
205
206	plpgsql_error_funcname = pstrdup(NameStr(procStruct->proname));
207
208	/*
209	 * Setup error traceback support for ereport()
210	 */
211	cbarg.proc_source = forValidator ? proc_source : NULL;
212	cbarg.yyscanner = scanner;
213	plerrcontext.callback = plpgsql_compile_error_callback;
214	plerrcontext.arg = &cbarg;
215	plerrcontext.previous = error_context_stack;
216	error_context_stack = &plerrcontext;
217
218	/*
219	 * Do extra syntax checks when validating the function definition. We skip
220	 * this when actually compiling functions for execution, for performance
221	 * reasons.
222	 */
223	plpgsql_check_syntax = forValidator;
224	plpgsql_curr_compile = function;
225
226	/*
227	 * All the permanent output of compilation (e.g. parse tree) is kept in a
228	 * per-function memory context, so it can be reclaimed easily.
229	 *
230	 * While the func_cxt needs to be long-lived, we initially make it a child
231	 * of the assumed-short-lived caller's context, and reparent it under
232	 * CacheMemoryContext only upon success.  This arrangement avoids memory
233	 * leakage during compilation of a faulty function.
234	 */
235	func_cxt = AllocSetContextCreate(CurrentMemoryContext,
236									 "PL/pgSQL function",
237									 ALLOCSET_DEFAULT_SIZES);
238	plpgsql_compile_tmp_cxt = MemoryContextSwitchTo(func_cxt);
239
240	function->fn_signature = format_procedure(fcinfo->flinfo->fn_oid);
241	MemoryContextSetIdentifier(func_cxt, function->fn_signature);
242	function->fn_oid = fcinfo->flinfo->fn_oid;
243	function->fn_input_collation = fcinfo->fncollation;
244	function->fn_cxt = func_cxt;
245	function->out_param_varno = -1; /* set up for no OUT param */
246	function->resolve_option = plpgsql_variable_conflict;
247	function->print_strict_params = plpgsql_print_strict_params;
248	/* only promote extra warnings and errors at CREATE FUNCTION time */
249	function->extra_warnings = forValidator ? plpgsql_extra_warnings : 0;
250	function->extra_errors = forValidator ? plpgsql_extra_errors : 0;
251
252	if (is_dml_trigger)
253		function->fn_is_trigger = PLPGSQL_DML_TRIGGER;
254	else if (is_event_trigger)
255		function->fn_is_trigger = PLPGSQL_EVENT_TRIGGER;
256	else
257		function->fn_is_trigger = PLPGSQL_NOT_TRIGGER;
258
259	function->fn_prokind = procStruct->prokind;
260
261	function->nstatements = 0;
262	function->requires_procedure_resowner = false;
263	function->has_exception_block = false;
264
265	/*
266	 * Initialize the compiler, particularly the namespace stack.  The
267	 * outermost namespace contains function parameters and other special
268	 * variables (such as FOUND), and is named after the function itself.
269	 */
270	plpgsql_ns_init();
271	plpgsql_ns_push(NameStr(procStruct->proname), PLPGSQL_LABEL_BLOCK);
272	plpgsql_DumpExecTree = false;
273	plpgsql_start_datums();
274
275	switch (function->fn_is_trigger)
276	{
277		case PLPGSQL_NOT_TRIGGER:
278
279			/*
280			 * Fetch info about the procedure's parameters. Allocations aren't
281			 * needed permanently, so make them in tmp cxt.
282			 *
283			 * We also need to resolve any polymorphic input or output
284			 * argument types.  In validation mode we won't be able to, so we
285			 * arbitrarily assume we are dealing with integers.
286			 */
287			MemoryContextSwitchTo(plpgsql_compile_tmp_cxt);
288
289			numargs = get_func_arg_info(procTup,
290										&argtypes, &argnames, &argmodes);
291
292			cfunc_resolve_polymorphic_argtypes(numargs, argtypes, argmodes,
293											   fcinfo->flinfo->fn_expr,
294											   forValidator,
295											   plpgsql_error_funcname);
296
297			in_arg_varnos = (int *) palloc(numargs * sizeof(int));
298			out_arg_variables = (PLpgSQL_variable **) palloc(numargs * sizeof(PLpgSQL_variable *));
299
300			MemoryContextSwitchTo(func_cxt);
301
302			/*
303			 * Create the variables for the procedure's parameters.
304			 */
305			for (i = 0; i < numargs; i++)
306			{
307				char		buf[32];
308				Oid			argtypeid = argtypes[i];
309				char		argmode = argmodes ? argmodes[i] : PROARGMODE_IN;
310				PLpgSQL_type *argdtype;
311				PLpgSQL_variable *argvariable;
312				PLpgSQL_nsitem_type argitemtype;
313
314				/* Create $n name for variable */
315				snprintf(buf, sizeof(buf), "$%d", i + 1);
316
317				/* Create datatype info */
318				argdtype = plpgsql_build_datatype(argtypeid,
319												  -1,
320												  function->fn_input_collation,
321												  NULL);
322
323				/* Disallow pseudotype argument */
324				/* (note we already replaced polymorphic types) */
325				/* (build_variable would do this, but wrong message) */
326				if (argdtype->ttype == PLPGSQL_TTYPE_PSEUDO)
327					ereport(ERROR,
328							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
329							 errmsg("PL/pgSQL functions cannot accept type %s",
330									format_type_be(argtypeid))));
331
332				/*
333				 * Build variable and add to datum list.  If there's a name
334				 * for the argument, use that as refname, else use $n name.
335				 */
336				argvariable = plpgsql_build_variable((argnames &&
337													  argnames[i][0] != '\0') ?
338													 argnames[i] : buf,
339													 0, argdtype, false);
340
341				if (argvariable->dtype == PLPGSQL_DTYPE_VAR)
342				{
343					argitemtype = PLPGSQL_NSTYPE_VAR;
344				}
345				else
346				{
347					Assert(argvariable->dtype == PLPGSQL_DTYPE_REC);
348					argitemtype = PLPGSQL_NSTYPE_REC;
349				}
350
351				/* Remember arguments in appropriate arrays */
352				if (argmode == PROARGMODE_IN ||
353					argmode == PROARGMODE_INOUT ||
354					argmode == PROARGMODE_VARIADIC)
355					in_arg_varnos[num_in_args++] = argvariable->dno;
356				if (argmode == PROARGMODE_OUT ||
357					argmode == PROARGMODE_INOUT ||
358					argmode == PROARGMODE_TABLE)
359					out_arg_variables[num_out_args++] = argvariable;
360
361				/* Add to namespace under the $n name */
362				add_parameter_name(argitemtype, argvariable->dno, buf);
363
364				/* If there's a name for the argument, make an alias */
365				if (argnames && argnames[i][0] != '\0')
366					add_parameter_name(argitemtype, argvariable->dno,
367									   argnames[i]);
368			}
369
370			/*
371			 * If there's just one OUT parameter, out_param_varno points
372			 * directly to it.  If there's more than one, build a row that
373			 * holds all of them.  Procedures return a row even for one OUT
374			 * parameter.
375			 */
376			if (num_out_args > 1 ||
377				(num_out_args == 1 && function->fn_prokind == PROKIND_PROCEDURE))
378			{
379				PLpgSQL_row *row = build_row_from_vars(out_arg_variables,
380													   num_out_args);
381
382				plpgsql_adddatum((PLpgSQL_datum *) row);
383				function->out_param_varno = row->dno;
384			}
385			else if (num_out_args == 1)
386				function->out_param_varno = out_arg_variables[0]->dno;
387
388			/*
389			 * Check for a polymorphic returntype. If found, use the actual
390			 * returntype type from the caller's FuncExpr node, if we have
391			 * one.  (In validation mode we arbitrarily assume we are dealing
392			 * with integers.)
393			 *
394			 * Note: errcode is FEATURE_NOT_SUPPORTED because it should always
395			 * work; if it doesn't we're in some context that fails to make
396			 * the info available.
397			 */
398			rettypeid = procStruct->prorettype;
399			if (IsPolymorphicType(rettypeid))
400			{
401				if (forValidator)
402				{
403					if (rettypeid == ANYARRAYOID ||
404						rettypeid == ANYCOMPATIBLEARRAYOID)
405						rettypeid = INT4ARRAYOID;
406					else if (rettypeid == ANYRANGEOID ||
407							 rettypeid == ANYCOMPATIBLERANGEOID)
408						rettypeid = INT4RANGEOID;
409					else if (rettypeid == ANYMULTIRANGEOID)
410						rettypeid = INT4MULTIRANGEOID;
411					else		/* ANYELEMENT or ANYNONARRAY or ANYCOMPATIBLE */
412						rettypeid = INT4OID;
413					/* XXX what could we use for ANYENUM? */
414				}
415				else
416				{
417					rettypeid = get_fn_expr_rettype(fcinfo->flinfo);
418					if (!OidIsValid(rettypeid))
419						ereport(ERROR,
420								(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
421								 errmsg("could not determine actual return type "
422										"for polymorphic function \"%s\"",
423										plpgsql_error_funcname)));
424				}
425			}
426
427			/*
428			 * Normal function has a defined returntype
429			 */
430			function->fn_rettype = rettypeid;
431			function->fn_retset = procStruct->proretset;
432
433			/*
434			 * Lookup the function's return type
435			 */
436			typeTup = SearchSysCache1(TYPEOID, ObjectIdGetDatum(rettypeid));
437			if (!HeapTupleIsValid(typeTup))
438				elog(ERROR, "cache lookup failed for type %u", rettypeid);
439			typeStruct = (Form_pg_type) GETSTRUCT(typeTup);
440
441			/* Disallow pseudotype result, except VOID or RECORD */
442			/* (note we already replaced polymorphic types) */
443			if (typeStruct->typtype == TYPTYPE_PSEUDO)
444			{
445				if (rettypeid == VOIDOID ||
446					rettypeid == RECORDOID)
447					 /* okay */ ;
448				else if (rettypeid == TRIGGEROID || rettypeid == EVENT_TRIGGEROID)
449					ereport(ERROR,
450							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
451							 errmsg("trigger functions can only be called as triggers")));
452				else
453					ereport(ERROR,
454							(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
455							 errmsg("PL/pgSQL functions cannot return type %s",
456									format_type_be(rettypeid))));
457			}
458
459			function->fn_retistuple = type_is_rowtype(rettypeid);
460			function->fn_retisdomain = (typeStruct->typtype == TYPTYPE_DOMAIN);
461			function->fn_retbyval = typeStruct->typbyval;
462			function->fn_rettyplen = typeStruct->typlen;
463
464			/*
465			 * install $0 reference, but only for polymorphic return types,
466			 * and not when the return is specified through an output
467			 * parameter.
468			 */
469			if (IsPolymorphicType(procStruct->prorettype) &&
470				num_out_args == 0)
471			{
472				(void) plpgsql_build_variable("$0", 0,
473											  build_datatype(typeTup,
474															 -1,
475															 function->fn_input_collation,
476															 NULL),
477											  true);
478			}
479
480			ReleaseSysCache(typeTup);
481			break;
482
483		case PLPGSQL_DML_TRIGGER:
484			/* Trigger procedure's return type is unknown yet */
485			function->fn_rettype = InvalidOid;
486			function->fn_retbyval = false;
487			function->fn_retistuple = true;
488			function->fn_retisdomain = false;
489			function->fn_retset = false;
490
491			/* shouldn't be any declared arguments */
492			if (procStruct->pronargs != 0)
493				ereport(ERROR,
494						(errcode(ERRCODE_INVALID_FUNCTION_DEFINITION),
495						 errmsg("trigger functions cannot have declared arguments"),
496						 errhint("The arguments of the trigger can be accessed through TG_NARGS and TG_ARGV instead.")));
497
498			/* Add the record for referencing NEW ROW */
499			rec = plpgsql_build_record("new", 0, NULL, RECORDOID, true);
500			function->new_varno = rec->dno;
501
502			/* Add the record for referencing OLD ROW */
503			rec = plpgsql_build_record("old", 0, NULL, RECORDOID, true);
504			function->old_varno = rec->dno;
505
506			/* Add the variable tg_name */
507			var = plpgsql_build_variable("tg_name", 0,
508										 plpgsql_build_datatype(NAMEOID,
509																-1,
510																function->fn_input_collation,
511																NULL),
512										 true);
513			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
514			var->dtype = PLPGSQL_DTYPE_PROMISE;
515			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_NAME;
516
517			/* Add the variable tg_when */
518			var = plpgsql_build_variable("tg_when", 0,
519										 plpgsql_build_datatype(TEXTOID,
520																-1,
521																function->fn_input_collation,
522																NULL),
523										 true);
524			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
525			var->dtype = PLPGSQL_DTYPE_PROMISE;
526			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_WHEN;
527
528			/* Add the variable tg_level */
529			var = plpgsql_build_variable("tg_level", 0,
530										 plpgsql_build_datatype(TEXTOID,
531																-1,
532																function->fn_input_collation,
533																NULL),
534										 true);
535			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
536			var->dtype = PLPGSQL_DTYPE_PROMISE;
537			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_LEVEL;
538
539			/* Add the variable tg_op */
540			var = plpgsql_build_variable("tg_op", 0,
541										 plpgsql_build_datatype(TEXTOID,
542																-1,
543																function->fn_input_collation,
544																NULL),
545										 true);
546			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
547			var->dtype = PLPGSQL_DTYPE_PROMISE;
548			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_OP;
549
550			/* Add the variable tg_relid */
551			var = plpgsql_build_variable("tg_relid", 0,
552										 plpgsql_build_datatype(OIDOID,
553																-1,
554																InvalidOid,
555																NULL),
556										 true);
557			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
558			var->dtype = PLPGSQL_DTYPE_PROMISE;
559			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_RELID;
560
561			/* Add the variable tg_relname */
562			var = plpgsql_build_variable("tg_relname", 0,
563										 plpgsql_build_datatype(NAMEOID,
564																-1,
565																function->fn_input_collation,
566																NULL),
567										 true);
568			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
569			var->dtype = PLPGSQL_DTYPE_PROMISE;
570			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_TABLE_NAME;
571
572			/* tg_table_name is now preferred to tg_relname */
573			var = plpgsql_build_variable("tg_table_name", 0,
574										 plpgsql_build_datatype(NAMEOID,
575																-1,
576																function->fn_input_collation,
577																NULL),
578										 true);
579			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
580			var->dtype = PLPGSQL_DTYPE_PROMISE;
581			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_TABLE_NAME;
582
583			/* add the variable tg_table_schema */
584			var = plpgsql_build_variable("tg_table_schema", 0,
585										 plpgsql_build_datatype(NAMEOID,
586																-1,
587																function->fn_input_collation,
588																NULL),
589										 true);
590			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
591			var->dtype = PLPGSQL_DTYPE_PROMISE;
592			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_TABLE_SCHEMA;
593
594			/* Add the variable tg_nargs */
595			var = plpgsql_build_variable("tg_nargs", 0,
596										 plpgsql_build_datatype(INT4OID,
597																-1,
598																InvalidOid,
599																NULL),
600										 true);
601			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
602			var->dtype = PLPGSQL_DTYPE_PROMISE;
603			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_NARGS;
604
605			/* Add the variable tg_argv */
606			var = plpgsql_build_variable("tg_argv", 0,
607										 plpgsql_build_datatype(TEXTARRAYOID,
608																-1,
609																function->fn_input_collation,
610																NULL),
611										 true);
612			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
613			var->dtype = PLPGSQL_DTYPE_PROMISE;
614			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_ARGV;
615
616			break;
617
618		case PLPGSQL_EVENT_TRIGGER:
619			function->fn_rettype = VOIDOID;
620			function->fn_retbyval = false;
621			function->fn_retistuple = true;
622			function->fn_retisdomain = false;
623			function->fn_retset = false;
624
625			/* shouldn't be any declared arguments */
626			if (procStruct->pronargs != 0)
627				ereport(ERROR,
628						(errcode(ERRCODE_INVALID_FUNCTION_DEFINITION),
629						 errmsg("event trigger functions cannot have declared arguments")));
630
631			/* Add the variable tg_event */
632			var = plpgsql_build_variable("tg_event", 0,
633										 plpgsql_build_datatype(TEXTOID,
634																-1,
635																function->fn_input_collation,
636																NULL),
637										 true);
638			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
639			var->dtype = PLPGSQL_DTYPE_PROMISE;
640			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_EVENT;
641
642			/* Add the variable tg_tag */
643			var = plpgsql_build_variable("tg_tag", 0,
644										 plpgsql_build_datatype(TEXTOID,
645																-1,
646																function->fn_input_collation,
647																NULL),
648										 true);
649			Assert(var->dtype == PLPGSQL_DTYPE_VAR);
650			var->dtype = PLPGSQL_DTYPE_PROMISE;
651			((PLpgSQL_var *) var)->promise = PLPGSQL_PROMISE_TG_TAG;
652
653			break;
654
655		default:
656			elog(ERROR, "unrecognized function typecode: %d",
657				 (int) function->fn_is_trigger);
658			break;
659	}
660
661	/* Remember if function is STABLE/IMMUTABLE */
662	function->fn_readonly = (procStruct->provolatile != PROVOLATILE_VOLATILE);
663
664	/*
665	 * Create the magic FOUND variable.
666	 */
667	var = plpgsql_build_variable("found", 0,
668								 plpgsql_build_datatype(BOOLOID,
669														-1,
670														InvalidOid,
671														NULL),
672								 true);
673	function->found_varno = var->dno;
674
675	/*
676	 * Now parse the function's text
677	 */
678	parse_rc = plpgsql_yyparse(&function->action, scanner);
679	if (parse_rc != 0)
680		elog(ERROR, "plpgsql parser returned %d", parse_rc);
681
682	plpgsql_scanner_finish(scanner);
683	pfree(proc_source);
684
685	/*
686	 * If it has OUT parameters or returns VOID or returns a set, we allow
687	 * control to fall off the end without an explicit RETURN statement. The
688	 * easiest way to implement this is to add a RETURN statement to the end
689	 * of the statement list during parsing.
690	 */
691	if (num_out_args > 0 || function->fn_rettype == VOIDOID ||
692		function->fn_retset)
693		add_dummy_return(function);
694
695	/*
696	 * Complete the function's info
697	 */
698	function->fn_nargs = procStruct->pronargs;
699	for (i = 0; i < function->fn_nargs; i++)
700		function->fn_argvarnos[i] = in_arg_varnos[i];
701
702	plpgsql_finish_datums(function);
703
704	if (function->has_exception_block)
705		plpgsql_mark_local_assignment_targets(function);
706
707	/* Debug dump for completed functions */
708	if (plpgsql_DumpExecTree)
709		plpgsql_dumptree(function);
710
711	/*
712	 * All is well, so make the func_cxt long-lived
713	 */
714	MemoryContextSetParent(func_cxt, CacheMemoryContext);
715
716	/*
717	 * Pop the error context stack
718	 */
719	error_context_stack = plerrcontext.previous;
720	plpgsql_error_funcname = NULL;
721
722	plpgsql_check_syntax = false;
723
724	MemoryContextSwitchTo(plpgsql_compile_tmp_cxt);
725	plpgsql_compile_tmp_cxt = NULL;
726}

728/* ----------
729 * plpgsql_compile_inline	Make an execution tree for an anonymous code block.
730 *
731 * Note: this is generally parallel to plpgsql_compile_callback(); is it worth
732 * trying to merge the two?
733 *
734 * Note: we assume the block will be thrown away so there is no need to build
735 * persistent data structures.
736 * ----------
737 */
738PLpgSQL_function *
739plpgsql_compile_inline(char *proc_source)
740{
741	yyscan_t	scanner;
742	char	   *func_name = "inline_code_block";
743	PLpgSQL_function *function;
744	struct compile_error_callback_arg cbarg;
745	ErrorContextCallback plerrcontext;
746	PLpgSQL_variable *var;
747	int			parse_rc;
748	MemoryContext func_cxt;
749
750	/*
751	 * Setup the scanner input and error info.
752	 */
753	scanner = plpgsql_scanner_init(proc_source);
754
755	plpgsql_error_funcname = func_name;
756
757	/*
758	 * Setup error traceback support for ereport()
759	 */
760	cbarg.proc_source = proc_source;
761	cbarg.yyscanner = scanner;
762	plerrcontext.callback = plpgsql_compile_error_callback;
763	plerrcontext.arg = &cbarg;
764	plerrcontext.previous = error_context_stack;
765	error_context_stack = &plerrcontext;
766
767	/* Do extra syntax checking if check_function_bodies is on */
768	plpgsql_check_syntax = check_function_bodies;
769
770	/* Function struct does not live past current statement */
771	function = (PLpgSQL_function *) palloc0(sizeof(PLpgSQL_function));
772
773	plpgsql_curr_compile = function;
774
775	/*
776	 * All the rest of the compile-time storage (e.g. parse tree) is kept in
777	 * its own memory context, so it can be reclaimed easily.
778	 */
779	func_cxt = AllocSetContextCreate(CurrentMemoryContext,
780									 "PL/pgSQL inline code context",
781									 ALLOCSET_DEFAULT_SIZES);
782	plpgsql_compile_tmp_cxt = MemoryContextSwitchTo(func_cxt);
783
784	function->fn_signature = pstrdup(func_name);
785	function->fn_is_trigger = PLPGSQL_NOT_TRIGGER;
786	function->fn_input_collation = InvalidOid;
787	function->fn_cxt = func_cxt;
788	function->out_param_varno = -1; /* set up for no OUT param */
789	function->resolve_option = plpgsql_variable_conflict;
790	function->print_strict_params = plpgsql_print_strict_params;
791
792	/*
793	 * don't do extra validation for inline code as we don't want to add spam
794	 * at runtime
795	 */
796	function->extra_warnings = 0;
797	function->extra_errors = 0;
798
799	function->nstatements = 0;
800	function->requires_procedure_resowner = false;
801	function->has_exception_block = false;
802
803	plpgsql_ns_init();
804	plpgsql_ns_push(func_name, PLPGSQL_LABEL_BLOCK);
805	plpgsql_DumpExecTree = false;
806	plpgsql_start_datums();
807
808	/* Set up as though in a function returning VOID */
809	function->fn_rettype = VOIDOID;
810	function->fn_retset = false;
811	function->fn_retistuple = false;
812	function->fn_retisdomain = false;
813	function->fn_prokind = PROKIND_FUNCTION;
814	/* a bit of hardwired knowledge about type VOID here */
815	function->fn_retbyval = true;
816	function->fn_rettyplen = sizeof(int32);
817
818	/*
819	 * Remember if function is STABLE/IMMUTABLE.  XXX would it be better to
820	 * set this true inside a read-only transaction?  Not clear.
821	 */
822	function->fn_readonly = false;
823
824	/*
825	 * Create the magic FOUND variable.
826	 */
827	var = plpgsql_build_variable("found", 0,
828								 plpgsql_build_datatype(BOOLOID,
829														-1,
830														InvalidOid,
831														NULL),
832								 true);
833	function->found_varno = var->dno;
834
835	/*
836	 * Now parse the function's text
837	 */
838	parse_rc = plpgsql_yyparse(&function->action, scanner);
839	if (parse_rc != 0)
840		elog(ERROR, "plpgsql parser returned %d", parse_rc);
841
842	plpgsql_scanner_finish(scanner);
843
844	/*
845	 * If it returns VOID (always true at the moment), we allow control to
846	 * fall off the end without an explicit RETURN statement.
847	 */
848	if (function->fn_rettype == VOIDOID)
849		add_dummy_return(function);
850
851	/*
852	 * Complete the function's info
853	 */
854	function->fn_nargs = 0;
855
856	plpgsql_finish_datums(function);
857
858	if (function->has_exception_block)
859		plpgsql_mark_local_assignment_targets(function);
860
861	/* Debug dump for completed functions */
862	if (plpgsql_DumpExecTree)
863		plpgsql_dumptree(function);
864
865	/*
866	 * Pop the error context stack
867	 */
868	error_context_stack = plerrcontext.previous;
869	plpgsql_error_funcname = NULL;
870
871	plpgsql_check_syntax = false;
872
873	MemoryContextSwitchTo(plpgsql_compile_tmp_cxt);
874	plpgsql_compile_tmp_cxt = NULL;
875	return function;
876}

879/*
880 * error context callback to let us supply a call-stack traceback.
881 * If we are validating or executing an anonymous code block, the function
882 * source text is passed as an argument.
883 */
884static void
885plpgsql_compile_error_callback(void *arg)
886{
887	struct compile_error_callback_arg *cbarg = (struct compile_error_callback_arg *) arg;
888	yyscan_t	yyscanner = cbarg->yyscanner;
889
890	if (cbarg->proc_source)
891	{
892		/*
893		 * Try to convert syntax error position to reference text of original
894		 * CREATE FUNCTION or DO command.
895		 */
896		if (function_parse_error_transpose(cbarg->proc_source))
897			return;
898
899		/*
900		 * Done if a syntax error position was reported; otherwise we have to
901		 * fall back to a "near line N" report.
902		 */
903	}
904
905	if (plpgsql_error_funcname)
906		errcontext("compilation of PL/pgSQL function \"%s\" near line %d",
907				   plpgsql_error_funcname, plpgsql_latest_lineno(yyscanner));
908}

911/*
912 * Add a name for a function parameter to the function's namespace
913 */
914static void
915add_parameter_name(PLpgSQL_nsitem_type itemtype, int itemno, const char *name)
916{
917	/*
918	 * Before adding the name, check for duplicates.  We need this even though
919	 * functioncmds.c has a similar check, because that code explicitly
920	 * doesn't complain about conflicting IN and OUT parameter names.  In
921	 * plpgsql, such names are in the same namespace, so there is no way to
922	 * disambiguate.
923	 */
924	if (plpgsql_ns_lookup(plpgsql_ns_top(), true,
925						  name, NULL, NULL,
926						  NULL) != NULL)
927		ereport(ERROR,
928				(errcode(ERRCODE_INVALID_FUNCTION_DEFINITION),
929				 errmsg("parameter name \"%s\" used more than once",
930						name)));
931
932	/* OK, add the name */
933	plpgsql_ns_additem(itemtype, itemno, name);
934}

936/*
937 * Add a dummy RETURN statement to the given function's body
938 */
939static void
940add_dummy_return(PLpgSQL_function *function)
941{
942	/*
943	 * If the outer block has an EXCEPTION clause, we need to make a new outer
944	 * block, since the added RETURN shouldn't act like it is inside the
945	 * EXCEPTION clause.  Likewise, if it has a label, wrap it in a new outer
946	 * block so that EXIT doesn't skip the RETURN.
947	 */
948	if (function->action->exceptions != NULL ||
949		function->action->label != NULL)
950	{
951		PLpgSQL_stmt_block *new;
952
953		new = palloc0(sizeof(PLpgSQL_stmt_block));
954		new->cmd_type = PLPGSQL_STMT_BLOCK;
955		new->stmtid = ++function->nstatements;
956		new->body = list_make1(function->action);
957
958		function->action = new;
959	}
960	if (function->action->body == NIL ||
961		((PLpgSQL_stmt *) llast(function->action->body))->cmd_type != PLPGSQL_STMT_RETURN)
962	{
963		PLpgSQL_stmt_return *new;
964
965		new = palloc0(sizeof(PLpgSQL_stmt_return));
966		new->cmd_type = PLPGSQL_STMT_RETURN;
967		new->stmtid = ++function->nstatements;
968		new->expr = NULL;
969		new->retvarno = function->out_param_varno;
970
971		function->action->body = lappend(function->action->body, new);
972	}
973}

978 *
979 * Note: this routine, and the hook functions it prepares for, are logically
980 * part of plpgsql parsing.  But they actually run during function execution,
981 * when we are ready to evaluate a SQL query or expression that has not
982 * previously been parsed and planned.

988	pstate->p_post_columnref_hook = plpgsql_post_column_ref;
989	pstate->p_paramref_hook = plpgsql_param_ref;
990	/* no need to use p_coerce_param_hook */
991	pstate->p_ref_hook_state = expr;

994/*
995 * plpgsql_pre_column_ref		parser callback before parsing a ColumnRef
996 */
997static Node *
998plpgsql_pre_column_ref(ParseState *pstate, ColumnRef *cref)
999{
1000	PLpgSQL_expr *expr = (PLpgSQL_expr *) pstate->p_ref_hook_state;
1001
1002	if (expr->func->resolve_option == PLPGSQL_RESOLVE_VARIABLE)
1003		return resolve_column_ref(pstate, expr, cref, false);
1004	else
1005		return NULL;
1006}
1007
1008/*
1009 * plpgsql_post_column_ref		parser callback after parsing a ColumnRef
1010 */
1011static Node *
1012plpgsql_post_column_ref(ParseState *pstate, ColumnRef *cref, Node *var)
1013{
1014	PLpgSQL_expr *expr = (PLpgSQL_expr *) pstate->p_ref_hook_state;
1015	Node	   *myvar;
1016
1017	if (expr->func->resolve_option == PLPGSQL_RESOLVE_VARIABLE)
1018		return NULL;			/* we already found there's no match */
1019
1020	if (expr->func->resolve_option == PLPGSQL_RESOLVE_COLUMN && var != NULL)
1021		return NULL;			/* there's a table column, prefer that */
1022
1023	/*
1024	 * If we find a record/row variable but can't match a field name, throw
1025	 * error if there was no core resolution for the ColumnRef either.  In
1026	 * that situation, the reference is inevitably going to fail, and
1027	 * complaining about the record/row variable is likely to be more on-point
1028	 * than the core parser's error message.  (It's too bad we don't have
1029	 * access to transformColumnRef's internal crerr state here, as in case of
1030	 * a conflict with a table name this could still be less than the most
1031	 * helpful error message possible.)
1032	 */
1033	myvar = resolve_column_ref(pstate, expr, cref, (var == NULL));
1034
1035	if (myvar != NULL && var != NULL)
1036	{
1037		/*
1038		 * We could leave it to the core parser to throw this error, but we
1039		 * can add a more useful detail message than the core could.
1040		 */
1041		ereport(ERROR,
1042				(errcode(ERRCODE_AMBIGUOUS_COLUMN),
1043				 errmsg("column reference \"%s\" is ambiguous",
1044						NameListToString(cref->fields)),
1045				 errdetail("It could refer to either a PL/pgSQL variable or a table column."),
1046				 parser_errposition(pstate, cref->location)));
1047	}
1048
1049	return myvar;
1050}

1052/*
1053 * plpgsql_param_ref		parser callback for ParamRefs ($n symbols)
1054 */
1055static Node *
1056plpgsql_param_ref(ParseState *pstate, ParamRef *pref)
1057{
1058	PLpgSQL_expr *expr = (PLpgSQL_expr *) pstate->p_ref_hook_state;
1059	char		pname[32];
1060	PLpgSQL_nsitem *nse;
1061
1062	snprintf(pname, sizeof(pname), "$%d", pref->number);
1063
1064	nse = plpgsql_ns_lookup(expr->ns, false,
1065							pname, NULL, NULL,
1066							NULL);
1067
1068	if (nse == NULL)
1069		return NULL;			/* name not known to plpgsql */
1070
1071	return make_datum_param(expr, nse->itemno, pref->location);
1072}

1074/*
1075 * resolve_column_ref		attempt to resolve a ColumnRef as a plpgsql var
1076 *
1077 * Returns the translated node structure, or NULL if name not found
1078 *
1079 * error_if_no_field tells whether to throw error or quietly return NULL if
1080 * we are able to match a record/row name but don't find a field name match.
1081 */
1082static Node *
1083resolve_column_ref(ParseState *pstate, PLpgSQL_expr *expr,
1084				   ColumnRef *cref, bool error_if_no_field)
1085{
1086	PLpgSQL_execstate *estate;
1087	PLpgSQL_nsitem *nse;
1088	const char *name1;
1089	const char *name2 = NULL;
1090	const char *name3 = NULL;
1091	const char *colname = NULL;
1092	int			nnames;
1093	int			nnames_scalar = 0;
1094	int			nnames_wholerow = 0;
1095	int			nnames_field = 0;
1096
1097	/*
1098	 * We use the function's current estate to resolve parameter data types.
1099	 * This is really pretty bogus because there is no provision for updating
1100	 * plans when those types change ...
1101	 */
1102	estate = expr->func->cur_estate;
1103
1104	/*----------
1105	 * The allowed syntaxes are:
1106	 *
1107	 * A		Scalar variable reference, or whole-row record reference.
1108	 * A.B		Qualified scalar or whole-row reference, or field reference.
1109	 * A.B.C	Qualified record field reference.
1110	 * A.*		Whole-row record reference.
1111	 * A.B.*	Qualified whole-row record reference.
1112	 *----------
1113	 */
1114	switch (list_length(cref->fields))
1115	{
1116		case 1:
1117			{
1118				Node	   *field1 = (Node *) linitial(cref->fields);
1119
1120				name1 = strVal(field1);
1121				nnames_scalar = 1;
1122				nnames_wholerow = 1;
1123				break;
1124			}
1125		case 2:
1126			{
1127				Node	   *field1 = (Node *) linitial(cref->fields);
1128				Node	   *field2 = (Node *) lsecond(cref->fields);
1129
1130				name1 = strVal(field1);
1131
1132				/* Whole-row reference? */
1133				if (IsA(field2, A_Star))
1134				{
1135					/* Set name2 to prevent matches to scalar variables */
1136					name2 = "*";
1137					nnames_wholerow = 1;
1138					break;
1139				}
1140
1141				name2 = strVal(field2);
1142				colname = name2;
1143				nnames_scalar = 2;
1144				nnames_wholerow = 2;
1145				nnames_field = 1;
1146				break;
1147			}
1148		case 3:
1149			{
1150				Node	   *field1 = (Node *) linitial(cref->fields);
1151				Node	   *field2 = (Node *) lsecond(cref->fields);
1152				Node	   *field3 = (Node *) lthird(cref->fields);
1153
1154				name1 = strVal(field1);
1155				name2 = strVal(field2);
1156
1157				/* Whole-row reference? */
1158				if (IsA(field3, A_Star))
1159				{
1160					/* Set name3 to prevent matches to scalar variables */
1161					name3 = "*";
1162					nnames_wholerow = 2;
1163					break;
1164				}
1165
1166				name3 = strVal(field3);
1167				colname = name3;
1168				nnames_field = 2;
1169				break;
1170			}
1171		default:
1172			/* too many names, ignore */
1173			return NULL;
1174	}
1175
1176	nse = plpgsql_ns_lookup(expr->ns, false,
1177							name1, name2, name3,
1178							&nnames);
1179
1180	if (nse == NULL)
1181		return NULL;			/* name not known to plpgsql */
1182
1183	switch (nse->itemtype)
1184	{
1185		case PLPGSQL_NSTYPE_VAR:
1186			if (nnames == nnames_scalar)
1187				return make_datum_param(expr, nse->itemno, cref->location);
1188			break;
1189		case PLPGSQL_NSTYPE_REC:
1190			if (nnames == nnames_wholerow)
1191				return make_datum_param(expr, nse->itemno, cref->location);
1192			if (nnames == nnames_field)
1193			{
1194				/* colname could be a field in this record */
1195				PLpgSQL_rec *rec = (PLpgSQL_rec *) estate->datums[nse->itemno];
1196				int			i;
1197
1198				/* search for a datum referencing this field */
1199				i = rec->firstfield;
1200				while (i >= 0)
1201				{
1202					PLpgSQL_recfield *fld = (PLpgSQL_recfield *) estate->datums[i];
1203
1204					Assert(fld->dtype == PLPGSQL_DTYPE_RECFIELD &&
1205						   fld->recparentno == nse->itemno);
1206					if (strcmp(fld->fieldname, colname) == 0)
1207					{
1208						return make_datum_param(expr, i, cref->location);
1209					}
1210					i = fld->nextfield;
1211				}
1212
1213				/*
1214				 * We should not get here, because a RECFIELD datum should
1215				 * have been built at parse time for every possible qualified
1216				 * reference to fields of this record.  But if we do, handle
1217				 * it like field-not-found: throw error or return NULL.
1218				 */
1219				if (error_if_no_field)
1220					ereport(ERROR,
1221							(errcode(ERRCODE_UNDEFINED_COLUMN),
1222							 errmsg("record \"%s\" has no field \"%s\"",
1223									(nnames_field == 1) ? name1 : name2,
1224									colname),
1225							 parser_errposition(pstate, cref->location)));
1226			}
1227			break;
1228		default:
1229			elog(ERROR, "unrecognized plpgsql itemtype: %d", nse->itemtype);
1230	}
1231
1232	/* Name format doesn't match the plpgsql variable type */
1233	return NULL;
1234}

1236/*
1237 * Helper for columnref parsing: build a Param referencing a plpgsql datum,
1238 * and make sure that that datum is listed in the expression's paramnos.
1239 */
1240static Node *
1241make_datum_param(PLpgSQL_expr *expr, int dno, int location)
1242{
1243	PLpgSQL_execstate *estate;
1244	PLpgSQL_datum *datum;
1245	Param	   *param;
1246	MemoryContext oldcontext;
1247
1248	/* see comment in resolve_column_ref */
1249	estate = expr->func->cur_estate;
1250	Assert(dno >= 0 && dno < estate->ndatums);
1251	datum = estate->datums[dno];
1252
1253	/*
1254	 * Bitmapset must be allocated in function's permanent memory context
1255	 */
1256	oldcontext = MemoryContextSwitchTo(expr->func->fn_cxt);
1257	expr->paramnos = bms_add_member(expr->paramnos, dno);
1258	MemoryContextSwitchTo(oldcontext);
1259
1260	param = makeNode(Param);
1261	param->paramkind = PARAM_EXTERN;
1262	param->paramid = dno + 1;
1263	plpgsql_exec_get_datum_type_info(estate,
1264									 datum,
1265									 &param->paramtype,
1266									 &param->paramtypmod,
1267									 &param->paramcollid);
1268	param->location = location;
1269
1270	return (Node *) param;
1271}

1274/* ----------
1275 * plpgsql_parse_word		The scanner calls this to postparse
1276 *				any single word that is not a reserved keyword.
1277 *
1278 * word1 is the downcased/dequoted identifier; it must be palloc'd in the
1279 * function's long-term memory context.
1280 *
1281 * yytxt is the original token text; we need this to check for quoting,
1282 * so that later checks for unreserved keywords work properly.
1283 *
1284 * We attempt to recognize the token as a variable only if lookup is true
1285 * and the plpgsql_IdentifierLookup context permits it.
1286 *
1287 * If recognized as a variable, fill in *wdatum and return true;
1288 * if not recognized, fill in *word and return false.
1289 * (Note: those two pointers actually point to members of the same union,
1290 * but for notational reasons we pass them separately.)
1291 * ----------
1292 */
1293bool
1294plpgsql_parse_word(char *word1, const char *yytxt, bool lookup,
1295				   PLwdatum *wdatum, PLword *word)
1296{
1297	PLpgSQL_nsitem *ns;
1298
1299	/*
1300	 * We should not lookup variables in DECLARE sections.  In SQL
1301	 * expressions, there's no need to do so either --- lookup will happen
1302	 * when the expression is compiled.
1303	 */
1304	if (lookup && plpgsql_IdentifierLookup == IDENTIFIER_LOOKUP_NORMAL)
1305	{
1306		/*
1307		 * Do a lookup in the current namespace stack
1308		 */
1309		ns = plpgsql_ns_lookup(plpgsql_ns_top(), false,
1310							   word1, NULL, NULL,
1311							   NULL);
1312
1313		if (ns != NULL)
1314		{
1315			switch (ns->itemtype)
1316			{
1317				case PLPGSQL_NSTYPE_VAR:
1318				case PLPGSQL_NSTYPE_REC:
1319					wdatum->datum = plpgsql_Datums[ns->itemno];
1320					wdatum->ident = word1;
1321					wdatum->quoted = (yytxt[0] == '"');
1322					wdatum->idents = NIL;
1323					return true;
1324
1325				default:
1326					/* plpgsql_ns_lookup should never return anything else */
1327					elog(ERROR, "unrecognized plpgsql itemtype: %d",
1328						 ns->itemtype);
1329			}
1330		}
1331	}
1332
1333	/*
1334	 * Nothing found - up to now it's a word without any special meaning for
1335	 * us.
1336	 */
1337	word->ident = word1;
1338	word->quoted = (yytxt[0] == '"');
1339	return false;
1340}

1343/* ----------
1344 * plpgsql_parse_dblword		Same lookup for two words
1345 *					separated by a dot.
1346 * ----------
1347 */
1348bool
1349plpgsql_parse_dblword(char *word1, char *word2,
1350					  PLwdatum *wdatum, PLcword *cword)
1351{
1352	PLpgSQL_nsitem *ns;
1353	List	   *idents;
1354	int			nnames;
1355
1356	idents = list_make2(makeString(word1),
1357						makeString(word2));
1358
1359	/*
1360	 * We should do nothing in DECLARE sections.  In SQL expressions, we
1361	 * really only need to make sure that RECFIELD datums are created when
1362	 * needed.  In all the cases handled by this function, returning a T_DATUM
1363	 * with a two-word idents string is the right thing.
1364	 */
1365	if (plpgsql_IdentifierLookup != IDENTIFIER_LOOKUP_DECLARE)
1366	{
1367		/*
1368		 * Do a lookup in the current namespace stack
1369		 */
1370		ns = plpgsql_ns_lookup(plpgsql_ns_top(), false,
1371							   word1, word2, NULL,
1372							   &nnames);
1373		if (ns != NULL)
1374		{
1375			switch (ns->itemtype)
1376			{
1377				case PLPGSQL_NSTYPE_VAR:
1378					/* Block-qualified reference to scalar variable. */
1379					wdatum->datum = plpgsql_Datums[ns->itemno];
1380					wdatum->ident = NULL;
1381					wdatum->quoted = false; /* not used */
1382					wdatum->idents = idents;
1383					return true;
1384
1385				case PLPGSQL_NSTYPE_REC:
1386					if (nnames == 1)
1387					{
1388						/*
1389						 * First word is a record name, so second word could
1390						 * be a field in this record.  We build a RECFIELD
1391						 * datum whether it is or not --- any error will be
1392						 * detected later.
1393						 */
1394						PLpgSQL_rec *rec;
1395						PLpgSQL_recfield *new;
1396
1397						rec = (PLpgSQL_rec *) (plpgsql_Datums[ns->itemno]);
1398						new = plpgsql_build_recfield(rec, word2);
1399
1400						wdatum->datum = (PLpgSQL_datum *) new;
1401					}
1402					else
1403					{
1404						/* Block-qualified reference to record variable. */
1405						wdatum->datum = plpgsql_Datums[ns->itemno];
1406					}
1407					wdatum->ident = NULL;
1408					wdatum->quoted = false; /* not used */
1409					wdatum->idents = idents;
1410					return true;
1411
1412				default:
1413					break;
1414			}
1415		}
1416	}
1417
1418	/* Nothing found */
1419	cword->idents = idents;
1420	return false;
1421}

1424/* ----------
1425 * plpgsql_parse_tripword		Same lookup for three words
1426 *					separated by dots.
1427 * ----------
1428 */
1429bool
1430plpgsql_parse_tripword(char *word1, char *word2, char *word3,
1431					   PLwdatum *wdatum, PLcword *cword)
1432{
1433	PLpgSQL_nsitem *ns;
1434	List	   *idents;
1435	int			nnames;
1436
1437	/*
1438	 * We should do nothing in DECLARE sections.  In SQL expressions, we need
1439	 * to make sure that RECFIELD datums are created when needed, and we need
1440	 * to be careful about how many names are reported as belonging to the
1441	 * T_DATUM: the third word could be a sub-field reference, which we don't
1442	 * care about here.
1443	 */
1444	if (plpgsql_IdentifierLookup != IDENTIFIER_LOOKUP_DECLARE)
1445	{
1446		/*
1447		 * Do a lookup in the current namespace stack.  Must find a record
1448		 * reference, else ignore.
1449		 */
1450		ns = plpgsql_ns_lookup(plpgsql_ns_top(), false,
1451							   word1, word2, word3,
1452							   &nnames);
1453		if (ns != NULL)
1454		{
1455			switch (ns->itemtype)
1456			{
1457				case PLPGSQL_NSTYPE_REC:
1458					{
1459						PLpgSQL_rec *rec;
1460						PLpgSQL_recfield *new;
1461
1462						rec = (PLpgSQL_rec *) (plpgsql_Datums[ns->itemno]);
1463						if (nnames == 1)
1464						{
1465							/*
1466							 * First word is a record name, so second word
1467							 * could be a field in this record (and the third,
1468							 * a sub-field).  We build a RECFIELD datum
1469							 * whether it is or not --- any error will be
1470							 * detected later.
1471							 */
1472							new = plpgsql_build_recfield(rec, word2);
1473							idents = list_make2(makeString(word1),
1474												makeString(word2));
1475						}
1476						else
1477						{
1478							/* Block-qualified reference to record variable. */
1479							new = plpgsql_build_recfield(rec, word3);
1480							idents = list_make3(makeString(word1),
1481												makeString(word2),
1482												makeString(word3));
1483						}
1484						wdatum->datum = (PLpgSQL_datum *) new;
1485						wdatum->ident = NULL;
1486						wdatum->quoted = false; /* not used */
1487						wdatum->idents = idents;
1488						return true;
1489					}
1490
1491				default:
1492					break;
1493			}
1494		}
1495	}
1496
1497	/* Nothing found */
1498	idents = list_make3(makeString(word1),
1499						makeString(word2),
1500						makeString(word3));
1501	cword->idents = idents;
1502	return false;
1503}

1546/* ----------
1547 * plpgsql_parse_cwordtype		Same lookup for compositeword%TYPE
1548 *
1549 * Here, we allow either a block-qualified variable name, or a reference
1550 * to a column of some table.  (If we must throw error, we assume that the
1551 * latter case was intended.)
1552 * ----------
1553 */
1554PLpgSQL_type *
1555plpgsql_parse_cwordtype(List *idents)
1556{
1557	PLpgSQL_type *dtype = NULL;
1558	PLpgSQL_nsitem *nse;
1559	int			nnames;
1560	RangeVar   *relvar = NULL;
1561	const char *fldname = NULL;
1562	Oid			classOid;
1563	HeapTuple	attrtup = NULL;
1564	HeapTuple	typetup = NULL;
1565	Form_pg_attribute attrStruct;
1566	MemoryContext oldCxt;
1567
1568	/* Avoid memory leaks in the long-term function context */
1569	oldCxt = MemoryContextSwitchTo(plpgsql_compile_tmp_cxt);
1570
1571	if (list_length(idents) == 2)
1572	{
1573		/*
1574		 * Do a lookup in the current namespace stack
1575		 */
1576		nse = plpgsql_ns_lookup(plpgsql_ns_top(), false,
1577								strVal(linitial(idents)),
1578								strVal(lsecond(idents)),
1579								NULL,
1580								&nnames);
1581
1582		if (nse != NULL && nse->itemtype == PLPGSQL_NSTYPE_VAR)
1583		{
1584			/* Block-qualified reference to scalar variable. */
1585			dtype = ((PLpgSQL_var *) (plpgsql_Datums[nse->itemno]))->datatype;
1586			goto done;
1587		}
1588		else if (nse != NULL && nse->itemtype == PLPGSQL_NSTYPE_REC &&
1589				 nnames == 2)
1590		{
1591			/* Block-qualified reference to record variable. */
1592			dtype = ((PLpgSQL_rec *) (plpgsql_Datums[nse->itemno]))->datatype;
1593			goto done;
1594		}
1595
1596		/*
1597		 * First word could also be a table name
1598		 */
1599		relvar = makeRangeVar(NULL,
1600							  strVal(linitial(idents)),
1601							  -1);
1602		fldname = strVal(lsecond(idents));
1603	}
1604	else
1605	{
1606		/*
1607		 * We could check for a block-qualified reference to a field of a
1608		 * record variable, but %TYPE is documented as applying to variables,
1609		 * not fields of variables.  Things would get rather ambiguous if we
1610		 * allowed either interpretation.
1611		 */
1612		List	   *rvnames;
1613
1614		Assert(list_length(idents) > 2);
1615		rvnames = list_delete_last(list_copy(idents));
1616		relvar = makeRangeVarFromNameList(rvnames);
1617		fldname = strVal(llast(idents));
1618	}
1619
1620	/* Look up relation name.  Can't lock it - we might not have privileges. */
1621	classOid = RangeVarGetRelid(relvar, NoLock, false);
1622
1623	/*
1624	 * Fetch the named table field and its type
1625	 */
1626	attrtup = SearchSysCacheAttName(classOid, fldname);
1627	if (!HeapTupleIsValid(attrtup))
1628		ereport(ERROR,
1629				(errcode(ERRCODE_UNDEFINED_COLUMN),
1630				 errmsg("column \"%s\" of relation \"%s\" does not exist",
1631						fldname, relvar->relname)));
1632	attrStruct = (Form_pg_attribute) GETSTRUCT(attrtup);
1633
1634	typetup = SearchSysCache1(TYPEOID,
1635							  ObjectIdGetDatum(attrStruct->atttypid));
1636	if (!HeapTupleIsValid(typetup))
1637		elog(ERROR, "cache lookup failed for type %u", attrStruct->atttypid);
1638
1639	/*
1640	 * Found that - build a compiler type struct in the caller's cxt and
1641	 * return it.  Note that we treat the type as being found-by-OID; no
1642	 * attempt to re-look-up the type name will happen during invalidations.
1643	 */
1644	MemoryContextSwitchTo(oldCxt);
1645	dtype = build_datatype(typetup,
1646						   attrStruct->atttypmod,
1647						   attrStruct->attcollation,
1648						   NULL);
1649	MemoryContextSwitchTo(plpgsql_compile_tmp_cxt);
1650
1651done:
1652	if (HeapTupleIsValid(attrtup))
1653		ReleaseSysCache(attrtup);
1654	if (HeapTupleIsValid(typetup))
1655		ReleaseSysCache(typetup);
1656
1657	MemoryContextSwitchTo(oldCxt);
1658	return dtype;
1659}

1661/* ----------
1662 * plpgsql_parse_wordrowtype		Scanner found word%ROWTYPE.
1663 *					So word must be a table name.
1664 * ----------
1665 */
1666PLpgSQL_type *
1667plpgsql_parse_wordrowtype(char *ident)
1668{
1669	Oid			classOid;
1670	Oid			typOid;
1671
1672	/*
1673	 * Look up the relation.  Note that because relation rowtypes have the
1674	 * same names as their relations, this could be handled as a type lookup
1675	 * equally well; we use the relation lookup code path only because the
1676	 * errors thrown here have traditionally referred to relations not types.
1677	 * But we'll make a TypeName in case we have to do re-look-up of the type.
1678	 */
1679	classOid = RelnameGetRelid(ident);
1680	if (!OidIsValid(classOid))
1681		ereport(ERROR,
1682				(errcode(ERRCODE_UNDEFINED_TABLE),
1683				 errmsg("relation \"%s\" does not exist", ident)));
1684
1685	/* Some relkinds lack type OIDs */
1686	typOid = get_rel_type_id(classOid);
1687	if (!OidIsValid(typOid))
1688		ereport(ERROR,
1689				(errcode(ERRCODE_WRONG_OBJECT_TYPE),
1690				 errmsg("relation \"%s\" does not have a composite type",
1691						ident)));
1692
1693	/* Build and return the row type struct */
1694	return plpgsql_build_datatype(typOid, -1, InvalidOid,
1695								  makeTypeName(ident));
1696}

1738/*
1739 * plpgsql_build_variable - build a datum-array entry of a given
1740 * datatype
1741 *
1742 * The returned struct may be a PLpgSQL_var or PLpgSQL_rec
1743 * depending on the given datatype, and is allocated via
1744 * palloc.  The struct is automatically added to the current datum
1745 * array, and optionally to the current namespace.
1746 */
1747PLpgSQL_variable *
1748plpgsql_build_variable(const char *refname, int lineno, PLpgSQL_type *dtype,
1749					   bool add2namespace)
1750{
1751	PLpgSQL_variable *result;
1752
1753	switch (dtype->ttype)
1754	{
1755		case PLPGSQL_TTYPE_SCALAR:
1756			{
1757				/* Ordinary scalar datatype */
1758				PLpgSQL_var *var;
1759
1760				var = palloc0(sizeof(PLpgSQL_var));
1761				var->dtype = PLPGSQL_DTYPE_VAR;
1762				var->refname = pstrdup(refname);
1763				var->lineno = lineno;
1764				var->datatype = dtype;
1765				/* other fields are left as 0, might be changed by caller */
1766
1767				/* preset to NULL */
1768				var->value = 0;
1769				var->isnull = true;
1770				var->freeval = false;
1771
1772				plpgsql_adddatum((PLpgSQL_datum *) var);
1773				if (add2namespace)
1774					plpgsql_ns_additem(PLPGSQL_NSTYPE_VAR,
1775									   var->dno,
1776									   refname);
1777				result = (PLpgSQL_variable *) var;
1778				break;
1779			}
1780		case PLPGSQL_TTYPE_REC:
1781			{
1782				/* Composite type -- build a record variable */
1783				PLpgSQL_rec *rec;
1784
1785				rec = plpgsql_build_record(refname, lineno,
1786										   dtype, dtype->typoid,
1787										   add2namespace);
1788				result = (PLpgSQL_variable *) rec;
1789				break;
1790			}
1791		case PLPGSQL_TTYPE_PSEUDO:
1792			ereport(ERROR,
1793					(errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
1794					 errmsg("variable \"%s\" has pseudo-type %s",
1795							refname, format_type_be(dtype->typoid))));
1796			result = NULL;		/* keep compiler quiet */
1797			break;
1798		default:
1799			elog(ERROR, "unrecognized ttype: %d", dtype->ttype);
1800			result = NULL;		/* keep compiler quiet */
1801			break;
1802	}
1803
1804	return result;
1805}

1833/*
1834 * Build a row-variable data structure given the component variables.
1835 * Include a rowtupdesc, since we will need to materialize the row result.
1836 */
1837static PLpgSQL_row *
1838build_row_from_vars(PLpgSQL_variable **vars, int numvars)
1839{
1840	PLpgSQL_row *row;
1841	int			i;
1842
1843	row = palloc0(sizeof(PLpgSQL_row));
1844	row->dtype = PLPGSQL_DTYPE_ROW;
1845	row->refname = "(unnamed row)";
1846	row->lineno = -1;
1847	row->rowtupdesc = CreateTemplateTupleDesc(numvars);
1848	row->nfields = numvars;
1849	row->fieldnames = palloc(numvars * sizeof(char *));
1850	row->varnos = palloc(numvars * sizeof(int));
1851
1852	for (i = 0; i < numvars; i++)
1853	{
1854		PLpgSQL_variable *var = vars[i];
1855		Oid			typoid;
1856		int32		typmod;
1857		Oid			typcoll;
1858
1859		/* Member vars of a row should never be const */
1860		Assert(!var->isconst);
1861
1862		switch (var->dtype)
1863		{
1864			case PLPGSQL_DTYPE_VAR:
1865			case PLPGSQL_DTYPE_PROMISE:
1866				typoid = ((PLpgSQL_var *) var)->datatype->typoid;
1867				typmod = ((PLpgSQL_var *) var)->datatype->atttypmod;
1868				typcoll = ((PLpgSQL_var *) var)->datatype->collation;
1869				break;
1870
1871			case PLPGSQL_DTYPE_REC:
1872				/* shouldn't need to revalidate rectypeid already... */
1873				typoid = ((PLpgSQL_rec *) var)->rectypeid;
1874				typmod = -1;	/* don't know typmod, if it's used at all */
1875				typcoll = InvalidOid;	/* composite types have no collation */
1876				break;
1877
1878			default:
1879				elog(ERROR, "unrecognized dtype: %d", var->dtype);
1880				typoid = InvalidOid;	/* keep compiler quiet */
1881				typmod = 0;
1882				typcoll = InvalidOid;
1883				break;
1884		}
1885
1886		row->fieldnames[i] = var->refname;
1887		row->varnos[i] = var->dno;
1888
1889		TupleDescInitEntry(row->rowtupdesc, i + 1,
1890						   var->refname,
1891						   typoid, typmod,
1892						   0);
1893		TupleDescInitEntryCollation(row->rowtupdesc, i + 1, typcoll);
1894	}
1895
1896	return row;
1897}

1899/*
1900 * Build a RECFIELD datum for the named field of the specified record variable
1901 *
1902 * If there's already such a datum, just return it; we don't need duplicates.
1903 */
1904PLpgSQL_recfield *
1905plpgsql_build_recfield(PLpgSQL_rec *rec, const char *fldname)
1906{
1907	PLpgSQL_recfield *recfield;
1908	int			i;
1909
1910	/* search for an existing datum referencing this field */
1911	i = rec->firstfield;
1912	while (i >= 0)
1913	{
1914		PLpgSQL_recfield *fld = (PLpgSQL_recfield *) plpgsql_Datums[i];
1915
1916		Assert(fld->dtype == PLPGSQL_DTYPE_RECFIELD &&
1917			   fld->recparentno == rec->dno);
1918		if (strcmp(fld->fieldname, fldname) == 0)
1919			return fld;
1920		i = fld->nextfield;
1921	}
1922
1923	/* nope, so make a new one */
1924	recfield = palloc0(sizeof(PLpgSQL_recfield));
1925	recfield->dtype = PLPGSQL_DTYPE_RECFIELD;
1926	recfield->fieldname = pstrdup(fldname);
1927	recfield->recparentno = rec->dno;
1928	recfield->rectupledescid = INVALID_TUPLEDESC_IDENTIFIER;
1929
1930	plpgsql_adddatum((PLpgSQL_datum *) recfield);
1931
1932	/* now we can link it into the parent's chain */
1933	recfield->nextfield = rec->firstfield;
1934	rec->firstfield = recfield->dno;
1935
1936	return recfield;
1937}

1940 * plpgsql_build_datatype
1941 *		Build PLpgSQL_type struct given type OID, typmod, collation,
1942 *		and type's parsed name.
1943 *
1944 * If collation is not InvalidOid then it overrides the type's default
1945 * collation.  But collation is ignored if the datatype is non-collatable.
1946 *
1947 * origtypname is the parsed form of what the user wrote as the type name.
1948 * It can be NULL if the type could not be a composite type, or if it was
1949 * identified by OID to begin with (e.g., it's a function argument type).
1950 */
1951PLpgSQL_type *
1952plpgsql_build_datatype(Oid typeOid, int32 typmod,
1953					   Oid collation, TypeName *origtypname)
1954{
1955	HeapTuple	typeTup;
1956	PLpgSQL_type *typ;
1957
1958	typeTup = SearchSysCache1(TYPEOID, ObjectIdGetDatum(typeOid));
1959	if (!HeapTupleIsValid(typeTup))
1960		elog(ERROR, "cache lookup failed for type %u", typeOid);
1961
1962	typ = build_datatype(typeTup, typmod, collation, origtypname);
1963
1964	ReleaseSysCache(typeTup);
1965
1966	return typ;
1967}

1969/*
1970 * Utility subroutine to make a PLpgSQL_type struct given a pg_type entry
1971 * and additional details (see comments for plpgsql_build_datatype).
1972 */
1973static PLpgSQL_type *
1974build_datatype(HeapTuple typeTup, int32 typmod,
1975			   Oid collation, TypeName *origtypname)
1976{
1977	Form_pg_type typeStruct = (Form_pg_type) GETSTRUCT(typeTup);
1978	PLpgSQL_type *typ;
1979
1980	if (!typeStruct->typisdefined)
1981		ereport(ERROR,
1982				(errcode(ERRCODE_UNDEFINED_OBJECT),
1983				 errmsg("type \"%s\" is only a shell",
1984						NameStr(typeStruct->typname))));
1985
1986	typ = (PLpgSQL_type *) palloc(sizeof(PLpgSQL_type));
1987
1988	typ->typname = pstrdup(NameStr(typeStruct->typname));
1989	typ->typoid = typeStruct->oid;
1990	switch (typeStruct->typtype)
1991	{
1992		case TYPTYPE_BASE:
1993		case TYPTYPE_ENUM:
1994		case TYPTYPE_RANGE:
1995		case TYPTYPE_MULTIRANGE:
1996			typ->ttype = PLPGSQL_TTYPE_SCALAR;
1997			break;
1998		case TYPTYPE_COMPOSITE:
1999			typ->ttype = PLPGSQL_TTYPE_REC;
2000			break;
2001		case TYPTYPE_DOMAIN:
2002			if (type_is_rowtype(typeStruct->typbasetype))
2003				typ->ttype = PLPGSQL_TTYPE_REC;
2004			else
2005				typ->ttype = PLPGSQL_TTYPE_SCALAR;
2006			break;
2007		case TYPTYPE_PSEUDO:
2008			if (typ->typoid == RECORDOID)
2009				typ->ttype = PLPGSQL_TTYPE_REC;
2010			else
2011				typ->ttype = PLPGSQL_TTYPE_PSEUDO;
2012			break;
2013		default:
2014			elog(ERROR, "unrecognized typtype: %d",
2015				 (int) typeStruct->typtype);
2016			break;
2017	}
2018	typ->typlen = typeStruct->typlen;
2019	typ->typbyval = typeStruct->typbyval;
2020	typ->typtype = typeStruct->typtype;
2021	typ->collation = typeStruct->typcollation;
2022	if (OidIsValid(collation) && OidIsValid(typ->collation))
2023		typ->collation = collation;
2024	/* Detect if type is true array, or domain thereof */
2025	/* NB: this is only used to decide whether to apply expand_array */
2026	if (typeStruct->typtype == TYPTYPE_BASE)
2027	{
2028		/*
2029		 * This test should include what get_element_type() checks.  We also
2030		 * disallow non-toastable array types (i.e. oidvector and int2vector).
2031		 */
2032		typ->typisarray = (IsTrueArrayType(typeStruct) &&
2033						   typeStruct->typstorage != TYPSTORAGE_PLAIN);
2034	}
2035	else if (typeStruct->typtype == TYPTYPE_DOMAIN)
2036	{
2037		/* we can short-circuit looking up base types if it's not varlena */
2038		typ->typisarray = (typeStruct->typlen == -1 &&
2039						   typeStruct->typstorage != TYPSTORAGE_PLAIN &&
2040						   OidIsValid(get_base_element_type(typeStruct->typbasetype)));
2041	}
2042	else
2043		typ->typisarray = false;
2044	typ->atttypmod = typmod;
2045
2046	/*
2047	 * If it's a named composite type (or domain over one), find the typcache
2048	 * entry and record the current tupdesc ID, so we can detect changes
2049	 * (including drops).  We don't currently support on-the-fly replacement
2050	 * of non-composite types, else we might want to do this for them too.
2051	 */
2052	if (typ->ttype == PLPGSQL_TTYPE_REC && typ->typoid != RECORDOID)
2053	{
2054		TypeCacheEntry *typentry;
2055
2056		typentry = lookup_type_cache(typ->typoid,
2057									 TYPECACHE_TUPDESC |
2058									 TYPECACHE_DOMAIN_BASE_INFO);
2059		if (typentry->typtype == TYPTYPE_DOMAIN)
2060			typentry = lookup_type_cache(typentry->domainBaseType,
2061										 TYPECACHE_TUPDESC);
2062		if (typentry->tupDesc == NULL)
2063			ereport(ERROR,
2064					(errcode(ERRCODE_WRONG_OBJECT_TYPE),
2065					 errmsg("type %s is not composite",
2066							format_type_be(typ->typoid))));
2067
2068		typ->origtypname = origtypname;
2069		typ->tcache = typentry;
2070		typ->tupdesc_id = typentry->tupDesc_identifier;
2071	}
2072	else
2073	{
2074		typ->origtypname = NULL;
2075		typ->tcache = NULL;
2076		typ->tupdesc_id = 0;
2077	}
2078
2079	return typ;
2080}

2083 * Build an array type for the element type specified as argument.

2085PLpgSQL_type *
2086plpgsql_build_datatype_arrayof(PLpgSQL_type *dtype)

2147 *		Generate PLpgSQL_condition entry(s) for an exception condition name

2264 * plpgsql_add_initdatums		Make an array of the datum numbers of
2265 *					all the initializable datums created since the last call
2266 *					to this function.
2267 *
2268 * If varnos is NULL, we just forget any datum entries created since the
2269 * last call.
2270 *
2271 * This is used around a DECLARE section to create a list of the datums
2272 * that have to be initialized at block entry.  Note that datums can also
2273 * be created elsewhere than DECLARE, eg by a FOR-loop, but it is then
2274 * the responsibility of special-purpose code to initialize them.
2275 * ----------
2276 */

2278plpgsql_add_initdatums(int **varnos)
2279{
2280	int			i;

2282
2283	/*
2284	 * The set of dtypes recognized here must match what exec_stmt_block()
2285	 * cares about (re)initializing at block entry.
2286	 */
2287	for (i = datums_last; i < plpgsql_nDatums; i++)
2288	{
2289		switch (plpgsql_Datums[i]->dtype)
2290		{
2291			case PLPGSQL_DTYPE_VAR:
2292			case PLPGSQL_DTYPE_REC:
2293				n++;
2294				break;
2295
2296			default:
2297				break;
2298		}
2299	}
2300
2301	if (varnos != NULL)
2302	{
2303		if (n > 0)
2304		{
2305			*varnos = (int *) palloc(sizeof(int) * n);
2306
2307			n = 0;
2308			for (i = datums_last; i < plpgsql_nDatums; i++)
2309			{
2310				switch (plpgsql_Datums[i]->dtype)
2311				{
2312					case PLPGSQL_DTYPE_VAR:
2313					case PLPGSQL_DTYPE_REC:
2314						(*varnos)[n++] = plpgsql_Datums[i]->dno;
2315
2316					default:
2317						break;
2318				}
2319			}
2320		}
2321		else
2322			*varnos = NULL;
2323	}
2324
2325	datums_last = plpgsql_nDatums;
