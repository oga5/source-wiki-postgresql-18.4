1-- Simple test to verify accessibility of the OLD and NEW trigger variables
2create table testtr (a int, b text);
3create function testtr_trigger() returns trigger language plpgsql as
4$$begin
5  raise notice 'tg_op = %', tg_op;
6  raise notice 'old(%) = %', old.a, row(old.*);
7  raise notice 'new(%) = %', new.a, row(new.*);
8  if (tg_op = 'DELETE') then
9    return old;
10  else
11    return new;
12  end if;
13end$$;
