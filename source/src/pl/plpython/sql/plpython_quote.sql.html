3CREATE FUNCTION quote(t text, how text) RETURNS text AS $$
4    if how == "literal":
5        return plpy.quote_literal(t)
6    elif how == "nullable":
7        return plpy.quote_nullable(t)
8    elif how == "ident":
9        return plpy.quote_ident(t)
10    else:
11        raise plpy.Error("unrecognized quote type %s" % how)
12$$ LANGUAGE plpython3u;
13
14SELECT quote(t, 'literal') FROM (VALUES
