1--
2-- Universal Newline Support
3--
4
5CREATE OR REPLACE FUNCTION newline_lf() RETURNS integer AS
6E'x = 100\ny = 23\nreturn x + y\n'
7LANGUAGE plpython3u;
8
9CREATE OR REPLACE FUNCTION newline_cr() RETURNS integer AS
10E'x = 100\ry = 23\rreturn x + y\r'
11LANGUAGE plpython3u;
12
13CREATE OR REPLACE FUNCTION newline_crlf() RETURNS integer AS
14E'x = 100\r\ny = 23\r\nreturn x + y\r\n'
15LANGUAGE plpython3u;
16
17
18SELECT newline_lf();
19SELECT newline_cr();
20SELECT newline_crlf();
