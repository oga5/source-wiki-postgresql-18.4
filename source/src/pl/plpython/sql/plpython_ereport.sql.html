43-- raise exception in python, handle exception in plpgsql
44CREATE OR REPLACE FUNCTION raise_exception(_message text, _detail text DEFAULT NULL, _hint text DEFAULT NULL,
45                                           _sqlstate text DEFAULT NULL,
46                                           _schema_name text DEFAULT NULL,
47                                           _table_name text DEFAULT NULL,
48                                           _column_name text DEFAULT NULL,
49                                           _datatype_name text DEFAULT NULL,
50                                           _constraint_name text DEFAULT NULL)
51RETURNS void AS $$
52kwargs = {
53    "message": _message, "detail": _detail, "hint": _hint,
54    "sqlstate": _sqlstate, "schema_name": _schema_name, "table_name": _table_name,
55    "column_name": _column_name, "datatype_name": _datatype_name,
56    "constraint_name": _constraint_name
57}
58# ignore None values
59plpy.error(**dict((k, v) for k, v in iter(kwargs.items()) if v))
60$$ LANGUAGE plpython3u;
61
62SELECT raise_exception('hello', 'world');
63SELECT raise_exception('message text', 'detail text', _sqlstate => 'YY333');
64SELECT raise_exception(_message => 'message text',
65                       _detail => 'detail text',
66                       _hint => 'hint text',
67                       _sqlstate => 'XX555',
68                       _schema_name => 'schema text',
69                       _table_name => 'table text',
70                       _column_name => 'column text',
71                       _datatype_name => 'datatype text',
72                       _constraint_name => 'constraint text');
73
74SELECT raise_exception(_message => 'message text',
75                       _hint => 'hint text',
76                       _schema_name => 'schema text',
77                       _column_name => 'column text',
78                       _constraint_name => 'constraint text');
79
80DO $$
81DECLARE
82  __message text;
83  __detail text;
84  __hint text;
85  __sqlstate text;
86  __schema_name text;
87  __table_name text;
88  __column_name text;
89  __datatype_name text;
90  __constraint_name text;
91BEGIN
92  BEGIN
93    PERFORM raise_exception(_message => 'message text',
94                            _detail => 'detail text',
95                            _hint => 'hint text',
96                            _sqlstate => 'XX555',
97                            _schema_name => 'schema text',
98                            _table_name => 'table text',
99                            _column_name => 'column text',
100                            _datatype_name => 'datatype text',
101                            _constraint_name => 'constraint text');
102  EXCEPTION WHEN SQLSTATE 'XX555' THEN
103    GET STACKED DIAGNOSTICS __message = MESSAGE_TEXT,
104                            __detail = PG_EXCEPTION_DETAIL,
105                            __hint = PG_EXCEPTION_HINT,
106                            __sqlstate = RETURNED_SQLSTATE,
107                            __schema_name = SCHEMA_NAME,
108                            __table_name = TABLE_NAME,
109                            __column_name = COLUMN_NAME,
110                            __datatype_name = PG_DATATYPE_NAME,
111                            __constraint_name = CONSTRAINT_NAME;
112    RAISE NOTICE 'handled exception'
113      USING DETAIL = format('message:(%s), detail:(%s), hint: (%s), sqlstate: (%s), '
114                            'schema_name:(%s), table_name:(%s), column_name:(%s), datatype_name:(%s), constraint_name:(%s)',
115                            __message, __detail, __hint, __sqlstate, __schema_name,
116                            __table_name, __column_name, __datatype_name, __constraint_name);
117  END;
118END;
119$$;
