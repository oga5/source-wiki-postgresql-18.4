33CREATE FUNCTION import_test_one(p text) RETURNS text
34	AS
35'try:
36    import hashlib
37    digest = hashlib.sha1(p.encode("ascii"))
38except ImportError:
39    import sha
40    digest = sha.new(p)
41return digest.hexdigest()'
42	LANGUAGE plpython3u;
