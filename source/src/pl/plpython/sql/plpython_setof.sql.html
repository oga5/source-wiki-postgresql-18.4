12CREATE FUNCTION test_setof_as_list(count integer, content text) RETURNS SETOF text AS $$
13return [ content ]*count
14$$ LANGUAGE plpython3u;
15
16CREATE FUNCTION test_setof_as_tuple(count integer, content text) RETURNS SETOF text AS $$
17t = ()
18for i in range(count):
19	t += ( content, )
20return t
21$$ LANGUAGE plpython3u;

23CREATE FUNCTION test_setof_as_set(count integer, content text) RETURNS SETOF text AS $$
24s = set()
25for i in range(count):
26	s.add(content * (i + 1) if content is not None else None)
27return s
28$$ LANGUAGE plpython3u;

30CREATE FUNCTION test_setof_as_iterator(count integer, content text) RETURNS SETOF text AS $$
31class producer:
32	def __init__ (self, icount, icontent):
33		self.icontent = icontent
34		self.icount = icount
35	def __iter__ (self):
36		return self
37	def __next__ (self):
38		if self.icount == 0:
39			raise StopIteration
40		self.icount -= 1
41		return self.icontent
42return producer(count, content)
43$$ LANGUAGE plpython3u;

78-- set-returning function that modifies its parameters
79CREATE OR REPLACE FUNCTION ugly(x int, lim int) RETURNS SETOF int AS $$
80global x
81while x <= lim:
82    yield x
83    x = x + 1
84$$ LANGUAGE plpython3u;
85
86SELECT ugly(1, 5);
