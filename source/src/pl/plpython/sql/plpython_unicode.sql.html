1--
2-- Unicode handling
3--
4-- This test case would fail if the database encoding is EUC_CN, EUC_JP,
5-- EUC_KR, or EUC_TW, for lack of any equivalent to U+00A0 (no-break space) in
6-- those encodings.  However, testing with plain ASCII data would be rather
7-- useless, so we must live with that.
8--
9SELECT getdatabaseencoding() IN ('EUC_CN', 'EUC_JP', 'EUC_KR', 'EUC_TW')
10  AS skip_test \gset
11\if :skip_test
12\quit
13\endif
14
15SET client_encoding TO UTF8;
16
17CREATE TABLE unicode_test (
18	testvalue  text NOT NULL
19);
20
21CREATE FUNCTION unicode_return() RETURNS text AS E'
22return "\\xA0"
23' LANGUAGE plpython3u;
24
25CREATE FUNCTION unicode_trigger() RETURNS trigger AS E'
26TD["new"]["testvalue"] = "\\xA0"
27return "MODIFY"
28' LANGUAGE plpython3u;
29
30CREATE TRIGGER unicode_test_bi BEFORE INSERT ON unicode_test
31  FOR EACH ROW EXECUTE PROCEDURE unicode_trigger();
32
33CREATE FUNCTION unicode_plan1() RETURNS text AS E'
34plan = plpy.prepare("SELECT $1 AS testvalue", ["text"])
35rv = plpy.execute(plan, ["\\xA0"], 1)
36return rv[0]["testvalue"]
37' LANGUAGE plpython3u;
38
39CREATE FUNCTION unicode_plan2() RETURNS text AS E'
40plan = plpy.prepare("SELECT $1 || $2 AS testvalue", ["text", "text"])
41rv = plpy.execute(plan, ["foo", "bar"], 1)
42return rv[0]["testvalue"]
43' LANGUAGE plpython3u;
44
45
46SELECT unicode_return();
47INSERT INTO unicode_test (testvalue) VALUES ('test');
48SELECT * FROM unicode_test;
49SELECT unicode_plan1();
50SELECT unicode_plan2();
